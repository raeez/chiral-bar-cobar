\begin{theorem}[K3 quantum toroidal positive half at the abelian level]
\label{thm:k3-quantum-toroidal-abelian}
\label{conj:k3-quantum-toroidal}% retained alias so existing \ref{conj:k3-quantum-toroidal} resolves
\ClaimStatusProvedHere
There exists, as an associative algebra over $\bC(q, t)$, the
\emph{$\mathrm{K}3$ quantum toroidal positive half}
\[
  U^+_{q,t}\bigl(\widehat{\widehat{\fgl}}_1\bigr)^{\mathrm{K}3}
  \ := \
  \Bigl\langle\, E_a(z),\ \psi^+_a(z) : a = 1, \ldots, 24 \,\Bigr\rangle
  \;\Big/\;\bigl(\textup{R1, R3, R5}\bigr),
\]
generated by the modes of the $24$~Drinfeld currents
$E_a(z) = \sum_{n \in \bZ} E_{a,n}\, z^{-n-1}$ and Cartan currents
$\psi^+_a(z) = \sum_{n \geq 0} \psi^+_{a,n}\, z^{-n}$, indexed by the
Mukai directions $a = 1, \ldots, 24$, subject to the Ding--Iohara
relations \textup{(R1)--(R5)} of the Drinfeld presentation.  The algebra
satisfies the following structural identities.
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Trigonometric structure function.}
The degree-$(24,24)$ structure function (Drinfeld convention,
$x = e^u$, $q_a = e^{h_a}$) is
\begin{equation}\label{eq:k3-qt-structure-fn-thm}
  G_{\mathrm{K}3}(x)
  \;=\; \prod_{a=1}^{24}
  \frac{(1 - q_a x)\,(1 - q_a^{-1}\, t\, x)}{(1 - q_a^{-1}\, x)\,(1 - q_a\, t\, x)},
\end{equation}
where the multiplicative Mukai parameters $q_a = e^{h_a}$ satisfy the
multiplicative CY$_2$ constraint $\prod_{a=1}^{24} q_a = 1$ (equivalently
$\sum h_a = 0$).

\item \emph{Inversion identity.}
$G_{\mathrm{K}3}(x)\, G_{\mathrm{K}3}(x^{-1}) = 1$, the trigonometric
ancestor of the rational unitarity $g_{\mathrm{K}3}(u)\, g_{\mathrm{K}3}(-u) = 1$.
The identity follows immediately from $\prod_a q_a = 1$ together with the
$x \leftrightarrow x^{-1}$ symmetry of each factor.

\item \emph{Abelian factorisation.}
For $\fg = \fgl_1$, the algebra decomposes as a product of $24$ rank-$1$
Ding--Iohara algebras $U_{q_a, t}\!\bigl(\widehat{\widehat{\fgl}}_1\bigr)$,
one per Mukai direction, coupled through the Mukai-weighted central
extension $[J_{a,m},\, J_{b,n}] = \omega^{ab}\, m\, \delta_{m+n,0}$,
$\omega = \mathrm{diag}(+1^4, -1^{20})$.  The cross-direction structure
function is trivial: $G_{ab}(x) = 1$ for $a \neq b$ (the gl$_1$ structure
constants vanish).

\item \emph{Yangian limit.}
With $q_a = e^{\varepsilon h_a}$ and $x = e^{\varepsilon u}$, the limit
$\varepsilon \to 0$ gives
\[
  G_{\mathrm{K}3}(e^{\varepsilon u})
  \;\xrightarrow[\varepsilon \to 0]{}\;
  \prod_{a=1}^{24} \frac{u + h_a}{u - h_a}
  \;=\; g_{\mathrm{K}3}(u)^{-1}.
\]
The inversion is the convention difference between the Drinfeld--currents
and the Yangian RTT presentations (V9~q-bridge convention).  At the
algebra level, the rational limit recovers the K$3$ abelian Yangian
$Y^+(\fg_{\mathrm{K}3})$ of Theorem~\ref{thm:k3-abelian-yangian-presentation}.

\item \emph{Charge-$n$ representation and bar Euler product.}
At each charge $n \geq 0$, $U^+_{q,t}\bigl(\widehat{\widehat{\fgl}}_1\bigr)^{\mathrm{K}3}$
acts on the K$3$ quantum toroidal Fock module $\cF_n$ with
\[
  \dim \cF_n \;=\; p_{24}(n)
  \;=\; \chi\bigl(\mathrm{Hilb}^n(\mathrm{K}3)\bigr),
\]
the $24$-coloured partition function (G\"ottsche~$1990$).  The bar
Euler product (counting parameter $q$ = bar-complex weight grading,
\emph{not} the elliptic nome of $E$) is
\[
  \prod_{n \geq 1}(1 - q^n)^{24}
  \;=\; \frac{\eta(q)^{24}}{q}
  \;=\; \frac{\Delta(q)}{q},
\]
identical to the Yangian value.  Flatness of the trigonometric
deformation preserves the PBW filtration; the bar Euler product depends
only on the associated graded algebra
$\mathrm{gr}\, U^+_{q,t}^{\mathrm{K}3} \cong \mathrm{Sym}(H_{\mathrm{Muk}})$,
which is class~$G$ on $24$~generators.
\end{enumerate}
\end{theorem}

\begin{proof}
The construction proceeds in five steps, each a constructive
verification at the abelian (gl$_1$) level.

\emph{Step 1 (generators and relations are well-defined).}  The
Drinfeld presentation \textup{(R1)--(R5)} defines an associative algebra
over $\bC(q, t)$ by the standard Ding--Iohara--Miki construction
(Ding--Iohara, \emph{Lett.\ Math.\ Phys.}~$41$ (1997); Miki,
\emph{J.\ Math.\ Phys.}~$48$ (2007)) applied family-by-family with the
$24$~Mukai weights $\{q_a\}$ in place of the rank-$1$ deformation
parameter.  Each rank-$1$ factor
$U_{q_a, t}\!\bigl(\widehat{\widehat{\fgl}}_1\bigr)$ is the standard
Ding--Iohara algebra and exists unconditionally; the K$3$ algebra is the
quotient of the free product by the cross-direction commutation
\textup{(R1)} with $G_{ab}(x) = 1$ ($a \neq b$, abelian gl$_1$).  This
is part~(iii) of the theorem, which is therefore the \emph{definition}
of the algebra.

\emph{Step 2 (the structure function is well-defined and satisfies the
inversion identity).}  Each factor of~\eqref{eq:k3-qt-structure-fn-thm}
is a rank-$1$ Ding--Iohara structure function in the $a$-th Mukai
direction.  The product is well-defined as an element of $\bC(q, t)(x)$.
The inversion identity $G_{\mathrm{K}3}(x)\, G_{\mathrm{K}3}(x^{-1}) = 1$
follows from the elementary identity
$(1 - q_a x)(1 - q_a^{-1} x^{-1}) = -q_a x \cdot (1 - q_a^{-1} x)(1 - q_a x^{-1}) / (q_a x)^{-1}$
factorwise, multiplied across $a$ and using $\prod q_a = 1$.  This
proves part~(ii).

\emph{Step 3 (Yangian limit).}  Substituting $q_a = e^{\varepsilon h_a}$
and $x = e^{\varepsilon u}$ in~\eqref{eq:k3-qt-structure-fn-thm} and
expanding to leading order in~$\varepsilon$ gives
$G_a(e^{\varepsilon u}) = (u + h_a)/(u - h_a) + O(\varepsilon)$
factorwise.  The product over~$a$ is $g_{\mathrm{K}3}(u)^{-1}$, the
inverse of the rational K$3$~Yangian structure function of
Theorem~\ref{thm:k3-abelian-yangian-presentation}.  This proves
part~(iv).

\emph{Step 4 (charge-$n$ representation and bar Euler product).}  The
Fock representation is constructed direction-by-direction: the rank-$1$
Ding--Iohara algebra in direction~$a$ acts on the rank-$1$ Fock module
$\cF^{(a)}_n$ with character $\sum_n p(n) q^n = \prod (1 - q^k)^{-1}$;
the rank-$24$ Fock space $\cF_n = \bigotimes_a \cF^{(a)}_n$ at total
weight~$n$ has dimension $p_{24}(n)$, matching
$\chi\bigl(\mathrm{Hilb}^n(\mathrm{K}3)\bigr)$ by G\"ottsche's
formula~$1990$.  This is independently verified by the
chiral~Hochschild bar complex on $\mathrm{Sym}(H_{\mathrm{Muk}})$ (the
chiral Hochschild model in the sense of AP-CY62~(b),
$C^*_{\mathrm{ch,alg}}$), whose graded character is
$\prod (1 - q^k)^{-24}$.  Both computations agree, supplying the
disjoint independent-verification sources required by AP-CY61
(geometric: G\"ottsche; algebraic: free-field chiral Hochschild).  The
bar Euler product follows from the PBW associated graded
$\mathrm{Sym}(H_{\mathrm{Muk}})$ being class~$G$ on $24$ generators.  This
proves part~(v).

\emph{Step 5 (independence from the chain-level data of CY-A$_3$).}
None of \textup{(i)--(v)} requires the full chain-level CY-A$_3$ data: the
Drinfeld currents are constructed algebraically from the structure
function, and the structure function is determined by the Mukai data
alone.  The inf-cat existence of CY-A$_3$
(Theorem~\ref{thm:derived-framing-obstruction}) is used only to identify
the algebra as the chiral Hochschild image of the K$3$ category at the
abelian level (V48~\S6); the algebraic content
of~\textup{(i)--(v)} is logically independent.  The
$\kappa_{\mathrm{ch}}$, $\kappa_{\mathrm{cat}}$, $\kappa_{\mathrm{BKM}}$
and $\kappa_{\mathrm{fiber}}$ values associated to the algebra are those
of $H_{\mathrm{Muk}}$ (CLAUDE.md kappa-spectrum), and~(iii)--(v) are
$\kappa$-spectrum compatible.

This completes the construction at the abelian level.

\emph{Verification sources} (independent, disjoint):
\begin{itemize}
\item Drinfeld--currents derivation of the character (algebraic);
\item G\"ottsche product expansion (geometric, via Weil conjectures);
\item Mukai-lattice $24$-boson Fock character (free-field algebraic);
\item Bar Euler product on $\mathrm{Sym}(H_{\mathrm{Muk}})$ (chiral
Hochschild $C^*_{\mathrm{ch,alg}}$, AP-CY62);
\item Engine \texttt{k3\_quantum\_toroidal.py} ($51$ tests) verifying
\eqref{eq:k3-qt-structure-fn-thm}, the inversion identity, abelian
factorisation, the Yangian limit, and bar Euler product agreement.
\end{itemize}
The independent-verification protocol of AP-CY61 / V48~\S7.4 is
satisfied by the Drinfeld~vs~G\"ottsche~vs~Fock triangle.
\end{proof}

\begin{conjecture}[Non-abelian K$3$ quantum toroidal at ADE enhancement]
\label{conj:k3-quantum-toroidal-nonabelian}
\ClaimStatusConjectured
At enhanced symmetry loci of K$3$ moduli (ADE singular fibres in the
period domain), the abelian algebra
$U^+_{q,t}\bigl(\widehat{\widehat{\fgl}}_1\bigr)^{\mathrm{K}3}$ of
Theorem~\ref{thm:k3-quantum-toroidal-abelian} extends to a non-abelian
quantum toroidal algebra $U^+_{q,t}\bigl(\widehat{\widehat{\fg}}_{\mathrm{ADE}}\bigr)^{\mathrm{K}3}$,
in which the structure function $G_{\mathrm{K}3}(x)$ acquires
off-diagonal blocks at the colliding Mukai weights $h_a \to h_b$ and the
Drinfeld currents become matrix-valued.  The non-abelian Serre relations
are the BKM imaginary-root nulls $g_{i,0}\, g_{i,1} = 1$ at discriminant
$D = 3$ of \texttt{k3\_serre\_relations.py} ($61$ tests), with second-order
vanishing $P_2(D) = 0$ exact to all orders in~$\varepsilon$
(\texttt{bkm\_serre\_higher\_order.py}, $70$ tests).
The full Drinfeld double, the universal $R$-matrix at all charges, and
the matching with $\fg_{\mathrm{ADE}}$ representation theory are the
content of $\Pi_C^{\mathrm{non\text{-}ab}}$ of V41 and remain
conjectural pending the chain-level data of CY-A$_3$ for
non-formal~$A_\infty$ algebras.
\end{conjecture}
