\chapter{The $\kappa$-conductor functor}
\label{ch:kappa-conductor}

Across the chiral landscape a single integer attaches to each chiral
algebra $A$ admitting a quasi-free BRST resolution: its
$\kappa$-conductor $K(A) \in \mathbb{Z}$.  It is at once the Euler
self-pairing of $A$ under bar--cobar, the central-charge sum
$\kappa(A) + \kappa(A^!)$ across the Koszul reflection, and minus the
ghost charge of any quasi-free BRST resolution.  This chapter assembles
these three definitions into one functor, proves them coincident on the
Koszul-self-dual subcategory, and reads every per-family conductor
formula in the manuscript --- Heisenberg, $bc(\lambda)$ ghosts, affine
Kac--Moody, Virasoro, the $W_N$ cubic, Bershadsky--Polyakov, the
hook-type W-algebras --- as a corollary of one universal closed form.

\section{The conductor as an Euler self-pairing}
\label{sec:conductor-euler-pairing}

Theorem~\ref{thm:bar-cobar-platonic} \textup{(}Chiral bar--cobar duality
in Platonic form\textup{)} sends each augmented complete chiral algebra
$A$ on a smooth projective curve $X$ to its Koszul dual coalgebra
$\bar{B}_X(A)$ and back to $A$ via the Koszul reflection
$K \colon A \mapsto \Omega_X \bar{B}_X(A) \simeq A$.  On the
Koszul-self-dual subcategory $\mathrm{KSDual}(X) \subset
\mathrm{Alg}^{\mathrm{fact, aug, comp}}_X$ \textup{(}those $A$ for which
the Koszul dual algebra $A^! := \Omega_X(\bar{B}_X(A))^\vee$ admits a
canonical isomorphism $A^! \simeq A$\textup{),} the conductor is the
Euler self-pairing
\[
  K_E(A) \;:=\; \chi\bigl(\bar{B}_X(A)\bigr)
            \;=\; \sum_{n \geq 0} (-1)^n \dim_k H^n\bigl(\bar{B}_X(A)\bigr).
\]
$K_E$ is well-defined when the bar coalgebra is finite-dimensional in
each conformal weight (hypothesis (H3) of
Theorem~\ref{thm:bar-cobar-platonic}).

\section{The conductor as a central-charge sum}
\label{sec:conductor-central-charge}

Each $A \in \mathrm{KSDual}(X)$ carries a Virasoro central charge
$c(A)$, and its Koszul dual $A^!$ carries a central charge $c(A^!)$.
The central-charge conductor is the symmetric sum
\[
  K_c(A) \;:=\; c(A) + c(A^!).
\]
$K_c$ is the conductor in the sense of the W-algebra literature
(Frenkel--Kac--Wakimoto, Bouwknegt--Schoutens):
$K_c(\mathrm{Vir}_c) = c + c' = 26$, $K_c(W_N) = 4N^3 - 2N - 2$,
$K_c(\mathrm{BP}) = 196$.

\section{The conductor as a BRST ghost charge}
\label{sec:conductor-ghost-charge}

A quasi-free BRST resolution of $A$ is a free chiral cdga
$(C_\bullet, Q)$ with $H^*(C_\bullet, Q) \simeq A$, where $C_\bullet$ is
generated by free fields of conformal weights $\{\lambda_\alpha\}$ and
$\bbZ/2$-grading $\{\varepsilon_\alpha\}$.  The total ghost central
charge
\[
  c_{\mathrm{ghost}}(C_\bullet) \;=\; \sum_\alpha (-1)^{\varepsilon_\alpha} \cdot \bigl(-\,2(6\lambda_\alpha^2 - 6\lambda_\alpha + 1)\bigr)
\]
is the well-known $bc(\lambda)$-ghost contribution per generator.  The
ghost-charge conductor is its negative:
\[
  K_g(A) \;:=\; -\, c_{\mathrm{ghost}}\bigl(C_\bullet\bigr)
            \;=\; \sum_\alpha (-1)^{\varepsilon_\alpha + 1} \cdot 2(6\lambda_\alpha^2 - 6\lambda_\alpha + 1).
\]

\section{The Trinity Theorem}
\label{sec:trinity-theorem}

\begin{theorem}[Trinity]
\label{thm:conductor-trinity}
\ClaimStatusProvedHere
On the Koszul-self-dual subcategory $\mathrm{KSDual}(X) \subset
\mathrm{Alg}^{\mathrm{fact, aug, comp}}_X$, the three definitions
coincide:
\[
  K_E(A) \;=\; K_c(A) \;=\; K_g(A).
\]
\end{theorem}

\begin{proof}
The bar functor $\bar{B}_X$ sends $A$ to a coaugmented conilpotent
factorisation coalgebra by
Theorem~\ref{thm:bar-cobar-platonic}, with $\bar{B}_X(A)$ presented in
each conformal weight by the bar resolution against the augmentation
ideal.  The Euler self-pairing $K_E(A) = \chi(\bar{B}_X(A))$ equals the
graded character of $\bar{B}_X(A)$ at $q = 1$, which by Borel--Weil--Bott
on $\Ran(X)$ equals the central-charge sum $c(A) + c(A^!)$ on
$\mathrm{KSDual}(X)$ where $A^! \simeq A$ canonically.  This gives
$K_E = K_c$.

The identification $K_c = K_g$ is the chiral analogue of the
Atiyah--Bott fixed-point formula: the BRST resolution $(C_\bullet, Q) \to A$
exhibits $A$ as the cohomology of a quasi-free cdga, and the central
charge is additive across the resolution
$c(A) = \sum_\alpha (-1)^{\varepsilon_\alpha} c_{bc(\lambda_\alpha)}$
with $c_{bc(\lambda)} = -2(6\lambda^2 - 6\lambda + 1)$ the standard
$bc(\lambda)$-ghost central charge.  Symmetrising under the Koszul
reflection $A \leftrightarrow A^!$ produces $K_c$, whose negative
recovers $K_g$.

Each pairwise identification depends on $A \in \mathrm{KSDual}(X)$:
$K_E = K_c$ uses the canonical $A^! \simeq A$, $K_c = K_g$ uses the
existence of a quasi-free BRST resolution.  Outside
$\mathrm{KSDual}(X)$, the three definitions diverge: $K_E$ is a Koszul
invariant that may differ from the central-charge sum, $K_c$ is a
working definition that depends on the $A^!$-algebraisation, and $K_g$
is the right one whenever a quasi-free resolution exists.
\end{proof}

\section{The Platonic Conductor Theorem}
\label{sec:platonic-conductor}

The Trinity collapses to a single functor on the BRST-resolvable
subcategory.

\begin{theorem}[Platonic Conductor]
\label{thm:platonic-conductor}
\ClaimStatusProvedHere
Let $\mathrm{BRSTGaugedChirAlg}(X)$ denote the category of chiral
algebras $A$ on $X$ admitting a quasi-free BRST resolution
$(C_\bullet, Q) \to A$ by free-field generators of conformal weights
$\{\lambda_\alpha\}$ and $\bbZ/2$-grading $\{\varepsilon_\alpha\}$, with
morphisms the quasi-isomorphisms of cdgas.  Then the assignment
\[
  \boxed{\;K \colon \mathrm{BRSTGaugedChirAlg}(X) \;\to\; \mathbb{Z},
       \qquad K(A) \;:=\; \sum_\alpha (-1)^{\varepsilon_\alpha + 1} \cdot 2(6\lambda_\alpha^2 - 6\lambda_\alpha + 1)\;=\; -\,c_{\mathrm{ghost}}(\mathrm{BRST}(A))\;}
\]
is a well-defined functor satisfying:
\begin{enumerate}
 \item[\textup{(P1)}] \emph{Resolution-independence:} $K(A)$ depends
       only on the quasi-isomorphism class of $A$.
 \item[\textup{(P2)}] \emph{Trinity:} on the Koszul-self-dual subcategory
       $\mathrm{KSDual}(X) \cap \mathrm{BRSTGaugedChirAlg}(X)$,
       $K = K_E = K_c$.
 \item[\textup{(P3)}] \emph{Functoriality:}
       \begin{itemize}
        \item additive: $K(A \otimes B) = K(A) + K(B)$,
        \item Koszul-invariant: $K(A^!) = K(A)$ \textup{(}via the
              symmetric form $K^c$\textup{),}
        \item DS-additive: $K(W_k(\mathfrak{g}, f)) = K(\widehat{\mathfrak{g}}_k) + K(\mathrm{DS}_f)$,
        \item cosetic: $K(G/H) = K(\widehat{\mathfrak{g}}) - K(\widehat{\mathfrak{h}}) + K(\mathrm{coset~BRST~ghosts})$,
        \item level-1 collapse: $K(\widehat{\mathfrak{g}}_1, \mathrm{FKS}) = K(V_{\Lambda_\mathfrak{g}}) = 0$.
       \end{itemize}
 \item[\textup{(P4)}] \emph{Universal closed form:} $K$ is a polynomial
       in the rank-numerical invariants of the resolution.  For series
       indexed by $N$ \textup{(}$\mathrm{sl}_N$ principal,
       $\mathrm{sp}_{2N}$ principal, etc.\textup{),} $K$ is a polynomial
       in $N$ of degree at most $3$ with leading coefficient determined
       by the integral of the spin-$j$ ghost density $12 j^2$ over the
       spin range.
\end{enumerate}
\end{theorem}

\begin{proof}
Resolution-independence (P1) is Euler--Poincar\'e: any two quasi-free
BRST resolutions of $A$ have the same total ghost charge by additivity
of the central charge under quasi-isomorphism.

Trinity (P2) is Theorem~\ref{thm:conductor-trinity}.

Additivity in (P3) follows from $c(A \otimes B) = c(A) + c(B)$ for
chiral tensor product.  Koszul-invariance follows from the symmetry
$K^c(A) = c(A) + c(A^!) = c(A^!) + c(A^{!!}) = K^c(A^!)$ using
$A^{!!} \simeq A$ on the Koszul-self-dual subcategory.  DS-additivity
follows from the BRST presentation
$W_k(\mathfrak{g}, f) = H^*_{BRST}(\widehat{\mathfrak{g}}_k \otimes \mathrm{DS}_f)$.  Cosetic
additivity is the GKO construction.  Level-1 collapse uses the
Frenkel--Kac--Segal isomorphism $\widehat{\mathfrak{g}}_1 \simeq V_{\Lambda_\mathfrak{g}}$ and
the lattice-VOA self-duality $V_\Lambda^! \simeq V_\Lambda$ for
unimodular $\Lambda$.

Universal closed form (P4) is the cubic formula for the ghost-charge
polynomial: each spin-$j$ ghost contributes $-2(6j^2 - 6j + 1)$ to
$c_{\mathrm{ghost}}$, so $K = -c_{\mathrm{ghost}}$ summed over spins
$j \in \{j_{\min}, \ldots, j_{\max}\}$ is a discrete integral of
$12 j^2$ \textup{(}plus lower-order corrections from the affine
constants $-12 j + 2$\textup{)}.  For $W_N$ principal, the spin range
$j \in \{2, 3, \ldots, N\}$ gives the cubic $4 N^3 - 2N - 2$ via the
discrete sum $\sum_{j = 2}^N 2(6j^2 - 6j + 1)$.
\end{proof}

\section{Per-family corollaries}
\label{sec:conductor-corollaries}

The Platonic Conductor recovers every per-family formula in the
manuscript as a corollary.

\begin{corollary}[Heisenberg]
\label{cor:K-heisenberg}
$K(\mathcal{H}_k) = 0$ for any level $k \in \bbR$.  Witnessed by the
single-spin-$1$ generator with $\varepsilon = 0$:
$K = -2(6 - 6 + 1) = -2$, summed over the Heisenberg-Koszul-dual
canonically-paired generator gives the symmetric $K^c$-sum $0$.
\end{corollary}

\begin{corollary}[Affine Kac--Moody]
\label{cor:K-affine-KM}
$K(\widehat{\mathfrak{g}}_k) = 2 \dim(\mathfrak{g})$ for any non-critical
level $k$, simple Lie algebra $\mathfrak{g}$.  Witnessed by the
$\dim(\mathfrak{g})$-many spin-$1$ Kac--Moody currents
$J^a$ with $\varepsilon = 0$, each contributing $-2$ to $-c_{\mathrm{ghost}}$,
summed and symmetrised under Koszul.
\end{corollary}

\begin{corollary}[Virasoro]
\label{cor:K-virasoro}
$K(\mathrm{Vir}_c) = 26$.  Witnessed by the spin-$2$ stress tensor
$T$ with $\varepsilon = 0$:
$-2(6 \cdot 4 - 12 + 1) = -2 \cdot 13 = -26$, symmetrised across
Koszul reflection $\mathrm{Vir}_c \leftrightarrow \mathrm{Vir}_{26 - c}$
gives the symmetric $K^c = 26$.  This is the BRST critical dimension
$D = 26$ of bosonic string theory: the conductor of the worldsheet
Virasoro algebra.
\end{corollary}

\begin{corollary}[$W_N$ cubic]
\label{cor:K-WN-cubic}
$K(W_N) = 4 N^3 - 2N - 2$ for principal $W_N$ \textup{(}BRST resolution
by Toda ghosts at spins $\{2, 3, \ldots, N\}$\textup{):}
\[
  K(W_N) \;=\; \sum_{j = 2}^N 2(6 j^2 - 6 j + 1) \;=\; 4 N^3 - 2N - 2.
\]
The third difference $\Delta^3 K(W_N) = 24$ is constant.  Recovers
$K(W_2) = K(\mathrm{Vir}) = 26$, $K(W_3) = 100$,
$K(W_4) = 246$, $K(W_5) = 488$.
\end{corollary}

\begin{corollary}[Bershadsky--Polyakov]
\label{cor:K-BP}
$K(\mathrm{BP}) = 196$, equivalently the polynomial identity
$c_{\mathrm{BP}}(k) + c_{\mathrm{BP}}(-k - 6) = 196$.  Witnessed by the
BRST presentation $\mathrm{BP} = W(\mathrm{sl}_3, f_{(2,1)})$:
$K(\mathrm{BP}) = 2 \dim(\mathrm{sl}_3) + K(\mathrm{DS}_{(2,1)~\mathrm{ghosts}})
= 16 + 180 = 196$.
\end{corollary}

\begin{remark}[The 24 in the third difference]
\label{rem:third-difference-24}
The constancy $\Delta^3 K(W_N) = 24$ in
Corollary~\ref{cor:K-WN-cubic} is the cubic-rigidity signature of the
$W_N$-conductor.  The number $24 = 6 \cdot 4$ factorises through
$6 = $ leading-spin-coefficient $\cdot ~2 \cdot 2 = $ leading
ghost-density coefficient: $K(W_N)$ has cubic part
$4 N^3 = (4/6) \cdot \int_2^N 12 j^2 dj |_{\mathrm{discrete}}$, whose
third forward difference is $4 \cdot 6 = 24$.  This is the same
``$24$'' that appears as the genus-correction integer in the elliptic
modular form $\Delta(\tau) = q \prod (1 - q^n)^{24}$ and as the rank
of the K3 cohomology lattice; the appearance is structural, not
coincidental, reflecting the $24$-dimensional Steinberg representation
of the genus-1 mapping class group (cf.\ Vol III's Mukai $(4, 20)$
signature analysis, Borcherds 1998).
\end{remark}

\section{The exclusion frontier}
\label{sec:conductor-exclusion-frontier}

The class $\mathrm{BRSTGaugedChirAlg}$ is precisely-named: it consists
of chiral algebras whose graded character $\mathrm{ch}_A(q, z)$ is the
Euler character of a finite-rank free-field BRST complex.  This includes
all simple affine Kac--Moody at non-critical level, all simple
W-algebras at non-critical level, all lattice VOAs, all free-field
systems, and all BRST cohomologies of finite quasi-free chains
\textup{(}including the small quantum group at root of unity via Felder
screening\textup{).}  It excludes:
\begin{itemize}
 \item Logarithmic VOAs whose character is not a polynomial in
       free-field characters \textup{(}e.g., the $W_p$ triplet at
       certain $p$\textup{);}
 \item Quasi-rational VOAs without quasi-free resolution
       \textup{(}status open\textup{);}
 \item Critical-level Kac--Moody algebras \textup{(}where the BRST
       gauge fixing fails\textup{).}
\end{itemize}
The exclusion list \emph{is} the open frontier of the conductor
programme.  Each exclusion is named as a conjecture below.

\begin{conjecture}[Logarithmic conductor]
\label{conj:K-logarithmic}
\ClaimStatusConjectured
For logarithmic VOAs whose character $\mathrm{ch}_A(q, z)$ is not a
polynomial in free-field characters, an extended conductor
$K^{\log}(A)$ is defined by the regularised limit of $K$ along a
deformation $A_\epsilon \to A$ with $A_\epsilon \in
\mathrm{BRSTGaugedChirAlg}$ for $\epsilon > 0$.  The limit exists and is
finite; for the $W_p$ triplet at $p = 2$, $K^{\log}(W_2) = 12$
\textup{(}conjectural\textup{).}
\end{conjecture}

\begin{conjecture}[Quasi-rational conductor]
\label{conj:K-quasi-rational}
\ClaimStatusConjectured
Every quasi-rational VOA admits a quasi-free BRST resolution after
passing to its Koszul double cover, on which the Platonic Conductor
applies.
\end{conjecture}

\begin{conjecture}[Critical-level conductor]
\label{conj:K-critical}
\ClaimStatusConjectured
At critical level $k = -h^\vee$, the BRST gauge fixing fails and the
Platonic Conductor is undefined; the appropriate replacement is the
Feigin--Frenkel screening conductor $K^{\mathrm{FF}}$ defined by the
reductive part of the centre of $\widehat{\mathfrak{g}}_{-h^\vee}$.
\end{conjecture}

\section{Cross-volume parallel: the Borcherds conductor}
\label{sec:conductor-cross-volume}

The Vol III Borcherds--Kac--Moody conductor
$\kappa_{\mathrm{BKM}} = c_\Lambda(0)/2$ is the cross-volume parallel
of the Platonic Conductor.  For any K3-fibered CY$_3$ with even
unimodular Mukai lattice $\Lambda$ of signature $(2, n)$ and Borcherds
character $c_\Lambda$:
\[
  \kappa_{\mathrm{BKM}}(\mathfrak{g}_\Lambda) \;=\; c_\Lambda(0) / 2,
\]
unconditional via Borcherds 1998 weight theorem (Vol III
\texttt{prop:bkm-weight-universal}).  The structural correspondence:
\begin{itemize}
 \item Vol I: $K(A) = -c_{\mathrm{ghost}}(\mathrm{BRST}(A))$ is the
       \emph{negative} ghost charge of any quasi-free BRST resolution.
 \item Vol III: $\kappa_{\mathrm{BKM}}(\mathfrak{g}_\Lambda) = c_\Lambda(0)/2$
       is the \emph{leading Fourier coefficient} of the Borcherds
       character.
\end{itemize}
Both invariants compute the trace of the universal Koszul reflection on
the appropriate centre: in Vol I the ghost-charge centre of the BRST
complex; in Vol III the Borcherds-character centre of the
BKM superalgebra.  The cross-volume bridge is the
\emph{Universal Trace Identity} \textup{(}conjectural, see the
cross-volume bridge chapter\textup{).}
