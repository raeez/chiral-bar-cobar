\chapter{Algebraic foundations and bar constructions}\label{ch:algebraic-foundations}

This chapter builds the workshop in which the rest of the volume operates.
The classical theory of quadratic algebras, Koszul pairs, and bar
constructions is assembled here not as an isolated algebraic toolkit but
as the first stratum of a deformation language that extends through
every later chapter.  The bar complex is the categorical logarithm;
the dual algebra is its leading coefficient; the twisting morphism
mediating between them is the visible endpoint of a Maurer--Cartan
story whose full habitat is the homotopy convolution algebra
$\Hom(\barB(\cA), \cA)$.

The homotopy upgrade matters from the outset.  In the operadic
literature (Loday--Vallette~\cite{LV12}), twisting morphisms, bar--cobar
adjunctions, and Koszul criteria are already organized by convolution
constructions whose Maurer--Cartan theory is intrinsically
homotopical, not merely strict.  A twisting morphism is not just an
arrow satisfying a differential equation: it is a degree-zero
cocycle in a convolution dg Lie algebra that is itself the strict
face of a larger $L_\infty$ package.  This chapter should therefore
be read as preparation for treating quadratic and nonquadratic, strict
and curved, algebraic and chiral constructions as successive strata
of one deformation language.

Which algebras have natural duals?  The answer is controlled by
a finiteness condition on the relations: when every relation involves
exactly two generators, the orthogonal complement of the relation
space defines a dual algebra, and a bar complex mediates between the
two.  This is classical Koszul duality (Priddy, Beilinson--Ginzburg--Soergel,
Loday--Vallette).  We recall its essential definitions, adapted
to the cohomological grading that will persist throughout the chiral
extension.\label{rem:algebraic-foundations-hms}

\section{Classical Koszul duality}
\label{sec:classical-koszul-foundation}

\subsection{Quadratic algebras and Koszul duality}

\begin{definition}[Quadratic algebra]
\label{def:quadratic-algebra}
\index{quadratic algebra|textbf}
A graded algebra $A = T(V)/I$ is \emph{quadratic} if:
\begin{enumerate}
\item $V$ is a graded vector space (generators)
\item $I \subset V \otimes V$ is a subspace of relations in degree 2
\item The defining ideal is $(I)$ generated by $I$
\end{enumerate}
We write $A = A(V, R)$ where $R \subset V \otimes V$ are the
relations.  The algebra carries a canonical \emph{augmentation}
$\varepsilon \colon A \twoheadrightarrow k$ sending all generators
to zero; the \emph{augmentation ideal} is
$\bar{A} = \ker(\varepsilon) = \bigoplus_{n \geq 1} V^{\otimes n}/(I)$.
The bar construction (Definition~\ref{def:bar-complex-algebraic})
is built from~$\bar{A}$.
\end{definition}

\begin{example}[Prototypical examples]
\label{ex:classical-quadratic-algebras}
\index{symmetric algebra}
\index{exterior algebra}
\index{universal enveloping algebra}
\index{tensor algebra}
The symmetric algebra $\mathrm{Sym}(V)$ has $R = \{v \otimes w - w \otimes v\}$; the exterior algebra $\Lambda(V)$ has $R = \{v \otimes w + w \otimes v\}$.  The universal enveloping algebra $U(\mathfrak{g})$ has:
\begin{align}
\text{Generators:} &\quad \mathfrak{g} \\
\text{Relations:} &\quad R_{\mathfrak{g}} = \{x \otimes y - y \otimes x - [x,y] : x,y \in \mathfrak{g}\}
\end{align}
\end{example}

\subsection{The Koszul dual coalgebra}

\begin{example}[Com--Lie: the coalgebra]\label{ex:com-lie-koszul-coalgebra}
\index{Koszul dual!Com--Lie coalgebra}
The symmetric algebra $\mathrm{Sym}(V)$ has relations
$R = \Lambda^2(V) \subset V^{\otimes 2}$ (antisymmetric tensors).
Its orthogonal complement is
$R^\perp = \mathrm{Sym}^2(V^*) \subset (V^*)^{\otimes 2}$.
The Koszul dual coalgebra is the sub-coalgebra
$\mathrm{Sym}^! \hookrightarrow T^c(sV^*)$
cogenerated by $s^2 R^\perp$; since the suspended generators $sV^*$
have degree~$1$, the Koszul sign rule makes graded-symmetric tensors
antisymmetric in the unsigned sense, and the result is the exterior
coalgebra $\Lambda^c(sV^*)$ with deconcatenation coproduct.
When $\dim V < \infty$, the graded linear dual is
\[
(\Lambda^c(sV^*))^{\vee} \;\cong\; \Lambda(sV)
\;\cong\; C^*_{\mathrm{CE}}(V),
\]
the Chevalley--Eilenberg cochain algebra.
The coalgebra $\Lambda^c(sV^*)$ and the algebra $\Lambda(sV)$
are related by linear dualizing, which requires
$\dim V < \infty$.
\end{example}

\begin{construction}[Koszul dual coalgebra]
\label{const:quadratic-dual}
\index{Koszul dual!quadratic|textbf}
Given a quadratic algebra $A = A(V,R)$ with $R \subset V^{\otimes 2}$,
define the orthogonal complement
\[
R^\perp = \{f \in V^* \otimes V^* :
\langle f, r \rangle = 0 \;\text{for all}\; r \in R\},
\qquad
\langle \alpha \otimes \beta, v \otimes w \rangle
= \langle \alpha, v \rangle \langle \beta, w \rangle.
\]
The \emph{Koszul dual coalgebra} is the sub-coalgebra
\[
A^! \;\hookrightarrow\; T^c(sV^*),
\]
defined as the maximal sub-coalgebra of the cofree coalgebra
$T^c(sV^*)$ whose degree-$2$ component lies in $s^2 R^\perp$.
Equivalently, the degree-$n$ component is the intersection
$(A^!)^{(n)} = \bigcap_{i+j=n-2}
(sV^*)^{\otimes i} \otimes s^2 R^\perp \otimes (sV^*)^{\otimes j}$
inside $(sV^*)^{\otimes n}$, with comultiplication given by
restricting deconcatenation.
This is a sub-coalgebra of the cofree coalgebra, not a quotient.
When each $(A^!)^{(n)}$ is finite-dimensional (in particular, when
$\dim V < \infty$), the graded linear dual $(A^!)^{\vee}$ is
the \emph{Koszul dual algebra}: a quotient of the free algebra
on the dual generators, with relations determined by $R$.
\end{construction}

\subsection{Koszul pairs: precise definition}

\begin{definition}[Koszul pair]
\label{def:koszul-pair-classical}
\index{Koszul pair!classical|textbf}
A pair of quadratic algebras $(A_1, A_2)$ is a \emph{Koszul pair} if:
\begin{enumerate}
\item $\bar{B}(A_1) \simeq A_2^!$ (as coalgebras)
\item $\bar{B}(A_2) \simeq A_1^!$ (as coalgebras)
\item $\Omega(\bar{B}(A_1)) \simeq A_1$ (cobar inverts bar)
\item $\Omega(\bar{B}(A_2)) \simeq A_2$ (cobar inverts bar)
\end{enumerate}
\end{definition}

\begin{definition}[Koszul locus]\label{def:koszul-locus}
\index{Koszul locus|textbf}
An augmented chiral algebra~$\cA$
(Definition~\ref{def:chiral-algebra}) lies on the
\emph{Koszul locus} if the bar coalgebra
$\barB_X(\cA)$ is \emph{formal}: quasi-isomorphic to its
cohomology, which is concentrated in bar degree~$1$.

Formality is the convergence condition for the categorical
logarithm.  Four equivalent formulations
(Theorem~\ref{thm:pbw-koszulness-criterion}):
\begin{enumerate}[label=\textup{(\alph*)}]
\item \emph{Spectral}: the PBW spectral sequence, induced by
  the conformal weight filtration on $\barB_X(\cA)$,
  degenerates at~$E_2$.
\item \emph{Resolution}: the cobar--bar counit
  $\Omega_X(\barB_X(\cA)) \xrightarrow{\sim} \cA$ is a
  quasi-isomorphism.
\item \emph{Concentration}: $H^*(\barB_X(\cA))$ is
  concentrated in bar degree~$1$.
\item \emph{Acyclicity}: the twisted tensor product
  $\barB_X(\cA) \otimes_\pi \cA^!$, formed with the universal
  twisting morphism~$\pi$
  (Proposition~\ref{prop:universal-twisting-adjunction}(i),
  Definition~\ref{def:twisted-tensor-product}), is acyclic.
\end{enumerate}
Off the Koszul locus, the bar complex is curved and lives in
the coderived category (Appendix~\ref{sec:coderived-models});
the logarithm has monodromy.
\end{definition}

\noindent
The chiral analogue is not defined by bar-cobar inversion directly.
Instead, Chapter~\ref{chap:koszul-pairs} introduces a chiral twisting
datum (Definition~\ref{def:chiral-twisting-datum}) and a Koszul twisting
morphism (Theorem~\ref{thm:fundamental-twisting-morphisms}) as antecedent
recognition data; bar-cobar identifications are then proved as
consequences on the Koszul locus.
The full chiral Koszul pair is
Definition~\ref{def:chiral-koszul-pair} (previewed in
\S\ref{subsec:chiral-koszul-pairs-precise} below);
the $\Eone$-chiral version appears in
Theorem~\ref{thm:e1-chiral-koszul-duality};
the modular (higher-genus) refinement is
Definition~\ref{def:modular-koszul-homotopy} (Chapter~\ref{chap:higher-genus}).

\begin{remark}[Two phenomena]
\label{rem:two-phenomena}
Conditions (1)--(2) establish \emph{Koszul duality}: bar of $A_1$ produces the dual coalgebra to $A_2$, and vice versa. Conditions (3)--(4) establish \emph{bar-cobar inversion}: $\Omega \circ \bar{B}$ is homotopy equivalent to the identity. These are logically independent.
\end{remark}

\subsection{Classical examples revisited}

\begin{theorem}[Classical Koszul pairs; \ClaimStatusProvedElsewhere{} \cite{Priddy70,BGS96,LV12}]
\label{thm:quadratic-koszul}
\index{Koszul duality!classical}
\index{Chevalley--Eilenberg complex}
The following are Koszul pairs in the sense of Definition~\ref{def:koszul-pair-classical}:
\begin{enumerate}
\item $(\text{Sym}(V), \Lambda(V^*))$ — commutative and exterior algebras
\item $(U(\mathfrak{g}), C^*_{\text{CE}}(\mathfrak{g}))$ — universal enveloping and Chevalley--Eilenberg cochains
\item $(T(V), T^c(V^*))$ — tensor algebra and tensor coalgebra
\end{enumerate}
Each pair satisfies all four conditions of Definition~\ref{def:koszul-pair-classical}.
\end{theorem}

The chiral enhancement appears in Chapter~\ref{ch:heisenberg-frame} (Heisenberg, double pole), \S\ref{sec:free-fermion} (free fermions, simple pole), and Chapter~\ref{chap:kac-moody-koszul} (Kac--Moody, mixed poles).

\subsection{Chiral Koszul pairs: preview}
\label{subsec:chiral-koszul-pairs-precise}

Definition~\ref{def:koszul-pair-classical} extends to chiral algebras (Definition~\ref{def:chiral-algebra}) via a two-step formulation: the \emph{construction} of bar and cobar exists for all augmented chiral algebras (Definition~\ref{def:chiral-twisting-datum}), while the Koszul \emph{recognition criterion} identifies when these constructions yield resolutions (Definition~\ref{def:chiral-koszul-morphism}; full treatment: Definition~\ref{def:chiral-koszul-pair}, Chapter~\ref{chap:koszul-pairs}).

The chiral Koszul pair structure has three layers
(cf.\ Definition~\ref{def:chiral-koszul-pair},
Chapter~\ref{chap:koszul-pairs}):
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Construction layer:} For every augmented chiral algebra
  $\mathcal{A}$, the bar construction
  $\bar{B}^{\mathrm{ch}}(\mathcal{A})$ and cobar construction
  $\Omega^{\mathrm{ch}}(\bar{B}^{\mathrm{ch}}(\mathcal{A}))$ exist as
  objects (Definition~\ref{def:chiral-twisting-datum}).

\item \emph{Recognition layer:} A chiral twisting datum
  $(\mathcal{A}, \mathcal{C}, \tau, F_\bullet)$ satisfying PBW
  degeneration and acyclicity of twisted tensor products
  provides \emph{antecedent} Koszul recognition data
  (Definition~\ref{def:chiral-koszul-morphism}).

\item \emph{Theorem layer:} For a chiral Koszul pair
  $(\mathcal{A}_1, \mathcal{A}_2)$, Theorem~A
  (Theorem~\ref{thm:bar-cobar-isomorphism-main}) proves
  $\bar{B}^{\mathrm{ch}}(\mathcal{A}_1) \simeq \mathcal{A}_2^!$
  and $\Omega^{\mathrm{ch}}(\mathcal{A}_i^!) \simeq \mathcal{A}_i$
  as \emph{consequences} of the recognition data, not as definitions.
\end{enumerate}
Off the Koszul locus, the bar-cobar object persists only in
the completed coderived category
(Appendix~\ref{sec:coderived-models}).%

\subsection{Twisting morphisms: the fundamental object}
\label{subsec:twisting-morphisms-fundamental}
\index{twisting morphism|textbf}

The single algebraic structure underlying bar-cobar duality, Koszul
resolutions, and the modular genus tower is the convolution algebra
$\mathrm{Hom}(C, A)$: at the dg~level it is a dg~Lie algebra whose
Maurer--Cartan elements are the twisting morphisms; at the full
homotopy level it carries an $L_\infty$-structure whose higher
brackets encode the homotopy transfer data
(\cite[Theorem~10.3.8]{LV12}).
More precisely, the convolution dg~Lie algebra is a strict model
$\Convstr(C,P)$ of the homotopy-invariant $L_\infty$-algebra
$\Convinf(C,P)$
(Theorem~\ref{thm:operadic-homotopy-convolution}); the strict model
suffices for all explicit computations in this monograph.
We begin with the classical dg theory
(\cite[\S2.1--2.3]{LV12}) before the chiral generalization
(Chapter~\ref{chap:koszul-pairs}); the hierarchy from dg through
$L_\infty$ to the modular quantum $L_\infty$ appears in
Remark~\ref{rem:convolution-hierarchy}.

\begin{definition}[Convolution dg Lie algebra]
\label{def:convolution-dg-lie}
\index{convolution dg Lie algebra|textbf}
Let $(C, \Delta, d_C)$ be a conilpotent dg coalgebra and
$(A, \mu, d_A)$ a dg algebra.  The \emph{convolution algebra}
$\mathrm{Hom}(C, A)$ is the graded vector space of all linear maps
$C \to A$, equipped with:
\begin{enumerate}[label=\textup{(\roman*)}]
\item the \emph{convolution product}
  $f \star g := \mu \circ (f \otimes g) \circ \Delta$;
\item the \emph{differential}
  $\partial(f) := d_A \circ f - (-1)^{|f|} f \circ d_C$;
\item the \emph{Lie bracket}
  $[f, g] := f \star g - (-1)^{|f|\,|g|}\, g \star f$.
\end{enumerate}
The triple $(\mathrm{Hom}(C, A), [-, -], \partial)$
is a dg Lie algebra \cite[Proposition~2.1.1]{LV12}.
\end{definition}

\begin{remark}[Strict model versus homotopy-invariant structure]
\label{rem:conv-strict-vs-homotopy}
The convolution dg~Lie algebra of Definition~\ref{def:convolution-dg-lie}
is a \emph{strict model} of an underlying homotopy-invariant object: the
$L_\infty$ convolution algebra $\operatorname{Conv}_\infty(\cC, \cP)$
whose higher brackets $\ell_k$ ($k \geq 3$) arise from the operadic
composition and the $A_\infty$/$L_\infty$ homotopy transfer
\cite[\S10.3]{LV12}.  The two structures have canonically isomorphic
Maurer--Cartan moduli:
\[
\mathrm{MC}\bigl(\operatorname{Conv}_{\mathrm{str}}(\cC,\cP)\bigr)
\;\cong\;
\mathrm{MC}\bigl(\operatorname{Conv}_\infty(\cC,\cP)\bigr),
\]
so for \emph{computing} Maurer--Cartan elements --- which is the primary
use throughout Part~I --- the strict dg~Lie model suffices.  The full
$L_\infty$ structure becomes essential for three purposes:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Homotopy transfer}: preserving MC data across
  quasi-isomorphisms (the reason bar-cobar preserves quasi-isos);
\item \emph{Formality}: proving that the deformation complex is formal
  in the sense of Kontsevich--Soibelman \cite{KontsevichSoibelman};
\item \emph{Gauge equivalence}: classifying deformations up to
  $L_\infty$ homotopy, not merely up to strict isomorphism.
\end{enumerate}
The hierarchy is:
\[
\text{modular } L_\infty \text{ object}
\;\xrightarrow{\;\text{strictify}\;}
\text{dg Lie model}
\;\xrightarrow{\;\text{MC}\;}
\text{algebra structures and genus tower}.
\]
All constructions in this monograph are functorial with respect to the
full $L_\infty$ structure; the strict model is the computational
avatar.
\end{remark}

\begin{theorem}[Convolution = master object identification;
\ClaimStatusProvedElsewhere]
\label{thm:convolution-master-identification}
\index{convolution!master object identification|textbf}
\index{master object!convolution identification}
For any cooperadic/operadic pair $(\cC, \cP)$ and twisting
morphism~$\alpha\colon \cC \to \cP$, the convolution dg~Lie algebra
$\Convstr(\cC, \cP)$ of
Definition~\textup{\ref{def:convolution-dg-lie}}
is the strict model of the convolution $sL_\infty$-algebra
\[
\Convstr(\cC, \cP)
\;\hookrightarrow\;
\operatorname{hom}_\alpha(\cC,\, \cP)
\]
of Robert-Nicoud--Wierstra~\textup{\cite[Theorem~4.1]{RNW19}}.
The $sL_\infty$ higher brackets $\ell_k$ ($k \geq 3$) on
$\operatorname{hom}_\alpha$ arise from the homotopy transfer through
the $\operatorname{Hom}$~functor
\textup{(}Loday--Vallette~\textup{\cite[Theorem~10.3.8]{LV12})}.
In particular, the modular convolution algebra
$\mathfrak{g}_\cA^{\mathrm{mod}}$ of
Definition~\textup{\ref{def:modular-convolution-dg-lie}}
is the chiral specialization with
$\cC = \cC^{\textup{!`}}_{\mathrm{ch}}$,
$\cP = \cP^{\mathrm{ch}}$.
\end{theorem}

\begin{proof}[References]
The general operadic convolution identification is
\cite[Theorem~4.1]{RNW19}; the strict model comparison is
\cite[Theorem~10.3.8]{LV12}.  The chiral specialization follows from
identifying the chiral operad as Koszul with the standard chiral
twisting morphism (Definition~\ref{def:twisting-morphism}).
\end{proof}

\begin{corollary}[MC element = twisting morphism;
\ClaimStatusProvedElsewhere]
\label{cor:theta-twisting-morphism}
\index{Maurer--Cartan element!twisting morphism identification}
The universal class $\Theta_\cA \in
\mathrm{MC}(\mathfrak{g}_\cA^{\mathrm{mod}})$ is a twisting
morphism:
\[
\mathrm{MC}\!\bigl(\operatorname{hom}_\alpha(\cC,\, \cP)\bigr)
\;\cong\;
\mathrm{Tw}_\alpha(\cC,\, \cP).
\]
The Chriss--Ginzburg principle
\textup{(}Proposition~\textup{\ref{prop:chriss-ginzburg-structure}):}
the chiral algebra structure IS the MC element\textup{)} is the chiral
instance of this general bijection
\textup{\cite[Theorem~4.6]{RNW19}}.
The all-arity master equation $\nabla_H(\mathrm{Sh}_r) + o^{(r)} = 0$
is the MC equation $\partial(\Theta) + \tfrac{1}{2}[\Theta, \Theta] = 0$
projected to arity~$r$ via the weight filtration.
\end{corollary}

\begin{proof}[References]
\cite[Theorem~4.6]{RNW19} and \cite[Theorem~6.5.7]{LV12}.
\end{proof}

\begin{definition}[Twisting morphism]
\label{def:twisting-morphism}
\index{twisting morphism!definition|textbf}
\index{Maurer--Cartan element|textbf}
A \emph{twisting morphism} $\tau\colon C \to A$ is a
degree-$+1$ linear map (degree $-1$ in the homological convention
of~\cite{LV12}) satisfying the \emph{Maurer--Cartan equation} in
the convolution algebra
(Definition~\ref{def:convolution-dg-lie}):
\begin{equation}\label{eq:twisting-mc}
\partial(\tau) + \tau \star \tau = 0,
\quad\text{i.e.,}\quad
d_A \circ \tau + \tau \circ d_C
+ \mu_A \circ (\tau \otimes \tau) \circ \Delta_C = 0.
\end{equation}
The set of all twisting morphisms is
$\mathrm{Tw}(C, A) := \mathrm{MC}(\mathrm{Hom}(C, A))$
\cite[\S2.1]{LV12}.
\end{definition}

\begin{definition}[Twisted tensor products]
\label{def:twisted-tensor-product}
\index{twisted tensor product|textbf}
For a twisting morphism $\tau\colon C \to A$, the
\emph{left twisted tensor product} $A \otimes_\tau C$ is the
graded vector space $A \otimes C$ with differential
\begin{equation}\label{eq:twisted-differential}
d_\tau(a \otimes c)
\;=\;
d_A(a) \otimes c
\;+\; (-1)^{|a|}\, a \otimes d_C(c)
\;+\; \sum (-1)^{|a|}\, a \cdot \tau(c_{(1)}) \otimes c_{(2)},
\end{equation}
where $\Delta(c) = \sum c_{(1)} \otimes c_{(2)}$ is the Sweedler
notation for the coproduct.  The \emph{right twisted tensor product}
$C \otimes_\tau A$ has differential
$d_\tau(c \otimes a) = d_C(c) \otimes a + (-1)^{|c|}\,
c \otimes d_A(a) + \sum c_{(1)} \otimes \tau(c_{(2)}) \cdot a$
(with Koszul signs; \cite[2.1.5]{LV12}).  The MC
equation~\eqref{eq:twisting-mc} is equivalent to
$d_\tau^2 = 0$ \cite[Lemma~2.1.4]{LV12}.
\end{definition}

\begin{proposition}[Universal twisting morphisms
{\cite{LV12}}; \ClaimStatusProvedElsewhere]\label{prop:universal-twisting-adjunction}
\index{twisting morphism!universal|textbf}
\index{bar-cobar adjunction!via twisting morphisms}
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Projection.}
  Project to bar degree~$1$ and desuspend to get
  $\pi\colon B(A) \to A$.
  This is a twisting morphism, and it is \emph{universal}:
  every twisting morphism $\tau\colon C \to A$ factors
  uniquely through~$\pi$ via a dg coalgebra map
  $g_\tau\colon C \to B(A)$
  \cite[Theorem~2.2.4]{LV12}.

\item \emph{Inclusion.}
  The composite
  $\iota\colon C \xrightarrow{s^{-1}} s^{-1}C
  \hookrightarrow \Omega(C)$
  (desuspension into the free algebra $\Omega(C) = T(s^{-1}C)$)
  is a twisting morphism.  It is \emph{universal}:
  every $\tau\colon C \to A$ factors uniquely as
  $\tau = f_\tau \circ \iota$ for a dg algebra map
  $f_\tau\colon \Omega(C) \to A$
  \cite[Theorem~2.2.7]{LV12}.

\item \emph{Natural bijection.}
  The two universal properties yield a natural bijection
  \begin{equation}\label{eq:tw-adjunction}
  \mathrm{Hom}_{\mathrm{dg\,CoAlg}}(C,\, B(A))
  \;\cong\;
  \mathrm{Tw}(C,\, A)
  \;\cong\;
  \mathrm{Hom}_{\mathrm{dg\,Alg}}(\Omega(C),\, A).
  \end{equation}
  This is the \emph{bar-cobar adjunction}: the bar functor $B$ is
  right adjoint to the cobar functor $\Omega$, and the set of
  twisting morphisms $\mathrm{Tw}(C, A)$ is the adjunction
  bifunctor \cite[Theorem~2.2.9]{LV12}.

\item \emph{Acyclicity.}
  Both canonical twisted tensor products are acyclic:
  $B(A) \otimes_\pi A \simeq \mathbb{k}$
  and $A \otimes_\pi B(A) \simeq \mathbb{k}$
  \cite[Corollary~2.2.5]{LV12}.
\end{enumerate}
\end{proposition}

\begin{remark}[The twisting morphism as geometric propagator]
\label{rem:twisting-as-propagator}
\index{twisting morphism!geometric realization}
\index{propagator!as twisting morphism}
In the chiral setting (Chapter~\ref{chap:bar-cobar}), the universal
twisting morphism $\pi\colon \barB_X(\cA) \to \cA$ is geometrically
realized as
\begin{equation}\label{eq:twisting-propagator}
\pi(c)(z)
\;=\;
\int_{\overline{C}_2(X)}
\mathrm{ev}^* c(w) \;\wedge\; S(z, w)\, dw,
\end{equation}
where $S(z,w)$ is the Szeg\H{o} propagator on the curve~$X$.
On a smooth curve~$X$, the M-level realization of the abstract
twisting morphism~$\pi$ is the Szeg\H{o} propagator integral.
The MC equation~\eqref{eq:twisting-mc} for~$\pi$ is equivalent to
$d_{\mathrm{bar}}^2 = 0$
(Theorem~\ref{thm:bar-nilpotency-complete}): the Fay trisecant
identity (Proposition~\ref{prop:fay-trisecant}) and the Arnold
relations (Theorem~\ref{thm:arnold-relations}) are the analytic
and combinatorial incarnations of the single algebraic MC equation.

The chiral generalization of the full bijection~\eqref{eq:tw-adjunction}
is the chiral twisting datum
(Definition~\ref{def:chiral-twisting-datum}) and the fundamental
theorem of chiral twisting morphisms
(Theorem~\ref{thm:fundamental-twisting-morphisms}).
\end{remark}

\cite[Theorem~10.3.8]{LV12}.
At genus $g \geq 1$, the quantum $L_\infty$ operations $\ell_n^{(g)}$
come from stable-graph Feynman amplitudes
(Theorem~\ref{thm:modular-quantum-linfty}).
For Koszul chiral algebras, the operadic convolution $L_\infty$ is
\emph{formal} (quasi-isomorphic to its cohomology with trivial higher
brackets), so the strict dg~Lie model suffices for computations; the
full $L_\infty$ structure becomes essential for non-Koszul algebras
and for homotopy-invariant constructions.

The $[1]$-shift in rows 2--3 makes the coderivation bracket
degree~$0$, matching the tangent Lie algebra convention.
The rows are related by successive enrichments:
row~1 $\to$ row~2 via the bar construction (coalgebra
becomes bar coalgebra);
row~2 $\to$ row~3 by completing the bar coalgebra and
restricting to cyclic coderivations
($\operatorname{CoDer}^{\mathrm{cyc}} \subset
\operatorname{CoDer}$);
row~3 $\to$ row~4 by tensoring with chains on
$\overline{\mathcal{M}}_{g,n}$
(Definition~\ref{def:modular-deformation-complex}).
The MC equation specializes at each level as shown in the
fourth column.  Row~4 is the \emph{master object}: rows~1--3 are
obtained by restricting to genus~$0$ ($\hbar \to 0$) and truncating
to lower homotopy operations.  The shadow Postnikov tower
$\Theta^{\leq r}$ in row~4 is the primary proved content;
the five main theorems are projections of its scalar level
$\Theta^{\leq 2} = \kappa$
(Remark~\ref{rem:unifying-principle}).

Each row admits a \emph{homotopy upgrade}: the dg~Lie bracket
$[-,-]$ is the shadow of a full $L_\infty$-structure whose higher
brackets $\ell_n$ ($n \ge 3$) arise from homotopy transfer through
the convolution Hom functor
(\cite[Theorem~10.3.8]{LV12}, van~der~Laan~\cite{vdL03}).  At the
modular level, this produces a \emph{quantum $L_\infty$-algebra}
with operations $\ell_n^{(g)}$ indexed by stable type $(g,n)$
(Theorem~\ref{thm:modular-quantum-linfty}).  The dg~level suffices
for the existence of MC elements; the full homotopy level is needed
for transfer to minimal models and for gauge equivalences
(Remark~\ref{rem:full-homotopy-why}).
\end{remark}

\begin{theorem}[Operadic identification of the convolution algebra;
\ClaimStatusProvedElsewhere]
\label{thm:operadic-homotopy-convolution}%
\index{convolution dg Lie algebra!operadic identification}%
\index{Robert-Nicoud--Wierstra theorem}%
Let $\cC$ be a cooperad, $\cP$ an operad, and
$\alpha\colon \cC \to \cP$ an operadic twisting morphism
$($i.e.\ $\alpha \in \operatorname{Tw}(\cC,\cP))$.
Then \textup{(Robert-Nicoud--Wierstra \cite{RNW19}, Theorem~4.1)}:
\begin{enumerate}[label=\textup{(\roman*)}]
\item The convolution operad
  $\operatorname{hom}(\cC,\cP)$ carries a natural
  $s\mathscr{L}_\infty$-algebra structure, and the assignment
  $\alpha \mapsto \mathsf{W}_\alpha$ defines a bijection
  \begin{equation}\label{eq:rnw-bijection}
    \operatorname{Tw}(\cC,\cP)
    \;\cong\;
    \operatorname{hom}_{\mathrm{Op}}(s\mathscr{L}_\infty,\;
      \operatorname{hom}(\cC,\cP)).
  \end{equation}
\item For $D$ a conilpotent $\cC$-coalgebra and $A$ a $\cP$-algebra,
  the chain complex $\operatorname{hom}(D, A)$ carries a natural
  $s\mathscr{L}_\infty$-algebra structure
  $\operatorname{hom}^\alpha(D,A)$ whose Maurer--Cartan elements
  are precisely the twisting morphisms relative to~$\alpha$:
  \begin{equation}\label{eq:rnw-mc-tw}
    \operatorname{MC}(\operatorname{hom}^\alpha(D,A))
    \;\cong\;
    \operatorname{Tw}_\alpha(D,A).
  \end{equation}
\item The bifunctor
  $\operatorname{hom}^\alpha\colon
  (\text{conil.\ }\cC\text{-cog})^{\mathrm{op}} \times
  \cP\text{-alg} \to s\mathscr{L}_\infty\text{-alg}$
  extends to $\infty_\alpha$-morphisms in either slot, but not
  both simultaneously \textup{(\cite{RNW19}, Theorem~5.1 and \S6)}.
\end{enumerate}
Applied to the chiral setting: the modular convolution dg~Lie algebra
$\gAmod$ of
Definition~\textup{\ref{def:modular-convolution-dg-lie}} is the
chiral instance of $\operatorname{hom}^\alpha(D,A)$ with
$\alpha = \tau_{\mathrm{ch}}$ the chiral twisting morphism
$($Theorem~\textup{\ref{thm:fundamental-twisting-morphisms}}$)$,
$D = \barB^{\mathrm{ch}}(\cA)$, and
$A = \cA^{!}$.  The master MC element
$\Theta_\cA \in \operatorname{MC}(\gAmod)$ is a twisting
morphism relative to~$\tau_{\mathrm{ch}}$,
and the Chriss--Ginzburg principle
$($Proposition~\textup{\ref{prop:chriss-ginzburg-structure}}$)$ is the
chiral specialization of~\eqref{eq:rnw-mc-tw}.
\end{theorem}

\begin{proof}
The identification of the dg~Lie structure on $\gAmod$ with
$\operatorname{hom}^\alpha(D,A)$ is immediate from
comparing Definition~\ref{def:modular-convolution-dg-lie}
(the product over stable types weighted by
$C_\bullet(\overline{\cM}_{g,n+1})$) with the general
construction \cite[Definition~4.4]{RNW19} applied to the
chiral cooperad and operad.  The bijection~\eqref{eq:rnw-mc-tw}
follows from \cite[Theorem~4.6]{RNW19}, which was proved
independently in \cite[Proposition~2.9]{RN18} and
\cite[Proposition~7.1]{Wie19}.
The one-slot functoriality is \cite[Corollary~5.4]{RNW19}.
\end{proof}

\subsection{The Gui--Li--Zeng quadratic duality framework}
\label{subsec:gui-li-zeng-framework}

\begin{definition}[Gui--Li--Zeng setup]
\label{def:glz}
Gui--Li--Zeng~\cite{GLZ22} define Koszul duality for chiral algebras through four ingredients.

\emph{Chiral quadratic data.}
A pair $(N, P)$ where $N$ is a sheaf of generators (chiral vector space) and $P \subset j_* j^* (N \boxtimes N)$ is a subsheaf of quadratic relations.
The quadratic chiral algebra is
\[\mathcal{A}(N, P) = \frac{\mathcal{A}(N)}{(P)}\]
where $\mathcal{A}(N)$ is the free chiral algebra on $N$.

\emph{Dualizable quadratic data.}
$(N, P)$ is \emph{dualizable} if $(s^{-1}N^{\vee}\omega^{-1}, P^{\perp})$ is also a chiral
quadratic datum, where $N^{\vee}\omega^{-1}$ is the dual with twist by inverse canonical bundle and $P^{\perp}$ is the chiral annihilator defined by
\[\mu(\langle P \otimes \omega_{X^2}, P^{\perp} \otimes s^2\omega_{X^2}\rangle) = 0\]
under the unit chiral operation $\mu$.

\emph{Quadratic dual.}
For dualizable $(N, P)$, the quadratic dual is
\[\mathcal{A}^! = \mathcal{A}(s^{-1}N^{\vee}\omega^{-1}, P^{\perp}).\]

\emph{Maurer--Cartan correspondence.}
There is a bijection
\[\operatorname{Hom}(\mathcal{A}, \mathcal{B}) \leftrightarrow MC(\mathcal{A}^! \otimes \mathcal{B})\]
between morphisms of chiral algebras and Maurer--Cartan elements in the tensor product.
\end{definition}

\begin{proposition}[Comparison: our approach vs GLZ; \ClaimStatusProvedHere]
\label{prop:comparison-our-glz}
\index{GLZ framework!comparison}
The geometric bar-cobar framework of this monograph and the algebraic
framework of Gui--Li--Zeng~\cite{GLZ22} are related as follows:
\begin{enumerate}
\item \emph{Quadratic case agreement.}
   For a quadratic chiral algebra $\mathcal{A}(N,P)$ (generators~$N$,
   relations $P \subset j_*j^*(N \boxtimes N)$), our bar construction
   gives $\bar{B}^{\mathrm{ch}}(\mathcal{A}) \simeq
   \mathcal{A}(s^{-1}N^{\vee}\omega^{-1}, P^{\perp})^!$,
   which coincides with the GLZ dual coalgebra.
\item \emph{Non-quadratic extension.}
   Our framework extends to non-quadratic algebras via the full
   configuration space bar complex (Chapter~\ref{chap:bar-cobar}),
   which replaces: quadratic relations $P$ with OPE structure of
   arbitrary pole order; the annihilator $P^{\perp}$ with residue
   extraction at collision divisors; algebraic dualization with
   geometric Poincar\'e--Verdier duality.
\item \emph{Maurer--Cartan elements.}
   The GLZ Maurer--Cartan element
   $\alpha \in MC(\mathcal{A}^! \otimes \mathcal{B})$
   corresponds to our twisting morphism
   $\tau\colon \mathcal{A}^! \to \mathcal{B}$,
   realized geometrically as an integration kernel on
   $\overline{C}_2(X)$:
   $\tau(c)(z) = \int_{\overline{C}_2(X)}
   \mathrm{ev}^* c(w) \wedge K(z, w)$.
\item \emph{Curved structures.}
   \index{$A_\infty$-algebra|textbf}
   GLZ's framework handles curved $A_{\infty}$ algebras through
   Maurer--Cartan deformations; our configuration space approach
   realizes the same data as genus-by-genus quantum corrections
   (curvature $=$ higher-genus corrections; Maurer--Cartan equation
   $=$ Stokes' theorem on $\overline{C}_n(X)$).
\end{enumerate}
\end{proposition}

\begin{proof}
For~(i): the bar complex $\bar{B}^{\mathrm{ch}}(\mathcal{A}(N,P))$
has generators $s^{-1}N^\vee$ in bar degree~1 and relations
$P^\perp \subset s^{-1}N^\vee \otimes s^{-1}N^\vee$ in bar degree~2,
extracted by the double-residue operation on $\overline{C}_3(X)$.  The
$\omega^{-1}$-twist arises from the desuspension convention
(Appendix~\ref{app:signs}, \S\ref{app:sign-conventions}).  Since the algebra is quadratic,
the bar complex is generated in degrees $\leq 2$, so the bar
coalgebra is the sub-coalgebra of the cofree coalgebra
$T^c(s^{-1}N^\vee \omega^{-1})$ cogenerated by $P^\perp$---which is
the definition of the GLZ dual coalgebra
(\cite{GLZ22}, Proposition~3.2).  For~(iii): the MC equation
$d\alpha + \alpha \star \alpha = 0$ translates under the integration
kernel to $d\tau + \mu_{\mathcal{B}} \circ (\tau \otimes \tau)
\circ \Delta_{\mathcal{A}^!} = 0$, which is the twisting morphism
condition (Definition~\ref{def:twisting-morphism}).  Items~(ii)
and~(iv) are descriptions of the respective frameworks' scopes.
\end{proof}

\begin{remark}[Complementarity of the two approaches]
\label{rem:advantages-comparison}
The GLZ algebraic framework provides conceptual clarity for quadratic
algebras; the geometric framework of this monograph extends to
non-quadratic examples (Virasoro, $\mathcal{W}$-algebras, Yangian)
via configuration space integrals, at the cost of more involved
compactification and convergence analysis.
\end{remark}

\section{Symmetric sequences and operads}

We work in $\mathcal{V} = \mathrm{Ch}_\mathbb{C}$, cochain complexes over $\mathbb{C}$ with cohomological grading ($|d| = +1$) and Koszul sign rule (Convention~\ref{conv:standing-assumptions}).

\begin{definition}[Symmetric sequence]
A \emph{symmetric sequence} is a collection $P = \{P(n)\}_{n \geq 0}$ where each $P(n)$ is an object of $\cV$ equipped with a right action of the symmetric group $S_n$. Morphisms of symmetric sequences are collections of $S_n$-equivariant maps. When $\cV$ carries a differential structure, we require that the $S_n$-action commutes with differentials.
\end{definition}
 
\begin{definition}[Composition product]
For symmetric sequences $A$ and $B$, their composition product is defined by:
\[
(A \circ B)(n) = \bigoplus_{k \geq 0} A(k) \otimes_{S_k} \left( \bigoplus_{i_1 + \cdots + i_k = n} \Ind_{S_{i_1} \times \cdots \times S_{i_k}}^{S_n}(B(i_1) \otimes \cdots \otimes B(i_k)) \right)
\]
where $\Ind$ denotes the induced representation functor, using the block diagonal embedding 
\[
S_{i_1} \times \cdots \times S_{i_k} \hookrightarrow S_n
\]
that acts on $\{1, \ldots, i_1\} \sqcup \{i_1 + 1, \ldots, i_1 + i_2\} \sqcup \cdots \sqcup \{i_1 + \cdots + i_{k-1} + 1, \ldots, n\}$.
\end{definition}
 
The composition product is associative up to canonical isomorphism, with unit given by the symmetric sequence $\mathbb{I}$ with $\mathbb{I}(1) = \C$ and $\mathbb{I}(n) = 0$ for $n \neq 1$.
 
\section{Chiral algebras and non-abelian Poincaré duality}
\label{sec:chiral-NAP}

\subsection{Factorization as local-to-global}

\begin{remark}\label{princ:factorization-locality}
The factorization axiom $\mathcal{A}(U \sqcup V) \simeq \mathcal{A}(U) \otimes \mathcal{A}(V)$ is excision for NAP duality: $\int_{M_1 \sqcup M_2} A = \int_{M_1} A \otimes \int_{M_2} A$.
\end{remark}

\subsection{Ran space and universal recipients}

\begin{definition}[Ran space]\label{def:ran-space}
The Ran space of X is:
\[\text{Ran}(X) = \varinjlim_{I \in \text{fSet}^{\text{surj}}} X^I\]
the space of finite non-empty subsets of $X$, where the colimit is taken over the category of non-empty finite sets with surjections. The surjections $I \twoheadrightarrow J$ induce diagonal embeddings $X^J \hookrightarrow X^I$, identifying configurations where points collide.

A factorization algebra on X is equivalent to a constructible sheaf on $\text{Ran}(X)$ satisfying compatibility conditions.
\end{definition}

\begin{definition}[Chiral algebra]\label{def:chiral-algebra}
\index{chiral algebra|textbf}
A \emph{chiral algebra} on a smooth algebraic curve~$X$ is a
right $\mathcal{D}_X$-module~$\cA$ equipped with a
\emph{chiral product}
\[
\mu \colon j_* j^*(\cA \boxtimes \cA) \longrightarrow
\Delta_!(\cA),
\]
where $j \colon X^2 \setminus \Delta \hookrightarrow X^2$ is the
complement of the diagonal and
$\Delta_! = \Delta_* \circ \mathcal{D}_{X \to X^2}$ is the
$\mathcal{D}$-module pushforward.  The three axioms are:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Locality}: $\mu$ factors through $j_* j^*$
  --- the product is meromorphic with poles only along the
  diagonal.
\item \emph{Associativity}: the Borcherds identity
  --- the unique algebraic condition under which the FM bar
  construction produces $d^2 = 0$
  (\S\ref{sec:selection-principle}).
\item \emph{Unit}: an embedding
  $\omega_X \hookrightarrow \cA$ (the vacuum).
\end{enumerate}
Axiom~(ii) is the selection principle: it is not imposed by fiat
but forced by the boundary stratification of
$\overline{C}_n(X)$ (\S\ref{sec:selection-principle}).

An \emph{augmented} chiral algebra carries a splitting
$\varepsilon \colon \cA \to \omega_X$ of the unit; the
augmentation ideal is
$\bar{\cA} = \ker(\varepsilon)$.
The operator product expansion
$a(z)\,b(w) \sim \sum_{n \geq 0} c_n(w)/(z-w)^{n+1}$
is the mode expansion of~$\mu$.  The \emph{chiral bracket}
$\{a, b\} = c_0$ extracts the simple-pole residue; the full
product~$\mu$ involves all pole orders
(Convention~\ref{conv:product-vs-bracket}).
The complete formulation, with the twisting datum
needed for the Koszul pair structure, is
Definition~\ref{def:chiral-twisting-datum}
(Chapter~\ref{chap:koszul-pairs}).
\end{definition}

\begin{theorem}[Chiral algebras on Ran space; \ClaimStatusProvedElsewhere]\label{thm:chiral-ran}
\textup{(Beilinson--Drinfeld \cite[Chapter~3]{BD04})}

A chiral algebra~$\cA$ on~$X$
(Definition~\ref{def:chiral-algebra}) determines a
$\mathcal{D}$-module $\mathcal{F}_{\cA}$ on
$\operatorname{Ran}(X)$ satisfying:
\begin{enumerate}
\item Factorization: $\mathcal{F}_{\mathcal{A}}(S \sqcup T) = \mathcal{F}_{\mathcal{A}}(S) \otimes \mathcal{F}_{\mathcal{A}}(T)$
\item Compatibility with embeddings
\item D-module structure encoding OPEs
\end{enumerate}

The bar-cobar duality acts on these sheaves on Ran space, realizing NAP duality for factorization algebras.
\end{theorem}

\begin{definition}[Operad]
\index{operad|textbf}
An \emph{operad} $P$ is a monoid for the composition product: composition maps $\gamma : P(k) \otimes P(i_1) \otimes \cdots \otimes P(i_k) \to P(i_1 + \cdots + i_k)$ and unit $\eta : \mathbb{I} \to P(1)$, satisfying associativity and $S_n$-equivariance. In $\cV = \mathrm{Ch}_\mathbb{C}$, all structure maps are chain maps.
\end{definition}
 
\begin{definition}[Cooperad]
\index{cooperad|textbf}
A \emph{cooperad} is a comonoid for the composition product, with structure maps dual to those of an operad. Explicitly, we have decomposition maps $\Delta : C(n) \to (C \circ C)(n)$ and a counit $\epsilon : C \to \mathbb{I}$ satisfying coassociativity and coequivariance axioms.
\end{definition}
 
\begin{example}[Endomorphism operad]
For any object $V \in \cV$, the endomorphism operad $\End_V$ has 
\[
\End_V(n) = \Hom_\cV(V^{\otimes n}, V)
\]
with composition given by substitution of multilinear operations. This is the fundamental example motivating the general theory.
\end{example}
 
\section{The cotriple bar construction}
 
An adjunction $F \dashv U : \mathcal{A} \rightleftarrows \mathcal{B}$ yields a comonad $G = FU$ on $\mathcal{B}$ with counit $\epsilon : FU \to \mathrm{id}$ and comultiplication $\delta : FU \to FUFU$.

\begin{definition}[Cotriple bar resolution]
The cotriple bar resolution of $B \in \mathcal{B}$ is the simplicial object:
\[
B^G_\bullet(B) : \cdots \rightrightarrows (FU)^3B \rightrightarrows (FU)^2B \rightrightarrows FUB \to B
\]
with face maps $d_i = (FU)^{i-1} \cdot \epsilon \cdot (FU)^{n-i-1}$ for $0 \le i \le n$ (applying the counit at position $i$) and degeneracy maps $s_i$ inserting the unit at position $i$.
\end{definition}
 
\begin{example}[Operadic bar construction]
For an operad $P$, the free-forgetful adjunction $F_P \dashv U : P\text{-Alg} \rightleftarrows \cV$ yields the classical bar construction $\barB^P_\bullet(A)$ for any $P$-algebra $A$. Explicitly:
\[
\barB^P_n(A) = P \circ \cdots \circ P \circ A \quad \text{($n$ copies of $P$)}
\]
This agrees with the construction via iterated insertions of operations from $P$. The differential is the alternating sum of face maps.
\end{example}
 
\subsection{The fundamental bar-cobar isomorphism}
\index{cobar construction!algebraic|textbf}

\begin{remark}[Operadic Koszul pairs]
The operadic version of Definition~\ref{def:koszul-pair-classical} replaces algebras by operads, generator duality by collection duality, and relation orthogonality by the operadic Koszul dual cooperad $P^! = \text{maximal sub-cooperad of } \mathcal{F}^c(V^*)$ cogenerated by $R^\perp$ (see Chapter~\ref{chap:koszul-pairs} for the chiral analog).
\end{remark}

\begin{example}[Com-Lie: the prototypical Koszul pair]
For the commutative and Lie operads:
\begin{itemize}
\item Generators: $\mu \in \text{Com}(2)$ (commutative product) and $\ell \in \text{Lie}(2)$ (Lie bracket)
\item Pairing: $\langle \mu, \ell \rangle = 1$ (canonical pairing between symmetry and antisymmetry)
\item Bar-cobar isomorphisms:
\begin{align*}
\barB(\text{Com}) &\simeq \text{co}\text{Lie} \quad \text{(partition complex gives Lie cooperad)} \\
\barB(\text{Lie}) &\simeq \text{co}\text{Com} \quad \text{(Chevalley--Eilenberg gives Com cooperad)} \\
\Omega(\text{co}\text{Lie}) &\simeq \text{Com} \quad \text{(cobar reconstructs commutative structure)} \\
\Omega(\text{co}\text{Com}) &\simeq \text{Lie} \quad \text{(cobar reconstructs Lie structure)}
\end{align*}
\end{itemize}

Concretely: the bar complex of the commutative operad is the chain complex of the partition lattice, whose homology is precisely the Lie operad (with sign).
\end{example}

\begin{remark}[Two levels of Koszul duality]
\label{rem:operadic-vs-algebraic-kd}
\index{Koszul duality!operadic vs.\ algebraic}
Koszul duality operates at two distinct levels:
\begin{enumerate}[label=\textup{(\alph*)}]
\item \emph{Operadic level}: the operad~$P$ itself has a Koszul
  dual cooperad $P^! = P^{{\text{\textexclamdown}},*}$
  (e.g.\ $\mathrm{Com}^! = \mathrm{Lie}$).  This governs the
  \emph{type} of algebraic structure.
\item \emph{Algebraic level}: a $P$-algebra~$A$ has a bar
  coalgebra $\barB(A)$ (the Koszul dual coalgebra;
  Convention~\ref{conv:bar-coalgebra-identity}).  This governs the
  \emph{specific} duality for a given algebra.
\end{enumerate}
In the chiral setting, both levels are active simultaneously.
The operadic structure supplies the geometry
(FM compactification, Arnold relation, logarithmic forms);
the algebraic structure supplies the OPE data (generators,
relations, pole orders).  The bar construction weaves them
together: it places OPE data on the operadic geometry and
extracts residues.  The result is a factorization coalgebra
that encodes both the type of structure (commutative,
associative, Lie) and the specific algebra~$\cA$.
Inversion holds on the Koszul locus
(Definition~\ref{def:koszul-locus}).
\end{remark}

\section{The operadic bar-cobar duality}
 
For an augmented operad $P$ with augmentation $\epsilon : P \to \mathbb{I}$, we construct the bar and cobar functors:
 
\begin{definition}[Operadic bar construction]
\index{bar construction!algebraic|textbf}
\index{suspension}
\index{cofree coalgebra}
\index{conilpotent!filtration|textbf}
The bar construction $\barB(P)$ is the cofree cooperad on the desuspension $s^{-1}\bar{P}$ (where $\bar{P} = \ker(\epsilon)$ is the augmentation ideal), in the cohomological convention ($|d| = +1$).  Explicitly:
\[
\barB(P) = T^c(s^{-1}\bar{P}) = \bigoplus_{n \geq 0} (s^{-1}\bar{P})^{\circ n}
\]
where $T^c$ denotes the cofree cooperad functor, $(-)^{\circ n}$ denotes the $n$-fold cooperadic composition,
The differential $d = d_{\mathrm{internal}} + d_{\mathrm{decomposition}}$ combines the internal differential of $P$ with edge contractions on trees decorated by operations from $P$.
\end{definition}

\section{Cotriple resolution and configuration spaces}

\begin{remark}[Configuration spaces and factorization]
By the Ayala--Francis theorem~\cite{AF15}, factorization homology $\int_X \mathcal{A}$ of a chiral algebra on a curve $X$ is computed by the colimit $\mathrm{colim}_{n} [ \mathcal{A}^{\otimes n} \otimes \Omega^*(\mathrm{Conf}_n(X)) ]$. The bar complex is the dual construction.

\subsection{The genus expansion}
\label{sec:genus_expansion_panorama}

\begin{theorem}[Operadic bar complex \cite{LV12}; \ClaimStatusProvedElsewhere]\label{thm:operadic-bar}
\index{operadic bar construction}
For an operad $\mathcal{P}$ and $\mathcal{P}$-algebra $A$, the bar complex is:
\[B_{\mathcal{P}}(A) = \bigoplus_{n \geq 1} (\mathcal{P}^{\text{\raisebox{0.5ex}{\tiny !`}}}(n) \otimes_{\Sigma_n} A^{\otimes n})\]
where $\mathcal{P}^{\text{\raisebox{0.5ex}{\tiny !`}}}$ is the Koszul dual cooperad (which already incorporates the appropriate suspension), and the differential combines the cooperad cocomposition with the algebra structure maps of $A$.
\end{theorem}

\begin{theorem}[Geometric realization; \ClaimStatusProvedHere]\label{thm:geometric-bridge}
For the chiral operad $\mathcal{P}_{\text{ch}}$ on a curve $X$:
\begin{enumerate}
\item $\mathcal{P}_{\text{ch}}(n) \cong \Omega^*_{\log}(\overline{C}_n(X))$ as a complex~\cite{KontsevichSoibelman}
\item The operadic composition corresponds to boundary stratification
\item The bar differential becomes residues at collision divisors
\end{enumerate}

This provides a canonical isomorphism:
\[B_{\mathcal{P}_{\text{ch}}}(\mathcal{A}) \cong \bar{B}^{\text{ch}}_{\text{geom}}(\mathcal{A})\]
\end{theorem}

\begin{proof}
We verify the isomorphism in two steps.

\emph{Underlying graded objects.}
The Koszul dual cooperad of the chiral operad satisfies
$\mathcal{P}_{\mathrm{ch}}^{!'}(n)
\cong H_*(\overline{C}_n(X), \mathbb{C})$
(the Borel--Moore homology of the FM compactification, dual to the
logarithmic cohomology
$\mathcal{P}_{\mathrm{ch}}(n) = \Omega^*_{\log}(\overline{C}_n(X))$).
Therefore the $n$-th component of the operadic bar complex is:
\[
\mathcal{P}_{\mathrm{ch}}^{!'}(n) \otimes_{\Sigma_n} \mathcal{A}^{\otimes n}
\cong \Gamma\bigl(\overline{C}_n(X),\,
j_* j^*(\mathcal{A}^{\boxtimes n}) \otimes \Omega^*_{\log}\bigr)^{\Sigma_n}
\]
which is the $n$-th term of the geometric bar complex
$\bar{B}^{\mathrm{ch}}_{\mathrm{geom}}(\mathcal{A})$.

\emph{Differentials.}
The operadic bar differential $d = d_{\mathrm{cooperad}} +
d_{\mathrm{alg}}$ translates as follows.
The cooperad cocomposition
$\Delta\colon \mathcal{P}^{!'}(n)
\to \bigoplus_{k} \mathcal{P}^{!'}(k) \otimes
\mathcal{P}^{!'}(n_1) \otimes \cdots \otimes \mathcal{P}^{!'}(n_k)$
corresponds to the boundary stratification of $\overline{C}_n(X)$:
each codimension-$1$ boundary divisor $D_S$ (labeled by a subset $S$ of
colliding points) contributes a residue term.  The algebra structure maps
$m_k\colon \mathcal{A}^{\otimes k} \to \mathcal{A}$ are the OPE
coefficients extracted by these residues.  Together,
$d_{\mathrm{cooperad}} + d_{\mathrm{alg}}$ becomes
$d_{\mathrm{dR}} + d_{\mathrm{res}}$
(the de~Rham plus residue differential on the geometric bar complex),
matching the differential of
Definition~\ref{def:geometric-bar}.
\end{proof}

\section{Quadratic operads and Koszul duality}
 
\begin{definition}[Quadratic operad]
A quadratic operad has the form $P = \Free(E)/(R)$ where $E$ is a collection of generating operations in arity $2$, $R \subset \Free(E)(3)$ consists of quadratic relations, and $(R)$ is the operadic ideal generated by $R$.
\end{definition}
 
\begin{definition}[Koszul dual cooperad]
\index{desuspension}
The Koszul dual cooperad $P^!$ is the maximal sub-cooperad of the cofree cooperad $T^c(s^{-1}E^\vee)$ cogenerated by the orthogonal relations $R^\perp \subset (s^{-1}E^\vee)^{\otimes 2}$, where the orthogonality is with respect to the natural pairing induced by evaluation.
\end{definition}
 
\begin{definition}[Koszul operad]\label{def:koszul-operad}
\index{Koszul operad|textbf}
An operad $P$ is \emph{Koszul} if the canonical map $\Omega(P^!) \to P$ is a quasi-isomorphism. Equivalently, the Koszul complex $K_\bullet(P) = P^! \circ P$ with differential induced by the cooperad and operad structures is acyclic in positive degrees.
\end{definition}
 
\section{Derivation of Com-Lie duality}
 
\begin{theorem}[Com-Lie Koszul duality; \ClaimStatusProvedElsewhere]\label{thm:com-lie}
\index{$L_\infty$-algebra}
We have canonical isomorphisms of operads:
\[
\Com^! \cong \Lie \quad \text{and} \quad \Lie^! \cong \Com
\]
At the cooperad level, $\barB(\Com) \simeq \mathrm{co}\Lie$ and $\barB(\Lie) \simeq \mathrm{co}\Com$.
Moreover, both $\Com$ and $\Lie$ are Koszul operads with quasi-isomorphisms:
\[
\Omega(\text{co}\Lie) \xrightarrow{\sim} \Com, \quad \Omega(\text{co}\Com) \xrightarrow{\sim} \Lie
\]
\end{theorem}
 
\begin{proof}[Proof via partition lattices]
By Theorem~\ref{thm:partition}, $\barB(\Com)(n) \simeq s^{n-1}\tilde{C}_{*}(\barPi_n) \otimes \sgn_n$.
 
Classical results of Björner--Wachs~\cite{BW83} and Stanley~\cite{Sta97} establish that the reduced homology of $\barPi_n$ is:
\begin{itemize}
\item The order complex $|\barPi_n|$ has dimension $n-3$ (maximal chains in $\barPi_n$ have length $n-3$)
\item $\tilde{H}_k(\barPi_n) = 0$ for $k \neq n-3$ (homology concentrated in the top degree)
\item The $S_n$-representation on $\tilde{H}_{n-3}(\barPi_n)$ is $\Lie(n) \otimes \sgn_n$ where $\Lie(n)$ is the Lie representation
\end{itemize}

The geometric realization $|\barPi_n|$ is homotopy equivalent to a wedge of $(n-1)!$ spheres of dimension $n-3$, with the $S_n$-action on the top homology given by the Lie representation tensored with the sign.

For Com-Lie duality, the bar construction gives:
\[\barB(\Com)(n) \simeq s^{n-1}\tilde{C}_{*}(\barPi_n) \otimes \sgn_n\]
Taking homology and using that $\barPi_n$ has homology concentrated in the single degree $n-3$:
\[H_*(\barB(\Com)(n)) \simeq s^{n-1}\tilde{H}_{n-3}(\barPi_n) \otimes \sgn_n = s^{n-1}\Lie(n) \otimes \sgn_n \otimes \sgn_n = s^{n-1}\Lie(n)\]
Since this is concentrated in a single degree, the bar complex is formal and we obtain $\barB(\Com) \simeq \text{co}\Lie$, where the arity-dependent shift~$s^{n-1}$ is absorbed into the cooperadic convention $\text{co}\Lie(n) := s^{n-1}\Lie(n)$ (LV12, \S7.2).
Applying $\Omega$ yields $\Omega(\text{co}\Lie) \simeq \Com$.
 
The dual statement $\Lie^! \cong \text{co}\Com$ follows by Schur--Weyl duality, using the characterization of $\Lie$ as the primitive part of the tensor coalgebra.
\end{proof}
 
\begin{proof}[Alternative proof via generating series]
The Poincaré series of the operads satisfy:
\begin{align}
P_{\Com}(x) &= e^x - 1 \\
P_{\Lie}(x) &= -\log(1 - x)
\end{align}
These are compositional inverses: $P_{\Lie}(-P_{\Com}(-x)) = x$. This functional equation characterizes Koszul dual pairs, providing an independent verification of the duality.
\end{proof}
 
\section{The quadratic dual and orthogonality}
 
For explicit computations, we need the quadratic presentations:
 
\begin{proposition}[Quadratic presentations~\cite{LV12}; \ClaimStatusProvedElsewhere]\label{prop:quadratic-presentations-com-lie}
The operads $\Com$ and $\Lie$ have quadratic presentations:
\begin{align}
\Com &= \Free(\mu)/(R_{\Com}) \text{ where } R_{\Com} = \langle \mu_{12,3} - \mu_{1,23}, \mu_{12} - \mu_{21} \rangle \\
\Lie &= \Free(\ell)/(R_{\Lie}) \text{ where } R_{\Lie} = \langle \ell_{12,3} + \ell_{23,1} + \ell_{31,2}, \ell_{12} + \ell_{21} \rangle
\end{align}
where subscripts denote inputs, and composition is denoted by adjacency. Here $\mu_{12,3}$ means $\mu \circ_1 \mu$ and $\mu_{1,23}$ means $\mu \circ_2 \mu$.
\end{proposition}
 
\begin{proposition}[Orthogonality; \ClaimStatusProvedHere]\label{prop:orthogonal}
With respect to the natural pairing between $\Free(\mu)(3)$ and
$\Free(\ell^*)(3)$ induced by $\langle \mu, \ell^* \rangle = 1$, one has
\[
R_{\Com} \perp R_{\Lie}
\]
This orthogonality is the concrete manifestation of Koszul duality.
\end{proposition}
 
\begin{proof}
We compute the pairing explicitly. The spaces have bases:
\begin{align}
\Free(\mu)(3) &= \text{span}\{\mu_{12,3}, \mu_{1,23}, \mu_{13,2}, \mu_{2,13}, \mu_{23,1}, \mu_{3,12}\} \\
\Free(\ell^*)(3) &= \text{span}\{\ell^*_{12,3}, \ell^*_{1,23}, \text{etc.}\}
\end{align}
 
The pairing $\langle \mu_{ij,k}, \ell^*_{pq,r} \rangle = 1$ if the tree structures match and $0$ otherwise. Computing:
\begin{align}
\langle \mu_{12,3} - \mu_{1,23}, \ell^*_{12,3} + \ell^*_{23,1} + \ell^*_{31,2} \rangle &= 1 + 0 + 0 - 0 - 0 - 1 = 0 \\
\langle \mu_{12,3} - \mu_{1,23}, \ell^*_{13,2} + \ell^*_{32,1} + \ell^*_{21,3} \rangle &= 0 - 1 + 0 + 0 + 1 + 0 = 0
\end{align}
Similar computations for all pairs verify the orthogonality.
\end{proof}

\section{Factorization algebra axioms: verification}
\label{sec:factorization-axioms-complete}

\subsection{Motivation}

\begin{construction}[Configuration space realization]
A factorization algebra $\mathcal{F}$ on a manifold $M$ assigns to each open $U \subset M$ a vector space $\mathcal{F}(U)$ satisfying
$\mathcal{F}(U \sqcup V) \cong \mathcal{F}(U) \otimes \mathcal{F}(V)$
and the colimit reconstruction
\[\mathcal{F}(U)= \operatorname{colim}_{(x_1,\ldots,x_n) \in C_n(U)}
   \mathcal{F}(\text{disk around } x_1) \otimes \cdots \otimes
   \mathcal{F}(\text{disk around } x_n)\]
over the configuration space $C_n(M) = \{(x_1, \ldots, x_n) \in M^n : x_i \neq x_j\}$.
Factorization algebras on $\mathbb{R}^n$ are
equivalent to $E_n$-algebras~\cite[Theorem~5.4.5.9]{HA}.
\end{construction}

\subsection{Weiss topology and factorization algebras}

The cosheaf condition defining factorization algebras requires a topology
finer than the ordinary open-cover topology.  The correct notion,
introduced by Weiss~\cite{Wei99} in the context of manifold calculus and
adopted by Costello--Gwilliam~\cite{CG17}, is the following.

\begin{definition}[Weiss cover]\label{def:weiss-cover}
\index{Weiss cover|textbf}
Let $M$ be a topological space.  A collection $\{U_i\}_{i \in I}$ of open
subsets of $M$ is a \emph{Weiss cover} of an open $V \subset M$ if every
\emph{finite} subset $S \subset V$ is contained in some single $U_i$.
\end{definition}

The \emph{Weiss topology} on $\mathrm{Opens}(M)$ is the Grothendieck
topology whose covering families are exactly the Weiss covers.
A \emph{prefactorization algebra} that satisfies descent for the Weiss
topology is a \emph{factorization algebra}.

\subsection{Ayala--Francis axioms}

\begin{definition}[Factorization algebra --- Ayala--Francis definition]
\label{def:factorization-algebra-AF}
Let $M$ be a smooth manifold and $\mathcal{V}$ a symmetric monoidal $\infty$-category
(in the sense of~\cite[§2.0, §4.1]{HA}).
A \emph{factorization algebra} $\mathcal{F}$ on $M$ with values in $\mathcal{V}$
consists of:

\emph{Data.}
\begin{enumerate}
\item For each open $U \subset M$: an object $\mathcal{F}(U) \in \mathcal{V}$.

\item For each inclusion $U \hookrightarrow V$ of opens: a morphism
   $\mathcal{F}(U) \to \mathcal{F}(V)$ in $\mathcal{V}$.

\item For each finite collection of pairwise disjoint opens $U_1, \ldots, U_n \subset V$:
   a \emph{factorization map}
   \[\mu_{U_1,\ldots,U_n}^V: \mathcal{F}(U_1) \otimes \cdots \otimes \mathcal{F}(U_n)
      \to \mathcal{F}(V).\]
\end{enumerate}

\emph{Axioms.}

(FA1) \emph{Functoriality.} The assignment $U \mapsto \mathcal{F}(U)$ is a functor
from $\text{Opens}(M)$ to $\mathcal{V}$:
\begin{itemize}
\item $\mathcal{F}(U) \xrightarrow{\text{id}} \mathcal{F}(U)$ is the identity.
\item For $U \hookrightarrow V \hookrightarrow W$: the composition
   $\mathcal{F}(U) \to \mathcal{F}(V) \to \mathcal{F}(W)$ equals
   $\mathcal{F}(U) \to \mathcal{F}(W)$.
\end{itemize}

(FA2) \emph{Multiplicativity.} For disjoint opens $U_1, \ldots, U_n \subset V$,
the factorization map is a structure map:
\[\mu_{U_1,\ldots,U_n}^V: \mathcal{F}(U_1) \otimes \cdots \otimes \mathcal{F}(U_n)
   \longrightarrow \mathcal{F}(V)\]
This map is a quasi-isomorphism when $\{U_1, \ldots, U_n\}$ forms a
Weiss cover of $V$ (Definition~\ref{def:weiss-cover}), but for arbitrary disjoint opens it is merely a morphism (not necessarily an equivalence).

(FA3) \emph{Associativity.} For nested collections
$U_{ij} \subset V_i \subset W$ (all disjoint), the diagram commutes:
\[\begin{tikzcd}[column sep=large]
\bigotimes_{i,j} \mathcal{F}(U_{ij})
   \arrow[r, "{\otimes_i \mu_{U_{i,*}}^{V_i}}"]
   \arrow[d, "{\mu_{\{U_{ij}\}}^W}"'] &
\bigotimes_i \mathcal{F}(V_i)
   \arrow[d, "{\mu_{\{V_i\}}^W}"] \\
\mathcal{F}(W) \arrow[r, equal] & \mathcal{F}(W)
\end{tikzcd}\]

(FA4) \emph{Unit.} For any open $U$:
\[\mathcal{F}(\emptyset) = \mathbb{1}_{\mathcal{V}}
   \quad \text{(unit object in } \mathcal{V}\text{)}\]
and $\mathcal{F}(\emptyset) \otimes \mathcal{F}(U) \xrightarrow{\sim} \mathcal{F}(U)$.

(FA5) \emph{Symmetry.} For any permutation $\sigma \in S_n$ and opens
$U_1, \ldots, U_n \subset V$, the diagram commutes:
\[\begin{tikzcd}
\mathcal{F}(U_1) \otimes \cdots \otimes \mathcal{F}(U_n)
   \arrow[r, "\sigma"] \arrow[d, "\mu"'] &
\mathcal{F}(U_{\sigma(1)}) \otimes \cdots \otimes \mathcal{F}(U_{\sigma(n)})
   \arrow[d, "\mu"] \\
\mathcal{F}(V) \arrow[r, equal] & \mathcal{F}(V)
\end{tikzcd}\]
\end{definition}

\subsection{Chiral algebras as factorization algebras}

\begin{theorem}[Chiral algebras are factorization algebras; \ClaimStatusProvedElsewhere]
\label{thm:chiral-factorization}
\textup{(Beilinson--Drinfeld \cite[Theorem~3.4.9, \S3.4.10--3.4.12]{BD04})}

Let $X$ be a smooth algebraic curve over $\mathbb{C}$.
Every chiral algebra $\mathcal{A}$ on $X$ in the sense of
\cite[\S3.3]{BD04} determines a factorization algebra of
$\mathcal{D}$-modules on $\operatorname{Ran}(X)$ satisfying
axioms (FA1)--(FA5).  More precisely, the functor
\[\mathcal{A} \;\longmapsto\; \mathcal{F}_{\mathcal{A}},
\qquad \mathcal{F}_{\mathcal{A}}(U) = \Gamma(U, \mathcal{A}),\]
defines an equivalence between the category of chiral algebras
on $X$ and the category of factorization algebras of
$\mathcal{D}$-modules on $\operatorname{Ran}(X)$.
\end{theorem}

\begin{remark}[Categorical distinction]
\label{rem:chiral-vs-factorization}
\index{factorization algebra!vs.\ chiral algebra}
Although Theorem~\ref{thm:chiral-factorization} establishes an
equivalence of categories, the two notions have different
signatures and live in different ambient categories:
\begin{itemize}
\item A \emph{chiral algebra}
  (Definition~\ref{def:chiral-algebra}) is a
  $\mathcal{D}_X$-module on the curve~$X$ equipped with a
  chiral product~$\mu$.
\item A \emph{factorization algebra}
  (Definition~\ref{def:factorization-algebra-AF}) is a
  cosheaf on $\operatorname{Ran}(X)$ with multiplicative
  structure.
\end{itemize}
The bar construction
(Chapter~\ref{chap:bar-cobar}) produces a factorization
\emph{coalgebra} on $\operatorname{Ran}(X)$
(Convention~\ref{conv:bar-coalgebra-identity}): it lives on
the factorization side of the equivalence.  When the chiral
perspective is needed (e.g.\ for the OPE, the pole
decomposition, the Borcherds identity), we work with the
chiral product~$\mu$; when the geometric perspective is needed
(e.g.\ for Verdier duality, factorization homology, the
Ran-space structure), we work with the factorization
coalgebra~$\barB_X(\cA)$.  The equivalence of
Theorem~\ref{thm:chiral-factorization} ensures that no
information is lost in the translation.
\end{remark}

\label{rem:bd-factorization-dictionary}

\subsection{Gluing formulas and excision}

\begin{theorem}[Excision property; \ClaimStatusProvedElsewhere]
\label{thm:excision-factorization}
\textup{(Costello--Gwilliam~\cite[§6.3]{CG17}; Lurie~\cite[Theorem~5.5.3.11]{HA})}

Let $\mathcal{F}$ be a factorization algebra on $M$, and let
$M = U \cup V$ be a \emph{Weiss cover} (i.e., every finite subset of $M$ is contained in $U$ or in $V$). Then there is a natural equivalence:
\[\mathcal{F}(M) \simeq \mathcal{F}(U) \otimes_{\mathcal{F}(U \cap V)} \mathcal{F}(V)\]
where the tensor product is the derived pushout over $\mathcal{F}(U \cap V)$.

For \emph{locally constant} factorization algebras ($E_n$-algebras), excision holds more generally for collar-gluing decompositions of manifolds, without the Weiss hypothesis~\cite[Theorem~5.5.3.11]{HA}.
\end{theorem}

\begin{proof}
This is a special case of the cosheaf property for Weiss covers
(Theorem~\ref{thm:factorization-cosheaf} below) applied to the
two-element Weiss cover $\{U, V\}$.  The pushout diagram
\[\begin{tikzcd}
U \cap V \arrow[r] \arrow[d] & V \arrow[d] \\
U \arrow[r] & M
\end{tikzcd}\]
in $\mathrm{Opens}(M)$ is sent by $\mathcal{F}$ to a pushout in the target
category by the descent condition for Weiss covers.
\end{proof}

\subsection{Cosheaf property}

\begin{theorem}[Factorization algebras are cosheaves for Weiss covers;
\ClaimStatusProvedElsewhere]
\label{thm:factorization-cosheaf}
\textup{(Costello--Gwilliam \cite[Theorem~6.3.1]{CG17})}

A prefactorization algebra $\mathcal{F}$ on $M$ is a factorization algebra
if and only if it satisfies the \emph{cosheaf property for Weiss covers}:
for every Weiss cover $\{U_i\}_{i \in I}$ of an open $V \subset M$
(Definition~\ref{def:weiss-cover}), the natural map
\[\operatorname{hocolim}_{\check{C}(\{U_i\})}
   \mathcal{F} \;\xrightarrow{\;\sim\;}\; \mathcal{F}(V)\]
is an equivalence, where the homotopy colimit is taken over the
\v{C}ech nerve of the Weiss cover.
\end{theorem}

\begin{remark}[Weiss covers versus ordinary covers]\label{rem:why-weiss}
Ordinary open covers detect only single-point behavior.  Weiss covers ensure every finite configuration is witnessed by a single open, so the cosheaf condition for Weiss covers is exactly equivalent to (FA1)--(FA5) with descent~\cite[Proposition~6.3.4]{CG17}.
\end{remark}
