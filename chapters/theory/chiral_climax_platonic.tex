\section{Setup: the Arnold connection, the KZ functor, BRST resolution}
\label{sec:climax-platonic-setup}

Four objects enter the climax: the ordered configuration space
$\Confnord(\Conf)$ with Arnold's $1$-forms; the universal
infinitesimal braid Lie algebra $\mathfrak{t}_n$; the chiral bar
coalgebra $\BarOrd(A)$; and the ghost charge of a quasi-free BRST
resolution. None is new. Each is recalled in the precise form the two
equations require, no more.

\subsection{The Arnold connection}
\label{ssec:climax-platonic-arnold}

Let $\Conf$ be a smooth complex curve and let
$\Confnord(\Conf) = \{(z_1,\dots,z_n) \in \Conf^{n}: z_i \neq z_j\}$
denote the open configuration space of $n$ ordered distinct points.
The Arnold $1$-forms are
\begin{equation}
\label{eq:arnold-eta}
   \eta_{ij} \;=\; \dlog(z_i - z_j)
   \;\in\; \Omega^{1}\bigl(\Confnord(\Conf)\bigr),
   \qquad 1 \leq i \neq j \leq n.
\end{equation}
Arnold's three-term identity reads
\begin{equation}
\label{eq:arnold-three-term}
   \eta_{ij} \wedge \eta_{jk}
   \;+\; \eta_{jk} \wedge \eta_{ki}
   \;+\; \eta_{ki} \wedge \eta_{ij}
   \;=\; 0
   \quad\text{in }\Omega^{2}\bigl(\Conf_3^{\mathrm{ord}}(\Conf)\bigr),
\end{equation}
for every triple of distinct indices.

\begin{theorem}[Arnold 1969]
\label{thm:arnold-cohomology-platonic}
\ClaimStatusProvedElsewhere
The de Rham cohomology ring $H^{*}(\Confnord(\Conf);\mathbb{C})$ is
the free graded-commutative algebra on the classes $[\eta_{ij}]$
modulo the three-term identity~\eqref{eq:arnold-three-term}.
\end{theorem}

\begin{remark}[Attribution]
\label{rem:arnold-attribution-platonic}
Theorem~\ref{thm:arnold-cohomology-platonic} is the Arnold--Orlik--Solomon
description of the cohomology of complements of hyperplane arrangements
applied to the braid arrangement; see Arnold (\textit{Mat.\ Zametki}~$5$, $1969$)
and Orlik--Solomon (\textit{Inventiones}~$56$, $1980$).
\end{remark}

The infinitesimal braid Lie algebra $\mathfrak{t}_n$ is the free
$\mathbb{Z}$-graded Lie algebra on generators
$t_{ij} = t_{ji}$ for $1 \leq i \neq j \leq n$ modulo the
infinitesimal braid relations
\begin{equation}
\label{eq:inf-braid-relations}
   [t_{ij}, t_{kl}] = 0 \quad \text{when } \{i,j\} \cap \{k,l\} = \emptyset,
   \qquad
   [t_{ij},\, t_{ik} + t_{jk}] = 0 \quad \text{for distinct } i,j,k.
\end{equation}

\begin{definition}[Arnold's universal KZ connection]
\label{def:arnold-universal-kz-connection-platonic}
\emph{Arnold's universal KZ connection} on the trivial
$U(\mathfrak{t}_n)$\nobreakdash-bundle over $\Confnord(\Conf)$ is
\begin{equation}
\label{eq:nabla-arnold}
   \nabla_{\Arn}
   \;=\; d \;-\; \sum_{1 \leq i < j \leq n} t_{ij}\,\eta_{ij},
\end{equation}
with $\eta_{ij}$ the Arnold $1$-form~\eqref{eq:arnold-eta} and
$t_{ij} \in \mathfrak{t}_n$ the universal infinitesimal braid generator.
\end{definition}

\begin{proposition}[Arnold flatness]
\label{prop:arnold-flatness-platonic}
\ClaimStatusProvedElsewhere
The connection $\nabla_{\Arn}$ is flat: $\nabla_{\Arn}^{2} = 0$. Flatness
is equivalent to the simultaneous vanishing of the three-term
identity~\eqref{eq:arnold-three-term} and the infinitesimal braid
relations~\eqref{eq:inf-braid-relations}.
\end{proposition}

\begin{proof}[Attribution]
The flatness $\nabla_{\Arn}^{2} = 0$ unpacks to
$\sum [t_{ij}, t_{kl}]\,\eta_{ij}\wedge\eta_{kl} = 0$, and the
hypotheses on $t_{ij}$ together with~\eqref{eq:arnold-three-term}
make every wedge sum vanish. Reference: Kohno (\textit{Annales
Inst.\ Fourier}~$37$, $1987$), Drinfel'd (\textit{Algebra i
Analiz}~$2$, $1989$).
\end{proof}

\subsection{The chiral bar complex $\BarOrd(A)$}
\label{ssec:climax-platonic-bar}

Let $A$ be an $E_\infty$\nobreakdash-chiral algebra on $\Conf$ in
the sense of Beilinson--Drinfeld: an augmented unital factorisation
$D$\nobreakdash-module on $\Conf$ with chiral multiplication
$\mu \colon j_!(A \boxtimes A)|_{\Conf^{2} \setminus \Delta} \to
\Delta_{!} A$. Write $\bar A = \ker\epsilon$ for the augmentation ideal.

\begin{definition}[Ordered chiral bar complex]
\label{def:bar-ord-platonic}
The \emph{ordered chiral bar coalgebra} of $A$ is the cofree conilpotent
graded coassociative coalgebra
\[
   \BarOrd(A)
   \;=\; \Tcoalg\bigl(s^{-1}\bar A\bigr)
   \;=\; \bigoplus_{n \geq 0}\bigl(s^{-1}\bar A\bigr)^{\otimes n},
\]
with deconcatenation coproduct, equipped with the bar differential
$\dbar = d_0 + d_1$ assembled from the augmentation $d_0$ and the
chiral multiplication $d_1$ via the standard Stasheff--Quillen
twisting morphism (Vol~I Chapter~\ref{chap:bar-construction}).
\end{definition}

The bar differential satisfies $\dbar^{2} = 0$, and $(\BarOrd(A),\dbar)$
is the categorical logarithm of $A$ in the chiral setting; see
Theorem~\ref{thm:theorem-a-platonic-pointer} below for the bar--cobar
adjoint equivalence on the Koszul locus.

\subsection{The KZ functor $\KZ$}
\label{ssec:climax-platonic-kz-functor}

Every chirally Koszul $E_\infty$-chiral algebra $A$ on $\Conf$
produces a matrix-valued flat connection on the configuration spaces
$\Confnord(\Conf)$ by absorbing its OPE residues into the Arnold
$1$-forms. The collection of all such connections forms a universal
arena $\ConnConf$ inside which Arnold's universal connection is the
initial object. The KZ functor $\KZ$ sends $A$ to its connection
$(V(A),\nabla(A))$ in this arena and is the geometric incarnation of
the question the pullback identity answers: does the bar differential
$\dbar$ equal $\KZ^{*}(\nabla_{\Arn})$?

\begin{definition}[The arena $\ConnConf$]
\label{def:connconf-platonic}
The category $\ConnConf = \ConnConf_{\Conf}$ has objects pairs
$(V, \nabla)$ where:
\begin{enumerate}
\item $V = \bigoplus_{n \geq 0} V_n$ is a graded vector space with
$V_n$ of finite rank in each total degree, $V_0 = \mathbb{C}$, equipped
with a graded-commutative unital product $\mu_V \colon V_m \otimes V_n
\to V_{m+n}$.
\item For each $n \geq 2$, $\nabla = \nabla_n$ is a meromorphic flat
connection on the trivial $V_n$\nobreakdash-bundle over $\Confnord(\Conf)$
with at-worst logarithmic poles along the collision divisors
$\Delta_{ij} = \{z_i = z_j\}$, of universal KZ form
\begin{equation}
\label{eq:nabla-n-form}
  \nabla_n \;=\; d \;-\; \sum_{1 \leq i < j \leq n} r^{(ij)}(z_i - z_j)\, d(z_i - z_j),
  \qquad
  r^{(ij)}(w) \in \End(V_n)[[w, w^{-1}]].
\end{equation}
\item Coherence: for each codimension-one boundary stratum
$\{z_{n+1} = w_0\} \subset \Confnord_{n+1}(\Conf)$, the restriction of
$\nabla_{n+1}$ to that stratum is the pullback of $\nabla_n$ on
$\Confnord_n(\Conf \setminus \{w_0\})$ along the forgetful map.
\end{enumerate}
Morphisms are pairs $(f,\phi)$ with $f \colon V \to V'$ a graded-ring
morphism and $\phi$ identifying $f^{*}\nabla'$ with $\nabla$ modulo
logarithmic gauge.
\end{definition}

\begin{remark}[Initial object]
\label{rem:connconf-initial-platonic}
The Arnold pair
$(U(\mathfrak{t}_\bullet), \nabla_{\Arn})$ from
Definition~\ref{def:arnold-universal-kz-connection-platonic} is the
initial object of $\ConnConf$, in the sense that for any
$(V, \nabla) \in \ConnConf$ the assignment
$t_{ij} \mapsto r^{(ij)}(z_{ij}) \cdot z_{ij}$ extends uniquely to a
ring morphism $U(\mathfrak{t}_n) \to V_n$ realising $\nabla$ as the
pullback of $\nabla_{\Arn}$.
\end{remark}

\begin{construction}[The KZ functor $\KZ$]
\label{constr:kz-functor-platonic}
For $A$ a chirally Koszul $E_\infty$\nobreakdash-chiral algebra on
$\Conf$, the KZ functor produces $\KZ(A) = (\BarOrd(A), \nabla(A))
\in \ConnConf$ in five steps.
\begin{description}
\item[(K1) Vector space.] Take $V(A) := \BarOrd(A)$ as a graded
vector space; the differential $\dbar$ will be repackaged as a
component of $\nabla(A)$.
\item[(K2) OPE residue.] For $a, b \in \bar A$ at distinct insertion
points, expand the chiral multiplication
$a(z) \cdot b(w) \sim \sum_{p \geq 1} \Res^{(p)}(a,b)(z-w)^{-p} +
\text{regular}$.
\item[(K3) $r$-matrix from OPE.] Form the $E_1$ $r$-matrix at the
$(i,j)$ pair of bar inputs by absorbing one $\dlog$ factor against
the leading OPE pole:
\begin{equation}
\label{eq:r-matrix-from-ope}
  r^{(ij)}(w) \;:=\; \sum_{p \geq 1} \Res^{(p)}(a_i, a_j)\,(w)^{-p+1}.
\end{equation}
\item[(K4) Connection.] Set
\begin{equation}
\label{eq:nabla-of-A}
  \nabla(A)_n
  \;:=\; d \;-\; \sum_{1 \leq i < j \leq n} r^{(ij)}(z_i - z_j)\,
                                            d(z_i - z_j).
\end{equation}
\item[(K5) Functoriality.] A chiral algebra morphism $f\colon A \to A'$
sends OPE residues to OPE residues, hence sends $r$-matrices to
$r$-matrices, inducing a morphism $\KZ(f) \in \ConnConf$.
\end{description}
\end{construction}

The $1$-form $d(z_i - z_j)$ in~\eqref{eq:nabla-of-A} is the
insertion-point differential, not the Arnold $\dlog$: the $\dlog$
factor was absorbed into step~(K3) via the trade $\dlog(z_i - z_j) =
(z_i - z_j)^{-1}\,d(z_i - z_j)$, which exchanges one unit of OPE
pole for one unit of $(z_i - z_j)^{-1}$.

\begin{example}[Affine Kac--Moody, both conventions]
\label{ex:kz-of-vk-g-platonic}
For $A = V^{k}(\fg)$ the universal affine vertex algebra, the OPE
residues are $\Res^{(2)}(J^a, J^b) = k\,\delta^{ab}$ and
$\Res^{(1)}(J^a, J^b) = f^{ab}{}_c\, J^c$. Step~(K3)
of Construction~\ref{constr:kz-functor-platonic} produces the
trace-form $r$-matrix
\begin{equation}
\label{eq:r-trace-form-vk-g}
  r^{(ij)}_{\mathrm{tr}}(z) \;=\; k\,\Omega_{ij}/z,
  \qquad \Omega = \sum_{a} I^{a} \otimes I_{a},
\end{equation}
which evaluates to $0$ at $k=0$ (abelian limit; with the level-prefix $k=0$ check passing) and to
$-\hv\,\Omega_{ij}/z$ at $k = -\hv$ (finite, nonzero; the Sugawara
critical level manifests in the KZ normalisation below, not in the
trace-form representative). The KZ normalisation
\begin{equation}
\label{eq:r-kz-form-vk-g}
  r^{(ij)}_{\KZ}(z)
  \;=\; \Omega_{ij}\big/\bigl((k+\hv)\,z\bigr)
\end{equation}
is the trace-form rescaled by $1/(k(k+\hv))$: the two conventions
represent the same $r$\nobreakdash-matrix up to normalisation of the
bilinear form $\Omega$. The KZ representative satisfies
$r|_{k=0} = \Omega/(\hv\,z) \neq 0$ for non-abelian $\fg$ (the Lie
bracket persists), and $r|_{k=-\hv}$ diverges. The trace-form
representative is finite at both endpoints; the KZ representative
diverges only at $k = -\hv$.
\end{example}

\subsection{Quasi-free BRST resolution and ghost charge}
\label{ssec:climax-platonic-brst}

The modular characteristic $\kappa$ varies per family
(Vol~I~$\S\ref{ch:landscape-census}$): $\kappa(V_k(\fg)) = \dim(\fg)(k
+ \hv)/(2\hv)$, $\kappa(\Vir_c) = c/2$, $\kappa(W_N) = c\,(H_N - 1)$
where $H_N = \sum_{j=1}^{N} 1/j$. Across the standard landscape these
values are unrelated in form, modular transformation, and physical
origin. The question is whether one integer-valued functor evaluated
at the canonical free-field resolutions recovers every one of them.
On the verified standard-landscape surface, the answer is yes: the
relevant scalar is the ghost charge of the corresponding quasi-free
BRST model.

Let $\BRSTGaugedChirAlg$ denote the category of pairs $(A, R_{\bullet}
\twoheadrightarrow A)$ consisting of an augmented unital $E_\infty$-chiral
algebra $A$ together with a quasi-free BRST resolution
$R_{\bullet} \to A$ in the homotopical sense: $R_{\bullet}$ is a free
chiral cdga generated in degrees $\leq 0$ by an augmentation-ideal
generator system together with a $bc$-ghost system, the
differential is the BRST charge $Q_{\BRST}$, and the morphism
$R_{\bullet} \to A$ is a quasi-isomorphism.

For each $bc(\lambda)$\nobreakdash-ghost pair (anticommuting at fermionic
$\lambda$), the central charge is $c_{\bc}(\lambda) = 1 - 3(2\lambda - 1)^{2}$
(/ B5). For each $\beta\gamma(\lambda)$\nobreakdash-system (commuting at
bosonic $\lambda$), the central charge is $c_{\beta\gamma}(\lambda)
= 2(6\lambda^{2} - 6\lambda + 1)$ (/ B6); the sign reverses
relative to $bc$. The total ghost charge of $R_{\bullet}$ is
\begin{equation}
\label{eq:c-ghost-formula}
  \cghost(R_{\bullet})
  \;=\; \sum_{\text{$bc$ pairs}} c_{\bc}(\lambda_{\alpha})
  \;+\; \sum_{\text{$\beta\gamma$ systems}} c_{\beta\gamma}(\lambda_{\beta}).
\end{equation}

\begin{remark}[Two free-field gradings]
\label{rem:bc-vs-bg-platonic}
The free-field grading determines the sign of the ghost contribution:
$bc$ at fermionic spin $\lambda = 2$ gives $c_{\bc}(2) = -26$ (the
bosonic-string $bc$); $\beta\gamma$ at bosonic spin $\lambda = 2$
gives $c_{\beta\gamma}(2) = +26$. Both signs appear in the standard
landscape (the former at Virasoro BRST; the latter as the matter
sector of the bosonic string). forbids cross-substitution.
\end{remark}

\section{The two equations}
\label{sec:climax-platonic-two-equations}

With the four objects in hand the two equations read off.

\subsection{Equation~(G2): bar differential is pullback of Arnold}
\label{ssec:climax-platonic-equation-g2}

The bar differential $\dbar$ on $\BarOrd(A)$ carries all the $E_\infty$
higher homotopies of $A$ into a single differential on the cofree
coalgebra. Under the KZ functor of
Construction~\ref{constr:kz-functor-platonic}, $\dbar$ is the pullback
of one universal flat connection, Arnold's $\nabla_{\Arn}$: no new
data is generated by the chiral bar beyond what the OPE already
encodes.

\begin{definition}[The pullback identity, structural form]
\label{def:pullback-identity-structural-platonic}
The \emph{pullback identity} for $A \in \ChirAlg^{E_\infty}$ at
genus zero is the equation in the category $\ConnConf_{\mathbb{P}^1}$,
\begin{equation}
\label{eq:climax-equation-G2}
  \dbar \;=\; \KZ^{*}(\nabla_{\Arn}),
\end{equation}
asserting that the bar differential of $\BarOrd(A)$ on
$\Confnord(\mathbb{P}^1)$ equals the pullback of Arnold's universal
KZ connection along the structure morphism
$\KZ(A) \to (U(\mathfrak{t}_\bullet), \nabla_{\Arn})$ induced by
$A \to \End(\BarOrd(A))$.
\end{definition}

\subsection{Equation~(G3): conductor is minus ghost charge}
\label{ssec:climax-platonic-equation-g3}

The per-family modular characteristic $\kappa(A)$ is recovered
uniformly across the standard landscape as the negative of the total
ghost central charge of any quasi-free BRST resolution of $A$. The
apparent family dependence of $\kappa$ is the surface manifestation
of a single integer-valued functor.

\begin{definition}[The ghost identity, structural form]
\label{def:ghost-identity-structural-platonic}
The \emph{ghost identity} for $(A, R_{\bullet}) \in \BRSTGaugedChirAlg$
is the equation in $\mathbb{Z}$,
\begin{equation}
\label{eq:climax-equation-G3}
  \kappa(A) \;=\; -\,\cghost\!\bigl(R_{\bullet}\bigr),
\end{equation}
asserting that the modular characteristic $\kappa(A)$ of $A$ equals
the negative of the total ghost central charge~\eqref{eq:c-ghost-formula}
of any quasi-free BRST resolution $R_{\bullet} \twoheadrightarrow A$.
\end{definition}

The right-hand side of~\eqref{eq:climax-equation-G3} depends a priori
on the choice of resolution. Theorem~\ref{thm:climax-vol1-platonic}
below verifies the identity on the canonical BRST models constructed for
the standard-landscape families and isolates the missing universal
functoriality as Conjecture~\ref{conj:climax-universal-brst-platonic}.

\section{The Climax Theorem}
\label{sec:climax-platonic-climax-theorem}

\begin{theorem}[Vol~I Climax surface: genus-zero pullback, verified
ghost families, and five-theorem readout]
\label{thm:climax-vol1-platonic}
\ClaimStatusProvedHere

a smooth projective curve $\Conf = \mathbb{P}^1$ over $\mathbb{C}$.
There exists a universal functor
\begin{equation}
\label{eq:KZ-functor-signature-platonic}
   \KZ \;\colon\; \ChirAlg^{E_\infty,\mathrm{Koszul}}_{\mathbb{P}^1}
   \;\longrightarrow\; \ConnConf_{\mathbb{P}^1},
\end{equation}
characterised by the following four properties.
\begin{enumerate}
\item \emph{(Pullback identity.)} The pullback identity
\eqref{eq:climax-equation-G2} holds:
$\dbar = \KZ^{*}(\nabla_{\Arn})$.
\item \emph{(Verified ghost identity on the standard landscape.)}
 For every family equipped with a canonical quasi-free BRST model,
\[
 A \in \{\mathcal{H}_k,\ \widehat{\fg}_k,\ \beta\gamma_\lambda,\
 bc_\lambda,\ \mathrm{Vir}_c,\ \mathcal{W}_3^k,\ \mathrm{BP}_k\},
\]

\[
 \kappa(A) \;=\; -\,\cghost(R_{\bullet}^{\mathrm{can}}).
\]
 The verification is family-by-family and is accompanied by three or

 a universal functorial BRST resolution is claimed here; that missing
 functoriality is exactly
 Conjecture~\ref{conj:climax-universal-brst-platonic}.
\item \emph{(Five-theorem readout, strongest honest form.)}
 Projecting \eqref{eq:climax-equation-G2} and the verified scalar
 ghost surface onto the five main theorem packages yields exactly the
 statements recorded in
 Theorems~\ref{thm:theorem-a-platonic-pointer}--%
 \ref{thm:theorem-h-platonic-pointer}: Theorem~A in chain-level and
 $(\infty,1)$-categorical bar-cobar adjunction form; Theorem~B as
 strict inversion on the Koszul locus together with the

 Theorem~C as C0/C1 complementarity plus the verified scalar constants
 $0,13,250/3,98/3$ on the classical $\mathsf{G}/\mathsf{L}/\mathsf{C}/\mathsf{M}$
 landmark table, enlarged to the canonical five-element bucket
 $\{0,8,13,250/3,98/3\}$ on the $\mathsf{G}/\mathsf{L}/\mathsf{C}/\mathsf{M}/\mathsf{B}$
 five-archetype ceiling via the $\mathsf{B}$-row Mukai-doubling witness

 (Convention~\ref{conv:theorem-c-bucket}); Theorem~D as the
 all-weight formula
 $F_g=\kappa\lambda_g^{\mathrm{FP}}+\delta F_g^{\mathrm{cross}}$
 with the exact scalar identity
 $\mathrm{obs}_g=\kappa\lambda_g$ only on the uniform-weight lane;
 and Theorem~H as the chiral Hochschild theorem, not a topological or
 categorical Hochschild statement, on its proved Koszul/generic
 surface. No broader scope than those native theorem surfaces is
 claimed in this chapter.
\item \emph{(Arnold reduction.)} The flatness identity
$\nabla_{\Arn}^{2} = 0$ on $\Confnord(\mathbb{P}^1)$ is equivalent,
after pullback under $\KZ$, to $\dbar^{2} = 0$ on $\BarOrd(A)$. The
classical Yang--Baxter equation on the family
$\{r^{(ij)}(z_{ij}) = \KZ^{*}(\eta_{ij})\}$ is the chain-level
incarnation of Arnold's three-term relation
\eqref{eq:arnold-three-term} at the codimension-two strata of
$\Confnord(\mathbb{P}^1)$.
\end{enumerate}
\end{theorem}

\begin{proof}[Proof architecture]
\label{prf:climax-vol1-platonic}
Property~(1) is proved in
Section~\ref{ssec:climax-platonic-path-A-fm} by identifying both

(Lemmas~\ref{lem:fm-residue-dbar-platonic},~\ref{lem:pullback-presentation-nabla-platonic}).

\section{Proof architecture: four genuinely disjoint paths}
\label{sec:climax-platonic-proof-architecture}

A single proof of Theorem~\ref{thm:climax-vol1-platonic} would already
close the identification at genus zero. Four disjoint proofs, routed
through four disjoint techniques, establish more: the identification
is forced by every one of the four directions from which the bar
differential can be read. The agreement of these four routes is the
substance of the climax, not a redundancy.

\subsection{Path~A: Fulton--MacPherson residues}
\label{ssec:climax-platonic-path-A-fm}

The first path is purely geometric. The chiral bar differential is
extracted from the Fulton--MacPherson compactification
$\overline{\mathrm{FM}}_n^{\mathrm{ord}}(\Conf)$ via residue maps at
the codimension-one boundary divisors corresponding to two-point
collisions.

\begin{lemma}[FM residue presentation of $\dbar$]
\label{lem:fm-residue-dbar-platonic}
\ClaimStatusProvedElsewhere
Under the Beilinson--Drinfeld factorisation construction, the chiral
bar differential $\dbar$ on $\BarOrd(A)$ at degree $n$ equals the sum
over collision pairs $(i,j)$ of the residue map
$\Res_{\Delta_{ij}} \colon \Omega^{n}\bigl(\Confnord(\Conf), A^{\boxtimes n}\bigr)
\to \Omega^{n-1}\bigl(\Confnord_{n-1}(\Conf), A^{\boxtimes(n-1)}\bigr)$
applied to the OPE pole expansion at $\Delta_{ij}$.
\end{lemma}

\begin{proof}[Attribution]
This is the standard Beilinson--Drinfeld presentation of the chiral
homology differential as boundary-residue extraction. See Vol~I
Chapter~\ref{chap:bar-construction}, and the original references
\textit{Chiral Algebras} (Beilinson--Drinfeld $2004$, $\S 4.2$).
\end{proof}

\begin{lemma}[Pullback presentation of $\nabla(A)$]
\label{lem:pullback-presentation-nabla-platonic}
\ClaimStatusProvedHere
Construction~\ref{constr:kz-functor-platonic} produces a connection
$\nabla(A)_n$ on $\Confnord(\Conf)$ whose connection $1$-form, modulo
logarithmic gauge, equals the OPE residue presentation of
Lemma~\ref{lem:fm-residue-dbar-platonic} at every codimension-two
collision stratum.
\end{lemma}

\begin{proof}
Step~(K3) of Construction~\ref{constr:kz-functor-platonic} extracts
$r^{(ij)}(w)$ from the OPE pole expansion at $\Delta_{ij}$ via
the trade $\dlog(z_i - z_j) \mapsto (z_i - z_j)^{-1}\,d(z_i - z_j)$.
At every codimension-two stratum $\Delta_{ij} \cap \Delta_{kl}$ with
$\{i,j\} \cap \{k,l\} = \emptyset$, the leading-pole coefficients of
$r^{(ij)}$ and $r^{(kl)}$ commute (Arnold property at disjoint pairs)
and their wedge contribution to $\nabla(A)^{2}$ vanishes by~\eqref{eq:inf-braid-relations}.
At every codimension-two stratum $\Delta_{ij} \cap \Delta_{ik}$ with
shared index $i$, the leading-pole contribution to $\nabla(A)^{2}$
is the chain-level incarnation of Arnold's three-term identity
\eqref{eq:arnold-three-term}; the OPE residue at the triple collision
is the chiral CYBE
\begin{equation}
\label{eq:cybe-on-r}
  [r^{(ij)},\, r^{(ik)} + r^{(jk)}]
  \;+\; [r^{(ik)},\, r^{(jk)}] \;=\; 0,
\end{equation}
verified in Lemma~\ref{lem:fm-residue-dbar-platonic} by directly
matching the wedge-product expansion to the bar-differential expansion
at the codimension-two FM stratum.
\end{proof}

\begin{remark}[Closure of Path~A]
\label{rem:path-A-closure-platonic}
Lemmas~\ref{lem:fm-residue-dbar-platonic} and~\ref{lem:pullback-presentation-nabla-platonic}
together establish the pullback identity~\eqref{eq:climax-equation-G2}
on the Koszul locus at genus zero: both $\dbar$ and $\KZ^{*}(\nabla_{\Arn})$
are presented as the same FM-residue map, hence they coincide.
\end{remark}

\subsection{Path~B: Kohno monodromy}
\label{ssec:climax-platonic-path-B-kohno}

The second path is representation-theoretic and computes the
monodromy of $\nabla(A)$ in evaluation modules.

\begin{theorem}[Kohno 1987]
\label{thm:kohno-monodromy-platonic}
\ClaimStatusProvedElsewhere
For $A = V^{k}(\fg)$ with $\fg$ simple at generic level $k$, the
monodromy of the connection $\nabla(A)$ on evaluation modules
$V_1 \otimes \cdots \otimes V_n$ is the quantum-group braid
representation of $U_{q}(\fg)$ at $q = \exp(2\pi i/(k+\hv))$.
\end{theorem}

\begin{proof}[Attribution]
Kohno (\textit{Annales Inst.\ Fourier}~$37$, $1987$) for the genus-zero
KZ monodromy; Drinfel'd (\textit{Algebra i Analiz}~$2$, $1989$) for
the associator construction; Kazhdan--Lusztig
(\textit{J.\ Amer.\ Math.\ Soc.}~$6$--$7$, $1993$--$1994$) for the
type-A category equivalence. See also Vol~I Section~\ref{sec:derived-dk}.
\end{proof}

\begin{remark}[Closure of Path~B for affine input]
\label{rem:path-B-closure-platonic}
Theorem~\ref{thm:kohno-monodromy-platonic} verifies the pullback
identity for the affine Kac--Moody specialisation: the monodromy of
the bar-differential side equals the monodromy of the Arnold side
both compute the quantum-group braid representation. The agreement
provides a non-circular cross-check on Path~A.
\end{remark}

\subsection{Path~C: Drinfeld associator}
\label{ssec:climax-platonic-path-C-drinfeld}

The third path is analytic and computes Arnold's monodromy via
iterated integrals. The Drinfeld associator $\Phi_{\KZ} \in U(\mathfrak{t}_3)
[[\hbar]]$ is the universal monodromy element for $\nabla_{\Arn}$ at
$n = 3$, controlling the failure of strict associativity at triple
collisions.

\begin{theorem}[Drinfeld $1989$]
\label{thm:drinfeld-associator-platonic}
\ClaimStatusProvedElsewhere
The monodromy of $\nabla_{\Arn}$ on $\Conf_3^{\mathrm{ord}}(\mathbb{C})$
through the iterated-integral construction equals the Drinfeld
associator $\Phi_{\KZ}$, satisfying the pentagon and hexagon coherence
identities and the duality relation
$\Phi_{\KZ}(t_{12}, t_{23})\,\Phi_{\KZ}(t_{23}, t_{12}) = 1$.
\end{theorem}

\begin{proof}[Attribution]
Drinfel'd (\textit{Algebra i Analiz}~$2$, $1989$, $\S 4$). See also
Bar-Natan (\textit{Selecta Math.}~$4$, $1998$) for an explicit
combinatorial presentation.
\end{proof}

\begin{remark}[Closure of Path~C]
\label{rem:path-C-closure-platonic}
The $\Phi_{\KZ}$ associator enters the bar-cobar machinery at degree
$3$ as the codimension-two strata correction to flatness. In
Theorem~\ref{thm:climax-vol1-platonic} this corresponds to property~(3)
universality: every $\KZ(A)$ is the pullback of $\nabla_{\Arn}$, hence
its monodromy factors through $\Phi_{\KZ}$.
\end{remark}

\subsection{Path~D: direct chain-level Yang--Baxter}
\label{ssec:climax-platonic-path-D-direct}

The fourth path is chain-theoretic and verifies the equivalence
between $\dbar^{2} = 0$ and the classical Yang--Baxter equation on
the family $\{r^{(ij)}\}$ extracted in step~(K3) of
Construction~\ref{constr:kz-functor-platonic}.

\begin{proposition}[CYBE equivalence]
\label{prop:cybe-equivalence-platonic}
\ClaimStatusProvedHere
The equation $\dbar^{2} = 0$ on $\BarOrd(A)$ is equivalent to the
classical Yang--Baxter equation~\eqref{eq:cybe-on-r} on the family
$\{r^{(ij)}(z_{ij})\}_{1 \leq i < j \leq n}$, for every $n \geq 3$
and every chirally Koszul $A$.
\end{proposition}

\begin{proof}
Expand $\dbar^{2}$ on a degree-$n$ bar word as a sum over
collision-pair compositions. By
Lemma~\ref{lem:fm-residue-dbar-platonic}, each term is an OPE
residue at a codimension-two stratum
$\Delta_{ij} \cap \Delta_{kl}$. At disjoint pairs the
contribution vanishes by~\eqref{eq:inf-braid-relations}; at
shared-index triples the contribution is exactly~\eqref{eq:cybe-on-r}.
Hence $\dbar^{2} = 0$ at degree $n$ if and only if the CYBE holds for
every triple of indices in $\{1,\dots,n\}$.
\end{proof}

\label{eq:climax-drinfeld-kohno-platonic}

\subsection{Verlinde: rational chiral algebra at genus zero}
\label{ssec:climax-platonic-cor-Verlinde}

\begin{corollary}[Verlinde formula from the climax]
\label{cor:climax-verlinde-platonic}
\ClaimStatusProvedHere
For $A$ a rational $E_\infty$\nobreakdash-chiral algebra with finite
fusion ring and modular $S$\nobreakdash-matrix $\{S_{ij}\}$, the
fusion coefficients $N_{ij}^{k}$ are diagonalised by $S$:
\begin{equation}
\label{eq:climax-verlinde-platonic}
   N_{ij}^{k}
   \;=\; \sum_{\ell}\frac{S_{i\ell}\,S_{j\ell}\,\overline{S_{k\ell}}}{S_{0\ell}}.
\end{equation}
\end{corollary}

\begin{proof}
Genus-zero conformal blocks
$\Hom_{\mathrm{ch}}(A, V_{i_1}\boxtimes\cdots\boxtimes V_{i_n})$ form
a flat bundle over $M_{0,n}$ under $\nabla(A)$, by
Theorem~\ref{thm:climax-vol1-platonic} property~(1). The codimension-one
boundary strata of $M_{0,n}$ correspond to fusion degenerations; the
factorisation of flat sections at these strata is governed by the
fusion rules $N_{ij}^{k}$. The monodromy of $\nabla(A)$ around the
modular $S$\nobreakdash-transformation diagonalises $N_{ij}^{k}$;
this is the Moore--Seiberg ($1989$) presentation of the Verlinde
formula. See Vol~I Chapter~\ref{ch:landscape-census}
for the standard-landscape verification.
\end{proof}

belongs to Theorem~\ref{thm:theorem-b-platonic-pointer}.

\end{proof}

\begin{lemma}[Humbert-wall failure mode at $H_1$]

$\phi^{(3)}\in C^3(\widehat{\mathfrak{t}_{2,[2]}^{\mathrm{Sieg}}
\oplus \mathfrak{n}_+^{\mathrm{imag}}})$ is the Chevalley--Eilenberg

component, valued in

\otimes H^0(\mathcal{M}_{1,1},\omega^{\otimes 12})^{-2}$, acquires
a simple zero on $H_3$ tracked by the Gritsenko--Nikulin

\end{proof}

\begin{lemma}[Humbert-wall failure mode at $H_4$]
\label{lem:cclimax-H4-failure}
\index{Humbert divisor!$H_4$ failure mode}
\index{Kuga--Satake!$\mathbb{Z}/2$-band}
On a formal neighbourhood of the Humbert divisor
$H_4\subset \overline{\mathcal{A}_2}$ (discriminant $4$, real
multiplication by $\mathbb{Z}[\sqrt{2}]$), the Gauss--Manin
monodromy action carries a $\mathbb{Z}/2$-twist
\cite[Thm.~9.3]{Bruinier2002LNM} that breaks \emph{strict}
chain-level inversion: the chiral bar complex acquires a
$\mathbb{Z}/2$-twisted factor that does not admit a global
trivialisation over $\overline{\mathcal{A}_2}$. Morrison's global
Torelli theorem for K3 surfaces \cite{Morrison84} lifts the
$\mathbb{Z}/2$-twist via the Kuga--Satake construction
\cite{KugaSatake67,Deligne72} to an \'etale $\mathbb{Z}/2$-gerbe
$\mathcal{G}_{H_4}\to H_4$ whose banding cocycle trivialises the
twist locally on \'etale neighbourhoods of $H_4$ but not globally.
The pro-object weight-completed bar complex descends through the
gerbe to yield Positselski coderived inversion; the strict
chain-level bar-cobar fails at $H_4$ in the direct-sum presentation
but is repaired by the banding cocycle on the complement
$\overline{\mathcal{A}_2}\setminus H_4$.
\end{lemma}

\begin{proof}
The $\mathbb{Z}/2$-twist arises because the Humbert divisor $H_4$
parametrises abelian surfaces $A$ with $\mathrm{End}(A) \supset
\mathbb{Z}[\sqrt{2}]$, and the real multiplication endomorphism
$\sqrt{2}$ acts with sign $\pm 1$ on the two isotypic components of
the Mukai lattice $\Lambda_{\mathrm{Muk}} = II_{4,20}$ restricted to
$H_4$. The Bruinier singular-theta lift
\cite[Thm.~9.3]{Bruinier2002LNM} produces a logarithmic branch of
$\Delta_5$ along $H_4$ whose monodromy is the involution swapping
the two $\pm 1$-isotypic components. This is precisely the twist
class in $H^1_{\text{\'et}}(H_4, \mathbb{Z}/2)$.
Morrison's global Torelli theorem for K3 surfaces \cite{Morrison84}
lifts the $\mathbb{Z}/2$-twist to a Kuga--Satake gerbe
$\mathcal{G}_{H_4}\to H_4$: the underlying abelian variety of
double dimension carries a canonical $\mathbb{Z}/2$-central
extension structure whose Chern-Simons-like banding cocycle is
\'etale-locally trivial but carries a global class in
$H^2_{\text{\'et}}(H_4, \mathbb{Z}/2)$ by the Brauer obstruction.
The banding cocycle trivialises the bar-complex twist on \'etale
neighbourhoods, yielding strict chain-level inversion on
$\overline{\mathcal{A}_2}\setminus (H_1\cup H_3\cup H_4)$; on
$H_4$ itself only the weight-completed / pro-object statement
holds, as the gerbe obstructs strict descent.
\end{proof}

$\mathbf{H}_{\Delta_5}$, and exclusion of other candidate curves.

\emph{Canonicity.} Three structural features force the curve to be
genus-$1$ with $24$ marked nodes.

(i) \emph{Genus from Siegel modularity.} The Siegel modular form $\Delta_5$
transforms with weight $5$ under $\mathrm{Sp}_4(\mathbb{Z})$ acting on the
Siegel upper half-space $\mathbb{H}_2$. Under the Kuga--Satake
correspondence \cite{KugaSatake1967,Deligne1972}, $\mathbb{H}_2$ is the

$A = A_{\mathrm{KS}}(K3)$ whose restriction to a $(2,2)$-isogeny locus
is $E\times E$ for an elliptic curve $E$. The chiral base has to carry
the modulus of $E$ as a deformation parameter, forcing $g(X_{\mathbf B})\geq 1$;
by dimension-counting of deformation parameters ($\dim\mathbb{H}_2 = 3$,
of which $1$ is the spectral direction transverse to the chiral base),
genus is exactly $1$.

(ii) \emph{$24$ marked nodes from Kodaira classification.} Kodaira's
topological classification of elliptic surfaces~\cite{Kodaira1963}
pins the number of singular fibres on a generic elliptic K3:
\[
   \sum_i (\text{Euler contribution of fibre }F_i) = \chi_{\mathrm{top}}(K3) = 24.
\]
For a generic elliptic K3 all $24$ singular fibres are of type $I_1$,
each contributing $1$ to the Euler number; this is the $I_1$-discriminant
specialisation recorded in Vol.~III
Chapter~\ref{ch:k3-chiral-bialgebra-platonic}. The $24$ marked points are
therefore forced by the topology of the elliptic-K3 fibration.

(iii) \emph{Nodal (not smooth marked) structure from $I_1$-stratum.} The
$I_1$ Kodaira type is a nodal cubic fibre; the chiral algebra acquires
a vanishing-cycle-monodromy $\mathcal{D}$-module at each node
(Vol.~III Lemma~\ref{lem:nearby-cycles-local-node}). Smooth marked
points would give no monodromy, but the BV-tower obstruction
$c_n = c_{\phi_{-2,1}}(-n)\cdot[H_n]$ of Costello 2018 requires the
local nodal vertex algebra $V^{I_1}$ at each node. Nodal structure is
therefore forced.

\emph{Exclusion.} The genus-$0$ curve $\mathbb{P}^1$ of the first four
archetype rows is ruled out: $\mathbb{P}^1$ has no elliptic modulus, so
the Siegel weight-$5$ covariance cannot be supported. The $24$-punctured
sphere $\Sigma_{0,24}$ (the class-$\mathcal{S}$ parent Riemann surface
of Vol.~III~\ref{def:classS-A1-Sigma024}) is also ruled out: this is the
$4$d parent surface, not the $2$d chiral-algebra curve; the
Beem--Rastelli reduction dualises to a different base
(cf.\ Remark~\ref{rem:confspaces-kuga-satake-mirror}).
The modular curve $X(1)=\mathbb{H}/\mathrm{SL}_2(\mathbb{Z})$
parametrises the modulus $\tau$, not supporting a chiral bracket.
The Seiberg--Witten spectral curve carries only Coulomb-branch data.

Only $E^{\mathrm{nod}}_{24}$ simultaneously supports all three structural
features (genus $1$, $24$-node $I_1$-marking, Beem--Rastelli chiral-base
duality); the exclusion is
Theorem~\ref{thm:confspaces-canonical-curve-HDeltaFive} of
Chapter~\ref{chap:configuration-spaces}.
\end{proof}

\begin{remark}[Why the crown row's curve is richer than $\mathbb{P}^1$]
\label{rem:climax-crown-curve-richer}
The four non-crown archetypes $\mathsf G/\mathsf L/\mathsf C/\mathsf M$
all live on $\mathbb{P}^1$: the Heisenberg, affine Kac--Moody,
$\beta\gamma/bc$, and Virasoro algebras are naturally $E_\infty$-chiral
on the sphere, and the Arnold KZ connection of Section~\ref{sec:climax-platonic-setup}
controls their bar differentials. The crown row $\mathbf B$ is where
the $E_1$-vs-$E_\infty$ distinction genuinely matters: the
Hall--Drinfeld double structure of $\mathbf{H}_{\Delta_5}$ is
$E_1$-only (associative, coassociative, with a non-trivial associator),
and the Borcherds automorphic-product structure adds genus and marked
nodes that sphere configurations cannot support. The curve upgrade
$\mathbb{P}^1 \leadsto E^{\mathrm{nod}}_{24}$ is the geometric signature
of the non-abelianity leap, not an accident of the K3 choice.
\end{remark}

\begin{remark}[The curve as a selection principle on the landscape]

residues, not a continuous family; see
Remark~\ref{rem:confspaces-1}.
\end{remark}
