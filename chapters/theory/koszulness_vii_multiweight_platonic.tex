\section{Factorization homology and scalar trace}
\label{sec:kos-vii-setup}
% ======================================================================

Characterization~\textup{(vii)} of Theorem~\textup{\ref{thm:koszul-equivalences-meta}}
has two logically separate layers. The first is a genus-$0$
factorization-homology criterion for chiral Koszulness, with the
detecting comparison hypothesis
$\mathsf{FH}^{\mathrm{det}}_{\mathbb{P}^1}(\cA)$ built into the
reverse implication. The second is the higher-genus scalar trace of
the modular bar complex. On a multi-weight algebra the second layer is
the diagonal term plus the mixed-channel residue:
\begin{equation}\label{eq:kos-vii-diagonal-plus-cross}
F_g(\cA)
\;=\;
F_g^{\mathrm{sc}}(\cA)
\;+\;
\delta F_g^{\mathrm{cross}}(\cA),
\qquad g\geq 1.
\end{equation}
Here
\[
F_g^{\mathrm{sc}}(\cA)
\;=\;
\kappa(\cA)\cdot\lambda_g^{\mathrm{FP}}
\]
is the diagonal scalar Faber--Pandharipande term,
$\kappa(\cA)=\sum_i\kappa_i$ is the total scalar modular
characteristic of the channel decomposition of
Theorem~\textup{\ref{thm:shadow-channel-decomposition}}, and
$\lambda_g^{\mathrm{FP}}
=\int_{\overline{\cM}_{g,1}}\psi_1^{2g-2}\lambda_g$ is the
Faber--Pandharipande number appearing in
Theorem~\textup{\ref{thm:genus-universality}}.

The correction is the mixed part of the stable-graph expansion:
\begin{equation}\label{eq:kos-vii-correction-graph-sum}
\delta F_g^{\mathrm{cross}}(\cA)
\;=\;
\sum_{\Gamma\in\mathcal{G}_{g,0}}
\frac{1}{|\operatorname{Aut}(\Gamma)|}
\sum_{\substack{\sigma\colon E(\Gamma)\to\{1,\ldots,r\}\\
\sigma\ \mathrm{mixed}}}
A_\Gamma(\sigma,\cA),
\end{equation}
where $\sigma$ is mixed when it is not constant on $E(\Gamma)$.
Construction~\textup{\ref{constr:cross-channel-graph-sum}} defines
$A_\Gamma(\sigma,\cA)$ from the inverse Zamolodchikov metric on the
edges and the OPE vertex amplitudes on the vertices. The sum is finite
for each fixed genus.

\section{Genus-zero detection and scalar trace}
\label{sec:kos-vii-theorem}
% ======================================================================

\begin{theorem}[Characterization~\textup{(vii)} with multi-weight scalar
trace; \ClaimStatusConditional]
\label{thm:koszulness-vii-multiweight-up-to-correction}
\index{Koszul equivalence (vii)!multi-weight scalar trace|textbf}
Let $\cA$ be a chiral algebra on a smooth projective curve with PBW
filtration $F_\bullet$. Assume
$\mathsf{FH}_{\mathbb{P}^1}(\cA)$ and
$\mathsf{FH}^{\mathrm{det}}_{\mathbb{P}^1}(\cA)$ in the sense of
Theorem~\textup{\ref{thm:koszul-equivalences-meta}}. Then
\[
\cA\ \textup{is chirally Koszul}
\quad\Longleftrightarrow\quad
H^k\!\left(\textstyle\int_{\mathbb{P}^1}\cA\right)=0
\quad(k\neq 0).
\]
If these equivalent conditions hold and $\cA$ satisfies the
higher-genus modular hypotheses of
Theorem~\textup{\ref{thm:multi-weight-genus-expansion}} with strong
generators $\varphi_1,\ldots,\varphi_r$ of conformal weights
$h_1,\ldots,h_r$, then the all-genera scalar trace is
\begin{equation}\label{eq:kos-vii-scalar-all-genera}
F_g(\cA)
\;=\;
F_g^{\mathrm{sc}}(\cA)
\;+\;
\delta F_g^{\mathrm{cross}}(\cA)
\qquad (g\geq 1),
\end{equation}
where
$F_g^{\mathrm{sc}}(\cA)=\kappa(\cA)\cdot\lambda_g^{\mathrm{FP}}$,
and $\delta F_g^{\mathrm{cross}}(\cA)$ is given by
\textup{\ref{eq:kos-vii-correction-graph-sum}} and
$\delta F_1^{\mathrm{cross}}(\cA)=0$.
\end{theorem}

\begin{proof}
The genus-$0$ equivalence is exactly the detected form of condition
\textup{(vii)} in
Theorem~\textup{\ref{thm:koszul-equivalences-meta}}. The comparison
$\mathsf{FH}_{\mathbb{P}^1}(\cA)$ transfers bar concentration to
$\int_{\mathbb{P}^1}\cA$, while
$\mathsf{FH}^{\mathrm{det}}_{\mathbb{P}^1}(\cA)$ reflects
degree-zero concentration back to the PBW bar model. The PBW criterion
of Theorem~\textup{\ref{thm:pbw-koszulness-criterion}} then identifies
that bar concentration with chiral Koszulness.

Assume now that these conditions hold and that the higher-genus
modular graph calculus applies to $\cA$. Theorem~\textup{\ref{thm:multi-weight-genus-expansion}}
decomposes the genus-$g$ stable-graph free energy into the diagonal
per-channel contribution and the mixed-channel contribution. The
diagonal scalar contribution is
\[
F_g^{\mathrm{sc}}(\cA)
\;=\;
\sum_i\kappa_i\lambda_g^{\mathrm{FP}}
\;=\;
\kappa(\cA)\lambda_g^{\mathrm{FP}}.
\]
This is the scalar-channel decomposition of
Theorem~\textup{\ref{thm:shadow-channel-decomposition}}. The mixed
contribution is
$\delta F_g^{\mathrm{cross}}(\cA)$ by
Construction~\textup{\ref{constr:cross-channel-graph-sum}}. At
genus~$1$ every boundary graph has at most one edge in the scalar
trace, so no mixed channel assignment occurs; equivalently,
Theorem~\textup{\ref{thm:multi-weight-genus-expansion}}\textup{(iii)}
gives $\delta F_1^{\mathrm{cross}}=0$.
\end{proof}

\begin{remark}[Scalar trace and full concentration]
\label{rem:kos-vii-scalar-not-full-concentration}
The equality \textup{\ref{eq:kos-vii-scalar-all-genera}} is an
identity after applying the scalar free-energy trace to the modular
bar complex. The full higher-genus factorization-homology
concentration statement remains the uniform-weight assertion of
Theorem~\textup{\ref{thm:koszul-equivalences-meta}}\textup{(vii)}.
The multi-weight all-genera invariant recorded here is the scalar
projection of the same bar geometry.
\end{remark}

\section{Finite-window graph formula}
\label{sec:kos-vii-universal}
% ======================================================================

\begin{proposition}[Finite-window form of the cross-channel correction;
\ClaimStatusConditional]
\label{prop:delta-f-cross-universal-formula}
\index{cross-channel correction!finite-window form|textbf}
Let $\cA$ be a modular Koszul chiral algebra whose strong-generator
OPE has channel basis
$\varphi_1,\ldots,\varphi_r$, inverse Zamolodchikov metric
$\eta^{ij}$ diagonal in that basis, and vertex amplitudes
$V_{g(v),n(v)}$. For every $g\geq 1$,
\begin{equation}\label{eq:kos-vii-universal-correction}
\delta F_g^{\mathrm{cross}}(\cA)
\;=\;
\sum_{\Gamma\in\mathcal{G}_{g,0}^{\mathrm{mix}}}
\frac{1}{|\operatorname{Aut}(\Gamma)|}
\sum_{\substack{\sigma\colon E(\Gamma)\to\{1,\ldots,r\}\\
\sigma\ \mathrm{mixed}}}
\left(
\prod_{e\in E(\Gamma)}
\eta^{\sigma(e)\sigma(e)}
\right)
\left(
\prod_{v\in V(\Gamma)}
V_{g(v),n(v)}\!\bigl(\sigma|_{H(v)}\bigr)
\right).
\end{equation}
The set $\mathcal{G}_{g,0}^{\mathrm{mix}}$ consists of stable
genus-$g$ graphs with at least two edges. If the OPE coefficients of
$\cA$ lie in the rational function field
$\mathbb{Q}(c,h_\bullet,f_\bullet,\eta_\bullet^{-1})$, then
$\delta F_g^{\mathrm{cross}}(\cA)$ also lies in that field.
\end{proposition}

\begin{proof}
Construction~\textup{\ref{constr:cross-channel-graph-sum}} writes
the full free energy as a finite sum over stable graphs and channel
assignments. Splitting the assignments into constant and mixed parts
gives the diagonal/cross decomposition. The diagonal part is
Theorem~\textup{\ref{thm:multi-weight-genus-expansion}}\textup{(i)};
the mixed part is the displayed expression. Graphs with zero or one
edge have no mixed channel assignment, so they contribute nothing to
the cross term. For fixed $g$ the set of stable graphs is finite.
Each summand is a product of finitely many inverse metric entries and
OPE vertex coefficients; rationality is preserved by finite sums and
products.
\end{proof}

\begin{theorem}[$\cW_3$ genus-$2$ cross-channel correction;
\ClaimStatusConditional]
\label{thm:delta-f-cross-w3-g2}
\index{cross-channel correction!W_3 at genus 2|textbf}
For the principal $\cW_3$ algebra, with strong generators $T$ and $W$
of conformal weights $2$ and $3$, the genus-$2$ mixed-channel
correction is
\begin{equation}\label{eq:kos-vii-w3-g2-explicit}
\delta F_2^{\mathrm{cross}}(\cW_3)
\;=\;
\frac{3}{c}+\frac{9}{2c}+\frac{1}{16}+\frac{21}{4c}
\;=\;
\frac{c+204}{16c}.
\end{equation}
It has a pole at $c=0$ and vanishes precisely at $c=-204$ among finite
values of the parameter.
\end{theorem}

\begin{proof}
The closed form is Theorem~\textup{\ref{thm:multi-weight-genus-expansion}}\textup{(vi)}
and Computation~\textup{\ref{comp:w3-genus2-multichannel}}. The last
sentence follows by factoring the displayed rational function:
$(c+204)/(16c)=0$ exactly when $c=-204$, and the denominator is zero
exactly when $c=0$.
\end{proof}

\begin{remark}[Concrete anchor]
\label{rem:kos-vii-w3-g2-anchor}
Equation~\textup{\ref{eq:kos-vii-w3-g2-explicit}} is a scalar
mixed-channel computation at one point of the landscape:
$(\cA,g)=(\cW_3,2)$. It does not supply a class-valued lift on
$R^*(\overline{\cM}_2)$, nor a closed form for every multi-weight
family.
\end{remark}

\section{Vanishing lanes}
\label{sec:kos-vii-corollaries}
% ======================================================================

\begin{corollary}[Uniform-weight scalar trace;
\ClaimStatusConditional]
\label{cor:uniform-weight-correction-vanishes}
\index{cross-channel correction!uniform-weight vanishing|textbf}
If $\cA$ lies on the uniform-weight scalar-diagonal lane, then
\[
\delta F_g^{\mathrm{cross}}(\cA)=0
\qquad(g\geq 1),
\]
and the scalar trace in Theorem~\textup{\ref{thm:koszulness-vii-multiweight-up-to-correction}}
is
\[
F_g(\cA)=F_g^{\mathrm{sc}}(\cA)
=\kappa(\cA)\cdot\lambda_g^{\mathrm{FP}}
\qquad(g\geq 1).
\]
\end{corollary}

\begin{proof}
This is Theorem~\textup{\ref{thm:multi-weight-genus-expansion}}\textup{(iv)}
together with the scalar-diagonal rigidity theorem
Theorem~\textup{\ref{thm:algebraic-family-rigidity}}.
\end{proof}

\begin{corollary}[Free-field scalar trace;
\ClaimStatusConditional]
\label{cor:free-field-correction-vanishes}
\index{cross-channel correction!free-field vanishing|textbf}
Let $\cA$ be a free-field chiral algebra: Heisenberg, lattice, $bc$, or
$\beta\gamma$. Then
\[
\delta F_g^{\mathrm{cross}}(\cA)=0
\qquad(g\geq 1),
\]
even when the strong generators have distinct conformal weights.
\end{corollary}

\begin{proof}
Proposition~\textup{\ref{prop:free-field-scalar-exact}} proves the
vanishing. In the graph formula
\textup{\ref{eq:kos-vii-universal-correction}}, the quadratic
free-field OPE makes the propagator block-diagonal by weight or by
ghost-number polarization. Every mixed assignment therefore contains
a zero edge factor or a zero vertex factor.
\end{proof}

\section{The all-genera form of condition \textup{(vii)}}
\label{sec:kos-vii-upgrade}
% ======================================================================

\begin{theorem}[All-genera form of characterization~\textup{(vii)};
\ClaimStatusConditional]
\label{thm:koszulness-vii-upgrade}
\index{Koszul equivalence (vii)!all-genera form|textbf}
Under the comparison hypotheses of
Theorem~\textup{\ref{thm:koszul-equivalences-meta}}, characterization
\textup{(vii)} has the following all-genera form.
\begin{enumerate}[label=\textup{(\roman*)}]
\item The genus-$0$ factorization-homology concentration criterion is
\[
\cA\ \textup{chirally Koszul}
\quad\Longleftrightarrow\quad
H^k\!\left(\textstyle\int_{\mathbb{P}^1}\cA\right)=0
\quad(k\neq 0),
\]
with the reverse implication requiring
$\mathsf{FH}^{\mathrm{det}}_{\mathbb{P}^1}(\cA)$.
\item On the uniform-weight scalar-diagonal lane, this extends to full
factorization-homology concentration on every smooth projective curve
$\Sigma_g$ for which $\mathsf{FH}_{\Sigma_g}(\cA)$ holds.
\item On the multi-weight lane, the all-genera invariant supplied by
the modular graph calculus is the scalar trace
\[
F_g(\cA)
\;=\;
F_g^{\mathrm{sc}}(\cA)
\;+\;
\delta F_g^{\mathrm{cross}}(\cA),
\qquad g\geq 1,
\]
with
$F_g^{\mathrm{sc}}(\cA)=\kappa(\cA)\cdot\lambda_g^{\mathrm{FP}}$ and
$\delta F_g^{\mathrm{cross}}$ computed by
\textup{\ref{eq:kos-vii-universal-correction}}.
\end{enumerate}
The equality $F_g=F_g^{\mathrm{sc}}$ is recovered on the
uniform-weight and free-field lanes. For $\cW_3$ at genus~$2$ the
cross-channel correction is \textup{\ref{eq:kos-vii-w3-g2-explicit}}.
\end{theorem}

\begin{proof}
Part~\textup{(i)} is the detected genus-$0$ form of
Theorem~\textup{\ref{thm:koszul-equivalences-meta}}\textup{(vii)}.
Part~\textup{(ii)} is the uniform-weight all-genera clause of the
same theorem, using
Theorem~\textup{\ref{thm:loop-order-collapse}} and
Theorem~\textup{\ref{thm:genus-universality}}. Part~\textup{(iii)} is
Theorem~\textup{\ref{thm:multi-weight-genus-expansion}} written with
the graph formula of
Proposition~\textup{\ref{prop:delta-f-cross-universal-formula}}. The
vanishing statements are
Corollaries~\textup{\ref{cor:uniform-weight-correction-vanishes}} and
\textup{\ref{cor:free-field-correction-vanishes}}; the $\cW_3$ value
is Theorem~\textup{\ref{thm:delta-f-cross-w3-g2}}.
\end{proof}

\begin{remark}[Boundaries of the scalar theorem]
\label{rem:kos-vii-platonic-is-II}
The scalar trace uses neither the derived centre
$Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)$ nor any Calabi--Yau, K3,
Igusa, or Yangian input. Its data are the PBW bar model, the
factorization-homology comparison at genus~$0$, the stable-graph
expansion over $\overline{\cM}_{g,0}$, the channel decomposition of
the modular characteristic, and the OPE coefficients of~$\cA$.
\end{remark}
