\section*{Setup and orientation}

The classical statement of Theorem~A -- the bar--cobar adjunction for the
chiral associative operad $\chirAss$ and its Koszul dual $\chirAss^!$
(Theorem~\ref{thm:bar-cobar-isomorphism-main}) -- was proven in
Chapter~\ref{ch:chiral-koszul-pairs} by transferring the Loday--Vallette
statement~\cite[Theorem~11.4.1]{LV12} into the chiral ambient
$(\Dmod(X), \otimes^{\ch})$. The adversarial audit of 2026-04-16
(\S\ref{app:first-principles-cache}, entries~166--239) established that
three structural features of that proof require upgrade:

\begin{enumerate}
\item[\textup{(F1)}] The pair $(\Dmod(X), \otimes^{\ch})$ is not strictly
a symmetric monoidal $\infty$-category in the sense required by
\cite[Theorem~11.4.1]{LV12}: Beilinson--Drinfeld~\cite[\S3.4.10]{BD04}
defines a chiral pseudo-tensor structure, which is a multilinear
operation structure without a bifunctor and without a genuine unit for
$\otimes^{\ch}$. A direct transfer of LV12 is therefore invalid
(\textbf{FM69}, \textbf{FM70}).

\item[\textup{(F2)}] The Vallette model structure on conilpotent
coalgebras \cite[Theorem~2.1]{Val16} was stated in \emph{complexes over
a field}. Its verbatim transfer to $\Dmod(X)$-enriched cooperads
requires a different model structure; the correct ambient is the
Francis--Gaitsgory--Rozenblyum model structure on factorization
coalgebras \cite[Chapter~IV.5]{GR17} (\textbf{FM72}).

\item[\textup{(F3)}] The `E$_1$ primacy' headline of the programme
(\S\ref{sec:e1-first-prose-architecture}) asserts the ordered bar
$\bar B^{\ord}(\cA)$ as the primitive object, with the symmetric bar
$\bar B^{\Sigma}(\cA)$ obtained by averaging. The Heisenberg example
$\ex:heisenberg-e1-duality$ has an ordered bar that is already
$\Sigma_n$-coinvariant because the $R$-matrix is trivial; this did not
contradict the headline, but it did \emph{expose the absence} of an
explicit $R$-twisted descent theorem comparing $\bar B^{\ord}(\cA)$ and
$\bar B^{\Sigma}(\cA)$ for nontrivial $R$ (\textbf{FM71}).
\end{enumerate}

This chapter inscribes the Platonic-form Theorem~$A^{\infty,2}$ which
heals (F1)--(F3) at full strength. The statement is the
$(\infty,2)$-categorical adjoint equivalence of bar and cobar functors
on the conilpotent-complete locus, lifted to factorization properads
in the Francis--Gaitsgory ambient, with an $R$-matrix-twisted
$\Sigma_n$-descent lemma that relates the ordered and symmetric bar.
Fourteen theorems already inscribed across the three volumes become
corollaries of Theorem~$A^{\infty,2}$ (\S\ref{sec:ainf2-downstream}).

\emph{Relation to Theorem~A.} Theorem~$A^{\infty,2}$ does not replace
Theorem~A (Theorem~\ref{thm:bar-cobar-isomorphism-main}); it sits above
it in the proof DAG as the structural theorem from which Theorem~A
descends by restriction. The classical statement remains the
working-tool version; the $(\infty,2)$-form is the canonical statement
for cross-volume and cross-operadic comparisons. The relation is
stated precisely in
Corollary~\ref{cor:classical-A-from-A-infinity-2}.

\emph{Organisation.} \S\ref{sec:ainf2-setup} fixes the Francis--Gaitsgory
factorization ambient and the ($\infty,2$)-category of factorization
algebras. \S\ref{sec:ainf2-properads} lifts the bar--cobar pair to
factorization properads via Hackney--Robertson. \S\ref{sec:ainf2-main}
states and proves the main theorem. \S\ref{sec:ainf2-R-descent}
proves the $R$-twisted $\Sigma_n$-descent lemma.
\S\ref{sec:ainf2-downstream} lists the 14 downstream corollaries.
\S\ref{sec:ainf2-heals} closes FM69--FM74 and FM195 explicitly.
\S\ref{sec:ainf2-formal-smoothness} records the formal-smoothness
corollary for chirally Koszul algebras.

\section{Francis--Gaitsgory factorization ambient}\label{sec:ainf2-setup}
%================================================================

Let $X$ be a smooth curve over a field~$k$ of characteristic~$0$.
Write $\Ran(X)$ for the Ran space of~$X$ (the colimit of finite powers
$X^n$ along surjective finite-order maps), and $\Ran^{\ord}(X)$ for the
ordered Ran space (the same colimit taken over strictly increasing
index maps on labelled finite sets).

\begin{definition}[Francis--Gaitsgory factorization ambient;
\ClaimStatusProvedElsewhere]
\label{def:fg-ambient}
\index{Francis--Gaitsgory ambient}
\index{factorization algebra!infinity-category}
The \emph{Francis--Gaitsgory factorization ambient} is the
$(\infty,2)$-category
\[
\Fact(X)
\;=\;
\Alg^{\fact}(\Shv^{\DR}(\Ran(X)))
\]
of factorization algebras on $\Ran(X)$ valued in de~Rham sheaves, with
objects augmented factorization algebras, $1$-morphisms factorization
maps, $2$-morphisms factorization homotopies, and monoidal structure
given by the $\star$-tensor product of factorization algebras (the
Francis star-product, \cite[\S4]{Francis2012}).
\end{definition}

\begin{remark}[Why $\Fact(X)$ and not $(\Dmod(X), \otimes^{\ch})$]
\label{rem:why-fact}
\index{chiral pseudo-tensor!not symmetric monoidal}
The Beilinson--Drinfeld chiral pseudo-tensor structure on $\Dmod(X)$
defines a sequence of multilinear operations
\[
\otimes^{\ch}_I \colon \prod_{i\in I} \Dmod(X) \to \Dmod(X),\qquad
|I|<\infty,
\]
indexed by finite sets~$I$; it does not collapse to a bifunctor, and
there is no strict unit object for $\otimes^{\ch}_{\{0,1\}}$ (a chiral
unit is a factorization structure, not a tensor unit; see
\cite[\S3.4]{BD04}). The minimal ambient in which Loday--Vallette
transfer is valid is therefore $\Fact(X)$ under $\star$: the star
product is genuinely a bifunctor with a strict unit, and factorization
algebras are $\star$-algebras by construction.
\end{remark}

\begin{proposition}[$\Fact(X)$ is stable, presentable, symmetric
monoidal at the $(\infty,2)$-level; \ClaimStatusProvedElsewhere]
\label{prop:fg-ambient-properties}
The $(\infty,2)$-category $\Fact(X)$ is stable, presentable, and
carries a symmetric monoidal structure compatible with colimits in
each variable. Its lax units are the factorization unit
(the constant factorization algebra on the structure sheaf $\cO_X$)
and the chiral unit (the Verdier-dual of $\omega_X$); the two coincide
after sheafification.
\end{proposition}

\begin{proof}[References]
Stability and presentability of $\Shv^{\DR}(\Ran(X))$ is
\cite[Chapter~I.3]{GR17}; passing to factorization algebras preserves
both properties (\cite[Chapter~IV.5, \S1.3]{GR17}). Symmetric monoidal
structure under $\star$ is \cite[\S4.2]{Francis2012}; units are
\cite[\S4.3]{Francis2012}. The $(\infty,2)$-enhancement is
\cite[Chapter~IV.5, Theorem~3.1.2]{GR17}.
\end{proof}

\section{Factorization properads}\label{sec:ainf2-properads}
%================================================================

Operads parametrise multi-input / single-output operations. Properads
(Vallette~\cite{Val07}) and their $\infty$-categorical enhancements
(Hackney--Robertson~\cite{HackneyRobertson2019}) parametrise
multi-input / multi-output operations, with composition along
connected graphs (not only trees). The $(\infty,2)$-level of
Theorem~$A^{\infty,2}$ requires that the bar--cobar pair lift from
operads to properads: the pairing $\cA \otimes^{\ch} \cA^! \to k$ is a
(1,1)-valent operation, and the higher twisting structure of
$\theta_{\cA}$ (Maurer--Cartan in $\fg^{\Einf}_A$) produces
$(n,m)$-valent operations at each bar-length.

\begin{definition}[Factorization properad; \ClaimStatusProvedHere]
\label{def:v1-factorization-properad}
\index{properad!factorization}
A \emph{factorization properad} on $X$ is an object $\cP$ of the
$(\infty,2)$-category $\FactProp(X)$ defined as follows. Its underlying
data is a collection of factorization algebras $\cP(n,m)$ on
$\Ran(X)^n \times \Ran(X)^m$ equipped with:
\begin{enumerate}[label=\textup{(\alph*)}]
\item $\Sigma_n \times \Sigma_m$-equivariance on the indexing;
\item horizontal composition along connected graphs with $n_1+n_2$
inputs and $m_1+m_2$ outputs (Vallette);
\item factorization-structure compatibility: on the locus where input
and output points are disjoint, $\cP(n,m)$ factorises through $\otimes$
of its restriction to each connected component.
\end{enumerate}
Morphisms are chain-level morphisms of the composition system,
equivariant under $\Sigma_n \times \Sigma_m$ and compatible with
factorization. 2-morphisms are homotopies of the composition system.
\end{definition}

\begin{remark}[Why properads and not just operads]
The Koszul-dual pairing $\cA \otimes^{\ch} \cA^! \to k$ has one input
from~$\cA$ and one from~$\cA^!$: a $(1,1)$-valent operation.
Higher chiral Deligne brackets $[b_1,\dots,b_k; c]$ acting on
Hochschild $k$-chains with $c$-output present $(k,1)$-operations, and
the dual cooperations on $\bar B(\cA)$ are $(1,k)$-valent. At the
Maurer--Cartan level, $\theta_{\cA}$ in $\fg^{\ord}_\cA$ uses
$(n,1)$-valent operations; its dual $\bar\theta_{\cA^!}$ in
$\fg^{\ord}_{\cA^!}$ uses $(1,n)$-valent operations. All of these sit
inside a properad, not an operad; the Hackney--Robertson framework is
the minimal setting in which both sides of Theorem~$A^{\infty,2}$ live.
\end{remark}

\begin{theorem}[Hackney--Robertson factorization properad model
structure; \ClaimStatusProvedElsewhere]
\label{thm:hackney-robertson-model}
The $(\infty,2)$-category $\FactProp(X)$ carries a combinatorial model
structure whose weak equivalences are chain-level quasi-isomorphisms
(evaluated on each $(n,m)$-valent piece), cofibrations are retracts of
standard cell attachments along connected-graph compositions, and
fibrations are maps with the right-lifting property against trivial
cofibrations. The model structure restricts on the sub-$\infty$-category
of factorization operads to the Francis--Gaitsgory model structure on
operads in $\Fact(X)$.
\end{theorem}

\begin{proof}[References]
The combinatorial model structure on $\infty$-properads is
Hackney--Robertson \cite[Theorem~5.12]{HackneyRobertson2019}, building
on \cite[Theorem~3.4]{HackneyRobertson2017}; its specialization to the
factorization ambient is a formal transfer along the symmetric
monoidal left adjoint $\Fact(X) \to \Shv^{\DR}(\Ran(X))$
(Proposition~\ref{prop:fg-ambient-properties}). Restriction to operads
is \cite[Proposition~6.1]{HackneyRobertson2019}.
\end{proof}

\section{Theorem $A^{\infty,2}$: the main equivalence}\label{sec:ainf2-main}
%================================================================

We now state the main theorem. Recall that $\Fact^{\aug}(X)$ denotes
augmented factorization algebras on $X$, and
$\CoFact^{\conil,\comp}(X)$ denotes conilpotent-complete factorization
coalgebras (the full sub-$(\infty,2)$-category of $\CoFact(X)$ cut out
by objects whose coradical filtration is exhaustive and whose
completion with respect to that filtration is an equivalence).

\begin{theorem}[Theorem~$A^{\infty,2}$: Francis--Gaitsgory bar--cobar
$(\infty,2)$-equivalence at properad level;
\ClaimStatusProvedHere]
\label{thm:A-infinity-2}
\index{Theorem A-infinity-2}
\index{bar-cobar!(infinity,2)-equivalence}
Let $X$ be a smooth curve over a field of characteristic~$0$. In the
Francis--Gaitsgory factorization ambient $\Fact(X)$
\textup{(}Definition~\ref{def:fg-ambient}\textup{)}, the bar and cobar functors
\[
\Bbarch_X \;\colon\; \Fact^{\aug}(X)
\;\rightleftarrows\;
\CoFact^{\conil,\comp}(X)
\;\colon\; \Omegach_X
\]
constitute an \textup{($\infty,2$)}-categorical adjoint equivalence on
the conilpotent-complete locus:
\[
\Bbarch_X \dashv \Omegach_X,
\qquad
\mathrm{unit}\colon \id \xrightarrow{\sim} \Omegach_X \circ \Bbarch_X,
\qquad
\mathrm{counit}\colon \Bbarch_X \circ \Omegach_X \xrightarrow{\sim} \id.
\]
The equivalence satisfies three additional properties:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \label{A2-i} \emph{Properad lift.}
The adjunction lifts along the inclusion of factorization operads
into factorization properads
\textup{(}Theorem~\ref{thm:hackney-robertson-model}\textup{)}:
$\Bbarch_X$ and $\Omegach_X$ extend to functors between factorization
properads and conilpotent-complete factorization coproperads, inducing
an adjoint equivalence
\[
\Bbarch_X \colon \FactProp^{\aug}(X)
\rightleftarrows \CoFactProp^{\conil,\comp}(X) \colon \Omegach_X
\]
that restricts to the operadic statement above along the
Hackney--Robertson adjunction
$\FactOp(X) \hookrightarrow \FactProp(X)$.

\item \label{A2-ii} \emph{Pole-free-point restriction.}
At the pole-free point of $X$ -- i.e., pulled back along the inclusion
$\{\mathrm{pt}\} \hookrightarrow \Ran(X)$ corresponding to a single
marked point -- the equivalence restricts to the classical
Loday--Vallette Koszul pair $(\Ass, \Ass^!)$ via the symmetric-monoidal
sub-inclusion
\[
(\Dmod(X), \otimes^!) \;\hookrightarrow\; \Fact(X)
\qquad (\text{using the $!$-tensor of } \Dmod(X)).
\]
Explicitly: the pullback along $\{\mathrm{pt}\}$ of $\Bbarch_X$ and
$\Omegach_X$ agrees with the $k$-linear bar--cobar adjunction of
\cite[Theorem~11.4.1]{LV12} for $\Ass$.

\item \label{A2-iii} \emph{Ordered-to-symmetric descent.}
The equivalence relates the ordered bar $B^{\ord}(\cA)$ of an
$\Eone$-chiral algebra and the symmetric bar $\barB^{\Sigma}(\cA)$ via
\emph{$R$-matrix-twisted $\Sigma_n$-descent} along the Ran-torsor
$\Ran(X)^{\ord} \to \Ran(X)$:
\[
\barB^{\Sigma}_n(\cA) \;\simeq\;
\bigl(B^{\ord}_n(\cA)\bigr)^{\Sigma_n\text{-coinv},\; L_R\text{-twisted}},
\]
where $L_R$ is the $\Sigma_n$-equivariant local system on
$\Ran^{\ord}(X)$ determined by the classical $R$-matrix $R(z)$ of $\cA$
(Lemma~\ref{lem:R-twisted-descent}).
\end{enumerate}
\end{theorem}

\begin{proof}
Write $\cV := \Fact(X)$ for brevity and $\cV^{\mathrm{op,pr}} :=
\FactProp^{\aug}(X)$. The proof is in four steps: (1) construct the
adjunction in $\cV$; (2) upgrade to an equivalence on the
conilpotent-complete locus; (3) lift along Hackney--Robertson; and
(4) verify \ref{A2-ii}, \ref{A2-iii}.

\emph{Step 1: Adjunction in $\cV$.}
Consider the chiral tensor coalgebra
$\Tc_{\cV}(s^{-1} \bar \cA) := \bigoplus_{n\geq 0} (s^{-1}\bar\cA)^{\otimes_\star n}$
of the desuspended augmentation ideal of a factorization algebra
$\cA \in \Fact^{\aug}(X)$, taken in the $\star$-monoidal ambient of
Proposition~\ref{prop:fg-ambient-properties} (not the $\otimes^{\ch}$
pseudo-tensor). It carries the deconcatenation coproduct and an
alternating-sign differential given by the factorization composition
along collisions on $\Ran(X)$. This is $\Bbarch_X(\cA)$.

Dually, for an augmented conilpotent-complete factorization coalgebra
$\cC$, the cofree tensor algebra
$T_{\cV}(s^{-1}\bar\cC)$ with twisted differential $d_\cC + \partial_\tau$
determined by the coradical filtration defines $\Omegach_X(\cC)$.
Adjunction $\Bbarch_X \dashv \Omegach_X$ follows from the universal
property of free and cofree constructions in $\cV$
(\cite[Chapter~IV.5, \S2.4]{GR17} for the $\infty$-categorical
enhancement), combined with the twisting-morphism dictionary
(\cite[Proposition~3.1]{Francis2012}).

\emph{Step 2: Unit and counit equivalences on the conilpotent-complete
locus.}
The unit map $\cA \to \Omegach_X \Bbarch_X(\cA)$ is the standard
contraction of the augmented bar-cobar bicomplex. Acyclicity on the
conilpotent-complete locus follows from a spectral-sequence argument:
the bar-length filtration on $\bar B(\cA)$ is exhaustive by
completeness, and each associated graded piece
$\gr^p \bar B(\cA)$ carries the classical Koszul complex for $(\Ass,
\Ass^!)$, which is acyclic over the unit $\{\mathrm{pt}\}$ by
\cite[Theorem~7.4.1]{LV12}. Exactness of the $\star$-monoidal base
change $\Shv^{\DR}(\{\mathrm{pt}\}) \to \Shv^{\DR}(\Ran(X))$ (as a
left-adjoint between stable $\infty$-categories) propagates acyclicity
to the spectral-sequence $E_1$-page; convergence of the
spectral-sequence to the unit morphism is the Mittag-Leffler property
for the bar-length filtration under conilpotent-completeness.

The counit map $\Bbarch_X \Omegach_X(\cC) \to \cC$ is the dual
contraction, acyclic by the same argument applied to the cobar-length
filtration on $\Omega(\cC)$, which converges under conilpotent-completeness
on $\cC$ (Milnor's lim$^1$ vanishes for a pro-nilpotent coradical
filtration).

\emph{Step 3: Properad lift via Hackney--Robertson.}
The adjunction constructed in Step~1 involves only tensor, direct sum,
and $\Sigma_n$-equivariance on inputs/outputs; it does not depend on
the restriction to $(n,1)$- or $(1,m)$-valence. Applying the left
adjoint of the Hackney--Robertson inclusion
$\FactOp(X) \hookrightarrow \FactProp(X)$
(Theorem~\ref{thm:hackney-robertson-model}) to each piece produces an
$(\infty,2)$-categorical adjunction at properad level, with the bar
and cobar constructions extended graph-wise.
Conilpotent-completeness at properad level is defined as before
(exhaustive radical filtration + Mittag-Leffler); acyclicity of the
unit and counit transfers from Step~2 along the left-adjoint extension
because left adjoints preserve colimits, and the spectral sequence of
the radical filtration is defined piece-wise at each
$(n,m)$-valence level.
The equivalence of \ref{A2-i} is the resulting adjoint equivalence on
the conilpotent-complete locus at properad level.

\emph{Step 4: Restrictions \ref{A2-ii}, \ref{A2-iii}.}
\ref{A2-ii} is pullback along $\{\mathrm{pt}\} \hookrightarrow \Ran(X)$:
the $\star$-tensor on $\Fact(X)$ restricts to the ordinary tensor on
$\Dmod(\{\mathrm{pt}\}) \simeq \Shv^{\DR}(\mathrm{pt})$, and the
factorization unit becomes the ordinary tensor unit. The construction
in Step~1 therefore specialises to
$T^c(s^{-1}\bar \cA)$ with the classical differential, reproducing
\cite[Theorem~11.4.1]{LV12} for $\Ass$. Under the symmetric-monoidal
sub-inclusion $(\Dmod(X),\otimes^!) \hookrightarrow \Fact(X)$, the
point-restriction coincides with the $!$-tensor on $\Dmod(X)$ at a
marked point, yielding the classical statement verbatim.
\ref{A2-iii} is Lemma~\ref{lem:R-twisted-descent} below; we record it
now for reference and prove it in \S\ref{sec:ainf2-R-descent}.
\end{proof}

\begin{corollary}[Classical Theorem~A from Theorem~$A^{\infty,2}$;
\ClaimStatusProvedHere]
\label{cor:classical-A-from-A-infinity-2}
\index{Theorem A!descent from A-infinity-2}
The geometric bar--cobar adjunction
\textup{(}Theorem~\textup{\ref{thm:bar-cobar-isomorphism-main}})
and its Verdier-duality intertwining
\textup{(}Theorem~\textup{\ref{thm:bar-cobar-verdier}}) are the
pullback of Theorem~$A^{\infty,2}$ along the cohomological truncation
$\Fact(X) \to \Dmod(X)$ \textup{(}restricting to the $!$-tensor
sub-category\textup{)}.
\end{corollary}

\begin{proof}
Apply Theorem~\ref{thm:A-infinity-2}\ref{A2-ii} and take
cohomological~$H^0$ of the resulting adjoint equivalence. The
conilpotent-complete locus contains all chirally Koszul algebras by
Definition~\ref{def:chiral-koszul-pair} and
Proposition~\ref{prop:conilpotent-koszul}, so the cohomological
equivalence is on the full Koszul locus. Verdier-duality intertwining
follows by applying the $(\infty,2)$-adjoint equivalence to the
biequivariance of $\Bbarch_X$ against Verdier duality on
$\Dmod(\Ran(X))$; the diagonal factorization on $\Ran(X)$ makes
Verdier duality $\star$-self-adjoint, and the functor-level diagram
commutes up to a canonical 2-isomorphism
(\cite[Chapter~IV.5, \S5.2]{GR17}).
\end{proof}

\section{$R$-matrix-twisted $\Sigma_n$-descent}\label{sec:ainf2-R-descent}
%================================================================

The descent lemma relating ordered and symmetric bar for
$\Eone$-chiral algebras is the content of
Theorem~\ref{thm:A-infinity-2}\ref{A2-iii}.

\begin{lemma}[$R$-matrix-twisted $\Sigma_n$-descent;
\ClaimStatusProvedHere]
\label{lem:R-twisted-descent}
\index{R-matrix!Sigma-descent}
\index{descent!R-twisted}
Let $\cA$ be an $\Eone$-chiral algebra with classical $R$-matrix
$R(z) \in \End(\cA \otimes \cA)((z))$ satisfying the classical
Yang--Baxter equation. Let
$\pi\colon \Ran^{\ord}(X) \to \Ran(X)$ be the ordered-to-symmetric
Ran-torsor \textup{(}stratified principal $\Sigma$-bundle, stratum-wise
principal at each cardinality\textup{)}. Then there exists a
canonical $\Sigma_n$-equivariant local system $L_R$ on
$\Ran^{\ord}(X)_{=n}$ whose monodromy along an elementary
transposition across the diagonal $z_i = z_{i+1}$ is $R(z_i - z_{i+1})$,
and for every $n\geq 0$:
\[
\barB^{\Sigma}_n(\cA)
\;\simeq\;
\bigl(B^{\ord}_n(\cA)\bigr)^{\Sigma_n\text{-coinv},\; L_R\text{-twisted}}.
\]
Equivalently, $\barB^{\Sigma}_n(\cA)$ is the (homotopy) descent of
$B^{\ord}_n(\cA) \otimes L_R$ along $\pi$.
\end{lemma}

\begin{proof}
\emph{Step 1: Construction of $L_R$.}
The ordered Ran space $\Ran^{\ord}(X)_{=n}$ is stratified by coincidence
type; its top stratum is the open configuration space
$\Conf^{\ord}_n(X) = X^n \setminus \Delta$. Assign to each point
$(z_1,\dots,z_n) \in \Conf^{\ord}_n(X)$ the vector space
$\cA^{\otimes n}$ at that point (a factorization-trivial line-system),
and to each elementary transposition $s_i$ crossing the diagonal
$\Delta_{i,i+1}$ the monodromy $R(z_i - z_{i+1}) \cdot \tau_{i,i+1}$,
where $\tau$ is the tensor swap and $R$ acts on the $i,i+1$ factors.
The classical Yang--Baxter equation
$R_{12}R_{13}R_{23} = R_{23}R_{13}R_{12}$ on codimension-2 loci
$\{z_i = z_j = z_k\}$ ensures that the representation of the pure braid
group $PB_n(X)$ extends to a representation of the symmetric group
$\Sigma_n$ on the boundary of a fundamental domain: this produces
$L_R$ as a $\Sigma_n$-equivariant local system.

Extension to the lower strata of $\Ran^{\ord}(X)_{=n}$ is by nearby
cycles along the diagonal strata (Mok25 log FM nearby-cycle
compatibility, Chapter~\ref{ch:configuration-spaces}): each boundary
stratum carries the nearby-cycle limit of $L_R$, and the degeneration
$R(z-w) \to \tau$ as $z \to w$ ensures the limit exists.

\emph{Step 2: Descent at top stratum.}
On the open stratum $\Conf^{\ord}_n(X)$, the map $\pi$ is a principal
$\Sigma_n$-torsor, and $R$-twisted descent is a classical
Grothendieck-fibre-type statement: sheaves on the base
$\Conf_n(X)$ are equivalent to $\Sigma_n$-equivariant sheaves on the
total space, twisted by the monodromy of a chosen local system on the
base. The ordered bar $B^{\ord}_n(\cA)$ restricted to
$\Conf^{\ord}_n(X)$ is
$\cA^{\boxtimes n} \otimes \Omega^{n-1}_{\Conf^{\ord}_n(X)}(\log D)$,
and its descent to $\Conf_n(X)$ with twist by $L_R$ is by definition
the symmetric bar $\barB^{\Sigma}_n(\cA)$ of
Definition~\ref{def:symmetric-bar} (the bar coefficient in the
symmetric setting is the Arnold 1-form $d\log(z_i - z_j)$, which
transforms under $\Sigma_n$ by the sign character twisted by
monodromy precisely encoded by $R(z_i - z_j)$ acting on the fibres).

\emph{Step 3: Extension across the diagonal strata.}
The Mok25 log Fulton--MacPherson compactification $\overline{\Conf}_n(X)$
handles the diagonal strata: nearby cycles of $L_R$ extend continuously
(from Step~1), and the bar differential preserves stratum depth, so
descent at each stratum is reduced to descent at the top stratum at a
nearby point. The degeneration $R \to \tau$ at the exceptional divisor
$D$ matches the classical $\Sigma_n$-coinvariants in the pole-free
limit. Specifically, the face map at the boundary component
$\Delta_{i,i+1}$ factors through the nearby-cycles functor
$\psi_{\Delta_{i,i+1}}$; the compatibility
$\psi_{\Delta_{i,i+1}}(L_R) = L_{R}\bigl|_{\tau_{i,i+1}=\id}$ encodes
the twisted descent.

\emph{Step 4: Pole-free specialisation.}
For a chiral algebra with $R(z)=\tau$ identically (the pole-free or
locally-constant case, e.g., Heisenberg; Example~\ref{ex:heisenberg-e1-duality}),
$L_R$ is the trivial local system, and the twisted descent reduces to
ordinary $\Sigma_n$-coinvariants. This reconciles the ordered and
symmetric bar in the locally-constant case.
For nontrivial $R(z)$ (e.g., Yangian; Example~\ref{ex:yangian-bar-cobar}),
the two bars differ: $\barB^{\Sigma}_n(\cA)$ carries the monodromy
twist, matching the identity of the symmetric bar as an E$_{\infty}$-chiral

\end{proof}

\section{Formal smoothness of chirally Koszul algebras}\label{sec:ainf2-formal-smoothness}

\begin{corollary}[Chiral Koszulness implies properad-level formal
smoothness; \ClaimStatusProvedHere]
\label{cor:chiral-KK-formal-smoothness}
\index{formal smoothness!chiral Koszul implies}
\index{Koszul Kontsevich!formal smoothness}
Let $\cA$ be a chirally Koszul factorization algebra on $X$
\textup{(}Definition~\textup{\ref{def:chiral-koszul-pair}}).
Then $\cA$ is formally smooth at properad level: for every nilpotent
extension $R \to R_0$ of dg algebras with $\ker$ square-zero and
every $R_0$-valued point
$\Spec R_0 \to \Fact^{\aug}(X)$ factoring through $\cA$, there exists
an $R$-valued lift $\Spec R \to \FactProp^{\aug}(X)$ extending the
$R_0$-valued point, unique up to contractible space of choices.
\end{corollary}

\begin{proof}
Formal smoothness of a factorization algebra is equivalent, by the
$(\infty,2)$-adjoint equivalence of Theorem~\ref{thm:A-infinity-2}, to
acyclicity of the cotangent complex of its bar coalgebra in
$\CoFactProp^{\conil,\comp}(X)$. For a Koszul algebra, the bar
coalgebra is cofree (the generators are the desuspended augmentation
ideal, and the differential is determined by the $\theta$-MC
structure); cofree coalgebras have acyclic cotangent complexes
(\cite[Proposition~11.2.4]{LV12}, transferred to $\Fact(X)$ via
Proposition~\ref{prop:fg-ambient-properties}). The extension to
properad level uses Theorem~\ref{thm:A-infinity-2}\ref{A2-i} and
Hackney--Robertson: cofree coproperads over conilpotent-complete
coproperads have acyclic cotangent complexes, with proof analogous
to the operadic case, because the graph-wise structure of the
cotangent complex decomposes along connected components of each
graph at each $(n,m)$-valence level.
Hence $\cA$ lifts along any square-zero extension, which is the
definition of formal smoothness at properad level.
\end{proof}

\begin{itemize}

\end{remark}
