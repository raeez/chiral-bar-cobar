\section{Genus-Graded Koszul Duality}

\begin{theorem}[Extended Koszul Duality; \ClaimStatusProvedHere]
\label{thm:extended-koszul}
Let $(\mathcal{A}, \mathcal{A}^!)$ form a genus-0 Koszul dual pair of chiral algebras
with central curvature at all genera.  Assume that $\mathcal{A}$ is locally
finite-dimensional and that the genus-graded bar complex
$\bar{B}^{\mathrm{full}}(\mathcal{A}) = \bigoplus_{g \geq 0} \bar{B}^{(g)}(\mathcal{A})$
converges in the $\hbar$-adic topology (where $\hbar$ tracks genus).  Then:
$$\left(\bigoplus_{g \geq 0} \mathcal{A}^{(g)}, \bigoplus_{g \geq 0} (\mathcal{A}^!)^{(g)}\right)$$
form a multi-genus Koszul dual pair with pairing:
$$\langle -, - \rangle: \mathcal{A}^{(g)} \otimes (\mathcal{A}^!)^{(g)} \to \mathbb{C}[\![\hbar]\!]$$
where $\hbar$ tracks the genus.
\end{theorem}

\begin{proof}
By the Prism Principle (Theorem~\ref{thm:prism-higher-genus}), the full
bar complex decomposes as a Feynman transform:
\[
\bar{B}^{\mathrm{full}}(\mathcal{A})
\simeq \mathrm{FT}_{\mathcal{M}od}(\mathcal{A})
= \bigoplus_{g \geq 0} \bar{B}^{(g)}(\mathcal{A}),
\]
and the differential satisfies $(d^{(g)})^2 = 0$ at each genus by
Theorem~\ref{thm:genus-induction-strict} (using central curvature).

\textbf{Step 1: Genus-$g$ bar-cobar duality.}
By the higher-genus bar-cobar inversion
(Theorem~\ref{thm:higher-genus-inversion}), we have
$\Omega_g(\bar{B}_g(\mathcal{A})) \simeq \mathcal{A}^{(g)}$
at each genus~$g$.  Since $(\mathcal{A}, \mathcal{A}^!)$ form a Koszul pair at genus~0,
$\bar{B}^{(0)}(\mathcal{A}) \simeq (\mathcal{A}^!)^{(0)}$ as coalgebras.

\textbf{Step 2: Higher-genus pairing.}
Define $(\mathcal{A}^!)^{(g)} := (\bar{B}^{(g)}(\mathcal{A}))^\vee$ (the linear dual of the genus-$g$ bar complex, which is a coalgebra) for
each $g \geq 0$.  The Verdier duality pairing on
$\overline{\mathcal{M}}_{g,n}$ (Theorem~\ref{thm:poincare-verdier})
extends the genus-0 Koszul pairing to a non-degenerate pairing
$\langle -, - \rangle_g\colon \mathcal{A}^{(g)} \otimes
(\mathcal{A}^!)^{(g)} \to \mathbb{C}$.  Summing over genera with
the bookkeeping parameter $\hbar^{2g}$ yields the formal pairing
with values in $\mathbb{C}[\![\hbar]\!]$.

\textbf{Step 3: Involutivity.}
The universal extension tower
(Theorem~\ref{thm:universal-extension-tower}) assembles the
genus-by-genus dualities into a compatible system.  Applying bar-cobar
inversion at each level and passing to the inverse limit shows
$(\mathcal{A}^!)^! \cong \mathcal{A}$ in $\mathbb{C}[\![\hbar]\!]$-modules,
establishing that the pair is Koszul in the multi-genus sense.
\end{proof}

\begin{remark}
The extension from genus 0 to all genera is \emph{not automatic}: the genus-$g$ bar complex involves integration over $\overline{\mathcal{M}}_{g,n}$ (not just $\overline{\mathcal{M}}_{0,n}$), and the resulting higher operations may fail to satisfy the required compatibility conditions. The main obstruction is that the Koszul property (diagonal Ext vanishing) at genus 0 does not formally imply the analogous vanishing at higher genus, where modular forms and period integrals introduce new cohomological contributions. Specific cases (Heisenberg, affine Kac-Moody at generic level) can be verified directly.
\end{remark}

\section{Definition and Basic Properties}

\begin{definition}[Genus-Graded Koszul Algebra]
A genus-graded associative algebra $\mathcal{A} = \bigoplus_{g \geq 0} \mathcal{A}^{(g)}$ is \emph{Koszul} if at each genus $g$:
$$\text{Ext}_{\mathcal{A}^{(g)}}^{i,j}(\mathbb{k}, \mathbb{k}) = 0 \quad \text{for } i \neq j$$
where $(i,j)$ is the bigrading by homological degree and internal (weight) degree, and additionally the genus-$g$ bar complex $B^{(g)}(\mathcal{A})$ is acyclic outside the diagonal $i = j$ for each $g \geq 0$.
\end{definition}

\begin{theorem}[Genus-Graded Koszul Duality Theorem; \ClaimStatusProvedHere]
\label{thm:genus-graded-koszul-duality}
If $\mathcal{A}$ is genus-graded Koszul with central curvature, then:
$$\mathcal{A}^! := \bigoplus_{g \geq 0} \text{Ext}_{\mathcal{A}^{(g)}}^*(\mathbb{k}, \mathbb{k})$$
is also genus-graded Koszul, and $(\mathcal{A}^!)^! \cong \mathcal{A}$.
\end{theorem}

\begin{proof}
We argue genus by genus, then assemble via the modular operad.

\textbf{Step 1: Ext computation at each genus.}
Fix $g \geq 0$.  The genus-$g$ bar complex $\bar{B}^{(g)}(\mathcal{A})$
computes $\mathrm{Ext}_{\mathcal{A}^{(g)}}^*(\mathbb{k}, \mathbb{k})$
(this is the defining property of the bar construction).  By
Theorem~\ref{thm:genus-induction-strict}, $(d^{(g)})^2 = 0$ strictly,
so $\mathrm{Ext}_{\mathcal{A}^{(g)}}^{i,j}(\mathbb{k}, \mathbb{k})$ is
well-defined.  The Koszul condition $\mathrm{Ext}^{i,j} = 0$ for
$i \neq j$ at genus~$g$ implies that the bar complex is concentrated on
the diagonal in the bigrading.

\textbf{Step 2: Koszulity of $\mathcal{A}^!$.}
Define $(\mathcal{A}^!)^{(g)} = \mathrm{Ext}_{\mathcal{A}^{(g)}}^*(\mathbb{k}, \mathbb{k})$.
By the $\Eone$-chiral Koszul duality theorem
(Theorem~\ref{thm:e1-chiral-koszul-duality}), the bar-cobar adjunction
gives a quasi-isomorphism
$\Omega^{\mathrm{ch}}(\bar{B}^{\mathrm{ch}}(\mathcal{A}^{(g)}))
\simeq \mathcal{A}^{(g)}$.  The cobar construction
$\Omega((\mathcal{A}^!)^{(g)})$ thus recovers $\mathcal{A}^{(g)}$, and the
bar construction $\bar{B}((\mathcal{A}^!)^{(g)})$ computes
$\mathrm{Ext}_{(\mathcal{A}^!)^{(g)}}^*(\mathbb{k}, \mathbb{k})$.  The
diagonal vanishing for $\mathcal{A}^!$ follows from the dual filtration
argument: the spectral sequence of the cobar filtration
(Theorem~\ref{thm:bar-cobar-spectral-sequence})
has $E_1$-page concentrated on the diagonal, hence collapses at $E_2$.
This gives $\mathrm{Ext}_{(\mathcal{A}^!)^{(g)}}^{i,j} = 0$ for
$i \neq j$, i.e., $\mathcal{A}^!$ is genus-graded Koszul.

\textbf{Step 3: Involutivity.}
The bar-cobar inversion
(Theorem~\ref{thm:bar-cobar-inversion-qi}) gives
$\Omega(\bar{B}(\mathcal{A})) \xrightarrow{\sim} \mathcal{A}$, which
combined with the analogous quasi-isomorphism for $\mathcal{A}^!$
yields $(\mathcal{A}^!)^! \simeq \mathcal{A}$.  More precisely:
the natural map $\mathcal{A} \to \Omega(\bar{B}(\mathcal{A}))$ is a QI
by the genus-graded bar-cobar theorem (applied at each genus using
Theorem~\ref{thm:higher-genus-inversion}), and the identification
$\bar{B}(\mathcal{A}) \simeq \mathcal{A}^!$ (as DG chiral coalgebras)
identifies $\Omega(\bar{B}(\bar{B}(\mathcal{A})))$ with the double dual
$(\mathcal{A}^!)^!$, which is quasi-isomorphic to $\mathcal{A}$ via
the unit-counit composition.
\end{proof}

\subsection{Genus-Graded Chiral Koszul Duality}

For chiral algebras across all genera, we need a modified definition:

\begin{definition}[Genus-Graded Chiral Koszul Duality]
Genus-graded chiral algebras $\mathcal{A} = \bigoplus_{g \geq 0} \mathcal{A}^{(g)}$ and $\mathcal{B} = \bigoplus_{g \geq 0} \mathcal{B}^{(g)}$ are Koszul dual if:
$$\mathrm{Ext}^{i,j}_{\mathcal{A}^{(g)}}(\mathbb{C}, \mathbb{C}) = 0 \quad \text{for } i \neq j$$
(diagonal vanishing in the bigraded Ext) in the derived category of chiral modules at each genus $g$, with $\mathcal{B}^{(g)} = (\mathcal{A}^{(g)})^!$ the quadratic dual, and modular covariance under $\mathrm{Sp}(2g, \mathbb{Z})$ transformations.
\end{definition}

\subsection{Curved and Filtered Generalizations Across Genera}

\begin{definition}[Curved Koszul Duality]
A genus-graded curved $A_\infty$ algebra
$(\mathcal{A}^{(g)}, m_0^{(g)}, m_1^{(g)}, m_2^{(g)}, \ldots)$ satisfies
\[
(m_1^{(g)})^2(a) = m_2^{(g)}(m_0^{(g)}, a) - m_2^{(g)}(a, m_0^{(g)}).
\]
Its Koszul dual is an \emph{uncurved} dg coalgebra
\[
((\mathcal{A}^{(g)})^!, d^{!(g)}) \quad \text{with } (d^{!(g)})^2 = 0.
\]
This asymmetry---the dual of a curved algebra is uncurved, and vice
versa---is fundamental to curved Koszul duality (Positselski), with modular
corrections from period integrals at each genus.
\end{definition}

\subsection{Computational Tools Across Genera}

\begin{lemma}[Genus-Graded Koszul Complex Resolution; \ClaimStatusProvedHere]
\label{lem:genus-graded-koszul-resolution}
For genus-graded Koszul $\mathcal{A}$ with central curvature, the minimal resolution of $\mathbb{k}$ at genus $g$ is:
$$\cdots \to \mathcal{A}^{(g)} \otimes (\mathcal{A}^!)_{(2)}^{(g)} \to \mathcal{A}^{(g)} \otimes (\mathcal{A}^!)_{(1)}^{(g)} \to \mathcal{A}^{(g)} \to \mathbb{k}$$
where $(\mathcal{A}^!)_{(n)}^{(g)}$ is the degree $n$ part of $\mathcal{A}^!$ at genus $g$, with the differential depending on the genus-$g$ conformal structure through the propagator on $\overline{C}_n(\Sigma_g)$.
\end{lemma}

\begin{proof}
The genus-$g$ bar complex $\bar{B}^{(g)}(\mathcal{A})$ computes
$\mathrm{Ext}_{\mathcal{A}^{(g)}}^*(\mathbb{k}, \mathbb{k})$
by construction.  By the Koszul hypothesis, this Ext is concentrated on
the diagonal ($\mathrm{Ext}^{i,j} = 0$ for $i \neq j$), so the
bar complex provides a minimal (i.e., linear) resolution.

Concretely, the $n$-th term of the bar complex at genus $g$ is:
\[
\bar{B}^{(g)}_n(\mathcal{A})
= \Gamma\!\left(\overline{C}_n(\Sigma_g),\;
j_* j^* \mathcal{A}^{\boxtimes n} \otimes
\Omega^n_{\log}\right).
\]
By Theorem~\ref{thm:geometric-equals-operadic-bar}, this equals
$\mathcal{A}^{(g)} \otimes (\mathcal{A}^!)_{(n)}^{(g)}$, where the
tensor product structure comes from splitting into the
``coefficient'' direction ($\mathcal{A}^{(g)}$-module structure)
and the ``Koszul dual'' direction (the cosyzygy
$(\mathcal{A}^!)_{(n)}^{(g)}$ extracted by integration over
$\overline{C}_n(\Sigma_g)$).

The differential is $d^{(g)} = d_{\mathrm{residue}} + d_{\mathrm{quantum}}$,
where:
\begin{itemize}
\item $d_{\mathrm{residue}}$ extracts OPE coefficients via residues along collision
divisors on $\overline{C}_n(\Sigma_g)$;
\item $d_{\mathrm{quantum}}$ incorporates the genus-$g$ quantum correction from
the propagator on $\Sigma_g$.
\end{itemize}
The exactness of this complex (away from $\mathbb{k}$) is the content of the
higher-genus bar-cobar inversion
(Theorem~\ref{thm:higher-genus-inversion}): the augmented complex
$\bar{B}^{(g)}(\mathcal{A}) \to \mathbb{k}$ is a resolution because
$\Omega_g(\bar{B}_g(\mathcal{A})) \simeq \mathcal{A}^{(g)}$.
\end{proof}

\subsection{Genus-Graded Maurer-Cartan Elements and Twisting}

\begin{theorem}[Genus-Graded MC Elements Parametrize Deformations; \ClaimStatusProvedHere]
\label{thm:genus-graded-mc}
For a genus-graded chiral algebra $\mathcal{A} = \bigoplus_{g \geq 0} \mathcal{A}^{(g)}$ with
central curvature and its bar complex $\bar{B}(\mathcal{A})$:

\textbf{1. Genus-Graded Maurer-Cartan Equation:}
$$\alpha^{(g)} \in \barBgeom^{(g)}(\mathcal{A}), \quad d^{(g)}\alpha^{(g)} + \frac{1}{2}[\alpha^{(g)}, \alpha^{(g)}] = 0$$
with modular corrections from period integrals.

\textbf{2. Genus-Graded Twisting:}
Each MC element $\alpha^{(g)}$ yields a twisted differential:
$$d_{\alpha^{(g)}}^{(g)} = d^{(g)} + [\alpha^{(g)}, -]$$
with $(d_{\alpha^{(g)}}^{(g)})^2 = 0$ and modular covariance.

\textbf{3. Genus-Graded Deformation:}
MC elements correspond to first-order deformations of $\mathcal{A}^{(g)}$:
$$\mu_{\alpha^{(g)}}^{(g)}(a \otimes b) = \mu^{(g)}(a \otimes b) + \langle \alpha^{(g)}, a \otimes b \rangle$$
with genus corrections.

\textbf{4. Geometric Interpretation Across Genera:}
On configuration spaces, MC elements are:
\begin{itemize}
\item Closed 1-forms on $\overline{C}_2^{(g)}(\Sigma_g)$ with prescribed residues and period integrals
\item Flat connections on the punctured configuration space with modular structure
\item Solutions to the classical Yang-Baxter equation with genus corrections
\end{itemize}

\textbf{5. Genus-Graded Moduli Space:}
$$\mathcal{M}_{\text{MC}}^{(g)}(\mathcal{A}) = \{\text{MC elements at genus } g\}/\text{gauge equivalence}$$
parametrizes deformations of the chiral algebra structure at each genus.
\end{theorem}

\begin{proof}
We establish each part by reducing to the established genus-0
Maurer-Cartan theory (Theorem~\ref{thm:mc-deformations}) and extending
via the modular operad structure.

\textbf{Part 1:}
The genus-$g$ bar complex $\barBgeom^{(g)}(\mathcal{A})$ carries a dg Lie
algebra structure (inherited from the Gerstenhaber bracket on the full
bar complex).  By Theorem~\ref{thm:genus-induction-strict}, the
differential $d^{(g)}$ satisfies $(d^{(g)})^2 = 0$ strictly.  The MC
equation $d^{(g)}\alpha^{(g)} + \frac{1}{2}[\alpha^{(g)}, \alpha^{(g)}]
= 0$ is then the standard MC equation for this dg Lie algebra.  The
modular corrections enter through the propagator on $\Sigma_g$: by
Theorem~\ref{thm:mc-periods}, MC elements arise from period integrals
$\alpha_g = \int_{\gamma \in H_1(\Sigma_g)} \omega_{\mathcal{A}}$, and the
MC equation is equivalent to the flatness condition on the connection
$\omega_{\mathcal{A}} \in \Omega^1(\Sigma_g, \mathcal{A})$.

\textbf{Part 2:}
Given an MC element $\alpha^{(g)}$, define $d_{\alpha^{(g)}}^{(g)} = d^{(g)}
+ [\alpha^{(g)}, -]$.  We compute:
\begin{align*}
(d_{\alpha^{(g)}}^{(g)})^2
&= (d^{(g)})^2 + d^{(g)}[\alpha^{(g)}, -]
  + [\alpha^{(g)}, d^{(g)}(-)]
  + [\alpha^{(g)}, [\alpha^{(g)}, -]] \\
&= 0 + [d^{(g)}\alpha^{(g)}, -]
  + \tfrac{1}{2}[[\alpha^{(g)}, \alpha^{(g)}], -] \\
&= [d^{(g)}\alpha^{(g)} + \tfrac{1}{2}[\alpha^{(g)}, \alpha^{(g)}], -]
= 0,
\end{align*}
where we used $(d^{(g)})^2 = 0$
(Theorem~\ref{thm:genus-induction-strict}), the derivation property
$d^{(g)}[\alpha, x] = [d^{(g)}\alpha, x] + [\alpha, d^{(g)}x]$, the
graded Jacobi identity $[\alpha, [\alpha, x]] =
\frac{1}{2}[[\alpha, \alpha], x]$ (since $|\alpha| = 1$), and the MC
equation.  Modular covariance follows because $d^{(g)}$ is equivariant
under the mapping class group $\Gamma_g$
(Theorem~\ref{thm:prism-higher-genus}(iii)) and MC elements transform
covariantly.

\textbf{Part 3:}
An MC element $\alpha^{(g)} \in \barBgeom^{(g),1}(\mathcal{A})$ is
a degree-1 element of the bar complex, hence a linear functional on
$\mathcal{A} \otimes \mathcal{A}$ (via the bar-cobar pairing).  The
deformed product $\mu_{\alpha^{(g)}}^{(g)} = \mu^{(g)} + \langle
\alpha^{(g)}, - \rangle$ satisfies associativity to first order precisely
because $\alpha^{(g)}$ is a Hochschild 2-cocycle; the MC equation
guarantees associativity to all orders.  This is the standard
deformation-MC correspondence (as in
Theorem~\ref{thm:mc-deformations}), applied at genus~$g$.

\textbf{Part 4:}
By Theorem~\ref{thm:mc-periods}, MC elements arise from flat connections
on $\Sigma_g$.  In the bar complex, such an element lives in
$\Gamma(\overline{C}_2(\Sigma_g), \mathcal{A}^{\boxtimes 2} \otimes
\Omega^1_{\log})$, i.e., it is a 1-form on the two-point
configuration space with logarithmic poles.  The flatness/MC equation
on $\Sigma_g$ reduces to the classical Yang-Baxter equation when
restricted to the genus-0 skeleton (via the degeneration map
$\partial\colon C_\bullet^{(g)} \to C_\bullet^{(g-1)}$ of the tower in
\S\ref{app:koszul_higher_genus}), with genus-$g$ corrections from
the period matrix of~$\Sigma_g$.

\textbf{Part 5:}
The gauge equivalence on MC elements is $\alpha \sim \alpha + d^{(g)}\xi
+ [\alpha, \xi]$ for $\xi \in \barBgeom^{(g),0}(\mathcal{A})$.  The
moduli space $\mathcal{M}_{\mathrm{MC}}^{(g)}(\mathcal{A})$ is a derived
stack (the derived zero locus of the MC equation in the dg Lie algebra
$\barBgeom^{(g)}(\mathcal{A})$).  By the deformation-MC correspondence
(Part~3), it parametrizes deformations of $\mathcal{A}^{(g)}$ up to
equivalence.  The tower structure of \S\ref{app:koszul_higher_genus}
assembles these into a pro-system
$\cdots \to \mathcal{M}_{\mathrm{MC}}^{(2)} \to
\mathcal{M}_{\mathrm{MC}}^{(1)} \to \mathcal{M}_{\mathrm{MC}}^{(0)}$.
\end{proof}

\subsection{Koszul Duality at Higher Genus: The Tower Structure}
\label{sec:koszul-tower-structure}

The genus 0 Koszul duality:
$$\Omega C_{\bullet}^{(0)}(\mathcal{A}) \simeq \mathcal{A}$$
extends to all genera by the modular operad structure.

For each $g \geq 0$, there is a duality:
$$\Omega^{(g)} C_{\bullet}^{(g)}(\mathcal{A}) \simeq \mathcal{A}^{(g)}$$
where:
\begin{itemize}
\item $\Omega^{(g)}$ is the genus $g$ cobar construction
\item $\mathcal{A}^{(g)}$ is the genus $g$ component of $\mathcal{A}$
\end{itemize}

\subsubsection{Compatibility}

The genus stratification satisfies:
$$\partial: C_{\bullet}^{(g)} \to C_{\bullet}^{(g-1)}$$
(boundary/degeneration maps) compatible with:
$$\rho: \mathcal{A}^{(g)} \to \mathcal{A}^{(g-1)}$$
(restriction maps, evaluating genus-$g$ structures on degenerate curves in the boundary of $\overline{\mathcal{M}}_{g,n}$).

This gives a \textbf{tower of Koszul dualities}:
\begin{center}
\begin{tikzcd}[column sep=small]
\cdots \arrow[r] & C_{\bullet}^{(2)}(\mathcal{A}) \arrow[r] \arrow[d, "\Omega^{(2)}"] & 
C_{\bullet}^{(1)}(\mathcal{A}) \arrow[r] \arrow[d, "\Omega^{(1)}"] & 
C_{\bullet}^{(0)}(\mathcal{A}) \arrow[d, "\Omega^{(0)}"] \\
\cdots \arrow[r] & \mathcal{A}^{(2)} \arrow[r] & 
\mathcal{A}^{(1)} \arrow[r] & 
\mathcal{A}^{(0)}
\end{tikzcd}
\end{center}

\subsubsection{The Limit}

Taking the inverse limit:
$$\mathcal{A}_{\text{complete}} = \varprojlim_g \mathcal{A}^{(g)}$$
gives the \textbf{completed chiral algebra}, encoding all genus contributions.

\subsubsection{Modular Invariance}

At each genus $g$, the duality respects the action of the mapping class group $\Gamma_g = \operatorname{MCG}(\Sigma_g)$:
$$\Omega^{(g)}(\sigma^* C_{\bullet}^{(g)}(\mathcal{A})) \simeq \sigma^* \mathcal{A}^{(g)}$$
for $\sigma \in \Gamma_g$.

This ensures that genus $g$ quantum corrections are modular-invariant.
