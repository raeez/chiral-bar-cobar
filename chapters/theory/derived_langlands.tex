\chapter{Derived structures and the geometric Langlands correspondence}
\label{ch:derived-langlands}
\index{geometric Langlands!derived structures|textbf}
\index{oper|textbf}
\index{bar complex!critical level|textbf}

At generic level~$k$, the bar complex $\barB(\widehat{\fg}_k)$ is
\emph{curved}: the curvature $m_0 = \kappa(\widehat{\fg}_k)$
measures the failure of $d^2 = 0$ on the reduced complex, and its
shadow is the central extension.  At the critical level
$k = -h^\vee$, the curvature vanishes, the bar complex becomes
an honest chain complex, and its cohomology acquires geometric
meaning: it computes functions on the space of
$\fg^\vee$-opers.  This identification---the bar complex at
critical level \emph{is} the derived oper space---is the entry
point of the geometric Langlands programme into bar-cobar duality.
Moving away from the critical level to admissible levels
$k = -h^\vee + p/q$, the bar complex acquires a periodic CDG
structure indexed by the root of unity
$q = e^{\pi i/(k + h^\vee)}$, and bar-cobar duality should
recover the Kazhdan--Lusztig equivalence between
$\widehat{\fg}_k$-modules and representations of the quantum
group $U_q(\fg)$.

This chapter develops the derived algebraic geometry needed to
make these identifications precise, states the full conjectures
with all hypotheses, and documents the proved ingredients.

\section{Derived algebraic geometry: prerequisites}
\label{sec:dag-prerequisites}
\index{derived algebraic geometry}

We recall the minimum framework of derived algebraic geometry
needed for the oper identification.  Our exposition follows
Lurie~\cite{HA} and To\"en--Vezzosi; we import definitions and
key properties without reproducing their proofs.

\begin{definition}[Derived stack]
\label{def:derived-stack}
\index{derived stack|textbf}
A \emph{derived stack} (over a field~$k$ of characteristic zero) is
a functor
$X\colon \mathrm{cdga}_k^{\mathrm{op}} \to \mathcal{S}$
from commutative dg-algebras to the $\infty$-category of spaces,
satisfying \'etale descent.  A derived stack~$X$ has:
\begin{enumerate}[label=\textup{(\roman*)}]
\item A \emph{classical truncation} $t_0(X)$, which is an ordinary
  algebraic stack obtained by restricting to ordinary (non-dg)
  commutative algebras.
\item A \emph{cotangent complex} $\mathbb{L}_X \in \mathrm{QCoh}(X)$,
  a quasi-coherent sheaf on~$X$ (in the derived sense) controlling
  deformations and obstructions.  For a smooth classical
  scheme~$Y$ viewed as a derived stack, $\mathbb{L}_Y = \Omega^1_Y[0]$
  is the ordinary sheaf of differentials.
\item A \emph{derived structure sheaf} $\cO_X^{\mathrm{der}}$,
  a sheaf of cdga's whose cohomology sheaves
  $\mathcal{H}^i(\cO_X^{\mathrm{der}})$ encode the derived
  information beyond the classical truncation.
\end{enumerate}
\end{definition}

\begin{definition}[Formal smoothness]
\label{def:formal-smoothness}
\index{formal smoothness}
A derived stack~$X$ is \emph{formally smooth} if its cotangent
complex $\mathbb{L}_X$ is concentrated in cohomological degree~$0$
(i.e., $\mathbb{L}_X \simeq \Omega^1_{t_0(X)}[0]$).  Equivalently,
$X$ has no higher obstructions: every infinitesimal deformation
extends.
\end{definition}

\begin{remark}[Formal smoothness vs.\ derived richness]
\label{rem:formal-smooth-derived}
The oper space $\mathrm{Op}_{\fg^\vee}(D)$ on a formal disk is
formally smooth (Frenkel--Gaitsgory~\cite{FG06}), so its cotangent
complex is classical.  Nevertheless, the bar complex
$\barB(\widehat{\fg}_{-h^\vee})$ naturally produces a chain complex
computing $\cO(\mathrm{Op}_{\fg^\vee}(D))$---the derived structure
is in the bar complex, not in the oper space.  The bar complex is the
\emph{derived enhancement} of the (classically smooth) oper space.
\end{remark}

\section{The oper space}
\label{sec:oper-space}
\index{oper!definition|textbf}

\begin{definition}[Opers; Beilinson--Drinfeld]
\label{def:opers}
\index{oper!Beilinson--Drinfeld}
Let $G^\vee$ be the Langlands dual group of~$G$, with
Lie algebra $\fg^\vee$, Borel subalgebra
$\mathfrak{b}^\vee \subset \fg^\vee$, and nilpotent radical
$\mathfrak{n}^\vee \subset \mathfrak{b}^\vee$.  Let $X$ be a smooth algebraic
curve.  A \emph{$G^\vee$-oper} on~$X$ is a triple
$(F_{G^\vee}, F_{B^\vee}, \nabla)$ where:
\begin{enumerate}[label=\textup{(\roman*)}]
\item $F_{G^\vee}$ is a principal $G^\vee$-bundle on~$X$;
\item $F_{B^\vee} \subset F_{G^\vee}$ is a reduction to the Borel
  subgroup~$B^\vee$;
\item $\nabla$ is a connection on $F_{G^\vee}$ satisfying the
  \emph{oper condition}: the induced map
  $TX \to F_{G^\vee}/F_{B^\vee} = F_{G^\vee} \times_{G^\vee} (G^\vee/B^\vee)$
  given by the connection lies in the open $B^\vee$-orbit.
\end{enumerate}
The moduli space of $G^\vee$-opers on~$X$ is denoted
$\mathrm{Op}_{G^\vee}(X)$.
\end{definition}

\begin{example}[$G = \mathrm{SL}_2$, opers = projective connections]
\label{ex:sl2-opers}
\index{oper!$\mathrm{SL}_2$}
For $G = \mathrm{SL}_2$, opers on a curve~$X$ are
\emph{projective connections}: second-order differential operators
$\partial_z^2 + t(z)$ on~$X$.  On a formal disk $D = \mathrm{Spec}\, k[[z]]$, the space of
$\mathrm{PGL}_2$-opers is $\mathrm{Op}_{\mathrm{PGL}_2}(D) \cong k[[z]]$,
parametrized by the ``projective connection'' $t(z) = \sum_{n \geq 0} t_n z^n$.
In general, $\mathrm{Op}_{G^\vee}(D) \cong k[[z]]^{\mathrm{rank}(\fg)}$
is a formal power series space of dimension $\mathrm{rank}(\fg)$.
\end{example}

\begin{theorem}[Feigin--Frenkel center; \ClaimStatusProvedElsewhere]
\label{thm:ff-center-dl}
\index{Feigin--Frenkel center|textbf}
\textup{(}Feigin--Frenkel~\cite{Feigin-Frenkel},
\cite{FG06}\textup{)}\quad
The center of the completed enveloping algebra of
$\widehat{\fg}$ at the critical level $k = -h^\vee$ is
canonically isomorphic to the algebra of functions on
$\fg^\vee$-opers:
\begin{equation}\label{eq:ff-center}
Z(\widehat{\fg}_{-h^\vee})
\;\cong\;
\mathrm{Fun}\bigl(\mathrm{Op}_{\fg^\vee}(D)\bigr).
\end{equation}
Moreover, the center $Z(\widehat{\fg}_{-h^\vee})$ is a polynomial
algebra on generators in conformal weights
$d_1, d_2, \ldots, d_r$ \textup{(}the exponents of $\fg$
plus~$1$\textup{)}.
\end{theorem}

\section{Bar complex at critical level = derived oper functions}
\label{sec:bar-critical}
\index{bar complex!critical level}

The critical level is the pivot point where bar-cobar duality
meets the geometric Langlands programme.

\begin{theorem}[Zeroth bar cohomology = oper functions;
\ClaimStatusProvedHere]
\label{thm:oper-bar-h0-dl}
\index{bar complex!$H^0 = \mathrm{Fun}(\mathrm{Op})$}
For $\fg$ a simple finite-dimensional Lie algebra, the zeroth
cohomology of the reduced bar complex at critical level is:
\begin{equation}\label{eq:h0-oper}
H^0\bigl(\barB(\widehat{\fg}_{-h^\vee})\bigr)
\;\cong\;
\mathrm{Fun}\bigl(\mathrm{Op}_{\fg^\vee}(D)\bigr).
\end{equation}
\end{theorem}

\begin{proof}
At the critical level, the curvature vanishes
($\kappa(\widehat{\fg}_{-h^\vee}) = 0$,
Corollary~\ref{cor:critical-level-universality}), so the bar
complex is an honest chain complex.
The PBW spectral sequence
(Theorem~\ref{thm:pbw-koszulness-criterion}) has
\[
E_1^{p,q}
= H^q\bigl(\fg;\, S^p(\fg[t^{-1}]t^{-1})\bigr)
\;\Longrightarrow\;
H^{p+q}\bigl(\barB(\widehat{\fg}_{-h^\vee})\bigr).
\]
At $p + q = 0$: the Whitehead lemma gives
$H^0(\fg; S^*(\fg[t^{-1}]t^{-1})) = S^*(\fg[t^{-1}]t^{-1})^\fg$,
the invariant subalgebra.  By the Feigin--Frenkel theorem
(Theorem~\ref{thm:ff-center-dl}), this invariant subalgebra is exactly
$\mathrm{Fun}(\mathrm{Op}_{\fg^\vee}(D))$.  The higher
differentials $d_r$ for $r \geq 1$ vanish on total degree~$0$
by degree considerations, so $E_\infty^{0,0} = E_1^{0,0}$.
\end{proof}

\begin{proposition}[First bar cohomology = oper $1$-forms;
\ClaimStatusProvedHere]
\label{prop:oper-bar-h1-dl}
\index{bar complex!$H^1 = \Omega^1(\mathrm{Op})$}
\begin{equation}\label{eq:h1-oper}
H^1\bigl(\barB(\widehat{\fg}_{-h^\vee})\bigr)
\;\cong\;
\Omega^1\bigl(\mathrm{Op}_{\fg^\vee}(D)\bigr).
\end{equation}
\end{proposition}

\begin{proof}
On the $E_1$ page of the PBW spectral sequence, the total degree-$1$
component is $E_1^{1,0} \oplus E_1^{0,1}$.  The term
$E_1^{0,1} = H^1(\fg; S^*(\fg[t^{-1}]t^{-1}))$ vanishes by the
Whitehead lemma ($H^1(\fg; M) = 0$ for any finite-dimensional
$\fg$-module~$M$ when $\fg$ is semisimple).  The surviving term
$E_1^{1,0} = \Omega^1(S^*(\fg[t^{-1}]t^{-1})^\fg)$ is the module of
K\"ahler differentials of the invariant subalgebra.  By formal
smoothness of the oper space (Frenkel--Gaitsgory~\cite{FG06}),
this is $\Omega^1(\mathrm{Op}_{\fg^\vee}(D))$.  Again, higher
differentials vanish by degree.
\end{proof}

\begin{conjecture}[Full derived identification;
\ClaimStatusConjectured]
\label{conj:oper-bar-dl}
\index{oper!derived identification}
For all $n \geq 0$:
\begin{equation}\label{eq:hn-oper}
H^n\bigl(\barB(\widehat{\fg}_{-h^\vee})\bigr)
\;\cong\;
\bigwedge\nolimits^n
\mathbb{L}_{\mathrm{Op}_{\fg^\vee}(D)}
\;=\;
\Omega^n\bigl(\mathrm{Op}_{\fg^\vee}(D)\bigr).
\end{equation}
In other words, the bar complex at critical level computes the
derived structure sheaf of the oper space:
\[
\barB(\widehat{\fg}_{-h^\vee})
\;\simeq\;
\cO^{\mathrm{der}}\bigl(\mathrm{Op}_{\fg^\vee}(D)\bigr).
\]
\end{conjecture}

\begin{remark}[Evidence and scope]
\label{rem:oper-evidence}
\index{oper!evidence for conjecture}
The conjecture is proved at $n = 0$ (Theorem~\ref{thm:oper-bar-h0-dl})
and $n = 1$ (Proposition~\ref{prop:oper-bar-h1-dl}).  Since the oper
space is formally smooth, $\mathbb{L}_{\mathrm{Op}} = \Omega^1[0]$ is
classical, and $\bigwedge^n \mathbb{L} = \Omega^n$.  The PBW spectral
sequence should give $E_2^{n,0} = \Omega^n(\mathrm{Op})$ with all
higher terms vanishing.  The gap is controlling $E_1^{p,q}$ for
$p + q = n$ with $q \geq 2$: this requires the full structure of
$H^q(\fg; S^p(\fg[t^{-1}]t^{-1}))$ as a module over the
Feigin--Frenkel center~$\mathfrak{z}$.

For $\fg = \mathfrak{sl}_2$, an explicit computation of
$H^2(\barB(\widehat{\mathfrak{sl}}_{2,-2}))$ via the PBW spectral
sequence, compared with $\Omega^2(\mathrm{Op}_{\mathrm{PGL}_2}(D))$,
would constitute a concrete test of the conjecture.

This conjecture is a chain-level refinement of the
Frenkel--Gaitsgory localization functor
$\Delta\colon D(\widehat{\fg}_{-h^\vee}\text{-mod})
\to D(\mathrm{QCoh}(\mathrm{Op}))$
applied to the vacuum module.  See
Conjecture~\ref{conj:oper-bar} for the version stated in
Chapter~\ref{chap:kac-moody}.
\end{remark}

\begin{proposition}[The bar complex as localization;
\ClaimStatusProvedHere]
\label{prop:bar-as-localization}
\index{localization functor!bar complex}
The map
\[
\barB(\widehat{\fg}_{-h^\vee})
\;\longrightarrow\;
\mathrm{Fun}(\mathrm{Op}_{\fg^\vee}(D))
\]
given by projecting to $H^0$ is a map of algebras: the bar
complex at critical level is a dg-algebra (since the curvature
vanishes), and the projection to cohomology respects
multiplication.  This map is the chain-level incarnation of the
Frenkel--Gaitsgory localization functor~$\Delta$ applied to the
vacuum module $V_{-h^\vee}(\fg)$.
\end{proposition}

\begin{proof}
The bar complex $\barB(\widehat{\fg}_{-h^\vee})$ carries a product
from the coproduct on the bar coalgebra (dual to the algebra
structure on the cobar construction).  At critical level, the
curvature vanishes and the bar complex is a genuine dg-algebra.
The PBW filtration is compatible with the product, and the
identification $H^0 = \mathrm{Fun}(\mathrm{Op})$ is an algebra
isomorphism by the Feigin--Frenkel theorem.  The claim that this
coincides with the Frenkel--Gaitsgory functor follows from
\cite{FG06}, Theorem~9.5: the localization of the vacuum module
is $\cO_{\mathrm{Op}}$, and our bar complex provides a specific
chain-level model.
\end{proof}

\section{Root-of-unity bar complexes}
\label{sec:root-of-unity}
\index{root of unity!bar complex|textbf}
\index{CDG algebra!periodic}

Moving away from the critical level to admissible levels introduces
curvature---but curvature with algebraic structure.

\subsection{Admissible levels and the quantum group parameter}

\begin{definition}[Admissible level]
\label{def:admissible-level-dl}
A level $k$ is \emph{admissible} if
$k = -h^\vee + p/q$ where $p, q \in \bZ_{> 0}$, $\gcd(p, q) = 1$,
and $p \geq h^\vee$ (for simply-laced~$\fg$; the condition is more
involved for non-simply-laced types).  The associated quantum group
parameter is $q_{\mathrm{QG}} = e^{\pi i/(k + h^\vee)} = e^{\pi i q/p}$.
\end{definition}

At admissible level, two structures interact:
\begin{enumerate}
\item The bar complex has curvature
  $m_0 = k \cdot \kappa \cdot |0\rangle$,
  where $\kappa$ is the normalized Killing form,
  which is non-zero but \emph{rational} at admissible level.
  This curvature arises directly from the level-$k$ central extension:
  the bilinear form $k\kappa(a, b)$ appears as the double-pole residue
  $a_{(1)}b = k\kappa(a,b)|0\rangle$ in the OPE, and obstructs
  $d^2 = 0$ on the unreduced bar complex
  (Remark~\ref{rem:bicomplex-obstruction}).
  The related Sugawara central charge $c = k\dim(\fg)/(k+h^\vee)$
  is a derived quantity, not the direct source of curvature.
\item The representation category $\cO_k^{\mathrm{int}}(\widehat{\fg})$
  is semisimple with $|\mathrm{Adm}_k|$ simple objects
  (Theorem~\ref{thm:admissible-rep-theory}).
\end{enumerate}

\subsection{The periodic CDG structure}

\begin{definition}[Curved dg-algebra (CDG)]
\label{def:cdg-algebra}
\index{CDG algebra|textbf}
A \emph{curved differential graded algebra} (CDG-algebra) is a
triple $(A, d, h)$ where $A$ is a graded algebra,
$d\colon A \to A[1]$ is a derivation, and $h \in A^2$ is the
\emph{curvature element}, satisfying:
\begin{enumerate}[label=\textup{(\roman*)}]
\item $d^2(a) = [h, a]$ for all $a \in A$;
\item $dh = 0$ (the curvature is closed).
\end{enumerate}
The bar complex $\barB(\widehat{\fg}_k)$ at non-critical level is
a CDG-coalgebra: the curvature $m_0 = \kappa$ satisfies $d(m_0) = 0$
and $d^2 = [m_0, -]$
(Proposition~\ref{prop:pole-decomposition}).
\end{definition}

\begin{conjecture}[Periodic CDG structure at admissible level;
\ClaimStatusConjectured]
\label{conj:periodic-cdg}
\index{CDG algebra!periodicity}
At admissible level $k = -h^\vee + p/q$, the bar complex
$\barB(\widehat{\fg}_k)$ admits a \emph{periodic CDG structure}:
there exists a natural isomorphism of CDG-coalgebras
\begin{equation}\label{eq:periodic-cdg}
\barB^{n + 2s}(\widehat{\fg}_k)
\;\cong\;
\barB^n(\widehat{\fg}_k)
\end{equation}
where $s = p$ \textup{(}the numerator of
$k + h^\vee = p/q$\textup{)}, so the bar complex is $2p$-periodic
in bar degree.
\end{conjecture}

\begin{remark}[Positselski's framework]
\label{rem:positselski}
\index{Positselski!CDG algebras}
Positselski~\cite{Positselski} develops the theory of CDG-algebras,
CDG-modules, and their derived categories.  The key features
relevant to our setting are:
\begin{enumerate}
\item CDG-algebras do \emph{not} have well-defined cohomology
  (since $d^2 \neq 0$), but they do have a well-defined
  \emph{coderived category} $D^{\mathrm{co}}(A, d, h)$ and a
  \emph{contraderived category} $D^{\mathrm{ctr}}(A, d, h)$.
\item Koszul duality for CDG-algebras exchanges the coderived and
  contraderived categories: if $(A, d_A, h_A)$ and $(B, d_B, h_B)$
  are Koszul dual CDG-algebras, then
  $D^{\mathrm{co}}(A) \simeq D^{\mathrm{ctr}}(B)$.
\item For periodic CDG-algebras, the periodicity~\eqref{eq:periodic-cdg}
  implies that the coderived category is equivalent to the
  \emph{stable category} of the finite-dimensional quotient---the
  analogue of the passage from $U_q(\fg)$ to the small quantum
  group $u_q(\fg)$ at a root of unity.
\end{enumerate}
Conjecture~\ref{conj:periodic-cdg} asserts that the bar complex at
admissible level is a periodic CDG-coalgebra in Positselski's sense.
\end{remark}


\subsection{Evidence for periodicity}

\begin{proposition}[$\widehat{\mathfrak{sl}}_2$ periodicity;
\ClaimStatusProvedHere]
\label{prop:sl2-periodicity-dl}
\index{periodicity!$\widehat{\mathfrak{sl}}_2$}
For $\fg = \mathfrak{sl}_2$, the bar cohomology
$H^n(\barB(\widehat{\mathfrak{sl}}_{2,k}))$ at generic level
satisfies the periodicity
\begin{equation}\label{eq:sl2-periodicity}
\dim H^{n+4}(\barB)
= \dim H^n(\barB)
\quad
\text{for all } n \geq 1,
\end{equation}
with period $2h = 4$ \textup{(}$h = 2$ for $\mathfrak{sl}_2$\textup{)}.
\end{proposition}

\begin{proof}
See the proof in Chapter~\ref{chap:kac-moody}
(Proposition~\ref{thm:affine-periodicity-critical}): the periodicity follows from
the rank-$1$ structure of the weight lattice and the $4$-periodicity
of the Riordan numbers (the bar cohomology dimensions for
$\mathfrak{sl}_2$; see Theorem~\ref{thm:sl2-koszul-dual}).
\end{proof}

\begin{remark}[Periodicity and quantum group]
\label{rem:periodicity-quantum}
\index{quantum group!from periodicity}
At admissible level $k = -2 + p/q$, the period $2h = 4$ divides
$2p$ when $p$ is even.  The $4$-periodicity of the bar cohomology
dimensions is a shadow of the representation-theoretic periodicity
of the quantum group $U_q(\mathfrak{sl}_2)$ at
$q = e^{\pi i q/p}$.  For the small quantum group $u_q$, the
representation category is $2p$-periodic, and Conjecture~\ref{conj:periodic-cdg} asserts
that the bar complex ``sees'' this periodicity via CDG structure.
\end{remark}

\section{The Kazhdan--Lusztig equivalence from bar-cobar}
\label{sec:kl-from-bar-cobar}
\index{Kazhdan--Lusztig equivalence!from bar-cobar|textbf}

The Kazhdan--Lusztig equivalence is a central result in
representation theory.  We formulate the conjecture that bar-cobar
duality provides a geometric proof.

\begin{theorem}[Kazhdan--Lusztig equivalence;
\ClaimStatusProvedElsewhere]
\label{thm:kl-equivalence}
\index{Kazhdan--Lusztig equivalence}
\textup{(}Kazhdan--Lusztig~\cite{KL93}\textup{)}\quad
For $\fg$ simply-laced and $k = -h^\vee + p/q$ admissible with
$p \geq h^\vee$, there is an equivalence of braided monoidal
categories:
\begin{equation}\label{eq:kl-equivalence-dl}
\cO_k^{\mathrm{int}}(\widehat{\fg})
\;\simeq\;
\mathrm{Rep}^{\mathrm{fd}}(U_q(\fg)),
\end{equation}
where $q = e^{\pi i q/p}$ and $\mathrm{Rep}^{\mathrm{fd}}$
denotes the category of finite-dimensional type~$1$ representations.
\end{theorem}

\begin{conjecture}[KL from bar-cobar; \ClaimStatusConjectured]
\label{conj:kl-bar-cobar-dl}
\index{Kazhdan--Lusztig equivalence!bar-cobar proof}
The Kazhdan--Lusztig equivalence~\eqref{eq:kl-equivalence-dl} lifts
to a chain-level equivalence mediated by bar-cobar duality:
\begin{equation}\label{eq:kl-bar-cobar-dl}
\barB_{\Eone}\bigl(\widehat{\fg}_k\text{-}\mathrm{mod}\bigr)
\;\simeq\;
\Omega_{\Eone}\bigl(U_q(\fg)\text{-}\mathrm{comod}\bigr),
\end{equation}
where $\barB_{\Eone}$ and $\Omega_{\Eone}$ are the $\Eone$-chiral
bar and cobar functors of
Chapter~\textup{\ref{chap:koszul-pairs}}.

More precisely, the five proved ingredients and one conjectural
ingredient are:
\begin{enumerate}[label=\textup{(\arabic*)}]
\item \textup{[Proved]}\quad
  Level-shifting Koszul duality:
  $\widehat{\fg}_k^! \cong \widehat{\fg}_{-k-2h^\vee}$
  \textup{(}Theorem~\textup{\ref{thm:universal-kac-moody-koszul})}.
\item \textup{[Proved]}\quad
  Module bar-cobar adjunction:
  $\Phi\colon \mathrm{Mod}(\cA) \rightleftarrows
  \mathrm{CoMod}(\cA^!)$
  is exact
  \textup{(}Theorem~\textup{\ref{thm:e1-module-koszul-duality})}.
\item \textup{[Proved]}\quad
  BGG resolution from bar complex: the bar spectral sequence
  recovers Bernstein--Gelfand--Gelfand multiplicities at
  generic level
  \textup{(}Corollary~\textup{\ref{cor:bgg-koszul-involution})}.
\item \textup{[Proved]}\quad
  Periodicity: bar cohomology is $2h$-periodic for rank-$1$
  algebras
  \textup{(}Proposition~\textup{\ref{prop:sl2-periodicity-dl})}.
\item \textup{[Proved]}\quad
  Finite-dimensionality: bar cohomology is finite-dimensional
  at admissible levels
  \textup{(}Corollary~\textup{\ref{cor:bar-admissible-finiteness})}.
\item \textup{[Conjectured]}\quad
  Periodic CDG structure: the bar complex at admissible level is
  $2p$-periodic as a CDG-coalgebra
  \textup{(}Conjecture~\textup{\ref{conj:periodic-cdg})}.
\end{enumerate}
Given ingredient~(6), the equivalence~\eqref{eq:kl-bar-cobar-dl} would
follow from Positselski's Koszul duality for periodic CDG-algebras
\cite{Positselski}: the coderived category of the periodic bar
complex is equivalent to the stable module category of the small
quantum group, which is equivalent to
$\mathrm{Rep}^{\mathrm{fd}}(U_q(\fg))$ by Lusztig's
theorem.
\end{conjecture}

\begin{remark}[Why this would be a geometric proof]
\label{rem:geometric-proof-kl}
\index{Kazhdan--Lusztig equivalence!geometric proof}
The original Kazhdan--Lusztig proof~\cite{KL93} uses
analytic continuation of vertex operator intertwining operators.
A bar-cobar proof would be geometric in the following sense:
\begin{enumerate}
\item The bar complex is built from configuration space integrals on
  algebraic curves (our Main Theorem~A).
\item The Koszul duality $\widehat{\fg}_k^! = \widehat{\fg}_{-k-2h^\vee}$
  is realized by Verdier duality on FM compactifications.
\item The periodic CDG structure arises from the interaction of
  bar curvature with the root-of-unity structure of
  $q = e^{\pi i/(k+h^\vee)}$.
\item The passage from the coderived category to finite-dimensional
  representations is Positselski's abstract Koszul duality, applied
  to the specific CDG-coalgebra produced by our bar complex.
\end{enumerate}
Each step has a concrete geometric content; the proof would avoid
the analytic continuation that makes the original KL proof
difficult to generalize.
\end{remark}


\subsection{Roadmap: from critical level to KL}

\begin{remark}[The critical-admissible deformation]
\label{rem:critical-admissible}
\index{critical level!deformation to admissible}
The critical level $k = -h^\vee$ and the admissible levels
$k = -h^\vee + p/q$ are related by a deformation: as
$p/q \to 0$, the admissible level approaches the critical level.
In bar-cobar terms:
\begin{enumerate}
\item At $k = -h^\vee$: the bar complex is uncurved, computes oper
  functions (Theorem~\ref{thm:oper-bar-h0-dl}).
\item At $k = -h^\vee + \varepsilon$: curvature $m_0 = O(\varepsilon)$
  appears, the bar complex becomes a CDG-coalgebra, and oper functions
  are deformed.
\item At $k = -h^\vee + p/q$ (rational): the CDG structure becomes
  periodic, and the bar cohomology ``quantizes'' oper functions to
  quantum group representations.
\end{enumerate}
This deformation family connects the geometric Langlands
programme (critical level, opers) to the representation-theoretic
Langlands programme (admissible level, quantum groups) through
bar-cobar duality.  The bar complex provides a \emph{single object}
that interpolates between these two regimes.
\end{remark}

\section{Connections to the geometric Langlands programme}
\label{sec:langlands-connections}
\index{geometric Langlands!connections}

\subsection{The Frenkel--Gaitsgory programme}

\begin{theorem}[Frenkel--Gaitsgory localization;
\ClaimStatusProvedElsewhere]
\label{thm:fg-localization}
\index{Frenkel--Gaitsgory!localization}
\textup{(}Frenkel--Gaitsgory~\cite{FG06}\textup{)}\quad
There is a localization functor
\[
\Delta\colon
D\bigl(\widehat{\fg}_{-h^\vee}\text{-mod}\bigr)
\;\longrightarrow\;
D\bigl(\mathrm{QCoh}(\mathrm{Op}_{\fg^\vee}(D))\bigr)
\]
that is an equivalence on the subcategory of modules on which the
center $Z(\widehat{\fg}_{-h^\vee})$ acts locally finitely.
\end{theorem}

\begin{remark}[Bar-cobar as localization]
\label{rem:bar-as-localization}
Our bar complex provides a chain-level model for the localization
functor~$\Delta$.  Specifically:
\begin{enumerate}
\item The vacuum module $V_{-h^\vee}(\fg)$ maps under $\Delta$ to
  $\cO_{\mathrm{Op}}$ (the structure sheaf of opers).  Our
  Theorem~\ref{thm:oper-bar-h0-dl} computes
  $\Delta(V_{-h^\vee}) = \cO_{\mathrm{Op}}$ at the chain level.
\item For a Verma module $M_\lambda$ at critical level, $\Delta(M_\lambda)$
  is a skyscraper sheaf at the oper corresponding to~$\lambda$.  The
  bar complex with module insertion
  $\barB(\widehat{\fg}_{-h^\vee}, M_\lambda)$ should compute this
  skyscraper at the chain level.
\item The full derived category equivalence would follow from
  Conjecture~\ref{conj:oper-bar-dl}: the bar complex computes the
  entire derived structure of $\mathrm{Op}$, not just $H^0$.
\end{enumerate}
\end{remark}

\subsection{From opers to Hecke eigensheaves}

\begin{remark}[The Langlands path]
\label{rem:langlands-path}
\index{geometric Langlands!path from bar complex}
The geometric Langlands conjecture asserts an equivalence
\[
D\bigl(\widehat{\fg}_{-h^\vee}\text{-mod}\bigr)
\;\simeq\;
D\bigl(\mathrm{QCoh}(\mathrm{Loc}_{G^\vee}(X))\bigr),
\]
where $\mathrm{Loc}_{G^\vee}(X)$ is the moduli stack of
$G^\vee$-local systems on~$X$.  The Frenkel--Gaitsgory programme
factors this through opers:
\[
\begin{tikzcd}
D(\widehat{\fg}_{-h^\vee}\text{-mod})
  \arrow[r, "\Delta"]
  \arrow[dr, "\text{Langlands}"']
& D(\mathrm{QCoh}(\mathrm{Op}))
  \arrow[d, "\text{forget reduction}"] \\
& D(\mathrm{QCoh}(\mathrm{Loc}_{G^\vee}))
\end{tikzcd}
\]
Our bar complex computes the top arrow (localization to opers).
The vertical arrow (forgetting the Borel reduction) is the oper-to-local-system map, which is a $G^\vee/B^\vee$-fibration.  Thus:
\emph{the bar complex at critical level computes the intermediate
step of the geometric Langlands correspondence.}
\end{remark}

\section{Summary}
\label{sec:derived-langlands-summary}

\noindent
\textbf{What is proved.}
\begin{enumerate}
\item $H^0(\barB(\widehat{\fg}_{-h^\vee})) \cong
  \mathrm{Fun}(\mathrm{Op}_{\fg^\vee}(D))$
  (Theorem~\ref{thm:oper-bar-h0-dl}): the zeroth bar cohomology at
  critical level computes oper functions.
\item $H^1(\barB(\widehat{\fg}_{-h^\vee})) \cong
  \Omega^1(\mathrm{Op}_{\fg^\vee}(D))$
  (Proposition~\ref{prop:oper-bar-h1-dl}): the first bar cohomology
  computes oper $1$-forms.
\item The bar complex at critical level is a chain-level model for
  the Frenkel--Gaitsgory localization functor
  (Proposition~\ref{prop:bar-as-localization}).
\item Five of six ingredients for the KL-from-bar-cobar conjecture
  are proved (Conjecture~\ref{conj:kl-bar-cobar-dl}).
\item Bar cohomology periodicity at rank~$1$
  (Proposition~\ref{prop:sl2-periodicity-dl}).
\end{enumerate}

\noindent
\textbf{What remains.}
\begin{enumerate}
\item The full derived identification
  $\barB(\widehat{\fg}_{-h^\vee}) \simeq
  \cO^{\mathrm{der}}(\mathrm{Op})$
  beyond $H^0$ and $H^1$
  (Conjecture~\ref{conj:oper-bar-dl}).
\item The periodic CDG structure at admissible levels
  (Conjecture~\ref{conj:periodic-cdg}): this is the single
  remaining ingredient for a bar-cobar proof of KL.
\item The full KL equivalence from bar-cobar
  (Conjecture~\ref{conj:kl-bar-cobar-dl}): depends on the periodic
  CDG structure.
\item Higher-rank periodicity: extending
  Proposition~\ref{prop:sl2-periodicity-dl} beyond rank~$1$
  to establish periodicity for general $\fg$.
\end{enumerate}

The critical-level bar complex sits at the intersection of three
programmes: geometric Langlands (Programme~I), the KL equivalence
(Programme~II), and fusion product preservation (Programme~III).
Bar-cobar duality provides a single algebraic object---the bar
complex---that unifies these apparently disparate research
directions.
