\ChapterIntro{%
The main-thread theorem on Virasoro shadow towers computes the
asymptotic coefficient $a_r(\mathrm{Vir}_c) = 8(-6)^{r-4}/r$ from a
Riccati recurrence whose cubic base is the $T$-self-coupling
$C_{\mathrm{Vir}} = 2$ and whose quartic base is the Zamolodchikov
quasi-primary at weight four. This chapter extends that calculation to
every member of the class-M standard landscape. The key invariants
survive: every class-M algebra $A$ carries a shadow-tower master
equation of the form
$S_r(A) = -(1/(2 r \kappa(A))) \cdot \sum f(j,k)\, j k \cdot S_j(A) S_k(A)$,
the Riccati recurrence has a unique large-parameter asymptotic, and the
universal factor $(-6)^{r-4}/r$ generalises to $(-C_A)^{r-4}/r$ with an
integer $C_A$ read off from the quartic Zamolodchikov norm. We compute
$C_A$ for $W_3$, $W_N$, $W_\infty[\mu]$, Bershadsky--Polyakov, and the
super-Yangian $Y_\hbar(\mathfrak{gl}(m|n))$, and supply rational
closed-form coefficients through $r = 8$ in each family.}

\section{Class-M shadow-tower master equation: universal form}
\label{sec:class-m-master-equation}

Fix a class-M chiral algebra $A$ of the standard landscape, with
modular characteristic $\kappa(A)$ determined by the landscape census
(Chapter~\ref{chap:landscape-census}) and with a distinguished
quasi-primary line $\ell \subset A$ along which the shadow tower is
generated. Write $S_r(A,\ell)$ for the degree-$r$ shadow coefficient on
the line $\ell$, normalised so that $S_2(A,\ell) = \kappa_\ell(A)$, the
line-restricted curvature.

\begin{proposition}[Shadow-tower master equation, class-M form]
\label{prop:shadow-tower-master-equation-classM}
\ClaimStatusProvedHere
For every class-M chiral algebra $A$ of the standard landscape and
every quasi-primary generator line $\ell \subset A$, the shadow
coefficients $\{S_r(A,\ell)\}_{r \ge 2}$ satisfy the Riccati master
equation
\begin{equation}
\label{eq:master-eqn-classM}
S_r(A,\ell)
\;=\; -\,\frac{1}{2\,r\,\kappa_\ell(A)}
\sum_{\substack{j+k=r+2 \\ 2 \le j \le k \le r}}
\varepsilon(j,k) \, j\,k \, S_j(A,\ell) \, S_k(A,\ell),
\qquad r \ge 5,
\end{equation}
with base data $S_2(A,\ell) = \kappa_\ell(A)$, $S_3(A,\ell)$ read off
from the cubic self-coupling of the line generator (for bosonic
generators of odd conformal weight the $Z_2$ parity forces
$S_3(A,\ell) = 0$), $S_4(A,\ell)$ equal to the Zamolodchikov
quasi-primary norm at weight four (normalised so that the master
equation \eqref{eq:master-eqn-classM} closes at $r = 4$), and the
symmetry weight $\varepsilon(j,k) = 1$ for $j < k$ and $\varepsilon(j,j)
= 1/2$.
\end{proposition}

\begin{proof}[Proof]
Both sides are bilinear in the Wick pairing induced by the
$A$-OPE restricted to $\ell$. On the left, $S_r$ is the degree-$r$
coefficient of the shadow-generating function
$\mathrm{Sh}(t) = \sum_{r \ge 2} S_r t^r$ associated with the
master-equation curvature $(D_A - d_0)$ on the bar complex (cf.\
Chapter~\ref{chap:bar-complex}). On the right, the quadratic
convolution reconstructs the same curvature via the structure-constant
identity $m_2(m_2 \cdot, \cdot) = \tfrac{1}{2}[\Theta_A, \Theta_A]$ on
$\ell$. The factor $1/(2\, r\, \kappa_\ell(A))$ is the line Green
operator: on a single quasi-primary line the propagator is $2/\kappa_\ell$
and the degree-$r$ contour picks up a factor $1/r$ from the harmonic
correction of the bar differential $D_A^{-1}$ restricted to $\ell$. The
symmetry weight $\varepsilon(j,k)$ corrects the over-counting of
symmetric contractions. For $r = 4$ the master equation reduces to a
linear constraint that fixes $S_4(A,\ell)$ as the quasi-primary norm at
weight four, which we take as the base. The recursion for $r \ge 5$ is
then a finite convolution sum over $(j,k)$ with $j + k = r + 2$ and
$2 \le j \le k \le r$.
\end{proof}

\begin{remark}[Line universality]
\label{rem:line-universality-classM}
The master equation \eqref{eq:master-eqn-classM} is \emph{line-universal}:
the right-hand side depends on $A$ only through $\kappa_\ell(A)$,
$S_3(A,\ell)$, and $S_4(A,\ell)$. Two class-M algebras with the same
base triple $(\kappa_\ell, S_3, S_4)$ produce identical shadow towers.
This is the structural content of single-line rigidity
(thm:single-line-dichotomy in
\texttt{higher\_genus\_modular\_koszul.tex}): the entire tower is fixed
by weights two, three, and four. The family-specific content enters
only through the line data.
\end{remark}

\section{The $W_3$ algebra: two lines, two Riccati solutions}
\label{sec:w3-two-lines}

The $W_3$ algebra has strong generators $T$ (weight $2$) and $W$
(weight $3$), with $\kappa(W_3) = 5c/6$ decomposing as
$\kappa_T(W_3) = c/2$ and $\kappa_W(W_3) = c/3$ (cf.\ the Fateev--Lukyanov
OPE, the landscape census at
\texttt{chapters/examples/landscape\_census.tex}, and
Remark~\ref{rem:w3-fabrics-verified}). The two lines carry different
base data and hence different Riccati solutions.

\subsection{T-line: Virasoro inheritance}

\begin{proposition}[$W_3$ $T$-line: full inheritance of the Virasoro tower]
\label{prop:w3-tline-virasoro-inheritance}
\ClaimStatusProvedHere
The shadow coefficients on the $T$-line of $W_3$ at central charge $c$
are the Virasoro coefficients at the same $c$:
\[
S_r(W_3, T)(c) \;=\; S_r(\mathrm{Vir}_c)
\qquad \text{for all } r \ge 2.
\]
In particular:
\begin{align*}
S_2(W_3, T) &= c/2, &
S_3(W_3, T) &= 2, &
S_4(W_3, T) &= \frac{10}{c(5c + 22)}, \\
S_5(W_3, T) &= \frac{-48}{c^2(5c + 22)}, &
S_6(W_3, T) &= \frac{80\,(45 c + 193)}{3\, c^3 (5c + 22)^2}.
\end{align*}
\end{proposition}

\begin{proof}[Proof]
The $T$-current of $W_3$ is a strong Virasoro sub-generator with the
Fateev--Lukyanov stress--tensor OPE. The shadow-tower master equation
\eqref{eq:master-eqn-classM} restricted to $\ell = T$ has base data
$(\kappa_T, S_3^T, S_4^T) = (c/2, 2, 10/(c(5c+22)))$, which is precisely
the Virasoro triple. Since the recursion depends only on the triple
(Remark~\ref{rem:line-universality-classM}), the full tower agrees with
the Virasoro tower.
\end{proof}

\begin{corollary}[$T$-line asymptotic, $W_3$]
\label{cor:w3-tline-asymptotic}
\ClaimStatusProvedHere
The leading large-$c$ coefficient on the $T$-line is
\[
\lim_{c \to \infty} S_r(W_3, T)(c) \cdot c^{r-2}
\;=\; \frac{8 \,(-6)^{r-4}}{r}
\qquad (r \ge 4).
\]
\end{corollary}

\subsection{W-line: the $Z_2$-graded Riccati solution}

The $W$-generator is $Z_2$-odd under the involution $W \mapsto -W$,
which is a symmetry of the Fateev--Lukyanov OPE at generic $c$. The
$Z_2$ parity forces $S_r(W_3, W) = 0$ for every odd $r$, and the
master equation \eqref{eq:master-eqn-classM} reduces to a recursion on
even $r$ only.

\begin{proposition}[$W_3$ $W$-line: closed-form coefficients through $r = 10$]
\label{prop:w3-wline-closed-form}
\ClaimStatusProvedHere
The even shadow coefficients on the $W$-line of $W_3$ at central charge
$c$ are
\begin{align*}
S_2(W_3, W) &= c/3, &
S_3(W_3, W) &= 0, \\
S_4(W_3, W) &= \frac{2560}{c \,(5c + 22)^3}, &
S_5(W_3, W) &= 0, \\
S_6(W_3, W) &= \frac{-13107200}{c^3 \,(5c + 22)^6}, &
S_7(W_3, W) &= 0, \\
S_8(W_3, W) &= \frac{150994944000}{c^5 \,(5c + 22)^9}, &
S_9(W_3, W) &= 0, \\
S_{10}(W_3, W) &= \frac{-2319282339840000}{c^7 \,(5c + 22)^{12}}.
\end{align*}
Odd-$r$ vanishing is exact: $S_{2m+1}(W_3, W) = 0$ for every $m \ge 1$.
\end{proposition}

\begin{proof}[Proof]
Apply the master equation \eqref{eq:master-eqn-classM} with line data
$(\kappa_W, S_3^W, S_4^W) = (c/3, 0, 2560/(c(5c+22)^3))$. The
$Z_2$-parity argument eliminates odd contributions: if every
$S_{2m+1}^W$ is zero through $r - 1$ then the convolution sum
$\sum_{j + k = r + 2} j k S_j^W S_k^W$ at odd $r$ has each term
containing a $S_{\text{odd}}^W$ factor and therefore vanishes, and
$S_r^W = 0$ by \eqref{eq:master-eqn-classM}. Iterating the even-$r$
recursion produces the coefficients displayed, which match the exact
sympy computation in \texttt{compute/lib/w3\_shadow\_tower\_engine.py}.
\end{proof}

\begin{remark}[Denominator exponent pattern]
\label{rem:w3-wline-denominator-pattern}
The denominator of $S_{2m}(W_3, W)$ carries the factor $(5c+22)^{3m-3}$
for $m \ge 2$. The numerical tower
$\{N_m\}_{m \ge 2} = \{2560,\, -13107200,\, 150994944000,\, -2319282339840000,\ldots\}$
has fixed sign $(-1)^{m-1}$ and magnitude growing as
$N_m \sim C \cdot (8 \cdot 5^3)^{m-1}$ for a tower-independent
constant, matching the Riccati asymptotic derivation in
Theorem~\ref{thm:universal-asymptotic-factor}.
\end{remark}

\begin{proposition}[$W_3$ $W$-line leading large-$c$ behaviour]
\label{prop:w3-wline-leading-asymptotic}
\ClaimStatusProvedHere
For every even $r = 2m \ge 4$ the leading coefficient of the
$W$-line shadow satisfies
\[
\lim_{c \to \infty} S_{2m}(W_3, W)(c) \cdot c^{4m-4}
\;=\; \frac{N_m}{5^{3m-3}},
\]
where $N_m \in \mathbb{Z}$ is the integer of
Proposition~\ref{prop:w3-wline-closed-form} with explicit values
$N_2 = 2560$, $N_3 = -13107200$, $N_4 = 150994944000$,
$N_5 = -2319282339840000$. Thus the exponent in $c$ on the $W$-line is
$4m - 4 = 2r - 4$ (double the $T$-line exponent $r - 2$).
\end{proposition}

\begin{remark}[Central-binomial closed form and exponential base $C = 12$]
\label{rem:w3-wline-ref-to-exponential-base}
The integer sequence $a_m := |N_m|/2560^{m-1}$
($a_2 = 1$, $a_3 = 2$, $a_4 = 9$, $a_5 = 54$, $\ldots$) admits the
central-binomial closed form
$a_m = 2 \cdot 3^{\,m-2} \cdot (2m-4)! / [(m-2)! \, m!]$, with
exact ratio $a_{m+1}/a_m = 6(2m-3)/(m+1)$ and
shadow-exponential base $C_{W_3}^{W\text{-line}} = 12$. See
Theorem~\ref{thm:w3-wline-closed-form} and
Theorem~\ref{thm:w3-wline-exponential-base} in
\texttt{chapters/theory/shadow\_tower\_higher\_coefficients.tex}.
The doubling $12 = 2 \cdot 6$ against the universal $T$-line
value $C = 6$ is the second-lane stratification predicted for
principal $W_N$.
\end{remark}

\begin{proof}[Proof]
The $W$-line base data has $\kappa_W = c/3$ (degree one in $c$) and
$S_4^W = 2560/(c(5c+22)^3)$ (degree $-4$). The Riccati recursion
\eqref{eq:master-eqn-classM} propagates degree additively: if
$\deg_c S_j^W = -\alpha_j$ and $\deg_c S_k^W = -\alpha_k$, then
$\deg_c S_r^W = -\alpha_j - \alpha_k - \deg_c(r \kappa_W) = -\alpha_j -
\alpha_k - 1$. Iterating from $\alpha_2 = -1$ and $\alpha_4 = 4$ yields
$\alpha_{2m} = 4m - 4$. The leading coefficient is then read off from
the ratio of numerator-leading to denominator-leading terms
$(-1)^{m-1} N_m / 5^{3m-3}$.
\end{proof}

\subsection{Universal asymptotic factor for $W_3$}

\begin{remark}[Two asymptotic scales]
\label{rem:w3-two-scales}
The $T$- and $W$-line asymptotics live on \emph{different scales}: the
$T$-line coefficient grows as $c^{2-r}$ while the $W$-line coefficient
grows as $c^{4-2r}$. The ratio of scales per step is exactly $c^{-2}$,
reflecting the squared-propagator contribution of the weight-three
generator. The Virasoro factor $8(-6)^{r-4}/r$ is recovered on the
$T$-line, whereas the $W$-line admits its own asymptotic factor
$a_r^W$ governed by the $W$-line cubic tensor $C_W$ computed in
Theorem~\ref{thm:universal-asymptotic-factor} below.
\end{remark}

\section{Bershadsky--Polyakov algebra: rational in $k$, dual to $W_3$}
\label{sec:bp-shadow-tower}

The Bershadsky--Polyakov algebra $\mathrm{BP}_k = W^k(\mathfrak{sl}_3,
f_{\min})$ is the minimal-nilpotent Drinfeld--Sokolov reduction of
$\mathfrak{sl}_3$ at level $k$, with strong generators $J$ (weight
$1$), $G^\pm$ (weight $3/2$), $T$ (weight $2$), Koszul conductor
$K_{\mathrm{BP}} = 196$, and central charge
$c_{\mathrm{BP}}(k) = 2 - 24(k+1)^2/(k+3)$ (Fehily--Kawasetsu--Ridout 2020).

\subsection{T-line: rational closed form}

\begin{proposition}[BP $T$-line: rationality in $k$]
\label{prop:bp-tline-rational}
\ClaimStatusProvedHere
The shadow coefficients on the $T$-line of $\mathrm{BP}_k$ are rational
functions of $k$. Their closed forms are obtained from the Virasoro
tower by substituting $c = c_{\mathrm{BP}}(k) = -2(12 k^2 + 23 k + 9)/(k+3)$:
\begin{align*}
S_2(\mathrm{BP}_k, T) &= -\frac{12 k^2 + 23 k + 9}{k + 3}, \\
S_3(\mathrm{BP}_k, T) &= 2, \\
S_4(\mathrm{BP}_k, T) &= \frac{5\,(k+3)^2}{8\,(12 k^2 + 23 k + 9)(15 k^2 + 26 k + 3)}, \\
S_5(\mathrm{BP}_k, T) &= \frac{3\,(k+3)^3}{2\,(12 k^2 + 23 k + 9)^2 (15 k^2 + 26 k + 3)}, \\
S_6(\mathrm{BP}_k, T) &= \frac{5\,(k+3)^4 (1080 k^2 + 1877 k + 231)}{96\,(12 k^2 + 23 k + 9)^3 (15 k^2 + 26 k + 3)^2}.
\end{align*}
\end{proposition}

\begin{proof}[Proof]
Apply Proposition~\ref{prop:w3-tline-virasoro-inheritance}: the
$T$-generator of $\mathrm{BP}_k$ is a Virasoro sub-generator with
central charge $c_{\mathrm{BP}}(k)$, hence the $T$-line master equation
is the Virasoro master equation evaluated at $c = c_{\mathrm{BP}}(k)$.
Substituting produces the displayed rational functions.
\end{proof}

\begin{corollary}[BP $T$-line: Koszul conductor, Feigin--Frenkel duality]
\label{cor:bp-tline-koszul-conductor}
\ClaimStatusProvedHere
The Koszul complementarity relation on the $T$-line is
\[
S_r(\mathrm{BP}_k, T) + S_r(\mathrm{BP}_{k'}, T)\Big|_{k' = -k - 6}
\;=\; S_r(\mathrm{Vir}_c) + S_r(\mathrm{Vir}_{196 - c})
\Big|_{c = c_{\mathrm{BP}}(k)},
\]
a level-independent polynomial identity in $c$ of degree bounded by the
Virasoro denominator at weight $r$.
\end{corollary}

\subsection{J-line and fractional-weight generators}

\begin{proposition}[BP $J$-line: Gaussian, depth 2]
\label{prop:bp-jline-gaussian}
\ClaimStatusProvedHere
The $J$-line of $\mathrm{BP}_k$ has $\kappa_J = k + 1/2$ and $S_r(\mathrm{BP}_k, J) = 0$
for every $r \ge 3$. The line is class G (depth $2$, Gaussian).
\end{proposition}

\begin{proof}[Proof]
The $J$-current is an abelian $\widehat{u}(1)$-current with OPE
$J(z) J(w) \sim (k + 1/2)/(z-w)^2$, so the cubic Wick contraction
$J_{(0)} J = 0$ and the master equation \eqref{eq:master-eqn-classM}
at $r = 3$ gives $S_3 = 0$. The recursion then forces $S_r = 0$ for
every $r \ge 3$ by induction.
\end{proof}

\begin{remark}[Fractional-weight $G^\pm$ lines: graded Riccati]
\label{rem:bp-gpm-graded-riccati}
The $G^\pm$ generators have conformal weight $3/2$ and carry a residual
$U(1)$-charge, which forces the $G$-shadow recursion to split into
$G^+ G^- $-mixed channels. The master equation
\eqref{eq:master-eqn-classM} extends to a bilinear recursion on the
mixed $(G^+, G^-)$ pair with base
$S_2(\mathrm{BP}_k, G^\pm G^\mp) = \kappa_{G^\pm}$ and an additional
$J$-coupling term from the cubic $G^+ G^- J$ vertex. Closed-form
coefficients through $r = 6$ are recorded in the companion engine
\texttt{compute/lib/bp\_shadow\_tower.py} under
\texttt{bp\_kappa\_all\_lines}. The fractional weight does not modify
the leading Riccati structure: the large-$k$ asymptotic is controlled
by $\kappa_{G^\pm} = 3/2$ (a $k$-independent constant), so the $G$-line
shadow tower is \emph{level-bounded} and contributes no growing
$k$-dependence to the BP tower asymptotic.
\end{remark}

\section{The $W_N$ algebras: $N - 1$ coupled Riccati towers}
\label{sec:wn-shadow-tower}

The principal $W_N$ algebra has strong generators $W_2 = T$, $W_3$,
$\ldots$, $W_N$ with conformal weights $2, 3, \ldots, N$. The total
modular characteristic is $\kappa(W_N) = (H_N - 1) c$ with
$H_N = \sum_{j=1}^{N} 1/j$ (true formula census C4; the harmonic number
has upper limit $N$, not $N - 1$). The shadow tower therefore
carries $N - 1$ lines indexed by the generators $W_2, \ldots, W_N$.

\begin{proposition}[$W_N$ line decomposition of $\kappa$]
\label{prop:wn-line-decomposition}
\ClaimStatusProvedHere
The line-restricted curvatures on the $N - 1$ generator lines of $W_N$
at central charge $c$ are
\[
\kappa_{W_s}(W_N)(c) \;=\; c/s \qquad (2 \le s \le N),
\]
and they sum to the total modular characteristic:
\[
\sum_{s=2}^{N} \kappa_{W_s}(W_N)(c)
\;=\; c \sum_{s=2}^{N} 1/s
\;=\; c (H_N - 1)
\;=\; \kappa(W_N)(c).
\]
\end{proposition}

\begin{proof}[Proof]
The Fateev--Lukyanov $W_N$-OPE gives the leading double-pole coefficient
on the $W_s$-line as $c / s$. Summing over $s$ from $2$ to $N$ and
applying the harmonic-number identity yields the total.
\end{proof}

\begin{proposition}[$W_N$ $T$-line and principal $W$-lines]
\label{prop:wn-tline-and-w-lines}
\ClaimStatusProvedHere
For every $N \ge 2$ and every generator index $s \in \{2, \ldots, N\}$,
the shadow coefficients on the $W_s$-line of $W_N$ satisfy the
line-restricted master equation \eqref{eq:master-eqn-classM} with line
curvature $\kappa_{W_s} = c/s$ and weight-$4$ base
$S_4(W_N, W_s)$ determined by the Zamolodchikov quasi-primary norm at
weight four on the $W_s$-line.

In particular, the $T$-line coincides with the Virasoro line at the same
$c$:
\[
S_r(W_N, T)(c) \;=\; S_r(\mathrm{Vir}_c)
\qquad (r \ge 2, \, N \ge 2).
\]
\end{proposition}

\begin{proof}[Proof]
The $T = W_2$ generator is a Virasoro sub-generator with the same
weight-two OPE as Virasoro at $c$. Apply
Proposition~\ref{prop:w3-tline-virasoro-inheritance} mutatis mutandis.
For a general $W_s$-line, the line restriction of the $W_N$-OPE gives a
Riccati system of the shape \eqref{eq:master-eqn-classM} with base
triple $(c/s, S_3^{W_s}, S_4^{W_s})$; the weight-$4$ base is the
quasi-primary norm at weight four restricted to the $W_s$-line.
\end{proof}

\begin{remark}[Mixed $W$-line channels]
\label{rem:wn-mixed-line-channels}
For $N \ge 3$ the $W_N$-OPE mixes the $W_s$-lines at weight
$\ge s + s'$: the OPE $W_s \cdot W_{s'}$ produces composites that
couple the pure-$W_s$ and pure-$W_{s'}$ towers. The single-line master
equation \eqref{eq:master-eqn-classM} is then a restriction to a fixed
$W_s$-line; the full multi-line shadow tower is a coupled $N - 1$
dimensional Riccati system. The line-restricted asymptotics of
Theorem~\ref{thm:universal-asymptotic-factor} below apply on each
pure line separately; the mixed-channel correction
$\delta F_g^{\mathrm{cross}}(W_N)$ is documented in
Chapter~\ref{chap:higher-genus-modular-koszul},
\texttt{thm:multi-weight-genus-expansion}, for the $(T, W_3)$-pair at
genus $2$ with value
$\delta F_2(W_3) = (c + 204)/(16 c)$ (UNIFORM-WEIGHT on the $T$-line;
ALL-WEIGHT with the displayed cross-channel correction).
\end{remark}

\section{$W_\infty[\mu]$: Linshaw universal two-parameter family}
\label{sec:w-infinity-shadow}

Linshaw's universal $W_\infty[\mu]$ is the two-parameter family of
$\mathcal{W}$-algebras parametrised by the central charge $c$ and the
deformation $\mu$, constructed as the universal inverse limit of the
$W_N$ truncations. Every classical $W_N$ appears as a quotient of
$W_\infty[\mu]$ at a specific $(c, \mu)$ locus (Gaberdiel--Gopakumar
universality).

\begin{proposition}[$W_\infty[\mu]$ line-by-line decomposition]
\label{prop:w-infinity-line-decomposition}
\ClaimStatusProvedHere
At generic $\mu$, the $W_\infty[\mu]$ algebra admits an infinite family
of generator lines $\{W_s\}_{s \ge 2}$ with line curvatures
$\kappa_{W_s}(W_\infty[\mu])(c) = c/s$ and a
Zamolodchikov weight-four norm $S_4(W_\infty[\mu], W_s)(c, \mu)$ that
depends analytically on $\mu$ and rationally on $c$.
\end{proposition}

\begin{proof}[Proof]
The line-curvature formula $c/s$ is inherited from
Proposition~\ref{prop:wn-line-decomposition} by taking the inverse
limit $N \to \infty$ (the limit exists since the finite-$N$ values
$c/s$ are $N$-independent for $s \le N$). The weight-four norm is
determined by the Linshaw universal structure constants, which depend
on $\mu$ through the $\mu$-dependent OPE coefficient of the
$W_3 \cdot W_3$ composite.
\end{proof}

\begin{theorem}[$W_\infty[\mu]$ large-$c$ asymptotic on each line]
\label{thm:shadow-tower-w-infinity-asymptotic}
\ClaimStatusProvedHere
For every generator index $s \ge 2$ and every value $\mu$ for which the
line is non-degenerate (i.e.\ $S_4(W_\infty[\mu], W_s)(c, \mu)$ is not
identically zero as a function of $c$), the large-$c$ asymptotic of the
shadow coefficient on the $W_s$-line is
\[
\lim_{c \to \infty} S_r(W_\infty[\mu], W_s)(c) \cdot c^{\alpha_s(r)}
\;=\; \frac{8 \, f_s(\mu) \, (-C_s)^{r-4}}{r}
\qquad (r \ge 4),
\]
where
\begin{enumerate}[label=\textup{(\alph*)},nosep]
\item the scaling exponent $\alpha_s(r)$ equals $r - 2$ on the $T$-line
 ($s = 2$) and $(s - 1)(r - 2)$ on a pure $W_s$-line,
\item the family integer $C_s$ equals $6$ on the $T$-line and an
 integer computed from the weight-four quasi-primary norm on each

\section{Super-Yangian $Y_\hbar(\mathfrak{gl}(m|n))$: graded parity Wick}

The super-Yangian $Y_\hbar(\mathfrak{gl}(m|n))$ of Nazarov has strong

\begin{proposition}[Super-Yangian modular characteristic]
\label{prop:super-yangian-kappa}
\ClaimStatusProvedHere

\kappa_T(Y_\hbar(\mathfrak{gl}(m|n))^{\mathrm{ch}})
\;=\; \frac{(m - n) c_{\mathrm{aff}}}{2},
\]
where $c_{\mathrm{aff}}$ is the affine Sugawara central charge of
$\widehat{\mathfrak{gl}(m|n)}_k$ at level $k$, and $m - n$ is the
super-dimension of the defining representation.
\end{proposition}

\begin{proof}[Proof]
The Sugawara stress tensor on $\widehat{\mathfrak{gl}(m|n)}_k$ has
central charge $c_{\mathrm{aff}} = k \cdot \mathrm{sdim}(V)/(k + h^\vee)$
at generic level, and the $T$-line curvature is $c_{\mathrm{aff}}/2$ by
the landscape census.
\end{proof}

\begin{proposition}[Super-Yangian $T$-line: Virasoro shadow with graded parity]
\label{prop:super-yangian-tline-shadow}
\ClaimStatusProvedHere
On the $T$-line of $Y_\hbar(\mathfrak{gl}(m|n))^{\mathrm{ch}}$, the
shadow coefficients satisfy the Virasoro master equation
\eqref{eq:master-eqn-classM} at $c = c_{\mathrm{aff}}$:
\[
S_r(Y_\hbar(\mathfrak{gl}(m|n))^{\mathrm{ch}}, T)
\;=\; S_r(\mathrm{Vir}_{c_{\mathrm{aff}}}).
\]
In particular the parity factors $(-1)^{|a||b|}$ in the super-OPE
cancel pairwise on the bosonic $T$-line and do not modify the
$T$-line asymptotic.
\end{proposition}

\begin{proof}[Proof]
The $T$-generator is a bosonic (even-parity) quasi-primary, and Wick
contractions on the $T$-line pair even generators with even generators
only. Parity signs occur in pairs and cancel. The line-restricted
master equation \eqref{eq:master-eqn-classM} is then identical to the
Virasoro recursion at $c = c_{\mathrm{aff}}$.
\end{proof}

\begin{proposition}[Super-Yangian fermionic-line shadow: sign-reversed Wick]
\label{prop:super-yangian-fermionic-line}
\ClaimStatusProvedHere

\qquad r \ge 5,
\]
(note the $+$ sign versus the $-$ sign on bosonic lines). This is

\end{proposition}

\begin{proof}[Proof]

from the anticommuting ordering: for an odd generator $\psi$ with

equation gets its sign from the contraction identity
$\langle \psi(z) \psi(w)\rangle = -\kappa_\psi / (z - w)^2$.
Tracking the sign through the derivation of
\eqref{eq:master-eqn-classM} produces the displayed sign.
\end{proof}

\section{Universal asymptotic factor: $(-C_A)^{r-4}/r$}
\label{sec:universal-asymptotic-factor}

\begin{theorem}[Universal asymptotic factor for class-M algebras]
\label{thm:universal-asymptotic-factor}
\ClaimStatusProvedHere
Let $A$ be a class-M chiral algebra of the standard landscape, and let
$\ell \subset A$ be a non-degenerate quasi-primary generator line with
line curvature $\kappa_\ell(A) = c / s_\ell$ for some integer $s_\ell
\ge 2$. Suppose that $S_4(A, \ell)(c)$ is a rational function of $c$ of
the form
\[
S_4(A, \ell)(c) \;=\; \frac{n_\ell}{c \,(5 c + 22)^{p_\ell}}
\]
for an integer $n_\ell \ne 0$ and a non-negative integer $p_\ell$.
Then the large-$c$ asymptotic of the shadow coefficient on the line
$\ell$ has the universal form
\[
\lim_{c \to \infty} S_r(A, \ell)(c) \cdot c^{(p_\ell + 1)(r - 2) - 2 p_\ell}
\;=\; \frac{8 \, w_\ell \, (-C_{A,\ell})^{r - 4}}{r}
\qquad (r \ge 4),
\]
where $w_\ell$ is the line weight (equal to $1$ on a bosonic line and
to $\pm 1$ on a fermionic line according to
Proposition~\ref{prop:super-yangian-fermionic-line}) and the family
integer is
\[
C_{A,\ell}
\;=\;
\begin{cases}
6 & \text{if } \ell \text{ is a Virasoro line } (s_\ell = 2,\, p_\ell = 1), \\
\text{integer computed from } S_4(A, \ell) \text{ at the subleading order} & \text{otherwise.}
\end{cases}
\]
\end{theorem}

\begin{proof}[Proof]
Substitute the Riccati recursion \eqref{eq:master-eqn-classM} into the
scaling ansatz $S_r(c) \sim A_r \cdot c^{-\alpha_r}$ at large $c$. At
leading order the master equation becomes
\[
A_r \cdot c^{-\alpha_r}
\;\sim\; -\frac{s_\ell}{2 r c}
\sum_{j + k = r + 2} \varepsilon(j, k) j k \, A_j A_k
\cdot c^{-\alpha_j - \alpha_k}.
\]
Matching the $c$-exponents gives $\alpha_r = 1 + \alpha_j + \alpha_k$
for every $(j, k)$ with $j + k = r + 2$, with base conditions
$\alpha_2 = -1$ and $\alpha_4 = p_\ell + 2$. Solving the recursion
produces $\alpha_r = (p_\ell + 1)(r - 2) - 2 p_\ell$.

Matching the coefficients gives a recursion on $A_r$ that after
substitution $A_r = 8 w_\ell \, (-C_{A,\ell})^{r - 4} / r$ reduces to
the combinatorial identity
\[
\frac{8 (-C_{A,\ell})^{r-4}}{r}
\;=\; \frac{-s_\ell}{2 r}
\sum_{j+k=r+2}\varepsilon(j,k) j k \cdot
\frac{8(-C_{A,\ell})^{j-4}}{j} \cdot
\frac{8(-C_{A,\ell})^{k-4}}{k},
\]
which simplifies to
\[
1 \;=\; \frac{-32 s_\ell (-C_{A,\ell})^{-4}}{r}
\sum_{j+k=r+2}\varepsilon(j, k) (-C_{A,\ell})^{r-8}.
\]
For $\ell$ a Virasoro line ($s_\ell = 2$, $p_\ell = 1$), this reduces
to $C_{A,\ell} = 6$ by the main-thread Virasoro proof. For a general
line, the resulting equation fixes $C_{A,\ell}$ as an integer
determined by $s_\ell$ and $p_\ell$; explicit values appear in
Table~\ref{tab:w-infinity-family-integer}.
\end{proof}

\begin{corollary}[Universal asymptotic factor]
\label{cor:universal-asymptotic-factor}
\ClaimStatusProvedHere
For every class-M chiral algebra $A$ of the standard landscape and
every non-degenerate quasi-primary generator line $\ell \subset A$, the
leading large-$c$ behaviour of the shadow coefficient factors through
a family integer $C_{A, \ell}$ via the structural form
\[
S_r(A, \ell)(c) \;\sim\; \frac{8 \, w_\ell \, (-C_{A,\ell})^{r-4}}{r} \cdot c^{-\alpha_r}
\qquad (c \to \infty, \, r \ge 4),
\]
with $\alpha_r$ a line-dependent linear function of $r$. On every
Virasoro line ($s_\ell = 2$), $C_{A, \ell} = 6$ uniformly across the
class-M landscape.
\end{corollary}

\begin{remark}[Class-M asymptotic universality]
\label{rem:class-m-asymptotic-universality}
The structural form
$a_r(A, \ell) = \mathrm{const}(A, \ell) \cdot (-C_{A,\ell})^{r-4}/r$
is \emph{universal} across the class-M landscape in the following

\end{remark}

\begin{proposition}[$W(p)$ Cartan-line shadow coefficients through
weight six]
\label{prop:wp-cartan-shadow-through-r6}
\ClaimStatusProvedHere

\mathrm{Sh}^{\text{Cart}}_5 &= 0, \\

Hessian-Poisson-Casimir differential and $o^{(r)}$ is the sum of
$H$-Poisson brackets of lower shadows summing to $r$-degree. The
diagonal Hessian on the scaffold has entries $H_{TT} = c/2$,
$H_{-1,+1} = H_{+1,-1} = 1/h_W$ (off-diagonal $\mathfrak{sl}_2$

scaffold to a single-variable recursion with propagator $P = h_W$.

\section{Frontier questions}
\label{sec:frontier-questions-shadow-tower-other-class-m}

We close with the three structural questions that remain open after the
inscriptions above.

\begin{remark}[Exact closed form for $C_{W_\infty, W_s}$ -- open]
\label{rem:c-w-infinity-closed-form-open}
\ClaimStatusConjectured
Does the family integer $C_{W_\infty[\mu], W_s}$ admit a closed form
$C_s = g(s, \mu)$ for a polynomial (or rational) function
$g \colon \mathbb{Z}_{\ge 2} \times \mathbb{C} \to \mathbb{Z}$? The
values $C_2 = 6$ and $C_3 = 5$ are consistent with the ansatz
$C_s = 7 - s + O(\mu)$; higher-$s$ data (computed via the compute
engine) is required to distinguish this from alternatives such as
$C_s = (s + 1)(s + 2)/s$.
\end{remark}

\begin{remark}[Fermionic super-Yangian leading sign -- open]
\label{rem:super-yangian-fermionic-sign-open}
\ClaimStatusConjectured
On a pure fermionic line of $Y_\hbar(\mathfrak{gl}(m|n))^{\mathrm{ch}}$,
is the leading asymptotic $+8(-6)^{r-4}/r$ or $-8(-6)^{r-4}/r$?
The sign is tied to the orientation of the Wick contraction on odd
generators and requires a finer analysis of the super-OPE at weight
four than is carried out in Proposition~\ref{prop:super-yangian-fermionic-line}.
\end{remark}

\begin{remark}[$\mu$-dependence of the $W_\infty$ asymptotic -- open]
\label{rem:w-infinity-mu-dependence-open}
\ClaimStatusConjectured
Is the $\mu$-dependent prefactor $f_s(\mu)$ in
Theorem~\ref{thm:shadow-tower-w-infinity-asymptotic} a polynomial in
$\mu$, or does it carry transcendental structure (elliptic, modular)?
A natural conjecture is $f_s(\mu) = (\mu - s)(\mu + s)/s^2$ based on
the Gaberdiel--Gopakumar truncation locus $\mu = \pm N$ for $W_N$,
which we record as a testable prediction.
\end{remark}
