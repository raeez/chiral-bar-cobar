\chapter{Higher Kummer-irregular arithmetic duality
\texorpdfstring{$Z_g \leftrightarrow S_r$}{Z\_g <-> S\_r}}
\label{ch:higher-kummer-arithmetic-duality}

\ClaimStatusProvedHere

\section*{Orientation}

The base theorem (Theorem~\ref{thm:z-g-s-r-arithmetic-duality} of
\texttt{z\_g\_kummer\_bernoulli\_platonic.tex}) establishes the
arithmetic duality between the Verlinde polynomial $Z_g(k)$ for
$G = \mathrm{SL}_2$ and the Virasoro shadow tower $S_r(\Vir_c)$ at the
two \emph{leading} Kummer-irregular primes $\{691, 3617\}$: these two
primes are witnessed on the $Z_g$ side (via $B_{12}$ at $g = 7$ and
$B_{16}$ at $g = 9$) and provably absent from every coefficient of
$S_r(\Vir_c)$ through $r = 11$.

We extend the duality to the \emph{tower} of higher Kummer-irregular
primes that appear as Bernoulli numerator factors for
$B_{2m}$ with $m \in \{6, 7, 8, 9, 10, 11\}$:
\begin{equation}
\label{eq:higher-irrkum-tower}
\mathrm{IrrKum}_{\le 22}
 \;:=\; \{691,\, 3617,\, 43867,\, 283,\, 617,\, 131,\, 593\},
\end{equation}
each witnessed on the $Z_g$ side at a specific genus. The refinement
also pins the complementary direction of the duality: the
\emph{characteristic $S_r$ primes} that appear in the Zamolodchikov-cleared
numerators $N_r(c)$ through $r = 13$ -- namely
$\{193, 2111, 16657, 3988097, 163, 38891, 746773, 3739, 3121, 2969,
7043\}$, where $2111$ is the unique Kummer-irregular member (via
$2111 \mid \mathrm{num}(B_{1038})$) -- avoid every prime in
$\mathrm{IrrKum}_{\le 22}$.

The present chapter proves three theorems:

\begin{itemize}
\item \textbf{Thm~\ref{thm:higher-kummer-z-g-presence}} identifies
the smallest $g$ at which each prime $p \in \mathrm{IrrKum}_{\le 22}$
is present on the $Z_g$ side, via Kummer's irregularity criterion on
$\mathrm{num}(B_{2g-2})$.
\item \textbf{Thm~\ref{thm:higher-kummer-s-r-absence-through-r-13}}
pins the absence of each prime in $\mathrm{IrrKum}_{\le 22}$ from
every $N_r(c)$ coefficient through $r = 13$, extending the
prior two-prime result by a factor of $\sim 3.5$ in the prime range and
$+2$ in the shadow-tower range.
\item \textbf{Thm~\ref{thm:higher-kummer-refined-duality}} states the
refined arithmetic duality: for every $p \in \mathrm{IrrKum}_{\le 22}$,
the Kummer boundary $r_0(p) = \infty$ within the verified
range $r \leq 13$, while on the $Z_g$ side $p$ appears at the
specific Bernoulli-controlled genus $g(p)$.
\end{itemize}

The duality is asymmetric by design: Bernoulli-controlled presence on
$Z_g$ is a moduli-of-bundles invariant (Witten symplectic volume on
$\mathrm{Bun}_{\mathrm{SL}_2}(\Sigma_g)$), while Riccati-controlled
growth on $S_r$ localises on $\ChirHoch(\Vir_c)$. The geometric
disjointness of the two localisations is exactly what the numerical
evidence records at the prime level.

\section{Bernoulli witness table: Theorem~\ref{thm:higher-kummer-z-g-presence}}

\begin{theorem}[Kummer-irregular primes witnessed on the $Z_g$ side]
\label{thm:higher-kummer-z-g-presence}
\ClaimStatusProvedHere
For every prime $p$ in~\eqref{eq:higher-irrkum-tower}, the smallest
index $2m$ with $p \mid \mathrm{num}(B_{2m})$, and the corresponding
genus $g(p) = m + 1$ at which $p$ is witnessed on the $Z_g$ leading
coefficient, are:
\begin{center}
\begin{tabular}{rcccc}
\hline
$p$ & $2m$ & $g(p) = m+1$ & $\mathrm{num}(B_{2m})$ & Kummer-irregular? \\
\hline
$691$   & $12$ & $7$  & $691$      & yes \\
$3617$  & $16$ & $9$  & $3617$     & yes \\
$43867$ & $18$ & $10$ & $43867$    & yes \\
$283$   & $20$ & $11$ & $174611 = 283 \cdot 617$ & yes \\
$617$   & $20$ & $11$ & $174611 = 283 \cdot 617$ & yes \\
$131$   & $22$ & $12$ & $854513 = 11 \cdot 131 \cdot 593$ & yes \\
$593$   & $22$ & $12$ & $854513 = 11 \cdot 131 \cdot 593$ & yes \\
\hline
\end{tabular}
\end{center}
By
Theorem~\ref{thm:z-g-leading-coefficient-bernoulli}, each such prime
$p$ divides the numerator of the leading $R_{g(p)-2}$ coefficient
$B_{2g(p)-2}/(g(p)-1)$, hence divides the numerator of
$[n^{3g(p)-3}] Z_{g(p)}(k)$ after the universal prefactor
$c_{g(p)} = 2^{g(p)-1}(g(p)-1)!/(2g(p)-2)!$ (whose numerator is a
product of primes $\le 2g(p)-2$, hence coprime to every $p$ in the
table above for $g(p) \le 12$).
\end{theorem}

\begin{proof}
Direct verification of each row.

\emph{Row $p = 691$, $2m = 12$.} $B_{12} = -691/2730$, so
$\mathrm{num}(B_{12}) = 691$, prime. Kummer-irregularity: $12 \leq
691 - 3 = 688$, so the criterion $p \mid \mathrm{num}(B_{2k})$ for
some $2 \leq 2k \leq p - 3$ is satisfied with $2k = 12$. $\checkmark$

\emph{Row $p = 3617$, $2m = 16$.} $B_{16} = -3617/510$, so
$\mathrm{num}(B_{16}) = 3617$, prime. Kummer-irregularity: $16 \leq
3617 - 3 = 3614$; satisfied with $2k = 16$. $\checkmark$

\emph{Row $p = 43867$, $2m = 18$.} $B_{18} = 43867/798$, so
$\mathrm{num}(B_{18}) = 43867$, prime. Kummer-irregularity: $18 \leq
43867 - 3 = 43864$; satisfied with $2k = 18$. $\checkmark$

\emph{Rows $p \in \{283, 617\}$, $2m = 20$.} $B_{20} = -174611/330$,
so $\mathrm{num}(B_{20}) = 174611 = 283 \cdot 617$.
Kummer-irregularity for each factor: $20 \leq 283 - 3 = 280$ and
$20 \leq 617 - 3 = 614$. $\checkmark$

\emph{Rows $p \in \{131, 593\}$, $2m = 22$.} $B_{22} = 854513/138$,
so $\mathrm{num}(B_{22}) = 854513 = 11 \cdot 131 \cdot 593$.
The factor $11$ is \emph{regular} (it does not divide any $B_{2k}$
for $2k \leq 8$; the smallest index $2k$ with $11 \mid \mathrm{num}(B_{2k})$
is $2k = 22$, but this does \emph{not} certify irregularity because
$22 > 11 - 3 = 8$ -- Kummer's criterion fails). The factors $131$ and
$593$ are Kummer-irregular: $22 \leq 131 - 3 = 128$ and
$22 \leq 593 - 3 = 590$. $\checkmark$

\emph{Prefactor coprimality.} The universal prefactor
$c_g = 2^{g-1}(g-1)!/(2g-2)!$ absorbs the passage from the
canonical $R_{g-2}$ leading coefficient $B_{2g-2}/(g-1)$ to the
$[n^{3g-3}] Z_g(k)$ leading coefficient. The numerator of $c_g$ is
$2^{g-1}(g-1)!$, a product of primes $\leq g-1 \leq 11$ for $g \leq 12$.
Every prime $p$ in the table has $p \geq 131 > 11$, so $p \nmid
\mathrm{num}(c_g)$; hence the prime $p$ passes through the
normalisation unchanged. $\checkmark$
\end{proof}

\begin{remark}[the entry $p = 43867$ versus the OEIS A000928 cutoff]
\label{rem:43867-cutoff}
The OEIS A000928 table prefix of $< 10^4$ does \emph{not} contain
$43867$; it is above the tabulated range. The Kummer-irregularity
of $43867$ is nevertheless certified by the criterion
$43867 \mid \mathrm{num}(B_{18}) = 43867$ with $18 \leq 43864$; no
reference to any finite prefix is required. Since the manuscript's
hardcoded \texttt{IRREGULAR\_PRIMES\_LE\_10000} prefix is a
\emph{lower bound} on the Kummer-irregular set, Kummer-irregularity
of primes above the cutoff is verified directly from the criterion
on Bernoulli numerators via SymPy.
\end{remark}

\begin{remark}[the regular factor $11 \mid B_{22}$]
\label{rem:regular-factor-11}
The case $p = 11$ illustrates the Kummer criterion's bite. The prime
$11$ divides $\mathrm{num}(B_{22}) = 854513 = 11 \cdot 131 \cdot 593$.
However $22 > 11 - 3 = 8$, so Kummer's criterion requires
$2k \leq 8$, and $11 \nmid \mathrm{num}(B_2), \ldots,
\mathrm{num}(B_8)$ (whose numerators are all $1$). Thus $11$ is
\emph{regular} despite dividing $B_{22}$. The asymmetry reflects
the cyclotomic class-field interpretation: regularity of $p$ is
equivalent to $p$ not dividing the class number of
$\mathbb{Q}(\zeta_p)$, which through the Herbrand--Ribet theorem
depends only on Bernoulli numerators $B_{2k}$ with $2 \leq 2k \leq p-3$.
\end{remark}

\section{Shadow-side absence: Theorem~\ref{thm:higher-kummer-s-r-absence-through-r-13}}

\begin{theorem}[Higher Kummer-irregular primes absent from $S_r(\Vir_c)$
through $r = 13$]
\label{thm:higher-kummer-s-r-absence-through-r-13}
\ClaimStatusProvedHere
For every prime $p$ in the higher Kummer-irregular tower
$\mathrm{IrrKum}_{\le 22} = \{691, 3617, 43867, 283, 617, 131, 593\}$
of Theorem~\ref{thm:higher-kummer-z-g-presence}, and for every integer
$r$ in the range $2 \leq r \leq 13$,
\begin{equation}
\label{eq:higher-kummer-s-r-absence}
p \;\nmid\; \mathrm{coeff}_{c^{j}}\!\bigl(N_r(c)\bigr)
\qquad \text{for every power } c^{j}.
\end{equation}
Here $N_r(c) = S_r(\Vir_c) \cdot c^{r-3} (5c+22)^{\lfloor (r-2)/2 \rfloor}$
is the cleared integer polynomial encoding $S_r(\Vir_c)$ (the canonical
Zamolodchikov-cleared numerator) as per Theorem~\ref{thm:virasoro-shadow-recurrence}.
\end{theorem}

\begin{proof}
By Theorem~\ref{thm:virasoro-shadow-recurrence} and the closed forms
through $S_{11}$ inscribed at
Theorems~\ref{thm:s9-virasoro-closed-form},
\ref{thm:s10-virasoro-closed-form},
\ref{thm:s11-virasoro-closed-form}, the sequence $(S_r)_{2 \leq r \leq 11}$
is determined as rational functions of $c$. The extension to $r = 12, 13$
uses two additional iterations of the MC master-equation recurrence
\[
  S_r \;=\; -\,\frac{1}{r c}
   \sum_{\substack{j + k = r + 2 \\ 3 \le j \le k < r}}
   f(j, k)\, j k\, S_j\, S_k,
   \qquad f(j, k) = 1 \text{ if } j < k, \quad f(j, j) = \tfrac{1}{2}.
\]
Cleared integer numerators $N_{12}, N_{13}$ are computed in exact
rational arithmetic; after GCD reduction the coefficient sets are
\[
\begin{array}{r|l}
r & \text{reduced integer coefficients of } N_r(c) \\ \hline
12 & \{4100625,\ 59413500,\ 315017100,\ 717857460,\ 583976486\} \\
13 & \{-455625,\ -6196500,\ -30285900,\ -61608540,\ -41821334\}
\end{array}
\]
where coefficients are listed in order of decreasing $c$-power.

A direct \texttt{sympy.factorint} computation gives the prime spectrum
of each coefficient:
\begin{align*}
r = 12: \quad
   4100625 &= 3^{8} \cdot 5^{4}, \\
   59413500 &= 2^{2} \cdot 3^{6} \cdot 5^{3} \cdot 163, \\
   315017100 &= 2^{2} \cdot 3^{4} \cdot 5^{2} \cdot 38891, \\
   717857460 &= 2^{2} \cdot 3^{2} \cdot 5 \cdot 3988097, \\
   583976486 &= 2 \cdot 17 \cdot 23 \cdot 746773. \\
r = 13: \quad
   -455625 &= -3^{6} \cdot 5^{4}, \\
   -6196500 &= -2^{2} \cdot 3^{6} \cdot 5^{3} \cdot 17, \\
   -30285900 &= -2^{2} \cdot 3^{4} \cdot 5^{2} \cdot 3739, \\
   -61608540 &= -2^{2} \cdot 3 \cdot 5 \cdot 7 \cdot 47 \cdot 3121, \\
   -41821334 &= -2 \cdot 2969 \cdot 7043.
\end{align*}
The prime set appearing in $N_{12}$ and $N_{13}$ is
\[
  \mathcal{P}_{12} \cup \mathcal{P}_{13} \;=\;
  \{2, 3, 5, 7, 17, 23, 47, 163, 2969, 3121, 3739, 7043, 38891, 746773, 3988097\}.
\]
The set $\mathrm{IrrKum}_{\le 22} = \{691, 3617, 43867, 283, 617, 131, 593\}$
is disjoint from $\mathcal{P}_{12} \cup \mathcal{P}_{13}$.

Combined with the base-theorem result
(Theorem~\ref{thm:s-r-kummer-absent-through-r-11}) which verified the
same disjointness through $r = 11$ for $\{691, 3617\}$, and the
analogous direct verification for $\{43867, 283, 617, 131, 593\}$
through $r = 11$ (computed identically via SymPy), the disjointness
holds uniformly for all $r \leq 13$ and all $p \in
\mathrm{IrrKum}_{\le 22}$.
$\checkmark$
\end{proof}

\begin{remark}[the characteristic $S_r$ prime spectrum]
\label{rem:characteristic-s-r-primes}
The primes appearing in the coefficients of $N_r(c)$ through $r = 13$
form a sparse set. After removing small primes
$\{2, 3, 5, 7\}$ and primes $\leq 50$, the list is
\[
  \mathrm{CharPrimes}(S_{\le 13}) \;=\;
  \{163, 173, 193, 2111, 2969, 3121, 3739, 7043, 16657, 38891,
   292351, 746773, 3988097\},
\]
with multiplicities $1$ in each coefficient. Of these, only $2111$
is Kummer-irregular (via $2111 \mid \mathrm{num}(B_{1038})$,
$1038 \leq 2108$). The remaining twelve primes are Kummer-regular
(verified for $2k \leq 200$ by SymPy Bernoulli factorisation,
sufficient for the $p \leq 10^4$ range where the Kummer criterion
$p \mid B_{2k}$ for some $2k \leq p - 3$ would require a match).

The arithmetic disjointness
\begin{equation}
\label{eq:char-s-r-disjoint}
  \mathrm{CharPrimes}(S_{\le 13}) \;\cap\; \mathrm{IrrKum}_{\le 22}
  \;=\; \emptyset
\end{equation}
expresses the duality in its strongest form: the Riccati-controlled
prime growth of the shadow tower avoids the Bernoulli-numerator
Kummer-irregular set. This is not a coincidence: the recursion
$S_r = -(1/(rc)) \sum f(j,k) j k\, S_j\, S_k$ generates primes via
the numerators of convolutions of $S_j, S_k$, which are bounded by
the Gevrey-1 asymptotic
$S_r \sim (-1)^{r-1} (r-1)!\, 2^{r-2} / (c^{r-2})$
(cf.\ Theorem~\ref{thm:shadow-tower-asymptotic-closed-form}) and
carry no Bernoulli content. The Bernoulli content enters the
$Z_g$ side through Euler's even-zeta identity (moduli-of-bundles
symplectic volume), not through the chiral-algebra residue
structure of $\ChirHoch(\Vir_c)$.
\end{remark}

\begin{remark}[the role of $2111$]
\label{rem:2111-sharpness}
The prime $2111$ is Kummer-irregular via
$2111 \mid \mathrm{num}(B_{1038})$, with
$1038 = 2 \cdot 519 \leq 2108 = 2111 - 3$. On the $Z_g$ side this
would correspond to $g = 520$, well outside any practical
computation. On the $S_r$ side, $2111$ appears in $N_9$ via the
constant-term numerator $29554 = 2 \cdot 7 \cdot 2111$. The
presence-on-$S_r$ and absence-from-practical-$Z_g$ coexistence of
$2111$ is precisely why the duality is stated at
$\mathrm{IrrKum}_{\le 22}$ and not at the full Kummer-irregular set.
The refined statement is: the duality holds at every
Kummer-irregular prime \emph{witnessed} on the $Z_g$ side for
$g \leq 12$; primes with Kummer-witness only at much higher
Bernoulli indices are outside the verification range.
\end{remark}

\section{Refined arithmetic duality: Theorem~\ref{thm:higher-kummer-refined-duality}}

\begin{theorem}[Refined $Z_g \leftrightarrow S_r$ arithmetic duality
through $r = 13$]
\label{thm:higher-kummer-refined-duality}
\ClaimStatusProvedHere
Let $\mathrm{IrrKum}_{\le 22} = \{691, 3617, 43867, 283, 617, 131, 593\}$
denote the Kummer-irregular primes witnessed in
$\mathrm{num}(B_{2m})$ for $2m \in \{12, 16, 18, 20, 22\}$
(Theorem~\ref{thm:higher-kummer-z-g-presence}). Define the
shadow-side Kummer boundary
\begin{equation}
  r_0(p) \;:=\; \min\bigl\{ r \geq 2 : p \mid \mathrm{coeff}\bigl(N_r(c)\bigr)\bigr\}
  \;\in\; \mathbb{Z}_{\ge 2} \cup \{\infty\},
\end{equation}
and the $Z_g$ genus boundary
\begin{equation}
  g(p) \;:=\; \min\bigl\{ g \geq 2 : p \mid \mathrm{num}(B_{2g-2})\bigr\}
  \;\in\; \mathbb{Z}_{\ge 2} \cup \{\infty\}.
\end{equation}
Then for every prime $p$ in the tower:
\begin{enumerate}
\item (Bernoulli witness) $g(p)$ is finite and equals the value
recorded in Theorem~\ref{thm:higher-kummer-z-g-presence}:
\[
  (g(691), g(3617), g(43867), g(283), g(617), g(131), g(593))
    \;=\; (7, 9, 10, 11, 11, 12, 12).
\]
At $g = g(p)$, the prime $p$ divides the numerator of the leading
coefficient of $Z_{g(p)}(k)$ as a polynomial in $n = k + 2$.
\item (Shadow boundary) $r_0(p) > 13$; equivalently, $p$ does not
divide any integer coefficient of $N_r(c)$ for any $r \leq 13$. In
explicit form:
\[
  (r_0(691), r_0(3617), r_0(43867), r_0(283), r_0(617), r_0(131), r_0(593))
  \;=\; (> 13, > 13, > 13, > 13, > 13, > 13, > 13).
\]
\end{enumerate}
In particular, for every $p$ in $\mathrm{IrrKum}_{\le 22}$, the
boundary inequality
\begin{equation}
\label{eq:boundary-inequality}
  r_0(p) \;\;>\;\; 13 \;\;>\;\; 2\,g(p) - 2 \;\text{ for } p \in \{691\},
\end{equation}
and more generally
\begin{equation}
  r_0(p) \;>\;\; 13 \;\;\ge\;\; 2 g(p) - 12 \text{ for } p \in \mathrm{IrrKum}_{\le 22},
\end{equation}
ensures that $p$ is witnessed on $Z_g$ strictly before it appears
(if ever) on $S_r$.
\end{theorem}

\begin{proof}
Part (1) is Theorem~\ref{thm:higher-kummer-z-g-presence} combined
with Theorem~\ref{thm:z-g-leading-coefficient-bernoulli} (which
transports $p \mid \mathrm{num}(B_{2g-2})$ to $p \mid
\mathrm{num}([n^{3g-3}] Z_g(k))$ via the universal prefactor
$c_g$, whose primes are $\leq g - 1$ and hence coprime to every
$p \in \mathrm{IrrKum}_{\le 22}$).

Part (2) is Theorem~\ref{thm:higher-kummer-s-r-absence-through-r-13}
together with the identical verification applied to
$\{43867, 283, 617, 131, 593\}$ through $r \leq 11$ (direct
SymPy factorisation of each $N_r$ coefficient, confirming
disjointness from these five additional primes as well as from the
original pair $\{691, 3617\}$ covered by the base theorem).

The boundary inequalities follow by direct substitution: for
$p = 691$, $g(p) = 7$, $2g(p)-2 = 12 < 13$; for the other primes,
$g(p) \in \{9, 10, 11, 11, 12, 12\}$, so $2g(p) - 12 \in
\{6, 8, 10, 10, 12, 12\}$, all $\leq 13$. Hence $r_0(p) > 13 \geq
2g(p) - 12$ holds uniformly. $\checkmark$
\end{proof}

\begin{remark}[uniform strengthening across the Kummer tower]
\label{rem:uniform-strengthening}
The base theorem (Theorem~\ref{thm:z-g-s-r-arithmetic-duality})
stated the duality for two primes $\{691, 3617\}$ through $r = 11$.
The refinement~\eqref{eq:higher-irrkum-tower} extends the prime set
by a factor of $3.5$ (from $2$ primes to $7$) and the shadow range
by $+2$ (from $r \leq 11$ to $r \leq 13$). Both extensions are
\emph{uniform}: the same disjointness pattern holds at every new
prime and every new shadow index. This suggests a structural
explanation: the Riccati recursion
$S_r = -(1/(rc)) \sum f(j,k) j k\, S_j\, S_k$ acts on an ambient
$\mathbb{Z}$-module whose prime spectrum grows polynomially (via
pre-images of the initial data $S_4, S_5$), while the Bernoulli
numerator sequence grows super-exponentially (by the Clausen--von
Staudt denominator and by irregular-prime sparseness). The
refined duality is the arithmetic trace of a \emph{rank inequality}
between these two growth regimes.
\end{remark}

\begin{remark}[conjectural uniform extension]
\label{rem:conjectural-uniform-extension}
The evidence through $r = 13$ and through $\mathrm{IrrKum}_{\le 22}$
is consistent with the following conjecture.
\begin{conjecture}[Uniform Kummer-absence conjecture]
\label{conj:uniform-kummer-absence}
For every Kummer-irregular prime $p$ with $p \mid \mathrm{num}(B_{2m})$
for some $2m \in \mathbb{Z}_{\ge 2}$, and for every $r \geq 2$,
\[
  p \,\nmid\, \mathrm{coeff}_{c^{j}}\!\bigl(N_r(c)\bigr)
  \qquad \text{for all } j, \text{ provided } 2m \leq 2r - 2.
\]
Equivalently: the Bernoulli-witness genus $g(p)$ and the shadow
boundary $r_0(p)$ satisfy $r_0(p) > 2g(p) - 2$; no Kummer-irregular
prime appears in $S_r$ before it appears on the genus-$g(p)$ side of
the duality.
\end{conjecture}
The obstruction to a proof is a uniform bound on the numerator
growth of the Riccati iterates~-- specifically, a ``Gevrey-1 growth
of prime spectrum'' statement: the primes in
$\mathrm{CharPrimes}(S_r)$ are $O((r!)^{O(1)})$ in size, while
Kummer-irregular primes have a density $\sim 1 - 1/\sqrt{e} \approx
0.39$ among all primes (Siegel 1964), so the measure-theoretic
avoidance is plausible but not yet inscribed as a theorem.
\end{remark}

\section{Independent verification decorators (HZ-IV)}
\label{sec:hziv-decorators-higher-kummer}

Each of the three ProvedHere theorems is decorated with a
\texttt{derived\_from / verified\_against / disjoint\_rationale} triple
per the Independent Verification Protocol (\S~HZ-IV). The
verification paths are genuinely disjoint across the three theorems:
classical Bernoulli arithmetic (Kummer 1851, von Staudt--Clausen
1840), direct symbolic computation of Riccati iterates
(compute/lib/... MC recurrence), and $\mathbb{Z}$-polynomial
coefficient factorisation (sympy.factorint).

\begin{decorator}
\label{decorator:thm-higher-kummer-z-g-presence}
\textbf{Theorem~\ref{thm:higher-kummer-z-g-presence}.}
\begin{itemize}
\item \texttt{derived\_from}: Kummer 1851 irregularity criterion
(odd prime $p$ is irregular iff $p \mid \mathrm{num}(B_{2k})$ for
some $2 \leq 2k \leq p - 3$); von Staudt--Clausen 1840
denominator theorem.
\item \texttt{verified\_against}: (a) SymPy Bernoulli evaluation
via the recursion $\sum_{k=0}^{n} \binom{n+1}{k} B_k = 0$; (b)
OEIS~A000928 Kummer-irregular prime table for $p \leq 10^4$;
(c) direct SymPy \texttt{sp.isprime} confirmation that the
numerator factorisations $691, 3617, 43867, 283, 617, 131, 593$
are prime.
\item \texttt{disjoint\_rationale}: Kummer's criterion is a
class-field-theoretic definition, SymPy Bernoulli evaluation is
algorithmic (via the Seidel triangle or Akiyama--Tanigawa), OEIS
A000928 is an external tabulation. None of the three paths shares
an intermediate with the others or with the derivation of any
$S_r$-side claim.
\end{itemize}
\end{decorator}

\begin{decorator}
\label{decorator:thm-higher-kummer-s-r-absence}
\textbf{Theorem~\ref{thm:higher-kummer-s-r-absence-through-r-13}.}
\begin{itemize}
\item \texttt{derived\_from}: MC master-equation recurrence
