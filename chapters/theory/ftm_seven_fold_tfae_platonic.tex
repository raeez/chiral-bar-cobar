\chapter{The FTM seven-fold TFAE at genus zero: hub-and-spoke theorem}
\label{chap:ftm-seven-fold-tfae-platonic}

\section{The finite-type genus-zero window}
\label{sec:ftm-sevenfold-setup}

The phrase ``seven-fold TFAE'' has a finite domain.  The intrinsic
fundamental-twisting-morphism package has six genus-zero conditions:
Koszulness of $\tau_{\mathrm{univ}}$, the bar--cobar counit,
the bar--cobar unit, twisted tensor acyclicity, diagonal bar
concentration, and PBW $E_2$-collapse.  The seventh condition,
Swiss-cheese formality, is not an intrinsic consequence of the first
six on the whole Koszul locus.  It becomes equivalent to them only on
the class-$G$ fibre of the shadow-archetype map.

\begin{definition}[Genus-zero finite-type PBW window]
\label{def:ftm-g0-pbw-window}
\index{FTM TFAE!finite-type PBW window}
A filtered augmented chiral algebra $(\cA,F_\bullet)$ lies in the
\emph{genus-zero finite-type PBW window} if the following hypotheses
hold.
\begin{enumerate}[label=\textup{(W\arabic*)}]
\item $X$ is a smooth genus-zero curve and
$\tau_{\mathrm{univ}}\colon \barB^{\ord}_X(\cA)\to\cA$ is the
universal twisting morphism of
Corollary~\ref{cor:three-bijections}.
\item The quadruple
$(\cA,\barB^{\ord}_X(\cA),\tau_{\mathrm{univ}},F_\bullet)$ is a
chiral twisting datum in the sense of
Definition~\ref{def:chiral-twisting-datum}: the filtration on
$\cA$, $\barB^{\ord}_X(\cA)$, and
$\Omega_X\barB^{\ord}_X(\cA)$ is exhaustive, complete, bounded
below, and compatible with the structure maps.
\item The PBW filtration is flat and finite in the sense of
Theorem~\ref{thm:pbw-koszulness-criterion}
\textup{\ref{item:pbw-flat}} and
\textup{\ref{item:pbw-bounded}}: every PBW graded piece is finite
rank in each conformal weight, and every fixed bar degree and
conformal weight of $\barB^{\ord}_X(\cA)$ is finite-dimensional.
\item The filtered bar, cobar, and twisted-tensor spectral sequences
converge strongly, so the intrinsic core of
Theorem~\ref{thm:koszul-equivalences-meta} applies:
chiral Koszulness of $\tau_{\mathrm{univ}}$ is equivalent to PBW
$E_2$-collapse.
\item When the notation $\cA^!$ is used, the finite graded-piece
duality needed for continuous linear or Verdier duality is present;
otherwise the theorem stops at the coalgebra
$\cA^{\mathrm i}=H^\bullet\barB^{\ord}_X(\cA)$ and makes no
algebra-level assertion about $\cA^!$.
\end{enumerate}
The generic standard-landscape witnesses cited below lie in this
window.  Critical affine level, non-generic simple quotients,
logarithmic examples, raw direct-sum class-$M$ completions, and
multi-weight higher-genus scalar reductions require their own
completion or perfectness hypotheses and are not included by the
word ``PBW'' alone.
\end{definition}

The proof uses the following local anchors:
Theorem~\ref{thm:fundamental-twisting-morphisms},
Theorem~\ref{thm:pbw-koszulness-criterion},
Theorem~\ref{thm:koszul-equivalences-meta},
Theorem~\ref{thm:bar-concentration},
Theorem~\ref{thm:ext-diagonal-vanishing},
Theorem~\ref{thm:bar-cobar-inversion-qi}, and
Proposition~\ref{prop:sc-formal-iff-class-g}.

\section{The seven conditions at genus zero}
\label{sec:ftm-seven-conditions}

Throughout, $(\cA,F_\bullet)$ denotes an augmented chiral algebra in
the genus-zero finite-type PBW window of
Definition~\ref{def:ftm-g0-pbw-window}, and
$\tau_{\mathrm{univ}}\colon \bar{B}^{\ord}_X(\cA) \to \cA$ the
universal twisting morphism of
Corollary~\ref{cor:three-bijections}. We restrict attention to the
genus-zero fiber: all statements below are made over a single smooth
quasi-projective genus-zero curve (e.g.\ $X = \mathbb{A}^1$ or
$\mathbb{P}^1$), so modular-family curvature $\kappa \cdot \omega_g$
is absent and the bar differential is flat.

\begin{definition}[The seven conditions]\label{def:seven-conditions-g0}
We label the following conditions on $\cA$ at genus zero.
\begin{enumerate}[label=\textup{(\roman*)}]
\item\label{seven:koszul}
\emph{Chiral Koszul morphism}: $\tau_{\mathrm{univ}}$ is a chiral Koszul
morphism in the sense of Definition~\ref{def:chiral-koszul-morphism}.
\item\label{seven:counit-qi}
\emph{Counit quasi-isomorphism}: the counit
$\varepsilon_{\tau_{\mathrm{univ}}} \colon \Omega_X(\bar B^{\ord}_X(\cA))
\to \cA$ of the bar--cobar adjunction is a quasi-isomorphism.
\item\label{seven:unit-we}
\emph{Unit weak equivalence}: the unit
$\eta_{\tau_{\mathrm{univ}}} \colon \bar B^{\ord}_X(\cA) \to
\bar B^{\ord}_X(\Omega_X(\bar B^{\ord}_X(\cA)))$ is a weak equivalence
of conilpotent complete factorisation coalgebras.  Under
Corollary~\ref{cor:three-bijections}, this is the unit corresponding
to the single universal twisting morphism
$\tau_{\mathrm{univ}}$; it is not a universal weak-equivalence
claim for every coalgebra $\cC$ mapping to $\cA$.
\item\label{seven:ttacyclic}
\emph{Twisted tensor acyclicity}: both twisted tensor products
$K_\tau^L(\cA, \bar B^{\ord}_X(\cA))$ and
$K_\tau^R(\bar B^{\ord}_X(\cA), \cA)$ are acyclic.
\item\label{seven:bar-conc-wt1}
\emph{Diagonal bar concentration}: the bar cohomology
$H^{p,q}(\bar B^{\ord}_X(\cA))$ vanishes for $q \neq 0$, and
$H^{p,0}$ is the $p$th bar-weight piece for every $p \geq 0$. In
particular,
$H^*(\bar B^{\ord}_X(\cA)) \cong \cA^\mathrm{i}$ as a weight-graded
chiral coalgebra, with the weight grading coinciding with bar degree
\textup{(}Theorem~\ref{thm:bar-concentration}\textup{)}.  The
historical phrase ``bar concentrated in weight~$1$'' means this
diagonal support condition; it does not assert that
$\cA^{\mathrm i}$ is cofree on $H^{1,0}$ outside the free/Gaussian
cases.
\item\label{seven:pbw-e2}
\emph{PBW $E_2$-collapse} (the hub): the PBW spectral sequence of
$\bar B^{\ord}_X(\cA)$ collapses at the $E_2$ page, i.e.\
$d_r = 0$ for all $r \geq 2$.
\item\label{seven:sc-formal}
\emph{SC-formality at genus zero}: the transferred Swiss-cheese
operation tower on $\cA$ has $m_k^{\mathrm{SC}} = 0$ for all
$k \geq 3$ (as in
Proposition~\ref{prop:sc-formal-iff-class-g}).
\end{enumerate}
\end{definition}

\begin{remark}[Object firewall]
\label{rem:ftm-object-firewall}
\index{object firewall!FTM seven-fold theorem}
Five objects remain separate throughout the theorem:
\[
\cA,\qquad
\barB^{\ord}_X(\cA),\qquad
\cA^{\mathrm i}=H^\bullet\barB^{\ord}_X(\cA),\qquad
\cA^!=(\cA^{\mathrm i})^\vee,\qquad
Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)=\ChirHoch^\bullet(\cA,\cA).
\]
The counit
$\Omega_X\barB^{\ord}_X(\cA)\to\cA$ is bar--cobar inversion.
The passage from $\cA^{\mathrm i}$ to $\cA^!$ is continuous linear
or Verdier duality on the finite-type lane.  The derived centre is
the chiral Hochschild complex.  Theorem~C's scalar complementarity
and Theorem~H's Hochschild amplitude are therefore not additional
spokes of this TFAE: they are consequences only after the scalar,
perfectness, and $E_\infty$-completion hypotheses of
Theorem~\ref{thm:derived-centre-complementarity-strengthened} and
Theorem~\ref{thm:main-koszul-hoch}.
\end{remark}

\begin{remark}[Why these seven, not some other seven]
\label{rem:seven-not-other-seven}
Other Koszulness characterisations exist.  Theorem~\ref{thm:koszul-equivalences-meta}
has an intrinsic equivalence core, a Barr--Beck--Lurie monadic
consequence, a conditional factorisation-homology criterion, a
one-way chiral Hochschild amplitude consequence, and conditional
perfectness/Lagrangian refinements.  The seven conditions of
Definition~\ref{def:seven-conditions-g0} are the FTM core of
Loday--Vallette \cite[\S2.3]{LV12}, the diagonal bar concentration
forced by Theorem~\ref{thm:bar-concentration}, and the
class-$G$ Swiss-cheese formality condition of
Proposition~\ref{prop:sc-formal-iff-class-g}.  No Hochschild
amplitude, scalar complementarity statement, or perfectness theorem is
silently promoted to a spoke.
\end{remark}

\section{The hub: PBW $E_2$-collapse}
\label{sec:ftm-hub}

The hub is condition~\ref{seven:pbw-e2}: PBW $E_2$-collapse. This
choice is forced by three considerations.

First, among the seven conditions, \ref{seven:pbw-e2} is the one for
which the forward implication from each other condition admits a
spectral-sequence proof at fixed level. Every other pairing (e.g.\
counit qi $\Leftrightarrow$ diagonal bar concentration) factors
through the PBW filtration.

Second, Theorem~\ref{thm:koszul-equivalences-meta} identifies
PBW $E_2$-collapse with chiral Koszulness inside the intrinsic
finite-type core, while
Theorem~\ref{thm:pbw-koszulness-criterion} supplies the
filtered-complete proof of the associated-graded-to-filtered
direction.  Every FTM arrow then routes through this primary
equivalence.

Third, the Vol~I landscape census
(\texttt{chapters/examples/landscape\_census.tex}) verifies
\ref{seven:pbw-e2} concretely for every named family (Heisenberg,
affine Kac--Moody away from the exceptional set, Virasoro at generic
$c$, principal $\cW_N$ at generic $\Psi$, $\beta\gamma$, lattice VOA,
bc).  The verification is case-by-case PBW finite-type verification,
not an inference from Swiss-cheese formality.

\begin{theorem}[Six-fold TFAE on the Koszul locus, with a class-$G$
seventh equivalence; \ClaimStatusConditional]
\label{thm:ftm-seven-fold-tfae-via-hub-spoke}
\index{FTM TFAE!six-fold on Koszul locus plus class-G seventh|textbf}
Let $(\cA,F_\bullet)$ lie in the genus-zero finite-type PBW window of
Definition~\ref{def:ftm-g0-pbw-window}.
\begin{enumerate}[label=\textup{(\Alph*)}]
\item \textup{(Six-fold intrinsic TFAE.)} The six
conditions
\ref{seven:koszul}, \ref{seven:counit-qi}, \ref{seven:unit-we},
\ref{seven:ttacyclic}, \ref{seven:bar-conc-wt1}, \ref{seven:pbw-e2}
are equivalent. The equivalence is organised as a hub-and-spoke
diagram with \ref{seven:pbw-e2} (PBW $E_2$-collapse) as the hub, and
with five independent bidirections
\[
\ref{seven:koszul} \,\Leftrightarrow\, \ref{seven:pbw-e2},
\quad
\ref{seven:counit-qi} \,\Leftrightarrow\, \ref{seven:pbw-e2},
\quad
\ref{seven:unit-we} \,\Leftrightarrow\, \ref{seven:pbw-e2},
\quad
\ref{seven:ttacyclic} \,\Leftrightarrow\, \ref{seven:pbw-e2},
\quad
\ref{seven:bar-conc-wt1} \,\Leftrightarrow\, \ref{seven:pbw-e2}.
\]
Transitivity then furnishes the remaining $\binom{5}{2} = 10$
bidirections of the six-fold equivalence.
\item \textup{(Class-$G$ seventh equivalence.)} On the standard
landscape, the seventh condition satisfies
\[
\ref{seven:sc-formal}
\quad\Longleftrightarrow\quad
\bigl(\ref{seven:pbw-e2}\ \text{and}\ \cA\in\mathsf{G}\bigr).
\]
After restricting the parameter space to the class-$G$ stratum
(Heisenberg, lattice VOA, free fermion), this becomes an honest
seventh equivalence with the six intrinsic conditions.  For generic
class-$L$, class-$C$, and class-$M$ members in the finite-type window,
the six intrinsic conditions hold but \ref{seven:sc-formal} fails
\textup{(Remark~\ref{rem:sc-formal-parametrised-scope})}.
\end{enumerate}
\end{theorem}

\begin{proof}
The five intrinsic spokes to the PBW $E_2$-collapse hub are proved
below, one section each, inside
Definition~\ref{def:ftm-g0-pbw-window}.  Transitivity of
``if and only if'' assembles the six-fold TFAE.  The class-$G$
seventh statement is Corollary~\ref{cor:ftm-seventh-class-G}.
\end{proof}

\begin{corollary}[Seventh (SC-formality) equivalence on class~$G$;
\ClaimStatusConditional]
\label{cor:ftm-seventh-class-G}
On the standard landscape,
\[
\ref{seven:sc-formal}
\quad\Longleftrightarrow\quad
\bigl(\ref{seven:pbw-e2}\ \text{and}\ \cA\in\mathsf{G}\bigr).
\]
Consequently, after restriction to the class-$G$ stratum, condition
\ref{seven:sc-formal} is equivalent to \ref{seven:pbw-e2}.
\end{corollary}

\begin{proof}
This is Proposition~\ref{prop:ftm-spoke-sc-pbw}.
\end{proof}

\section{Spoke 1: Koszul \texorpdfstring{$\Leftrightarrow$}{iff} PBW
$E_2$-collapse}
\label{sec:ftm-spoke-1}

\begin{proposition}[Spoke 1; \ClaimStatusProvedHere]
\label{prop:ftm-spoke-koszul-pbw}
Condition~\ref{seven:koszul} is equivalent to
condition~\ref{seven:pbw-e2}.
\end{proposition}

\begin{proof}
This is the intrinsic core
\textup{(i)}$\Leftrightarrow$\textup{(ii)} of
Theorem~\ref{thm:koszul-equivalences-meta}, restricted to the
genus-zero finite-type PBW window.  The forward direction is
Koszulness of $\tau_{\mathrm{univ}}$ forcing degeneration of the
PBW bar spectral sequence.  The reverse direction is the
filtered-complete Priddy transfer: PBW $E_2$-collapse identifies the
associated-graded Koszul complex with a classically Koszul datum, and
Lemma~\ref{lem:filtered-comparison} lifts the associated-graded
acyclicity to the filtered chiral twisted tensor products.
\end{proof}

\section{Spoke 2: counit qi \texorpdfstring{$\Leftrightarrow$}{iff}
PBW $E_2$-collapse}
\label{sec:ftm-spoke-2}

\begin{proposition}[Spoke 2; \ClaimStatusProvedHere]
\label{prop:ftm-spoke-counit-pbw}
Condition~\ref{seven:counit-qi} is equivalent to
condition~\ref{seven:pbw-e2}.
\end{proposition}

\begin{proof}
Both directions route through Spoke 1 combined with the classical FTM
bidirection
\ref{seven:koszul} $\Leftrightarrow$ \ref{seven:counit-qi} of
Theorem~\ref{thm:fundamental-twisting-morphisms}\ref{ftm:koszul}%
--\ref{ftm:counit}. Explicitly:
\ref{seven:pbw-e2} $\stackrel{\mathrm{Spoke\,1}}{\Rightarrow}$
\ref{seven:koszul} $\stackrel{\mathrm{FTM}}{\Rightarrow}$
\ref{seven:counit-qi}, and
\ref{seven:counit-qi} $\stackrel{\mathrm{FTM}}{\Rightarrow}$
\ref{seven:koszul} $\stackrel{\mathrm{Spoke\,1}}{\Rightarrow}$
\ref{seven:pbw-e2}. Both composite routes use only
Theorem~\ref{thm:fundamental-twisting-morphisms} and
Theorem~\ref{thm:pbw-koszulness-criterion}, which are independently
proved. The reverse direction is genuine content, not tautology,
because \ref{seven:counit-qi} is a statement about a single specific
map ($\Omega_X \bar B^{\ord}_X \cA \to \cA$) while \ref{seven:pbw-e2}
is a filtration-level statement about all differentials $d_r$
simultaneously.
\end{proof}

\section{Spoke 3: unit weak equivalence
\texorpdfstring{$\Leftrightarrow$}{iff} PBW $E_2$-collapse}
\label{sec:ftm-spoke-3}

\begin{proposition}[Spoke 3; \ClaimStatusProvedHere]
\label{prop:ftm-spoke-unit-pbw}
Condition~\ref{seven:unit-we} is equivalent to
condition~\ref{seven:pbw-e2}.
\end{proposition}

\begin{proof}
For the single universal twisting datum
$(\cA,\barB^{\ord}_X(\cA),\tau_{\mathrm{univ}},F_\bullet)$,
Theorem~\ref{thm:fundamental-twisting-morphisms}
\ref{ftm:counit}--\ref{ftm:unit} identifies the counit
quasi-isomorphism and the unit weak equivalence.  The triangle
identities of the bar--cobar adjunction supply the comparison of the
two canonical maps; no assertion about all maps
$\cC\to\barB^{\ord}_X(\cA)$ is used.  Composing with Spoke 2 gives
\ref{seven:unit-we} $\Leftrightarrow$ \ref{seven:counit-qi}
$\Leftrightarrow$ \ref{seven:pbw-e2}.
\end{proof}

\section{Spoke 4: twisted tensor acyclicity
\texorpdfstring{$\Leftrightarrow$}{iff} PBW $E_2$-collapse}
\label{sec:ftm-spoke-4}

\begin{proposition}[Spoke 4; \ClaimStatusConditional]
\label{prop:ftm-spoke-ttacyclic-pbw}
Condition~\ref{seven:ttacyclic} is equivalent to
condition~\ref{seven:pbw-e2}. This spoke is the
non-tautological backbone of the hub-and-spoke TFAE (see
Proposition~\ref{prop:no-tautology-at-g0}).
\end{proposition}

\begin{proof}
\emph{Forward} \ref{seven:ttacyclic} $\Rightarrow$
\ref{seven:pbw-e2}.
By Lemma~\ref{lem:twisted-product-cone-counit}, the left-twisted
tensor product $K_\tau^L \simeq \mathrm{Cone}(\varepsilon_\tau)[-1]$.
Acyclicity of $K_\tau^L$ forces $\varepsilon_\tau$ to be a
quasi-isomorphism, which is condition~\ref{seven:counit-qi}; Spoke 2
then gives~\ref{seven:pbw-e2}.

\emph{Reverse} \ref{seven:pbw-e2} $\Rightarrow$ \ref{seven:ttacyclic}.
This direction contains the filtered step.  $E_2$-collapse on
$\bar B^{\ord}_X(\cA)$ means every transferred differential
$d_r = 0$ for $r \geq 2$, hence the associated graded
$\mathrm{gr}_F \bar B^{\ord}_X(\cA)$ is the classical bar complex of
$\mathrm{gr}_F \cA$.  Inside
Definition~\ref{def:ftm-g0-pbw-window}, the PBW recognition circuit
identifies the associated-graded twisting datum as classically
Koszul.  Both classical twisted
tensor products $K_{\bar\tau}^L$ and $K_{\bar\tau}^R$ on the
associated-graded side are acyclic by Priddy 1970
\cite[Proposition~2.1]{Priddy1970}. The strong-convergence hypothesis
built into the twisting datum
(Definition~\ref{def:chiral-twisting-datum}) then lifts this
acyclicity from the associated-graded to the filtered complexes
$K_\tau^L(\cA,\barB^{\ord}_X(\cA))$ and
$K_\tau^R(\barB^{\ord}_X(\cA),\cA)$; the lift is
Lemma~\ref{lem:filtered-comparison}. This uses genuine filtered
spectral-sequence input, not merely the abstract Koszul bidirection,
and is therefore the spoke where non-tautology is visible.
\end{proof}

\section{Spoke 5: diagonal bar concentration
\texorpdfstring{$\Leftrightarrow$}{iff} PBW $E_2$-collapse}
\label{sec:ftm-spoke-5}

\begin{proposition}[Spoke 5; \ClaimStatusProvedHere]
\label{prop:ftm-spoke-bar-conc-pbw}
Condition~\ref{seven:bar-conc-wt1} is equivalent to
condition~\ref{seven:pbw-e2}.
\end{proposition}

\begin{proof}
\emph{Forward} \ref{seven:bar-conc-wt1} $\Rightarrow$
\ref{seven:pbw-e2}: diagonal bar concentration makes the bar
resolution compute a diagonal Ext algebra.  The
Ext-diagonal criterion
(Theorem~\ref{thm:ext-diagonal-vanishing}) then forces the PBW
differentials $d_r$ for $r \geq 2$ to vanish: such a differential
has bidegree $(r,1-r)$ and would create an off-diagonal Ext class in
the strongly convergent spectral sequence.

\emph{Reverse} \ref{seven:pbw-e2} $\Rightarrow$
\ref{seven:bar-conc-wt1}: Theorem~\ref{thm:bar-concentration} states
that for a chiral Koszul pair, bar cohomology is concentrated in
bar-differential degree~$0$ with
$H^{p,0}\cong(\cA^{\mathrm i})_p$. The
bar-degree grading coincides with the weight grading at genus zero
because the bar differential preserves weight, and the bar-degree
grading is the weight grading of the Koszul dual (see
Corollary~\ref{cor:bar-cohomology-koszul-dual}). Spoke 1 turns
\ref{seven:pbw-e2} into Koszulness of the universal twisting morphism,
which is the input of Theorem~\ref{thm:bar-concentration}.

This is a statement about the coalgebra
$\cA^{\mathrm i}=H^\bullet\barB^{\ord}_X(\cA)$, not about the
Verdier-dual algebra $\cA^!$ unless the finite-type duality
hypothesis of Definition~\ref{def:ftm-g0-pbw-window}\textup{(W5)}
is present.  For non-abelian examples such as affine Kac--Moody,
the $E_1$ exterior coalgebra is cut by the Chevalley--Eilenberg
$d_1$ differential; the surviving $\cA^{\mathrm i}$ is not the free
exterior coalgebra on the weight-$1$ generators
\textup{(}Remark~\ref{rem:ce-vs-exterior}\textup{)}.
\end{proof}

\section{Spoke 6: SC-formality
\texorpdfstring{$\Leftrightarrow$}{iff} class G, and the PBW hub}
\label{sec:ftm-spoke-6}

SC-formality characterises the Gaussian locus (class $G$), not the
whole Koszul locus.  The correct unparametrised statement is
\[
\ref{seven:sc-formal}
\quad\Longleftrightarrow\quad
\bigl(\ref{seven:pbw-e2}\ \text{and}\ \cA\in\mathsf{G}\bigr)
\]
inside the standard landscape.  The displayed equivalence reduces to
\ref{seven:sc-formal} $\Leftrightarrow$ \ref{seven:pbw-e2} only after
the parameter space has already been restricted to class~$G$.

\begin{proposition}[Spoke 6, parametrised; \ClaimStatusConditional]
\label{prop:ftm-spoke-sc-pbw}
For $\cA$ in the standard landscape and in the genus-zero finite-type
PBW window,
\[
\ref{seven:sc-formal}
\quad\Longleftrightarrow\quad
\bigl(\ref{seven:pbw-e2}\ \text{and}\ \cA\in\mathsf{G}\bigr).
\]
After restriction to class~$G$ (Heisenberg, lattice VOA, free
fermion), this is the bidirection
\ref{seven:sc-formal} $\Leftrightarrow$ \ref{seven:pbw-e2}.
\end{proposition}

\begin{proof}
\emph{Forward.}  Proposition~\ref{prop:sc-formal-iff-class-g} gives
SC-formality $\Rightarrow \cA\in\mathsf{G}$.  The class-$G$ families
are genus-zero PBW-Koszul in the finite-type window, so Spoke~1 gives
\ref{seven:pbw-e2}.

\emph{Reverse.}  If \ref{seven:pbw-e2} holds and
$\cA\in\mathsf{G}$, then
Proposition~\ref{prop:sc-formal-iff-class-g} gives
SC-formality.  The class condition is part of the premise; PBW
$E_2$-collapse alone does not imply SC-formality.
\end{proof}

\begin{remark}[Parametrisation is forced]
\label{rem:sc-formal-parametrised-scope}
The \emph{unparametrised} bidirection \ref{seven:sc-formal}
$\Leftrightarrow$ \ref{seven:pbw-e2} is false.  Generic class-$L$,
class-$C$, and class-$M$ algebras in the standard landscape satisfy
PBW $E_2$-collapse in the finite-type window, but they are not
SC-formal
(Proposition~\ref{prop:swiss-cheese-nonformality-by-class}).  The
seven-fold TFAE is therefore a theorem on the class-$G$ stratum; on
the larger Koszul locus the first six conditions remain equivalent,
and the seventh cuts out the Gaussian subclass.
\end{remark}

\section{Non-tautology theorem}
\label{sec:ftm-sevenfold-nontautology}

\begin{proposition}[Non-tautology at genus zero; \ClaimStatusConditional]
\label{prop:no-tautology-at-g0}
\index{FTM seven-fold TFAE!non-tautology}
The hub-and-spoke bidirection
\ref{seven:ttacyclic} $\Leftrightarrow$ \ref{seven:pbw-e2}
(Spoke~4) is not tautological: the direction
\ref{seven:ttacyclic} $\Rightarrow$ \ref{seven:pbw-e2} requires the
cone identification
$K_\tau^L \simeq \mathrm{Cone}(\varepsilon_\tau)[-1]$, a
chain-level computation not implied by definition unfolding, and the
direction
\ref{seven:pbw-e2} $\Rightarrow$ \ref{seven:ttacyclic} requires the
filtered-comparison argument transferring associated-graded
acyclicity to filtered acyclicity. Consequently the six-fold
TFAE of clause~(A) of
Theorem~\ref{thm:ftm-seven-fold-tfae-via-hub-spoke} is genuine
content: deleting Spoke~4's filtered-comparison lemma breaks the
chain.
\end{proposition}

\begin{proof}
The forward direction of Spoke~4 invokes
Lemma~\ref{lem:twisted-product-cone-counit}, whose proof computes the
cone of the bar--cobar counit and identifies it with the suspension
shift of the left twisted tensor product. Without this lemma the
implication from acyclicity of $K_\tau^L$ to quasi-isomorphism of
$\varepsilon_\tau$ is not available at the chain level; it is the
cone identification that upgrades ``acyclic'' to
``quasi-isomorphism''. The cone identification is a chain-level
calculation, not a definition.

The reverse direction of Spoke~4 uses
Lemma~\ref{lem:filtered-comparison} to transfer associated-graded
acyclicity to filtered acyclicity. The transfer requires exhaustive,
complete, and bounded-below filtration (standard hypotheses on the
twisting datum), and strong convergence of the associated spectral
sequence. Without the exhaustion + completion + bounded-below package,
classical Koszulity of $\mathrm{gr}_F \cA$ does not imply chiral
Koszulity of $\cA$: associated-graded acyclicity can coexist with
persistence of the Koszul dual in the $F^\infty$ residue. The
filtered-comparison lemma is therefore load-bearing, not routine.

Since a broken Spoke~4 disconnects \ref{seven:ttacyclic} from the
rest of the diagram (Spokes 1--3, 5 do not supply an independent
route from acyclicity to $E_2$-collapse), the full six-fold TFAE
would fail. Hence the theorem is not tautological.
\end{proof}

\begin{remark}[Combinatorics of the hub]
\label{rem:ftm-hub-combinatorics}
In a hub-and-spoke diagram with hub $H$ and spokes
$S_1,\dots,S_5$, the statement
``$H,S_1,\dots,S_5$ are TFAE'' is proved from five independent
bidirections $S_i\Leftrightarrow H$; the
$\binom{6}{2}-5=10$ remaining bidirections are consequences of
transitivity.  Adding the class-$G$-scoped SC-formality spoke adds
one more independent bidirection only on the class-$G$ stratum.
Thus the proof obligations are five intrinsic spokes plus one
parametrised spoke, not $21$ direct arrows.  Spoke~4 is the visible
non-formal step because it uses both the cone computation and the
filtered-comparison lemma.
\end{remark}

\section{Higher-genus survival of the spokes}
\label{sec:ftm-higher-genus}

The seven conditions of
Definition~\ref{def:seven-conditions-g0} behave differently under
lift to genus $g \geq 1$.  The FTM arrows live on completed curved
twisting data; the PBW spoke requires a named all-genera PBW theorem;
diagonal bar concentration is a separate curved-bar assertion; and
the scalar formula $\mathrm{obs}_g=\kappa(\cA)\lambda_g$ is available
only on the scalar lane of Definition~\ref{def:scalar-lane} or at
the explicitly stated genus-$1$ exception.

\begin{corollary}[Higher-genus survival, with scalar-lane scope;
\ClaimStatusConditional]
\label{cor:TFAE-extends-to-genus-1-uniform-weight}
\index{FTM seven-fold TFAE!genus extension}
Fix $g\geq 1$.  Let $\cA$ be a standard-landscape chiral algebra
whose completed genus-$g$ bar datum lies on the higher-genus Koszul
locus of Theorem~\ref{thm:higher-genus-inversion}.  Assume the
relevant all-genera PBW theorem is available for the family
\textup{(}for example
Theorems~\ref{thm:pbw-allgenera-km},
\ref{thm:pbw-allgenera-virasoro}, and
\ref{thm:pbw-allgenera-principal-w}\textup{)}.  Then:
\begin{itemize}
\item[\textup{(a)}] The FTM equivalence among Koszulness, counit
quasi-isomorphism, unit weak equivalence, and twisted tensor
acyclicity persists for the completed genus-$g$ twisting datum.  The
PBW hub remains equivalent to this FTM core precisely for the families
covered by the named all-genera PBW theorem.
\item[\textup{(b)}] Diagonal bar concentration is not controlled by
the scalar number $\kappa(\cA)$ alone.  It persists at genus~$g$ only
when the curved PBW spectral sequence remains diagonal after the
genus-$g$ correction.  On the scalar lane, the obstruction class is
the scalar projection
$\mathrm{obs}_g(\cA)=\kappa(\cA)\lambda_g$
\textup{(}Proposition~\ref{prop:scalar-obstruction-hodge-euler},
Theorem~\ref{thm:genus-universality}\textup{)}; off the scalar lane,
multi-weight cross-channel terms of
Definition~\ref{def:scalar-lane} intervene.  Thus the genus-zero
Spoke~5 is not promoted to a higher-genus equivalence without a
separate diagonal-concentration theorem.
\item[\textup{(c)}] The SC-formality spoke remains class-$G$-scoped.
For Gaussian families with all-genus Wick factorisation, such as the
Heisenberg/lattice/free-fermion witnesses, the higher operations
remain zero
\textup{(}Corollary~\ref{cor:lattice-postnikov-termination}\textup{)}.
For non-Gaussian classes, PBW survival does not imply
SC-formality.
\end{itemize}
\end{corollary}

\begin{proof}
(a) The completed genus-$g$ bar object is the ambient object of
Theorem~\ref{thm:higher-genus-inversion}; on that locus the
bar--cobar counit is a quasi-isomorphism and the FTM proof of
Theorem~\ref{thm:fundamental-twisting-morphisms} applies to the
completed curved twisting datum.  The PBW condition is not automatic
from the word ``genus''.  It is supplied by the family-specific
all-genera PBW results named in the statement, where the genus
correction preserves the relevant filtered comparison.

(b) The scalar obstruction formula is a projection of the universal
curved bar datum, not the whole curved spectral sequence.  On the
scalar lane, Proposition~\ref{prop:scalar-obstruction-hodge-euler}
and Theorem~\ref{thm:genus-universality} identify the diagonal scalar
obstruction with $\kappa(\cA)\lambda_g$.  Definition~\ref{def:scalar-lane}
records the failure of this one-variable description for multi-weight
algebras at $g\geq 2$, where cross-channel corrections appear.  Hence
bar diagonal concentration at higher genus requires a direct
curved-PBW diagonal theorem; it is not equivalent to the vanishing of
a scalar $\kappa$.

(c) Proposition~\ref{prop:sc-formal-iff-class-g} is genus-zero
Swiss-cheese formality.  For Gaussian free-field families, Wick
factorisation gives the same vanishing of primitive higher operations
after genus lift; the lattice case is recorded in
Corollary~\ref{cor:lattice-postnikov-termination}.  The non-Gaussian
classes retain their first nonzero higher Swiss-cheese operation by
Proposition~\ref{prop:swiss-cheese-nonformality-by-class}, so PBW
survival gives no reverse implication to SC-formality.
\end{proof}

\begin{remark}[The SC-formality spoke at class $M$]
\label{rem:spoke-6-class-M}
Class $M$ (Virasoro, $\cW_N$) has $S_r \neq 0$ for all $r \geq 3$,
hence $m_k^{\mathrm{SC}} \neq 0$ for all $k \geq 3$. Spoke 6
therefore rules out class $M$ from the SC-formality $\Leftrightarrow$
PBW bidirection at genus $0$. The residual Koszulness remains via
Spokes 1--5: class-$M$ algebras are genus-$0$ Koszul even though the
SC-formality spoke excludes them.
\end{remark}

\section{Relation to the Koszul meta-theorem}
\label{sec:ftm-relation-meta}

Theorem~\ref{thm:koszul-equivalences-meta} has four layers.  Its
intrinsic equivalence core contains Koszulness, PBW $E_2$-collapse,
minimal $A_\infty$ formality, Ext diagonal vanishing, and the
bar--cobar counit.  Its Barr--Beck--Lurie statement is a monadic
consequence.  Its factorisation-homology criterion is conditional on
the detecting comparison hypotheses.  Its chiral Hochschild amplitude
$[0,2]$ is a one-way consequence on the Koszul locus, and its
Lagrangian/perfectness and $\cD$-module purity clauses have their own
conditional or one-way scopes.

The seven-fold TFAE of the present chapter is the genus-zero
sub-diagram on
$\{\,\text{(i)},\,\text{(ii)},\,\text{(v)},\,\text{(iii)-hub-image},\,
\text{diagonal bar concentration},\,\text{tt-acyclic},\,
\text{SC-formal}\,\}$
restricted to genus zero, with hub (ii) and spokes~1--6 as assigned
above.

\begin{remark}[What's new relative to the meta-theorem]
\label{rem:ftm-sevenfold-delta}
The meta-theorem gives the PBW--Koszul--counit core.  The present
chapter adds diagonal bar concentration as an explicit spoke by using
Theorem~\ref{thm:bar-concentration} and
Theorem~\ref{thm:ext-diagonal-vanishing}.  It adds SC-formality only
as a parametrised spoke: Proposition~\ref{prop:sc-formal-iff-class-g}
turns SC-formality into class-$G$ membership, not into an intrinsic
property of the whole Koszul locus.
\end{remark}

\begin{remark}[Why we do not assert $21$-arrow independence]
\label{rem:no-21-independence}
Asserting $21$ independent bidirections would require proving, for
example, ``SC-formality $\Leftrightarrow$ counit qi'' without routing
through PBW $E_2$-collapse and without imposing class~$G$.  No such
direct theorem is used here.  The proof obligation is five intrinsic
bidirections with the hub, plus the class-$G$ parametrised spoke.
\end{remark}

\section{Non-tautology witness: class $L$}
\label{sec:ftm-counterexample}

To make non-tautology at the structural level operational, we exhibit
an algebra where Spoke~4 (twisted tensor acyclicity $\Leftrightarrow$
PBW $E_2$-collapse) is non-trivial.

\begin{proposition}[Class-$L$ witness; \ClaimStatusConditional]
\label{prop:class-L-witness}
For the affine Kac--Moody vertex algebra $V_k(\mathfrak{sl}_2)$ at a
generic level $k\notin\Sigma(\mathfrak{sl}_2)$
\textup{(}in particular away from the critical level
$k=-h^\vee=-2$\textup{)}, at genus zero:
\begin{itemize}
\item[\textup{(i)}] PBW $E_2$-collapse holds, by the Kac--Shapovalov
non-degeneracy criterion
(Theorem~\ref{thm:kac-shapovalov-koszulness}) at generic $k$.
\item[\textup{(ii)}] Twisted tensor acyclicity of
$K_{\tau_{\mathrm{univ}}}^L$ and $K_{\tau_{\mathrm{univ}}}^R$ holds,
following from the filtered-comparison argument of Spoke~4.
\item[\textup{(iii)}] The filtered-comparison lemma is load-bearing:
naive associated-graded computation gives acyclicity of the
classical Koszul complex of $\mathrm{gr}_F V_k(\mathfrak{sl}_2) =
\Sym(\mathfrak{sl}_2\otimes t^{-1}\bC[t^{-1}])$, but the filtered
lift to the chiral twisted tensor product on $X = \mathbb{P}^1$
requires the Kashiwara/conformal-weight completeness hypothesis on
the vacuum module (see
Remark~\ref{rem:kac-moody-filtered-comparison}).  Without this
finite-type completion, the associated-graded calculation does not
prove acyclicity of the filtered chiral twisted tensor product.
\end{itemize}
In particular, the bidirection of Spoke~4 contains genuine
filtration-level content for affine Kac--Moody, confirming
Proposition~\ref{prop:no-tautology-at-g0}.
\end{proposition}

\begin{proof}
(i) is Theorem~\ref{thm:kac-shapovalov-koszulness}
(Kac--Shapovalov determinant non-vanishing at
$k\notin\Sigma(\mathfrak{sl}_2)$ implies PBW-Koszulness, hence
$E_2$-collapse). (ii) follows from Spoke~4 of
Theorem~\ref{thm:ftm-seven-fold-tfae-via-hub-spoke}. (iii) observes
that the passage from associated-graded Koszulity to filtered
Koszulity in the Kac--Moody case requires the Kashiwara filtration
identification used in the proof of
Proposition~\ref{prop:d-module-purity-km-equivalence}.  Without that
identification, the proof has only associated-graded acyclicity, not
filtered twisted-tensor acyclicity.
\end{proof}

\begin{remark}[Kashiwara filtration on the Kac--Moody vacuum module;
\ClaimStatusConditional]
\label{rem:kac-moody-filtered-comparison}
\index{Kashiwara filtration!on vacuum module}
\index{filtered comparison!non-tautology at the chiral level}
The filtration step in clause~(iii) of
Proposition~\ref{prop:class-L-witness} is the following.  For
$\cA=V_k(\mathfrak{sl}_2)$ at
$k\notin\Sigma(\mathfrak{sl}_2)$, the
Kashiwara filtration $F^{\mathrm{Kash}}_\bullet V_k(\mathfrak{sl}_2)$
on the vacuum module is the filtration by $L_0$-eigenspaces (the
conformal-weight filtration), equivalently the PBW filtration on
$U(\widehat{\mathfrak{sl}_2})$ restricted to the vacuum
Verma-quotient. The associated graded
$\mathrm{gr}^F V_k(\mathfrak{sl}_2)$ is the symmetric algebra
$\Sym\bigl(\mathfrak{sl}_2\otimes t^{-1}\bC[t^{-1}]\bigr)$, which
is Koszul classically.
The filtered-comparison lemma
(Lemma~\ref{lem:filtered-comparison}) demands that the filtration
be exhaustive, complete, and bounded below, and that the spectral
sequence of the bar complex converges strongly. Exhaustion and
bounded-below are immediate from the grading. \emph{Completeness is
the non-routine step at the chiral level}: it asks that
$V_k(\mathfrak{sl}_2)\cong
\varprojlim_n V_k(\mathfrak{sl}_2)/F^{\mathrm{Kash}}_{\geq n}
V_k(\mathfrak{sl}_2)$ as a chiral module, not merely as a vector
space. The verification proceeds through the Kac--Kazhdan determinant
formula: at generic $k$ the Shapovalov form on each conformal-weight
component is non-degenerate, so the Verma multiplicity in
$\mathrm{gr}^F V_k(\mathfrak{sl}_2)$ is finite in every degree and
matches the PBW estimate. This \emph{finiteness-at-each-weight} is
what closes the inverse limit at the chiral level.

The non-tautology is visible by contrast with nearby cases outside
the generic proof:
\begin{itemize}
\item[(a)] at the \emph{critical} level $k=-h^\vee=-2$, the
Shapovalov form degenerates (the Feigin--Frenkel centre enters
$\mathrm{gr}^F$), Kac--Kazhdan singular vectors at every bounded
weight invalidate completeness, and the spectral sequence acquires
non-trivial higher differentials.  Spoke~4's reverse direction is
not available at $k=-h^\vee$;
\item[(b)] for non-generic simple quotients and logarithmic
quotients, the finite-type and strong-convergence hypotheses must be
verified separately before Spoke~4 can be invoked.
\end{itemize}
Thus ``classical associated-graded Koszulness implies filtered chiral
Koszulness'' is a non-vacuous statement whose verification at
$V_k(\mathfrak{sl}_2)$ requires the Kac--Kazhdan-generic
multiplicity estimate, and is outside the theorem's scope off the
generic locus until an equivalent finite-type completion is supplied. The
filtered-comparison lemma is therefore load-bearing for the
$V_k(\mathfrak{sl}_2)$
witness; the step is not supplied by definition unfolding. Primary
sources: Kac--Kazhdan 1979 (determinant formula for
$\widehat{\mathfrak{sl}_2}$), Kac 1998 Chapter~11 (PBW on vacuum
Verma), Frenkel--Ben-Zvi 2004 \S5.3 (chiral completeness of vacuum
modules at generic level).
\end{remark}

\begin{remark}[Failure-mode under relaxed hypotheses]
\label{rem:ftm-sevenfold-failure-mode}
If the PBW filtration is not exhaustive, complete, and bounded below, the
spectral sequence of the bar complex may fail to converge strongly.
In such cases: $E_2$-collapse can hold (locally in each bar degree)
while twisted tensor acyclicity fails (the $F^\infty$-residue carries
unkilled cohomology).  Critical affine level, non-generic simple
quotients, and logarithmic $\cW(p)$ examples lie outside
Definition~\ref{def:ftm-g0-pbw-window} until the relevant completed
finite-type filtered model is proved.  The hub-and-spoke theorem uses
that window as a hypothesis, not as an automatic consequence of
having a PBW filtration.
\end{remark}

\section{Compact theorem}
\label{sec:ftm-compact-theorem}

The theorem is a finite-type genus-zero hub-and-spoke diagram with
hub \ref{seven:pbw-e2} (PBW $E_2$-collapse).  The intrinsic theorem
has five spokes and six equivalent conditions.  The seventh spoke is
SC-formality, which is equivalent to the hub only on the class-$G$
stratum, equivalently on the shadow-truncation locus
$S_r=0$ for $r\geq 3$.  Diagonal bar concentration is a coalgebra
statement about $\cA^{\mathrm i}$; scalar complementarity for
$\cA^!$ and Hochschild amplitude for
$Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)$ are separate theorem packages.

In one sentence:
\begin{quote}
\textit{Genus-0 finite-type FTM theorem: hub $=$ PBW
$E_2$-collapse; Spokes $=$ Koszul morphism, counit qi, unit weak eq,
twisted tensor acyclic, diagonal bar concentration, and
SC-formality with the class-$G$ parameter; non-tautology is visible at
Spoke 4 via the cone computation and the filtered-comparison lemma
\textup{(Lemmas~\ref{lem:twisted-product-cone-counit},
\ref{lem:filtered-comparison})}. Higher genus requires the completed
Koszul locus, a named all-genera PBW theorem, and scalar-lane
hypotheses for any $\kappa\lambda_g$ statement.}
\end{quote}
