\chapter{Non-abelian Poincaré duality and the construction of Koszul dual cooperads}
\label{chap:NAP-koszul-derivation}

\index{Poincare duality!chiral|textbf}
\index{Verdier duality!bar construction is dual}

The MC element $\Theta_\cA$ lives in a convolution algebra built
from the bar coalgebra $\barB(\cA)$ and the algebra~$\cA$ itself.
Its Verdier dual produces a second algebra: the homotopy Koszul dual
$\cA^!_\infty := \mathbb{D}_{\Ran}\barB(\cA)$, which is the
factorization algebra through which the complementarity theorem
(Theorem~C) and the modular characteristic (Theorem~D) operate.
The MC equation for~$\Theta_\cA$ splits genus-$g$ obstructions
into $\cA$-visible and $\cA^!$-visible halves; this chapter
constructs the $\cA^!$-side of that decomposition.

Concretely, $\barB(\cA)$ is a factorization coalgebra on $\Ran(X)$.
Verdier duality
$\mathbb{D}_{\Ran} \colon
D(\mathcal{D}\text{-mod}(\Ran(X)))
\to D(\mathcal{D}\text{-mod}(\Ran(X)))$
exchanges $!$-tensor and $*$-tensor, hence factorization
coalgebras and factorization algebras.  Define
\begin{equation}\label{eq:homotopy-koszul-dual-def}
\cA^!_\infty \;:=\; \mathbb{D}_{\Ran}\barB(\cA).
\end{equation}
This is the \emph{homotopy Koszul dual}, unconditionally
a factorization algebra on $\Ran(X)$, carrying an
$A_\infty$-structure whose operations are:
\begin{center}
\begin{tabular}{r@{\;$\longleftrightarrow$\;}l}
bar coproduct $\Delta$ & chiral product $\mu$ of $\cA^!_\infty$, \\
bar differential $d_{\barB}$ & $A_\infty$-differential of $\cA^!_\infty$, \\
logarithmic forms on $\overline{C}_n(X)$ & distributional
  currents on $C_n(X)$.
\end{tabular}
\end{center}
No quadratic presentation, no Koszul hypothesis, no
classical $R^\perp$ is invoked.

Three regimes organize the theory:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{On the Koszul locus}, $\cA^!_\infty$ is
  formal: the bar cohomology $H^*(\barB(\cA))$ is
  concentrated in bar degree~$1$, the transferred
  $A_\infty$-structure has $m_k = 0$ for $k \geq 3$,
  and $\cA^!_\infty \simeq \cA^!$ as chiral algebras.
\item \emph{Off the Koszul locus}, $\cA^!_\infty$
  carries genuine higher operations.
  For $A = k[x]/(x^n)$: $m_{n-1}(a,\ldots,a) = b$ is
  the unique nontrivial Massey product
  (Example~\ref{ex:truncated-polynomial-ainfty}).
\item \emph{At higher genus}, the bar differential
  acquires curvature $(\dfib)^2 = \kappa(\cA)\cdot\omega_g$
  (Chapter~\ref{chap:higher-genus}), and $\cA^!_\infty$
  becomes a curved $A_\infty$-algebra.  The modular
  convolution algebra $\Convinf(\barB(\cA), \cA)$ is the
  natural home: its Maurer--Cartan element $\Theta_\cA$
  (Theorem~\ref{thm:mc2-bar-intrinsic}) encodes the
  full Koszul duality data at all genera, with the Verdier
  pairing of logarithmic forms against distributional
  currents as its chain-level realization.
\end{enumerate}

The classical objects are cohomological shadows:
\begin{center}
\begin{tabular}{r@{\;:\quad}l}
$\bar{B}(\cA)$ & factorization coalgebra, the
  \emph{bar side}, \\
$\cA^!_\infty = \mathbb{D}_{\Ran}\barB(\cA)$ &
  factorization algebra, the \emph{homotopy dual}, \\
$\cA^! = (H^*(\barB(\cA)))^\vee$ & the \emph{classical
  dual}, the cohomological shadow of $\cA^!_\infty$.
\end{tabular}
\end{center}
The bar construction does not \emph{compute} the dual;
its Verdier dual \emph{is} the dual.  The classical
$\cA^!_{\mathrm{coal}} = H^*(\barB(\cA))$ is visible
only when $\cA$ is Koszul; in general, $\barB(\cA)$
retains cohomology in all bar degrees, and the correct
object is $\cA^!_\infty$.\label{rem:poincare-duality-hms}

\section{The circularity problem}

\begin{problem}[Independent construction of Koszul dual]\label{prob:independent-dual}
The statement $\bar{B}^{\text{ch}}(\mathcal{A}_1) \simeq \mathcal{A}_2^!$ is circular as it stands: the dual $\mathcal{A}_2^!$ is ``defined'' as the Koszul dual cooperad, but no independent construction of it as a chiral coalgebra exists, and for non-quadratic algebras the classical criterion $R_1 \perp R_2$ does not apply.

Four things are needed: an intrinsic definition of $\mathcal{A}_2^!$ using only $\mathcal{A}_2$; a proof that $\mathcal{A}_2^!$ is inherently a coalgebra; a geometric identification of $\bar{B}^{\text{ch}}(\mathcal{A}_1)$ with this dual; and an extension to non-quadratic and curved cases.
\end{problem}

\begin{remark}[NAP duality]\label{rem:NAP-solution}
Non-abelian Poincar\'e duality resolves this circularity.  For a factorization algebra $\mathcal{A}$ on a manifold $M$:
\[\int_M \mathcal{A} \xleftrightarrow{\text{NAP}} \mathbb{D}\left(\int_{-M} \mathcal{A}^!\right)\]
where $\int_M$ is factorization homology, $\mathbb{D}$ is Verdier duality, $-M$ is $M$ with opposite orientation, and $\mathcal{A}^!$ is the Koszul dual defined via this duality.

\emph{The construction \textup{(Kontsevich)}.} Factorization homology is computed as a colimit over Weiss covers:
\[\int_M \mathcal{A} = \operatorname{colim}_{\coprod_i D_i \hookrightarrow M} \bigotimes_i \mathcal{A}(D_i)\]
where the colimit ranges over collections of disjoint disk embeddings $\coprod_i D_i \hookrightarrow M$ (not over arbitrary configuration spaces; cf.\ the excision property for Weiss covers, Definition~\ref{def:weiss-cover}).

Verdier duality exchanges integration over $\overline{C}_n(M)$ with distributions on $C_n(M)$, logarithmic forms with delta functions, and residues at collisions with singularity insertions.  Construction~\ref{const:A-dual-intrinsic} makes this precise.
\end{remark}

\section{Logarithmic forms and distributional currents}
\label{sec:geometric-arena}
\index{collision geometry|textbf}

Configuration spaces carry a natural duality that resolves the circularity.  On $\overline{C}_k(X)$, two dual sheaf complexes coexist; one builds the bar construction, the other the cobar.

The sheaf $\Omega^*_{\log}(\overline{C}_k(X))$ consists of
differential forms with at worst logarithmic poles along the
boundary divisors $D_I = \{z_i = z_j : i, j \in I\}$.  The
propagator $\eta_{ij} = d\log(z_i - z_j)$ is the simplest
section: it has a simple pole on $D_{ij}$ and is smooth elsewhere.
The wedge products $\bigwedge_{e \in E(T)} \eta_e$, indexed by edges
of planar trees, are the integrands of the bar construction.

The sheaf $\mathcal{D}^*_{\mathrm{dist}}(C_k(X))$ consists of
distributional currents on the open configuration space.  The delta
current $\delta_{ij} = \delta(z_i - z_j)\, dz_i \wedge d\bar{z}_j$,
supported on the collision locus $D_{ij}$, is the simplest section.
The distributional insertions $\prod_{e \in E(T)} \delta_e$ are the
integrands of the cobar construction.

Verdier duality exchanges these two sheaves: there is a canonical
perfect pairing
\begin{equation}\label{eq:log-dist-pairing-pd}
\langle \cdot, \cdot \rangle \colon
\Omega^*_{\log}(\overline{C}_k(X))
\otimes
\mathcal{D}^*_{\mathrm{dist}}(C_k(X))
\longrightarrow \mathbb{C}
\end{equation}
given by $\langle \omega, K \rangle = \int_{\overline{C}_k(X)}
\omega \wedge \iota^* K$, which identifies residue extraction (bar)
with singularity insertion (cobar) as Poincar\'e--Lefschetz dual
operations on $(\overline{C}_k(X), D)$.

\begin{example}[Two points]
For $k = 2$ on $X = \mathbb{P}^1$:
$\langle \eta_{12}, \delta_{12} \rangle
= \operatorname{Res}_{z_1 = z_2} \eta_{12} = 1$.
The OPE coefficient of the identity channel is the residue of the
propagator, paired with the delta current at collision.
\end{example}

\begin{example}[Three points and the Arnold relation]
For $k = 3$, the three boundary divisors $D_{12}$, $D_{23}$,
$D_{31}$ of $\overline{C}_3(\mathbb{P}^1)$ satisfy
$\eta_{12} \wedge \eta_{23} + \eta_{23} \wedge \eta_{31}
+ \eta_{31} \wedge \eta_{12} = 0$
(Theorem~\ref{thm:arnold-three}).  The pairing with the
distributional currents $\delta_{12} \wedge \delta_{23}$, etc.,
gives the associativity constraint: the signed sum of iterated
residues vanishes.
\end{example}

\section{Verdier duality on configuration spaces}

\begin{setup}[Configuration space duality]\label{setup:config-verdier}
Let $X$ be a smooth curve (or more generally, an $n$-dimensional manifold). The configuration space of $k$ points is:
\[C_k(X) = \{(z_1, \ldots, z_k) \in X^k : z_i \neq z_j \text{ for } i \neq j\}\]

Its Fulton--MacPherson compactification $\overline{C}_k(X)$ is a smooth manifold with corners, with boundary divisors parametrizing collision patterns.
\end{setup}

\begin{theorem}[Verdier duality for configuration spaces; \ClaimStatusProvedElsewhere{} \cite{KS90}]\label{thm:verdier-config}
\index{Verdier duality!configuration spaces}
There exists a canonical perfect pairing:
\[\langle \cdot, \cdot \rangle: \Omega^*_{\log}(\overline{C}_k(X)) \otimes \mathcal{D}^*_{\text{dist}}(C_k(X)) \to \mathbb{C}\]

given by integration:
\[\langle \omega, K \rangle = \int_{\overline{C}_k(X)} \omega \wedge \iota^* K\]

Here $\Omega^*_{\log}(\overline{C}_k(X))$ denotes differential forms with logarithmic poles on boundary divisors, $\mathcal{D}^*_{\text{dist}}(C_k(X))$ denotes distributional currents on the open configuration space, and $\iota: C_k(X) \hookrightarrow \overline{C}_k(X)$ is the inclusion.

\emph{Verdier duality.}
\[\mathbb{D}: \Omega^*_{\log}(\overline{C}_k(X)) \xrightarrow{\simeq} \mathcal{D}^*_{\text{dist}}(C_k(X))[d]\]
where $d = \dim_{\mathbb{R}}(\overline{C}_k(X))$ and the isomorphism is in the derived category $D^b_c(\overline{C}_k(X))$ (not degree-by-degree).
\end{theorem}

\begin{proof}
We apply Verdier duality for constructible
sheaves~\cite{KS90} to the specific geometry of
configuration spaces.

\emph{Step 1.}
For any constructible complex $\mathcal{F}$ on $\overline{C}_k(X)$:
\[\mathbb{D}(\mathcal{F}) = \mathcal{RHom}_{\overline{C}_k(X)}(\mathcal{F}, \omega_{\overline{C}_k(X)}[d])\]

\emph{Step 2.}
Since $\overline{C}_k(X)$ is smooth of complex dimension $k$ (for $X$ a smooth curve), the constant sheaf is self-dual up to shift:
\[\mathbb{D}(\mathbb{C}_{\overline{C}_k(X)}) \cong \mathbb{C}_{\overline{C}_k(X)}[2k]\]
For the $j_!$-extension from the open stratum $C_k(X) \hookrightarrow \overline{C}_k(X)$, Verdier duality exchanges $j_!$ and $Rj_*$:
\[\mathbb{D}(j_! \mathbb{C}_{C_k(X)}) \cong Rj_* \mathbb{C}_{C_k(X)}[2k]\]

\emph{Step 3.}
Taking hypercohomology with respect to the de Rham complex:
\begin{align*}
\mathbb{H}^*(\overline{C}_k(X), \Omega^*_{\log}) &= H^*_{\text{dR}}(\overline{C}_k(X), \log D)\\
\mathbb{H}^*_{C_k(X)}(\Omega^*_{\text{dist}}) &= H^*_{c,\text{dR}}(C_k(X))
\end{align*}

The Verdier duality pairing descends to the de Rham pairing.

\emph{Step 4.}
The explicit pairing is computed by residues:
\[\langle \omega_{\text{log}}, K_{\text{dist}} \rangle = \sum_{\text{strata } S} \int_S \text{Res}_S(\omega_{\text{log}}) \wedge K_{\text{dist}}|_S\]

Verdier duality for constructible complexes on smooth varieties~\cite{KS90} gives perfectness.
\end{proof}

\begin{remark}[Algebraic vs.\ analytic Verdier duality]\label{rem:verdier-bridge}
\index{Verdier duality!algebraic vs.\ analytic}
Theorem~\ref{thm:verdier-config} uses ``Verdier duality'' in two senses that
must be carefully distinguished:
\begin{enumerate}
\item \emph{Algebraic ($\mathcal{D}$-module) Verdier duality.}
The functor $\mathbb{D}_{\mathrm{alg}} = \mathcal{RH}om(-,
\omega_{\overline{C}_k}[\dim])$ acts on holonomic
$\mathcal{D}$-modules on $\overline{C}_k(X)$.  It exchanges
$j_!$ and $j_*$ extensions (Lemma~\ref{lem:verdier-extension-exchange}).
This is the sense used in Definition~\ref{def:geom-cobar-intrinsic} to
construct the cobar complex intrinsically.
\item \emph{Analytic (Borel--Moore) Verdier duality.}
On the real manifold underlying $C_k(X)$, Poincar\'e--Lefschetz
duality identifies $H^*(\overline{C}_k)$ with $H^{d-*}_c(C_k)$,
realized concretely by the integration pairing of logarithmic forms
against distributional currents.
\end{enumerate}
The bridge between (1) and (2) is the Riemann--Hilbert correspondence:
applying the de~Rham functor to holonomic $\mathcal{D}$-modules produces
constructible sheaves, and the algebraic duality $\mathbb{D}_{\mathrm{alg}}$
becomes the topological Verdier duality on constructible complexes
\cite{KS90}.  See Theorem~\ref{thm:cobar-distributional-model}
for the analytic realization as a consequence of the algebraic statement.
\end{remark}

\subsection{The dual operations}

\begin{theorem}[Dual differentials; \ClaimStatusProvedHere]\label{thm:dual-differentials}
Under Verdier duality, the following operations are precisely dual:

\emph{Residue vs.\ delta insertion.}
\begin{align*}
\text{Bar: } &\text{Res}_{D_{ij}}: \Omega^*_{\log}(\overline{C}_k) \to \Omega^*_{\log}(\overline{C}_{k-1})\\
\text{Cobar: } &\delta_{ij}: \mathcal{D}^*_{\text{dist}}(C_{k-1}) \to \mathcal{D}^*_{\text{dist}}(C_k)
\end{align*}

Explicitly:
\[\langle \text{Res}_{D_{ij}}(\omega), K \rangle = \langle \omega, \delta_{ij}(K) \rangle\]

\emph{Collapsing vs.\ splitting.}
\begin{align*}
\text{Bar: } &\text{Collapse at } D: C_k \dashrightarrow C_{k-1}\\
\text{Cobar: } &\text{Split along diagonal: } C_{k-1} \to C_k
\end{align*}

\emph{Composition product vs.\ coproduct.}
\begin{align*}
\text{Bar: } &\circ: \Omega^*_{\log}(\overline{C}_k) \times \Omega^*_{\log}(\overline{C}_l) \to \Omega^*_{\log}(\overline{C}_{k+l-1})\\
\text{Cobar: } &\Delta: \mathcal{D}^*_{\text{dist}}(C_{k+l-1}) \to \mathcal{D}^*_{\text{dist}}(C_k) \otimes \mathcal{D}^*_{\text{dist}}(C_l)
\end{align*}
\end{theorem}

\begin{proof}[Geometric proof via stratifications]
\emph{Observation \textup{(Serre)}.} Boundary divisors $D \subset \overline{C}_k(X)$ biject with collision patterns (combinatorial data) and with diagonal subspaces in $C_k(X)$ (geometric data).

\emph{Residue operation.}
For a form $\omega$ with logarithmic pole along $D$:
\[\omega = f \cdot \eta_{ij} \wedge \omega' + \text{regular}\]
where $\eta_{ij} = d\log(z_i - z_j)$.

The residue is:
\[\text{Res}_D(\omega) = f|_D \cdot \omega'|_D\]

\emph{Delta operation.}
For a distribution $K$ on $C_{k-1}(X)$, insertion of delta function gives:
\[\delta_{ij}(K)(z_1, \ldots, z_k) = K(z_1, \ldots, z_{i-1}, z_{i+1}, \ldots, z_k) \cdot \delta(z_i - z_j)\]

\emph{The duality.}
\begin{align*}
\langle \omega, \delta_{ij}(K) \rangle &= \int_{\overline{C}_k} (f \cdot \eta_{ij} \wedge \omega') \wedge \delta(z_i - z_j) \cdot K\\
&= \int_{D} f|_D \cdot \omega'|_D \wedge K|_D \quad \text{(delta function localizes to diagonal)}\\
&= \int_{\overline{C}_{k-1}} \text{Res}_D(f \cdot \eta_{ij} \wedge \omega') \wedge K\\
&= \langle \text{Res}_D(\omega), K \rangle
\end{align*}
\end{proof}

\begin{example}[Heisenberg specialization]\label{ex:dual-diff-heisenberg}
For the Heisenberg algebra $\cH_k$ with generator $J(z)$,
Theorem~\ref{thm:dual-differentials} reduces to:
the residue $\operatorname{Res}_{z_1=z_2} J(z_1) J(z_2)\, d\log(z_1-z_2) = k$
on the bar side is dual to the delta-function insertion
$\delta(z_1-z_2)$ on the cobar side, recovering the
level-$k$ pairing that identifies $\cH_k^! = \mathrm{Sym}^{\mathrm{ch}}(V^*)$
in \S\ref{sec:frame-koszul-dual}.
\end{example}

\begin{remark}[Analytic framework]\label{rem:analytic-framework}
The computation above is formal: the expression $\eta_{ij} \wedge \delta(z_i - z_j)$ is not well-defined as a pointwise product of distributions, since $\eta_{ij} = d\log(z_i - z_j)$ has a singularity precisely at the support of $\delta(z_i - z_j)$. The correct framework is Verdier duality as a functor on constructible sheaves (see \cite[Expos\'e XVIII]{SGA4}): the pairing is defined at the level of the derived category $D^b_c(\overline{C}_k(X))$ via the proper push-forward theorem, not by pointwise multiplication. The formula above is a shorthand for the sheaf-theoretic adjunction, and the passage from line 1 to line 2 is justified by the localization triangle for the pair $(C_k(X), D_{ij})$ rather than by literal integration of a distribution against a singular function.
\end{remark}

\section{From Verdier duality to cooperad structure}

\begin{construction}[Intrinsic definition of \texorpdfstring{$\mathcal{A}^!$}{A!} via Verdier duality]\label{const:A-dual-intrinsic}
Let $\mathcal{A}$ be a chiral algebra on $X$. We define the \emph{Koszul dual chiral coalgebra} $\cA^{\scriptstyle \text{\normalfont !`}}$ intrinsically as follows.  (The Koszul dual \emph{algebra}~$\cA^!$ is obtained by linear duality: $\cA^! = (\cA^{\scriptstyle \text{\normalfont !`}})^\vee$; cf.\ Remark~\ref{rem:four-objects}.)

\emph{Step 1: Configuration space extension.}

The factorization structure of~$\mathcal{A}$ extends to configuration spaces:
\[\mathcal{A}^{\otimes k}: C_k(X) \to \text{Vect},\qquad
(z_1, \ldots, z_k) \mapsto \mathcal{A}(z_1) \otimes \cdots \otimes \mathcal{A}(z_k).\]

\emph{Step 2: Apply Verdier duality.}

Define the \emph{dual bundle} on configuration spaces:
\[(\mathcal{A}^!)^{\boxtimes k} := \mathbb{D}\left(\mathcal{A}^{\otimes k}\right) \otimes \omega_{C_k(X)}^{-1}\]

where $\mathbb{D}$ is Verdier duality and we have included an orientation twist.

\emph{Step 3: Extract the chiral coalgebra structure.}

The factorization structure of $\mathcal{A}$ (composition of insertions) dualizes to a coproduct on $\mathcal{A}^!$ (decomposition of fields), a counit (projection onto the vacuum), and a differential (dual to the chiral product).

\emph{Explicit formulas.}

\emph{Coproduct.} For $\phi^* \in \mathcal{A}^!$,
\[\Delta(\phi^*) = \sum_{\substack{\text{collision}\\\text{patterns}}} \text{Res}_{D}(\phi^* \cdot \text{propagator}_{D})\]

where the sum is over all ways to split points into two groups.

\emph{Differential.} For $\phi^*_1 \otimes \cdots \otimes \phi^*_k \in (\mathcal{A}^!)^{\otimes k}$,
\[d(\phi^*_1 \otimes \cdots \otimes \phi^*_k) = \sum_{i<j} \langle \text{OPE}_{ij}, \phi^*_1 \otimes \cdots \otimes \phi^*_k \rangle\]

where $\text{OPE}_{ij}$ is the operator product expansion of $\mathcal{A}$.

\emph{Counit.}
\[\epsilon: \mathcal{A}^! \to \omega_X\]
\[\epsilon(\phi^*) = \langle \phi^*, \mathbb{1}_{\mathcal{A}} \rangle\]
\end{construction}

\begin{remark}\label{rem:intrinsic-nature}
Construction~\ref{const:A-dual-intrinsic} uses no bar construction, orthogonal relations, or quadraticity
hypothesis.
The Koszul dual is defined by the geometry of configuration spaces
and Verdier duality alone; the bar construction is then identified
with it (Theorem~\ref{thm:bar-computes-dual}), rather than defining it.
\end{remark}

\subsection{Verification of coalgebra axioms}

\begin{theorem}[Coalgebra structure via NAP; \ClaimStatusProvedHere]\label{thm:coalgebra-via-NAP}
The construction of $\mathcal{A}^!$ via Verdier duality yields a well-defined conilpotent chiral coalgebra:
\begin{enumerate}
\item \emph{Coassociativity:}
$(\Delta \otimes \operatorname{id}) \circ \Delta = (\operatorname{id} \otimes \Delta) \circ \Delta$.
\item \emph{Counit:}
$(\epsilon \otimes \operatorname{id}) \circ \Delta = \operatorname{id} = (\operatorname{id} \otimes \epsilon) \circ \Delta$.
\item \emph{Coderivation (graded):}
$\Delta(d(c)) = \sum d(c_{(1)}) \otimes c_{(2)} + (-1)^{|c_{(1)}|} c_{(1)} \otimes d(c_{(2)})$.
\item \emph{Conilpotency:}
For each $\phi^* \in \mathcal{A}^!$, there exists $N$ such that $\Delta^{(N)}(\phi^*) = 0$.
\end{enumerate}
\end{theorem}

\begin{proof}[Proof via geometric stratifications]
\emph{Part 1: Coassociativity.}

The coproduct $\Delta: \mathcal{A}^! \to \mathcal{A}^! \otimes \mathcal{A}^!$ arises from the geometric map:
\[C_k(X) \to \bigcup_{I \sqcup J = [k]} C_{|I|}(X) \times C_{|J|}(X)\]
that splits points into two groups.

The composition $(\Delta \otimes \text{id}) \circ \Delta$ corresponds to:
\[C_k(X) \to C_{|I|}(X) \times C_{|J|}(X) \times C_{|K|}(X)\]
for $I \sqcup J \sqcup K = [k]$.

This is manifestly independent of how we bracket the split $(I \sqcup J) \sqcup K$ vs. $I \sqcup (J \sqcup K)$, giving coassociativity.

\emph{Part 2: Counit property.}

The counit $\epsilon: \mathcal{A}^! \to \omega_X$ corresponds to the projection:
\[C_k(X) \to X\]
selecting a single point (say, the first).

The composition $(\epsilon \otimes \text{id}) \circ \Delta$ corresponds to:
\[C_k(X) \xrightarrow{\Delta} C_1(X) \times C_{k-1}(X) \xrightarrow{\epsilon \times \text{id}} X \times C_{k-1}(X) \cong C_{k-1}(X)\]

This is the identity on $C_{k-1}(X)$, verifying the counit axiom.

\emph{Part 3: Coderivation property.}

The coproduct $\Delta$ on $\mathcal{A}^!$ is the Verdier dual of the chiral product on $\mathcal{A}$:
\[\Delta: \mathcal{A}^! \to \mathcal{A}^! \otimes \mathcal{A}^!\]

Geometrically, this corresponds to:
\[d: C_k(X) \to \bigcup_{i<j} D_{ij} \times C_{k-2}(X)\]
where $D_{ij}$ is the collision divisor.

The coderivation property:
\[\Delta \circ d = (d \otimes \text{id} + \text{id} \otimes d) \circ \Delta\]

follows from the identity:
\[\text{(split then collide)} = \text{(collide then split on left)} + \text{(collide then split on right)}\]

This is the combinatorial identity for configuration space strata, which holds because boundary divisors satisfy:
\[\partial(\overline{C}_k(X)) = \bigcup_{\text{strata}} D_{\sigma}\]
with compatible orientations.

\emph{Part 4: Conilpotency.}

The chiral coalgebra $\mathcal{A}^!$ is conilpotent because it inherits a \emph{weight grading} from the conformal weight grading on $\mathcal{A}$.
Since $\mathcal{A}$ is graded by conformal weight with finite-dimensional graded pieces, $\mathcal{A}^!$ inherits a dual weight grading.  The coproduct strictly increases total weight, so the iterated coproduct $\Delta^{(N)}$ vanishes on elements of weight $\leq N$.

This can also be seen from the bar construction perspective: once Theorem~\ref{thm:bar-computes-dual} identifies $\mathcal{A}^! \simeq \bar{B}^{\mathrm{ch}}(\mathcal{A})$, conilpotency is manifest because $\bar{B}^{\mathrm{ch}}(\mathcal{A})$ is filtered by bar degree and the coproduct (deconcatenation) strictly increases the number of tensor factors, ensuring $\Delta^{(N)} = 0$ on bar degree $\leq N$.
\end{proof}

\section{\texorpdfstring{The bar construction computes $\mathcal{A}^!$}{The bar construction computes A!}}

\begin{theorem}[Bar construction = Verdier dual via NAP; \ClaimStatusProvedHere]\label{thm:bar-computes-dual}
\index{Verdier duality!bar computes dual|textbf}
\index{non-abelian Poincare duality}
\textup{[Regime: quadratic
\textup{(}Convention~\textup{\ref{conv:regime-tags})}.]}

Let $\mathcal{A}$ be an augmented chiral algebra on a smooth curve $X$,
valued in holonomic $\mathcal{D}$-modules
(Convention~\ref{subsec:ambient-category}).  Assume that the configuration-space
pushforwards $\pi_*(\mathcal{A}^{\boxtimes(n+1)} \otimes \omega^{\log}_{C_{n+1}(X)})$
are well-defined in the derived category of holonomic $\mathcal{D}$-modules on $X$.

Then there is a canonical quasi-isomorphism of chiral coalgebras:
\[\bar{B}^{\mathrm{ch}}(\mathcal{A}) \xrightarrow{\sim} \cA^{\scriptstyle \text{\normalfont !`}}\]

Here $\bar{B}^{\mathrm{ch}}(\mathcal{A})$ is the geometric bar complex (Definition~\ref{def:geometric-bar}) and $\cA^{\scriptstyle \text{\normalfont !`}}$ is the Verdier dual chiral coalgebra (Construction~\ref{const:A-dual-intrinsic}).  The Koszul dual \emph{algebra} is $\cA^! = (\cA^{\scriptstyle \text{\normalfont !`}})^\vee$.

\emph{The isomorphism.}
\[\Phi: \bar{B}^{\text{ch}}(\mathcal{A}) \to \mathcal{A}^!\]
\[\Phi(\phi_1 \otimes \cdots \otimes \phi_k \otimes \omega) = \mathbb{D}(\phi_1 \otimes \cdots \otimes \phi_k) \otimes \iota_*(\omega)\]

with $\mathbb{D}$ the Verdier duality on the factorization algebra and $\iota_*$ the direct image along $\iota: C_k(X) \hookrightarrow \overline{C}_k(X)$.
\end{theorem}

\begin{proof}
\emph{Step 1: Well-definedness of $\Phi$.}

\emph{Claim:} The map $\Phi$ is a morphism of differential graded vector spaces.

\emph{Proof of claim:} We must verify $\Phi \circ d_{\text{bar}} = d_{\mathcal{A}^!} \circ \Phi$.

The bar differential is:
\[d_{\text{bar}} = d_{\text{int}} + d_{\text{res}} + d_{\text{dR}}\]

The dual differential is:
\[d_{\mathcal{A}^!} = d_{\text{Verdier}}\]

Under Verdier duality:
\begin{align*}
\mathbb{D}(d_{\text{int}}) &= d_{\text{int}}^{\text{dual}} \quad \text{(internal differential)}\\
\mathbb{D}(d_{\text{res}}) &= d_{\delta} \quad \text{(delta function insertion)}\\
\mathbb{D}(d_{\text{dR}}) &= d_{\text{dR}}^{\text{dist}} \quad \text{(de Rham on distributions)}
\end{align*}

These sum to give $d_{\mathcal{A}^!}$, so $\Phi$ is a chain map.

\emph{Step 2: $\Phi$ preserves coalgebra structure.}

\emph{Claim:} $\Phi$ is a morphism of coalgebras, i.e., $\Delta_{\mathcal{A}^!} \circ \Phi = (\Phi \otimes \Phi) \circ \Delta_{\bar{B}}$.

\emph{Proof of claim:} The coproduct on the bar side comes from splitting configurations:
\[\Delta_{\bar{B}}: \overline{C}_k(X) \to \bigcup_{I \sqcup J} \overline{C}_{|I|}(X) \times \overline{C}_{|J|}(X)\]

Under Verdier duality, this becomes:
\[\mathbb{D}(\Delta_{\bar{B}}): \mathcal{D}^*(C_k(X)) \to \mathcal{D}^*(C_{|I|}(X)) \otimes \mathcal{D}^*(C_{|J|}(X))\]

which is precisely $\Delta_{\mathcal{A}^!}$ by construction.

\emph{Step 3: $\Phi$ is a quasi-isomorphism.}

\emph{Claim:} $\Phi$ induces an isomorphism on cohomology.

\emph{Proof of claim:} By the foundational theorem of Verdier duality (SGA 4, Exposé XVIII):
\[\mathbb{H}^*(\mathbb{D}(\mathcal{F})) \cong \mathbb{H}^{d-*}(\mathcal{F})^{\vee}\]

Applying to $\mathcal{F} = \mathcal{A}^{\otimes k}$ as a factorization algebra on configuration spaces:
\[H^*(\mathcal{A}^!) \cong H^{d-*}(\bar{B}^{\text{ch}}(\mathcal{A}))^{\vee}\]

In general, $\Phi$ is a quasi-isomorphism of complexes because Verdier duality is an exact involutive equivalence ($\mathbb{D}^2 \cong \operatorname{id}$) on the bounded derived category of constructible sheaves.  When
$\mathcal{A}$ is moreover Koszul (Definition~\ref{def:chiral-koszul-pair}),
both sides are concentrated in a single degree, and the quasi-isomorphism
reduces to a strict isomorphism of coalgebras.

\emph{Step 4: Naturality and uniqueness.}

The construction is functorial in $\mathcal{A}$ and respects all structure (products, factorization, etc.). Any other natural map $\bar{B}^{\text{ch}}(\mathcal{A}) \to \mathcal{A}^!$ must agree with $\Phi$ by uniqueness of Verdier duality.
\end{proof}

\begin{example}[Heisenberg specialization]\label{ex:bar-dual-heisenberg}
For $\cA = \cH_k$, Theorem~\ref{thm:bar-computes-dual} gives
$\bar{B}^{\mathrm{ch}}(\cH_k) \simeq \mathrm{Sym}^{\mathrm{ch}}(V^*)^!$:
the bar complex, built from logarithmic forms on $\overline{C}_n(X)$,
is quasi-isomorphic to the Verdier dual coalgebra.
Since $\cH_k$ is Koszul, both sides are concentrated in a single
degree per arity, and $H^n(\bar{B}(\cH_k)) \cong \mathbb{C}$ with
generating function $\prod(1-q^n)^{-1}$
(cf.\ \S\ref{sec:frame-inversion}).
\end{example}

\subsection{Explicit computation in low degrees}

\begin{computation}[Degree 0 and 1; \ClaimStatusProvedHere]\label{comp:bar-dual-low-degrees}
We verify the theorem explicitly in low degrees for a quadratic chiral algebra $\mathcal{A} = T_{\text{ch}}(V)/(R)$.

\emph{Degree 0.}
\[\bar{B}^0(\mathcal{A}) = \mathcal{A}\]
\[(\mathcal{A}^!)^{(0)} = \mathbb{D}(\mathcal{A}) \otimes \omega_X\]

The isomorphism is:
\[\Phi_0: \mathcal{A} \to \mathbb{D}(\mathcal{A}) \otimes \omega_X\]
\[\phi \mapsto \langle \cdot, \phi \rangle \otimes \omega_X\]

This is the canonical pairing twisted by the canonical bundle.

\emph{Degree 1.}
\[\bar{B}^1(\mathcal{A}) = \Gamma(\overline{C}_2(X), \mathcal{A}^{\boxtimes 2} \otimes \eta_{12})\]

The dual is:
\[(\mathcal{A}^!)^{(1)} = \mathcal{D}^*(C_2(X), (\mathcal{A}^!)^{\otimes 2})\]

The isomorphism is:
\[\Phi_1(\phi_1 \otimes \phi_2 \otimes \eta_{12}) = \mathbb{D}(\phi_1 \otimes \phi_2) \otimes \delta(z_1 - z_2)\]

with $\eta_{12} = d\log(z_1 - z_2)$ the logarithmic form and $\delta(z_1 - z_2)$ the delta distribution.

The pairing is given by the residue:
\[\langle \eta_{12}, \delta_{12} \rangle = \operatorname{Res}_{z_1 = z_2}\, d\log(z_1 - z_2) = 1\]

(The pointwise product $\frac{1}{z}\delta(z)$ is not defined in distribution theory;
the correct formulation uses the residue pairing between logarithmic forms and
delta currents supported on the diagonal.)  This is the fundamental Verdier pairing.

\emph{Degree 2 (first nontrivial case).}
\[\bar{B}^2(\mathcal{A}) = \Gamma(\overline{C}_3(X), \mathcal{A}^{\boxtimes 3} \otimes (\eta_{12} \wedge \eta_{23} + \text{cyc}))\]

The dual is:
\[(\mathcal{A}^!)^{(2)} = \mathcal{D}^*(C_3(X), (\mathcal{A}^!)^{\otimes 3})\]

The isomorphism involves:
\[\Phi_2(\phi_1 \otimes \phi_2 \otimes \phi_3 \otimes \eta_{12} \wedge \eta_{23}) = \mathbb{D}(\phi_1 \otimes \phi_2 \otimes \phi_3) \otimes \delta(z_1 - z_2) \delta(z_2 - z_3)\]

The Arnold relations on the bar side:
\[\eta_{12} \wedge \eta_{23} + \eta_{23} \wedge \eta_{31} + \eta_{31} \wedge \eta_{12} = 0\]

correspond, under Verdier duality, to relations among constructible sheaves
supported on the diagonal strata $\{z_i = z_j\}$.  The precise formulation
requires working with $\mathcal{D}$-modules or perverse sheaves on $C_3(X)$
rather than literal products of delta distributions (the naive products
$\delta(z_1{-}z_2)\delta(z_2{-}z_3)$, $\delta(z_2{-}z_3)\delta(z_3{-}z_1)$,
$\delta(z_3{-}z_1)\delta(z_1{-}z_2)$ are each well-defined as distributions on $\mathbb{C}^3$,
but they are all supported on the full diagonal $\{z_1=z_2=z_3\}$ and sum to
$3\delta_{123} \neq 0$, so the Arnold relation is \emph{not} preserved by naive Verdier duality
at the level of distributions).
The correct dual statement is that the Verdier dual of the Arnold relation,
viewed as a relation in $H^2(C_3(X))$, yields the corresponding relation in
the homology $H_{2n-2}$ of the Borel--Moore compactification.
\end{computation}

\section{Koszul pairs and symmetric duality}

\begin{proposition}[Chiral Koszul pair via NAP; \ClaimStatusProvedHere]\label{prop:koszul-pair-NAP}
\index{Koszul pair!via non-abelian Poincare duality}
Two chiral algebras $(\mathcal{A}_1, \mathcal{A}_2)$ on $X$ form a chiral Koszul pair (Definition~\ref{def:chiral-koszul-pair}) if and only if there exist quasi-isomorphisms of chiral coalgebras:
\begin{align}
\bar{B}^{\text{ch}}(\mathcal{A}_1) &\xrightarrow{\sim} (\mathcal{A}_2)^!\\
\bar{B}^{\text{ch}}(\mathcal{A}_2) &\xrightarrow{\sim} (\mathcal{A}_1)^!
\end{align}

where $\mathcal{A}_i^!$ is defined via Verdier duality as in Construction \ref{const:A-dual-intrinsic}.

\emph{Equivalent characterization (NAP).}
\[\int_X \mathcal{A}_1 \simeq \mathbb{D}\left(\int_{-X} \mathcal{A}_2\right)\]

with $\int_X$ the factorization homology, $-X$ the orientation reversal, and $\mathbb{D}$ Verdier duality.
\end{proposition}

\begin{theorem}[Symmetric Koszul duality; \ClaimStatusProvedHere]\label{thm:symmetric-koszul}
\textup{[Regime: quadratic on the Koszul locus
\textup{(}Convention~\textup{\ref{conv:regime-tags})}.]}

If $(\mathcal{A}_1, \mathcal{A}_2)$ is a Koszul pair, then:
\begin{align}
(\mathcal{A}_1)^! &\simeq \bar{B}^{\text{ch}}(\mathcal{A}_2) \quad \text{(bar of } \mathcal{A}_2)\\
(\mathcal{A}_2)^! &\simeq \bar{B}^{\text{ch}}(\mathcal{A}_1) \quad \text{(bar of } \mathcal{A}_1)\\
\Omega^{\text{ch}}((\mathcal{A}_1)^!) &\simeq \mathcal{A}_2 \quad \text{(cobar reconstructs } \mathcal{A}_2)\\
\Omega^{\text{ch}}((\mathcal{A}_2)^!) &\simeq \mathcal{A}_1 \quad \text{(cobar reconstructs } \mathcal{A}_1)
\end{align}

\emph{Diagram of mutual duality.}
\begin{center}
\begin{tikzcd}
\mathcal{A}_1 \arrow[r, "\bar{B}"] \arrow[d, "\Omega"', dashed] & (\mathcal{A}_2)^! \arrow[d, "\simeq"] \\
\Omega((\mathcal{A}_2)^!) \arrow[r, "\simeq"] & \bar{B}^{\text{ch}}(\mathcal{A}_2)
\end{tikzcd}
\end{center}
\end{theorem}

\begin{proof}
By Theorem~\ref{thm:bar-computes-dual}, the bar construction computes the Verdier dual: $\bar{B}^{\text{ch}}(\mathcal{A}_1) \simeq (\mathcal{A}_1)^!_{\text{Verdier}}$. For a Koszul pair $(\mathcal{A}_1, \mathcal{A}_2)$, the defining condition $\bar{B}^{\text{ch}}(\mathcal{A}_1) \simeq (\mathcal{A}_2)^!$ combined with uniqueness of Verdier duals gives $(\mathcal{A}_2)^! \simeq (\mathcal{A}_1)^!_{\text{Verdier}}$. Applying the involution property of Verdier duality ($\mathbb{D} \circ \mathbb{D} \simeq \operatorname{id}$), we obtain the symmetric statement: $\bar{B}^{\text{ch}}(\mathcal{A}_2) \simeq (\mathcal{A}_1)^!$.

The cobar reconstruction $\Omega^{\text{ch}}((\mathcal{A}_i)^!) \simeq \mathcal{A}_i$ follows from the bar-cobar adjunction (Theorem~\ref{thm:bar-cobar-adjunction}), since for Koszul algebras the counit $\Omega \circ B \to \operatorname{id}$ is a quasi-isomorphism.
\end{proof}

\begin{example}[Heisenberg specialization]\label{ex:symmetric-heisenberg}
For the Koszul pair $(\cH_k, \mathrm{Sym}^{\mathrm{ch}}(V^*))$,
Theorem~\ref{thm:symmetric-koszul} gives:
$\bar{B}^{\mathrm{ch}}(\cH_k) \simeq (\mathrm{Sym}^{\mathrm{ch}}(V^*))^!$
and $\Omega^{\mathrm{ch}}((\cH_k)^!) \simeq \mathrm{Sym}^{\mathrm{ch}}(V^*)$.
The symmetric diagram specializes to the Lie--Com exchange:
the chiral Lie structure of $\cH_k$ dualizes to the chiral commutative
structure of $\mathrm{Sym}^{\mathrm{ch}}(V^*)$, and vice versa
(\S\ref{sec:frame-koszul-dual}).
\end{example}

\begin{remark}[Surface observables via factorization homology]
\label{rem:surface-observables-fh}
\index{factorization homology!surface observables}
\index{Ayala--Francis!Poincare--Koszul duality@Poincar\'e--Koszul duality}
For a closed genus-$g$ surface $\Sigma_g$, the factorization
homology
$H_{\Sigma_g}(\cA) := \int_{\Sigma_g} \cA$
computes the space of \emph{surface observables} of the chiral
algebra~$\cA$.  The Ayala--Francis Poincar\'e/Koszul duality
theorem~\cite{AF15} gives
\[
\int_{\Sigma_g} \cA
\;\simeq\;
\mathbb{D}\!\int_{-\Sigma_g} \cA^!,
\]
which is the topological shadow of the modular bar-cobar story:
the left side is factorization homology with~$\cA$; the right
side is factorization cohomology with~$\cA^!$.  At genus~$0$,
this recovers Theorem~A.  At genus~$g \geq 1$, the surface
observables are the derived global sections of the bar family
over $\overline{\mathcal{M}}_g$: $H_{\Sigma_g}(\cA) \simeq
R\Gamma(\overline{\mathcal{M}}_g, \barB_g(\cA))$.  The
modular characteristic package extracts numerical invariants
from this derived object.
\end{remark}

\section{Non-quadratic cases and completion}

\begin{remark}[Necessity of completion]\label{rem:why-completion}
For non-quadratic chiral algebras (W-algebras, Yangians, etc.), the bar construction produces infinitely many generators in each degree:

\emph{Problem.} The bar complex is not finitely generated, so the Koszul dual $\mathcal{A}^!$ is not a finitely presented coalgebra.

\emph{Solution \textup{(Beilinson--Drinfeld)}.} Use I-adic completion:
\[\widehat{\bar{B}}(\mathcal{A}) = \varprojlim_n \bar{B}(\mathcal{A})/I^n\]
where $I = \ker(\epsilon: \bar{B}(\mathcal{A}) \to \mathbb{C})$ is the coaugmentation ideal.

\emph{Geometric interpretation.} The completion is the inverse limit
\[\widehat{\bar{B}}(\mathcal{A}) = \varprojlim_n \bar{B}(\mathcal{A})/F^n\]
over the conilpotent filtration by bar degree (see Appendix~\ref{app:nilpotent-completion}), which organizes the bar complex by the number of collision points on~$\overline{C}_n(X)$.
\end{remark}

\begin{definition}[Completed Koszul dual]\label{def:completed-dual}
For a chiral algebra $\mathcal{A}$, define the \emph{completed Koszul dual} via:
\[\widehat{\mathcal{A}^!} := \varprojlim_n (\mathcal{A}^!)/I^n\]

where $I^n$ is the $n$-th power of the conilpotent filtration.

\emph{Alternatively.} Using Verdier duality directly,
\[\widehat{\mathcal{A}^!} = \mathbb{D}(\mathcal{A}) \otimes \widehat{\Omega}^*_{\text{comp}}\]
where $\widehat{\Omega}^*_{\text{comp}}$ are completed differential forms on compactified configuration spaces.
\end{definition}

\begin{theorem}[Completion and Koszul duality; \ClaimStatusProvedHere]\label{thm:completion-koszul}
For chiral algebras satisfying:
\begin{enumerate}
\item Finite generation over $\mathcal{D}_X$
\item Polynomial growth of structure constants
\item Formal smoothness
\end{enumerate}

The completed bar construction:
\[\widehat{\bar{B}}^{\text{ch}}(\mathcal{A}) \xrightarrow{\sim} \widehat{\mathcal{A}^!}\]
is a quasi-isomorphism of completed chiral coalgebras.

The cobar construction:
\[\Omega^{\text{ch}}(\widehat{\mathcal{A}^!}) \xrightarrow{\sim} \mathcal{A}\]
recovers the original algebra.
\end{theorem}

\begin{proof}
\emph{Step 1: Verdier duality extends to completed sheaves.}
The bar complex carries a filtration $F_n \bar{B}^{\text{ch}}(\mathcal{A})$ by bar degree $\leq n$. Since $\mathcal{A}$ is finitely generated over $\mathcal{D}_X$ (hypothesis (1)), each $\bar{B}^{\text{ch}}_n(\mathcal{A})$ is a holonomic $\mathcal{D}$-module on $\overline{C}_n(X)$ (the bar construction at degree~$n$ involves $n$-fold chiral tensor products of~$\mathcal{A}$, which are holonomic by closure of holonomic $\mathcal{D}$-modules under tensor product and proper pushforward). Verdier duality commutes with filtered colimits of holonomic $\mathcal{D}$-modules and exchanges $\varprojlim$ with $\varinjlim$ by the standard adjunction:
\[\mathbb{D}: \varprojlim_n \mathcal{F}_n \xrightarrow{\sim} \varinjlim_n \mathbb{D}(\mathcal{F}_n).\]

\emph{Step 2: Completion filtration is compatible with Verdier duality.}
The coaugmentation ideal $I = \ker(\epsilon: \bar{B}^{\text{ch}} \to \mathbb{C})$ defines the $I$-adic filtration.  Each quotient $\bar{B}^{\text{ch}}/I^n$ involves only configuration spaces $C_k(X)$ with $k \leq n$, hence is geometric. The polynomial growth condition (hypothesis (2)) ensures $\dim(I^n/I^{n+1}) \leq C \cdot n^d$ for constants $C, d$, so the Mittag-Leffler condition holds for the inverse system $\{H^*(\bar{B}^{\text{ch}}/I^n)\}$.

\emph{Step 3: Lifting to completions.}
By Theorem~\ref{thm:bar-computes-dual}, we have a
quasi-isomorphism
$\Phi\colon \bar{B}^{\text{ch}}(\mathcal{A})
\xrightarrow{\sim} \mathcal{A}^!$ at each finite level.
The universal property of the inverse limit, together with
the Mittag-Leffler condition from Step~2, yields:
\[\widehat{\Phi}: \widehat{\bar{B}}^{\text{ch}}(\mathcal{A}) = \varprojlim_n \bar{B}^{\text{ch}}/I^n \xrightarrow{\sim} \varprojlim_n \mathcal{A}^!/J^n = \widehat{\mathcal{A}^!}\]
as a quasi-isomorphism of completed coalgebras.

\emph{Step~4: Cobar.}
By hypothesis~(1), each bar component is holonomic
(Step~1), so $\dim \mathrm{CH}^n(\mathcal{A}) < \infty$
(de~Rham cohomology of holonomic $\mathcal{D}$-modules is
finite-dimensional).  Hence the cobar
construction $\Omega^{\text{ch}}(\widehat{\mathcal{A}^!})$ is
well-defined and carries a complete filtration compatible with
the differential.  The bar-cobar adjunction unit
$\Omega(\bar{B}(\mathcal{A})) \to \mathcal{A}$
(Theorem~\ref{thm:bar-cobar-adjunction}) is a quasi-isomorphism
for Koszul algebras; in the completed setting, the same argument
applies since all operations are continuous with respect to the
$I$-adic topology and the Mittag-Leffler condition ensures
$\varprojlim^1 = 0$.
\end{proof}

\subsection{Application: $\mathcal{W}$-algebras}

\begin{example}[W-algebra Koszul duality via completion]\label{ex:W-completion}
For the $W_3$ algebra with generators $L(z)$ (weight 2) and $W(z)$ (weight 3):

\emph{The composite field.}
\[\Lambda = :T \cdot T: - \frac{3}{10}\partial^2 T\]
is the quasi-primary composite of weight 4 (NOT in $W_3$ itself, but appearing in $\bar{B}^{\text{ch}}(W_3)$). In the $W$-$W$ OPE, $\Lambda$ appears with coefficient $\frac{16}{22+5c}$.

\emph{Completed Koszul dual.}
\[\widehat{W_3^!} = \text{Free coalgebra}(L^*, W^*, \Lambda^*, (\Lambda^*)^{(2)}, (\Lambda^*)^{(3)}, \ldots)\]

where $(\Lambda^*)^{(n)}$ are descendant towers.

\emph{Coproduct \textup{(computed via Verdier duality)}.}
\begin{align*}
\Delta(L^*) &= 0 \quad \text{(primitive)}\\
\Delta(W^*) &= 0 \quad \text{(primitive)}\\
\Delta(\Lambda^*) &= L^* \otimes L^* - \frac{3}{10}\left(\partial^2 L^* \otimes 1 + 1 \otimes \partial^2 L^*\right) \quad \text{(composite)}
\end{align*}

\emph{The differential.}
The differential on $\Lambda^*$ encodes the $W$--$W$ OPE:
since $W(z)W(w) \sim \frac{16}{22+5c}\frac{\Lambda(w)}{(z-w)^2} + \cdots$, the bar differential acts by
\[d(\Lambda^*) = \frac{16}{22+5c}\, W^* \otimes W^*\]
with the remaining terms determined by the full $W$--$W$ OPE (see Definition~\ref{def:lambda-complete} and Computation~\ref{comp:w3-bar-degree2} for the explicit structure constants).

\emph{Cobar reconstruction.}
\[\Omega^{\text{ch}}(\widehat{W_3^!}) = W_3\]

The relation $\Lambda = :T \cdot T: - \frac{3}{10}\partial^2 T$ (with OPE coefficient $\frac{16}{22+5c}$) becomes a Maurer--Cartan element in the cobar complex.
\end{example}

\section{Resolution of the circularity}

\begin{theorem}[Resolution of circularity; \ClaimStatusProvedHere]\label{thm:main-NAP-resolution}
Construction~\ref{const:A-dual-intrinsic} defines the Koszul dual $\mathcal{A}^!$ intrinsically via Verdier duality on factorization homology, with no reference to bar constructions or orthogonal relations. Theorem~\ref{thm:bar-computes-dual} then identifies $\bar{B}^{\text{ch}}(\mathcal{A}) \xrightarrow{\sim} \mathcal{A}^!$ as dg coalgebras, and for Koszul pairs $(\mathcal{A}_1, \mathcal{A}_2)$, Theorem~\ref{thm:symmetric-koszul} gives the symmetric statement
\[\bar{B}^{\text{ch}}(\mathcal{A}_1) \simeq (\mathcal{A}_2)^! \quad \text{and} \quad \bar{B}^{\text{ch}}(\mathcal{A}_2) \simeq (\mathcal{A}_1)^!\]
derived from the NAP identity $\int_X \mathcal{A}_1 \simeq \mathbb{D}\bigl(\int_{-X} \mathcal{A}_2\bigr)$.
\end{theorem}

\begin{proof}
The intrinsic Koszul dual $\mathcal{A}^! := \mathbb{D}(\int_{-X} \mathcal{A})$ is defined in Definition~\ref{def:geom-cobar-intrinsic}.  The identification $\bar{B}^{\mathrm{ch}}(\mathcal{A}) \xrightarrow{\sim} \mathcal{A}^!$ is Theorem~\ref{thm:verdier-bar-cobar}, proved by comparing the geometric bar complex with the Verdier dual via the configuration space model (Theorem~\ref{thm:bar-NAP-homology}).  The Koszul pair statement follows from applying this to both $\mathcal{A}_1$ and $\mathcal{A}_2$ and using NAP (Theorem~\ref{thm:verdier-NAP}).
\end{proof}

The completed bar construction (Theorem~\ref{thm:completion-koszul}) extends this to non-quadratic algebras.  The quantum corrections (curvature from periods on $\overline{\mathcal{M}}_g$) appear in \S\ref{sec:koszul-defects} and Chapter~\ref{chap:higher-genus}.

\begin{remark}[Downstream dependencies]\label{rem:nap-usage}
Theorem~\ref{thm:main-NAP-resolution} supplies the intrinsic
Koszul dual for Theorem~A$_2$
(Theorem~\ref{thm:bar-cobar-isomorphism-main});
Theorem~\ref{thm:verdier-bar-cobar} supplies the Verdier
compatibility input for Theorem~C
(Theorem~\ref{thm:quantum-complementarity-main}); and the
explicit NAP computations of
Chapter~\ref{chap:NAP-computations} independently verify
Heisenberg, Kac--Moody, and $\beta\gamma$/bc Koszul pairs.
\end{remark}
