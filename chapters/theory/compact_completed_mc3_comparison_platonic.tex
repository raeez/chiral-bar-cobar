\section{Setup: two ambient DK categories, one truncation topology}
\label{sec:compact-completed-mc3-setup}

Fix a simple Lie algebra $\mathfrak{g}$ (or
$\mathfrak{g} = \mathfrak{gl}_{m|n}$, or an elliptic base
$\mathcal{E}_\rho$) and let $Y_\hbar(\mathfrak{g})^{\mathrm{ch}}$ be
the $E_1$-chiral Yangian of Chapter~\ref{chap:mc3-five-family-platonic}.
This chapter concerns the gap between two ambient categories in
which MC3 thick generation can be stated:

\begin{enumerate}[label=\textup{(A\arabic*)},nosep]
\item\label{it:ambient-ev-core}
 $\mathrm{DK}^{\mathrm{ev}}_{\mathfrak{g}}$ -- the
 \emph{evaluation-generated core}: the thick idempotent-complete
 subcategory of $\mathrm{DK}_{\mathfrak{g}}$ generated by evaluation
 modules $V_{\omega_i}(a)$ and their shifted prefundamental companions
 $L^\pm_{\omega_i,b}$ at generic spectral parameters. This is the
 domain of Theorem~\ref{thm:mc3-evaluation-core-five-family}.
\item\label{it:ambient-completed}
 $\mathrm{DK}^{\mathrm{completed}}_{\mathfrak{g}}$ -- the
 \emph{truncation-completed DK category}: the presentable stable
 $\infty$-category obtained from $\mathrm{DK}^{\mathrm{ev}}_{\mathfrak{g}}$
 by the Francis--Gaitsgory pro-nilpotent completion with respect to
 the truncation-by-conformal-weight filtration
 $\mathrm{DK}^{\mathrm{ev},\le N}_{\mathfrak{g}}$ of
 Definition~\ref{def:truncation-topology-mc3}.
\end{enumerate}

\begin{definition}[Truncation topology on $\mathrm{DK}^{\mathrm{ev}}_\mathfrak{g}$]
\label{def:truncation-topology-mc3}
\index{truncation topology!DK category}%
\index{conformal weight!truncation filtration}%
For an integer $N \ge 0$, let
$\mathrm{DK}^{\mathrm{ev},\le N}_\mathfrak{g} \subset
\mathrm{DK}^{\mathrm{ev}}_\mathfrak{g}$ be the full subcategory
spanned by modules whose $L_0$-generalized-eigenspace decomposition is
concentrated in conformal weight~$\le N$. The inclusions
$\mathrm{DK}^{\mathrm{ev},\le N}_\mathfrak{g} \hookrightarrow
\mathrm{DK}^{\mathrm{ev},\le N+1}_\mathfrak{g}$ are fully faithful and
exact; their union is $\mathrm{DK}^{\mathrm{ev}}_\mathfrak{g}$. The
truncation topology on $\mathrm{DK}^{\mathrm{ev}}_\mathfrak{g}$ is the
Grothendieck topology whose open sets are truncation-closed sieves.
\end{definition}

\begin{definition}[Completed DK category via truncation pro-limit]
\label{def:completed-dk-category}
\index{DK category!completed}%
\index{Francis-Gaitsgory!pro-nilpotent completion}%
The completed DK category is
\[
 \mathrm{DK}^{\mathrm{completed}}_\mathfrak{g}
 \;:=\;
 \lim_{N \to \infty}\bigl(
 \mathrm{DK}^{\mathrm{ev},\le N}_\mathfrak{g}
 \bigr)^{\mathrm{Ind}},
\]
the inverse limit of Ind-completions along the truncation filtration,
taken in presentable stable $\infty$-categories. Equivalently, using
the Francis--Gaitsgory pro-nilpotent formalism:
\[
 \mathrm{DK}^{\mathrm{completed}}_\mathfrak{g}
 \;\simeq\;
 \Pro_{\mathrm{nilp}}\bigl(
 \mathrm{DK}^{\mathrm{ev}}_\mathfrak{g}
 \bigr)
\]
with nilpotents the truncation ideal
$I_N := \ker(\mathrm{DK}^{\mathrm{ev}}_\mathfrak{g} \to
\mathrm{DK}^{\mathrm{ev},\le N}_\mathfrak{g})$.
\end{definition}

\begin{remark}[Why a completion is natural for the $E_1$-chiral setting]
\label{rem:why-completion-natural}
The $E_1$-chiral Yangian acts on conformal-weight-graded modules
whose formal characters are power series in $q$; generic objects
such as the prefundamentals $L^\pm_{\omega_i,b}$ and the Verma-type
modules are genuinely infinite-dimensional and have no finite
truncation in a single ambient. The truncation topology is the
minimal topology adapted to $L_0$-grading in which all relevant
Bethe-algebra eigenspace constructions converge. The completion does
not introduce new chiral-primary elements or new OPE data; it packages
the already-present pro-finite tail of conformal-weight levels.
Ambient upgrade, not scope downgrade.
\end{remark}

\begin{principle}[Ambient upgrade as the correct resolution of ]
\label{princ:ambient-upgrade-ap233}
 (compact/completed comparison) was flagged in the 732-agent
adversarial campaign as a gap between the unconditional thick
generation theorem (domain
$\mathrm{DK}^{\mathrm{ev}}_\mathfrak{g}$) and the preface claim
that MC3 governs the full DK factorization category. The LOSSLESS
resolution is not to restrict the preface claim but to prove the
stronger theorem in the larger ambient: the evaluation-generated core
is \emph{dense} in the completed category, and a thick-generation
lifting lemma (Theorem~\ref{thm:compact-completed-mc3-comparison})
promotes dense thick generation to completed thick generation. The
result is MC3 in the completed ambient, which is strictly stronger
than MC3 on the evaluation-generated core.
\end{principle}

\section{Central theorem: density plus thick-generation lifting}
\label{sec:central-compact-completed-theorem}

\begin{theorem}[Compact/completed MC3 comparison; \ClaimStatusProvedHere]
\label{thm:compact-completed-mc3-comparison}
\index{MC3!compact-completed comparison}%
\index{thick generation!lifting to completed category}%
%
Let $\mathfrak{g}$ be as in
Theorem~\ref{thm:mc3-evaluation-core-five-family}. Then:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item \emph{Density.} The inclusion
 $\mathrm{DK}^{\mathrm{ev}}_\mathfrak{g} \hookrightarrow
 \mathrm{DK}^{\mathrm{completed}}_\mathfrak{g}$ is dense: every
 object $M \in \mathrm{DK}^{\mathrm{completed}}_\mathfrak{g}$ is the
 truncation-topology limit of its evaluation-core truncations
 $\{M_N\}_{N \ge 0}$ with $M_N \in
 \mathrm{DK}^{\mathrm{ev},\le N}_\mathfrak{g}$.
\item \emph{Lifting of thick generators.} If
 $\mathcal{S} \subset \mathrm{DK}^{\mathrm{ev}}_\mathfrak{g}$ is a
 countable seed thickly generating
 $\mathrm{DK}^{\mathrm{ev}}_\mathfrak{g}$ (the evaluation-type
 modules at generic spectral parameters of
 Theorem~\ref{thm:mc3-evaluation-core-five-family}), then
 $\mathcal{S}$ thickly generates
 $\mathrm{DK}^{\mathrm{completed}}_\mathfrak{g}$ in the
 completed-category sense: the thick idempotent-complete
 subcategory of $\mathrm{DK}^{\mathrm{completed}}_\mathfrak{g}$
 generated by $\mathcal{S}$ and closed under the truncation-topology
 limit equals $\mathrm{DK}^{\mathrm{completed}}_\mathfrak{g}$.
\item \emph{Scope.} The two statements hold for every family covered
 by Theorem~\ref{thm:mc3-evaluation-core-five-family}: rational
 type~$A$, twisted $B/C/D$, uniform simple $\mathfrak{g}$, elliptic
 $\mathfrak{sl}_N$, super $\mathfrak{gl}_{m|n}$.
\end{enumerate}
Consequently, MC3 upgrades from ``thick generation of the
evaluation-generated core'' to ``thick generation of the completed
DK factorization category'' under the truncation topology of
Definition~\ref{def:truncation-topology-mc3}.
\end{theorem}

\begin{proof}
Part~(i) is Proposition~\ref{prop:compact-approximation-exists}.
Part~(ii) is Theorem~\ref{thm:mc3-full-DK-in-completed-category},
which is an application of the classical thick-generation lifting
lemma to the dense inclusion of~(i) together with the existence of
the countable seed
$\mathcal{S}$ from the five-family
Theorem~\ref{thm:mc3-evaluation-core-five-family}. Part~(iii) is
uniform across the five families because both the truncation
filtration (by $L_0$-weight) and the density argument (via
Verma-filtration and PBW, see
Proposition~\ref{prop:compact-approximation-exists}) apply to each
family without modification: every evaluation-type module has a
well-defined $L_0$-grading, and every prefundamental is constructed
as a pro-limit of KR modules that is automatically
truncation-convergent.
\end{proof}

\section{Compact approximation: every completed object is a truncation limit}
\label{sec:compact-approximation}

\begin{proposition}[Compact approximation exists; \ClaimStatusProvedHere]
\label{prop:compact-approximation-exists}
\index{compact approximation!completed DK category}%
\index{Verma filtration!PBW}%
\index{truncation limit!completed module}%
Let $M \in \mathrm{DK}^{\mathrm{completed}}_\mathfrak{g}$. Then there
exists a tower
$\cdots \twoheadrightarrow M_{N+1} \twoheadrightarrow M_N
\twoheadrightarrow \cdots$ with $M_N \in
\mathrm{DK}^{\mathrm{ev},\le N}_\mathfrak{g}$ and an isomorphism in
$\mathrm{DK}^{\mathrm{completed}}_\mathfrak{g}$,
\[
 M \;\simeq\; \lim_{N \to \infty} M_N,
\]
the limit being taken in the truncation topology of
Definition~\ref{def:truncation-topology-mc3}. Each $M_N$ is
equivalent to a finite extension of objects in
$\mathcal{S} \subset \mathrm{DK}^{\mathrm{ev}}_\mathfrak{g}$.
\end{proposition}

\begin{proof}
Step~1 (Verma filtration). The action of $Y_\hbar(\mathfrak{g})^{\mathrm{ch}}$
on $M$ preserves $L_0$-generalized-eigenspaces, so $M$ inherits a
weight-graded decomposition
$M = \bigoplus_h M_h$. The universal Verma filtration of
$Y_\hbar(\mathfrak{g})^{\mathrm{ch}}$ (Kac--Kazhdan for affine level
$k$, analogous for Yangian via PBW) gives, on each $M_h$, a
highest-weight filtration whose composition factors are evaluation-type
simples at generic spectral parameters by
Theorem~\ref{thm:mc3-evaluation-core-five-family}.

Step~2 (PBW compact approximation). Let $M_N := M_{\le N} / \mathrm{nil}^{\ge N+1}$
where $M_{\le N} := \bigoplus_{h \le N} M_h$ and
$\mathrm{nil}^{\ge N+1}$ is the nilpotent ideal of operators of
$L_0$-weight shift $\ge N+1$ restricted to $M_{\le N}$. By the PBW
theorem for $Y_\hbar(\mathfrak{g})^{\mathrm{ch}}$ (with respect to
the weight-grading; see
Proposition~\ref{prop:mc3-uniform-chari-moura-platonic}) the
composition factors of $M_N$ lie in evaluation-type simples of total
$L_0$-weight $\le N$, so $M_N \in
\mathrm{DK}^{\mathrm{ev},\le N}_\mathfrak{g}$.

Step~3 (truncation limit). The canonical surjections $M_{N+1}
\twoheadrightarrow M_N$ are induced from the weight-truncation
functors and form a tower whose inverse limit in
$\mathrm{DK}^{\mathrm{completed}}_\mathfrak{g}$ recovers $M$ up to
quasi-isomorphism because the truncation topology is precisely the
nilpotent topology of
Definition~\ref{def:completed-dk-category} and $M$ is complete by
construction in
$\mathrm{DK}^{\mathrm{completed}}_\mathfrak{g}$. Finally, each $M_N$
has a finite composition series by evaluation-type simples (the
$L_0 \le N$ eigenspaces of a chiral module of bounded Bethe data are
finite-dimensional in each Cartan weight), so $M_N$ is a finite
extension of objects in the evaluation-generated core.
\end{proof}

\begin{remark}[Where PBW is used: not for the five-family mechanisms themselves]
\label{rem:pbw-usage}
The PBW argument in Step~2 is classical for the Yangian and independent
of the five-family mechanisms; it is the standard Drinfeld--Olshanski
PBW for $Y_\hbar(\mathfrak{g})$ in its loop presentation, which
descends to $Y_\hbar(\mathfrak{g})^{\mathrm{ch}}$ via the Gautam--Toledano
Laredo equivalence. Compact approximation is therefore a statement of
algebraic structure, not of thick generation.
\end{remark}

\begin{remark}[Comparison with the Francis--Gaitsgory formalism]
\label{rem:francis-gaitsgory-comparison}
In Francis--Gaitsgory formalism, the truncation ideals $I_N$ of
Definition~\ref{def:completed-dk-category} generate the pro-nilpotent
completion. Proposition~\ref{prop:compact-approximation-exists}
states that the completed objects are precisely the $I_N$-adically
complete objects, which is the standard conclusion of the
Francis--Gaitsgory completion theorem in the presentable stable
$\infty$-categorical framework.
\end{remark}

\section{Lifting thick generation to the completed category}
\label{sec:lifting-thick-generation}

\begin{lemma}[Dense-core thick-generation lifting; \ClaimStatusProvedElsewhere]
\label{lem:dense-thick-generation-lifting}
\index{thick generation!dense core!lifting lemma}%
Let $\mathcal{A} \hookrightarrow \mathcal{B}$ be a dense inclusion of
presentable stable $\infty$-categories (every $B \in \mathcal{B}$ is a
filtered colimit of compact objects in $\mathcal{A}$) such that
compact objects of $\mathcal{B}$ lie in the essential image of
$\mathcal{A}$. If $\mathcal{S} \subset \mathcal{A}$ thickly
idempotent-completely generates $\mathcal{A}$, then $\mathcal{S}$
thickly idempotent-completely generates $\mathcal{B}$.
\end{lemma}

\begin{proof}[Proof sketch]
This is Neeman's classical thick-generation / compact-generation
comparison (see Neeman, \emph{Triangulated Categories}, Ch.~8), in
the presentable stable $\infty$-categorical form established by
Lurie, \emph{Higher Algebra}~1.4. The argument: a compact object of
$\mathcal{B}$ which is in $\mathcal{A}$ lies in $\mathrm{Thick}(\mathcal{S})$
by hypothesis; an arbitrary object is a filtered colimit of such
compacts, and thick-idempotent closure is filtered-colimit-complete
under the compact-generation assumption. The presentable enhancement
of this lifting appears in Lurie HA~1.4 and is cited here as
\ClaimStatusProvedElsewhere.
\end{proof}

\begin{theorem}[MC3 thick generation in the completed category; \ClaimStatusProvedHere]
\label{thm:mc3-full-DK-in-completed-category}
\index{MC3!completed DK category!thick generation}%
Let $\mathcal{S} \subset \mathrm{DK}^{\mathrm{ev}}_\mathfrak{g}$ be
the countable seed of evaluation-type modules at generic spectral
parameters thickly generating
$\mathrm{DK}^{\mathrm{ev}}_\mathfrak{g}$ via
Theorem~\ref{thm:mc3-evaluation-core-five-family}. Then $\mathcal{S}$
thickly idempotent-completely generates
$\mathrm{DK}^{\mathrm{completed}}_\mathfrak{g}$.
\end{theorem}

\begin{proof}
Apply Lemma~\ref{lem:dense-thick-generation-lifting} with
$\mathcal{A} = \mathrm{DK}^{\mathrm{ev}}_\mathfrak{g}$,
$\mathcal{B} = \mathrm{DK}^{\mathrm{completed}}_\mathfrak{g}$. The
density hypothesis is
Proposition~\ref{prop:compact-approximation-exists}; the compact
inclusion in $\mathcal{A}$ holds by Step~2 of that proposition (every
$M_N$ is a finite extension of objects of $\mathcal{S}$). The
thick generation of $\mathcal{A}$ by $\mathcal{S}$ is
Theorem~\ref{thm:mc3-evaluation-core-five-family}. The conclusion
is thick generation of $\mathcal{B}$ by $\mathcal{S}$.
\end{proof}

\begin{remark}[Compatibility with the Drinfeld-centre derived equivalence]
\label{rem:compatibility-drinfeld-centre}
Theorem~\ref{thm:mc3-full-DK-in-completed-category} is compatible
with the E_3 identification
of Theorem~\ref{thm:e3-identification-reductive}: the derived centre
$Z^{\mathrm{der}}_{\mathrm{ch}}(Y_\hbar(\mathfrak{g})^{\mathrm{ch}})$
inherits an E_3-algebraic structure that respects the truncation
filtration, so the factorization category of completed modules is
the natural target for the full DK equivalence
$\Phi_{\mathrm{DK}}$ in the sense of
Section~\ref{sec:mc3-platonic-full-DK-conjectural}.
\end{remark}
