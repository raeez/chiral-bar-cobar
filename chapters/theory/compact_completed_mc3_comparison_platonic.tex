\section{Finite-window DK completion}
\label{sec:compact-completed-mc3-setup}

Fix a simple Lie algebra $\mathfrak{g}$, the super series
$\mathfrak{g}=\mathfrak{gl}_{m|n}$, or an elliptic base
$\mathcal E_\rho$, and let
$Y_\hbar(\mathfrak g)^{\mathrm{ch}}$ be the corresponding
$E_1$-chiral Yangian object of
Chapter~\ref{chap:mc3-five-family-platonic}.  Write
$\mathrm{DK}^{\mathrm{ev}}_{\mathfrak g}$ for the
evaluation-generated core: the small idempotent-complete stable
subcategory generated by the evaluation modules
$V_{\omega_i}(a)$ and the prefundamental companions
$L^\pm_{\omega_i,b}$ at generic parameters.

The comparison with a completed DK category is not a comparison with
the full factorization category $\mathrm{DK}_{\mathfrak g}$.  It is a
comparison with a specified finite-window completion of
$\mathrm{DK}^{\mathrm{ev}}_{\mathfrak g}$.  A full-DK identification
requires a separate compact-object theorem.

\begin{definition}[Truncation topology on
$\mathrm{DK}^{\mathrm{ev}}_\mathfrak{g}$]
\label{def:truncation-topology-mc3}
\index{truncation topology!DK category}%
\index{finite-window topology!DK category}%
Choose a separated exhaustive finite-window parameter~$\nu$:
conformal weight, reflection-equation weight, theta-divisor height,
or the corresponding super weight.  For every $N\geq0$ let
\[
 \tau_{\le N}\colon
 \mathrm{DK}^{\mathrm{ev}}_{\mathfrak g}
 \longrightarrow
 \mathrm{DK}^{\mathrm{ev},\le N}_{\mathfrak g,\nu}
\]
be the exact idempotent-complete window quotient.  Its kernel
$I_{N+1}:=\ker(\tau_{\le N})$ consists of morphisms and objects
invisible in all $\nu$-weights $\le N$.  The quotients carry exact
transition functors
\[
 p_N\colon
 \mathrm{DK}^{\mathrm{ev},\le N+1}_{\mathfrak g,\nu}
 \longrightarrow
 \mathrm{DK}^{\mathrm{ev},\le N}_{\mathfrak g,\nu},
 \qquad
 \tau_{\le N}=p_N\tau_{\le N+1}.
\]
The truncation topology is the topology generated by the ideals
$I_{N+1}$.  A map is a truncation equivalence if all its
$\tau_{\le N}$-images are equivalences.
\end{definition}

\begin{definition}[Completed DK category via finite windows]
\label{def:completed-dk-category}
\index{DK category!completed}%
\index{Francis-Gaitsgory!pro-nilpotent completion}%
The finite-window completed DK category is the presentable stable
$\infty$-category
\[
 \mathrm{DK}^{\mathrm{completed}}_{\mathfrak g,\nu}
 :=
 \varprojlim_N
 \operatorname{Ind}
 \bigl(
 \mathrm{DK}^{\mathrm{ev},\le N}_{\mathfrak g,\nu}
 \bigr),
\]
taken along the exact functors $p_N^{\operatorname{Ind}}$.  An object
is a compatible tower
\[
 M=\{M_N,\alpha_N:p_NM_{N+1}\xrightarrow{\sim}M_N\}_{N\ge0}.
\]
The evaluation core maps to the completion by
\[
 j_\nu(E)=\{\tau_{\le N}E\}_{N\ge0}.
\]
This is the Francis--Gaitsgory pro-nilpotent completion attached to
the tower of nilpotent ideals $\{I_{N+1}\}$, in the same sense that
Appendix~\ref{app:nilpotent-completion} realizes a completed bar
coalgebra as $\varprojlim_N\bar B_X(A_{\le N})$.
\end{definition}

\begin{definition}[Finite-window MC3 tower]
\label{def:finite-window-mc3-tower}
\index{MC3!finite-window tower}%
Let $\mathcal S\subset\mathrm{DK}^{\mathrm{ev}}_{\mathfrak g}$ be
the countable seed of Theorem~\ref{thm:mc3-evaluation-core-five-family}.
The chosen finite-window completion is an \emph{MC3 finite-window
tower} if:
\begin{enumerate}[label=\textup{(FW\arabic*)},nosep]
\item each compact window is thickly generated by the seed:
 \[
 \bigl(
 \mathrm{DK}^{\mathrm{ev},\le N}_{\mathfrak g,\nu}
 \bigr)^\omega
 =
 \operatorname{Thick}^{\mathrm{idem}}
 \bigl(\tau_{\le N}\mathcal S\bigr);
 \]
\item the transition functors are exact, preserve compact objects,
 and are essentially surjective on the thick subcategories generated
 by $\tau_{\le N+1}\mathcal S$;
\item for every compatible tower of mapping spectra arising from
 compact-window objects, the homotopy groups form a
 Mittag--Leffler inverse system.  Surjective transition maps on
 finite-dimensional Hom groups are sufficient.
\end{enumerate}
In a curved CDG realization, the tower also carries the second-kind
condition: every finite-window comparison cone is coacyclic, and the
cohomology tower of cones is Mittag--Leffler.  In a square-zero total
model, replace ``coacyclic'' by ``acyclic''.
\end{definition}

\begin{definition}[Completed thick-localizing closure]
\label{def:completed-thick-closure}
\index{thick generation!completed closure}%
The completed closure of $\mathcal S$ is the full subcategory
\[
 \widehat{\operatorname{Loc}}_\nu(\mathcal S)
 \subset
 \mathrm{DK}^{\mathrm{completed}}_{\mathfrak g,\nu}
\]
consisting of compatible towers $M=\{M_N\}$ with
$M_N\in
\operatorname{Loc}_{\operatorname{Ind}(
\mathrm{DK}^{\mathrm{ev},\le N}_{\mathfrak g,\nu})}
(\tau_{\le N}\mathcal S)$ for every~$N$.  Its compact-window part
\[
 \widehat{\operatorname{Thick}}_\nu(\mathcal S)
\]
is obtained by replacing $\operatorname{Loc}$ by
$\operatorname{Thick}^{\mathrm{idem}}$ in every finite window.  Thus
``thick generation in the completed category'' means compact thick
generation in each finite window, followed by the specified
presentable Ind-completion and the inverse limit over the
truncation tower.  It does not mean that an ordinary thick subcategory
is closed under arbitrary filtered colimits.
\end{definition}

\begin{remark}[Why completion is the natural ambient]
\label{rem:why-completion-natural}
Prefundamental and Verma-type objects carry infinite positive tails in
the chosen finite-window parameter.  A single finite window records a
bounded part of this tail; the completed category records the
compatible system of all windows.  The completion adds no new
evaluation generators and no new OPE constants.  It changes the
ambient topology in which the old generators are allowed to converge.
\end{remark}

\section{The compact/completed comparison theorem}
\label{sec:central-compact-completed-theorem}

\begin{theorem}[Compact/completed MC3 comparison;
\ClaimStatusProvedHere]
\label{thm:compact-completed-mc3-comparison}
\index{MC3!compact-completed comparison}%
\index{thick generation!lifting to completed category}%
Let $\mathfrak g$ be in the range of
Theorem~\ref{thm:mc3-evaluation-core-five-family}, and let~$\nu$ be a
finite-window parameter for which
$\mathrm{DK}^{\mathrm{ev}}_{\mathfrak g}$ is an MC3 finite-window
tower in the sense of Definition~\ref{def:finite-window-mc3-tower}.
Then:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item \emph{Windowwise density.}  The functor
	 $j_\nu\colon\mathrm{DK}^{\mathrm{ev}}_{\mathfrak g}\to
	 \mathrm{DK}^{\mathrm{completed}}_{\mathfrak g,\nu}$
	 is continuous for the truncation topology, the finite-window
	 projections jointly detect equivalences, and the image of the seed
	 generates every compact window by
	 Definition~\ref{def:finite-window-mc3-tower}\textup{(FW1)}.
\item \emph{Completed generation.}  The seed~$\mathcal S$ of
 Theorem~\ref{thm:mc3-evaluation-core-five-family} generates
 $\mathrm{DK}^{\mathrm{completed}}_{\mathfrak g,\nu}$ in the
 completed sense:
 \[
 \widehat{\operatorname{Loc}}_\nu(\mathcal S)
 =
 \mathrm{DK}^{\mathrm{completed}}_{\mathfrak g,\nu},
 \qquad
 \widehat{\operatorname{Thick}}_\nu(\mathcal S)
 =
 \bigl(\mathrm{DK}^{\mathrm{completed}}_{\mathfrak g,\nu}\bigr)^\omega
 \]
 whenever compact objects in the inverse limit are detected
 windowwise.
\item \emph{Curved boundary.}  If the finite-window comparison lives
 in a curved CDG category, the completed comparison is a coacyclic-cone
 statement in the second-kind coderived category.  If a square-zero
 total model is chosen, the same proof gives an ordinary
 quasi-isomorphism.
\end{enumerate}
The theorem proves MC3 in the specified finite-window completed
ambient.  It does not identify
$\mathrm{DK}^{\mathrm{completed}}_{\mathfrak g,\nu}$ with the full
factorization category $\mathrm{DK}_{\mathfrak g}$.
\end{theorem}

\begin{proof}
For (i), an object of
$\mathrm{DK}^{\mathrm{completed}}_{\mathfrak g,\nu}$ is, by
Definition~\ref{def:completed-dk-category}, a compatible tower
$\{M_N\}$.  The projections to the $N$-th window are jointly
conservative because the completion is the inverse limit of those
windows.  The comparison functor $j_\nu$ sends an evaluation-core
object to its tower of finite-window quotients.  Definition
\ref{def:finite-window-mc3-tower}\textup{(FW1)} says that the images of
the seed thickly generate each compact window.  This proves
windowwise density; it does not assert that every compatible tower is
the truncation tower of a single evaluation-core object.

For (ii), Definition~\ref{def:finite-window-mc3-tower}(FW1) gives
compact thick generation in every window.  Passing to
$\operatorname{Ind}$ turns compact thick generation into presentable
localizing generation in each
$\operatorname{Ind}(\mathrm{DK}^{\mathrm{ev},\le N}_{\mathfrak g,\nu})$.
The transition functors preserve the generated subcategories by
(FW2).  Therefore a compatible tower belongs to
$\widehat{\operatorname{Loc}}_\nu(\mathcal S)$ precisely when each
projection does; every completed object satisfies this.  If compact
objects of the inverse limit are detected by compact window
projections, the same argument with
$\operatorname{Thick}^{\mathrm{idem}}$ gives the displayed compact
statement.

For (iii), let $K_N$ be the finite-window cone of the comparison map.
In the square-zero total model, $H^\bullet(K_N)=0$ for all~$N$.
The Milnor exact sequence
\[
0\to
\varprojlim\nolimits_N^1 H^{m-1}(K_N)
\to
H^m(\varprojlim_N K_N)
\to
\varprojlim_N H^m(K_N)
\to 0
\]
shows that the inverse-limit cone is acyclic because the right term
vanishes and the left term vanishes by (FW3).  In the curved CDG
ambient the same inverse-limit cone is interpreted in Positselski's
second-kind coderived category; the finite-window coacyclic cones
generate a coacyclic limit under the same Mittag--Leffler hypothesis.
\end{proof}

\begin{remark}[Per-family truncation filtrations]
\label{rem:per-family-truncation-filtrations}
\index{truncation filtration!per family}%
The letter~$\nu$ in
Definition~\ref{def:truncation-topology-mc3} has the following
family-specific meanings.
\begin{itemize}[nosep]
\item Rational type~$A$: conformal weight, equivalently the
 positive-energy $L_0$ window on the affine/Yangian side.
\item Uniform simple $\mathfrak g$: conformal weight for the
 Sugawara normalization in the non-critical positive-energy regime.
\item Super $\mathfrak{gl}_{m|n}$: total conformal weight with parity
 carried as an independent $\mathbb Z/2$-grading.
\item Twisted $B/C/D$: reflection-equation Cartan weight, the
 filtration used by
 Proposition~\ref{prop:mc3-type-BCD-reflection-shapovalov-platonic}.
\item Elliptic $E_{\rho,\eta}(\mathfrak{sl}_N)$:
 theta-divisor height on
 $\mathcal E_\rho^{N-1}\setminus\Theta$, the filtration used by
 Proposition~\ref{prop:mc3-elliptic-theta-divisor-platonic}.
\end{itemize}
The theorem requires the finite-window tower axioms for the chosen
filtration.  It does not replace those axioms by polynomial growth or
by a bare PBW basis.
\end{remark}

\begin{remark}[Bar-cobar inversion and Verdier--Koszul duality]
\label{rem:bar-cobar-verdier-boundary-mc3}
\index{bar-cobar inversion!not Koszul duality}%
The comparison above is a category-completion statement.  It is not
the bar-cobar counit and it is not Verdier--Koszul duality.  The
bar-cobar counit is the inversion map
\[
 \Omega_X\bar B_X(A)\longrightarrow A .
\]
The bar cohomology $H^\bullet\bar B_X(A)=A^{\mathrm i}$ is the
Koszul dual coalgebra.  The Koszul dual algebra is obtained only
after the Verdier step on the Ran space:
\[
 A^{!}:=\mathbb D_{\Ran}(A^{\mathrm i}),
\]
which becomes $(A^{\mathrm i})^\vee$ in finite-type notation.
MC3 concerns generation in a DK module category; it cannot prove a
Verdier identification by changing the completion topology.
\end{remark}

\section{Finite-window approximation and Mittag--Leffler cones}
\label{sec:compact-approximation}

\begin{proposition}[Finite-window approximation exists;
\ClaimStatusProvedHere]
\label{prop:compact-approximation-exists}
\index{compact approximation!completed DK category}%
\index{truncation limit!completed module}%
\index{Mittag--Leffler!completed cone}%
Let $M\in\mathrm{DK}^{\mathrm{completed}}_{\mathfrak g,\nu}$.
Then $M$ has a canonical tower of approximations
\[
 \cdots \longrightarrow M_{\le N+1}
 \longrightarrow M_{\le N}
 \longrightarrow \cdots,
 \qquad
 M_{\le N}:=\pi_N(M),
\]
where
$\pi_N\colon\mathrm{DK}^{\mathrm{completed}}_{\mathfrak g,\nu}\to
\operatorname{Ind}(
\mathrm{DK}^{\mathrm{ev},\le N}_{\mathfrak g,\nu})$
is the $N$-th projection.  The canonical map
\[
 M\longrightarrow \varprojlim_N M_{\le N}
\]
is an equivalence in the completed category.  If $M$ is compact and
compactness is detected by the windows, then
$M_{\le N}\in\operatorname{Thick}^{\mathrm{idem}}(\tau_{\le N}\mathcal S)$
for every~$N$.

For a compatible tower of finite-window comparison maps
$f_N\colon X_N\to Y_N$ with cones~$K_N$, if the tower
$\{H^m(K_N)\}_N$ is Mittag--Leffler for every~$m$, then:
\[
H^m(\operatorname{Cone}(\varprojlim_N f_N))
\cong
\varprojlim_N H^m(K_N).
\]
In particular, acyclic finite-window cones give an acyclic completed
cone; coacyclic finite-window cones give a coacyclic completed cone in
the second-kind coderived category.
\end{proposition}

\begin{proof}
The first assertion is the defining universal property of
Definition~\ref{def:completed-dk-category}: $M$ is the compatible
tower of its projections.  Compact-window generation follows from
Definition~\ref{def:finite-window-mc3-tower}(FW1).

For the cone statement, apply the Milnor exact sequence to the inverse
system of cones:
\[
0\to
\varprojlim\nolimits_N^1H^{m-1}(K_N)
\to
H^m(\varprojlim_NK_N)
\to
\varprojlim_NH^m(K_N)
\to0.
\]
The Mittag--Leffler hypothesis kills the derived limit term.  Exactness
of inverse limits on the chosen tower identifies
$\varprojlim_NK_N$ with the cone of $\varprojlim_N f_N$.  Positselski's
second-kind category replaces ordinary acyclicity by membership in
the coacyclic thick subcategory; that condition is preserved by the
same finite-window limit when the tower of cone cohomologies is
Mittag--Leffler.
\end{proof}

\begin{remark}[PBW supplies windows, not convergence]
\label{rem:pbw-usage}
The PBW theorem for Yangians identifies finite monomial windows and
supports the finite-window generation statements used in MC3.  It does
not by itself prove that an arbitrary inverse system is complete, nor
does it kill $\varprojlim^1$.  Completion requires the topology of
Definition~\ref{def:truncation-topology-mc3} and the
Mittag--Leffler hypothesis in
Definition~\ref{def:finite-window-mc3-tower}(FW3), exactly as in
Theorem~\ref{thm:completion-convergence}.
\end{remark}

\begin{remark}[Francis--Gaitsgory comparison]
\label{rem:francis-gaitsgory-comparison}
The ideals $I_{N+1}$ form the pro-nilpotent tower.  The completed DK
category is the category of objects complete for this tower, just as
the completed bar coalgebra of Appendix~\ref{app:nilpotent-completion}
is the inverse limit of finite quotients.  The finite-window
comparison is therefore a statement about continuous towers, not about
raw direct sums.
\end{remark}

\section{Lifting generation through completion}
\label{sec:lifting-thick-generation}

\begin{lemma}[Finite-window dense generation lifting;
\ClaimStatusProvedElsewhere]
\label{lem:dense-thick-generation-lifting}
\index{thick generation!dense core!lifting lemma}%
Let $\{\mathcal C_N,p_N\}$ be an inverse system of compactly generated
presentable stable $\infty$-categories with exact transition functors,
and let
$\mathcal C=\varprojlim_N\mathcal C_N$.  Let
$S_N\subset\mathcal C_N^\omega$ be compatible compact sets such that
$\operatorname{Loc}_{\mathcal C_N}(S_N)=\mathcal C_N$ for every~$N$.
If the transition maps satisfy the Mittag--Leffler condition on
mapping spectra for the compact towers generated by the $S_N$, then
the compatible seed generates~$\mathcal C$ under finite colimits,
retracts, windowwise filtered colimits, and the inverse limit over
the tower.  If compactness in $\mathcal C$ is detected by compactness
in every window, the compact objects of~$\mathcal C$ are generated
windowwise by $\operatorname{Thick}^{\mathrm{idem}}(S_N)$.
\end{lemma}

\begin{proof}
This is the Neeman compact-generation argument applied before taking
the inverse limit, together with the presentable stable
$\infty$-categorical form of compact generation in Lurie,
\emph{Higher Algebra}, Section~1.4.  If an object
$M\in\mathcal C$ is right-orthogonal to the compatible seed, then
each projection $\pi_NM$ is right-orthogonal to $S_N$ in
$\mathcal C_N$.  Since $S_N$ generates $\mathcal C_N$, all
$\pi_NM$ vanish.  The projections are jointly conservative for the
inverse limit, so $M=0$.  The compact assertion is the same argument
inside the compact-window subcategories, with the
Mittag--Leffler condition ensuring that mapping spectra are computed
by ordinary inverse limits rather than by a nonzero derived-limit
extension.
\end{proof}

\begin{theorem}[MC3 generation in the completed finite-window category;
\ClaimStatusProvedHere]
\label{thm:mc3-full-DK-in-completed-category}
\index{MC3!completed DK category!thick generation}%
Let $\mathcal S\subset\mathrm{DK}^{\mathrm{ev}}_{\mathfrak g}$ be
the seed of Theorem~\ref{thm:mc3-evaluation-core-five-family}.
Under the finite-window tower hypotheses of
Definition~\ref{def:finite-window-mc3-tower},
\[
 \widehat{\operatorname{Loc}}_\nu(\mathcal S)
 =
 \mathrm{DK}^{\mathrm{completed}}_{\mathfrak g,\nu}.
\]
If compact objects of the completed category are detected windowwise,
then
\[
 \widehat{\operatorname{Thick}}_\nu(\mathcal S)
 =
 \bigl(\mathrm{DK}^{\mathrm{completed}}_{\mathfrak g,\nu}\bigr)^\omega.
\]
\end{theorem}

\begin{proof}
Apply Lemma~\ref{lem:dense-thick-generation-lifting} to
\[
\mathcal C_N=
\operatorname{Ind}
\bigl(\mathrm{DK}^{\mathrm{ev},\le N}_{\mathfrak g,\nu}\bigr),
\qquad
S_N=\tau_{\le N}\mathcal S.
\]
The windowwise generation is
Definition~\ref{def:finite-window-mc3-tower}(FW1), inherited from
Theorem~\ref{thm:mc3-evaluation-core-five-family} after projection to
the chosen finite windows.  Compatibility and Mittag--Leffler control
are (FW2)--(FW3).  The conclusion is exactly the displayed equality
of completed closures.
\end{proof}

\begin{remark}[Compatibility with the derived centre]
\label{rem:compatibility-drinfeld-centre}
The theorem is compatible with the abstract $E_3$ derived-centre
identification of Theorem~\ref{thm:e3-identification}: both use the
same $E_1$-chiral Yangian input and the same evaluation seed on the
core.  This chapter does not prove that the $E_3$ structure preserves
the finite-window filtration, nor that the completed DK category is
the full DK factorization category.  Such a statement requires an
additional filtration-respect theorem or a compact-object
identification.
\end{remark}

\section{Scope of the completed statement}
\label{sec:comparison-gap-resolution}

\begin{corollary}[Compact/completed comparison inside the
finite-window ambient; \ClaimStatusProvedHere]
\label{cor:comparison-gap-resolved-completed}
\index{compact-completed comparison!MC3}%
The MC3 statement on the evaluation-generated core and the MC3
statement on the finite-window completed category are related by the
comparison functor
\[
 j_\nu\colon
 \mathrm{DK}^{\mathrm{ev}}_{\mathfrak g}
 \longrightarrow
 \mathrm{DK}^{\mathrm{completed}}_{\mathfrak g,\nu}.
\]
The first is Theorem~\ref{thm:mc3-evaluation-core-five-family}.  The
second is Theorem~\ref{thm:mc3-full-DK-in-completed-category}.  Their
bridge is finite-window density plus Mittag--Leffler lifting.
\end{corollary}

\begin{proof}
The comparison functor is Definition~\ref{def:completed-dk-category}.
Density and cone control are
Proposition~\ref{prop:compact-approximation-exists}.  Generation in
the completed ambient is
Theorem~\ref{thm:mc3-full-DK-in-completed-category}.
\end{proof}

\begin{remark}[Scope statement]
\label{rem:honest-scope-statement}
\index{MC3!scope}%
Three categories remain distinct.
\begin{itemize}[nosep]
\item The evaluation-generated core
 $\mathrm{DK}^{\mathrm{ev}}_{\mathfrak g}$ is generated by
 $\mathcal S$ under ordinary thick idempotent-complete closure.
\item The finite-window completion
 $\mathrm{DK}^{\mathrm{completed}}_{\mathfrak g,\nu}$ is generated by
 $\mathcal S$ in the completed sense of
 Definition~\ref{def:completed-thick-closure}.
\item The full factorization category $\mathrm{DK}_{\mathfrak g}$ is
 reached only after an additional identification
 $\mathrm{DK}^{\mathrm{completed}}_{\mathfrak g,\nu}\simeq
 \mathrm{DK}_{\mathfrak g}$.  That is the compact-completion problem,
 not a consequence of the formal finite-window theorem.
\end{itemize}
\end{remark}

\begin{remark}[Type~$A$ compact-object boundary]
\label{rem:type-A-completion-conjecture-resolved}
\index{Conjecture!dk-compacts-completion!type A reduction}%
For $\mathfrak g=\mathfrak{sl}_N$ at generic level, the
Kazhdan--Lusztig--Finkelberg comparison identifies the completed
finite-window side with the corresponding KL--F admissible category.
This does not by itself identify the full
$\mathrm{DK}_{\mathfrak{sl}_N}$ with the completed category.  The
remaining type-$A$ problem is the compact-object check:
every compact object of the full DK category must lie in the essential
image of the completed finite-window category.  A compact object of
$\mathrm{DK}_{\mathfrak{sl}_N}$ whose finite-window image is not in
the completed closure of the evaluation seed would falsify this
upgrade, while leaving the finite-window theorem above intact.
\end{remark}

\begin{remark}[Vol~II and Vol~III interfaces]
\label{rem:coherence-vol2-vol3-mc3-completed}
The Vol~II bar-chain models use the same type of finite-window
control: finite stages carry strict chain statements; the completed
object is obtained by a Mittag--Leffler inverse limit, with coacyclic
cones in the curved lane.  The Vol~III functor
$\Phi_d=\SpCh_{\Sigma_{d-1},C}\circ\PhiFA_d$ lands, after stage~$2$,
in an $E_1$-chiral category on the reference curve.  The theorem here
applies to that stage-$2$ shadow only after the chosen specialization
cycle supplies the same finite-window tower.  Different cycles
$\Sigma_{d-1}$ may produce different towers.
\end{remark}

\section{Failure loci}
\label{sec:falsification-surfaces-comparison-gap-completed}

\begin{remark}[Failure loci for the completed MC3 theorem]
\label{rem:falsification-plan-comparison-gap-completed}
The completed theorem can fail only through one of the finite-window
hypotheses.
\begin{enumerate}[label=\textup{(F\arabic*)},nosep]
\item \emph{Window generation failure.}  For some $N$, the compact
	 category
	 $(\mathrm{DK}^{\mathrm{ev},\le N}_{\mathfrak g,\nu})^\omega$
	 contains an object not in
	 $\operatorname{Thick}^{\mathrm{idem}}(\tau_{\le N}\mathcal S)$.
	 This violates the finite-window MC3 input.
\item \emph{Derived-limit failure.}  A compatible cone tower
 $\{K_N\}$ has nonzero
 $\varprojlim^1H^{m-1}(K_N)$ for some~$m$.  Then finite-window
 equivalences do not assemble to a completed equivalence.
\item \emph{Curved-lane failure.}  A finite-window cone is acyclic as
 a raw complex but not coacyclic in the chosen second-kind coderived
 category.  Then the curved CDG conclusion is false even if the
 square-zero total model survives.
\item \emph{Full-DK upgrade failure.}  A compact object of
	 $\mathrm{DK}_{\mathfrak g}$ is absent from the essential image of
	 $\mathrm{DK}^{\mathrm{completed}}_{\mathfrak g,\nu}$.  This violates
	 the full-DK compact-completion conjecture, not
	 Theorem~\ref{thm:compact-completed-mc3-comparison}.
\end{enumerate}
\end{remark}
