\chapter{The Universal Conductor: ordered-to-symmetric averaging}
\label{chap:universal-conductor}

\epigraph{The conductor is the map that forgets order without pretending
that the lost order-data was never there.}{}

The universal conductor of Volume~I is the averaging morphism
\[
 K_\cA \;=\; \Av_\cA \colon \gAE \longrightarrow \gAmod
\]
from the ordered convolution dg Lie algebra to the symmetric modular
quotient. At chain level it is the Reynolds projector followed by the
coinvariant quotient
\[
 B^{\Ord}(\cA)=T^c(s^{-1}\bar\cA)
 \longrightarrow
 B^{\Sigma}(\cA)=\Symm^c(s^{-1}\bar\cA),
\]
and at the factorisation level it is left Kan extension along the
quotient map
\[
 q \colon \Ranord(X) \longrightarrow \Ran(X)=\Ranord(X)/\Sigma_\bullet.
\]
This is the map that sends the ordered bar package to its modular
shadow. Bar--cobar inversion, Verdier Koszul duality, the derived
chiral centre $\DerZ(\cA)$, and the categorical Drinfeld centre
$\mathcal Z(\Rep^{E_1}(\cA))$ are different constructions. The five objects
\[
 \cA,\qquad B(\cA),\qquad \cA^i,\qquad \cA^!,\qquad \DerZ(\cA)
\]
remain distinct throughout.

On standard witness lanes three scalar packages are kept separate from
the averaging morphism:
\[
 K^c(A,A^!)=c(A)+c(A^!),\qquad
 K^\kappa(A,A^!)=\kappa(A)+\kappa(A^!),\qquad
 K_{\mathrm{ghost}}^{\mathrm{leg}}(A)
 =-c_{\mathrm{ghost}}(\BRST_{\mathrm{can}}(A)).
\]
They are central-charge, modular-characteristic, and BRST-presentation
numbers, respectively.

$K$ has named chain-level and $(\infty,1)$-categorical witnesses. Its
image is simultaneously a coinvariant quotient, a Reynolds
fixed-point summand in characteristic zero, and a homotopy-coinvariant
factorisation shadow. Its kernel contains the ordered classes that
separate the four archetypes $G/L/C/M$. The landscape map
$\mathfrak U\colon \cA\mapsto\{G,L,C,M\}$ is the supremal
$G/L/C/M$ depth of the averaged shadow, checked row-by-row against the
census. The Vol~III
comparison is restricted to the ordered Koszul locus. The scalar
packages $K^c$, $K^\kappa$, and $K_{\mathrm{ghost}}^{\mathrm{leg}}$
are family-dependent and remain separate unless a displayed
calculation identifies them.

\section{Definition of the averaging conductor}
\label{sec:uc-definition}

An ordered chiral algebra $\cA$ on $X$ is
\emph{conductor-eligible} if its ordered bar package carries a
specified $\Sigma_\bullet$-action, or a specified
$R$-twisted $\Sigma_\bullet$-descent datum, for which the ordered
differential, residue operations, and convolution bracket are
equivariant, and each arity is finite type or completed so that finite
group averaging is continuous. The displayed formulas in this chapter
use the untwisted case $R=\mathrm{id}$. In the twisted case
$\Sigma_n$-coinvariants are replaced everywhere by the corresponding
$R$-twisted coinvariants. Outside this conductor domain the aritywise
Reynolds projection is only a linear shadow map; no dg Lie morphism is
asserted.

\subsection{Chain-level witness}
\label{subsec:uc-chain-level}

\begin{definition}[Reynolds--coinvariant averaging]
\label{def:uc-reynolds-coinvariant}
Let $V=s^{-1}\bar\cA$. For each $n\geq 0$, let
\[
 \Av^{\mathrm{Rey}}_{\Sigma_n}
 \;:=\;
 \frac{1}{n!}\sum_{\sigma\in\Sigma_n}\sigma
 \colon
 V^{\otimes n}\longrightarrow V^{\otimes n}
\]
be the Reynolds projector and let
\[
 q_n \colon V^{\otimes n}\longrightarrow V^{\otimes n}_{\Sigma_n}
 \cong \Symm^n(V)
\]
be the coinvariant quotient. The \emph{chain-level universal
conductor} on the underlying arity-graded bar object is
\[
 K_{\cA}^{\mathrm{ch}}
 \;:=\;
 \bigoplus_{n\geq 0} q_n\circ \Av^{\mathrm{Rey}}_{\Sigma_n}
 \colon
 B^{\Ord}(\cA)=T^c(V)\longrightarrow B^{\Sigma}(\cA)=\Symm^c(V).
\]
When the ordered differential is $\Sigma_\bullet$-equivariant this is a
chain map to the symmetric coinvariant quotient.
\end{definition}

\begin{definition}[Ordered-to-symmetric chiral homology descent]
\label{def:uc-totalization}
For a smooth complex curve $X$, write
\[
 \int_X^{\Ord}\cA
 :=
 \Tot\!\Bigl(\bigoplus_{n\geq 0}
 C_\bullet(\Conf_n^{\Ord}(X),\cA^{\boxtimes n})\Bigr)
\]
for ordered chiral homology and
\[
 \int_X^{\mathrm{mod}}\cA
 :=
 \Tot\!\Bigl(\bigoplus_{n\geq 0}
 C_\bullet(\Conf_n(X),(\cA^{\boxtimes n})_{\Sigma_n})\Bigr)
\]
for the symmetric modular shadow. The induced descent map
\[
 K^{\mathrm{hom}}_{X,\cA}\colon
 \int_X^{\Ord}\cA\longrightarrow\int_X^{\mathrm{mod}}\cA
\]
for a conductor-eligible $\cA$ is defined on the $n$-th
totalisation summand by the same Reynolds
operator
$\frac{1}{n!}\sum_{\sigma\in\Sigma_n}\sigma$ followed by passage to
coinvariants. This is the totalisation-level realisation of the
ordered-to-symmetric descent of
\eqref{eq:ordered-descent}.
\end{definition}

\subsection{$(\infty,1)$-categorical witness}
\label{subsec:uc-infinity-level}

\begin{definition}[Homotopy-coinvariant factorisation functor]
\label{def:uc-homotopy-coinvariants}
Let
\[
 q\colon \Ranord(X)\longrightarrow \Ran(X)=\Ranord(X)/\Sigma_\bullet
\]
be the quotient map of \eqref{eq:ran-as-quotient}. For an ordered
factorisation algebra $\mathcal F$ on $\Ranord(X)$ with the descent
datum above, define
\[
 K^\infty(\mathcal F)
 \;:=\;
 q_!(\mathcal F)
 \;\simeq\;
 \mathcal F_{h\Sigma_\bullet},
\]
the left Kan extension along $q$, equivalently the homotopy
coinvariants of the symmetric-group action. Its right adjoint is the
ordered pullback $q^*$, whose fibrewise description is given by
homotopy fixed points.
\end{definition}

\begin{theorem}[\textbf{Universal conductor as ordered-to-symmetric descent}]
\label{thm:uc-universal-conductor}
\ClaimStatusProvedHere
For every conductor-eligible ordered chiral algebra $\cA$ on a smooth
complex curve $X$,
the assignment
\[
 K_\cA=\Av_\cA\colon \gAE\longrightarrow \gAmod
\]
is a well-defined natural dg Lie morphism with the following two named
witnesses.
\begin{enumerate}
\item[\textup{(i)}] \emph{Chain-level witness.}
 $K_\cA$ is dual to the Reynolds--coinvariant map
 $K_{\cA}^{\mathrm{ch}}$ of
 Definition~\ref{def:uc-reynolds-coinvariant} on
 $B^{\Ord}(\cA)\to B^{\Sigma}(\cA)$, equivalently to the
 totalisation-level descent
 $K_{X,\cA}^{\mathrm{hom}}$ of
 Definition~\ref{def:uc-totalization}.
\item[\textup{(ii)}] \emph{$(\infty,1)$-categorical witness.}
 $K_\cA$ is induced by the homotopy-coinvariant functor
 $q_!\simeq(-)_{h\Sigma_\bullet}$ of
 Definition~\ref{def:uc-homotopy-coinvariants}.
\end{enumerate}
It satisfies:
\begin{enumerate}
\item[\textup{(a)}] $K_\cA$ preserves the dg Lie differential and the
 bracket because these operations are $\Sigma_\bullet$-equivariant on
 the conductor domain.
\item[\textup{(b)}] $K_\cA$ is lossy on the nontrivial
 $\Sigma_\bullet$-isotypic ordered sector: it remembers the symmetric
 modular shadow and forgets the ordered commutator, associator, and
 higher braid data not represented by the trivial summand.
\item[\textup{(c)}] $K_\cA$ is distinct from
 $\Omega(B(\cA))\to \cA$, from the bar-dual coalgebra
 $\cA^i=H^*(B^{\mathrm{ch}}(\cA))$, from Verdier-Koszul duality
 $\cA^i\mapsto \cA^!$, from the derived centre
 $\DerZ(\cA)$, and from the categorical Drinfeld centre
 $\mathcal Z(\Rep^{E_1}(\cA))$.
\end{enumerate}
\end{theorem}

\begin{proof}
The ordered bar differential and the ordered convolution bracket are
defined by sums over splittings and collision residues. On the
conductor domain these operations are equivariant for the specified
$\Sigma_\bullet$-descent datum; in the untwisted symmetric case this is
the usual compatibility of the collision forms with relabelling,
together with the Arnold relation in the Orlik--Solomon algebra. Hence
the Reynolds projector commutes with the differential, and the quotient
$q_n\colon V^{\otimes n}\to \Symm^n(V)$ is a chain map. Without this
equivariance the same formula is only an aritywise linear projection,
not a dg Lie morphism. This proves the chain-level part.

The factorisation-theoretic part is the ordered-to-symmetric descent
already used in the ordered modular package: since
$\Ran(X)=\Ranord(X)/\Sigma_\bullet$, left Kan extension along $q$
identifies the symmetric object with fibrewise homotopy coinvariants.
Passing to the dual convolution algebras gives the dg Lie morphism
$K_\cA\colon \gAE\to \gAmod$.

Finally, the source and target objects distinguish $K$ from the other
load-bearing constructions. The bar--cobar counit
$\Omega(B(\cA))\to \cA$ is an inversion map; the bar-dual coalgebra is
$\cA^i=H^*(B^{\mathrm{ch}}(\cA))$; Verdier duality turns $\cA^i$ into
the Koszul dual algebra $\cA^!$ only under the finite-type or completed
continuous duality hypotheses; $\DerZ(\cA)$ is a bulk algebra built
from chiral Hochschild cochains; and
$\mathcal Z(\Rep^{E_1}(\cA))$ is a braided monoidal category. The
universal conductor is the ordered-to-symmetric projection.
\end{proof}

\section{Image of the conductor}
\label{sec:uc-image}

\begin{theorem}[\textbf{Three descriptions of the image}]
\label{thm:uc-trinity}
\ClaimStatusProvedHere
Let $V=s^{-1}\bar\cA$, work over characteristic zero, and restrict to
the conductor domain. For each $n\geq 0$, the image of the universal
conductor on $V^{\otimes n}$ is characterised by three independent
descriptions.
\begin{enumerate}
\item[\textup{(Q)}] \emph{Quotient description.}
 \[
 \operatorname{im}(K_{\cA,n}^{\mathrm{ch}})
 \;=\;
 V^{\otimes n}_{\Sigma_n}
 \;\cong\;
 \Symm^n(V).
 \]
 Thus the image is, by construction, the symmetric coinvariant
 quotient.
\item[\textup{(F)}] \emph{Fixed-point splitting.}
 The Reynolds projector identifies the same image with the fixed-point
 summand
 \[
 \operatorname{im}(\Av^{\mathrm{Rey}}_{\Sigma_n})
 =
 (V^{\otimes n})^{\Sigma_n}
 \cong
 \Symm^n(V).
 \]
 Thus the image may be treated as a fixed-point object only after the
 characteristic-zero Reynolds splitting.
\item[\textup{(H)}] \emph{Homotopy-coinvariant description.}
 On factorisation algebras the same image is the essential image of
 \[
 q_!\simeq(-)_{h\Sigma_\bullet}
 \colon
 \Fact(\Ranord(X))\longrightarrow \Fact(\Ran(X)).
 \]
 Hence the image is the homotopy-coinvariant shadow of the ordered
 theory.
\end{enumerate}
Consequently the answer to ``subalgebra, fixed point, or quotient?''
is precise: the image is a coinvariant quotient by definition, a
fixed-point summand after Reynolds splitting, and a sub-dg Lie algebra
of $\gAmod$ only because $\gAmod$ is already the symmetric target.
\end{theorem}

\begin{proof}
Description~(Q) is the definition of
Definition~\ref{def:uc-reynolds-coinvariant}: the target of
$K_{\cA,n}^{\mathrm{ch}}$ is the coinvariant quotient.
Description~(F) is the characteristic-zero Reynolds splitting:
averaging projects $V^{\otimes n}$ onto its $\Sigma_n$-invariant
summand, and invariants are canonically identified with symmetric
tensors. Description~(H) is the factorisation-theoretic reformulation
of the quotient map $\Ranord(X)\to \Ran(X)$ from
Definition~\ref{def:uc-homotopy-coinvariants}. The final sentence just
unpacks the three descriptions.
\end{proof}

\begin{proposition}[Schur--Weyl kernel count]
\label{prop:uc-kernel-dimension}
\ClaimStatusProvedHere
On a finite-dimensional purely even arity window with $d=\dim V$, one
has
\[
 \dim\ker(K_{\cA,n}^{\mathrm{ch}})
 \;=\;
 d^n-\binom{n+d-1}{d-1}.
\]
Equivalently, the image has dimension $\binom{n+d-1}{d-1}$.
For a graded or super window this binomial coefficient is replaced by
the coefficient of the graded-symmetric Hilbert series.
\end{proposition}

\begin{proof}
This is the explicit kernel count of
Proposition~\ref{prop:ker-av-schur-weyl}: the image of averaging on
$V^{\otimes n}$ is $\Symm^n(V)$ and stars-and-bars gives its
dimension. Subtracting from $d^n$ yields the kernel dimension.
\end{proof}

\section{Kernel classes for the four archetypes}
\label{sec:uc-kernel}

\begin{theorem}[Named kernel witnesses by archetype]
\label{thm:uc-kernel-archetypes}
\ClaimStatusProvedHere
The kernel of the universal conductor contains explicit ordered
witnesses for each of the four archetypes $G/L/C/M$.
\begin{enumerate}
\item[\textup{(G)}] \emph{Heisenberg / Gaussian witness.}
 For currents $a_i,a_j$, the antisymmetric tensor
 \[
 \Xi_G(a_i,a_j)
 :=
 [s^{-1}a_i\,|\,s^{-1}a_j]
 -
 [s^{-1}a_j\,|\,s^{-1}a_i]
 \]
 lies in $\ker K_{\mathcal H}$.
\item[\textup{(L)}] \emph{Affine Kac--Moody / Lie witness.}
 Let $t_{ij}$ denote the ordered Casimir insertion operators on
 $\Conf_3(X)$. The Lie-commutator class
 \[
 \Xi_L
 :=
 [t_{12},t_{23}]
 \]
 lies in $\ker K_{V_k(\fg)}$.
\item[\textup{(C)}] \emph{$\beta\gamma$ / contact witness.}
 The boson--fermion obstruction
 \[
 \Xi_C
 :=
 R_{\beta\gamma}
 =
 \beta\otimes\gamma-\gamma\otimes\beta
 \]
 lies in $\ker K_{\beta\gamma}$.
\item[\textup{(M)}] \emph{Virasoro / collision-channel witness.}
 The degree-$3$ non-invariant collision-channel component
 \[
 \Xi_M
 :=
 [T|T]_{12}\,\eta_{12}
 -
 [T|T]_{13}\,\eta_{13}
 \]
 lies in $\ker K_{\Vir}$.
\end{enumerate}
For each archetype the kernel claim has three independent
checks: direct averaging, a family-specific ordered
computation, and the shadow-depth classification in
\texttt{landscape\_census.tex}.
\end{theorem}

\begin{proof}
\textup{(G)} Direct averaging kills $\Xi_G$ because the transposition
$i\leftrightarrow j$ changes its sign while
$\Av^{\mathrm{Rey}}_{\Sigma_2}$ symmetrises. Independently,
Proposition~\ref{prop:uc-kernel-dimension} shows that every
finite-rank degree-$2$ ordered sector has a nontrivial antisymmetric
kernel unless $d=1$. Third, the Heisenberg rows in the census have
$r_{\max}=2$: nothing survives beyond the scalar shadow, so the ordered
antisymmetric residue must disappear under averaging.

\textup{(L)} The affine ordered residue carries the Lie bracket, and
the commutator $[t_{12},t_{23}]$ is exactly the bracket-asymmetry term.
Its symmetric average is zero because the symmetric shadow keeps the
Casimir scalar but forgets the ordered commutator. Independently,
Remark~\ref{rem:lie-associative-dichotomy} identifies the linearised
associator / Lie commutator as a degree-$3$ kernel class. Third, the
affine rows of the census are class~L with $r_{\max}=3$, so the cubic
ordered witness is the last surviving ordered obstruction before
symmetrisation kills higher terms.

\textup{(C)} Directly,
$\Av^{\mathrm{Rey}}_{\Sigma_2}(\beta\otimes\gamma-\gamma\otimes\beta)=0$.
Independently, Proposition~\ref{prop:bc-betagamma-orthogonality}
detects the same class by its nontrivial pairing with the Clifford dual
$R_{bc}=b\otimes c+c\otimes b$. Third, the class-C census rows are
precisely the quartic-contact families: the cubic average vanishes,
while the quartic contact class survives.

\textup{(M)} The three collision channels
$[T|T]_{12}\eta_{12}$, $[T|T]_{13}\eta_{13}$, and
$[T|T]_{23}\eta_{23}$ carry the permutation representation of
$\Sigma_3$. The Reynolds projector keeps the trivial line spanned by
their symmetric sum and kills its standard two-dimensional complement;
$\Xi_M$ has coefficient sum zero, so its average is zero. This is the
correct Arnold-compatible statement: Proposition~\ref{prop:arnold-virasoro-deg3}
shows that the linear symmetric leakage
$\frac c2[T](\eta_{12}+\eta_{13}+\eta_{23})$ survives, and that the
Arnold relation cancels only quadratic wedge expressions. Third, the
Virasoro and class-M rows of the census have $r_{\max}=\infty$ with
nonzero higher shadows $S_4,S_5,\dots$; the kernel witness is the
nontrivial ordered channel component, not extinction of the tower.
\end{proof}

\section{The landscape map to \texorpdfstring{$\{G,L,C,M\}$}{G/L/C/M}}
\label{sec:uc-landscape-map}

\begin{definition}[Averaged shadow depth]
\label{def:uc-averaged-shadow-depth}
Let $\cA$ be a standard row whose ordered shadow tower has a
$G/L/C/M$ Virasoro or free-field shadow projection, and let
$\Theta^{E_1}_{\cA,\le r}$ denote that ordered shadow tower. Define the
\emph{averaged shadow depth} by
\[
 r_{\mathrm{avg}}(\cA)
 :=
 \sup\bigl\{
 r\geq 2 \;:\; K_\cA(\Theta^{E_1}_{\cA,\le r})\neq 0
 \bigr\}
 \in \{2,3,4,\infty\}.
\]
Then define
\[
 \mathfrak U(\cA)
 :=
 \begin{cases}
 G, & r_{\mathrm{avg}}(\cA)=2,\\
 L, & r_{\mathrm{avg}}(\cA)=3,\\
 C, & r_{\mathrm{avg}}(\cA)=4,\\
 M, & r_{\mathrm{avg}}(\cA)=\infty.
 \end{cases}
\]
\end{definition}

\begin{theorem}[Constructed universality map on \texorpdfstring{$G/L/C/M$}{G/L/C/M} census rows]
\label{thm:uc-landscape-universality}
\ClaimStatusProvedHere
On the $G/L/C/M$-classified standard-landscape rows of
\texttt{landscape\_census.tex}, the map
$\mathfrak U\colon \cA\mapsto\{G,L,C,M\}$ of
Definition~\ref{def:uc-averaged-shadow-depth} reproduces the archetype
column entry-by-entry. The row count is:
\[
 G:5,\qquad L:2,\qquad C:4,\qquad M:10.
\]
These $21$ rows are exactly the census rows whose archetype column is
one of $G,L,C,M$. The Yangian row is the separate $E_1$-RTT row with
no Virasoro shadow depth and no scalar $c$ or $\kappa$ in this sense.
The strict-$E_1$ quantum lattice row remains unclassified there, and the
wild Kronecker row lies outside the domain because its bar spectral
sequence does not collapse.
\end{theorem}

\begin{proof}
The construction of $\mathfrak U$ is intrinsic: it is defined by the
supremum of the stages whose averaged projection is nonzero. The
comparison is finite because the census table is finite.

\emph{Class~G ($5$ rows).} Free fermion, Heisenberg, lattice VOA,
Niemeier lattice, and $Y_{1,1,1}[\Psi]$ are the Gaussian rows. For
each row three checks are available: the archetype table records
$r_{\max}=2$; the completion-kinematics table records class~G again;
and the family chapter shows vanishing of higher shadows after the
scalar level.

\emph{Class~L ($2$ rows).} The simply-laced and non-simply-laced
affine Kac--Moody rows are the two Lie rows. Three checks are again
available: the archetype table records $r_{\max}=3$; the affine family
chapters identify the cubic Lie witness; and
Theorem~\ref{thm:uc-kernel-archetypes}(L) exhibits the ordered
commutator killed by averaging.

\emph{Class~C ($4$ rows).} The $\beta\gamma$, $bc$, symplectic boson,
and symplectic fermion rows are the quartic-contact rows. They are
identified by the table entry $r_{\max}=4$, by the explicit quartic
contact class recorded in the family chapters, and by the
boson--fermion obstruction of
Theorem~\ref{thm:uc-kernel-archetypes}(C).

\emph{Class~M ($10$ rows).} Virasoro, finite $\mathcal W_N$,
principal $\mathcal W_3$, Ising, $\mathcal N{=}2$ SCA, triplet
$\mathcal W(p)$, $Y_{0,0,N}[\Psi]$, $\mathcal W_\infty$, Monster,
and Bershadsky--Polyakov are the mixed/infinite rows. Each row has three
checks: the archetype table records $r_{\max}=\infty$; the
shadow-invariants table records a nonvanishing higher shadow or
discriminant; and the family chapter supplies the infinite-tower
witness, starting with the Virasoro collision-channel witness of
Theorem~\ref{thm:uc-kernel-archetypes}(M).

The Yangian row is typed in the census as $E_1$-RTT: it has an
ordered RTT $R$-matrix tower and no Virasoro shadow depth. The theorem
therefore does not fold it into class~M. The quantum lattice row is
explicitly marked unknown in the census, so the theorem does not
classify it. The Kronecker wild row is outside the domain because the
modular-Koszul shadow tower does not exist there.
\end{proof}

\begin{remark}[Census counts by archetype]
\label{rem:uc-census-counts}
The $G/L/C/M$ row counts of
Theorem~\ref{thm:uc-landscape-universality} are themselves checked in
three independent ways: by direct count in the archetype table, by the
completion-kinematics table of
Section~\ref{sec:completion-kinematics-census}, and by the family
chapter list attached to each row in the census. The theorem uses the
intersection of those three surfaces and excludes the $E_1$-RTT
Yangian row from the four-archetype count.
\end{remark}

\section{Vol~III scope and boundary}
\label{sec:uc-voliii-boundary}

\begin{theorem}[Ordered-Koszul boundary for Vol~III comparisons]
\label{thm:uc-K-Atiyah}
\ClaimStatusProvedHere
The Vol~III RTT-Yangian / chiral-quantum-group comparison can
compare ordered data with the symmetric modular shadow through the
universal conductor only on the \emph{ordered Koszul locus}, namely
where the following three conditions hold simultaneously:
\begin{enumerate}
\item[\textup{(K1)}] the ordered counit
 \[
 \Omega(B^{\Ord}(\cA))\xrightarrow{\sim}\cA
 \]
 is a chain-level quasi-isomorphism, not merely a coderived one;
\item[\textup{(K2)}] the ordered bar complex is pure enough for the
 shadow tower $\Theta^{E_1}_\cA$ to be defined without extra
 completion obstructions;
\item[\textup{(K3)}] chiral Hochschild cohomology remains concentrated
 in degrees $\{0,1,2\}$.
\end{enumerate}
Off this locus the first named failure modes are:
\begin{enumerate}
\item[\textup{(B1)}] \emph{bar non-purity:} ordered bar cohomology
 spreads into extra weights;
\item[\textup{(B2)}] \emph{widened Hochschild amplitude:}
 $\mathrm{HH}^{\mathrm{ch},\bullet}(\cA)$ need not remain in $\{0,1,2\}$;
\item[\textup{(B3)}] \emph{coderived-only inversion:} the counit is no
 longer a chain-level quasi-isomorphism;
\item[\textup{(B4)}] \emph{RTT-Yangian bar obstruction:} the missing
 step is still the identification of $B(Y)$ as a factorisation
 coalgebra on $\Ran(C)$ with shuffle coproduct;
\item[\textup{(B5)}] \emph{elliptic/toroidal boundary:} the KZB/Fay
 replacement of the Arnold kernel is only partial, so no toroidal
 analogue of the universal conductor theorem is inscribed.
\end{enumerate}
\end{theorem}

\begin{proof}
Condition~(K1) is the explicit chain-level content of the ordered
Koszul locus. The failure off the locus is recorded in the ordered bar
package: for class~M and beyond the counit survives only in the
coderived sense. Condition~(K2) is the purity needed to define the
ordered tower itself; its failure is the non-pure bar obstruction.
Condition~(K3) is exactly the Vol~I/III Hochschild concentration
boundary.

The boundary terms are concrete obstructions. The RTT-Yangian
obstruction is the missing factorisation-coalgebra identification in
the Vol~III comparison.
The elliptic/toroidal boundary is the point where the rational Arnold
relation has been replaced by the Fay/KZB kernel only at the
propagator level; the comparison with ordered-to-symmetric averaging
remains open there.
\end{proof}

\begin{conjecture}[Universal trace comparison downstream of averaging]
\label{thm:uc-universal-trace-identity}
\ClaimStatusConjectured
Every universal-trace comparison between the Vol~I conductor and a
Vol~III chiral quantum-group invariant must first factor through the
ordered-to-symmetric averaging map
$K_\cA\colon \gAE\to \gAmod$ on the ordered Koszul locus of
Theorem~\ref{thm:uc-K-Atiyah}. The elliptic/KZB and toroidal regimes
have no theorem asserting such a factorisation.
\end{conjecture}

\section{Central-charge, \texorpdfstring{$\kappa$}{kappa}, and ghost scalar packages}
\label{sec:uc-scalar-packages}

The averaging conductor $K_\cA=\Av_\cA$ is a morphism of dg Lie
algebras. The scalar packages live on witness lanes where the bar-dual
coalgebra $A^i=H^*(B^{\mathrm{ch}}(A))$ is finite type in each weight,
or carries a completed continuous Verdier pairing, so that
$A^!=(A^i)^\vee$ is defined. A Virasoro element and a canonical BRST
presentation are additional data.

\begin{definition}[Three scalar conductor packages]
\label{def:uc-scalar-packages}
For such a Verdier pair $(A,A^!)$, set
\[
 K^c(A,A^!) := c(A)+c(A^!),\qquad
 K^\kappa(A,A^!) := \kappa(A)+\kappa(A^!).
\]
For a chosen finite BRST presentation set
\[
 K_{\mathrm{ghost}}^{\mathrm{leg}}(\cA)
 :=
 -\,c_{\mathrm{ghost}}(\BRST_{\mathrm{can}}(\cA)).
\]
The first number is central-charge complementarity; the second is
modular-characteristic complementarity; the third is a
presentation-sensitive ghost central-charge scalar.
\end{definition}

\begin{corollary}[Heisenberg scalar packages]
\label{cor:uc-K-heisenberg}
\ClaimStatusProvedHere

\section{Averaging and scalar identities}
\label{sec:uc-identities}

The universal conductor of Volume~I is the averaging map
\[
 K_\cA=\Av_\cA\colon \gAE\longrightarrow \gAmod,
\]
with named witnesses
$K_{\cA}^{\mathrm{ch}}=q_\bullet\circ\Av^{\mathrm{Rey}}_{\Sigma_\bullet}$
and $K^\infty=q_!\simeq(-)_{h\Sigma_\bullet}$. Its image is the
symmetric quotient, equivalently the Reynolds fixed-point summand,
equivalently the homotopy-coinvariant factorisation shadow
(Theorem~\ref{thm:uc-trinity}). Its kernel contains explicit ordered
classes for the four archetypes:
antisymmetric Heisenberg tensors, the affine Lie commutator,
$R_{\beta\gamma}$, and the Virasoro Arnold witness
(Theorem~\ref{thm:uc-kernel-archetypes}).

The map to $\{G,L,C,M\}$ is the $G/L/C/M$ depth of the averaged shadow
and agrees with the $21$ $G/L/C/M$-classified census rows; the Yangian row
is $E_1$-RTT and lies outside this four-archetype count
(Theorem~\ref{thm:uc-landscape-universality}). The Vol~III comparison
is restricted to the ordered Koszul locus, with named failure modes off
that locus (Theorem~\ref{thm:uc-K-Atiyah}). The scalar packages are
separate:
\[
 K^c,\qquad K^\kappa,\qquad K_{\mathrm{ghost}}^{\mathrm{leg}}.
\]
They coincide only on the family lanes where the displayed formulas
prove the coincidence.
