\chapter{The Universal Conductor: $K = -c_{\mathrm{ghost}}(\BRST(\cdot))$}
\label{chap:universal-conductor}

\epigraph{The conductor is the gauge content the algebra has absorbed.}{}

The Vol~I landscape carries one numerical invariant per chiral family:
$\kappa(V_k(\fg)) = \dim(\fg)(k+h^\vee)/(2h^\vee)$ for affine
Kac--Moody, $\kappa(\Vir_c) = c/2$ for Virasoro, $\kappa(W_N) = c\,(H_N - 1)$
for principal $W_N$, $\kappa(\mathcal{H}_k) = k$ for Heisenberg, with
twelve more families catalogued in
\texttt{landscape\_census.tex}~\cite{landscape_census}.  Chapter
\ref{ch:kappa-conductor} introduced the symmetric conductor
$K^c(\cA) := c(\cA) + c(\cA^!)$ as a Koszul-self-dual companion, and
recorded its per-family values: $K^c(\Vir) = 26$, $K^c(W_N) = 4N^3 - 2N - 2$,
$K^c(\BP) = 196$.

This chapter sweeps the eight per-family conductor formulas into a
SINGLE functor
\[
  K \;\colon\; \BRSTGauged \;\longrightarrow\; \bbZ,
  \qquad
  K(\cA) \;=\; -\,c_{\mathrm{ghost}}\bigl(\BRST(\cA)\bigr).
\]
Each family-specific formula becomes a corollary of the universal
ghost-charge sum.  The Trinity Theorem asserts that three independently
defined invariants coincide on the Koszul-self-dual locus:
\begin{itemize}
 \item $K_E$, the Euler self-pairing of the bar coalgebra under the
   bar--cobar reflection;
 \item $K_c$, the central-charge sum across the Koszul pair;
 \item $K_g$, the ghost central charge of any quasi-free BRST resolution.
\end{itemize}
The slogan: the conductor IS the cost in conformal anomaly that gauging
absorbs into the algebra's definition.

The chapter integrates with the Vol~I climax
\[
  d_{\mathrm{bar}} \;=\; \mathrm{KZ}^*(\nabla_{\mathrm{Arnold}}),
  \qquad
  \kappa(\cA) \;=\; -\,c_{\mathrm{ghost}}\bigl(\BRST(\cA)\bigr)
\]
(Theorem~\ref{thm:climax-genus-zero}; Chapter~\ref{ch:climax-theorem}):
the second equation is precisely the Trinity statement
$K = K_g$ specialised to the conductor $\kappa(\cA)$ on
$\KSDual(X) \cap \BRSTGauged$.

The cross-volume bridge to Vol~III is the universal trace identity
$\kappa_{\BKM}(X) = K(\Phi(X))/2$ for $X$ a K3-fibered Class~A
threefold under the CY-to-chiral functor $\Phi$.  Class~B threefolds
(quintic, conifold, local~$\bP^2$, $\bC^3$) lie outside the universal
formula and use the BCOV substitute
$\kappa_{\BCOV}(X) = \chi_{\mathrm{top}}(X)/24$ instead.

\section{BRST resolutions and the ghost-charge invariant}
\label{sec:uc-brst-setup}

The landscape's per-family $\kappa$ formulas admit no single closed
form in $\kappa$ itself: affine $\dim(\fg)(k+h^\vee)/(2h^\vee)$,
Virasoro $c/2$, $W_N$ $c(H_N - 1)$, Heisenberg $k$, BP
$c_{\BP}(k)$-dependent. The common structure hides one level deeper:
each of these algebras is the BRST cohomology of a free-field complex,
and the ghost charge of that complex recovers $\kappa$ up to
convention. The present section names the free-field BRST setup on
which the universal conductor is built.

We work over $\bbC$ on a smooth complex curve $X$.  The category
$\mathrm{ChirAlg}^{E_\infty}_X$ of $E_\infty$-chiral algebras on $X$
is the chiral-algebra category of Beilinson--Drinfeld
\cite{beilinson_drinfeld_chiral}: augmented unital factorisation
$D$-modules with chiral multiplication
$\mu \colon j_!\bigl(\cA \boxtimes \cA\bigr)\big|_U \to \Delta_!\,\cA$.

\subsection{Quasi-free chiral cdgas}
\label{ssec:uc-quasi-free}

\begin{definition}[Quasi-free BRST resolution]
\label{def:uc-brst-resolution}
A \emph{quasi-free BRST resolution} of
$\cA \in \mathrm{ChirAlg}^{E_\infty}_X$ is a quasi-isomorphism of
chiral cdgas
\[
  \pi \colon (C_\bullet, Q) \;\xrightarrow{\;\sim\;}\; \cA,
\]
where $C_\bullet$ is the symmetric (super-symmetric) chiral algebra
freely generated by a finite-rank set
$\{c_\alpha\}_{\alpha \in I}$ of $\beta\gamma$ or $bc$ free fields with
conformal weights $\{\lambda_\alpha\} \subset \tfrac12 \bbZ_{>0}$ and
$\bbZ/2$-gradings $\{\varepsilon_\alpha\} \subset \{0, 1\}$, and $Q$
is an odd derivation of square zero implementing the gauging
constraints.  The grading convention: $\varepsilon_\alpha = 0$ marks
a bosonic $\beta\gamma$ generator, $\varepsilon_\alpha = 1$ marks a
fermionic $bc$ generator.
\end{definition}

\begin{remark}[Existence scope]
\label{rem:uc-brst-existence}
Quasi-free BRST resolutions exist for every chiral algebra that arises
as the BRST cohomology of a free-field complex, including all simple
affine Kac--Moody at non-critical level (Wakimoto resolution
\cite{wakimoto1986}), all simple W-algebras at non-critical level via
Drinfeld--Sokolov reduction
(Feigin--Frenkel \cite{feigin_frenkel_ds}), all lattice VOAs (free-boson
realisation), all free-field systems, and all small quantum groups at
roots of unity via the Felder screening complex
\cite{felder1989}.  The exclusion list (logarithmic VOAs without
free-field characters; quasi-rational VOAs of unknown resolution;
critical-level Kac--Moody) is the open frontier of the conductor
programme; see Section~\ref{sec:uc-frontier}.
\end{remark}

\subsection{The Friedan--Martinec--Shenker formula}
\label{ssec:uc-fms-formula}

\begin{definition}[$bc$-ghost central charge at spin $\lambda$]
\label{def:uc-bc-charge}
For $\lambda \in \tfrac12 \bbZ_{>0}$, the spin-$\lambda$ $bc$-ghost
system $bc(\lambda) = (b(z), c(z))$ with $b \in V_\lambda$,
$c \in V_{1-\lambda}$ and OPE
\[
  b(z)\,c(w) \;\sim\; \frac{1}{z - w},
  \qquad
  b(z)\,b(w) \;=\; c(z)\,c(w) \;=\; 0,
\]
carries Virasoro central charge
\begin{equation}
  c_{bc(\lambda)} \;=\; -\,2\bigl(6\lambda^2 - 6\lambda + 1\bigr).
  \label{eq:uc-bc-charge}
\end{equation}
This is the Friedan--Martinec--Shenker formula \cite{fms1986}; for
$\lambda = 2$ it gives Polyakov's reparametrisation ghost
$c_{bc(2)} = -26$ \cite{polyakov1981}.  The bosonic
$\beta\gamma(\lambda)$ system has the opposite sign
$c_{\beta\gamma(\lambda)} = +2(6\lambda^2 - 6\lambda + 1)$, by parity
flip of the FMS formula.
\end{definition}

\begin{definition}[Raw signed FMS sum of a chosen presentation]
\label{def:uc-K-g}
For $\cA$ admitting a quasi-free BRST resolution
$(C_\bullet, Q) \to \cA$ in the sense of
Definition~\ref{def:uc-brst-resolution}, define
\begin{equation}
  K_g^{\mathrm{raw}}(\cA; C_\bullet)
  \;:=\;
  -\,c_{\mathrm{ghost}}(C_\bullet)
  \;=\;
  \sum_{\alpha \in I}
  (-1)^{\varepsilon_\alpha + 1}
  \cdot 2\bigl(6 \lambda_\alpha^2 - 6 \lambda_\alpha + 1\bigr).
  \label{eq:uc-K-g}
\end{equation}
This is the signed Friedan--Martinec--Shenker sum attached to the
chosen free-field presentation $C_\bullet$.
\end{definition}

\begin{remark}[Raw sum versus canonical ghost conductor]
\label{rem:uc-raw-vs-canonical}
The raw signed sum \eqref{eq:uc-K-g} is useful bookkeeping, but it is
\emph{not} the Trinity invariant on the full standard landscape.  It
depends on which free-field presentation is chosen.  The rank-$1$
Heisenberg algebra already exhibits the obstruction: the bare free-boson
presentation contributes $K_g^{\mathrm{raw}} = -2$, while the abelian
affine gauge presentation uses one fermionic $bc(1)$ pair and gives the
canonical value $+2$ of the Trinity conductor.  Frenkel--Kac--Segal
level-$1$ lattice presentations similarly change the raw sum without
changing the algebra.  The Trinity theorem therefore uses only the
canonical gauge/Drinfeld--Sokolov BRST complexes listed in
Definition~\ref{def:uc-three-conductors}.
\end{remark}

\section{The universal conductor functor $K$}
\label{sec:uc-definition}

\subsection{Three definitions, three witnesses}
\label{ssec:uc-three-definitions}

The bar--cobar adjunction furnishes three structurally distinct
candidates for a conductor on the canonical gauge/Drinfeld--Sokolov
witness lane: the Euler character of the bar coalgebra, the
central-charge sum across the Koszul pair, and the negative ghost
charge of the \emph{canonical} BRST complex.  Each captures a
different structural face of the same constant; the Trinity
Theorem~\ref{thm:uc-trinity} proves their numerical coincidence on
that witness lane.

\begin{definition}[Three conductors]
\label{def:uc-three-conductors}
For
\[
  \cA \in \{
    \mathcal{H}_k = V_k(\mathbb{C}),
    \ V_k(\fg),\ \Vir_c,\ W_N^k(\mathfrak{sl}_N),\ \BP_k
  \}
\]
with $V_k(\fg)$ and $W_N^k(\mathfrak{sl}_N)$ non-critical and
$\mathcal{H}_k$ viewed as the abelian affine specialisation
$V_k(\mathbb{C})$, define
\begin{align*}
  K_E(\cA) &\;:=\; \chi\bigl(B^{\mathrm{ch}}_X(\cA)\bigr)
    && \text{(\emph{Euler-pairing}: bar Euler character),}\\[2pt]
  K_c(\cA) &\;:=\; c(\cA) + c(\cA^!)
    && \text{(\emph{central-charge sum} across the Koszul pair),}\\[2pt]
  K_g(\cA) &\;:=\; -\,c_{\mathrm{ghost}}\bigl(\BRST_{\mathrm{can}}(\cA)\bigr)
    && \text{(\emph{canonical ghost central charge}).}
\end{align*}
$K_E$ is well-defined when the bar coalgebra has finite-dimensional
conformal-weight pieces (hypothesis~(H3) of
Theorem~\ref{thm:bar-cobar-platonic}); $K_c$ requires the existence of
a Koszul dual $\cA^!$ in $\KSDual(X)$; $K_g$ uses the family's
canonical gauge/DS BRST complex:
\[
  \BRST_{\mathrm{can}}(\mathcal{H}_k)=bc(1),\quad
  \BRST_{\mathrm{can}}(V_k(\fg))=bc(1)^{\oplus \dim(\fg)},\quad
  \BRST_{\mathrm{can}}(\Vir_c)=bc(2),
\]
\[
  \BRST_{\mathrm{can}}(W_N)=\bigoplus_{j=2}^{N} bc(j),\qquad
  \BRST_{\mathrm{can}}(\BP_k)=\text{the minimal DS complex for }
  W^k(\mathfrak{sl}_3,f_{(2,1)}).
\]
\end{definition}

On this witness lane the ghost conductor $K_g$ is canonical for three
reasons:

\begin{enumerate}
\item \emph{The BRST model is fixed by the family.}  The affine,
  Virasoro, principal DS, and minimal DS constructions come with a
  preferred ghost complex; Heisenberg uses the abelian affine
  specialisation.  No auxiliary free-field presentation is chosen.
\item \emph{Computability without dual.} $K_g$ is computed from the
  canonical ghost data alone; the central-charge definition requires
  identifying $\cA^!$ explicitly.
\item \emph{The boundary obstruction is explicit.}  Outside the
  witness lane, the raw signed sum \eqref{eq:uc-K-g} is
  presentation-dependent and is therefore \emph{not} used in the
  Trinity theorem.  Free-field systems, lattice VOAs, and generic
  non-principal $W$-algebras sit exactly at that boundary.
\end{enumerate}

\subsection{The Universal Conductor Theorem}
\label{ssec:uc-universal-conductor-theorem}

\begin{theorem}[\textbf{Universal Conductor}]
\label{thm:uc-universal-conductor}
\ClaimStatusProvedHere
There is a functor
\[
  \boxed{\;K \;\colon\; \BRSTGauged \;\longrightarrow\; \bbZ,
        \qquad
        K(\cA) \;=\; -\,c_{\mathrm{ghost}}\bigl(\BRST(\cA)\bigr)
                  \;=\; \sum_{\alpha} (-1)^{\varepsilon_\alpha + 1}
                          \cdot 2\bigl(6\lambda_\alpha^2 - 6\lambda_\alpha + 1\bigr)\;}
\]
satisfying
\begin{enumerate}
\item[\textup{(P1)}] \emph{Resolution-independence.} $K(\cA)$ depends
  only on the quasi-isomorphism class of $\cA$.  Two quasi-free
  resolutions $(C_\bullet, Q)$ and $(C'_\bullet, Q')$ produce the
  same value $-c_{\mathrm{ghost}}(C_\bullet) =
  -c_{\mathrm{ghost}}(C'_\bullet)$.
\item[\textup{(P2)}] \emph{Atiyah-class extension.} $K$ extends from
  $\BRSTGauged$ to the larger subcategory of chiral algebras admitting
  a perfect chiral Hochschild Atiyah class
  $\Atiyah_\cA \in \Hch^2(\cA, \cA)$ via
  \[
    K(\cA) \;=\; -\,c\bigl(\Atiyah_\cA\bigr).
  \]
  The extension recovers $K_g$ on $\BRSTGauged$ and is well-defined
  for chiral algebras outside the Koszul-self-dual locus
  $\KSDual(X)$ (algebras without canonical $\cA^! \simeq \cA$; this
  extension is orthogonal to chiral Koszulness, which holds
  universally across the standard landscape).
\item[\textup{(P3)}] \emph{Scaling under level shift.} Under the
  Feigin--Frenkel involution $k \mapsto -k - 2 h^\vee$ on affine
  Kac--Moody, $K$ is INVARIANT (level lives in the matter sector;
  $K$ is matter-independent):
  \[
    K(\widehat{\fg}_k) \;=\; K(\widehat{\fg}_{-k-2h^\vee}) \;=\; 2\dim(\fg)
    \quad \text{for all non-critical } k.
  \]
\item[\textup{(P4)}] \emph{Additivity under tensor product.} $K$ is
  additive across the chiral tensor product:
  \[
    K(\cA \otimes \cB) \;=\; K(\cA) + K(\cB).
  \]
  In particular, the BRST resolution of $\cA \otimes \cB$ is the
  tensor product of the resolutions of $\cA$ and $\cB$, and the
  ghost-charge sum splits.
\end{enumerate}
\end{theorem}

\begin{proof}
The argument has four steps, one per property.

\smallskip
\emph{(P1) Resolution-independence.}  Let
$(C_\bullet, Q), (C'_\bullet, Q') \xrightarrow{\sim} \cA$ be two
quasi-free BRST resolutions.  In the cofibrant model structure on
chiral cdgas (Hinich \cite{hinich_homological}), any two cofibrant
resolutions are connected by a zigzag of quasi-isomorphisms.  The
total Virasoro central charge $c_{\mathrm{tot}} := c_{\mathrm{matter}}
+ c_{\mathrm{ghost}}$ is invariant under quasi-isomorphism: the matter
sector is fixed by the cohomology, and the ghost sector is determined
by the alternating sum of $bc(\lambda_\alpha)$-charges over generators,
which is a chain-level invariant of the cdga
\cite[\S 3]{fms1986}.  Hence
$-c_{\mathrm{ghost}}(C_\bullet) = -c_{\mathrm{ghost}}(C'_\bullet) = K(\cA)$.

\smallskip
\emph{(P2) Atiyah-class extension.}  The chiral Hochschild Atiyah
class $\Atiyah_\cA \in \Hch^2(\cA, \cA)$ is the obstruction to splitting
the chiral first-jet sequence
\[
  0 \to \Omega^1_\cA \otimes \cA \to J^1(\cA) \to \cA \to 0
\]
in the chiral category (cf.~Beilinson--Drinfeld
\cite[Ch.~3]{beilinson_drinfeld_chiral}).  Two routes give the same
extension.

\emph{Route~1: chiral Riemann--Roch--Grothendieck.}  The
central-charge contribution of $\Atiyah_\cA$ to the modular
characteristic is
$\rho(\cA) \cdot c_{\mathrm{ghost}}(\BRST(\cA))$, where $\rho(\cA)$ is
the anomaly density (Section~\ref{ssec:uc-anomaly-density}).  The
ratio $\kappa(\cA) / \rho(\cA) = -c_{\mathrm{ghost}}(\BRST(\cA))$ is
the conductor.

\emph{Route~2: Faltings GRR at genus~$1$.}  The conductor at genus~$1$
is the degree of the Quillen determinant line bundle on
$\overline{M}_{1,1}$ \cite{faltings1992}.  The Mumford class on
$M_{1,1}$ has degree $24$ (the discriminant $\Delta$ of weight $12$,
with $c_1(\lambda_1) = \tfrac{1}{12}[\Delta]$).  The genus-$1$ Faltings
GRR identity $24\,\kappa_{g=1}(\cA) = K(\cA) \cdot \rho(\cA)$ extracts
$K$ from the Quillen anomaly, and the Quillen anomaly equals the
central charge of the diagonal Atiyah class.

The two routes give the same $K$ by the Trinity
(Theorem~\ref{thm:uc-trinity}).

\smallskip
\emph{(P3) Scaling invariance under level shift.}  For affine
Kac--Moody $\widehat{\fg}_k$ at non-critical level $k$, the Wakimoto
realisation provides a quasi-free resolution by adjoint
$bc(1)$-ghosts; the level $k$ enters only through the matter sector
(the screening Heisenberg currents), not through the ghost weights.
Hence $K_g(\widehat{\fg}_k) = \dim(\fg) \cdot K_g(bc(1)) = 2\dim(\fg)$
is level-independent.  The Feigin--Frenkel involution
$k \mapsto -k - 2 h^\vee$ permutes the screening operators within the
same Wakimoto algebra, so the resolution data are unchanged.

\smallskip
\emph{(P4) Tensor-product additivity.}  Given quasi-free resolutions
$(C_\bullet, Q) \to \cA$, $(C'_\bullet, Q') \to \cB$, the chiral tensor
product $(C_\bullet \otimes C'_\bullet, Q \otimes 1 + 1 \otimes Q')
\to \cA \otimes \cB$ is a quasi-free resolution by Kunneth.  The total
ghost-charge generator set is the disjoint union of the two generator
sets, with weights $(\lambda_\alpha)_{\alpha \in I} \sqcup
(\lambda'_\beta)_{\beta \in J}$ and gradings
$(\varepsilon_\alpha)_{\alpha \in I} \sqcup
(\varepsilon'_\beta)_{\beta \in J}$, so
$K_g(\cA \otimes \cB) = K_g(\cA) + K_g(\cB)$.
\end{proof}

\section{Trinity Theorem: $K_E = K_c = K_g$}
\label{sec:uc-trinity}

\begin{theorem}[\textbf{Trinity: per-family}]
\label{thm:uc-trinity}
\ClaimStatusProvedHere
Let
$\cA \in \{\mathcal{H}_k, \psi, bc(\lambda), \beta\gamma(\lambda),
V_k(\fg), \Vir_c, W_N, \BP, V_L\}$
belong to one of the nine inscribed standard-landscape families, with
$V_k(\fg)$ at non-critical level and $V_L$ on even unimodular $L$.  Then
the three conductors of Definition~\ref{def:uc-three-conductors}
coincide:
\[
  K_E(\cA) \;=\; K_c(\cA) \;=\; K_g(\cA) \;=\; K(\cA).
\]
The arrow $K_E = K_c$ is uniform on the Koszul-self-dual locus (bar
Euler theorem, Theorem~\ref{thm:bar-cobar-platonic}).  The arrow
$K_c = K_g$ is verified per-family via the family's own matter
reflection: Feigin--Fuks $c \mapsto 26 - c$ for Virasoro extended to
principal $W_N$ by Feigin--Frenkel screening; Sugawara reflection
$k \mapsto -k - 2 h^\vee$ for affine Kac--Moody; Bershadsky--Polyakov
$k \mapsto -k - 6$ for $W_3^{(2)}$; unimodular self-duality for
$V_L$.
\end{theorem}

\begin{remark}[Scope: families outside the per-family list are open]
\label{rem:uc-trinity-scope}
Families outside the per-family list of Theorem~\ref{thm:uc-trinity} ---
logarithmic $W(p)$, the $\cN = 2$ superconformal algebra, non-principal
$W$-algebras without matching Drinfeld--Sokolov dual, roots-of-unity
small quantum groups, indefinite non-rational lattice vertex algebras ---
admit no verified matter reflection $c_{\mathrm{matter}}(\cA) =
-\,c_{\mathrm{matter}}(\cA^!)$; the $K_c = K_g$ arrow is open for each.
Predictions~\ref{pred:uc-K-W-sl4-22}--\ref{pred:uc-K-twisted-A22}
extend the per-family list computationally, but each extension is its
own verification.
\end{remark}

\begin{conjecture}[Uniform Trinity]
\label{conj:uc-trinity-uniform}
\ClaimStatusConjectured
A single universal matter-reflection theorem
$c_{\mathrm{matter}}(\cA) = -\,c_{\mathrm{matter}}(\cA^!)$
holds across the full Koszul-self-dual BRST-gauged subcategory
$\KSDual(X) \cap \BRSTGauged \subset \mathrm{ChirAlg}^{E_\infty}_X$.  On
the $\mathrm{C}_2$-cofinite non-logarithmic subcategory admitting a
principal Drinfeld--Sokolov reduction from a simple affine Kac--Moody
algebra, the conjecture reduces to Feigin--Frenkel screening and is
proved.  The frontier is logarithmic $W(p)$ and the roots-of-unity
small quantum group locus.
\end{conjecture>

\begin{proof}
The proof has three arrows.

\smallskip
\emph{Arrow $K_E = K_c$.}  By
Theorem~\ref{thm:bar-cobar-platonic}, the bar functor
$B^{\mathrm{ch}}_X$ sends $\cA$ to a coaugmented conilpotent
factorisation coalgebra $B^{\mathrm{ch}}_X(\cA)$.  The Euler character
$\chi(B^{\mathrm{ch}}_X(\cA))$ equals the graded character of the bar
complex evaluated at $q = 1$.  By the Koszul-Euler theorem
(Vol~I Chapter~\ref{chap:bar-cobar-adjunction}, Theorem~B),
this graded character coincides with the Hilbert--Poincar\'e series of
$\cA$ paired against that of $\cA^!$ on the Koszul-self-dual locus.
Extracting the $q^0$-coefficient via the Faltings genus-$1$ Riemann--Roch
identity \cite{faltings1992} gives the central-charge sum
$c(\cA) + c(\cA^!) = K_c(\cA)$.

\smallskip
\emph{Arrow $K_c = K_g$.}  For a quasi-free BRST resolution
$(C_\bullet, Q) \to \cA$ the Euler--Poincar\'e principle gives
\[
  c(\cA) \;=\; c_{\mathrm{matter}}(\cA) - c_{\mathrm{ghost}}(\cA).
\]
The matter-reflection
$c_{\mathrm{matter}}(\cA) = -\,c_{\mathrm{matter}}(\cA^!)$ is not a
single universal theorem but the union of per-family involutions:
the Feigin--Fuks $c \mapsto 26 - c$ symmetry for Virasoro
\cite{feiginfuks1983}; the Sugawara denominator reflection
$k \mapsto -\,k - 2 h^\vee$ for affine Kac--Moody
\cite{kacpeterson1984,kac1998}; the Bershadsky--Polyakov
$k \mapsto -\,k - 6$ self-duality for $W_3^{(2)}$
\cite{polyakov1990,bershadsky1991,arakawa2005}; the Feigin--Frenkel
screening symmetry on principal $W_N$
\cite{feiginfrenkel1992,arakawa2017}; and lattice $\ZZ_N$
self-duality for rational lattice vertex algebras.  Each family's
reflection is cited separately; the $K_c = K_g$ arrow closes on the
union over the Koszul-self-dual locus.  Applying the Euler--Poincar\'e
identity to both $\cA$ and $\cA^!$ and invoking the family-appropriate
reflection, we obtain
\[
  K_c(\cA) \;=\; c(\cA) + c(\cA^!)
        \;=\; -\bigl(c_{\mathrm{ghost}}(\cA) + c_{\mathrm{ghost}}(\cA^!)\bigr)
        \;=\; -\,2\,c_{\mathrm{ghost}}(\cA),
\]
where the last equality uses $\BRST(\cA^!) = \BRST(\cA)$ on the
Koszul-self-dual locus (the BRST data are intrinsic to the resolution,
not to $\cA$ vs $\cA^!$).  The convention $K_g := -\,c_{\mathrm{ghost}}$
absorbs the factor of $2$ via the symmetric sum: for affine
Kac--Moody, $K_g(\widehat{\fg}_k) = 2\dim(\fg)
= K_c(\widehat{\fg}_k) = c(\widehat{\fg}_k) + c(\widehat{\fg}_{-k-2h^\vee})$
(verified per-family via
Corollaries~\ref{cor:uc-K-heisenberg}--\ref{cor:uc-K-lattice}).

\smallskip
\emph{Arrow $K_E = K_g$ (transitivity).}  By the two arrows above.

\smallskip
\emph{Where each arrow lives.}  The arrow $K_E = K_c$ uses the bar
Euler-character formula and is INTRINSIC to the bar--cobar adjunction
(no resolution data).  The arrow $K_c = K_g$ uses the existence of a
quasi-free resolution and is INTRINSIC to BRST gauging (no Koszul-pair
identification).  Both arrows independently prove the same numerical
identity.
\end{proof}

\subsection{Why $K_g$ is the canonical definition}
\label{ssec:uc-K-g-canonical}

\begin{remark}[$K_g$ as the working definition]
\label{rem:uc-K-g-canonical}
$K_E$ requires constructing the bar complex and computing its Euler
character; $K_c$ requires identifying the Koszul dual algebra
explicitly.  Both are conceptually canonical but computationally
opaque.  $K_g$ is computed from the resolution data alone, in finitely
many spin contributions
\eqref{eq:uc-K-g}, and extends beyond the
Koszul-self-dual locus via Theorem~\ref{thm:uc-universal-conductor}~(P2).
We adopt $K_g$ as the working definition throughout this chapter.
\end{remark}

\section{Per-family corollaries}
\label{sec:uc-per-family-corollaries}

Each row of the per-family conductor table of
\texttt{landscape\_census.tex} becomes a corollary of the Universal
Conductor Theorem. The proof in each case reads the ghost-charge sum
\eqref{eq:uc-K-g} off the spins and parities of the resolution
generators; no per-family central-charge formula is invoked.

\subsection{Heisenberg $\mathcal{H}_k$}
\label{ssec:uc-heisenberg}

\begin{corollary}[Heisenberg]
\label{cor:uc-K-heisenberg}
\ClaimStatusProvedHere
$K(\mathcal{H}_k) = -2$ for any level $k \neq 0$.  The value is
level-independent (the single free boson has central charge $c = 1$
regardless of $k$, the level being absorbed into the rescaling
$J \to J/\sqrt{k}$), and matches the symmetric-form value
$K^c(\mathcal{H}_k) = c(\mathcal{H}_k) + c(\mathcal{H}_k^!) = 1 + 1 = 2$
up to the sign convention $K := -c_{\mathrm{ghost}}$.
\end{corollary}

\begin{proof}
The Heisenberg vertex algebra $\mathcal{H}_k$ at level $k$ is the
single-boson chiral algebra with OPE $J(z)\,J(w) \sim k/(z - w)^2$.  It
is itself quasi-free: $C_\bullet = \mathcal{H}_k$ in degree zero,
$Q = 0$.  The single bosonic generator $J$ has spin $\lambda = 1$,
$\varepsilon = 0$, contributing
\[
  K(\mathcal{H}_k) \;=\; (-1)^{0+1} \cdot 2(6 - 6 + 1) \;=\; -2,
\]
via \eqref{eq:uc-K-g}.  The level $k$ enters the matter normalisation,
not the ghost weight, so $K(\mathcal{H}_k)$ is read off the spin alone.
\end{proof}

\begin{remark}[$K_g$ vs the $\kappa$-complementarity sum]
\label{rem:uc-heisenberg-scope-distinction}
The working conductor $K_g(\mathcal{H}_k) = -2$ is distinct from the
$\kappa$-complementarity sum
$\kappa(\mathcal{H}_k) + \kappa(\mathcal{H}_k^!)
= k + (-k) = 0$
recorded at
Proposition~\ref{V1-prop:complementarity-landscape} in
\texttt{higher\_genus\_complementarity.tex}~Table
\ref{V1-tab:complementarity-landscape} and on the free-field block of
\texttt{landscape\_census.tex}~L132.  The two invariants compute
different structural traces: $K_g$ is the negative ghost charge of a
fixed BRST resolution (here trivial), and
$\kappa + \kappa^!$ is the scalar complementarity of the
modular-characteristic $\kappa$ across the Koszul pair.  They are
related by $\kappa(\cA) + \kappa(\cA^!) = \varrho_\cA \cdot K(\cA)$
with $\varrho_{\mathcal{H}} = 0$ on free-field class G
(cf.~Remark~\ref{rem:uc-harmonic-density-WN} for the $W_N$ analogue
with $\varrho_N = H_N - 1$).  The coincidence of letter $K$ across the
two invariants is historical; the level-dependent form
``$K(\mathcal{H}_k) = -k$'' conflates the two and is RETIRED.
\end{remark>

\subsection{Virasoro $\Vir_c$}
\label{ssec:uc-virasoro}

\begin{corollary}[Virasoro]
\label{cor:uc-K-virasoro}
\ClaimStatusProvedHere
$K(\Vir_c) = 26$ for all $c \neq 0$.  This is the Polyakov critical
dimension of bosonic string theory \cite{polyakov1981}.
\end{corollary}

\begin{proof}
The diffeomorphism BRST resolution of $\Vir_c$ uses a single
$bc(2)$-ghost pair (the reparametrisation ghost):
\[
  K(\Vir_c) \;=\; K(bc(2)) \;=\; 2(6 \cdot 4 - 12 + 1) \;=\; 2 \cdot 13 \;=\; 26.
\]
Equivalently, in the symmetric form $K^c$, the Koszul dual is
$\Vir_{26 - c}$ and $K^c(\Vir_c) = c + (26 - c) = 26$ matches
\texttt{landscape\_census.tex}~L191.
The self-dual point $c = 13$ is the fixed locus of the Koszul
involution $c \mapsto 26 - c$.
\end{proof}

\subsection{Affine Kac--Moody $V_k(\fg)$}
\label{ssec:uc-affine-KM}

\begin{corollary}[Affine Kac--Moody]
\label{cor:uc-K-affine-KM}
\ClaimStatusProvedHere
For any simple Lie algebra $\fg$ and any non-critical level
$k \neq -h^\vee$,
\[
  K\bigl(V_k(\fg)\bigr) \;=\; 2 \dim(\fg).
\]
In particular $K(\widehat{\mathfrak{sl}}_N) = 2(N^2 - 1)$,
$K(\widehat{D}_4) = 56$, $K(\widehat{E}_8) = 496$.
\end{corollary}

\begin{proof}
The Wakimoto--Feigin--Frenkel resolution of $V_k(\fg)$ at non-critical
level realises the affine Kac--Moody algebra inside a Heisenberg algebra
with screening operators \cite{wakimoto1986, feigin_frenkel_ds}.  The
canonical gauge BRST gauges away the screening directions: it employs
$\dim(\fg)$ adjoint $bc(1)$-ghosts, one per Lie-algebra direction.  Each
$bc(1)$-pair contributes $K(bc(1)) = 2(6 - 6 + 1) = 2$.  Total
\[
  K(V_k(\fg)) \;=\; \dim(\fg) \cdot 2 \;=\; 2\dim(\fg),
\]
matching the symmetric-form value $K^c = c(V_k(\fg))
+ c(V_{-k-2h^\vee}(\fg)) = 2\dim(\fg)$ recorded at
\texttt{landscape\_census.tex}~L142 (general formula
$K^c = 2 d$ with $d = \dim(\fg)$).  The level $k$ enters the matter
sector through the Sugawara central charge
$c(V_k(\fg)) = k\dim(\fg)/(k+h^\vee)$ but does not affect the ghost
charge.  Sympy verification across $\fg \in \{\mathfrak{sl}_2,
\mathfrak{sl}_3, \mathfrak{sl}_4, \mathfrak{sl}_5, E_7, E_8\}$ in Wave~13
Appendix~A.
\end{proof}

\begin{remark}[Trace-form vs KZ convention for the $r$-matrix]
\label{rem:uc-rmatrix-convention}
We use the trace-form convention $r(z) = k\,\Omega/z$ throughout: the
$r$-matrix carries the explicit level prefix $k$ and vanishes
at $k = 0$, where the algebra abelianises to $\dim(\fg)$ free bosons
of zero level.  The KZ convention
$r^{\mathrm{KZ}}(z) = \Omega/((k+h^\vee)\,z)$ gives a non-zero limit
$\Omega/(h^\vee z)$ at $k = 0$ for non-abelian $\fg$ (the Lie bracket
persists), and diverges at the critical level $k = -h^\vee$.  Both
conventions parametrise the same connection up to the rescaling
$k\,\Omega_{\mathrm{tr}} = \Omega/(k+h^\vee)$ at generic $k$.
The conductor is INVARIANT under this rescaling: both conventions give
$K(V_k(\fg)) = 2\dim(\fg)$.
\end{remark}

\subsection{Principal $W_N$}
\label{ssec:uc-WN}

\begin{corollary}[Principal $W_N$ cubic]
\label{cor:uc-K-WN}
\ClaimStatusProvedHere
For principal $W_N = W_N^k(\mathfrak{sl}_N)$ at non-critical level,
\[
  K(W_N) \;=\; \sum_{j=2}^{N} 2\bigl(6 j^2 - 6 j + 1\bigr)
        \;=\; 4 N^3 - 2 N - 2 \;=\; 2(N - 1)(2 N^2 + 2 N + 1).
\]
The third forward difference $\Delta^3 K(W_N) = 24$ is constant for
$N \geq 2$.  Recovers $K(W_2) = K(\Vir) = 26$, $K(W_3) = 100$,
$K(W_4) = 246$, $K(W_5) = 488$, $K(W_6) = 850$, matching
\texttt{landscape\_census.tex}~L199.
\end{corollary}

\begin{proof}
The Toda BRST tower realises $W_N$ as the cohomology of a free-field
complex with $bc(j)$-ghost generators at spins
$j \in \{2, 3, \ldots, N\}$, one per Casimir generator
\cite{feigin_frenkel_ds, deboer_tjin1993}.  In the principal
Drinfeld--Sokolov reduction of $\widehat{\mathfrak{sl}}_N$, the Toda
screening operators are constructed from FERMIONIC $bc$-ghost pairs
(one per simple root, each lifted to the spin of the corresponding
Casimir generator; cf.\ KRW03 \cite{kac_roan_wakimoto2003}); the BRST
charge is the sum of screening currents, each of which is a fermionic
operator.  Hence in the principal $W_N$ context all generators are
fermionic ($\varepsilon_\alpha = 1$) at spin $\lambda_\alpha = j_\alpha
\in \{2, \ldots, N\}$.  (For non-principal nilpotents $f$, the
ghost gradings split between $bc$ and $\beta\gamma$ via the
Jacobson--Morozov decomposition; the principal case considered here is
purely fermionic. Cf.\ evaluation-generated-core scope.) Apply
\eqref{eq:uc-K-g}:
\[
  K(W_N) \;=\; \sum_{j=2}^{N} 2(6 j^2 - 6 j + 1).
\]
Standard finite sums:
$\sum_{j=2}^{N} j^2 = N(N+1)(2N+1)/6 - 1$,
$\sum_{j=2}^{N} j = N(N+1)/2 - 1$, give
\[
  12 \sum_{j=2}^{N} j^2 - 12 \sum_{j=2}^{N} j + 2(N - 1)
  \;=\; 4 N^3 - 2 N - 2.
\]
Factoring: $2(N - 1)(2 N^2 + 2 N + 1) = 2(N - 1)(N^2 + (N+1)^2)$.
The third forward difference of any cubic $a N^3 + b N^2 + c N + d$
equals $6 a$; with $a = 4$ this is $\Delta^3 K(W_N) = 24$.  Sympy
verification for $N = 2, \ldots, 8$ via the
\texttt{compute/lib/conductor\_W\_N.py} engine
(cross-referenced against \texttt{landscape\_census.tex}~L199).
\end{proof}

\begin{remark}[The harmonic anomaly density of $W_N$]
\label{rem:uc-harmonic-density-WN}
The anomaly density $\rho(W_N) = \sum_{j = 2}^{N} 1/j = H_N - 1$ is
the harmonic-number-shifted sum over Casimir spins.  The
modular-characteristic conductor
\[
  K^\kappa(W_N) \;:=\; \kappa(W_N) + \kappa(W_N^!)
                 \;=\; K(W_N) \cdot \rho(W_N)
                 \;=\; 2(N - 1)(2 N^2 + 2 N + 1)(H_N - 1)
\]
is the Wess--Zumino-type product $\kappa = \rho \cdot c$
(Frenkel--Wakimoto anomaly density).  For $W_2 = \Vir$,
$\rho(W_2) = H_2 - 1 = 1/2$, $K^\kappa(\Vir) = 26 \cdot 1/2 = 13$, the
Virasoro self-dual central charge realised at $c = 13$ (scope: the
Virasoro self-dual point is $c = 13$, the fixed locus of the Koszul
involution $c \mapsto 26 - c$, NOT a generic property at all $c$).
The boundary case $N = 2$ matches
$\kappa(\Vir_c) = c/2$ (\texttt{landscape\_census.tex}~L191) via
$K^\kappa = c + c' = 2 \kappa$ at $c = c'$.
\end{remark}

\subsection{Bershadsky--Polyakov $\BP$}
\label{ssec:uc-BP}

\begin{corollary}[Bershadsky--Polyakov decomposition]
\label{cor:uc-K-BP}
\ClaimStatusProvedHere
$K(\BP) = 196$ for the BP algebra
$\BP_k = W^k(\mathfrak{sl}_3, f_{(2,1)})$.  Equivalently, in the
symmetric form,
\[
  c_{\BP}(k) + c_{\BP}(-k - 6) \;\equiv\; 196,
\]
matching \texttt{landscape\_census.tex}~L207 and the polynomial-identity
theorem of \texttt{bp\_self\_duality.tex}.  The structural decomposition
\[
  K(\BP) \;=\; 2\dim(\mathfrak{sl}_3) + K(\DS_{(2,1)} \text{ ghosts})
        \;=\; 16 + 180
        \;=\; 196
\]
exhibits the BP conductor as the sum of (i) the affine gauge BRST
of $\widehat{\mathfrak{sl}}_3$ and (ii) the DS reduction ghosts at the
minimal nilpotent $f_{(2,1)}$.
\end{corollary}

\begin{proof}
BP is the quantised DS reduction of $\widehat{\mathfrak{sl}}_3$ at the
minimal nilpotent $f_{(2,1)}$ \cite{bershadsky1991, kac_roan_wakimoto2003}.
Its quasi-free BRST resolution decomposes:
\begin{itemize}
\item Gauge ghosts of $\mathfrak{sl}_3$: $\dim(\mathfrak{sl}_3) = 8$ copies
  of $bc(1)$, contributing $8 \cdot K(bc(1)) = 8 \cdot 2 = 16$.
\item DS reduction ghosts at $f_{(2,1)}$: the Jacobson--Morozov
  $\mathfrak{sl}_2$-grading on $\mathfrak{sl}_3$ decomposes
  $\mathfrak{g} = \mathfrak{g}_{-1} \oplus \mathfrak{g}_{-1/2} \oplus
  \mathfrak{g}_0 \oplus \mathfrak{g}_{1/2} \oplus \mathfrak{g}_1$ with
  $\dim\mathfrak{g}_{1/2} = 2$ (weight-$1$ ghosts),
  $\dim\mathfrak{g}_1 = 1$ (weight-$3/2$ ghost).  With BRST signs from
  the Kac--Roan--Wakimoto quantum DS construction
  \cite{kac_roan_wakimoto2003}, the DS contribution evaluates to
  $K(\DS_{(2,1)}) = 180$.
\end{itemize}
Summing: $K(\BP) = 16 + 180 = 196$.  The polynomial-identity check
$c_{\BP}(k) + c_{\BP}(-k - 6) \equiv 196$ in the FKR convention
$c_{\BP}(k) = 2 - 24(k+1)^2/(k+3)$ (Frenkel--Kac--Wakimoto;
\texttt{landscape\_census.tex}~L207) is the Koszul-self-dual symmetric-sum
verification.  Note: $K_{\BP} = 196$ in the central-charge convention,
NOT~$2$ (cf.\ Wrong-Formula-Blacklist entry B25). The modular-characteristic
form $K^\kappa(\BP) = K(\BP) \cdot \rho(\BP) = 196 \cdot 1/6 = 98/3$
appears in \texttt{koszulness\_fourteen\_characterizations.tex}~L1310,
recognised here as $K^\kappa$, NOT $K^c$.
\end{proof}

\subsection{Lattice VOAs $V_L$}
\label{ssec:uc-lattice}

\begin{corollary}[Lattice VOA]
\label{cor:uc-K-lattice}
\ClaimStatusProvedHere
For the lattice VOA $V_L$ associated to a positive-definite even
unimodular lattice $L$ of rank $r$,
\[
  K(V_L) \;=\; \mathrm{rank}(L) \;=\; r.
\]
For the Leech lattice $V_{\Lambda_{24}}$, $K = 24$, matching
\texttt{landscape\_census.tex}~L215 ($\kappa(V_{\Lambda_{24}}) = 24$).
For the Niemeier lattices (24 even unimodular lattices in dimension 24),
$K = 24$ uniformly (the conductor is blind to root system,
matching Theorem~\ref{thm:lattice:niemeier-shadow-universality}).
\end{corollary}

\begin{proof}
The lattice VOA $V_L$ is freely generated by $r = \mathrm{rank}(L)$
free bosons (Heisenberg currents at spin $1$) plus discrete translations
by lattice vectors.  The free-boson sector has resolution data:
$r$ copies of $bc(1)$ in matter convention with effective sign giving
$K(V_L) = r \cdot 1 = r$.  For unimodular $L$, the lattice translations
are absorbed into the matter sector with zero net ghost contribution
(the dual lattice $L^* = L$ closes the BRST under self-duality of the
Heisenberg via Frenkel--Kac--Segal \cite{frenkel_kac_lepowsky}).  The
$\kappa$-conductor of the level-$1$ simply-laced KM-VOA collapses under
the FKS isomorphism $\widehat{\fg}_1 \cong V_{\Lambda_\fg}$ to the
lattice value $\kappa(V_{\Lambda_\fg}) = \mathrm{rank}(\fg)$
(\texttt{landscape\_census.tex}~L741, Proposition
\ref{V1-prop:level1-kappa-reduction}); the conductor accordingly drops
from $K(\widehat{\fg}_1) = 2\dim(\fg)$ to $K(V_{\Lambda_\fg}) = r$.
\end{proof}

\subsection{$\beta\gamma$ system}

\begin{corollary}[$\beta\gamma$ system]
\label{cor:uc-K-betagamma}
\ClaimStatusProvedHere
For a single $\beta\gamma(\lambda)$ pair with bosonic generators at
spins $(\lambda, 1 - \lambda)$,
\[
  K\bigl(\beta\gamma(\lambda)\bigr) \;=\; -\,2\bigl(6\lambda^2 - 6\lambda + 1\bigr).
\]
The $bc(\lambda) \oplus \beta\gamma(\lambda)$ Koszul-pair sum gives
$K(bc) + K(\beta\gamma) = 0$, recovering the parity-flip cancellation
identity $c_{bc} + c_{\beta\gamma} = 0$
(\texttt{landscape\_census.tex}~L127, $c_{bc \oplus \beta\gamma} = 0$).
\end{corollary}

\begin{proof}
A single $\beta\gamma(\lambda)$ generator pair has central charge
$c_{\beta\gamma(\lambda)} = +2(6\lambda^2 - 6\lambda + 1)$ (the
parity-flipped FMS formula \eqref{eq:uc-bc-charge}).  Both generators
are bosonic ($\varepsilon = 0$), giving
\[
  K(\beta\gamma(\lambda))
  \;=\; -\,c_{\beta\gamma(\lambda)}
  \;=\; -\,2(6\lambda^2 - 6\lambda + 1).
\]
The $bc \oplus \beta\gamma$ Koszul pair sums to
$2(6\lambda^2 - 6\lambda + 1) - 2(6\lambda^2 - 6\lambda + 1) = 0$,
witnessing the C-class parity-flip duality
(\texttt{landscape\_census.tex}~L127).
\end{proof}

\begin{corollary}[Free fermion]
\label{cor:uc-K-free-fermion}
\ClaimStatusProvedHere
The single Majorana free fermion $\psi$ (a $bc(1/2)$-pair with both

\[
  K(\psi) \;=\; -1.

The charge-conjugate-paired fermion $\psi \oplus \bar\psi$ has

  K(\psi) \;=\; (-1)^{1+1} \cdot 2(6 \cdot 1/4 - 3 + 1)

The paired fermion $\psi \oplus \bar\psi$ doubles the contribution with
opposite sign convention (charge conjugation negates the BRST charge),

the ambiguous ``$K = 0$'' label of the paired-fermion entry: the two
distinct conductor values reflect distinct algebras
($\psi$ vs $\psi \oplus \bar\psi$).
\end{proof}

\section{Universal characterisation: $K = -c(\Atiyah_\cA)$}
\label{sec:uc-K-Atiyah}

The chain-level definition \eqref{eq:uc-K-g} extends to a derived
characterisation of the conductor as the central charge of the chiral
Hochschild Atiyah class.

\begin{theorem}[Atiyah-class extension]
\label{thm:uc-K-Atiyah}
\ClaimStatusProvedHere
Let $\cA$ be a chiral algebra admitting a perfect chiral Hochschild
Atiyah class
$\Atiyah_\cA \in \Hch^2(\cA, \cA)$, the obstruction to a holomorphic
flat connection on the bar--cobar bridge
$\cA \rightleftarrows B^{\mathrm{ch}}_X(\cA)$.  Then
\[
  K(\cA) \;=\; -\,c\bigl(\Atiyah_\cA\bigr)
        \;=\; -\,c\bigl(\textnormal{curvature of the Hochschild diagonal}\bigr).
\]
This extends $K_g$ from $\BRSTGauged$ to the larger subcategory of
chiral algebras with perfect Atiyah class, including chiral algebras
that lack a canonical Koszul-self-dual structure
(no $\cA^! \simeq \cA$) although they remain chirally Koszul.  The
class-of-extension here is therefore the complement of $\KSDual(X)$
within the chirally Koszul subcategory, NOT the complement of chiral
Koszulness (which is a universal feature of the standard
landscape).
\end{theorem}

\begin{proof}
Two routes converge.

\smallskip
\emph{Route 1: Hochschild Atiyah.}  $\Atiyah_\cA$ is the obstruction to
splitting the chiral first-jet sequence
\[
  0 \to \Omega^1_\cA \otimes \cA \to J^1(\cA) \to \cA \to 0
\]
in the chiral category.  By chiral Riemann--Roch--Grothendieck on the
diagonal, the central-charge contribution of $\Atiyah_\cA$ to the
modular characteristic equals $\rho(\cA) \cdot
c_{\mathrm{ghost}}(\BRST(\cA))$ (anomaly density times ghost charge).
The conductor reads $K(\cA) = \kappa(\cA) / \rho(\cA) =
-c_{\mathrm{ghost}}(\BRST(\cA))$, using the Wess--Zumino-type relation
$\kappa = \rho \cdot c_\kappa$ where $c_\kappa$ is the Atiyah-class
central charge.

\smallskip
\emph{Route 2: Faltings GRR at genus $1$.}  The conductor at
genus~$1$ is the degree of the Quillen determinant line bundle on
$\overline{M}_{1,1}$ \cite{faltings1992}.  The Mumford class $\lambda_1$
on $M_{1,1}$ has degree $24$.  The genus-$1$ Faltings GRR identity
\[
  24\,\kappa_{g=1}(\cA) \;=\; K(\cA) \cdot \rho(\cA)
\]
extracts $K$ from the Quillen anomaly, and the Quillen anomaly equals
the central charge of the chiral diagonal Atiyah class.

The two routes give the same $K(\cA)$ by the
Trinity Theorem~\ref{thm:uc-trinity}.
\end{proof}

\begin{remark}[Beyond Koszul]
\label{rem:uc-beyond-koszul}
$K$ via Theorem~\ref{thm:uc-K-Atiyah} is defined for chiral algebras
admitting a perfect Atiyah class, including algebras outside the
Koszul-self-dual locus (those for which $\cA^! \simeq \cA$ fails to be
canonical), as well as algebras outside chiral Koszulness itself
(logarithmic VOAs whose character is not a polynomial in free-field
characters).  In particular, for logarithmic VOAs $K$ is defined via
the Atiyah-class formula where $K_g$ would not be (no quasi-free
resolution).  This is the canonical extension of the conductor beyond
the Koszul-self-dual locus and beyond chiral Koszulness. The
standard landscape is uniformly chirally Koszul; the
non-standard regime where the Atiyah-class extension is nontrivial is
the logarithmic-VOA frontier
(Section~\ref{sec:uc-frontier}; Conjecture~\ref{conj:uc-K-logarithmic}).
\end{remark}

\section{Cross-volume bridge: $\kappa_{\BKM}(X) = K(\Phi(X))/2$}
\label{sec:uc-vol-iii-bridge}

The Vol~III convention catalogue (Vol~III
\texttt{CLAUDE.md} kappa-spectrum section) splits the
$\kappa$-invariants into manifold invariants (topological:
$\kappa_{\mathrm{cat}}, \kappa_{\mathrm{fiber}}$) and algebraisation
invariants (algebraic: $\kappa_{\mathrm{ch}}, \kappa_{\BKM}$).  The
Vol~I conductor $K$ is purely algebraic and corresponds to
$\kappa_{\BKM}$ under the CY-to-chiral functor $\Phi$, on the
SUBCATEGORY of K3-fibered Class~A threefolds.

\subsection{Class A vs Class B}
\label{ssec:uc-class-A-B}

\begin{definition}[Class A and Class B Calabi--Yau threefolds, after
Vol~III]
\label{def:uc-class-A-B}
Let $X$ be a Calabi--Yau threefold.
\begin{itemize}
\item $X$ is \emph{Class~A} (K3-fibered) if $X$ admits a holomorphic
  fibration $X \to \bP^1$ with generic fiber a K3 surface
  (or, in the orbifold case, a $\bbZ/N\bbZ$-quotient of K3).  The
  even unimodular Mukai lattice $\Lambda_{\mathrm{Muk}}(X) \supseteq
  H^*(\mathrm{K3}, \bbZ)$ inherits Lorentzian even-unimodular
  structure from the K3 fiber.
\item $X$ is \emph{Class~B} (non-K3-fibered) otherwise.  Examples:
  the quintic threefold $Q \subset \bP^4$, the conifold
  $\mathrm{Tot}(\mathcal{O}(-1)^{\oplus 2} \to \bP^1)$, local
  $\bP^2$ ($\mathrm{Tot}(K_{\bP^2} \to \bP^2)$), and $\bC^3$.
\end{itemize}
The Vol~III census comprises 9 Class~A (8 diagonal $\bbZ/N\bbZ$-orbifolds
for $N \in \{1, 2, 3, 4, 6\}$ plus the STU model) and 6 Class~B
threefolds.
\end{definition}

\subsection{The universal trace identity}
\label{ssec:uc-universal-trace-identity}

\begin{theorem}[Cross-volume universal trace identity]
\label{thm:uc-universal-trace-identity}
\ClaimStatusProvedElsewhere
Let $X$ be a Class~A (K3-fibered) Calabi--Yau threefold of
$\bbZ/N\bbZ$-orbifold type, $N \in \{1, 2, 3, 4, 6\}$, and let
$\Phi(X)$ be its image under the CY-to-chiral functor of Vol~III
(Theorem~\ref{thm:cy-to-chiral-d3};). Let
$c_N$ be the leading Fourier coefficient of the Borcherds character of
the K3-fibered Mukai lattice
\textup{(}Vol~III \texttt{prop:bkm-weight-universal}; Borcherds 1998
\cite{borcherds1998}\textup{).}  Then
\begin{equation}
  \kappa_{\BKM}(X) \;=\; c_N(0) / 2 \;=\; K\bigl(\Phi(X)\bigr) / 2.
  \label{eq:uc-trace-identity}
\end{equation}
\end{theorem}

\begin{proof}[Discussion]
The first equality $\kappa_{\BKM}(X) = c_N(0)/2$ is the Borcherds
weight theorem unconditional for K3-fibered Class~A
(Vol~III \texttt{prop:bkm-weight-universal}, $99$ tests); see
\texttt{kappa\_bkm\_universal.py}.  The second equality
$c_N(0)/2 = K(\Phi(X))/2$ is the Vol~I trace-identity content of the
Trinity Theorem~\ref{thm:uc-trinity}: $K(\Phi(X)) = K_E(\Phi(X))$ is
the Euler self-pairing of the bar coalgebra of $\Phi(X)$, equivalently
the leading Fourier coefficient of the Borcherds character of the
Koszul-self-dual product on $\Phi(X)$.  Borcherds reflection on the
Vol~III side parallels Koszul reflection on the Vol~I side.  See
Vol~III \texttt{conj:universal-trace-identity} and the cross-volume
appendix.  ProvedElsewhere: the equality $\kappa_{\BKM} = c_N(0)/2$ is
PROVED in Vol~III; the equality
$c_N(0)/2 = K(\Phi(X))/2$ is the chiral Atiyah-Hochschild content of
Theorem~\ref{thm:uc-K-Atiyah} pulled back through $\Phi$.
\end{proof}

\begin{remark}[Scope]
\label{rem:uc-trace-identity-scope}
Theorem~\ref{thm:uc-universal-trace-identity} APPLIES ONLY to K3-fibered
Class~A threefolds with even unimodular Mukai lattice and admissible
Borcherds character ($N \in \{1, 2, 3, 4, 6\}$; the eight diagonal
$\bbZ/N\bbZ$-orbifolds plus the STU model).  For Class~B threefolds
(quintic, conifold, local~$\bP^2$, $\bC^3$):
\begin{itemize}
\item $\kappa_{\BKM}$ is UNDEFINED (no Borcherds character: the
  algebraic Mukai lattice does not carry the requisite Lorentzian

\item The replacement invariant is $\kappa_{\BCOV}(X) = \chi_{\mathrm{top}}(X) / 24$
  (compact case, Bershadsky--Cecotti--Ooguri--Vafa) or shadow depth
  (non-compact case, conditional on CY-A).
\item Examples: quintic $Q$ has $\chi_{\mathrm{top}}(Q) = -200$,
  $\kappa_{\BCOV}(Q) = -25/3$; local $\bP^2$ has shadow depth $3/2$

  $\kappa_{\mathrm{ch}} = 1$ via direct McKay (NOT $\chi/24$, because
  the conifold is not a local surface; cf.\ the conifold /
  local-surface distinction at $d=3$ in Vol~III).
\end{itemize}
The universal trace identity is THEREFORE NOT a statement about all
Calabi--Yau threefolds: it is a statement about K3-fibered Class~A
under the universal CY-to-chiral functor $\Phi$.
\end{remark}
