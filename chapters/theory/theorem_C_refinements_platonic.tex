\section{Scalar, Hochschild, self-dual, and K3 comparison surfaces}
\label{sec:theorem-C-surface-separation}

\begin{proposition}[Five separated Theorem~C surfaces;
\ClaimStatusConditional]
\label{prop:theorem-C-surface-separation}
\index{Theorem C!surface separation}
\index{derived center!not a scalar conductor}
\index{K3 Hall--Drinfeld double!not a Yangian}
Let $(\cA,\cA^!)$ be a chiral Koszul pair for which
Theorem~\textup{\ref{thm:quantum-complementarity-main}} is defined.
Then the following data are logically distinct.
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Verdier complementarity.}
 Theorem~\textup{\ref{thm:quantum-complementarity-main}} gives the
 decomposition
 \[
 \mathbf{C}_g(\cA)\simeq
 \mathbf{Q}_g(\cA)\oplus\mathbf{Q}_g(\cA^!)
 \]
 and, for $g\geq 1$ under the stated non-degeneracy hypothesis, the
 cross-pairing of degree $-(3g{-}3)$.

\item \emph{Scalar conductor.}
 The number
 \[
 K^\kappa(\cA)=\kappa(\cA)+\kappa(\cA^!)
 \]
 is a scalar shadow of the bar package. On the Vol.~I scalar ceiling
 its possible landmark values are
 $\{0,13,250/3,98/3\}$:
 free-field and affine rows give~$0$, Virasoro gives~$13$,
 principal $\cW_3$ gives~$250/3$, and Bershadsky--Polyakov gives
 $98/3$. The Vol.~III Mukai-enhanced $\mathsf B$-row adjoins
 \[
  K^{\kappa_{\mathrm{ch}}}
  =2c_+(\mathrm{Mukai}(K3))=8,
 \]
 producing the five-element extension
 $\{0,8,13,250/3,98/3\}$.

\item \emph{Derived center.}
 The object
 $Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)$ is the chiral Hochschild
 cochain complex, hence a brace dg algebra in completed chiral
 cochains once the chiral Hochschild model has been fixed. A physical
 bulk realization requires an open-closed comparison map
 \[
  \mathrm{OC}_{\cA}\colon
  Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)
  \longrightarrow
  \mathcal{B}^{\mathrm{phys}}_{\cA}
 \]
 to an analytically constructed closed-sector factorization algebra.
 In the scalar finite-window calculations,
 Proposition~\textup{\ref{prop:derived-center-explicit}} gives
 Heisenberg cohomology of shape $(1,1,1,0,\ldots)$ and Virasoro
 cohomology of shape $(1,0,1,0,\ldots)$. These dimensions are
 Hochschild invariants and leave the scalar
 $K^\kappa(\cA)$ undetermined.

\item \emph{Self-dual halving.}
 The halving formula
 \[
 \dim Q_g(\cA)=\dim Q_g(\cA^!)
=\tfrac12\dim H^*(\overline{\mathcal{M}}_g,Z(\cA))
 \]
 is the conditional conclusion of
 Theorem~\textup{\ref{thm:self-dual-halving}} and requires a specified
 isomorphism $f\colon\cA\simeq\cA^!$ of chiral algebras whose induced
 automorphism $\tau_f$ on $\mathbf{C}_g(\cA)$ satisfies
 \[
  \tau_f\sigma=-\sigma\tau_f .
 \]
 In the witness landscape this applies on the real same-family locus to
 $\mathrm{Vir}_{13}$, where
 $\kappa(\mathrm{Vir}_{13})=13/2$. Heisenberg at $k=0$, the affine
 critical level, and $\beta\gamma_{\lambda=1/2}$ are scalar or
 parameter fixed points; they do not provide the required
 chiral-algebra self-duality and anti-equivariant automorphism.

\item \emph{K3/Hall--Drinfeld comparison.}
 The $\mathsf B$-row value~$8$ is a conditional K3-facing
 enlargement through the Mukai lattice and the Vol.~III
 Hall--Drinfeld construction. The K3 BKM object has infinite-rank
 generalized Cartan data, imaginary simple roots, and no Drinfeld
 $J$-presentation. The taxonomic target is the completed K3 chiral
 Hall--Drinfeld double, not a Drinfeld Yangian
 $Y_\hbar(\mathfrak g_{\Delta_5})$.
\end{enumerate}
\end{proposition}

\begin{proof}
Clause~\textup{(i)} is the statement of
Theorem~\ref{thm:quantum-complementarity-main}. Clause~\textup{(ii)}
is the scalar complementarity landscape:
Proposition~\ref{prop:complementarity-landscape} gives the free,
affine, Virasoro, and principal $\cW_N$ formulas, while
Proposition~\ref{prop:archetype-complementarity-bridge} gives the
$\cW_3$, Bershadsky--Polyakov, and Mukai-enhanced $\mathsf B$ rows.
The displayed constants are the direct evaluations
\[
\frac{c}{2}+\frac{26-c}{2}=13,\qquad
\frac56\cdot 100=\frac{250}{3},\qquad
\frac16\cdot 196=\frac{98}{3},\qquad
2c_+(\mathrm{Mukai}(K3))=8 .
\]
Clause~\textup{(iii)} is
Proposition~\ref{prop:derived-center-explicit}; its output is a
graded Hochschild cohomology object. The scalar $K^\kappa$ is supplied
by the modular-characteristic computation. Clause~\textup{(iv)} is
Theorem~\ref{thm:self-dual-halving} together with
Theorem~\ref{thm:census-self-dual-locus}, which separates
Koszul self-duality, anti-equivariant halving data, and scalar fixed
points. Clause~\textup{(v)} is the K3 taxonomy fixed in the
concordance and in the Vol.~III Hall--Drinfeld double construction:
the K3 target is quasi-Hopf/Hall--Drinfeld, with no finite-type
Drinfeld $J$-presentation asserted.
\end{proof}

\section{Derived center: naive recovery; brace equivalence conjecture}
\label{sec:derived-center-koszul-equivalence}

Let $(\cA, \cA^!)$ be a chiral Koszul pair on a smooth complex
curve~$X$ in the sense of
Definition~\ref{def:chiral-koszul-pair}. Write
$Z(\cA) = \{a \in \cA \mid [a, b] = 0\ \forall b\}$ for the naive
center and $Z^{\mathrm{der}}_{\mathrm{ch}}(\cA) =
\ChirHoch^\bullet(\cA, \cA)$ for the chiral-Hochschild derived
center.

\begin{lemma}[Central-pairing criterion for the naive centers;
\ClaimStatusConditional]
\label{lem:naive-center-koszul-identification}
\index{center!naive!Koszul identification}%
For a chiral Koszul pair $(\cA, \cA^!)$ on the Koszul locus, assume
that the naive centers are finite-dimensional, or more generally are
separated and complete for the conformal-weight topology with
finite-dimensional graded pieces. Assume also the central
orthogonality condition
\[
 Z(\cA)^\perp \cap Z(\cA^!) = 0,
 \qquad
 Z(\cA^!)^\perp \cap Z(\cA) = 0
\]
inside the Koszul pairing $\langle -, - \rangle_K$ of
Theorem~\ref{thm:chiral-koszul-duality}. Then
$\langle -, - \rangle_K$ restricts weightwise to a non-degenerate
pairing
\[
 \langle -, - \rangle_K \colon Z(\cA) \otimes Z(\cA^!)
 \longrightarrow \mathbb{C}
\]
and induces a natural isomorphism from the completed center
$\widehat{Z(\cA)}$ to the continuous graded dual
$\widehat{Z(\cA^!)}^{\,\vee,\mathrm{cont}}$.
\end{lemma}

\begin{proof}
The Koszul pairing is non-degenerate on the full pair
$(\cA, \cA^!)$ by Theorem~\ref{thm:chiral-koszul-duality}. The two
central orthogonality equalities say exactly that no non-zero element
of either naive center pairs trivially against the opposite naive
center. Hence the restricted bilinear form is non-degenerate on both
factors. In finite dimension this gives the ordinary algebraic dual.
In the completed case it gives the inverse-limit statement over
finite conformal-weight windows; the dual is the continuous graded
dual of the completed center. This is an $H^0$ center statement; it
does not assert the Verdier eigenspace decomposition or the
$g\geq 1$ Lagrangian pairing of Theorem~C.
\end{proof}

\begin{conjecture}[Derived center brace-algebra equivalence]
\label{conj:derived-center-koszul-equivalence}%
\ClaimStatusConjectured
\index{derived center!Koszul equivalence!conjecture}%
For a chiral Koszul pair $(\cA, \cA^!)$ on the Koszul locus, there
exists a natural zig-zag of quasi-isomorphisms of brace dg algebras
\[
 Z^{\mathrm{der}}_{\mathrm{ch}}(\cA) \simeq
 Z^{\mathrm{der}}_{\mathrm{ch}}(\cA^!)
\]
that becomes an equivalence of $E_2$-algebras after Deligne--Tamarkin
rectification. On~$H^0$ this zig-zag is required to be compatible
with the completed central pairing of
Lemma~\ref{lem:naive-center-koszul-identification} whenever the
central orthogonality hypotheses of that lemma hold.
\end{conjecture}

\begin{proposition}[Derived-Morita criterion for the brace center;
\ClaimStatusConditional]
\label{prop:derived-center-morita-criterion}
\index{derived center!Morita criterion}
Let $(\cA,\cA^!)$ be a chiral Koszul pair on the Koszul locus. Assume
there is a perfect chiral $\cA$--$\cA^!$ bimodule
$\mathcal K_{\cA}$ whose left and right duals exist and whose tensor
functors induce inverse equivalences
\[
 \Perf_{\mathrm{ch}}(\cA)
 \;\simeq\;
 \Perf_{\mathrm{ch}}(\cA^!)
\]
in the Morita $(\infty,2)$-category of chiral dg categories. Assume
also that the chiral Hochschild cochain complex is computed by the
endomorphism operad model of
Theorem~\ref{thm:thqg-brace-dg-algebra}, and that this model is
Morita functorial. Then there is a natural zig-zag of brace
dg-algebra quasi-isomorphisms
\[
 Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)
 \;\simeq\;
 Z^{\mathrm{der}}_{\mathrm{ch}}(\cA^!).
\]
After Deligne--Tamarkin rectification this is an equivalence of
$E_2$-algebras. If, in addition, the central orthogonality
hypotheses of Lemma~\ref{lem:naive-center-koszul-identification}
hold and $\mathcal K_{\cA}$ is normalized by the Koszul evaluation
pairing, then the induced $H^0$ isomorphism is compatible with the
completed central pairing of that lemma. Without this last
normalization, Morita invariance supplies only the degree-zero
center isomorphism; a preferred bilinear pairing requires the
normalization.
\end{proposition}

\begin{proof}
Keller and To\"en identify derived Morita equivalences with
equivalences of the corresponding dg categories of perfect modules
\cite{Keller06,Toen07}. Hochschild cochains are the derived
endomorphism object of the identity functor on this category. An
equivalence transports the identity functor, its derived endomorphism
object, and the operadic insertions defining the brace operations.
Thus the Morita equivalence induced by $\mathcal K_{\cA}$ gives a
brace dg-algebra quasi-isomorphism between the Hochschild cochains of
$\Perf_{\mathrm{ch}}(\cA)$ and $\Perf_{\mathrm{ch}}(\cA^!)$.
The stated chiral endomorphism-operad comparison identifies these
Hochschild cochains with
$Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)$ and
$Z^{\mathrm{der}}_{\mathrm{ch}}(\cA^!)$. Deligne--Tamarkin
rectification sends brace quasi-isomorphisms to $E_2$
quasi-isomorphisms. The degree-zero term is the endomorphism ring of
the unit object, hence the naive center after completion. The Morita
construction identifies these completed degree-zero centers. The
further identification with the bilinear central pairing is exactly
the added central-orthogonality and Koszul-normalization condition,
and is therefore not a formal consequence of Morita invariance alone.
\end{proof}

\begin{remark}[Morita obstruction]
\label{rem:derived-center-koszul-scope}
Proposition~\ref{prop:derived-center-morita-criterion} isolates the
load-bearing obstruction. The brace center equivalence follows from a
perfect chiral Morita kernel together with Morita functoriality for
the chiral endomorphism operad. The proposition is a sufficient
criterion. A direct brace-level proof must construct the same
zig-zag, preserve operadic insertions, and recover the normalized
$H^0$ central pairing.

Three checks fix the normalization. On $H^0$ the degree-zero
comparison reduces to
Lemma~\ref{lem:naive-center-koszul-identification}. For the Heisenberg
family $\cA=\cH_k$, the chiral Hochschild complex has shape
$(1,1,1,0,\ldots)$, so the brace center can be computed directly. At
$E_2$ cohomology level, Chiral Higher Deligne in Vol.~II supplies the
individual $E_3$-chiral structure on each derived center. The
cross-pair map still has to come from the Morita kernel. For
class~$\mathsf M$ the raw direct-sum complex lacks the required
continuous homotopies; the completed/pro surface is the ambient for
the Morita kernel.
\end{remark}

\begin{remark}[Intrinsic-chiral routes and Deligne--Tamarkin;
falsification test]
\label{rem:derived-center-sharpened-scope}%
\index{derived center!falsification test}%
Two intrinsic chiral inputs clarify the Deligne--Tamarkin dependence.
\emph{Hinich rectification~\textup{\cite{Hin97}}} gives a
chain-level $E_2$ model of any brace dg algebra, and hence a
chain-level $E_2$ model of each side of the putative equivalence
separately. The comparison zig-zag still comes from the Morita kernel
of Proposition~\ref{prop:derived-center-morita-criterion}.
\emph{Gaitsgory--Rozenblyum~\textup{\cite{GR17}}}
provide a six-functor formalism on Ran prestacks that lifts
factorization algebras to $(\infty, 2)$-properads; this supports
Theorem~A at the chiral base. The derived-center identification
across Koszul-dual pairs still requires the brace/Morita comparison.
The chiral Higher Deligne theorem of
Vol.~II upgrades $Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)$ to a
universal $E_3$-chiral algebra. The Deligne--Tamarkin rectification
remains the substantive input at the $E_2$ brace level.

\emph{Falsification test.} A Koszul-normalized Morita kernel as in
Proposition~\ref{prop:derived-center-morita-criterion} produces, at
the level of~$H^2$, a pairing
\[
 \langle -, - \rangle_{\ChirHoch^2}
 \colon
 \ChirHoch^2_{\mathrm{ch}}(\cA, \cA)
 \;\otimes\;
 \ChirHoch^2_{\mathrm{ch}}(\cA^!, \cA^!)
 \;\longrightarrow\; \mathbb{C}
\]
by transporting one factor through the brace quasi-isomorphism and
then applying the Koszul evaluation pairing. Compatibility requires
this pairing to coincide with the restriction of the Koszul pairing
to the cohomological-degree-$2$ scalar central-cocycle lines under the
$\sigma$-involution. On the Heisenberg family, where
$\cA = \cH_k$ and
$\cA^! = \operatorname{Sym}^{\mathrm{ch}}(V^*)$ is the chiral
symmetric coalgebra of the Heisenberg linear-dual
\textup{(}the Koszul partner has opposite scalar curvature
$\kappa(\cH_k^!)=-k$ in the scalar complementarity lane, while
$\cH_k^!$ is distinct from the chiral algebra $\cH_{-k}$\textup{)},
the first $H^2$ line is one-dimensional by
Proposition~\textup{\ref{prop:derived-center-explicit}} and the
second is the image line selected by the chosen brace equivalence.
The identity checks the normalization of Koszul evaluation against
the chiral central extension; a numerical mismatch in the ratio of
the two sides falsifies the normalized Morita criterion at its
simplest test point. For Virasoro at generic~$c$ the same test is
expected to return the family-dependent constant
$c/2 = \kappa(\mathrm{Vir}_c)$ on the appropriate cocycle
normalization. This is a normalization test for the $H^2$ cocycle
pairing. The scalar conductor remains a separate invariant of the
family. A mismatch falsifies the conjecture on
class~$\mathsf{M}$.
\end{remark}

\section{Perfectness on the standard landscape: family-by-family}
\label{sec:perfectness-standard-landscape}

The perfectness criterion
Lemma~\ref{lem:perfectness-criterion} is hypothesis-conditional: it
requires \textup{(i)} PBW filterability with Koszul-acyclic
associated graded, \textup{(ii)} finite-dimensional total flat fiber
cohomology at every closed point of $\overline{\mathcal{M}}_g$, and
\textup{(iii)} a pseudo-coherent bounded family whose filtration is
finite on every cohomological truncation.
Hypothesis~\textup{(i)} is built into the modular pre-Koszul datum
(axioms \textup{MK1}, \textup{MK3} of
Definition~\ref{def:modular-koszul-chiral}). Hypotheses
\textup{(ii)} and~\textup{(iii)} are the substantive content on the
compactification, where total fiber cohomology, base change, and
bounded pseudo-coherence must be checked at nodal fibers of
$\overline{\mathcal{M}}_g$.

The proposition records the family-specific inputs used by the
criterion. TUY finite-rank conformal blocks verify a
total-finiteness input for positive-integer integrable affine
Kac--Moody after the bar-conformal-block exactness hypotheses are
present. Arakawa's
$C_2$-cofiniteness applies to simple admissible quotients. The
universal vacuum algebra requires an additional quotient and
bar-Koszulness comparison. Generic non-critical Kac--Moody,
admissible simple quotients, and class~$\mathsf{M}$ boundary strata
therefore remain separate conditional or conjectural loci.

\begin{proposition}[Perfectness on the standard landscape,
conditional;
\ClaimStatusConditional]
\label{prop:perfectness-standard-landscape}
\index{perfectness!standard landscape|textbf}%
Let $\cA$ be an object of the standard landscape in the sense of
Part~\ref{part:standard-landscape}
\textup{(}Heisenberg, affine Kac--Moody,
Virasoro, principal $\cW_N$, lattice, $\beta\gamma$\textup{)}. Assume
the three hypotheses of
Lemma~\textup{\ref{lem:perfectness-criterion}}: PBW filterability,
total finite-dimensional flat fiber cohomology at every closed point
of $\overline{\mathcal{M}}_g$, and pseudo-coherent boundedness with
finite filtration on cohomological truncations. Then
$R\pi_{g*}\bar{B}^{(g)}_{\mathrm{flat}}(\cA)$ is a perfect complex
on~$\overline{\mathcal{M}}_g$. The following entries describe the
family-specific inputs for the total-fiber-finiteness part of the
criterion on the stated loci:
\begin{enumerate}[label=\textup{(\alph*)}]
\item \emph{Heisenberg $\cA = H_k$.} Each conformal-weight piece
 $H_{k,h}$ is finite-dimensional by the Poincar\'e series
 $\prod_{n\geq 1}(1-q^n)^{-\operatorname{rank}}$, and the bar
 coalgebra $\bar{B}(H_k)$ has finite-dimensional cohomology at each
 fixed conformal weight. This is a weightwise input. Total
 flat-fiber finiteness follows only after the
 bar-conformal-block exactness and degree-zero concentration
 hypotheses used in the proof below. At nodal boundary strata of
 $\overline{\mathcal{M}}_g$ the fiber bar complex reduces via the
 sewing axiom to the Heisenberg bar complex on the normalized
 curve, which is of lower genus and inductively
 finite-dimensional in each fixed sector.
\item \emph{Affine Kac--Moody at positive integer level
 $\cA = V_k(\fg)$, $k \in \mathbb{Z}_{> 0}$.} At integrable level
 the space of conformal blocks at each stable $n$-marked curve is
 finite-dimensional by Tsuchiya--Ueno--Yamada sheaf-theoretic
 propagation across $\overline{\mathcal{M}}_{g,n}$, and the
 factorization-algebra sewing at nodal fibers preserves finite
 rank because the sewing kernel is the finite sum over integrable
 highest weights. The bar cohomology at each fiber is finite
 dimensional at each conformal weight.

\item \emph{Affine Kac--Moody at admissible level.} Arakawa's
 $C_2$-cofiniteness theorem applies to simple admissible affine
 vertex algebras $L_k(\fg)$. It supplies the finite-conformal-blocks
 input for the simple quotient after the quotient comparison,
 admissible bar-Koszulness, nodal sewing, and pseudo-coherent
 boundedness are supplied. The universal vacuum algebra
 $V_k(\fg)$ requires its own quotient-comparison and bar-Koszulness
 input.
\end{enumerate}
\end{proposition}

\begin{proof}
The main statement is exactly
Lemma~\ref{lem:perfectness-criterion} applied with its three
hypotheses. The remaining paragraphs isolate the family-specific
inputs needed for hypothesis~\textup{(ii)} on the stated standard
rows. The pseudo-coherent boundedness hypothesis~\textup{(iii)}
remains a separate input.

For the Heisenberg case, the Poincar\'e series of~$H_k$ is
$\prod_{n \geq 1}(1 - q^n)^{-\operatorname{rank}}$; each weight
piece is finite-dimensional. The bar coalgebra inherits
finite-dimensional weight pieces. This proves weightwise control
only. Total flat-fiber cohomology is
obtained only after the bar-conformal-block bridge identifies the
surviving degree-$0$ cohomology with the fixed-sector Heisenberg
conformal block and the PBW spectral sequence kills higher
cohomology. The sewing procedure expresses a nodal fiber as the
finite fixed-sector sewing of the normalized curve; induction on the
topological type of $\Sigma$ preserves this fixed-sector finiteness.

For affine Kac--Moody at positive integer level, the conformal
blocks at every stable $n$-marked curve form a vector bundle of
finite rank on $\overline{\mathcal{M}}_{g,n}$ with the TUY
propagation isomorphism identifying the fiber at a nodal curve
with the finite direct sum, over integrable highest weights
$\lambda$, of tensor products of conformal blocks on the
normalized components coupled through the evaluation pairing at
$\lambda$. Proposition~\ref{prop:conformal-blocks-bar}, together
with degree-zero concentration from
Lemma~\ref{lem:perfectness-criterion}, identifies the surviving
flat-bar cohomology with this finite-rank conformal-block space.
This is the total-finiteness input; finite conformal-weight pieces
alone do not imply it.
For admissible level, Arakawa's theorem supplies finite conformal
blocks for the simple quotient $L_k(\fg)$ under its hypotheses. The
passage from that quotient statement to the universal algebra
$V_k(\fg)$, and from conformal blocks to the flat bar family, remains
conditional on the quotient comparison and bar-Koszulness hypotheses
named in the statement.

\emph{Weightwise to total: the bar-conformal-blocks bridge.}
Weightwise finite-dimensionality alone is strictly weaker than the
total finite-dimensional flat fiber cohomology required by
hypothesis~\textup{(ii)} of
Lemma~\ref{lem:perfectness-criterion}; infinitely many nonzero
conformal weights can still make the total fiber
infinite-dimensional. The upgrade requires two inputs. First,
Proposition~\ref{prop:conformal-blocks-bar} identifies
$H^0$ of the flat fiber bar complex with the space of conformal
blocks only under its degreewise-finiteness and exactness hypotheses,
so the TUY finite-rank conclusion controls
$\dim H^0(\bar B^{(g)}_{\mathrm{flat}}(\cA)|_\Sigma)$ as a single
finite-rank conformal-block space. Second, the spectral-sequence collapse
recorded inside Lemma~\ref{lem:perfectness-criterion} concentrates
the flat-bar cohomology in degree~$0$ on the Koszul locus at
integer level, and on any admissible simple-quotient locus where the
same PBW/bar-Koszulness hypotheses have been established: on the
PBW-associated graded
the higher pages vanish by the Orlik--Solomon Koszul
transfer~\textup{(cf.~Theorem~\ref{thm:pbw-koszulness-criterion})},
and the flat filtration is finite, so the degree-$0$ concentration
lifts to the strict flat bar complex. Combining these two, total
fiber cohomology equals $H^0$ equals the space of conformal blocks,
which is finite-dimensional by TUY at positive integer level.  At
admissible level the same conclusion is conditional on the simple
quotient and bar-comparison hypotheses above.

The proposition therefore proves a conditional perfectness criterion.
On Heisenberg and positive-integer affine Kac--Moody it supplies the
main total-finiteness input; ordinary perfectness still uses the
bounded pseudo-coherent family hypothesis of
Lemma~\ref{lem:perfectness-criterion}.  On admissible simple
quotients it supplies only the Arakawa finite-conformal-blocks input;
the universal vacuum algebra and the bar/Ext comparison remain outside
the proved locus.
\end{proof}

\begin{corollary}[$C_0$ on the Heisenberg and affine Kac--Moody
integer locus;
\ClaimStatusConditional]
\label{cor:c0-unconditional-class-G-L}
On the Heisenberg family and on affine Kac--Moody at positive
integer level, the fiber-center identification
Theorem~\ref{thm:fiber-center-identification} follows from
Proposition~\ref{prop:perfectness-standard-landscape} once the
bounded pseudo-coherent family hypothesis of
Lemma~\ref{lem:perfectness-criterion} is supplied. For affine
Kac--Moody at generic non-critical $k$ outside the positive-integer
locus, and for admissible simple quotients, the same conclusion is
conditional on the corresponding boundary-perfectness and quotient
comparison hypotheses.
\end{corollary}

\begin{proof}
Combine
Proposition~\ref{prop:perfectness-standard-landscape}, with the
bounded pseudo-coherent family hypothesis retained, with
Theorem~\ref{thm:fiber-center-identification}.
\end{proof}

\section{Genus zero clarification}
\label{sec:theorem-C-g0}

At $g = 0$, $\overline{\mathcal{M}}_0$ is a point once three marked
points are fixed (or is empty for fewer than three marked points);
the Verdier involution $\sigma$ acts trivially on the unique point
class. The content of Theorem~C at $g = 0$ is therefore
\[
 Q_0(\cA) = H^0\bigl(\mathrm{pt}, \cZ(\cA)\bigr),
 \qquad Q_0(\cA^!) = 0,
\]
as established in the genus-zero clause of
Theorem~\ref{thm:quantum-complementarity-main}.

\begin{remark}[Domain of the Verdier degree formula at $g=0$]
\label{rem:plus-three-shift-nonissue}%
\index{Verdier pairing!g=0 non-issue}%
The formula $-(3g{-}3)$ for the cohomological degree of the Verdier
pairing on~$\mathbf{C}_g(\cA)$, applied formally at $g = 0$, gives
$+3$. This is not a contradiction: the Verdier pairing is part of
the $g \geq 1$ clause of
Theorem~\ref{thm:quantum-complementarity-main}, and there is no
Verdier pairing at $g = 0$ in the chiral Koszul decomposition,
because the involution is trivial and $\mathbf{Q}_0(\cA^!)$
vanishes. The degree formula has domain $g \geq 1$.
\end{remark}

\section{PTVV compatibility criterion}
\label{sec:C-PTVV-alternative}

The Pantev--To\"en--Vaqui\'e--Vezzosi mapping-stack theorem needs a
$d$-oriented $\mathcal{O}$-compact source~$Y$ and an $n$-shifted
symplectic derived Artin target~$Z$; the output shift is $n-d$. A
PTVV route to Theorem~C therefore begins with an intrinsic
$0$-shifted symplectic derived stack $S_\cA$ whose tangent is the
chiral center, an orientation trace on a derived enhancement of
$\overline{\mathcal{M}}_g$, and an equality between the induced
tangent duality map and Verdier duality. For a classifying stack
$BG_\cA$, the shifted form must be constructed in its natural degree;
a declared $0$-shifted form supplies no PTVV input. Classifying stacks
with invariant pairings are the standard source of shifted symplectic
structures of different degree.

The theorem is a compatibility criterion. It records the data under
which a PTVV construction reproduces the Verdier pairing, using
Theorem~\ref{thm:quantum-complementarity-main} as input.

\begin{theorem}[PTVV compatibility criterion for~$C_1$; conditional]
\label{thm:C-PTVV-alternative}
\index{PTVV!mapping-stack presentation|textbf}%
\ClaimStatusConditional
Let $(\cA, \cA^!)$ be a chiral Koszul pair on a smooth complex
curve~$X$, and fix $g \geq 2$. Assume the following PTVV datum:
\begin{enumerate}[label=\textup{(\alph*)}]
\item a proper $\mathcal{O}$-compact derived enhancement
$Y_g \to \overline{\mathcal{M}}_g$ with orientation trace
\[
 R\Gamma(Y_g,\mathcal{O}_{Y_g})
 \longrightarrow
 \mathbb{C}[-(3g-3)];
\]
\item a derived Artin stack $S_\cA$ locally of finite presentation,
carrying a $0$-shifted symplectic form
\[
 \omega_\cA^\flat\colon
 T_{S_\cA}\xrightarrow{\sim}L_{S_\cA};
\]
\item a base point $s\in S_\cA$ with
$T_sS_\cA\simeq\mathcal{Z}(\cA)$ and an identification
\[
 T_{[s]}R\operatorname{Map}(Y_g,S_\cA)
 \simeq
 R\Gamma(\overline{\mathcal{M}}_g,\mathcal{Z}(\cA))
 = \mathbf{C}_g(\cA);
\]
\item equality between the linearized PTVV duality morphism
\[
 \omega_{\mathrm{PTVV},[s]}^\flat\colon
 \mathbf{C}_g(\cA)
 \longrightarrow
 \mathbf{C}_g(\cA)^\vee[-(3g{-}3)]
\]
and the Verdier pairing morphism of
Theorem~\textup{\ref{thm:quantum-complementarity-main}(ii)}.
\end{enumerate}
Then\textup{:}
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Shifted symplectic form.}
\ClaimStatusConditional%
 The derived mapping stack
 $R\operatorname{Map}(Y_g,S_\cA)$
 carries a canonical $(-(3g{-}3))$-shifted symplectic form by the
 mapping-stack construction
 \textup{\cite[Theorem~2.5]{PTVV13}}.

\item \emph{Comparison to the Verdier pairing.}
\ClaimStatusConditional%
 Under hypotheses \textup{(c)}--\textup{(d)}, the map
 $\omega_{\mathrm{PTVV},[s]}^\flat$ on
 $T_{[s]}R\operatorname{Map}(Y_g,S_\cA)$ is the Verdier duality map
 on $\mathbf{C}_g(\cA)$.

\item \emph{Lagrangian complementarity.}
\ClaimStatusConditional%
 The embeddings of $\mathbf{Q}_g(\cA)$ and $\mathbf{Q}_g(\cA^!)$ as
 complementary Lagrangians in the PTVV sense are consequences of
 Theorem~\textup{\ref{thm:quantum-complementarity-main}} via
 Proposition~\textup{\ref{prop:ptvv-lagrangian}}; an intrinsically
 derived-stack-level construction of the anti-symplectic involution
 and a consequent derivation of the eigenspace decomposition require
 data beyond the mapping-stack hypotheses above.
\end{enumerate}
\end{theorem}

\begin{proof}
\textup{(i)} PTVV~\cite[Theorem~2.5]{PTVV13} sends a
$d$-oriented $\mathcal{O}$-compact source and an $n$-shifted
symplectic derived Artin target to an $(n-d)$-shifted symplectic
mapping stack. Here $n=0$ and $d=3g-3$, hence the shift is
$-(3g-3)$.

\textup{(ii)} The tangent-complex formula for derived mapping stacks
\textup{\cite[3.3.3]{HAG-II}} gives
$T_{[s]}R\operatorname{Map}(Y_g,S_\cA)\simeq
R\Gamma(Y_g,s^*T_sS_\cA)$. Hypothesis~\textup{(c)} identifies this
with $\mathbf{C}_g(\cA)$. The PTVV form is obtained by applying the
orientation trace of~$Y_g$ to the target isomorphism
$T_{S_\cA}\simeq L_{S_\cA}$. Hypothesis~\textup{(d)} identifies the
resulting tangent-duality morphism with the Verdier duality morphism
on $\mathbf{C}_g(\cA)$. This proves compatibility after the Verdier
pairing has been supplied by Theorem~C.

\textup{(iii)} The Lagrangian property of $\mathbf{Q}_g(\cA)$ inside
the PTVV shifted-symplectic linear derived scheme is
Proposition~\textup{\ref{prop:ptvv-lagrangian}}; its proof explicitly
cites $\mathbf{C}_g(\cA) = \mathbf{Q}_g(\cA) \oplus
\mathbf{Q}_g(\cA^!)$ from
Theorem~\textup{\ref{thm:quantum-complementarity-main}}. A proof
of~(iii) intrinsic to the PTVV side requires constructing the
Verdier involution on $R\operatorname{Map}(Y_g,S_\cA)$ directly at
derived-stack level. That construction is outside the hypotheses of
this theorem, so clause~\textup{(iii)} remains
\ClaimStatusConditional{}.
\end{proof}

\section{Loci and hypotheses}
\label{sec:theorem-C-refinements-summary}

Proposition~\ref{prop:theorem-C-surface-separation} fixes the
ambient taxonomy. Verdier complementarity is the paired-complex
decomposition of Theorem~\ref{thm:quantum-complementarity-main}.
The scalar conductor
$K^\kappa(\cA)=\kappa(\cA)+\kappa(\cA^!)$ is a family-level shadow:
the Vol.~I classical values are $\{0,13,250/3,98/3\}$, and the
Vol.~III Mukai-enhanced $\mathsf B$-row adjoins
$K^{\kappa_{\mathrm{ch}}}=2c_+(\mathrm{Mukai}(K3))=8$. The
derived center is the Hochschild complex
$Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)$; its finite-window cohomology
does not compute the scalar conductor.

The derived-center comparison has three levels. The
brace-dg equivalence
$Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)\simeq
Z^{\mathrm{der}}_{\mathrm{ch}}(\cA^!)$ remains
Conjecture~\ref{conj:derived-center-koszul-equivalence}. It follows
from the Morita-kernel criterion of
Proposition~\ref{prop:derived-center-morita-criterion}. The naive
$H^0$ center pairing is the conditional criterion of
Lemma~\ref{lem:naive-center-koszul-identification}.

Perfectness is governed by
Lemma~\ref{lem:perfectness-criterion}.  Proposition~\ref{prop:perfectness-standard-landscape}
records the conditional criterion and the family-specific
total-finiteness evidence: Heisenberg and positive-integer affine
Kac--Moody supply the strongest current input; admissible simple
quotients, generic non-critical affine Kac--Moody, and the
class-$\mathsf{M}$ boundary stratum remain under
Conjectures~\ref{conj:perfectness-boundary-km-generic}
and~\ref{conj:perfectness-boundary-class-M} or their named quotient
comparison hypotheses.

The $C_1$ reflexivity statement follows on the loci of
Corollary~\ref{cor:c0-unconditional-class-G-L} with the same
pseudo-coherent and perfectness hypotheses retained, and is
conditional on the fiber-finiteness conjectures elsewhere. The
$g=0$ comparison is the separate statement of
Remark~\ref{rem:plus-three-shift-nonissue}; the Verdier-pairing
degree formula has domain $g\geq 1$.

Self-dual halving is a same-family statement. It applies only under
the hypotheses of Theorem~\ref{thm:self-dual-halving}, where a
specified chiral-algebra isomorphism $f\colon\cA\simeq\cA^!$ induces
an automorphism $\tau_f$ satisfying
$\tau_f\sigma=-\sigma\tau_f$. In the real witness landscape the scalar
fixed points at Heisenberg $k=0$, affine critical level, and
$\beta\gamma_{\lambda=1/2}$ lack this chiral-algebra self-duality
package; the Virasoro same-family point is $c=13$ with
$\kappa(\mathrm{Vir}_{13})=13/2$.

The shifted-symplectic refinement keeps the hypotheses of
Theorem~\ref{thm:shifted-symplectic-complementarity}. The MC5
class-$\mathsf{M}$ input is the weight-completed/pro-object statement

continuous homotopies. The explicit cross-channel correction at
$(W_3,g)=(W_3,2)$ is
Theorem~\textup{\ref{thm:delta-f-cross-w3-g2}}. The PTVV statement
is Theorem~\ref{thm:C-PTVV-alternative}: it imports the Verdier
pairing of $C_1$ and tests compatibility of a derived-stack
construction with that pairing.

The K3-facing comparison is external to the Vol.~I classical scalar

completed K3 chiral Hall--Drinfeld double. No Drinfeld Yangian
attached to $\mathfrak g_{\Delta_5}$ is asserted.
