\chapter{Comparison of existence regimes for Koszul duals}
\label{app:existence-criteria}


For which chiral algebras does a Koszul dual exist? The strict
finite-type theory of the core monograph, the completion package of
Appendix~\ref{app:nilpotent-completion}, and the classical
existence/non-existence criteria in the literature each answer a
different version of this question.

\section{The existence problem}
\label{sec:existence-problem-statement}

\subsection{Question and ambient}

\begin{question}[Central question]\label{q:existence-central}
Given a chiral algebra $\mathcal{A}$ on $X$, when does there exist a
Koszul dual coalgebra $\mathcal{A}^i$ such that:
\begin{enumerate}
\item The bar construction resolves the dual coalgebra:
 $\bar{B}(\mathcal{A}) \xrightarrow{\sim} \mathcal{A}^i$
\item The cobar of the bar recovers $\mathcal{A}$:
 $\Omega(\bar{B}(\mathcal{A})) \xrightarrow{\sim} \mathcal{A}$
\item On additional strict finite-type module-theoretic surfaces, one
 can ask for the separate module/comodule comparison:
 \[\mathrm{Mod}^{\mathrm{compl}}(\mathcal{A}) \simeq
 \mathrm{CoMod}^{\mathrm{conil}}(\mathcal{A}^i)\]
\end{enumerate}
The base problem in this appendix is conditions~(1)--(2): existence of
the quadratic dual coalgebra and its bar/cobar resolution. Item~(3) is
a separate module-theoretic package with its own finiteness and
flatness hypotheses.

When conditions~(1)--(2) hold, the Koszul dual \emph{algebra} is
obtained only after the appropriate finite or continuous duality step:
\[
\mathcal{A}^! := \mathbb D_{\Ran}(\mathcal{A}^i)
\quad\text{or, on a finite-type model,}\quad
\mathcal{A}^! := (\mathcal{A}^i)^{\vee}_{\mathrm{cont}}
 \simeq \bar{B}(\mathcal{A})^{\vee}_{\mathrm{cont}} .
\]
\end{question}

\subsection{Preliminary definitions}

\begin{definition}[Koszul dual: quadratic finite-dual case (Priddy)]\label{def:koszul-dual-quadratic}
Let $\mathcal{A} = T_{\mathcal{D}}(V) / (R)$ be a \emph{quadratic} chiral algebra,
where $V$ is a graded $\mathcal{D}_X$-module of generators and
$R \subseteq V \otimes_{\mathcal{D}} V$ is a submodule of quadratic
relations. Assume that $V$ is dualizable in each conformal weight and
that $R$ is a direct summand, so that annihilators and biduals are
formed by perfect pairings. In finite-dual coordinates the
\emph{quadratic Koszul-dual coalgebra} of $\mathcal{A}$ is
\begin{equation}\label{eq:koszul-dual-quadratic}
\mathcal{A}^i \;\hookrightarrow\; \text{Cofree}_{\mathcal{D}}(sV^*),
\end{equation}
defined as the maximal sub-coalgebra of
$\text{Cofree}_{\mathcal{D}}(sV^*)$ whose quadratic core lies in
$s^2R^\perp$,
where:
\begin{itemize}
\item $V^* = \text{Hom}_{\mathcal{D}}(V, \mathcal{D}_X)$ is the $\mathcal{D}$-linear dual,
\item $R^\perp = \{ f \in (V^*)^{\otimes 2} : \langle f, r \rangle = 0
 \text{ for all } r \in R \}$ is the annihilator of $R$ under the
 natural pairing $(V^*)^{\otimes 2} \otimes V^{\otimes 2} \to \mathbb{C}$,
\item $s$ denotes the cohomological suspension (degree shift by $+1$).
\end{itemize}
Equivalently, in the coalgebra convention of
Loday--Vallette, one uses the subcoalgebra of $T^c_{\mathcal D}(sV)$
with corelations $s^2R$ and obtains the displayed annihilator
presentation after finite duality. The two descriptions agree exactly
on the finite-dual surface above.

The construction is intrinsic to the quadratic presentation. It does
not use the bar construction and it does not assert Koszulness. For a
chiral algebra presented by generators and OPE relations, the same
definition is applied to the quadratic associated graded or to a
finite-window quotient. The Koszul dual algebra is the separate object
$\mathcal{A}^! := \mathbb D_{\Ran}(\mathcal{A}^i)$, or
$(\mathcal{A}^i)^\vee_{\mathrm{cont}}$ on a finite-type linear model,
when that duality is defined.
\end{definition}

\begin{remark}[Relationship to the bar construction]\label{rem:bar-vs-quadratic-dual}
The bar construction $\bar{B}(\mathcal{A})$ is defined \emph{independently} of the
Koszul dual: it is the cofree coalgebra on the desuspension of the augmentation
ideal $\bar{\mathcal{A}} = \ker(\varepsilon)$,
\[\bar{B}(\mathcal{A}) \;=\; \bigl(T^c(s^{-1}\bar{\mathcal{A}}),\; d\bigr),\]
equipped with the differential $d$ induced by the multiplication of $\mathcal{A}$.
No knowledge of $\mathcal{A}^i$ is required to define $\bar{B}(\mathcal{A})$.

The \emph{Koszulness condition} (Definition~\ref{def:koszul-property-existence}
below) is precisely the statement that the natural projection
$\bar{B}(\mathcal{A}) \twoheadrightarrow \mathcal{A}^i$ is a quasi-isomorphism
of dg coalgebras.
Thus the bar construction provides a dg coalgebra \emph{resolution} of
$\mathcal{A}^i$ when $\mathcal{A}$ is Koszul, not its definition.
To obtain the Koszul dual algebra from the coalgebra~$\mathcal{A}^i$,
one applies Verdier duality on the Ran space, or takes the continuous
graded linear dual in a linearly compact finite-window model. This
requires the corresponding finite-dimensionality or completeness
hypotheses; it is not part of the definition of the bar coalgebra.
\end{remark}

\begin{definition}[Koszul property; \ClaimStatusDefinitional]\label{def:koszul-property-existence}
Let $\mathcal{A} = T_{\mathcal{D}}(V)/(R)$ be a quadratic chiral algebra with
Koszul dual $\mathcal{A}^i$ as in Definition~\ref{def:koszul-dual-quadratic}.
Then $\mathcal{A}$ is \emph{Koszul} if:
\begin{enumerate}
\item The natural projection from the bar coalgebra to the Koszul dual
 coalgebra is a quasi-isomorphism of dg coalgebras:
 \[\bar{B}(\mathcal{A}) \xrightarrow{\;\sim\;} \mathcal{A}^i\]
 (After Verdier or continuous finite-window duality, this identifies
 the corresponding Koszul dual \emph{algebra}; see
 Remark~\ref{rem:bar-vs-quadratic-dual}.)
\item Equivalently, the cobar of the bar recovers $\mathcal{A}$:
 \[\Omega(\bar{B}(\mathcal{A})) \xrightarrow{\;\sim\;} \mathcal{A}\]
\end{enumerate}
Conditions (1) and (2) are equivalent for conilpotent coalgebras
(see \cite{LV12}, Theorem 2.3.3). Under the finite-dual hypotheses of
Definition~\ref{def:koszul-dual-quadratic}, the quadratic coalgebra
$\mathcal{A}^i$ exists; the content of the Koszul property is that the
bar construction provides a \emph{resolution} of $\mathcal{A}^i$: the
projection carries the homotopy type of the dual coalgebra.
\end{definition}

\begin{remark}[Module-category comparison is extra data]
\label{rem:existence-module-comparison-extra}
The ordinary module/comodule comparison is not part of the base
definition above. On the manuscript's current proved surface, the
intrinsic higher-genus statement is the bar-coalgebra
coderived/contraderived comparison
\textup{(}Theorems~\ref{thm:positselski-chiral} and
\ref{thm:full-derived-module-equiv}\textup{)}, while the ordinary
module-level equivalence is the separate quadratic genus-$0$
$\Eone$ theorem
\textup{(}Theorem~\ref{thm:e1-module-koszul-duality}\textup{)}.
Any broader module-category comparison for general chiral Koszul pairs
requires extra hypotheses beyond the quadratic existence package
recorded in this appendix.
\end{remark}

\begin{proposition}[Four loci in the existence problem;
\ClaimStatusProvedHere]
\label{prop:existence-four-loci}
Let $\mathcal A$ be an augmented chiral algebra with bar coalgebra
$\bar B(\mathcal A)$ and, when a quadratic finite-dual presentation is
chosen, quadratic coalgebra $\mathcal A^i_{\mathrm{quad}}$.
There are four distinct questions.
\begin{enumerate}
\item \emph{Bar-cobar inversion.} The counit
 \[
 \epsilon_{\mathcal A}\colon
 \Omega\bar B(\mathcal A)\longrightarrow \mathcal A
 \]
 is defined from the universal twisting cochain
 $\tau_{\mathcal A}\colon \bar B(\mathcal A)\to\mathcal A$ before any
 quadratic dual is chosen. Dually, for a conilpotent coalgebra~$C$
 the unit
 \[
 \eta_C\colon C\longrightarrow \bar B\Omega(C)
 \]
 is defined by the same universal twisting cochain
 $C\to\Omega(C)$. These maps are quasi-isomorphisms only under the
 conilpotent strict hypotheses, or under the completed hypotheses
 recorded below.
\item \emph{Koszul coalgebra locus.} A quadratic finite-dual
 coalgebra $\mathcal A^i_{\mathrm{quad}}$ exists when
 $V$ is weightwise dualizable, $R$ is a direct summand, and
 $(R^\perp)^\perp=R$. The point $\mathcal A$ lies on the Koszul locus
 exactly when
 \[
 q_{\mathcal A}\colon \bar B(\mathcal A)\to
 \mathcal A^i_{\mathrm{quad}}
 \]
 is a quasi-isomorphism of dg coalgebras.
\item \emph{Strict duality locus.} The notation
 $\mathcal A^!$ denotes an algebra only after applying
 $\mathbb D_{\Ran}$, or the continuous finite-window dual. Thus the
 strict duality locus is the intersection of the Koszul coalgebra
 locus with the finiteness, completeness, and continuity hypotheses
 making this dual algebra and the induced comparison maps defined.
\item \emph{Derived centre.} The derived chiral centre is
 $Z^{\mathrm{der}}_{\mathrm{ch}}(\mathcal A)
 = C^\bullet_{\mathrm{ch}}(\mathcal A,\mathcal A)$ on its Hochschild
 surface. It is not $\mathcal A^i$, not $\mathcal A^!$, and not the
 bar-cobar counit.
\end{enumerate}
Consequently
\[
\{\text{strict duality locus}\}\subseteq
\{\text{Koszul coalgebra locus}\}\subseteq
\{\text{bar-cobar inversion locus in a chosen ambient}\},
\]
and none of the inclusions is a definition of the derived centre.
\end{proposition}

\begin{proof}
The bar coalgebra and the universal twisting cochain are constructed
from the augmentation and multiplication of $\mathcal A$; the
coalgebra $\mathcal A^i_{\mathrm{quad}}$ is constructed instead from
the quadratic finite-dual presentation. The map
$q_{\mathcal A}$ compares these two independently defined coalgebras.
Its being a quasi-isomorphism is precisely the Koszul condition of
Definition~\ref{def:koszul-property-existence}. Applying
$\mathbb D_{\Ran}$ or a continuous dual is a further functorial
operation whose source and target must be defined in the chosen
topological category. Hochschild cochains compute derived endomorphisms
of $\mathcal A$ as an algebra object, so they live on a third surface.
The unit and counit are the ordinary bar-cobar adjunction maps and are
typed before any assertion about their homology is made.
\end{proof}

\begin{proposition}[Recognition targets are not existence criteria;
\ClaimStatusProvedHere]
\label{prop:recognition-not-existence}
Let $\mathscr C$ be a category of augmented chiral algebras with bar
construction, and let
$F\colon\mathscr C\to\mathscr D$ be a comparison or transport functor.
A Koszul-dual existence theorem for $\mathcal A\in\mathscr C$ is not
supplied by an isomorphism, MC solution, or recognition theorem for
$F(\mathcal A)$ unless the following source-level data are also
present:
\begin{enumerate}
\item a coalgebra $C_{\mathcal A}$ in the chosen strict or completed
 chiral coalgebra category;
\item a typed comparison
 \[
 q_{\mathcal A}\colon\bar B(\mathcal A)\longrightarrow C_{\mathcal A}
 \]
 whose cone is acyclic in the declared ambient;
\item preservation of the bar/cobar maps by $F$, and conservativity of
 $F$ on the class of cones containing $\operatorname{Cone}(q_{\mathcal A})$.
\end{enumerate}
Consequently associator or $\operatorname{GRT}$ transport, a
Positselski co/contra comparison, a Yangian recognition theorem, a K3
Hall-double comparison, or a topological enhancement through
$\mathsf{SC}^{\mathrm{ch,top}}$ is a recognition surface until it is
accompanied by these source-level bar data. None of them replaces the
finite-dual test, the PBW spectral sequence, or the completed
Mittag--Leffler test.
\end{proposition}

\begin{proof}
The assertion that $\mathcal A$ has a Koszul dual coalgebra includes
both an object of the source coalgebra category and the acyclicity of
the map from the source bar coalgebra. Data living only after applying
$F$ neither construct that source coalgebra nor prove acyclicity of the
source cone. If $F$ preserves the displayed comparison and is
conservative on its cone, then acyclicity of $F(q_{\mathcal A})$
reflects acyclicity of $q_{\mathcal A}$; without those two hypotheses
the target statement is only a comparison statement.
\end{proof}


\section{Quadratic case: constructive existence proof}
\label{sec:quadratic-existence}

\subsection{Statement of result}

\begin{theorem}[Quadratic finite-dual coalgebras exist \cite{LV12, Priddy70}; \ClaimStatusProvedElsewhere]\label{thm:quadratic-have-duals}
Let $\mathcal{A} = \text{Free}_{\mathcal{D}}(V) / (R)$ be a \emph{quadratic}
chiral algebra, meaning:
\begin{enumerate}
\item $V$ is dualizable in each conformal weight;
\item $R \subseteq V^{\otimes 2}$ is a direct-summand relation object;
\item the annihilator pairing gives $(R^\perp)^\perp=R$.
\end{enumerate}

Then the quadratic coalgebra $\mathcal{A}^i_{\mathrm{quad}}$ of
Definition~\ref{def:koszul-dual-quadratic} exists and is conilpotent
in the strict finite-dual lane. There is a canonical projection of dg
coalgebras
\[
q_{\mathcal A}\colon \bar B(\mathcal A)\longrightarrow
\mathcal{A}^i_{\mathrm{quad}}
\]
obtained by retaining the quadratic core. The algebra $\mathcal A$ is
Koszul if and only if $q_{\mathcal A}$ is a quasi-isomorphism. Thus
quadratic finite-dual existence is automatic under the displayed
dualisability hypotheses; Koszulness is a further homological
condition.
\end{theorem}

\begin{proof}
The perfect pairing identifies $R^\perp$ as a direct summand of
$(V^*)^{\otimes 2}$ and makes biduality exact. The cofree conilpotent
coalgebra $T^c_{\mathcal D}(sV^*)$ carries the deconcatenation
coproduct. The intersection of the inverse images of $s^2R^\perp$
under all adjacent length-two projections is stable under
deconcatenation, hence is the maximal subcoalgebra with that quadratic
core. This is $\mathcal A^i_{\mathrm{quad}}$.

The bar differential on $\bar B(\mathcal A)$ is the coderivation
determined by the product of $\mathcal A$. Projection to the quadratic
core kills precisely the tensors violating the relation object $R$ and
therefore gives a morphism of dg coalgebras
$q_{\mathcal A}$. By definition, $\mathcal A$ is Koszul exactly when
this morphism is a quasi-isomorphism. No implication from the mere
existence of $\mathcal A^i_{\mathrm{quad}}$ to Koszulness is used.
\end{proof}

\begin{construction}[Quadratic finite-dual coordinates]\label{constr:quadratic-finite-dual-coordinates}
The finite-coordinate construction has four inputs.

\begin{enumerate}
\item \emph{Dual space.}
\[V^* = \text{Hom}(V, \mathbb{C})\]
with dual grading: $|v^*| = -|v|$.

\item \emph{Dual relations.}
The relations $R \subseteq V^{\otimes 2}$ induce:
\[R^\perp \subseteq (V^*)^{\otimes 2}\]
defined by:
\[R^\perp = \{f \in (V^*)^{\otimes 2} : f(r) = 0 \text{ for all } r \in R\}\]

\item \emph{Dual coalgebra.}
Define $\mathcal{A}^i_{\mathrm{quad}}$ as the maximal sub-coalgebra of
$\text{Cofree}_{\mathcal{D}}(sV^*)$ with quadratic core $s^2R^\perp$:
\[\mathcal{A}^i_{\mathrm{quad}} \hookrightarrow \text{Cofree}_{\mathcal{D}}(sV^*)\]
where $s$ denotes suspension (degree shift by 1). The dual
\emph{coalgebra} is a sub-object of the cofree coalgebra, not a
quotient. The quotient construction
$\text{Free}(sV^*)/(sR^\perp)$ is the finite-dual quadratic algebra
presentation; identifying it with $\mathcal A^!$ still requires the
Verdier or continuous-duality hypotheses above.

\item \emph{Koszul comparison.}
The coalgebra structure follows from the universal property of the
cofree conilpotent coalgebra; see \cite{LV12}*{Theorem~7.1.2}.
Koszulness is the separate check that
$\bar{B}(\mathcal{A}) \to \mathcal{A}^i_{\mathrm{quad}}$ is a
quasi-isomorphism, equivalently that the Koszul complex is acyclic
away from degree~$0$.
\end{enumerate}
\end{construction}

\subsection{Explicit construction}

\begin{construction}[Building \texorpdfstring{$\mathcal{A}^i$}{Ai} from \texorpdfstring{$\mathcal{A}$}{A}]\label{const:quadratic-dual-criteria}
Given a finite-dual quadratic presentation
$\mathcal{A} = \text{Free}(V) / (R)$, construct
$\mathcal{A}^i_{\mathrm{quad}}$ as follows:

\emph{Input.}
Generators $V = \text{span}\{v_1, \ldots, v_n\}$ with degrees $|v_i|$, and relations $R = \text{span}\{r_1, \ldots, r_m\}$ where each $r_j \in V^{\otimes 2}$.

\emph{Dual generators.}
\[V^* = \text{span}\{v_1^*, \ldots, v_n^*\} \quad \text{with } |v_i^*| = -|v_i| + 1\]

(The shift by 1 is the suspension $s$.)

\emph{Coordinate relations.}
Express each relation as:
\[r_j = \sum_{i,k} a_{ik}^j \, v_i \otimes v_k\]

\emph{Orthogonal complement.}
Under the natural pairing $\langle v_i^* \otimes v_k^*, v_l \otimes v_m \rangle
= \delta_{il}\delta_{km}$, define:
\[R^\perp = \{f \in (V^*)^{\otimes 2} : \langle f, r \rangle = 0
\;\text{for all}\; r \in R\}\]
This is a subspace of codimension $\dim R$ in $(V^*)^{\otimes 2}$.
\emph{Annihilator convention.} $R^\perp$ is the annihilator of $R$ under the
Koszul pairing. It is defined only after choosing the duality
$V^*\otimes V\to\mathbb C$; it is not the image of $R$ under a basis
identification $V\simeq V^*$.

\emph{Koszul dual coalgebra.}
\[
\mathcal{A}^i_{\mathrm{quad}}
 = \text{sub-coalgebra of } \text{Cofree}(sV^*)
 \text{ with quadratic core } s^2R^\perp .
\]

That is, $\mathcal{A}^i_{\mathrm{quad}}$ is the maximal sub-coalgebra
of $\text{Cofree}(sV^*)$ whose degree-2 component lies in
$s^2R^\perp \subset (sV^*)^{\otimes 2}$. Here
$\text{Cofree}(sV^*)$ denotes the cofree coalgebra:
\[\text{Cofree}(sV^*) = \bigoplus_{n=0}^\infty (sV^*)^{\otimes n}\]
with comultiplication:
\[\Delta(x_1 \otimes \cdots \otimes x_n) = \sum_{i=0}^n
(x_1 \otimes \cdots \otimes x_i) \otimes (x_{i+1} \otimes \cdots \otimes x_n),
\qquad x_j\in sV^* .\]
\end{construction}

\begin{example}[Heisenberg algebra]\label{ex:heisenberg-quadratic-dual}
The Heisenberg chiral algebra is:
\[\mathcal{H}_k = \text{Free}(a, a^*) / ([a, a^*] - k)\]

\emph{Generators.} $V = \text{span}\{a, a^*\}$ with $|a| = 1$, $|a^*| = 1$.

\emph{Relation (quadratic part).} $R = \operatorname{span}\{a \otimes a^* - a^* \otimes a\} \subset V^{\otimes 2}$

(The full relation $[a, a^*] = k$ is inhomogeneous, where $k$ is the
Heisenberg level; for Heisenberg this level also equals the modular
characteristic, $\kappa(\mathcal{H}_k) = k$. This identification is
specific to the abelian case: for non-abelian Kac--Moody one has
instead $\kappa(\widehat{\mathfrak{g}}_k) = \dim(\mathfrak{g})(k+h^\vee)/(2h^\vee)$.
The constant $k$ contributes curvature $m_0$ in the bar complex,
not a quadratic relation. For Koszul duality, we work with the
quadratic part.)

\emph{Dual generators.} $V^* = \operatorname{span}\{a^{\dagger}, (a^*)^{\dagger}\}$ with
$|a^{\dagger}| = 0$, $|(a^*)^{\dagger}| = 0$ (after suspension).

\emph{Orthogonal complement.} $R$ is the antisymmetric part of $V^{\otimes 2}$,
so its annihilator is the symmetric part:
\[R^\perp = \operatorname{Sym}^2(V^*) = \operatorname{span}\{(a^{\dagger})^2,\;
((a^*)^{\dagger})^2,\; a^{\dagger}(a^*)^{\dagger} + (a^*)^{\dagger}a^{\dagger}\}\]

\emph{Dual coalgebra.}
\[\mathcal{H}^i \hookrightarrow \text{Cofree}(a^{\dagger}, (a^*)^{\dagger})\]
(the maximal sub-coalgebra with quadratic core $s^2R^\perp$)

This has comultiplication:
\[\Delta(a^{\dagger}) = a^{\dagger} \otimes 1 + 1 \otimes a^{\dagger}\]
\[\Delta((a^*)^{\dagger}) = (a^*)^{\dagger} \otimes 1 + 1 \otimes (a^*)^{\dagger}\]

(These are primitive.)

\emph{Verification.} The bar construction resolves the dual coalgebra,
and the cobar of the bar recovers the original algebra:
\[\bar{B}(\mathcal{H}) \xrightarrow{\sim} \mathcal{H}^i
\quad \text{(dg coalgebras)} \quad \text{and} \quad
\Omega(\bar{B}(\mathcal{H})) \xrightarrow{\sim} \mathcal{H}
\quad \text{(chiral algebras)}.\]
The Koszul-dual algebra is
$\mathcal H^!=(\mathcal H^i)^\vee_{\mathrm{cont}}$ on the
finite-window continuous-dual surface.
\end{example}

\subsection{PBW regularity conditions}

\begin{definition}[Quadratic PBW regularity]\label{def:quadratic-regularity}
A quadratic or filtered-quadratic presentation
$\mathcal{A}=\text{Free}(V)/(R)$ is \emph{PBW regular} if:
\begin{enumerate}
\item \emph{Finite dualisability.} The pairing:
 \[V \otimes V^* \to \mathbb{C}\]
 is perfect in each conformal weight, and $R$ is a direct summand of
 $V^{\otimes 2}$.

\item \emph{Bidual compatibility.} The relations $R$ and $R^\perp$ satisfy:
 \[(R^\perp)^\perp = R\]
 with respect to the chosen pairing.

\item \emph{Conilpotency.} The coalgebra $\mathcal{A}^i_{\mathrm{quad}}$ is conilpotent: every element
 of the coaugmentation coideal
 $\overline{\mathcal{A}^i_{\mathrm{quad}}} = \ker(\varepsilon)$ is
 annihilated by a sufficiently high iterate of the reduced coproduct $\bar{\Delta}$:
 \[\bigcup_{n=0}^\infty \ker(\bar{\Delta}^n) =
 \overline{\mathcal{A}^i_{\mathrm{quad}}}\]

\item \emph{PBW flatness.} There is a multiplicative filtration $F$
 whose Rees algebra is flat and whose associated graded
 $\operatorname{gr}_F\mathcal A$ is the quadratic algebra
 determined by $(V,R)$.

\item \emph{Koszul associated graded.} The quadratic algebra
 $\operatorname{gr}_F\mathcal A$ is classically Koszul, and the
 bar chain groups are finite-dimensional in each conformal
 weight and bar degree.
\end{enumerate}
\end{definition}

\begin{theorem}[PBW regularity implies Koszul \cite{BGS96, LV12}; \ClaimStatusConditional]\label{thm:regular-implies-koszul}
If $\mathcal{A} = \text{Free}(V) / (R)$ is PBW regular in the sense of
Definition~\ref{def:quadratic-regularity}, then:
\begin{enumerate}
\item $\mathcal{A}^i_{\mathrm{quad}}$ exists and is given by
 Construction~\ref{const:quadratic-dual-criteria};
\item $\mathcal{A}$ is Koszul, i.e.
 $\bar B(\mathcal A)\to\mathcal A^i_{\mathrm{quad}}$ is a
 quasi-isomorphism;
\item the bar-cobar counit
 $\epsilon_{\mathcal A}\colon\Omega\bar B(\mathcal A)\to\mathcal A$
 is a quasi-isomorphism in the strict conilpotent lane.
\end{enumerate}
\end{theorem}

\begin{proof}
Items (1)--(3) of Definition~\ref{def:quadratic-regularity} give the
finite-dual conilpotent coalgebra by
Theorem~\ref{thm:quadratic-have-duals}. Items (4)--(5) put the bar
complex under the PBW spectral sequence. Its $E_1$ page is the Koszul
complex of $\operatorname{gr}_F\mathcal A$, which is acyclic in
positive homological degree by the associated-graded hypothesis. The
finite-dimensionality condition gives strong convergence. Hence the
bar complex of $\mathcal A$ has the same diagonal concentration and
$\bar B(\mathcal A)\to\mathcal A^i_{\mathrm{quad}}$ is a
quasi-isomorphism.

The strict bar-cobar counit is the morphism induced by the universal
twisting cochain $\tau_{\mathcal A}\colon\bar B(\mathcal A)\to
\mathcal A$. For conilpotent coalgebras, the standard bar-cobar
comparison theorem \cite{LV12} identifies acyclicity of
$\tau_{\mathcal A}$ with the quasi-isomorphism property of
\[
\epsilon_{\mathcal A}\colon\Omega\bar B(\mathcal A)\to\mathcal A .
\]
This proves the counit directly from the PBW spectral sequence and
conilpotent bar-cobar adjunction. Theorem~\ref{thm:bar-cobar-inversion-qi}
is the global chiral packaging of this strict-lane assertion together
with the separate completed/coderived off-locus statement; it is not a
premise in the present argument.
\end{proof}


\section{Non-quadratic case: completion required}
\label{sec:non-quadratic-existence}

\subsection{Necessity of completion}

\begin{remark}[Obstructions in non-quadratic case]\label{rem:non-quadratic-obstructions}
For higher-order relations ($n \ge 3$), the dual relations
$r^\perp \in (V^*)^{\otimes n}$ need not define a conilpotent
quadratic coalgebra, the bidual map may have infinite-dimensional
kernel, and $\mathcal{A}^*$ lacks a natural continuous
comultiplication. Filtered completions remove these obstructions only
after the topology, finite-window quotients, and
Mittag--Leffler convergence have been supplied.
\end{remark}

\begin{definition}[Filtered completion]\label{def:filtered-completion}
Let $\mathcal{A} = \text{Free}(V) / (R)$ carry a descending
multiplicative filtration
\[
\mathcal A=F^0\mathcal A \supset F^1\mathcal A \supset
F^2\mathcal A\supset\cdots,\qquad
F^p\mathcal A\cdot F^q\mathcal A\subset F^{p+q}\mathcal A .
\]
The \emph{filtered completion} is:

\emph{Separating topology.}
The filtration is required to be separated and complete:
\[
\bigcap_nF^n\mathcal A=0,\qquad
\mathcal A\simeq\varprojlim_n\mathcal A/F^n\mathcal A .
\]
For an augmented algebra one usually takes $F^n\mathcal A=I^n$,
where $I=\ker(\varepsilon)$.

\emph{Completion.}
\[\widehat{\mathcal{A}} = \varprojlim_n \mathcal{A} / F^n\mathcal{A}.\]

The useful finite-window regime assumes that the quotients
$\mathcal A/F^n\mathcal A$ are finite in each cohomological degree and
that all chiral operations are continuous for this topology.
\end{definition}

\begin{theorem}[Completed Koszul dual \cite{Positselski11}; \ClaimStatusProvedElsewhere]\label{thm:completed-koszul-dual}
Let $\mathcal A$ be a separated complete filtered chiral algebra with
finite quotients $\mathcal A_n=\mathcal A/F^n\mathcal A$. Assume:
\begin{enumerate}
\item the chiral operations are strongly filtered, so every finite
 quotient sees only finitely many arities;
\item each $\mathcal A_n$ has a finite-dual Koszul coalgebra
 $\mathcal A_n^i$ and a bar-cobar counit
 \[
 \epsilon_n\colon\Omega\bar B(\mathcal A_n)\to\mathcal A_n
 \]
 which is a
 quasi-isomorphism in the square-zero total model, or a coacyclic-cone
 equivalence in Positselski's second-kind curved category;
\item for $m\ge n$ the quotient maps $\rho_{m,n}\colon
 \mathcal A_m\to\mathcal A_n$ and coalgebra maps
 $\rho^i_{m,n}\colon\mathcal A_m^i\to\mathcal A_n^i$ make the
 finite-stage counits commute:
 \[
 \begin{tikzcd}
 \Omega\bar B(\mathcal A_m) \ar[r,"\epsilon_m"]
   \ar[d,"\Omega\bar B(\rho_{m,n})"']&
 \mathcal A_m \ar[d,"\rho_{m,n}"]\\
 \Omega\bar B(\mathcal A_n) \ar[r,"\epsilon_n"']&
 \mathcal A_n ;
 \end{tikzcd}
 \]
\item the inverse systems
 $\{H^j(\operatorname{Cone}(\epsilon_n))\}_n$ and
 $\{H^j(\bar B(\mathcal A_n))\}_n$ satisfy the
 Mittag--Leffler condition for every~$j$.
\end{enumerate}
Then the completed dual coalgebra
\[
\widehat{\mathcal{A}^i}
 = \varprojlim_n \mathcal A_n^i
\]
exists as a separated complete pro-conilpotent chiral coalgebra and
satisfies:
\begin{enumerate}
\item $\widehat{\mathcal{A}^i}$ is a completed chiral coalgebra;
\item the completed Koszul dual algebra is
 $\widehat{\mathcal A}^{\,!}
 :=\mathbb D_{\Ran}(\widehat{\mathcal{A}^i})$, or
 $(\widehat{\mathcal{A}^i})^\vee_{\mathrm{cont}}$ in a
 linearly compact model, whenever the relevant duality is defined;
\item Bar-cobar constructions converge in completion:
 \[
 \widehat\Omega(\widehat{\bar{B}}(\widehat{\mathcal{A}}))
 \simeq \widehat{\mathcal{A}},
 \qquad
 H^\bullet\widehat{\bar B}(\widehat{\mathcal A})
 \simeq \widehat{\mathcal{A}^i};
 \]
\item module/comodule comparisons require the Positselski
 coderived/contraderived hypotheses; they are not part of the
 existence of $\widehat{\mathcal{A}^i}$ itself.
\end{enumerate}
\end{theorem}

\begin{proof}
Strong filtration makes the finite-stage bar differentials finite
sums. The displayed commutative square makes the finite-stage dual
coalgebras and counits an inverse system. Taking inverse limits gives
a complete
pro-conilpotent coalgebra because each stage is conilpotent and the
transition maps respect coaugmentation and coproduct. The Milnor exact
sequence for the cone of the stagewise counits has a
$\varprojlim^1 H^{j-1}(\operatorname{Cone}(\epsilon_n))$ term; the
Mittag--Leffler hypothesis kills it, so the limit counit is again a
quasi-isomorphism in the square-zero model.
In the curved CDG model the same argument is read in the
coderived/contraderived category, where the limiting cone is
coacyclic. The final assertion is only the definition of the
post-Verdier or continuous dual algebra and requires its own
topological finiteness hypotheses.
\end{proof}

\begin{proposition}[Finite-stage obstruction classes;
\ClaimStatusProvedHere]
\label{prop:finite-stage-obstruction-classes}
Let $\mathcal A$ satisfy the filtration hypotheses of
Theorem~\ref{thm:completed-koszul-dual}, and write
$\mathcal A_n=\mathcal A/F^n\mathcal A$. A completed Koszul-dual
coalgebra is obtained only after the following finite-stage
obstructions vanish.
\begin{enumerate}
\item \emph{Stage Koszul obstruction.} If
 $q_n\colon\bar B(\mathcal A_n)\to\mathcal A_n^i$ is the finite-stage
 bar comparison, then
 \[
 \mathsf K_n^j=H^j(\operatorname{Cone}(q_n))
 \]
 must vanish for every~$j$ in the strict square-zero model; in the
 curved model the cone must be coacyclic in the specified second-kind
 category.
\item \emph{Counit obstruction.} For the finite-stage counit
 $\epsilon_n\colon\Omega\bar B(\mathcal A_n)\to\mathcal A_n$,
 \[
 \mathsf E_n^j=H^j(\operatorname{Cone}(\epsilon_n))
 \]
 must vanish, or have coacyclic cone in the curved ambient.
\item \emph{Compatibility obstruction.} For $m\ge n$ the class of
 \[
 \delta_{m,n}:=
 \rho_{m,n}\circ\epsilon_m
 -\epsilon_n\circ\Omega\bar B(\rho_{m,n})
 \]
 in
 \[
 H^0\operatorname{Hom}\bigl(
 \Omega\bar B(\mathcal A_m),\mathcal A_n\bigr)
 \]
 must vanish, with coherent null-homotopies for triples
 $\ell\ge m\ge n$.
\item \emph{Milnor obstruction.} The inverse-limit passage has
 obstruction
 \[
 \varprojlim\nolimits^1_n\,\mathsf E_n^{j-1}
 \]
 in degree~$j$. The Mittag--Leffler condition kills this term.
\item \emph{Maurer--Cartan lifting obstruction.} Let
 \[
 \mathfrak g_{\mathcal A}
 =\operatorname{Conv}(\bar B(\mathcal A),\mathcal A)
 \]
 with its filtration induced from $F^\bullet\mathcal A$. A partial
 solution $\Theta_{\le N}$ of the MC equation modulo $F^{N+1}$ lifts
 across $F^{N+1}/F^{N+2}$ only if
 \[
 \mathsf O_{N+1}(\Theta_{\le N})
 =
	 \left[
	 \partial\Theta_{\le N}
	 +\frac12[\Theta_{\le N},\Theta_{\le N}]
	 \right]_{N+1}
	 \in H^2\!\left(
	 \operatorname{gr}^{N+1}\mathfrak g_{\mathcal A},
	 d_{\Theta_{\le N}}
	 \right),
	 \qquad
	 d_{\Theta_{\le N}}=\partial+[\Theta_{\le N},-],
	 \]
	 vanishes. When it vanishes, the choices of lift form a torsor under
	 $H^1(\operatorname{gr}^{N+1}\mathfrak g_{\mathcal A},
	 d_{\Theta_{\le N}})$.
\end{enumerate}
These classes are the finite-stage tests; Positselski completion,
MC solvability, and target recognition do not remove them.
\end{proposition}

\begin{proof}
The first two classes are the homology of the cones that define the
finite-stage bar comparison and counit. The third is the obstruction to
making the finite-stage counits a morphism of inverse systems; coherent
null-homotopies are exactly the higher compatibility data. The Milnor
exact sequence for the inverse limit of cones contains
$\varprojlim^1_n\mathsf E_n^{j-1}$, so the fourth class is precisely
the obstruction to passing acyclicity to the limit. The last formula is
the standard obstruction class for lifting a filtered Maurer--Cartan
solution: the MC curvature of a partial solution has first possible
non-zero component in
$\operatorname{gr}^{N+1}\mathfrak g_{\mathcal A}$, is closed for
$d_{\Theta_{\le N}}$ by the Bianchi identity, and changes by a
$d_{\Theta_{\le N}}$-coboundary under a change of lift.
\end{proof}

\subsection{I-adic completion}

\begin{definition}[I-adic completion]\label{def:i-adic-completion}
Let $I \subseteq \mathcal{A}$ be an ideal (the ``augmentation ideal''). The
\emph{$I$-adic completion} is:
\[\widehat{\mathcal{A}}_I = \varprojlim_n \mathcal{A} / I^n\]

\emph{Geometric meaning.} This is completion along the formal
neighbourhood of the augmentation character, when such a spectrum
exists.
\end{definition}

\begin{example}[Virasoro algebra]\label{ex:virasoro-i-adic}
The Virasoro algebra is:
\[\text{Vir} = \text{span}\{L_n, c : n \in \mathbb{Z}\} /
([L_m, L_n] - (m-n)L_{m+n} - \frac{c}{12}(m^3 - m)\delta_{m+n,0})\]

This is \emph{inhomogeneous quadratic}: the relation $[L_m, L_n] = (m-n)L_{m+n} + \frac{c}{12}(m^3-m)\delta_{m+n,0}$ is bilinear in generators plus a constant (the central charge defines a Lie algebra 2-cocycle in $H^2(\mathrm{Witt}, \mathbb{C})$).

\emph{Augmentation ideal.} $I = \text{span}\{L_n : n \in \mathbb{Z}\}$ (the kernel of the counit $\varepsilon: U(\mathrm{Vir}) \to \mathbb{C}$ sending all $L_n \mapsto 0$ and $c \mapsto c_0$)

\emph{Completion.}
\[\widehat{\text{Vir}} = \varprojlim_n \text{Vir} / I^n\]

In this completion, the Virasoro algebra acquires a completed curved
bar coalgebra
\[
\widehat{\text{Vir}^{\,i}}
 = \text{completed cofree coalgebra on } L_n^*, c^* .
\]
This line is only a description of the completed candidate. Its
bar-cobar inversion requires the finite-window convergence hypotheses:
finite quotient stages, compatible counits, and the
Mittag--Leffler condition on the cohomology of the stagewise cones.
The same-family shadow $\mathrm{Vir}_{26-c}$ belongs to the
Verdier/Koszul-dual algebra lane and requires the separate finite-type
or completed comparison. It is not produced by the completed counit
$\Omega\widehat{\bar B}(\mathrm{Vir}_c)\to \mathrm{Vir}_c$.

Without the PBW filtration or an $I$-adic completion,
$\mathrm{Vir}_c$ is outside the strict finite-dual quadratic criterion.
\end{example}

\subsection{Convergence criteria}

\begin{theorem}[Finite-window convergence criterion; \ClaimStatusProvedHere]\label{thm:completion-convergence-criteria}
Let $\mathcal{A}$ carry a descending multiplicative filtration
$F^\bullet\mathcal A$ and set
$\widehat{\mathcal A}=\varprojlim_n\mathcal A/F^n\mathcal A$.
Assume:
\begin{enumerate}
\item $\bigcap_n F^n\mathcal A=0$ and $\mathcal A$ is complete for
 $F^\bullet$;
\item each quotient $\mathcal A/F^n\mathcal A$ is finite over the
 coefficient field in each cohomological degree;
\item every chiral operation is strongly filtered:
 \[
 \mu_r(F^{i_1}\mathcal A,\ldots,F^{i_r}\mathcal A)
 \subset F^{i_1+\cdots+i_r}\mathcal A;
 \]
\item the finite-stage bar-cobar counits
 \[
 \epsilon_n\colon
 \Omega\bar B(\mathcal A/F^n\mathcal A)\to
 \mathcal A/F^n\mathcal A
 \]
 are quasi-isomorphisms in the square-zero
 total model, or coacyclic-cone equivalences in the curved
 second-kind category, and are compatible with the quotient maps;
\item for every $j$, the inverse systems
 $\{H^j(\operatorname{Cone}(\epsilon_n))\}_n$ and
 $\{H^j(\bar B(\mathcal A/F^n\mathcal A))\}_n$ satisfy the
 Mittag--Leffler condition.
\end{enumerate}
Then the completed bar-cobar counit
\[
\widehat\Omega\widehat{\bar B}(\widehat{\mathcal A})\longrightarrow
\widehat{\mathcal A}
\]
is a quasi-isomorphism in the square-zero total model, and has
coacyclic cone in the curved second-kind category. The completed
Koszul-dual coalgebra is the inverse limit
$\widehat{\mathcal A^i}=\varprojlim_n(\mathcal A/F^n\mathcal A)^i$.
Its Verdier or continuous dual gives the completed Koszul-dual algebra
when that duality exists.
\end{theorem}

\begin{proof}
The strong filtration makes each arity-$r$ operation continuous and
forces every fixed quotient $\mathcal A/F^n\mathcal A$ to see only
finitely many filtration windows. The bar construction is therefore
computed stagewise. By hypothesis, the stagewise counits are
quasi-isomorphisms, or coacyclic-cone equivalences in the curved
ambient, and form a compatible inverse system. The Mittag--Leffler
condition kills the $\varprojlim^1$ term in the Milnor exact sequence,
so taking inverse limits preserves the quasi-isomorphism in the
square-zero model. The curved statement is the same cone calculation
read in Positselski's second-kind category. The description of
$\widehat{\mathcal A^i}$ is the same inverse-limit construction
applied to the finite-stage Koszul dual coalgebras.
\end{proof}

\section{Finite-stage existence tests}
\label{sec:algorithmic-test}

\subsection{Regime classifier}

\begin{construction}[Existence test for Koszul duals]\label{con:koszul-dual-existence}
Given an augmented chiral algebra
$\mathcal{A}=\operatorname{Free}(V)/(R)$, the test returns the
strongest regime justified by the data: strict finite-dual coalgebra,
strict Koszul coalgebra, strict post-Verdier algebra, completed
pro-coalgebra, or separate analysis.

\emph{Quadratic test.}
If all relations in $R$ lie in $V^{\otimes 2}$, the strict
finite-dual test is available. Otherwise the strict quadratic test is
silent and the completed test is the only construction in this
appendix.

\emph{PBW regularity.}
Compute the dual space $V^*$ and dual relations
$R^\perp \subseteq (V^*)^{\otimes 2}$. If $(R^\perp)^\perp = R$,
the pairing $V \otimes V^* \to \mathbb{C}$ is perfect in each
weight, and $R$ is a direct summand, the strict finite-dual coalgebra
is defined. If one of these tests fails, the chosen presentation has
not produced a strict finite-dual coalgebra.

\emph{Construction and conilpotency.}
Form the candidate dual
$\mathcal{A}^i_{\mathrm{quad}} \hookrightarrow
\operatorname{Cofree}(sV^*)$ as the maximal sub-coalgebra with
quadratic core $s^2R^\perp$. In the direct-sum cofree coalgebra this
object is conilpotent by tensor length. In an inverse-limit or
completed model, replace conilpotence by separated pro-conilpotence
and check continuity of the coproduct and differential.

The strict Koszul conclusion is stronger than construction of the
candidate coalgebra. It requires the comparison
$q_{\mathcal A}\colon\bar B(\mathcal A)\to
\mathcal A^i_{\mathrm{quad}}$ to be a quasi-isomorphism. PBW
regularity proves this by Theorem~\ref{thm:regular-implies-koszul};
without PBW regularity the output is only a finite-dual candidate.
The post-Verdier algebra $\mathcal A^!$ is returned only after the
finite-type or continuous duality hypotheses of
Proposition~\ref{prop:existence-four-loci}.

\emph{Non-quadratic case.}
Compute the arities of the higher relations. Bounded arity alone is
not a convergence theorem. Choose a descending multiplicative
filtration, verify strong finite-window continuity, verify
finite-stage Koszulness of the quotients, and check the
Mittag--Leffler condition of
Theorem~\ref{thm:completion-convergence-criteria}. The completed dual
coalgebra exists on that surface.

\emph{Failure of the test.}
Failure of the strict finite-dual test is not non-existence in every
ambient category. It means that the chosen presentation has not
produced a strict coalgebra or a strict post-Verdier algebra. Failure
of the completed test means that at least one of finite-stage
Koszulness, finite-stage counit compatibility, continuity, or
Mittag--Leffler convergence remains to be proved.
\end{construction}

\begin{proposition}[Correctness of the existence test;
\ClaimStatusConditional]
\label{prop:existence-test-correctness}
The construction above has the following logical content.
\begin{enumerate}
\item If the quadratic finite-dual and conilpotency tests pass without
 invoking completion, then
 $\mathcal A^i_{\mathrm{quad}}$ exists as a conilpotent strict
 coalgebra. If PBW regularity also holds, then
 $q_{\mathcal A}$ and the counit
 $\epsilon_{\mathcal A}\colon\Omega\bar B(\mathcal A)\to\mathcal A$
 are quasi-isomorphisms.
\item If the completed finite-window test passes, then
 $\widehat{\mathcal A^i}=\varprojlim_n(\mathcal A/F^n\mathcal A)^i$
 exists as a separated complete pro-conilpotent coalgebra, and the
 completed counit is a quasi-isomorphism in the square-zero total
 model, or has coacyclic cone in the second-kind curved category.
\item If the post-Verdier algebra $\mathcal A^!$ is asserted, the
 finite-type Verdier or continuous-dual hypotheses must be supplied in
 addition to \textup{(1)} or \textup{(2)}.
\end{enumerate}
\end{proposition}

\begin{proof}
The first assertion is Theorem~\ref{thm:quadratic-have-duals} plus
Theorem~\ref{thm:regular-implies-koszul}. The second is
Theorem~\ref{thm:completion-convergence-criteria}. The third is
Proposition~\ref{prop:existence-four-loci}. The construction therefore
uses the order
\[
\text{finite-dual data}\to\text{Koszul comparison}\to
\text{bar-cobar counit}\to\text{Verdier or continuous duality},
\]
	and never proves a source-level arrow by assuming a target-side
	comparison.
\end{proof}

\subsection{Examples of algorithm application}

\begin{example}[Heisenberg finite-dual test]\label{ex:algorithm-heisenberg}
For
$\mathcal{H}_k = \text{Free}(a, a^*) / ([a, a^*] - k)$
the homogeneous part of the relation space is
\[R = \{a \otimes a^* - a^* \otimes a\} \subseteq V^{\otimes 2}\]
and the central level~$k$ is curvature, not a quadratic relation.

The dual data are
\[V^* = \text{span}\{a^{\dagger}, (a^*)^{\dagger}\}\]
\[R^\perp = \mathrm{Sym}^2(V^*) = \text{span}\{a^{\dagger} \otimes a^{\dagger},\; (a^*)^{\dagger} \otimes (a^*)^{\dagger},\; a^{\dagger} \otimes (a^*)^{\dagger} + (a^*)^{\dagger} \otimes a^{\dagger}\}\]

Here $(R^\perp)^\perp=R$ and the pairing
$V\otimes V^*\to\mathbb C$ is perfect.

The finite-dual coalgebra is
\[
\mathcal{H}^{i}
 = \text{sub-coalgebra of }
 \text{Cofree}(sa^{\dagger}, s(a^*)^{\dagger})
 \text{ with quadratic core } s^2R^\perp .
\]
\[\Delta(a^{\dagger}) = a^{\dagger} \otimes 1 + 1 \otimes a^{\dagger}\]
The generator is primitive, hence conilpotent by tensor length. The
Koszul-dual coalgebra is $\mathcal H^i$ as constructed, and the
Koszul-dual algebra is
$\mathcal H^!=(\mathcal H^i)^\vee_{\mathrm{cont}}$ when the
continuous graded dual is defined.
\end{example}

\begin{example}[Finite-window recognition for \texorpdfstring{$\mathcal{W}_\infty$}{W\_infinity}]\label{ex:algorithm-w-infinity}
The algebra $\mathcal{W}_\infty$ has infinitely many generators and
relations of unbounded arity. The strict finite-dual quadratic test
therefore returns no coalgebra. The bar coalgebra
$\bar{B}(\mathcal{W}_\infty)$ still exists, but a completed dual is
obtained only from a tower
$\{\mathcal W_{\infty,\le N}\}_N$ satisfying the hypotheses of
Theorem~\ref{thm:completion-convergence-criteria}: finite-stage
Koszulness, compatible counits, continuity of all operations, and the
Mittag--Leffler condition. Theorem~\ref{thm:completed-bar-cobar-strong}
is the comparison theorem after those tower hypotheses have been
verified; it is not a bounded-arity recognition shortcut.
\end{example}

\subsection{Detailed analysis: Kac--Moody}

\begin{proposition}[Affine Kac--Moody existence criterion \cite{FBZ04, Feigin-Frenkel}; \ClaimStatusProvedElsewhere]\label{prop:kac-moody-koszul-duals}
Let $V_k(\mathfrak g)$ be the universal affine vertex algebra of a
simple Lie algebra $\mathfrak g$ at level $k\in\mathbb C$. Then:
\begin{enumerate}
\item $V_k(\mathfrak g)$ satisfies the PBW existence criterion:
 its PBW associated graded is
 $\operatorname{Sym}(\mathfrak g\otimes t^{-1}\mathbb C[t^{-1}])$,
 finite in each conformal weight, hence classically Koszul.
\item The bar-dual coalgebra is the chiral Chevalley--Eilenberg
 coalgebra determined by the dual current generators and the
 Kac--Moody cocycle. The central term is a curved contribution to
 the bar differential, not a new quadratic relation.
\item A post-Verdier algebra notation such as
 $(V_k(\mathfrak g))^!$ requires the finite-type or continuous
 duality hypotheses of this appendix. The genus-$0$ module lane can
 identify the corresponding algebra-side partner with the familiar
 level reflection $k\mapsto -k-2h^\vee$; that is additional
 Verdier/module input, not the existence criterion for the coalgebra.
\item For a simple quotient $L_k(\mathfrak g)$, PBW for the universal
 algebra does not decide Koszulness. Singular vectors in the quotient
 can create off-diagonal bar cohomology; admissible and critical
 levels therefore require the separate null-vector or completed
 analysis recorded in the main Koszulness chapter.
\end{enumerate}
\end{proposition}

\begin{proof}
The universal affine algebra has generators $J^a_{-n}$ and relations
\[
[J^a_m,J^b_n]=f^{ab}_{c}J^c_{m+n}
 + k\,m\,\langle a,b\rangle\delta_{m+n,0}.
\]
The PBW filtration by the number of current modes has associated
graded the symmetric algebra on
$\mathfrak g\otimes t^{-1}\mathbb C[t^{-1}]$. Each conformal-weight
piece is finite-dimensional, and the symmetric algebra is classically
Koszul by Priddy. Proposition~\ref{prop:pbw-universality} gives
chiral Koszulness for the universal algebra.

The orthogonal complement of the current-current relation gives the
Chevalley--Eilenberg coalgebra on dual current generators; the
level-$k$ cocycle supplies curvature in the bar differential. The
level-reflection and Feigin--Frenkel/Langlands formulas live on
different comparison surfaces: they identify algebra-side partners
after Verdier or module-category input. They are not needed to prove
existence of the bar-dual coalgebra. Passing to $L_k(\mathfrak g)$
changes the associated graded by quotienting singular vectors, so the
universal PBW proof no longer applies without an additional
null-vector calculation.
\end{proof}

\subsection{Detailed analysis: $\mathcal{W}$-algebras}

\begin{proposition}[\texorpdfstring{$\mathcal{W}$}{W}-algebra Koszul analysis]\label{prop:w-algebra-koszul}
\begin{enumerate}
\item\emph{(\ClaimStatusConditional)}
The universal $\mathcal{W}$-algebra $\mathcal{W}^k(\fg, f_{\mathrm{prin}})$
is chirally Koszul at every level~$k$, for every simple~$\fg$.
Its bar-dual coalgebra exists by PBW. The post-Verdier algebra
$(\mathcal{W}^k)^!$ exists only on the finite-type or completed
continuous-dual surface: for finitely generated finite-window
$\mathcal{W}_N$ by Theorem~\textup{\ref{thm:completed-bar-cobar-strong}},
and for $\mathcal{W}_\infty$ by weight stabilisation
\textup{(}Theorem~\textup{\ref{thm:stabilized-completion-positive}}\textup{)}.

\item For the simple quotient $\mathcal{W}_k(\fg)$ at admissible
level~$k$, $C_2$-cofiniteness holds~\cite{Arakawa15}, and the Zhu
algebra is finite-dimensional. These are not bar-Koszul criteria. The
remaining open mathematical condition is the vanishing of the null-vector
obstruction
\[
H^j\!\left(
\operatorname{Cone}\bigl(
\bar B(\mathcal W_k(\fg))\to \mathcal W_k(\fg)^i_{\mathrm{quad}}
\bigr)\right)=0
\quad(j\ne0)
\]
in the chiral bar complex, or the corresponding coacyclic-cone
statement in a completed curved ambient.
\end{enumerate}
\end{proposition}

\begin{proof}[Proof of item~\textup{1}]
$\mathcal{W}^k(\fg, f_{\mathrm{prin}})$ is freely strongly generated
by the Feigin--Frenkel theorem.
Proposition~\ref{prop:pbw-universality} gives the bar-dual
coalgebra. The final sentence is the completed finite-window
comparison, not a consequence of PBW alone.
\end{proof}

\section{Finite-stage computation of Koszul duals}
\label{sec:practical-computation}

\subsection{Computable data}

\begin{procedure}[Computing \texorpdfstring{$\mathcal{A}^i$}{Ai} in practice]\label{proc:compute-koszul-dual}
Given $\mathcal{A} = \text{Free}(V) / (R)$:

\emph{Presentation data.}
\begin{enumerate}
\item Write generators $V = \text{span}\{v_1, \ldots, v_n\}$ with explicit degrees
\item Write relations $R = \text{span}\{r_1, \ldots, r_m\}$ in normal form
\item Identify which relations are quadratic, cubic, etc.
\end{enumerate}

\emph{Quadratic finite-dual data.}
\begin{enumerate}
\item Extract quadratic relations $R_2 \subseteq R$
\item Form dual space $V^* = \{v_1^*, \ldots, v_n^*\}$
\item Compute dual quadratic relations $R_2^\perp \subseteq (V^*)^{\otimes 2}$
\item Build quadratic part of dual:
 \[
 \mathcal{A}^i_{\text{quad}}
 = \text{sub-coalgebra of } \text{Cofree}(sV^*)
 \text{ with quadratic core } s^2R_2^\perp .
 \]
\end{enumerate}

\emph{Higher-relation data.}
\begin{enumerate}
\item If $R$ contains only quadratic relations, set
 $\mathcal{A}^i = \mathcal{A}^i_{\text{quad}}$.
\item If $R$ contains higher relations:
 \begin{enumerate}
 \item Choose a descending multiplicative filtration $F^n\mathcal A$
 and verify separated completeness.
 \item Compute successive finite windows
 $\mathcal{A}_n = \mathcal{A} / F^n\mathcal A$.
 \item Compute duals $\mathcal{A}_n^i$ for each $n$ and verify
 stagewise Koszulness.
 \item Check compatibility and the Mittag--Leffler condition, then take
 the inverse limit:
 $\widehat{\mathcal{A}^i} = \varprojlim_n \mathcal{A}_n^i$.
 \end{enumerate}
\end{enumerate}

\emph{Verification data.}
\begin{enumerate}
\item Check conilpotency: $\ker(\varepsilon) = \bigcup_n \ker(\bar{\Delta}^{(n)})$ (every element of the coaugmentation coideal is annihilated by a sufficiently high iterate of the reduced coproduct)
\item Verify bar resolution: Compute $\bar{B}(\mathcal{A})$ and check
 $\bar{B}(\mathcal{A}) \xrightarrow{\sim} \mathcal{A}^i$ as a
 quasi-isomorphism of dg coalgebras, or the corresponding
 finite-window inverse-limit statement in the completed lane.
\item Verify cobar-bar: Compute $\Omega(\bar{B}(\mathcal{A}))$ and
 compare to $\mathcal{A}$ in the strict Koszul lane; in the curved
 completed lane compare
 $\widehat\Omega(\widehat{\bar B}(\widehat{\mathcal A}))$ with
 $\widehat{\mathcal A}$ in the square-zero total model or in the
 second-kind coderived category.
\end{enumerate}
\end{procedure}

\subsection{\texorpdfstring{Worked example: free fermion ($bc$ system)}{Worked example: free fermion}}

\begin{example}[Free fermion (\texorpdfstring{$bc$}{bc} ghost) system]\label{ex:free-fermion-koszul-dual}
The free fermion ($bc$ ghost) system is:
\[\mathcal{F} = \text{Free}(b, c) / (b \otimes c + c \otimes b)\]

\emph{Generators.} $V = \text{span}\{b, c\}$ with $|b| = \lambda$,
$|c| = 1 - \lambda$.

\emph{Relation.} $R = \{b \otimes c + c \otimes b\}$ (anticommutation).

\emph{Dual space.}
\[V^* = \text{span}\{b^*, c^*\}\]
with $|b^*| = -\lambda + 1$, $|c^*| = \lambda$.

\emph{Dual relation.}
\[R^\perp = \operatorname{span}\{b^* \otimes b^*,\; c^* \otimes c^*,\; b^* \otimes c^* - c^* \otimes b^*\}\]

This is 3-dimensional (codimension 1 in $(V^*)^{\otimes 2}$).
Check: $\langle b^* \otimes c^* - c^* \otimes b^*,\, b \otimes c + c \otimes b \rangle = 1 - 1 = 0$.
The symmetric generator $b^* \otimes c^* + c^* \otimes b^*$ has pairing $1 + 1 = 2 \neq 0$, so it is \emph{not} in $R^\perp$.

\emph{Dual coalgebra.}
\[
\mathcal{F}^i
 = \text{sub-coalgebra of } \text{Cofree}(sb^*, sc^*)
 \text{ with quadratic core } s^2R^\perp .
\]

Since $R^\perp$ contains the symmetric tensors $b^{*2}$, $c^{*2}$, and $b^*c^* - c^*b^*$,
this is the \emph{symmetric (polynomial) coalgebra}:
\[\mathcal{F}^i = \mathrm{Sym}^{\bullet,c}(sb^*, sc^*)\]

with the cocommutative shuffle comultiplication.
This is the $\beta\gamma$ system (cf.\ Theorem~\ref{thm:betagamma-fermion-koszul}), consistent with the Lie$\leftrightarrow$Com duality: the $bc$ system has antisymmetric (Lie-type) relations, so its Koszul dual has symmetric (Com-type) relations.

\emph{Verification.}
\begin{itemize}
\item Conilpotency follows from the coradical filtration by polynomial degree.
\item The Koszul-dual coalgebra is detected by bar cohomology
 concentrated on
 the Koszul diagonal, and the resulting coalgebra is
 $\mathcal F^i=\mathrm{Sym}^{\bullet,c}(sb^*,sc^*)$.
\item The Koszul-dual algebra is the graded continuous dual
 $(\mathcal F^i)^\vee_{\mathrm{cont}}\cong\mathcal F^!\cong\beta\gamma$
 on the finite-dual surface.
\item Bar-cobar inversion is the quasi-isomorphism
 $\Omega(\bar{B}(\mathcal{F})) \xrightarrow{\sim} \mathcal{F}$,
 equivalently acyclicity of the Koszul complex off degree~$0$.
\end{itemize}

Hence the free fermion ($bc$ ghost) system is Koszul with symmetric coalgebra dual ($\beta\gamma$ system).
\end{example}

\begin{remark}[Comparison of existence regimes
 \cite{LV12, Positselski11}]
\label{rem:existence-summary}
Three existence regimes govern this monograph and the cited literature:
\begin{enumerate}
\item \emph{Strict quadratic/conilpotent regime.} In the classical
 and quadratic literature, finite-dual quadratic data give an explicit
 strict coalgebra
 \textup{(}Construction~\ref{const:quadratic-dual-criteria}\textup{)}.
 PBW regularity is the additional hypothesis that makes this coalgebra
 the bar-cobar Koszul resolution.
\item \emph{Completed/filter-complete regime.} In the non-quadratic
 setting one seeks a completed dual under explicit convergence and
 inverse-limit hypotheses; Appendix~\ref{app:nilpotent-completion}
 records this frontier package and its standard towers.
\item \emph{Negative or comparison statements.} Claims that no dual
 exists depend on the ambient category, the allowed completion, and the
 level at which one asks for inversion. They are regime-specific
 comparison statements, not a single monograph theorem overriding the
 family-by-family theorem surface.
\end{enumerate}
There is no single constitutional ``if and only if'' criterion for the
whole monograph.
\end{remark}

\subsection{Alternative ambients}

\begin{remark}[Alternatives outside a chosen Koszul-dual regime]\label{rem:no-dual-alternatives}
If a chiral algebra $\mathcal{A}$ has no Koszul dual in the strict
finite-dual or completed regime being used, four replacement surfaces
remain available.

\emph{$A_\infty$ replacement.}
Under the conilpotence and completion hypotheses of the bar-cobar
adjunction, $\Omega(\bar{B}(\mathcal{A}))$ gives an $A_\infty$
model of~$\mathcal{A}$ even when the Koszul-dual coalgebra is not
concentrated in the quadratic degree.

\emph{Derived module category.}
The derived category $\mathcal{D}^b(\mathcal{A}\text{-mod})$ can be
formed without an explicit Koszul dual, although finiteness,
compact-generation, and comparison theorems remain separate
hypotheses.

\emph{Factorization homology.}
Factorization homology $\int_X \mathcal{A}$ can be computed without Koszul dual
(Costello--Gwilliam factorization algebra formalism).

\emph{Finite or compactly generated subcategory.}
Some subcategories of $\mathcal{A}\text{-mod}$ satisfy the
finite-type or compact-generation hypotheses needed for a
module/comodule comparison.
\end{remark}
