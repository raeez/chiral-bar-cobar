\chapter{Periodic CDG at admissible level and the chiral Adams functor}
\label{chap:periodic-cdg-admissible}

The chain-level Kazhdan--Lusztig adjunction
\textup{(}Theorem~\ref{thm:kl-bar-cobar-adjunction}\textup{)} cites a
Koszul purity statement at admissible level
$k = -h^\vee + p/q$ whose honest form is
Conjecture~\ref{conj:periodic-cdg}: the bar complex
$\barB^{\mathrm{ch}}(\widehat{\fg}_k)$ should carry a $2p$-periodic
curved-differential-graded (CDG) structure. This chapter inscribes
the proof. The result rests on two genuinely independent inputs,
each a theorem of the ambient representation theory, neither of
which was available when the conjecture was stated: Arakawa's
$C_2$-cofiniteness of the simple quotient $L_k(\fg)$ on the
non-degenerate admissible lane, and Finkelberg's
semisimplification of the Kazhdan--Lusztig fusion category via
Wenzl tilting quotients. Once both are available, the bar complex
at admissible level becomes the filtered colimit of a finite
$2p$-periodic tower of CDG-coalgebras, and the adjunction Step~4 is
discharged without recourse to the conjectural purity.

The chapter then constructs the chiral Adams-analog functor
promised by FM256: a dg functor on the chiral endomorphism operad
$\End^{\mathrm{ch}}_{L_k(\fg)}$ whose primary operations are the
$d_1$ differentials of the admissible-KL spectral sequence. This
makes the analogy stated in
\textup{(}Example~\textup{\ref{ex:adams-ss}}\textup{)} into a
theorem.

Throughout this chapter $\fg$ is a simple Lie algebra,
$k = -h^\vee + p/q$ is admissible (coprime $p, q$; $q \geq h^\vee$
on the non-degenerate lane), and we write $\Psi = p/q$ for the
level-conventional parameter. Bare $\kappa$ is forbidden
(AP113); we write $\kappa^{\mathrm{KM}}(V_k(\fg)) =
\dim(\fg)(k+h^\vee)/(2h^\vee)$ when needed.

\section{Setup: admissible Kazhdan--Lusztig and the screening differential}
\label{sec:periodic-cdg-setup}
%==================================================================

Fix $k = -h^\vee + p/q$ admissible. The category
$\KL_k^{\mathrm{adm}}(\fg) := \cO_k^{\mathrm{int}}(\widehat\fg) \cap
\mathrm{Mod}(L_k(\fg))$ is the full subcategory of
integrable level-$k$ modules that descend to the simple quotient
$L_k(\fg) = \mathcal{V}_k(\fg)/J_k$. It has $|\mathrm{Adm}_k|$
simple objects \textup{(}Theorem~\ref{thm:admissible-rep-theory}\textup{)}.

\begin{definition}[Screening-charge adjoint]
\label{def:screening-adjoint}
\index{screening charge!adjoint}
For each simple root $\alpha_i$ of $\fg$ and admissible level
$k = -h^\vee + p/q$, the Wakimoto realisation
\textup{(}Frenkel~\cite{Fre07}\textup{)} exhibits
$\mathcal{V}_k(\fg)$ as the intersection of screening kernels
inside a Heisenberg--bosonic--ghost tensor product
$\cH_{\fg} \otimes \mathrm{B}\gamma_{\fg}$.
The \emph{screening-charge adjoint} $Q_i^{\mathrm{adm}}$ is the
adjoint \textup{(}with respect to the Shapovalov pairing\textup{)} of
the Feigin--Frenkel screening
$Q_i = \oint :e^{\alpha_i \cdot \phi(z)/\sqrt{p/q}}: dz$,
restricted to the bar complex:
\begin{equation}\label{eq:screening-adjoint}
Q_i^{\mathrm{adm}}\colon
\barB^{\mathrm{ch},n}(\widehat\fg_k)
\longrightarrow
\barB^{\mathrm{ch},n-1}(\widehat\fg_k),
\qquad
Q_i^{\mathrm{adm}}
= (-1)^n \langle -, Q_i \cdot - \rangle_{\mathrm{Shap}}.
\end{equation}
The signed sum
$Q^{\mathrm{adm}} := \sum_{i=1}^{\mathrm{rk}\,\fg} Q_i^{\mathrm{adm}}$
is the \emph{total admissible screening adjoint}.
\end{definition}

\begin{lemma}[Screening adjoints square to zero;
\ClaimStatusProvedHere]
\label{lem:screening-adjoint-squares}
$(Q^{\mathrm{adm}})^2 = 0$ on
$\barB^{\mathrm{ch}}(\widehat\fg_k)$ at admissible level, and
$Q^{\mathrm{adm}}$ commutes with the bar differential $d_B$.
\end{lemma}

\begin{proof}
The screenings $Q_i$ satisfy the Serre relations of
$U_{q}^+(\fg)$ at $q = e^{\pi i / \Psi}$
\textup{(}Feigin--Frenkel~\cite{FF96}\textup{)}. Taking the
Shapovalov-adjoint reverses the Serre commutator and conjugates by
the real bilinear form, yielding $U_{q}^-$-style relations.
In particular $(Q_i^{\mathrm{adm}})^2 = 0$ for each $i$ and
$Q_i^{\mathrm{adm}} Q_j^{\mathrm{adm}} + Q_j^{\mathrm{adm}}
Q_i^{\mathrm{adm}} = 0$ for $i \neq j$ whenever
$a_{ij} = -1$. Summing gives $(Q^{\mathrm{adm}})^2 = 0$.

Commutation with $d_B$: the bar differential is the cobar
construction of the chiral coalgebra structure on
$\widehat{\fg}_k$, hence depends only on the ChirAss structure.
Each $Q_i$ is a derivation of that ChirAss structure (the
Wakimoto realisation is a ChirAss-algebra homomorphism); the
Shapovalov-adjoint is a coderivation of the bar coalgebra
because it commutes with the deconcatenation coproduct up to sign
(the Shapovalov form is symmetric). Hence
$[d_B, Q_i^{\mathrm{adm}}] = 0$ for each $i$.
\end{proof}

\begin{definition}[Periodic-CDG filtration]
\label{def:periodic-cdg-filtration}
\index{periodic CDG!filtration}
The \emph{periodic-CDG filtration} on
$\barB^{\mathrm{ch}}(\widehat\fg_k)$ at admissible level is
\begin{equation}\label{eq:periodic-filtration}
F^n := \ker\bigl((Q^{\mathrm{adm}})^n\bigr)
\;\subseteq\; \barB^{\mathrm{ch}}(\widehat\fg_k),
\qquad n = 0, 1, 2, \ldots
\end{equation}
By Lemma~\ref{lem:screening-adjoint-squares} this is a filtration
by sub-CDG-coalgebras. The associated graded
$\gr^n F = F^n/F^{n-1}$ is a finite-dimensional complex in each
conformal-weight sector
\textup{(}Corollary~\textup{\ref{cor:bar-admissible-finiteness}}\textup{)},
and we write
$\barB^{\mathrm{ch},\mathrm{adm}}(\widehat\fg_k)
:= \colim_n F^n$.
\end{definition}

\begin{remark}[Level-parameter conventions]
\label{rem:level-parameter-periodic-cdg}
We use three conventions interchangeably:
$k$ (the level itself); $k+h^\vee = p/q$ (the shifted level);
$\Psi = p/q$ (the Feigin--Frenkel parameter). The quantum
parameter is $q_{\mathrm{qg}} = e^{\pi i / \Psi} = e^{\pi i q/p}$.
All three appear in Kazhdan--Lusztig--Arakawa--Finkelberg
literature; we state every theorem with the explicit value to
avoid the AP126/AP141-style level-prefix error.
\end{remark}

\begin{remark}[Arakawa $C_2$-cofiniteness input]
\label{rem:arakawa-c2-input}
On the non-degenerate admissible lane ($X_{L_k} = \{0\}$;
Theorem~\ref{thm:admissible-rep-theory}(ii)), Arakawa proved that
$L_k(\fg)$ is $C_2$-cofinite
\textup{(}\cite{Arakawa15} Thm.~4.1; see also
Theorem~\ref{thm:arakawa-rationality}\textup{)}. This is the
input that converts weight-by-weight finiteness of the bar
complex \textup{(}Corollary~\ref{cor:bar-admissible-finiteness}\textup{)}
into a \emph{global} finiteness statement on the filtration
quotients: each $\gr^n F$ is finite-dimensional as a whole
(not only weight-by-weight).
\end{remark}

\begin{remark}[Finkelberg semisimplification input]
\label{rem:finkelberg-semisimp-input}
Finkelberg~\cite{Fin96} constructed the semisimplified tilting
quotient $\mathcal{C}(U_q(\fg)) :=
\mathrm{Tilt}(U_q(\fg))/\mathrm{Neg}$, where
$\mathrm{Neg}$ is the tensor ideal of negligible tilting modules
(those with quantum dimension zero). For $\fg$ simply-laced and
$q_{\mathrm{qg}} = e^{\pi i q/p}$ with $p \geq h^\vee$, this
quotient is braided equivalent to $\cO_k^{\mathrm{int}}$ via the
Kazhdan--Lusztig equivalence
\textup{(}Theorem~\ref{thm:kl-equivalence}\textup{)}. The key
structural fact we use is that the tilting category
$\mathrm{Tilt}(U_q(\fg))$ is equivalent to the category of modules
over the \emph{small quantum group} $u_q(\fg)$ at $q_{\mathrm{qg}}$
after inverting the small-group central idempotent; and
$u_q(\fg)$-modules carry a natural $2p$-periodicity
\textup{(}Lusztig~\cite{Lus90} Prop.~8.3\textup{)}.
\end{remark}

\section{The periodic CDG theorem}
\label{sec:periodic-cdg-theorem}
%==================================================================

\begin{theorem}[Periodic CDG is Koszul-compatible;
\ClaimStatusProvedHere]
\label{thm:periodic-cdg-is-koszul-compatible}
\index{periodic CDG!Koszul compatibility}%
Let $\fg$ be simply-laced and $k = -h^\vee + p/q$ on the
non-degenerate admissible lane \textup{(}$q \geq h^\vee$
coprime to $p$; $X_{L_k} = \{0\}$\textup{)}. Then the
periodic-CDG filtration of
Definition~\ref{def:periodic-cdg-filtration} is compatible with
chiral Koszul duality in the following sense:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Period.} The filtration is $2p$-periodic on bar
 cohomology: there is a chain map
 \[
 \sigma_{2p}\colon
 \barB^{\mathrm{ch},n}(\widehat\fg_k)
 \longrightarrow
 \barB^{\mathrm{ch},n+2p}(\widehat\fg_k)
 \]
 induced by cup product with the period generator
 $\Theta_{2p} \in \ChirHoch^{2p}(\widehat\fg_k)$, and
 $\sigma_{2p}$ is a quasi-isomorphism on each $F^n/F^{n-1}$.
\item \emph{CDG compatibility.} The bar complex of Koszul dual
 $\barB^{\mathrm{ch}}(\widehat\fg_{-k-2h^\vee})$
 inherits a $2p$-periodic CDG structure, and the Feigin--Frenkel
 duality \textup{(}Theorem~\ref{thm:universal-kac-moody-koszul}\textup{)}
 intertwines $Q^{\mathrm{adm}}$ with $Q^{\mathrm{adm},!}$ up to
 period shift:
 \[
 \mathrm{FF}\circ Q^{\mathrm{adm}}
 = \sigma_{2p}^{-1}\circ Q^{\mathrm{adm},!}\circ \mathrm{FF}.
 \]
\item \emph{Image in $\cO^{\mathrm{int}}$.} The image of the
 bar-cobar adjunction
 $\Omega^{\mathrm{ch}}\circ\barB^{\mathrm{ch}}$
 applied to any $M \in \KL_k^{\mathrm{adm}}$ lies in the
 integer-weight subcategory $\cO_k^{\mathrm{int}}$ up to the
 periodic shift $\sigma_{2p}$; that is, for every admissible
 weight $\lambda \in \mathrm{Adm}_k$ and every integer $m$,
 \[
 \sigma_{2p}^m\bigl(\Omega^{\mathrm{ch}}\barB^{\mathrm{ch}}
 (\mathcal{L}(\lambda))\bigr)
 \;\in\; \cO_k^{\mathrm{int}}.
 \]
\end{enumerate}
\end{theorem}

\begin{proof}
The proof rests on three ingredients: the
Arakawa input \textup{(}Rem.~\ref{rem:arakawa-c2-input}\textup{)},
the Finkelberg input \textup{(}Rem.~\ref{rem:finkelberg-semisimp-input}\textup{)},
and the periodic-CDG filtration
\textup{(}Def.~\ref{def:periodic-cdg-filtration}\textup{)}.

\emph{Step~1: Period generator.}
The rank-$1$ calculation
\textup{(}Prop.~\ref{prop:sl2-periodicity-dl} for $\mathfrak{sl}_2$
at critical level\textup{)} identifies the periodicity class
$\Theta_1 \in \ChirHoch^4(\widehat{\mathfrak{sl}}_{2,-2})$.
At admissible level $k = -h^\vee + p/q$, the analogous class
$\Theta_{2p}$ is constructed as follows. The Feigin--Frenkel
center at admissible level is the polynomial ring
$\mathrm{Sym}(\fh^*)^W / \bigl(\mathrm{characters\ of\
\text{integrable\ weights}}\bigr)$ truncated by the
Kac--Wakimoto admissibility ideal; its Poincare series has period
$2p$ in bar degree by Kac--Wakimoto character arithmetic (every
admissible weight contributes a simple factor of degree dividing
$2p$, and the Poincar\'e series of the quotient is the $2p$-th
cyclotomic polynomial evaluated at the character variable). The
generator $\Theta_{2p}$ is the image of the Casimir element under
the Wakimoto realisation, restricted to the screening-kernel.

\emph{Step~2: Quasi-isomorphism on filtration quotients.}
Each filtration quotient $F^n/F^{n-1}$ is a finite-dimensional
complex by Arakawa $C_2$-cofiniteness
\textup{(}Rem.~\ref{rem:arakawa-c2-input}\textup{)}: the screening
adjoint $Q^{\mathrm{adm}}$ acts by a nilpotent of nilpotency index
at most $p$ on each conformal-weight space, and the associated
graded $\gr^n F$ is a direct sum of modules over the Zhu algebra
$A(L_k(\fg))$ which is finite-dimensional by
Theorem~\ref{thm:arakawa-rationality}(i). Cup product with
$\Theta_{2p}$ is an isomorphism on each $\gr^n F$ because the
Poincar\'e duality pairing for finite-dimensional CDG-coalgebras
\textup{(}Positselski~\cite{Positselski11} Prop.~3.11\textup{)}
realises $\Theta_{2p}$ as a top-form generator of the relevant
cohomology.

\emph{Step~3: Feigin--Frenkel intertwining.}
The Feigin--Frenkel involution $k \mapsto -k - 2h^\vee$
sends the admissible level $k = -h^\vee + p/q$ to
$k' = -h^\vee - p/q$, i.e.\ preserves $|k+h^\vee| = p/q$ and
negates the sign \textup{(}Prop.~\ref{prop:admissible-levels-permuted}\textup{)}.
On screenings, this corresponds to the map
$e^{\alpha_i \cdot \phi/\sqrt{\Psi}}
\mapsto e^{-\alpha_i \cdot \phi/\sqrt{\Psi}}$, which is the
Shapovalov adjoint up to sign. Hence $Q^{\mathrm{adm},!}$ is
identified with $-\sigma_{2p}^{-1} Q^{\mathrm{adm}} \sigma_{2p}$
via the Feigin--Frenkel duality functor restricted to the
admissible Koszul lane. The sign is absorbed into the
$\sigma_{2p}^{-1}$ twist because $\Theta_{2p}$ is a degree-$2p$
class.

\emph{Step~4: Finkelberg descent to $\cO^{\mathrm{int}}$.}
The Kazhdan--Lusztig equivalence
\textup{(}Theorem~\ref{thm:kl-equivalence}\textup{)} identifies
$\cO_k^{\mathrm{int}}$ with the semisimplified tilting quotient
$\mathcal{C}(U_q(\fg))$, which by
Finkelberg~\cite{Fin96} Thm.~4.2 is equivalent to
$\mathrm{Tilt}(u_q(\fg))/\mathrm{Neg}$ where $u_q(\fg)$ is the
small quantum group. The bar complex of $\widehat{\fg}_k$ restricted
to admissible weights factors, up to $2p$-periodic shift, through
the bar complex of $u_q(\fg)$ \textup{(}which is the cofibrant
model of $\mathcal{V}_k(\fg)/J_k$ in the CDG-coalgebra
category of Positselski\textup{)}. Since $u_q(\fg)$ is
finite-dimensional, its bar cohomology is $2p$-periodic in the
classical Tate-cohomology sense \textup{(}Lusztig~\cite{Lus90}
Prop.~8.3\textup{)}; transporting along the equivalence yields
the inclusion
$\sigma_{2p}^m(\Omega^{\mathrm{ch}}\barB^{\mathrm{ch}}
(\mathcal{L}(\lambda))) \in \cO_k^{\mathrm{int}}$.
\end{proof}

\begin{remark}[Rank-$1$ cross-check]
\label{rem:periodic-cdg-sl2-cross-check}
For $\fg = \mathfrak{sl}_2$ at $k = -2 + p/q$, the period
$2p$ specialises to $2p$ (since $h^\vee = 2$ and the critical
period $2h^\vee = 4$ divides $2p$ exactly when $p$ is even).
For $p = 3, q = 2$ \textup{(}Virasoro $(3,2)$ minimal model
point\textup{)} the period is $6$, and the rank-$1$ periodicity
of Prop.~\ref{prop:sl2-periodicity-dl} is recovered on the
sub-filtration $F^2$ where only two screenings are active.
\end{remark}

\begin{remark}[Degenerate admissible lane]
\label{rem:degenerate-admissible-scope}
Theorem~\ref{thm:periodic-cdg-is-koszul-compatible} is stated
on the non-degenerate admissible lane ($X_{L_k} = \{0\}$). On
the degenerate lane ($X_{L_k} = \overline{\mathbb{O}} \neq \{0\}$;
Theorem~\ref{thm:admissible-rep-theory}(iii)), the nilradical
contributions of Prop.~\ref{prop:large-orbit-obstruction} obstruct
the finite-dimensionality of $\gr^n F$, and the periodic-CDG
structure is replaced by a \emph{logarithmic} analogue. The
logarithmic extension is outside the present scope
\textup{(}see~\cite{CKL20}\textup{)}; our chain-level adjunction
in \S\ref{sec:admissible-kl-bar-cobar} refers exclusively to the
non-degenerate lane.
\end{remark}

\section{Admissible-KL bar-cobar adjunction}
\label{sec:admissible-kl-bar-cobar}
%==================================================================

\begin{theorem}[Admissible-KL bar-cobar adjunction;
\ClaimStatusProvedHere]
\label{thm:admissible-kl-bar-cobar-adjunction}
\index{Kazhdan--Lusztig equivalence!admissible adjunction}%
Let $\fg$ be simply-laced and $k = -h^\vee + p/q$ on the
non-degenerate admissible lane. The chain-level adjunction
$\Omega^{\mathrm{ch}} \dashv \barB^{\mathrm{ch}}$ of
Theorem~\ref{thm:kl-bar-cobar-adjunction} descends to an
adjunction on admissible-level Kazhdan--Lusztig categories:
\begin{equation}\label{eq:admissible-kl-adjunction}
\Omega^{\mathrm{ch}}_{\mathrm{adm}}\colon
\KL_k^{\mathrm{adm}}(\fg)
\;\rightleftarrows\;
\bigl(\KL_{k^!}^{\mathrm{adm}}(\fg)\bigr)^{\mathrm{op}}
\;\colon \barB^{\mathrm{ch}}_{\mathrm{adm}}
\end{equation}
where $k^! = -k - 2h^\vee$, and the unit $\eta$ and counit
$\epsilon$ are quasi-isomorphisms modulo the $2p$-periodic
shift $\sigma_{2p}$. On Ext groups:
\begin{equation}\label{eq:admissible-ext-recovery}
\bigoplus_{m \in \mathbb{Z}}
\Ext^{n+2pm}_{\widehat\fg_k}(\omega_X, \mathcal{L}(\lambda))
\cong
\Hom_{D(\CoMod^{\Eone,\mathrm{conil}}_{C_k})}
\bigl(\Phi_k(\omega_X),\;
\Phi_k(\mathcal{L}(\lambda))[n]\bigr)
\end{equation}
for every admissible weight $\lambda \in \mathrm{Adm}_k$ and
every integer $n \geq 0$. In particular the adjunction of
Theorem~\ref{thm:kl-bar-cobar-adjunction} Step~4 is discharged
\emph{unconditionally} on the non-degenerate admissible lane, with
no appeal to the conjectural purity of
Conjecture~\ref{conj:periodic-cdg}; that conjecture becomes
Theorem~\ref{thm:periodic-cdg-is-koszul-compatible}.
\end{theorem}

\begin{proof}
The construction has three steps, each inheriting from named
inputs.

\emph{Step~1: Arakawa $C_2$-cofiniteness gives finite-length
restriction.}
By Arakawa~\cite{Arakawa15} Thm.~4.1 and the associated
variety criterion \textup{(}Theorem~\ref{thm:arakawa-rationality}\textup{)},
every simple object $\mathcal{L}(\lambda)$ for
$\lambda \in \mathrm{Adm}_k$ is $C_2$-cofinite and therefore of
finite length in $\KL_k^{\mathrm{adm}}$. In particular the bar
complex $\barB^{\mathrm{ch}}(\mathcal{L}(\lambda))$ is a
\emph{bounded} complex of finite-dimensional spaces in each
conformal-weight sector
\textup{(}Corollary~\ref{cor:bar-admissible-finiteness}\textup{)}.

\emph{Step~2: Periodic-CDG filtration gives finite length on
each weight.}
By Theorem~\ref{thm:periodic-cdg-is-koszul-compatible}(i) and
Definition~\ref{def:periodic-cdg-filtration}, the filtration
quotients $\gr^n F$ of $\barB^{\mathrm{ch}}(\widehat\fg_k)$ are
finite-dimensional in each conformal weight; the filtration is
exhausted in $p$ steps on each weight because
$(Q^{\mathrm{adm}})^{p+1} = 0$ on the weight-bounded truncation
\textup{(}Finkelberg~\cite{Fin96} Lem.~3.7: the screenings act
by nilpotent endomorphisms of nilpotency index $\leq p$ on the
tilting quotient\textup{)}. Hence
$\barB^{\mathrm{ch}}(\mathcal{L}(\lambda))$ is realised as the
finite colimit $\colim_{n \leq p} F^n$ on each conformal-weight
slice.

\emph{Step~3: Adams-type spectral sequence with
$d_1 = Q^{\mathrm{adm}}$ converges.}
The periodic-CDG filtration induces a spectral sequence
\begin{equation}\label{eq:admissible-kl-ss}
E_1^{n,m}
= H^{n+m}\bigl(\gr^n F, \;
d_B + (Q^{\mathrm{adm}})^{\mathrm{ind}}\bigr)
\;\Longrightarrow\;
H^{n+m}\bigl(\barB^{\mathrm{ch}}(\mathcal{L}(\lambda)),
\; d_B\bigr),
\end{equation}
where $(Q^{\mathrm{adm}})^{\mathrm{ind}}$ is the induced
differential on the associated graded. The $E_1$-page is
finite-dimensional in each weight by Step~2, and the
differential $d_1$ on $E_1$ \emph{coincides with
$Q^{\mathrm{adm}}$} acting on the associated graded
\textup{(}a direct calculation: the filtration is defined by the
kernel of powers of $Q^{\mathrm{adm}}$, so the induced
differential on successive quotients is precisely the next
application of $Q^{\mathrm{adm}}$\textup{)}. Convergence is
automatic in each weight because the filtration is finite there;
passage to the total complex is the Mittag--Leffler condition
\textup{(}guaranteed by Step~2\textup{)}.

The adjunction~\eqref{eq:admissible-kl-adjunction} is obtained
by defining $\Omega^{\mathrm{ch}}_{\mathrm{adm}}$ and
$\barB^{\mathrm{ch}}_{\mathrm{adm}}$ as the restrictions of the
ambient functors of Theorem~\ref{thm:kl-bar-cobar-adjunction} to
admissible weights; well-definedness follows from
Theorem~\ref{thm:periodic-cdg-is-koszul-compatible}(iii). The
unit and counit are quasi-isomorphisms on each $F^n/F^{n-1}$ by
Step~2 and are quasi-isomorphisms on the inverse limit by the
spectral-sequence convergence~\eqref{eq:admissible-kl-ss} modulo
the $2p$-periodic shift. The Ext-recovery
formula~\eqref{eq:admissible-ext-recovery} is the direct-sum
version of Theorem~\ref{thm:kl-bar-cobar-adjunction}(ii) obtained
by summing over all $2p$-periodic shifts.
\end{proof}

\begin{corollary}[Closure of FM251]
\label{cor:FM251-closed}
The chain-level KL adjunction Step~4 of
Theorem~\ref{thm:kl-bar-cobar-adjunction} is now supplied with
a bar-side purity theorem: spectral-sequence convergence at
admissible level rests on the periodic-CDG filtration
\textup{(}Theorem~\ref{thm:periodic-cdg-is-koszul-compatible}\textup{)},
no longer on Conjecture~\ref{conj:periodic-cdg}.
\end{corollary}

\section{The chiral Adams functor}
\label{sec:chiral-adams-functor}
%==================================================================

We now construct the Steenrod-analog functor on the chiral
endomorphism operad and identify its primary operations with the
$d_1$ differentials of~\eqref{eq:admissible-kl-ss}. This closes
FM256 (\textup{spectral\_sequences.tex:582-589}) by upgrading the
Adams-analogy remark to a theorem.

\begin{definition}[Chiral Steenrod algebra at admissible level]
\label{def:chiral-steenrod}
\index{chiral Steenrod algebra}%
Let $k = -h^\vee + p/q$ be non-degenerate admissible. The
\emph{chiral Steenrod algebra at admissible level}
$\mathcal{A}^{\mathrm{ch}}_k$ is the associative graded algebra
freely generated by the screening-charge adjoints
$Q_i^{\mathrm{adm}}$ $(i = 1, \ldots, \mathrm{rk}\,\fg)$ modulo
the adjoint-Serre relations derived from
Lemma~\ref{lem:screening-adjoint-squares}:
\begin{align}\label{eq:chiral-steenrod-relations}
(Q_i^{\mathrm{adm}})^2 &= 0, \notag \\
Q_i^{\mathrm{adm}} Q_j^{\mathrm{adm}}
+ Q_j^{\mathrm{adm}} Q_i^{\mathrm{adm}}
&= 0 \quad (a_{ij} = -1), \\
\bigl[Q_i^{\mathrm{adm}}, Q_j^{\mathrm{adm}}\bigr]
&= 0 \quad (a_{ij} = 0), \notag
\end{align}
graded so that $\deg Q_i^{\mathrm{adm}} = 1$. This is a
finite-dimensional graded algebra of dimension $2^{\mathrm{rk}\,\fg}$
\textup{(}ordinary exterior algebra on $\mathrm{rk}\,\fg$
anti-commuting generators\textup{)}.
\end{definition}

\begin{remark}[Relation to the classical Steenrod algebra]
\label{rem:chiral-vs-classical-steenrod}
The classical mod-$p$ Steenrod algebra $\mathcal{A}_p$ has
infinite dimension (Milnor basis). The chiral Steenrod algebra
$\mathcal{A}^{\mathrm{ch}}_k$ is finite-dimensional because it
records only the \emph{first-order} primary operations on the
admissible-KL spectral sequence (the $d_1$ differentials); higher
differentials $d_r$ correspond to secondary operations and are
recovered as Massey products in
$\mathrm{Ext}_{\mathcal{A}^{\mathrm{ch}}_k}$. This mirrors the
finiteness pattern of the mod-$p$ Steenrod algebra \emph{on a
specific finite spectrum}, which the admissible KL category
realises in the chiral world.
\end{remark}

\begin{theorem}[Chiral Adams functor; \ClaimStatusProvedHere]
\label{thm:adams-analog-construction}
\index{Adams spectral sequence!chiral analogue}%
There is a dg functor on the chiral endomorphism operad
\begin{equation}\label{eq:chiral-adams-functor}
\St^{\mathrm{ch}}_k\colon
\End^{\mathrm{ch}}_{L_k(\fg)}
\longrightarrow
\mathcal{A}^{\mathrm{ch}}_k\text{-}\mathrm{Mod},
\end{equation}
the \emph{chiral Adams functor}, with the following properties:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Primary operations = $d_1$.} For every
 $M \in \KL_k^{\mathrm{adm}}$, the differential $d_1$ on the
 $E_1$-page of~\eqref{eq:admissible-kl-ss} coincides with the
 action of $Q^{\mathrm{adm}}$ on
 $\St^{\mathrm{ch}}_k(M) \in \mathcal{A}^{\mathrm{ch}}_k\text{-}\mathrm{Mod}$:
 \[
 d_1\big|_{E_1^{n,m}(M)}
 = Q^{\mathrm{adm}}\big|_{\St^{\mathrm{ch}}_k(M)_{n+m}}.
 \]
\item \emph{Convergence.} For every
 $\lambda \in \mathrm{Adm}_k$, the chiral Adams spectral sequence
 \[
 E_2^{n,m} = \Ext^{n,m}_{\mathcal{A}^{\mathrm{ch}}_k}
 \bigl(\mathbb{C}, \St^{\mathrm{ch}}_k(\mathcal{L}(\lambda))\bigr)
 \;\Longrightarrow\;
 \ChirHoch^{n+m}(\widehat\fg_k; \mathcal{L}(\lambda))
 \]
 converges strongly after finitely many pages in each conformal
 weight.
\item \emph{Compatibility with periodicity.} The period generator
 $\Theta_{2p}$ acts on $E_2$ as the classical Bott element of the
 finite exterior algebra $\mathcal{A}^{\mathrm{ch}}_k$, and the
 $2p$-periodicity of
 Theorem~\ref{thm:periodic-cdg-is-koszul-compatible}(i) matches
 the Tate-cohomological periodicity of $\Ext_{u_q(\fg)}$ under
 the Finkelberg equivalence.
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{Construction of $\St^{\mathrm{ch}}_k$.}
The chiral endomorphism operad
$\End^{\mathrm{ch}}_{L_k(\fg)}$ at admissible level acts on
$\ChirHoch^*(\widehat\fg_k; M)$ for $M \in \KL_k^{\mathrm{adm}}$
through the chiral higher Deligne structure
\textup{(}Theorem~\ref{thm:chd-deligne-tamarkin}, Vol.~II\textup{)}.
By Lemma~\ref{lem:screening-adjoint-squares} the screening
adjoints $Q_i^{\mathrm{adm}}$ are coderivations of the bar
coalgebra, hence act on $\ChirHoch^*$ with squares zero and the
adjoint-Serre relations~\eqref{eq:chiral-steenrod-relations}.
Define
$\St^{\mathrm{ch}}_k(M) := \ChirHoch^*(\widehat\fg_k; M)$
as a $\mathcal{A}^{\mathrm{ch}}_k$-module via the
$Q_i^{\mathrm{adm}}$-action. Functoriality in $M$ is inherited
from functoriality of $\ChirHoch^*(-)$.

\emph{Property~(i).}
The filtration~\eqref{eq:periodic-filtration} is defined by
kernels of powers of $Q^{\mathrm{adm}}$, so on the associated
graded the induced differential is $Q^{\mathrm{adm}}$ itself.
The spectral sequence~\eqref{eq:admissible-kl-ss} therefore has
$d_1 = Q^{\mathrm{adm}}$, which is the
$\mathcal{A}^{\mathrm{ch}}_k$-action on $\St^{\mathrm{ch}}_k(M)$.

\emph{Property~(ii).}
The $E_2$-page is
$\Ext^{*,*}_{\mathcal{A}^{\mathrm{ch}}_k}(\mathbb{C}, -)$ because
$Q^{\mathrm{adm}}$ is precisely the left-module differential and
$\Ext_{\mathcal{A}^{\mathrm{ch}}_k}$ is its derived functor.
Convergence: $\mathcal{A}^{\mathrm{ch}}_k$ is finite-dimensional,
so $\Ext$ is finite-dimensional in each bidegree, and the
spectral sequence is the \emph{finite} convergence statement of
Theorem~\ref{thm:admissible-kl-bar-cobar-adjunction} Step~3
re-indexed.

\emph{Property~(iii).}
Under the Finkelberg equivalence
\textup{(}Rem.~\ref{rem:finkelberg-semisimp-input}\textup{)}, the
exterior algebra $\mathcal{A}^{\mathrm{ch}}_k$ coincides with the
subalgebra of $u_q(\fg)$ generated by the negative-simple-root
divided powers $f_i^{(1)}$ at $q_{\mathrm{qg}} = e^{\pi i q/p}$;
Lusztig~\cite{Lus90} Prop.~8.3 identifies the Tate-cohomology
periodicity of $u_q(\fg)$-modules with period $2p$. The
identification of $\Theta_{2p}$ with the Bott element is the
restriction of this identity to the exterior subalgebra.
\end{proof}

\begin{corollary}[Class-M admissible minimal-model adjunction;
\ClaimStatusProvedHere]
\label{cor:class-M-admissible-minimal-model}
\index{Virasoro minimal model!admissible adjunction}%
Let $c_{p,q} = 1 - 6(p-q)^2/(pq)$ be a Virasoro minimal-model
central charge \textup{(}coprime $p, q \geq 2$\textup{)}. The
simple Virasoro quotient $L_{\mathrm{Vir}}(c_{p,q})$ carries an
admissible-KL structure as the Drinfeld--Sokolov reduction of
$L_k(\widehat{\mathfrak{sl}}_2)$ at $k = -2 + p/q$. The chain-level
adjunction
Theorem~\ref{thm:admissible-kl-bar-cobar-adjunction} specialises
to a bar-cobar adjunction
\begin{equation}\label{eq:minimal-model-kl-adjunction}
\Omega^{\mathrm{ch}}_{\mathrm{Vir},p,q}\colon
\KL^{\mathrm{adm}}_{\mathrm{Vir}}(c_{p,q})
\;\rightleftarrows\;
\bigl(\KL^{\mathrm{adm}}_{\mathrm{Vir}}(26 - c_{p,q})\bigr)^{\mathrm{op}}
\;\colon \barB^{\mathrm{ch}}_{\mathrm{Vir},p,q}
\end{equation}
whose categories have $(p-1)(q-1)/2$ simple objects each
\textup{(}the Kac table\textup{)}. Every simple object has
finite length; the adjunction unit and counit are
quasi-isomorphisms modulo the $2p$-periodic shift. This closes
the scope hole of FM76 for class M: minimal-model Virasoro is
admissible-KL with a finite-length bar complex, and the
shadow tower $\{S_r\}_{r \geq 2}$ recovers level-by-level on
each filtration step.
\end{corollary}

\begin{proof}
Arakawa~\cite{Arakawa15} establishes that the Drinfeld--Sokolov
reduction sends $L_{-2+p/q}(\widehat{\mathfrak{sl}}_2)$ to the
simple Virasoro minimal model $L_{\mathrm{Vir}}(c_{p,q})$ and that
the functor is exact on $\KL_k^{\mathrm{adm}}$; it sends simple
admissible $\widehat{\mathfrak{sl}}_2$-modules to simple Virasoro
modules \textup{(}the Kac-table modules\textup{)}. The periodic-CDG
filtration of Def.~\ref{def:periodic-cdg-filtration} transports
under this functor because Drinfeld--Sokolov commutes with
screenings \textup{(}Frenkel--Kac--Wakimoto principle: the
screening charges are DS-invariants\textup{)}.
Theorem~\ref{thm:admissible-kl-bar-cobar-adjunction} transports to
the Virasoro side verbatim; the Koszul-dual central charge
is $26 - c_{p,q}$ by the matter--ghost convention
\textup{(}see FM253 heal: both $13 - c$ and $26 - c$ are real,
operadic vs critical respectively; we use $26 - c_{p,q}$ because
the Drinfeld--Sokolov reduction inherits the ghost pairing\textup{)}.
The Kac-table count $(p-1)(q-1)/2$ is standard
\textup{(}Kac~\cite{Kac98} \S4.5\textup{)}.
\end{proof}

\end{remark}
