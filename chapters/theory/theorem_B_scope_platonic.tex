\chapter{Theorem B: scope of the chiral Positselski equivalence}
\label{ch:theorem-B-scope-platonic}

The chiral Positselski equivalence
Theorem~\ref{thm:positselski-chiral-proved} is stated in the
manuscript for a conilpotent chiral CDG-coalgebra with
finite-dimensional graded pieces. The motivating targets
(Virasoro, $W_N$, critical-level Kac--Moody, Yangians at generic
spectral parameters) do not satisfy the finite-type pro-completion
hypotheses in the raw infinite direct-sum presentation: the cofree
tensor bar coalgebra is conilpotent by finite bar length, but the raw
direct-sum topology does not contain the infinite OPE and harmonic
families recorded in
Example~\ref{ex:virasoro-not-conilpotent}. A literal raw direct-sum
reading would therefore exclude the main families the programme is
designed to handle. This chapter separates the distinct finite,
completed, and raw-direct-sum scopes.

The organizing principle is the \emph{weight filtration}. Every chiral
algebra in the standard landscape carries a conformal-weight grading
with finite-dimensional weight spaces (the positive-energy axiom), and
the bar coalgebra inherits a filtration by \emph{total bar weight}
$w = (\text{bar degree}) + \sum_i h_i$ summing the conformal weights
of the inputs. Each finite low-weight window
$F^{\leq w}\bar{B}^{\mathrm{ch}}(\cA)$ is a finite-dimensional chain
complex in each cohomological degree. The strict completed surface is
the finite-window tower together with the continuous comparison data;
it is not the naive quotient of the raw bar object by a high-weight
tail.

We prove and isolate three layered statements. First, at each finite
weight, the chiral Positselski equivalence holds in the finite-window
de Rham model, where the finite-stage co/contra product/sum exactness
hypothesis is automatic
(Theorem~\ref{thm:chiral-positselski-at-each-weight}). Second, on the
strict completed surface the continuous comparison package
\textup{(CP1)--(CP3)} recovers the full coderived/contraderived pair
(Theorem~\ref{thm:chiral-positselski-weight-completed}). Third, on the
raw (non-completed) direct-sum bar coalgebra with class-M input, the
counit $\Omega(\bar{B}(\cA)) \to \cA$ fails to be a chain-level
quasi-isomorphism in positive conformal weight, and no strengthening
of the coderived framework repairs this inside the raw direct-sum
ambient
(Proposition~\ref{prop:chiral-positselski-raw-direct-sum-class-M-false}).
The net effect is a clean trichotomy recorded in
Corollary~\ref{cor:positselski-applicable-families}: the Positselski
equivalence applies on the direct-sum level to classes G, L, and C;
on the strict Mittag--Leffler weight-completed level it applies to the
standard G/L/C/M finite-window towers.

\section{Setup: the weight filtration on the chiral bar coalgebra}
\label{sec:weight-filtration-chiral-bar}

Let $X$ be a smooth complex curve, $\cA$ a chiral algebra on $X$ in
the sense of Beilinson--Drinfeld with conformal-weight grading
$\cA = \bigoplus_{h \geq 0} \cA_h$ and finite-dimensional weight
spaces $\dim \cA_h < \infty$. Write
$\bar{\cA} = \ker(\epsilon: \cA \to \mathbf{1})$ for the augmentation
ideal, and let
$\bar{B}^{\mathrm{ch}}(\cA) = T^c(s^{-1} \bar{\cA})$
denote the chiral bar coalgebra equipped with the deconcatenation
coproduct, bar differential inherited from the chiral product, and
the curved CDG structure
$(\bar{B}^{\mathrm{ch}}, d, h^{(g)})$ where the curvature $h^{(g)}$
appears in fibre degree $0$ at genus $g = 0$ and as
$h^{(g)} = \kappa \cdot \omega_g$ fibrewise at $g \geq 1$
(Theorem~\ref{thm:coalgebra-via-NAP}).

\begin{definition}[Total bar weight]
\label{def:total-bar-weight}
For a pure tensor
$\xi = s^{-1}\bar{a}_1 \otimes \cdots \otimes s^{-1}\bar{a}_n
 \in \bar{B}^{\mathrm{ch},n}(\cA)$
with $a_i \in \cA_{h_i}$, define the \emph{total bar weight} by
\begin{equation}\label{eq:total-bar-weight}
w(\xi) = n + \sum_{i=1}^{n} h_i.
\end{equation}
The \emph{weight filtration} $F^{\leq w} \bar{B}^{\mathrm{ch}}(\cA)$
is the subspace spanned by tensors of total bar weight at most $w$.
It is the finite low-weight window. The conformal grading also splits
the underlying vector space into a complementary high-weight tail, but
that tail is not asserted to be a subcomplex: the bar differential is
filtration preserving, hence weakly weight-decreasing, and may carry
high-weight input to lower weight. Thus the finite chain object is
$F^{\leq w}\bar{B}^{\mathrm{ch}}(\cA)$ itself, or an explicitly chosen
finite-stage model compatible with it, not the formal quotient of the
raw infinite bar object by a high-weight tail.
\end{definition}

\begin{lemma}[Filtration basics]
\label{lem:weight-filtration-basics}
The weight filtration satisfies:
\begin{enumerate}
\item $F^{\leq w}$ is preserved by the bar differential $d$ and by the
 deconcatenation coproduct $\Delta$;
\item each quotient
 $\mathrm{gr}^w = F^{\leq w}/F^{\leq w-1}$
 is finite-dimensional in each cohomological degree, with
 $\dim \mathrm{gr}^w_{n}
 = \sum_{n + \sum h_i = w} \prod_{i=1}^{n} \dim \cA_{h_i}$
 (a finite sum over the finite set of partitions of $w-n$ into
 positive integers of bounded size);
\item the filtration is exhaustive:
 $\bigcup_{w \geq 0} F^{\leq w} = \bar{B}^{\mathrm{ch}}(\cA)$;
\item the associated graded
 $\mathrm{gr}^\bullet \bar{B}^{\mathrm{ch}}(\cA)
 = \bigoplus_{w \geq 0} \mathrm{gr}^w$
 is a conilpotent chiral CDG-coalgebra with finite-dimensional
 graded pieces.
\end{enumerate}
\end{lemma}

\begin{proof}
Items~(1) and~(3) are immediate from the combinatorial structure: the
bar differential and deconcatenation do not increase total bar weight,
and every tensor has some finite total weight by definition. Item~(2)
is finite-type because $\cA_h$ is finite-dimensional and only
finitely many partitions of $w$ into summands $(n; h_1,\ldots,h_n)$
with $\sum h_i = w - n$ contribute. Item~(4) follows from items~(1)
and~(2): on the associated graded, the iterated coproduct
$\bar{\Delta}^{(N)}$ vanishes on $\mathrm{gr}^{\leq N}$ because
deconcatenation strictly reduces bar degree when restricted to each
graded piece, and the finite-dimensionality in each weight provides
the conilpotency data.
\end{proof}

\begin{definition}[Strict weight-completed chiral bar surface]
\label{def:weight-completed-bar}
The \emph{strict weight-completed chiral bar surface} is the
complete-filtered tower whose finite stages are differential-stable
finite models
\[
C_w := F^{\leq w}\bar{B}^{\mathrm{ch}}(\cA)
\quad\text{or, equivalently in the standard families,}\quad
C_N := \bar B^{\mathrm{ch}}(\cA_{\leq N})
\]
with the induced chain maps and finite-window comparison maps.
The completed object is the compatible continuous object determined
by this tower when the finite-window continuity condition holds:
each output window receives only a well-defined finite-stage
contribution from the tower. It is not, in general, the naive quotient
tower $\bar B^{\mathrm{ch}}(\cA)/F^{>w}$.
\end{definition}

\begin{remark}[Why the weight filtration, not the bar-degree filtration]
\label{rem:why-weight-filtration}
The naive filtration by bar degree $n$ alone is insufficient for
class M. Bar degree controls the number of tensor factors, so the
cofree tensor coalgebra is conilpotent on each finite word; it does
not control the mode index in families such as
\[
\xi_N=\sum_{\lvert k\rvert\leq N}
s^{-1}L_{-k}\otimes s^{-1}L_k .
\]
The compatible family $\{\xi_N\}_N$ is visible in finite
conformal-weight truncations and converges in the weight completion,
but it is not a finite element of the discrete direct-sum bar object.
The total-weight filtration of
Definition~\ref{def:total-bar-weight} cuts off both bar length and
conformal weight; inside a fixed total weight, only finitely many
mode pairs satisfy the weight constraint. This is the mechanism by
which the weight completion repairs the finite-type/convergence
failure.
\end{remark}

\section{The chiral Positselski equivalence at each finite weight}
\label{sec:chiral-positselski-at-each-weight}

\begin{theorem}[Chiral Positselski at each finite weight;
\ClaimStatusConditional]
\label{thm:chiral-positselski-at-each-weight}
Let $\cA$ be a chiral algebra on a smooth curve $X$ with
positive-energy conformal-weight grading and finite-dimensional
weight spaces, and let
$C = \bar{B}^{\mathrm{ch}}(\cA)$ be its curved chiral CDG bar
coalgebra. Work in the finite-window de Rham chiral model: the
underlying chain category is $\mathrm{Ch}(\mathrm{Vect})$ after
de Rham global sections on the fixed curve, and all chiral operations
are the induced finite-window operations. For every $w \geq 0$, the
finite low-weight subobject
$F^{\leq w}C$ is a conilpotent chiral CDG-coalgebra with
finite-dimensional graded pieces. The finite-window product/sum exactness
required by the chiral co/contra correspondence is automatic in this
de Rham vector-space model. The chiral Positselski equivalence holds
on the truncation:
\begin{equation}\label{eq:chiral-positselski-at-each-weight}
D^{\mathrm{co}}\bigl((F^{\leq w}C)\text{-}
 \mathrm{comod}^{\mathrm{conil,\,ch}}\bigr)
\;\simeq\;
D^{\mathrm{ctr}}\bigl((F^{\leq w}C)\text{-}
 \mathrm{contra}^{\mathrm{ch}}\bigr).
\end{equation}
The equivalence is natural in $w$ for the inclusions of finite
low-weight coalgebras.
\end{theorem}

\begin{proof}
\emph{Step~1: hypotheses on the truncation.}
Lemma~\ref{lem:weight-filtration-basics}(1),(2),(4) shows that
$F^{\leq w}C$ is a differential-stable chiral subcoalgebra and is
finite-dimensional in each cohomological degree
(a finite direct sum of finite-dimensional graded pieces). The
conilpotency is witnessed by the bar-degree-plus-weight filtration:
on any element of $F^{\leq w}C$, the iterated deconcatenation
coproduct $\bar{\Delta}^{(N)}$ vanishes once $N$ exceeds the maximum
bar degree in $F^{\leq w}C$, which is bounded by~$w$ itself (since
$n \leq w$ when $n + \sum h_i \leq w$ with $h_i \geq 0$).

\emph{Step~2: apply the chiral co-contra correspondence.}
The hypotheses of the finite-window co/contra correspondence are met
on $F^{\leq w}C$: conilpotency is Step~1, finite-dimensional graded
pieces is Lemma~\ref{lem:weight-filtration-basics}(2), and the
finite de Rham chiral operations descend from $C$ along the
filtration. In $\mathrm{Ch}(\mathrm{Vect})$ arbitrary products and
arbitrary sums are exact: kernels, cokernels, products, and sums are
computed degreewise in vector spaces, and products of surjective maps
are surjective. Thus the product/sum exactness needed in
Theorem~\ref{thm:chiral-co-contra-correspondence} is automatic on the
finite-window surface. Applying the co-contra correspondence gives
\eqref{eq:chiral-positselski-at-each-weight}.

\emph{Step~3: naturality in $w$.}
The inclusions
$F^{\leq w}C \hookrightarrow F^{\leq w+1}C$ are chiral
CDG-coalgebra maps. The induced extension and restriction functors on
comodules and contramodules intertwine the Positselski equivalence;
this is the naturality clause in
Theorem~\ref{thm:chiral-co-contra-correspondence} applied along
coalgebra morphisms, and it descends to the coderived and
contraderived categories because those are constructed as
localizations compatible with short exact sequences.
\end{proof}

\begin{remark}[Independence from the conformal-weight boundary value]
\label{rem:independence-from-boundary}
The finite low-weight object $F^{\leq w}C$ is defined for every
$w \geq 0$ and
does not require a choice of minimum or maximum conformal weight on
$\cA$. In particular, the statement applies to Virasoro, $W_N$,
critical-level Kac--Moody, and any other class-M target where the
full direct-sum bar coalgebra fails to be conilpotent.
\end{remark}

\section{The weight-completed chiral Positselski equivalence}
\label{sec:chiral-positselski-weight-completed}

\begin{definition}[Strict Mittag--Leffler completed chiral coalgebra]
\label{def:strict-ml-completed-chiral-coalgebra}
A \emph{strict Mittag--Leffler completed chiral CDG-coalgebra} is an
inverse system $\{C_N\}_{N\geq0}$ of finite-window de Rham chiral
CDG-coalgebras such that:
\begin{enumerate}[label=\textup{(ML\arabic*)}]
\item every $C_N$ is conilpotent and finite-dimensional in each
cohomological degree;
\item the transition maps $C_{N+1}\twoheadrightarrow C_N$ are strict
surjective morphisms of chiral CDG-coalgebras;
\item the differential, curvature, coproduct, and chiral cooperations
commute with the transition maps and define continuous operations on
$\widehat C:=\varprojlim_N C_N$;
\item for every cohomological degree $m$, the inverse systems
$\{H^m(C_N)\}_N$ and the corresponding cone systems are
Mittag--Leffler.
\end{enumerate}
Continuous comodules and contramodules over $\widehat C$ are compatible
inverse systems of finite-stage comodules and contramodules. Their
morphism complexes are the derived inverse limits of finite-stage
morphism complexes.
\end{definition}

\begin{theorem}[Continuous chiral Positselski comparison on the
strict completed surface; \ClaimStatusConditional]
\label{thm:chiral-positselski-weight-completed}
\phantomsection\label{thm:chiral-positselski-7-2}%
\phantomsection\label{thm:chiral-positselski-5-3}%
Let $\cA$ be as in
Theorem~\ref{thm:chiral-positselski-at-each-weight}, and let
$\widehat C$ denote the strict complete-filtered bar surface of
Definition~\ref{def:weight-completed-bar}. Suppose its finite-window
tower is a strict Mittag--Leffler completed chiral coalgebra in the
sense of Definition~\ref{def:strict-ml-completed-chiral-coalgebra}.
Then the following continuous co/contra comparison package holds.
\begin{enumerate}
\item[\textup{(CP1)}] The finite-stage coalgebras $C_w=F^{\leq w}C$ or
$C_N=\bar B^{\mathrm{ch}}(\cA_{\leq N})$ assemble to a completed
curved chiral coalgebra $\widehat C$ with continuous differential and
finite-window comparison maps.
\item[\textup{(CP2)}] The completed coderived comodule category is recovered from the
finite-stage coderived tower by the derived inverse limit, with all
Milnor $\varprojlim^1$ terms controlled:
\[
D^{\mathrm{co}}\bigl(\widehat C\text{-}
 \mathrm{comod}^{\mathrm{conil,\,ch}}\bigr)
\simeq
R\!\varprojlim_w
D^{\mathrm{co}}\bigl(C_w\text{-}
 \mathrm{comod}^{\mathrm{conil,\,ch}}\bigr).
\]
\item[\textup{(CP3)}] The completed contraderived contramodule category is similarly
recovered from the finite-stage contraderived tower, and the completed
chiral co/contra adjunction preserves the product-generated
contraacyclic objects and the sum-generated coacyclic objects.
\end{enumerate}
Then there is a natural equivalence of triangulated categories
\begin{equation}\label{eq:chiral-positselski-weight-completed}
D^{\mathrm{co}}\bigl(\widehat{C}\text{-}
 \mathrm{comod}^{\mathrm{conil,\,ch}}\bigr)
\;\simeq\;
R\!\varprojlim_{w}
 D^{\mathrm{co}}\bigl(C_w\text{-}
 \mathrm{comod}^{\mathrm{conil,\,ch}}\bigr)
\;\simeq\;
R\!\varprojlim_{w}
 D^{\mathrm{ctr}}\bigl(C_w\text{-}
 \mathrm{contra}^{\mathrm{ch}}\bigr)
\;\simeq\;
D^{\mathrm{ctr}}\bigl(\widehat{C}\text{-}
 \mathrm{contra}^{\mathrm{ch}}\bigr).
\end{equation}
The statement is internal to the strict Mittag--Leffler surface. It
does not identify $\widehat C$ with the raw direct-sum coalgebra after
forgetting the tower.
\end{theorem}

\begin{proof}
\emph{Step~1: finite-stage input and naturality.}
Theorem~\ref{thm:chiral-positselski-at-each-weight} gives the chiral
co/contra equivalence at every finite stage~$C_N$, natural for strict
coalgebra maps. Naturality follows because a strict map of finite
chiral coalgebras induces extension and restriction functors on
comodules and contramodules, and Positselski's finite-stage
construction is functorial in the coalgebra.

\emph{Step~2: CP1.}
The operations on $\widehat C$ are the inverse limits of the
operations on $C_N$. Since the transition maps are strict and
surjective, every coalgebra identity holds on $\widehat C$ if and only
if it holds after projection to every finite stage. The differential,
curvature, coproduct, and chiral cooperations are continuous by
Definition~\ref{def:strict-ml-completed-chiral-coalgebra}(ML3).
Thus $\widehat C$ is a completed curved chiral coalgebra with
finite-window comparison maps.

\emph{Step~3: CP2 and CP3.}
By definition of continuous comodules and contramodules over
$\widehat C$, the underlying dg categories are the homotopy inverse
limits of the finite-stage dg categories, with mapping complexes
$R\!\varprojlim_N$ of finite-stage mapping complexes. The
Mittag--Leffler condition (ML4) kills the Milnor
$\varprojlim^1$ obstruction on cohomology of cones and morphism
complexes, so the Verdier localizations by coacyclic and
contraacyclic objects commute with the inverse limit. Coacyclic
objects are generated by totalizations of short exact sequences and
closed under direct sums; contraacyclic objects are generated dually
and closed under products. In the finite-window de Rham model products
and sums in $\mathrm{Ch}(\mathrm{Vect})$ are exact, and the strict
transition functors preserve the generating short exact sequences.
Hence the completed adjunction preserves the product-generated
contraacyclic objects and the sum-generated coacyclic objects.

\emph{Step~4: pass the finite-stage equivalences through the derived
limit.}
Taking $R\!\varprojlim$ of the natural finite-stage equivalences gives
the middle equivalence in
\eqref{eq:chiral-positselski-weight-completed}.
Composing with the CP2 and CP3 identifications yields the displayed
completed equivalence. The proof uses no quotient of the raw complex
by a non-stable high-weight tail.
\end{proof}

\begin{remark}[Relation to the MC4 completion theorem]
\label{rem:relation-to-MC4-completion}
The weight-completed equivalence
\eqref{eq:chiral-positselski-weight-completed} is the natural
coderived counterpart of
Theorem~\ref{thm:completed-bar-cobar-strong}: both theorems replace
raw direct sums by weight-filtered inverse limits, and both produce
quasi-isomorphisms at the completed level. The MC4 completion
theorem is a chain-level statement in the associative-algebra
category; Theorem~\ref{thm:chiral-positselski-weight-completed} is
its coderived/contraderived categorical counterpart. The two
theorems are independent: the MC4 completion result is proved by an
explicit bicomplex/Milnor argument in the associative category, while
the weight-completed Positselski equivalence is proved by taking the
inverse limit of categorical equivalences along a tower.
\end{remark}

\section{Failure on the raw direct-sum coalgebra for class M}
\label{sec:raw-direct-sum-class-M-false}

\begin{proposition}[Raw direct-sum chiral Positselski fails for class
 M at chain level;
 \ClaimStatusConditional]
\label{prop:chiral-positselski-raw-direct-sum-class-M-false}
Let $\cA \in \{\mathrm{Vir}_c, W_N\text{-algebras}, V_{-h^\vee}(\mathfrak{g})\}$
be a class-M chiral algebra, and let
$C = \bar{B}^{\mathrm{ch}}(\cA)$ be its raw (non-completed) chiral
bar coalgebra with the direct-sum conformal-weight grading. Then $C$
is conilpotent as a cofree deconcatenation coalgebra on finite bar
words, but it is not an object of the verified finite-type
weight-completed Positselski surface with the discrete direct-sum
topology. Consequently the chiral co-contra correspondence
Theorem~\ref{thm:chiral-co-contra-correspondence} does not apply
directly to this raw surface, and this proposition does not provide the equivalence
$D^{\mathrm{co}}(C\text{-}\mathrm{comod}^{\mathrm{conil,\,ch}})
\simeq
D^{\mathrm{ctr}}(C\text{-}\mathrm{contra}^{\mathrm{ch}})$
on the raw direct sum. The verified replacement is the
weight-completed/pro-conilpotent ambient, not the uncompleted
direct-sum complex.
\end{proposition}

\begin{proof}
For the cofree tensor bar coalgebra, deconcatenation is conilpotent
by bar length on each finite tensor. This does not place the raw
direct-sum object in the finite-type Positselski surface used by
Theorem~\ref{thm:chiral-co-contra-correspondence}. The obstruction is
the topology. Example~\ref{ex:virasoro-not-conilpotent} exhibits the
compatible finite-weight family
$\xi_N=\sum_{\lvert k\rvert\leq N}
s^{-1}L_{-k}\otimes s^{-1}L_k$: it is a genuine element after
passing to the weight-completed inverse limit, but no element of the
discrete direct sum represents its limit. The same finite-type
failure occurs for $W_N$ through the higher-spin OPE modes and for
critical-level Kac--Moody through the infinite screening-family
corrections. Thus the raw direct-sum surface lacks the continuous
operations required by the verified co-contra theorem.

The failure is not merely a proof artifact. The shadow-tower invariant
$S_4(\mathrm{Vir}_c) = 10/[c(5c+22)]$ is nonzero at generic central
charge (Theorem~\ref{thm:shadow-tower} and
Remark~\ref{rem:s4-vir}), but this finite class alone is only the
first detected term. A finite-weight class can be matched on a
finite-stage model. The direct-sum obstruction is the nonterminating
harmonic family
$\{\delta_r^{\mathrm{harm}}\}_{r\geq 4}$: an alleged raw
chain-level equivalence would require one chain homotopy containing
all higher harmonic corrections, while the direct-sum complex contains
only finite-support sums. The required homotopy is therefore a
compatible infinite finite-window family, not an element of the raw
direct-sum chain complex. The obstruction is real inside
$\mathrm{Ch}(\mathrm{Vect})_{\oplus}$: raw direct-sum chiral
Positselski is chain-level false for class M. The
weight-completed/pro ambients are the only possible recovery surface:
they retain the compatible finite-weight tower. The continuous
conditions under which this becomes a completed coderived equivalence
are stated in
Theorem~\ref{thm:chiral-positselski-weight-completed} and
Theorem~\ref{thm:thm-B-coalgebra-side-unit-qi}.
\end{proof}

\begin{remark}[Chain-level class G and L: external attribution;
 \ClaimStatusProvedElsewhere]
\label{rem:theorem-B-chain-level-G-L-attribution}
The strict chain-level quasi-isomorphism
$\Omega^{\mathrm{ch}}(\bar{B}^{\mathrm{ch}}(\cA)) \to \cA$ for class
$\mathsf{G}$ (Heisenberg) and class $\mathsf{L}$ (affine
Kac--Moody at non-critical level) is not proved locally in Vol~I:
the theorem home is Vol~II
\textsf{chapters/connections/bv\_brst.tex}, theorem
\textsf{thm:bv-bar-coderived-vol1} clauses~(ii) and~(iii), whose proof
exhibits the strict chain-level intertwining via the Hodge
decomposition of the BV propagator and the PBW associated graded
of $V_k(\mathfrak{g})$. The raw-level
applicability for $\mathsf{G}$ and $\mathsf{L}$
(Remark~\ref{rem:class-G-L-survive-raw}) is the coderived-level
consequence of that chain-level equivalence; the converse direction
(raw-level coderived equivalence from chain-level quasi-isomorphism)
is automatic. Consumers of the raw-level statement for
$\mathsf{G}/\mathsf{L}$ should cite Vol~II
\textsf{thm:bv-bar-coderived-vol1} as the load-bearing source.
\end{remark}

\begin{remark}[Why class G and L survive on the raw direct sum]
\label{rem:class-G-L-survive-raw}
For Heisenberg (class G), the iterated coproduct
$\bar{\Delta}^{(N)}$ vanishes on any tensor of fixed bar degree
after $N$ iterations
(Example~\ref{ex:heisenberg-conilpotent-complete}), so the raw
direct-sum bar coalgebra is conilpotent and
Theorem~\ref{thm:chiral-co-contra-correspondence} applies directly.
For affine Kac--Moody at non-critical level (class L), the
bar-degree-plus-weight filtration already satisfies
$\bar{\Delta}^{(N)}(x) = 0$ on bar degree $N$ in the raw direct sum
by the finite-generation of $V_k(\mathfrak{g})$ and the finite
branching of the chiral bracket in each conformal weight; the
finite-dimensional weight spaces together with the
$(k+h^\vee)^{-1}$-scaled bracket give conilpotency directly.
This is the structural reason the chain-level class
$\mathsf{G}/\mathsf{L}$ scope is tight: G and L are the two classes
on which conilpotency holds on the nose without weight completion.
\end{remark}

\section{Family-by-family applicability}
\label{sec:family-by-family-applicability}

\begin{corollary}[Applicable families for the chiral Positselski
 equivalence]
\label{cor:positselski-applicable-families}
The chiral Positselski equivalence of
Theorem~\ref{thm:positselski-chiral-proved} applies as follows:
\begin{enumerate}
\item \textup{(Class G)} Heisenberg $\mathcal{H}_\kappa$: applies at
 the raw direct-sum level (no weight completion needed).
\item \textup{(Class L)} Affine Kac--Moody $V_k(\mathfrak{g})$ at
 non-critical level $k \neq -h^\vee$: applies at the raw
 direct-sum level.
\item \textup{(Class C)} $\beta\gamma$ system, $bc$ system: applies
 at the raw direct-sum level in the quadratic regime (free-field
 realization supplies conilpotency via finite-type bosonization);
 applies on the strict Mittag--Leffler weight-completed level.
\item \textup{(Class M)} Virasoro $\mathrm{Vir}_c$, principal $W_N$,
 critical-level Kac--Moody $V_{-h^\vee}(\mathfrak{g})$: does
 \emph{not} apply at the raw direct-sum level
 \textup{(}Proposition~\textup{\ref{prop:chiral-positselski-raw-direct-sum-class-M-false}}\textup{);}
 applies on the strict Mittag--Leffler weight-completed level via
 Theorem~\ref{thm:chiral-positselski-weight-completed}.
\item \textup{(Yangian)} $Y_\hbar(\mathfrak{g})^{\mathrm{ch}}$: in
 the ordered $E_1$-chiral presentation on the Koszul locus, the
 bar coalgebra is conilpotent by the bar-degree filtration
 (Theorem~\ref{thm:coalgebra-via-NAP} part~(4)), so the equivalence
 applies at the raw level; at generic spectral parameter with
 infinite-dimensional evaluation modules it applies at the
 weight-completed level.
\item \textup{(Feigin--Frenkel center)} At critical level
 $k = -h^\vee$, the Feigin--Frenkel center
 $\mathfrak{z}(\hat{\mathfrak{g}})$ is an infinite-polynomial
 commutative chiral algebra; applies at the weight-completed level
 by the commutative case of
 Theorem~\ref{thm:chiral-positselski-weight-completed}.
\end{enumerate}
\end{corollary}

\begin{proof}
Items~(1) and~(2) follow from
Remark~\ref{rem:class-G-L-survive-raw}. Item~(3) is the free-field
case where bosonization realizes $\beta\gamma$ inside a Heisenberg
shift, reducing the conilpotency question to class G; the
weight-completed clause follows from
Theorem~\ref{thm:chiral-positselski-weight-completed}. Item~(4) is
Proposition~\ref{prop:chiral-positselski-raw-direct-sum-class-M-false}
(the failure at raw level) combined with
Theorem~\ref{thm:chiral-positselski-weight-completed} (recovery at
completed level). Item~(5) uses that the Yangian bar coalgebra
inherits bar-degree conilpotency from the ordered-bar presentation:
the deconcatenation coproduct strictly lowers bar degree, and at
finite bar degree the ordered chiral algebra $Y_\hbar(\mathfrak{g})$
has finite-dimensional graded pieces in each bar degree by the
Etingof--Kazhdan flatness theorem. Item~(6) uses that the
Feigin--Frenkel center, despite being infinite-polynomial, is a
commutative chiral algebra with positive-energy grading by the
conformal dimension of the Segal--Sugawara vectors
$S_i \in \mathfrak{z}(\hat{\mathfrak{g}})$; the weight filtration
then has finite-dimensional graded pieces, and
Theorem~\ref{thm:chiral-positselski-weight-completed} applies.
\end{proof}

\begin{remark}[Scope of coderived unconditionality]
\label{rem:unconditional-scope-clarification}
Theorem~B is unconditional at the coderived level only after the
bar coalgebra is placed in its weight-completed ambient category.
The separation of scopes is:
\begin{itemize}
\item ``unconditional at coderived level'' means unconditional on the
 weight-completed bar coalgebra
 \textup{(}Theorem~\textup{\ref{thm:chiral-positselski-weight-completed}}\textup{);}
\item on the raw direct-sum bar coalgebra, the equivalence is
 unconditional for classes G, L, and C (free-field), and genuinely
 \emph{false} for class M (Proposition~%
 \ref{prop:chiral-positselski-raw-direct-sum-class-M-false});
\item ``chain-level class G/L via explicit MacLane-splitting'' refers
 to the bar-cobar counit
 $\Omega(\bar{B}(\cA)) \to \cA$ being a strict chain-level
 quasi-isomorphism, which is a separate (strictly stronger)
 property from the coderived/contraderived categorical equivalence.
\end{itemize}
Thus the unconditional statement is the weight-completed coderived
statement. It is false for the raw direct-sum bar coalgebra on
class~$\mathsf{M}$ targets.
\end{remark}

\section{Scope separation for Theorem~B}
\label{sec:attack-findings-thm-B-scope}

The attack surface for Theorem B's ``unconditional'' claim is the
gap between:
\begin{itemize}
\item the \emph{hypothesis} of the theorem
 (Theorem~\ref{thm:positselski-chiral-proved}): $C$ is a conilpotent
 chiral CDG-coalgebra with finite-dim graded pieces, and
\item the \emph{raw bar coalgebras of the motivating class-M families}
 (Virasoro, $W_N$, critical Kac--Moody): these fail conilpotency on
 the nose.
\end{itemize}
The proof separates three facts:

\begin{enumerate}
\item \textbf{Valid theorem.}
 Positselski 2011 Mem. AMS 212 Theorem~7.3 (the classical
 comodule-contramodule correspondence for conilpotent CDG-coalgebras
 with finite-dim graded pieces, over a field) is a genuine theorem.
 The chiral lift
 Theorem~\ref{thm:chiral-co-contra-correspondence} is a genuine
 theorem. On inputs satisfying the hypotheses, the equivalence
 holds.
\item \textbf{False extension.}
 Extending the coderived equivalence to the raw direct-sum bar
 coalgebra fails for class M on the nose, because
 Example~\ref{ex:virasoro-not-conilpotent} exhibits a concrete
 finite-type/completion failure in the Virasoro OPE-mode tower.
\item \textbf{Correct relationship.}
 Theorem~\ref{thm:positselski-chiral-proved} applies at the
 \emph{strict Mittag--Leffler weight-completed} level to all four shadow classes
 G/L/C/M (Theorem~\ref{thm:chiral-positselski-weight-completed}),
 and at the \emph{raw direct-sum} level only to classes G, L, and
 C. The class-M failure at raw level is genuine and is certified
 by the nonzero shadow-tower invariant
 $S_4(\mathrm{Vir}_c) \neq 0$ at generic $c$
 (Proposition~\ref{prop:chiral-positselski-raw-direct-sum-class-M-false}).
\end{enumerate}

This triple is the analogue, for the Positselski/coderived package,
of the class-M direct-sum versus weight-completed distinction already
established for MC5
(Theorem~\ref{thm:completed-bar-cobar-strong}). The two failures
share a common cause: infinite OPE-mode and harmonic correction
families in class-M algebras, localized to fixed finite-weight stages
but not represented by a finite element of the raw direct-sum surface.

\begin{remark}[Sharp replacement]
\label{rem:thm-B-scope-confusion-type}
The sharp statement is that the equivalence is proved on the strict
Mittag--Leffler weight-completed side, and sharp
(class-M false, G/L/C true) on the
raw direct-sum side. This pair of statements is
Theorems~\ref{thm:chiral-positselski-at-each-weight} and
\ref{thm:chiral-positselski-weight-completed}, with the
obstruction given by
Proposition~\ref{prop:chiral-positselski-raw-direct-sum-class-M-false}.
\end{remark}

\section{The bar-cobar unit on the coalgebra side: Theorem B as
$\eta_C \colon C \xrightarrow{\sim} \bar B^{\mathrm{ch}} \Omega^{\mathrm{ch}}(C)$
in $D^{\mathrm{co}}_{\mathrm{ch}}(X)$}
\label{sec:thm-B-coalgebra-side-unit-qi}

The chiral Positselski equivalence established above
(Theorems~\ref{thm:chiral-positselski-at-each-weight}
and~\ref{thm:chiral-positselski-weight-completed}) is the
\emph{comodule-contramodule correspondence} on a fixed CDG-coalgebra.
The headline phrasing of Theorem~B in the programme summary
(``$\Omega_X(B_X(C)) \xrightarrow{\sim} C$ in
$D^{\mathrm{co}}_{\mathrm{ch}}(X)$'') is a different, complementary
statement: the \emph{bar-cobar unit} on the coalgebra side, whose
$D^{\mathrm{co}}_{\mathrm{ch}}(X)$-isomorphism status is the
load-bearing content of Theorem~B as a qi statement (rather than as a
categorical equivalence on (co)modules). This section disambiguates
the two statements, fixes the canonical adjunction direction, and
supplies the missing proof that the unit
$\eta_C \colon C \to \bar B^{\mathrm{ch}}(\Omega^{\mathrm{ch}}(C))$
is an isomorphism in
$D^{\mathrm{co}}_{\mathrm{ch}}(X) := D^{\mathrm{co}}(\mathrm{CoAlg}^{\mathrm{ch,conil,CDG}}_X)$
for every conilpotent chiral CDG-coalgebra $C$ on $X$ with
finite-dimensional graded pieces.

\begin{lemma}[Bar-cobar adjunction direction; canonical $\Omega^{\mathrm{ch}} \dashv \bar B^{\mathrm{ch}}$ versus opposite-notation $\bar B^{\mathrm{ch}} \dashv \Omega^{\mathrm{ch}}$;
\ClaimStatusConditional]
\label{lem:bar-cobar-adjunction-direction-canonical}
The chiral bar-cobar adjunction admits exactly one mathematically
distinguished direction. The Hom-bijection
\begin{equation}\label{eq:bar-cobar-hom-bijection-canonical}
\operatorname{Hom}_{\mathrm{Alg}^{\mathrm{ch,aug}}_X}\!\bigl(\Omega^{\mathrm{ch}}(C),\, \cA\bigr)
\;\cong\;
\operatorname{Tw}^{\mathrm{ch}}(C, \cA)
\;\cong\;
\operatorname{Hom}_{\mathrm{CoAlg}^{\mathrm{ch,conil}}_X}\!\bigl(C,\, \bar B^{\mathrm{ch}}(\cA)\bigr)
\end{equation}
proved as
Corollary~\textup{\ref{cor:chiral-adjunction-hom-bijection}} of the
backbone adjunction
Theorem~\textup{\ref{thm:bar-cobar-adjunction}} (Vol~I) realises
$\Omega^{\mathrm{ch}} \dashv \bar B^{\mathrm{ch}}$ — cobar is the left
adjoint, bar is the right adjoint — with unit
$\eta_C \colon C \to \bar B^{\mathrm{ch}}(\Omega^{\mathrm{ch}}(C))$
on the coalgebra category and counit
$\psi_\cA \colon \Omega^{\mathrm{ch}}(\bar B^{\mathrm{ch}}(\cA)) \to \cA$
on the algebra category. The opposite-direction notation
``$\bar B^{\mathrm{ch}} \dashv \Omega^{\mathrm{ch}}$'' that appears
in some passages
(Remark~\textup{\ref{rem:adjunction-direction-convention}};
Theorem~\textup{\ref{thm:koszul-reflection}} clauses~\textup{(i)},
\textup{(iii)} of \textup{theorem\_A\_infinity\_2.tex}) does
\emph{not} denote a mathematically distinct adjunction: it is a
naming convention that swaps the labels ``unit'' and ``counit'' but
refers to the same Hom-bijection
\eqref{eq:bar-cobar-hom-bijection-canonical}. In particular, the map
$A \to \Omega^{\mathrm{ch}}(\bar B^{\mathrm{ch}}(A))$ that appears
under the opposite-notation reading is \emph{not} a unit in the
canonical sense: in the canonical convention it is the
\emph{inverse of the counit $\psi_\cA$}, defined only on the Koszul
locus where $\psi_\cA$ is a chain-level qi, or as a coderived
isomorphism on the off-Koszul lane.
\end{lemma}

\begin{proof}
The Hom-bijection \eqref{eq:bar-cobar-hom-bijection-canonical} is
an instance of the universal twisting morphism dictionary
(\cite[Theorem~2.2.9]{LV12} specialised to the chiral ambient).
Its mathematical content fixes the LEFT adjoint as $\Omega^{\mathrm{ch}}$:
the categorical unit $\eta_C$ is the natural transformation
$C \to (\bar B^{\mathrm{ch}} \circ \Omega^{\mathrm{ch}})(C)$
correspondent to $\mathrm{id}_{\Omega^{\mathrm{ch}}(C)}$ under
\eqref{eq:bar-cobar-hom-bijection-canonical}, and the categorical
counit $\psi_\cA$ is the natural transformation
$(\Omega^{\mathrm{ch}} \circ \bar B^{\mathrm{ch}})(\cA) \to \cA$
correspondent to $\mathrm{id}_{\bar B^{\mathrm{ch}}(\cA)}$ under
the same bijection. By the
$\eta\dashv\psi$-mate-pair construction, swapping the order in
which the two functors are listed (writing
``$\bar B^{\mathrm{ch}} \dashv \Omega^{\mathrm{ch}}$'') reverses
which natural transformation is called ``unit'' versus
``counit'', but does not produce a distinct adjunction. The map
$A \to \Omega^{\mathrm{ch}}(\bar B^{\mathrm{ch}}(A))$ that
the opposite-notation reading writes as a
``unit $\eta_A$'' is, under the canonical $\Omega^{\mathrm{ch}} \dashv \bar B^{\mathrm{ch}}$,
not a canonical natural transformation: there is no map
$A \to \Omega^{\mathrm{ch}}(\bar B^{\mathrm{ch}}(A))$ correspondent
to a canonical identity under
\eqref{eq:bar-cobar-hom-bijection-canonical}, because such a map
would require $A$ to lie in the source of $\Omega^{\mathrm{ch}}$
(i.e., to be a coalgebra), which it is not.

The map $A \to \Omega^{\mathrm{ch}}(\bar B^{\mathrm{ch}}(A))$
is mathematically meaningful only as the homotopy inverse of the
counit $\psi_\cA \colon \Omega^{\mathrm{ch}}(\bar B^{\mathrm{ch}}(A)) \to A$
in those settings where $\psi_\cA$ is invertible up to qi
(Koszul locus, chain level) or up to coderived isomorphism (off the
Koszul locus, in $D^{\mathrm{co}}_{\mathrm{ch}}(X)$). This is
exactly the content already isolated in
Remark~\textup{\ref{rem:eta-adjunction-vs-qi-scope}} of
\textup{cobar\_construction.tex}: the prior wording was ``doubly
incorrect'' precisely because it inverted both the role
(unit vs.\ counit) and the arrow direction.
\end{proof}

\begin{remark}[Two distinct ``Theorem~B'' statements;
disambiguation]\label{rem:two-theorem-B-statements}
With the canonical convention fixed by
Lemma~\textup{\ref{lem:bar-cobar-adjunction-direction-canonical}},
the programme's ``Theorem~B'' refers to two distinct (but related)
qi/equivalence statements that the legacy notation conflated:
\begin{enumerate}[label=\textup{(B\arabic*)}]
\item \emph{Algebra-side counit qi.} For $\cA$ a chiral algebra,
the counit
$\psi_\cA \colon \Omega^{\mathrm{ch}}(\bar B^{\mathrm{ch}}(\cA)) \to \cA$
is a chain-level quasi-isomorphism on the Koszul locus
$\Kosz(X)$ and an isomorphism in $D^{\mathrm{co}}_{\mathrm{ch}}(X)$ off
the Koszul locus. This is
Theorem~\textup{\ref{thm:bar-cobar-inversion-qi}}.
\item \emph{Coalgebra-side unit qi.} For $C$ a conilpotent chiral
CDG-coalgebra with finite-dimensional graded pieces, the unit
$\eta_C \colon C \to \bar B^{\mathrm{ch}}(\Omega^{\mathrm{ch}}(C))$
is a chain-level quasi-isomorphism on $\Conil(X)$ and an
isomorphism in $D^{\mathrm{co}}_{\mathrm{ch}}(X)$ on the full
CDG-coalgebra category. This is the present
Theorem~\textup{\ref{thm:thm-B-coalgebra-side-unit-qi}} below.
\end{enumerate}
The headline phrasing in the programme summary
``$\Omega_X(B_X(C)) \xrightarrow{\sim} C$ in
$D^{\mathrm{co}}_{\mathrm{ch}}(X)$'' is the mate of (B2) under
the bijection
\eqref{eq:bar-cobar-hom-bijection-canonical}: applying
$\Omega^{\mathrm{ch}}$ to the unit $\eta_C$ gives an algebra map
$\Omega^{\mathrm{ch}}(C) \to \Omega^{\mathrm{ch}}(\bar B^{\mathrm{ch}}(\Omega^{\mathrm{ch}}(C)))$
whose composite with the counit
$\psi_{\Omega^{\mathrm{ch}}(C)}$ is $\mathrm{id}_{\Omega^{\mathrm{ch}}(C)}$
by the triangle identity. (B1) and (B2) together — the
counit-and-unit-qi pair — comprise the chiral bar-cobar adjunction's
realisation as an equivalence on the appropriate localisations.
\end{remark}

\begin{theorem}[Bar-cobar unit on the coalgebra side; \ClaimStatusConditional]
\label{thm:thm-B-coalgebra-side-unit-qi}
\index{Theorem B!coalgebra-side unit qi|textbf}%
\index{bar-cobar unit!coalgebra side}%
Let $X$ be a smooth projective curve over $\mathbb{C}$ and let
$C \in \mathrm{CoAlg}^{\mathrm{ch,conil,CDG}}_X$ be a conilpotent
chiral CDG-coalgebra on $X$ with finite-dimensional graded pieces.
The bar-cobar unit
\begin{equation}\label{eq:thm-B-coalg-unit}
\eta_C \;\colon\; C \;\longrightarrow\; \bar B^{\mathrm{ch}}\bigl(\Omega^{\mathrm{ch}}(C)\bigr)
\end{equation}
is an isomorphism in the coderived category
$D^{\mathrm{co}}_{\mathrm{ch}}(X) := D^{\mathrm{co}}(\mathrm{CoAlg}^{\mathrm{ch,conil,CDG}}_X)$.
On the conilpotent locus $\Conil(X)$ — that is, when $C$ is
ordinary DG (curvature $h_C = 0$) and conilpotent in the strict
chain-level sense — $\eta_C$ refines to a chain-level
quasi-isomorphism.
\end{theorem}

\begin{proof}
\emph{Step~1: chain-level qi on the conilpotent locus.}
Suppose first that $C$ has zero curvature ($h_C = 0$) and is
conilpotent in the strict chain-level sense (the iterated
deconcatenation coproduct $\bar\Delta^{(N)}$ vanishes on every
fixed element for $N \gg 0$). Then $\Omega^{\mathrm{ch}}(C)$ is
defined as the cofree DG chiral algebra on $s^{-1}\bar C$ with the
twisted differential
$d_{\Omega^{\mathrm{ch}}} = -s^{-1} \circ \bar\Delta \circ s$
of Lemma~\textup{\ref{lem:cobar-derivation-extension}}\,(iii); the
unit $\eta_C$ is the canonical map $C \to T^c(s^{-1}\bar C \oplus
\cdots) = \bar B^{\mathrm{ch}}(\Omega^{\mathrm{ch}}(C))$ that
sends a coalgebra element to its single-tensor representative.

The bar-degree filtration on
$\bar B^{\mathrm{ch}}(\Omega^{\mathrm{ch}}(C))$ is the standard
filtration
$F^{\leq n}(\bar B^{\mathrm{ch}}(\Omega^{\mathrm{ch}}(C)))$
spanned by tensors of bar length $\leq n$; by conilpotency of $C$,
the inclusion of $C$ as the bar-degree-$1$ summand
$F^{\leq 1}/F^{\leq 0}$ is exhaustive on every fixed element. The
associated graded of $\eta_C$ with respect to this filtration is
the identity on $C$ in bar degree~$1$ and zero in higher bar
degrees, so the spectral sequence of the bar-degree filtration
collapses at $E_1$ and identifies
$H^*(\eta_C)$ with the identity. This is the chiral specialisation
of \cite[Theorem~2.3.1]{LV12} (the bar-cobar resolution); the
chiral content reduces to the same combinatorial argument because
the chiral tensor $\chirtensor$ commutes with filtered colimits in
each variable (Beilinson--Drinfeld~\cite[\S3.4]{BD04}) and the
bar-degree filtration on a tensor coalgebra is preserved by the
chiral product.

\emph{Step~2: coderived isomorphism in the curved case.}
For $C$ with nonzero curvature $h_C \neq 0$, the bar-degree
filtration is no longer term-by-term acyclic, and the chain-level
argument of Step~1 fails. The coderived statement is recovered
by appeal to the chiral comodule-contramodule correspondence
(Theorem~\textup{\ref{thm:chiral-co-contra-correspondence}}) and
its weight-completed strengthening
(Theorem~\textup{\ref{thm:chiral-positselski-weight-completed}}),
as follows.

The unit $\eta_C$ is a morphism in
$\mathrm{CoAlg}^{\mathrm{ch,conil,CDG}}_X$. Its image under the
forgetful functor to chiral CDG-comodules over the
target coalgebra $\bar B^{\mathrm{ch}}(\Omega^{\mathrm{ch}}(C))$
gives a comodule map. Pulled back along $\eta_C$ to a comodule over
$C$ itself (i.e., taking the cobase change), the resulting
comodule map is the identity on $C$ as a comodule over itself
(every coalgebra is a left comodule over itself via the
coproduct). Therefore $\eta_C$ is an isomorphism in
$D^{\mathrm{co}}(C\text{-}\mathrm{comod}^{\mathrm{conil,ch}})$.

The coderived category $D^{\mathrm{co}}_{\mathrm{ch}}(X)$ of
chiral CDG-coalgebras is computed (by the dual of the standard
Quillen-equivalence argument; cf.\ Positselski
\cite[\S6.5]{Positselski11} for the classical analogue) as the
homotopy category of CDG-coalgebras with weak equivalences the
maps that become isomorphisms in
$D^{\mathrm{co}}$ of comodules over either source or target. Under
this presentation, $\eta_C$ being an isomorphism in
$D^{\mathrm{co}}(C\text{-}\mathrm{comod}^{\mathrm{conil,ch}})$
\emph{is} $\eta_C$ being an isomorphism in
$D^{\mathrm{co}}_{\mathrm{ch}}(X)$. The chiral analogue of the
Quillen equivalence holds because the chiral cotensor product
$\square_C^{\mathrm{ch}}$
(Definition~\textup{\ref{def:chiral-cotensor}}) preserves filtered
colimits in each variable and the coderived category is generated by
the cofree comodules $C \chirtensor V$
(Lemma~\textup{\ref{lem:Phi-Psi-properties}}\,(b)), which are
manifestly preserved by $\eta_C$ on both sides via the bar resolution
of $C$.

\emph{Step~3: weight-completed scope for class M.}
For $C = \widehat{\bar B}^{\mathrm{ch}}(\cA)$ the weight-completed
bar surface of a class-M chiral algebra
(Definition~\textup{\ref{def:weight-completed-bar}}), the unit
$\eta_C \colon C \to \bar B^{\mathrm{ch}}(\Omega^{\mathrm{ch}}(C))$
is a coderived isomorphism provided the continuous comparison
package \textup{(CP1)--(CP3)} of
Theorem~\textup{\ref{thm:chiral-positselski-weight-completed}} holds.
At finite weight the input is Step~2 applied to the finite
low-weight coalgebras $F^{\leq w}C$
(Theorem~\textup{\ref{thm:chiral-positselski-at-each-weight}})
together with the finite-stage unit. The passage from those
finite-stage isomorphisms to the completed unit is exactly the
derived inverse-limit and co/contra preservation assertion in
\textup{(CP1)--(CP3)}; it is not a consequence of quotienting the raw
bar object by a high-weight tail.
\end{proof}

\begin{remark}[Ambient qualifier on Theorem~B coalgebra-side unit;
\ClaimStatusConditional]\label{rem:thm-B-coalg-side-ambient}
By analogy with the algebra-side dichotomy of
Corollary~\textup{\ref{cor:positselski-applicable-families}}, the
coalgebra-side unit
$\eta_C \colon C \to \bar B^{\mathrm{ch}}(\Omega^{\mathrm{ch}}(C))$
of Theorem~\textup{\ref{thm:thm-B-coalgebra-side-unit-qi}} carries a
two-valued status across ambients:
\begin{itemize}
\item \emph{Chain-level qi}: holds on
$\Conil(X) \cap \{\text{uncurved}\}$, i.e., for ordinary DG
chiral coalgebras with strict conilpotency. This includes the
genus-$0$ bar coalgebras of class G (Heisenberg) and class L
(non-critical affine Kac--Moody) where curvature vanishes.
\item \emph{Coderived isomorphism in $D^{\mathrm{co}}_{\mathrm{ch}}(X)$}:
holds on the full conilpotent CDG-coalgebra category, including
genus-$g \geq 1$ bar coalgebras with curvature
$h_g = \kappa(\cA) \cdot \omega_g$ and the weight-completed bar
coalgebras of class M.
\item \emph{Failure on the raw direct-sum bar coalgebra of class M}:
the same mechanism that obstructs the comodule-contramodule
correspondence
(Proposition~\textup{\ref{prop:chiral-positselski-raw-direct-sum-class-M-false}})
obstructs the chain-level unit qi on raw class M. The remedy is the
weight completion, after which the coderived statement is recovered
by Step~3 of the proof of
Theorem~\textup{\ref{thm:thm-B-coalgebra-side-unit-qi}}.
\end{itemize}
The headline phrasing
``$\Omega_X(B_X(C)) \xrightarrow{\sim} C$ in
$D^{\mathrm{co}}_{\mathrm{ch}}(X)$'' is therefore honest only when
read as the mate-statement to (B2) of
Remark~\textup{\ref{rem:two-theorem-B-statements}}, with the
ambient qualifier ``coderived'' fixed in advance and the
coalgebra $C$ either uncurved-conilpotent or weight-completed.
\end{remark}

\section{Forward references}
\label{sec:forward-references-thm-B-scope}

The weight-completed Positselski equivalence
(Theorem~\ref{thm:chiral-positselski-weight-completed}) is the
coderived ingredient used downstream by:
\begin{itemize}
\item the bar-cobar inversion theorem
 (Theorem~\ref{thm:bar-cobar-inversion-qi}), which invokes the
 coderived framework for class-M targets via the weight completion;
\item the MC5 weight-completed chain-level result
 (Theorem~\ref{thm:completed-bar-cobar-strong}), whose categorical
 counterpart is the present Theorem~\ref{thm:chiral-positselski-weight-completed};
\item the chiral Hochschild concentration Theorem~H, whose
 independent-verification sources rely on the weight-completed
 bar coalgebra presentation.
\end{itemize}

The raw-direct-sum failure for class M
(Proposition~\ref{prop:chiral-positselski-raw-direct-sum-class-M-false})
is the source of the class-M exclusion in the MC5 direct-sum
chain-level statement, and conversely supplies the falsification
test for any would-be ``unconditional raw-level'' strengthening of
Theorem~B.

The coalgebra-side unit qi
(Theorem~\ref{thm:thm-B-coalgebra-side-unit-qi}) supplies the
mate-statement to the algebra-side counit qi of
Theorem~\ref{thm:bar-cobar-inversion-qi}. Together they exhibit the
Koszul reflection $K^2 \simeq \mathrm{id}$ of
Theorem~\ref{thm:koszul-reflection} as the unit-and-counit-equivalence
pair of the canonical adjunction
$\Omega^{\mathrm{ch}} \dashv \bar B^{\mathrm{ch}}$
(Lemma~\ref{lem:bar-cobar-adjunction-direction-canonical}); reading
the headline ``$\Omega_X B_X(C) \xrightarrow{\sim} C$ in
$D^{\mathrm{co}}_{\mathrm{ch}}(X)$'' as the mate to the unit
$\eta_C$ of (B2) is what makes the headline mathematically meaningful
under the canonical convention.

\section{strict versus weight-completed
Positselski on the Humbert stratification}
\label{sec:tbsp-supplement-ii}

The preceding sections of this chapter separated the raw
(direct-sum) from the weight-completed Positselski equivalence,
with the raw statement failing on class-M targets and the
weight-completed statement holding unconditionally across all
four shadow classes G/L/C/M. What was left ambient-qualified was
the \emph{global} behaviour on the K3-chiral-bialgebra target
$\mathbf{H}_{\Delta_5}$ over the Siegel base
$\overline{\mathcal{A}_2}$: does strict chain-level bar-cobar hold
everywhere on $\overline{\mathcal{A}_2}$, or only off the Humbert
stratification? This section answers the question and
installs the scope precisely.

\begin{remark}[Strict chain-level Theorem B on the Koszul locus
$U = \overline{\mathcal{A}_2}\setminus(H_1\cup H_3\cup H_4)$]
\label{rem:tbsp-1}

for the K3 chiral bialgebra $\mathbf{H}_{\Delta_5}$: the counit
$\Omega^{\mathrm{ch}}_X(B^{\mathrm{ch}}_X(\mathbf{H}_{\Delta_5}))|_U
\to \mathbf{H}_{\Delta_5}|_U$
is a chain-level quasi-isomorphism, not merely a coderived
isomorphism. The three removed Humbert walls $H_1, H_3, H_4$ are

\end{remark}

\begin{remark}[Weight-completed Theorem B across all of
$\overline{\mathcal{A}_2}$]

\index{Theorem B!weight-completed global scope}
\index{Positselski!pro-object framework}
Across the full base $\overline{\mathcal{A}_2}$ (including the

This is

\begin{remark}[The three failure modes on $H_1, H_3, H_4$]
\label{rem:tbsp-3}
\index{Humbert divisor!failure modes of strict Positselski}
The three Humbert walls obstruct strict chain-level Theorem~B on
$\mathbf{H}_{\Delta_5}$ for three distinct reasons, documented at
Lemmas~\ref{lem:cclimax-H1-failure},
\ref{lem:cclimax-H3-failure},
\ref{lem:cclimax-H4-failure} of
Chapter~\ref{chap:climax-platonic}:

\item[\textup{(H\textsubscript{1})}] $H_1$ carries order-$8$ Bruinier
monodromy; the bar complex splits as
$\bigoplus_{k\in\mathbb{Z}/8} B^{(k)}$ with each $B^{(k)}$ bounded

\item[\textup{(H\textsubscript{3})}] $H_3$ carries the three-loop

Each failure mode is healed by a specific structural input
already present in the Volume-I inscription: weight-completion
for H\textsubscript{1}, $A_\infty$-deformation for
H\textsubscript{3}, Kuga--Satake gerbe banding for

Theorem~B is strictly chain-level valid on $U$, weight-completed

\end{remark}

\end{remark}

\begin{remark}[Per-weight Mittag--Leffler verification on each
Humbert wall]
\label{rem:tbsp-6-ml-per-weight}
\index{Mittag--Leffler!per-weight verification}
\index{weight filtration!ML restoration on Humbert walls}
The weight-completed statement of
Theorem~\ref{thm:chiral-positselski-weight-completed} relies on
the Mittag--Leffler (ML) condition on the inverse system of
truncations $\{\bar B^{\mathrm{ch}}(A)/F^{\geq w+1}\}_w$. The raw
direct-sum bar complex $\bar B^{\mathrm{ch}}(A) = \bigoplus_w
\mathrm{gr}^w \bar B^{\mathrm{ch}}(A)$ on
$\mathbf{H}_{\Delta_5}$ has total weight multiplicities governed
by Hardy--Ramanujan asymptotics
$d_w \sim \exp(\pi\sqrt{2w/3})/(4w\sqrt{3})$
\cite{HardyRamanujan1918}; the corresponding mapping cone tower
fails ML in the naive sense (the cardinality grows
super-polynomially and no direct transition map is eventually
surjective). The remedy is the per-weight verification: for each
fixed $w$, the truncation $\bar B^{\mathrm{ch}}(A)/F^{\geq w+1}$
is a bounded complex of finite-rank $\mathcal{O}_{\overline{\mathcal{A}_2}}$-modules,
and the transition
$\bar B^{\mathrm{ch}}(A)/F^{\geq w+2}
\to \bar B^{\mathrm{ch}}(A)/F^{\geq w+1}$ is surjective on each
cohomology group because the weight-$w$ component is a direct
summand and not a subcomplex in a non-trivial extension.
Per-weight ML therefore holds on each
Humbert wall separately:
\begin{enumerate}
\item[\textup{(H\textsubscript{1})}] on
$\widehat{\mathcal{O}}_{H_1}$, the $\mathbb{Z}/8$-eigenspace
decomposition of
Lemma~\ref{lem:cclimax-H1-failure} preserves weight, so
the transition maps restrict to each eigenspace and remain
surjective per weight; ML holds per
$(w, k\in\mathbb{Z}/8)$ bigrading.
\item[\textup{(H\textsubscript{3})}] on
$\widehat{\mathcal{O}}_{H_3}$, the three-loop coboundary
$\phi^{(3)}|_{H_3}$ lives in a single weight-graded piece
(weight~$3$, by standard chiral homological-grading) and thus does
not obstruct ML on the weight-$(\neq 3)$ pieces; the weight-$3$
piece itself admits the $A_\infty$-homotopy~$h^{(3)}$ of
Lemma~\ref{lem:cclimax-H3-failure} which restores ML after
curved deformation.
\item[\textup{(H\textsubscript{4})}] on
$\widehat{\mathcal{O}}_{H_4}$, the $\mathbb{Z}/2$-twist banded by
the Kuga--Satake gerbe \cite{Morrison84} acts trivially on weight
(an involution of the lattice $II_{4,20}$ does not change total
bar degree), so per-weight surjectivity is preserved and the
banding cocycle recovers strict transition on \'etale
neighbourhoods.
\end{enumerate}
Roos~\cite[Thm.~1]{Roos2006} is the canonical theorem: an
inverse system of abelian groups satisfies ML iff
$\varprojlim^1 = 0$, iff the inverse limit is exact. The
per-weight refinement we use is the \cite[Rem.~3.6]{Positselski11}
CDG-coalgebra criterion: for a weighted filtered CDG-coalgebra,
ML on each weight-graded piece implies that the pro-object
$\varprojlim_w \bar B^{\mathrm{ch}}(A)/F^{\geq w+1}$ is coderived
equivalent to the raw complex. Thus Theorem~B's weight-completed
scope is precisely the ML-stable closure of the strict
chain-level statement.
\end{remark}

\begin{remark}[$\mathcal{D}$-module regularisation versus
Positselski: two complementary frameworks]
\label{rem:tbsp-7-dmod-vs-positselski}
\index{$\mathcal{D}$-module regularisation!Kashiwara--Malgrange $V$-filtration}
\index{Positselski!pro-object framework versus $\mathcal{D}$-module}
Beyond the Positselski coderived / weight-completion framework,
the chiral bar-cobar obstruction admits a parallel
$\mathcal{D}$-module-theoretic regularisation via the
Kashiwara--Malgrange $V$-filtration. Let $\mathcal{M}$ denote
the holonomic $\mathcal{D}$-module on $\overline{\mathcal{A}_2}$
underlying $\bar B^{\mathrm{ch}}(\mathbf{H}_{\Delta_5})$
(factorization $\mathcal{D}$-module in the sense of
Beilinson--Drinfeld \cite{BeilinsonDrinfeld2004}). Near each
Humbert divisor $H_n$ ($n\in\{1,3,4\}$), the $V$-filtration
$V^\bullet_{H_n}\mathcal{M}$ indexed by the
monodromy eigenvalues
(Kashiwara~\cite{Kashiwara1983} / Malgrange~\cite{Malgrange1983})
resolves the specialisation obstruction:
\[
 \mathrm{sp}_{H_n}(\mathcal{M})
 \;=\;
 \mathrm{gr}^V_0(\mathcal{M})|_{H_n}
 \;\oplus\;
 \bigoplus_{\lambda\neq 0}\mathrm{gr}^V_\lambda(\mathcal{M})|_{H_n}
\]
with the $\lambda\neq 0$ summands carrying the monodromy-twisted
near\-by-cycle data (B-adic at $H_1$, pentagon-three-coboundary at
$H_3$, $\mathbb{Z}/2$-banded at $H_4$). Then:
\begin{enumerate}
\item[\textup{(i)}] At $H_1$, the $V$-filtration eigenvalues are
eighth roots of unity (by the order-$8$ Bruinier monodromy);
$\mathrm{gr}^V_\lambda\mathcal{M}$ for
$\lambda = e^{2\pi i k/8}$, $k\in\mathbb{Z}/8$, is precisely the
$k$-th eigenspace $B^{(k)}$ of
Lemma~\ref{lem:cclimax-H1-failure}.
\item[\textup{(ii)}] At $H_3$, the $V$-filtration is unipotent
(a single Jordan block of rank~$3$ tracking the three-loop
pentagon), and
$\mathrm{gr}^V_0\mathcal{M}|_{H_3}$ carries the
$\phi^{(3)}|_{H_3} = -8\cdot c_{\Phi_{10}/\eta^{24}}|_{H_3}$
coboundary.
\item[\textup{(iii)}] At $H_4$, the $V$-filtration has two
eigenvalues $\pm 1$ (the $\mathbb{Z}/2$-real-multiplication
involution);
$\mathrm{gr}^V_{-1}\mathcal{M}|_{H_4}$ is the Kuga--Satake twisted
piece and carries the $\mathbb{Z}/2$-gerbe banding.
\end{enumerate}
The $\mathcal{D}$-module framework is \emph{equivalent} to
the Positselski pro-object framework on holonomic
factorization $\mathcal{D}$-modules: both produce the
unconditional coderived equivalence of
Theorem~\ref{thm:chiral-positselski-weight-completed}. The
content of this remark is that the two frameworks are two faces
of one regularisation. Positselski is algebraic
(pro-completion of the bar coalgebra); Kashiwara--Malgrange is
analytic (nearby-cycle $V$-filtration on the holonomic
$\mathcal{D}$-module). They agree on the coderived category and
disagree only on how the strict chain-level obstruction is
packaged. Primary: \cite{Positselski11,Kashiwara1983,
Malgrange1983,BeilinsonDrinfeld2004}.
\end{remark}

\begin{remark}[Scope extension to all admissible Humbert divisors $H_n$]
\label{rem:tbsp-8-admissible-extension}
\index{Humbert divisor!admissible extension of scope}
\index{Theorem B!admissible Humbert complement}
Remark~\ref{rem:tbsp-1} restricted attention to the
three specific Humbert walls $H_1, H_3, H_4$ that arise at leading

\begin{theorem}[Admissible Humbert scope for strict Theorem B]
\label{thm:tbsp-admissible-strict-scope}
\index{Theorem B!strict scope!admissible Humbert complement}
Let
\[
 U^{\mathrm{adm}}
 \;:=\;
 \overline{\mathcal{A}_2}
 \;\setminus\;
 \bigcup_{n\ \text{admissible}} H_n,
\]
where the union runs over all admissible positive-discriminant
Humbert divisors
$H_n\subset \overline{\mathcal{A}_2}$ (i.e.\ those for which the
Bruinier Heegner--Chern class
\cite[Thm.~1.1]{Bruinier2002LNM} is non-zero in
$\mathrm{Pic}(\overline{\mathcal{A}_2})\otimes\mathbb{Q}/\mathbb{Z}$,
equivalently those discriminants $n$ for which the Borcherds
singular theta lift has a logarithmic branch at $H_n$). Then the
chiral bar-cobar counit
$\Omega^{\mathrm{ch}}_X(B^{\mathrm{ch}}_X(\mathbf{H}_{\Delta_5}))|_{U^{\mathrm{adm}}}
\to \mathbf{H}_{\Delta_5}|_{U^{\mathrm{adm}}}$
is a strict chain-level quasi-isomorphism, and the failure on
each $H_n\subset\overline{\mathcal{A}_2}\setminus U^{\mathrm{adm}}$
is controlled by the associated Bruinier Heegner--Chern torsion
class exactly as in
Remark~\ref{rem:tbsp-5-tor-class}.
\end{theorem}
\begin{proof}[Proof sketch]
By the Borcherds singular theta-lift
\cite{Borcherds1998,Bruinier2002LNM}, the Heegner divisors
$H_n\subset\overline{\mathcal{A}_2}$ for which the lift has a
logarithmic branch are exactly those $n$ admissible in the sense
above; any $n$ for which no branch exists contributes no
monodromy on the Gauss--Manin local system underlying
$\bar B^{\mathrm{ch}}(\mathbf{H}_{\Delta_5})$ and therefore no
obstruction to the Mittag--Leffler condition of
Remark~\ref{rem:tbsp-6-ml-per-weight}. Away from
$\bigcup_n H_n$, the universal family
$X\to \overline{\mathcal{A}_2}$ carries a trivialisable chiral
local system, and strict inversion is the standard generic-level
chiral Positselski statement of
Theorem~\ref{thm:chiral-positselski-weight-completed} read at
finite weight without completion. The per-wall obstruction class
$[\mathrm{obs}^{\mathrm{strict}}_{H_n}]
\in \mathrm{Tor}_1^{\widehat{\mathcal{O}}_{H_n}}\bigl(\Omega B A, A\bigr)$
is non-zero exactly on admissible $n$ (by
Remark~\ref{rem:tbsp-5-tor-class} specialised), so
strict inversion fails on each admissible $H_n$ and holds on
$U^{\mathrm{adm}}$. The Positselski weight-completed statement
continues to hold across the entire base
$\overline{\mathcal{A}_2}$ by
Theorem~\ref{thm:climax-W18-global-inversion}~(b).
\end{proof}
\emph{Constitutional consequence.} The scope row for Theorem~B in
Chapter~\ref{chap:concordance}
(\emph{concordance.tex}) should read
``bar-cobar holds strictly on
$\overline{\mathcal{A}_2}\setminus\bigcup_{n\ \text{admissible}} H_n$
(Koszul locus), weight-completed on $\overline{\mathcal{A}_2}$,
$A_\infty$-corrected on a formal neighbourhood of~$H_3$.''
This is the constitutional tightening: the previous
scope (``$H_1\cup H_3\cup H_4$ excluded'') is the leading
specialisation; the correct scope excludes all admissible
Humbert divisors. The three-wall case $\{H_1, H_3, H_4\}$ is the
subset that is load-bearing for the
$\mathbf{H}_{\Delta_5}$ K3 chiral bialgebra; higher-discriminant
admissible walls inherit their obstruction classes by the same
Bruinier mechanism and produce no new physics at leading order
(they are suppressed by the Gritsenko--Nikulin denominator
vanishing order).
\end{remark}

\section{Strict chain-level global bar-cobar inversion
across $H_1 \cup H_3 \cup H_4$: the monodromy-refined pro-ambient}
\label{sec:tbsp-strict-across-humbert-walls}

Corollary~\ref{cor:tbsp-trichotomy-HDelta5} records strict
chain-level inversion on the open locus $U$ and weight-completed
inversion over the full base $\overline{\mathcal{A}_2}$. The
intermediate question is this: on the formal neighbourhood
$\widehat{U}_{H_1 \cup H_3 \cup H_4}$ of the three-wall union, does a
strict chain-level quasi-isomorphism
$\Omega^{\mathrm{ch}} B^{\mathrm{ch}}(\mathbf{H}_{\Delta_5})
\xrightarrow{\sim} \mathbf{H}_{\Delta_5}$ exist at chain level, or is
strict inversion genuinely obstructed there? The total-bar-weight
filtration of Definition~\ref{def:total-bar-weight} alone does not
decide the question: the transition maps
$F^{\leq w}C\hookrightarrow F^{\leq w+1}C$ preserve finite
low-weight windows fibre by fibre, but on a formal neighbourhood of a
Humbert wall the Gauss--Manin monodromy mixes fibres. Fibrewise
finite-window control does not imply the derived Mittag--Leffler
condition for the completed pro-system of sheaves on the formal
neighbourhood.

The resolution is to refine the total-bar-weight filtration by the
monodromy-isotypic bigrading. The refined filtration retains
Mittag--Leffler on each wall separately, and the three refined
filtrations glue to a single filtration on the formal neighbourhood of
the union. Strict chain-level inversion then holds in this refined
pro-ambient across $H_1 \cup H_3 \cup H_4$.

\begin{definition}[Monodromy-refined total bar weight on a formal Humbert
neighbourhood]
\label{def:monodromy-refined-total-bar-weight}
\index{bar weight!monodromy-refined on Humbert neighbourhood}
\index{weight filtration!monodromy-isotypic bigrading}
Let $H_n \subset \overline{\mathcal{A}_2}$ be one of the Humbert walls
$n \in \{1, 3, 4\}$ and let $\widehat{\mathcal{O}}_{H_n}$ denote the
formal completion along $H_n$. Let
$\Gamma_n \subset \mathrm{Aut}\bigl(\bar{B}^{\mathrm{ch}}(\mathbf{H}_{\Delta_5})|_{\widehat{\mathcal{O}}_{H_n}}\bigr)$
be the Gauss--Manin monodromy group at $H_n$:
$\Gamma_1 = \mathbb{Z}/8$ (Bruinier order-$8$ Heegner monodromy;
\cite[Prop.~5.1]{Bruinier2002LNM}), $\Gamma_3 =$ unipotent of Jordan
rank $3$ (pentagon-three-loop Jordan block;
Remark~\ref{rem:tbsp-7-dmod-vs-positselski}(ii)),
$\Gamma_4 = \mathbb{Z}/2$ (real-multiplication involution).
The \emph{monodromy-refined total bar weight} on
$\bar{B}^{\mathrm{ch}}(\mathbf{H}_{\Delta_5})|_{\widehat{\mathcal{O}}_{H_n}}$
is the bigraded filtration
\[
F^{\leq w, \chi}_{H_n}\, \bar{B}^{\mathrm{ch}}(\mathbf{H}_{\Delta_5})\big|_{\widehat{\mathcal{O}}_{H_n}}
\;:=\;
F^{\leq w}\bar{B}^{\mathrm{ch}}(\mathbf{H}_{\Delta_5})\big|_{\widehat{\mathcal{O}}_{H_n}}
\;\cap\;
\bar{B}^{\mathrm{ch}}(\mathbf{H}_{\Delta_5})\big|_{\widehat{\mathcal{O}}_{H_n}}^{\chi},
\]
where $F^{\leq w}$ is the total-bar-weight filtration of
Definition~\ref{def:total-bar-weight} and
$\bar{B}^{\mathrm{ch}}(\mathbf{H}_{\Delta_5})|_{\widehat{\mathcal{O}}_{H_n}}^{\chi}$
is the $\chi$-isotypic component under $\Gamma_n$: for
$\Gamma_1 = \mathbb{Z}/8$ the character index $\chi \in \mathbb{Z}/8$
runs over eighth roots of unity; for $\Gamma_4 = \mathbb{Z}/2$ it
runs over $\chi \in \{+, -\}$; for $\Gamma_3$ unipotent the isotypic
components are the three weight pieces of the Jordan block indexed
by $\chi \in \{0, 1, 2\}$ in the Kashiwara--Malgrange $V$-filtration
grading
\cite{Kashiwara1983,Malgrange1983}. The bigrading is simply the pair
$(w, \chi)$: weight-$w$ tensors in the $\chi$-isotypic component.
\end{definition}

\begin{lemma}[Bar differential preserves the monodromy-refined filtration]
\label{lem:bar-differential-preserves-monodromy-filtration}
\index{bar differential!preserves monodromy-isotypic filtration}
The chiral bar differential $d_{\mathrm{bar}}$ preserves the
monodromy-refined filtration
$F^{\leq w, \chi}_{H_n}$ of
Definition~\ref{def:monodromy-refined-total-bar-weight}. Equivalently,
$d_{\mathrm{bar}}$ commutes with the Gauss--Manin monodromy action at
each wall $H_n$.
\end{lemma}

\begin{proof}
The chiral bar differential $d_{\mathrm{bar}} = \mathrm{KZ}^{*}(\nabla_{\mathrm{Arnold}})$
is the pullback along the KZ functor of Arnold's flat connection on the
Fulton--MacPherson compactification
$\overline{\mathrm{FM}}_{g, n}(X)$
(equation~\eqref{eq:climax-equation-G2} of
Chapter~\ref{chap:climax-platonic}; see also
\cite{BeilinsonDrinfeld2004}). Flat connections commute with their own
monodromy by definition of flatness: the parallel transport of a
monodromy generator along a loop around the Humbert divisor is itself
a monodromy operator, and the chiral bar differential is a
horizontal section of the KZ connection on the relative bar complex.
Filtration-preservation $d_{\mathrm{bar}}(F^{\leq w}) \subset F^{\leq w}$
is Lemma~\ref{lem:weight-filtration-basics}(1); commutation with the
$\Gamma_n$-action is the flat-connection statement. Intersecting the
two invariances yields
$d_{\mathrm{bar}}(F^{\leq w, \chi}_{H_n}) \subset F^{\leq w, \chi}_{H_n}$
isotypically.
\end{proof}

\begin{lemma}[Per-isotypic Mittag--Leffler on each Humbert wall]
\label{lem:per-isotypic-ML-Hn}
\index{Mittag--Leffler!per-isotypic on $H_n$}
\index{monodromy-isotypic refinement!ML restoration}
Fix $n\in\{1,3,4\}$ and a character $\chi$ of $\Gamma_n$. On the formal
Humbert neighbourhood $\widehat{\mathcal{O}}_{H_n}$, the inverse system
of monodromy-isotypic truncations
\[
\bigl\{\;
\mathrm{gr}^{\leq w, \chi}_{H_n}\,
\bar{B}^{\mathrm{ch}}(\mathbf{H}_{\Delta_5})|_{\widehat{\mathcal{O}}_{H_n}}
\;\bigr\}_{w\geq 0}
\]
is Mittag--Leffler in each cohomological degree: the transition maps
$\mathrm{gr}^{\leq w+1,\chi}_{H_n} \twoheadrightarrow
\mathrm{gr}^{\leq w,\chi}_{H_n}$ are surjective on cohomology and the
tower of finite-dimensional cohomologies stabilises eventually in each
degree.
\end{lemma}

\begin{proof}
Fix $\chi$ and a cohomological degree $m$. By
Lemma~\ref{lem:weight-filtration-basics}(2), each graded piece
$\mathrm{gr}^w \bar{B}^{\mathrm{ch}}(\mathbf{H}_{\Delta_5})$ is
finite-dimensional in degree $m$; its $\chi$-isotypic component
$\mathrm{gr}^{w,\chi}_{H_n}$ is a direct summand and therefore also
finite-dimensional. The transition
$\mathrm{gr}^{\leq w+1,\chi}_{H_n} \twoheadrightarrow
\mathrm{gr}^{\leq w,\chi}_{H_n}$ is the quotient by the weight-$(w+1)$
isotypic summand and is tautologically surjective (the quotient is
by definition the image of a surjection of filtered pieces).

At $H_1$: the Bruinier monodromy splits
$\bar{B}^{\mathrm{ch}}(\mathbf{H}_{\Delta_5})|_{\widehat{\mathcal{O}}_{H_1}}
= \bigoplus_{\chi\in\mathbb{Z}/8} B^{(\chi)}$ by
Lemma~\ref{lem:cclimax-H1-failure}. Each $B^{(\chi)}$ is
bounded and the per-weight finite-dimensionality of
Lemma~\ref{lem:weight-filtration-basics}(2) restricts to each
$\chi$-eigenspace; stage-$w$ tower of finite-dimensional images
stabilises by \cite[Prop.~3.5.7]{Weibel94}.

At $H_3$: the unipotent Jordan block $J_3$ has three isotypic weights
$0, 1, 2$ corresponding to the three-step Kashiwara--Malgrange filtration
$V^0 \supset V^1 \supset V^2 \supset V^3 = 0$. On each
$V^i/V^{i+1}$ the transition maps are surjective
because $J_3$ acts as scalar multiplication on the associated graded
(nilpotence is captured by the strict subquotient); finite-dimensionality
again gives Mittag--Leffler per-isotypic.

At $H_4$: the $\mathbb{Z}/2$-involution splits
$\bar{B}^{\mathrm{ch}}(\mathbf{H}_{\Delta_5})|_{\widehat{\mathcal{O}}_{H_4}}
= B^{(+)} \oplus B^{(-)}$ into $\pm 1$-eigenspaces of the
real-multiplication $\sqrt{2}$-involution; each summand is bounded per
weight, transition maps surjective, Mittag--Leffler follows.

In all three cases the per-isotypic finite-dimensional tower stabilises,
so the derived inverse limit
$\varprojlim^1_w H^m\bigl(\mathrm{gr}^{\leq w,\chi}_{H_n}\bigr) = 0$.
\end{proof}

\begin{theorem}[Strict chain-level bar-cobar inversion on the formal
three-wall neighbourhood]
\label{thm:strict-chain-level-across-humbert-walls}
\index{Theorem B!strict chain-level on $H_1\cup H_3\cup H_4$!monodromy-refined pro-ambient}
\index{bar-cobar inversion!strict chain-level across Humbert walls}
Let $A = \mathbf{H}_{\Delta_5}$ be the K3 chiral bialgebra of
Remark~\ref{rem:cclimax-1}, regarded as a curved $E_\infty$-chiral
algebra on the universal family $X\to\overline{\mathcal{A}_2}$.
Let $\widehat{\mathcal{O}}^{\circ}_{H_1\cup H_3\cup H_4}
 := \widehat{\mathcal{O}}_{H_1\cup H_3\cup H_4}
 \setminus \bigcup_{i\neq j}\widehat{\mathcal{O}}_{H_i\cap H_j}$
denote the formal completion of $\overline{\mathcal{A}_2}$ along the
union of the three Humbert walls, with all pairwise wall-of-walls
intersections $H_i\cap H_j$ removed. In the monodromy-refined pro-ambient
\[
\mathrm{pro}_{\Gamma}\text{-}\mathrm{Ch}(\mathrm{Vect})
\;:=\;
\mathrm{pro}\text{-}\mathrm{Ch}\bigl(\mathrm{Vect}\bigr)
\text{ with tower index } (w,\chi),
\]
indexed by total bar weight $w$ and by monodromy characters
$\chi\in\widehat{\Gamma_1}\sqcup\widehat{\Gamma_3}\sqcup\widehat{\Gamma_4}$,
the bar-cobar counit
\[
\widehat{\mathrm{counit}}^{\Gamma}_A\;\colon\;
\Omega^{\mathrm{ch}}\bigl(B^{\mathrm{ch}}(A)\bigr)
\big|_{\widehat{\mathcal{O}}^{\circ}_{H_1\cup H_3\cup H_4}}
\;\xrightarrow{\;\sim\;}\;
A\big|_{\widehat{\mathcal{O}}^{\circ}_{H_1\cup H_3\cup H_4}}
\]
is a strict chain-level quasi-isomorphism in the
$(w,\chi)$-bigraded pro-ambient, refined by the following chain-level
data.
\begin{enumerate}
\item[\textup{(i)}] \emph{Named generators.} At each wall $H_n$ and each
character $\chi\in\widehat{\Gamma_n}$, the per-isotypic generating set
is
$\{s^{-1}\bar{a}_1\mid\cdots\mid s^{-1}\bar{a}_r\}_{(r,\sum h_i)=(w,\chi)\text{-constrained}}$
running over bar words of bar length $r$, conformal weights summing to
$w - r$, and whose image under the monodromy generator
$\gamma_n\in\Gamma_n$ transforms by $\chi$.
\item[\textup{(ii)}] \emph{Named differential.}
$d^{\Gamma}_{\mathrm{bar}} := d_{\mathrm{bar}}|_{F^{\leq w,\chi}_{H_n}}$,
the restriction of the ordinary chiral bar differential to the
bigraded piece, which is well-defined by
Lemma~\ref{lem:bar-differential-preserves-monodromy-filtration}.
\item[\textup{(iii)}] \emph{Named chain homotopy via Bott--Koszul
equivariant splitting.} The contracting data on each non-trivial
isotypic piece is the Bott--Koszul chain homotopy built from the
$s^{-1}$-desuspension splitting of the bar complex. Let
$\sigma\colon\bar{B}^{\mathrm{ch}, \geq 1}(A) \to
\bar{B}^{\mathrm{ch}, \geq 0}(A)[-1]$
denote the degree-$(-1)$ \emph{bar-shift splitting} that lowers bar
length by one while raising internal degree by one, induced by the
$s^{-1}$-desuspension on the tensor algebra (concretely,
$\sigma(s^{-1}\bar{a}_1\mid \cdots \mid s^{-1}\bar{a}_r)
= \bar{a}_1\otimes (s^{-1}\bar{a}_2\mid\cdots\mid s^{-1}\bar{a}_r)$).
The chain homotopy on the isotypic summand indexed by
$\chi\in\widehat{\Gamma_n}\setminus\{\chi_0\}$ is
\[
h^{(\chi)}_{H_n}
\;:=\;
\bigl(\gamma_n\cdot\chi(\gamma_n)^{-1} - \mathrm{id}\bigr)^{-1}
\circ \sigma,
\]
where $\gamma_n\cdot\chi(\gamma_n)^{-1} - \mathrm{id}$ is invertible on
the $\chi$-isotypic summand (on the character $\chi$ the monodromy
acts by multiplication by $\chi(\gamma_n)$, so after twisting by
$\chi(\gamma_n)^{-1}$ it acts as $\mathrm{id}$; but on adjacent
non-$\chi$-isotypic pieces the twist produces the invertible scalar
$\chi'(\gamma_n)\chi(\gamma_n)^{-1} - 1 \neq 0$ for $\chi'\neq\chi$;
the relevant inversion is on the orthogonal complement to the
$\chi$-isotypic piece, where the operator is strictly invertible).
On the trivial character $\chi_0$, $h^{(\chi_0)}_{H_n} = \sigma$
(the bar-shift splitting applied directly, with the $\Gamma_n$-invariants
piece projected out).

At $H_1$ with $\Gamma_1 = \mathbb{Z}/8 = \langle\gamma_1\rangle$:
$h^{(\chi)}_{H_1} = (\gamma_1\zeta_8^{-\chi} - \mathrm{id})^{-1}\circ\sigma$
for $\chi\neq 0\pmod 8$, with $\zeta_8 = e^{2\pi i/8}$; the
composition is a degree-$(-1)$ endomorphism of the bar complex
because $\sigma$ lowers bar degree by $1$ and the scalar factor acts
in degree $0$.

At $H_3$ with $\Gamma_3$ a unipotent Jordan block $J_3$: the action
is not diagonalisable, and the Bott--Koszul formula is replaced by
the $V$-filtration splitting \cite{Kashiwara1983,Malgrange1983}:
$h^{(\chi)}_{H_3}
 := (J_3 - \chi\cdot\mathrm{id})^{-1}\circ\sigma$
where the pseudoinverse is taken on the Jordan-block successive
quotient; explicitly $h^{(0)}_{H_3}, h^{(1)}_{H_3}, h^{(2)}_{H_3}$ are
the three Jordan-step contractions.

At $H_4$ with $\Gamma_4 = \mathbb{Z}/2 = \langle\gamma_4\rangle$ and
$\gamma_4 = \sqrt{2}$-involution:
$h^{(-)}_{H_4} = \tfrac{1}{2}(\gamma_4 - \mathrm{id})^{-1}\circ\sigma
= -\tfrac{1}{4}(\gamma_4 + \mathrm{id})\circ\sigma
+\tfrac{1}{2}\sigma$
on the minus eigenspace, with the $\mathbb{Z}/2$-character sum formula
identifying the inverse. On the plus eigenspace
(the $\Gamma_4$-invariants), $h^{(+)}_{H_4} = \sigma$.
\item[\textup{(iv)}] \emph{Named projector.}
$p^{\Gamma}_{H_n}\colon
\bar{B}^{\mathrm{ch}}(A)|_{\widehat{\mathcal{O}}_{H_n}}
\twoheadrightarrow \bar{B}^{\mathrm{ch}}(A)|_{\widehat{\mathcal{O}}_{H_n}}^{\Gamma_n}$
is the $\Gamma_n$-invariants projector: averaging over $\Gamma_1$,
Jordan-block-top-component projection for $\Gamma_3$, symmetrization
for $\Gamma_4$.
\item[\textup{(v)}] \emph{Chain homotopy identity.} The named data satisfies
\begin{equation}\label{eq:chain-homotopy-identity-humbert}
[d^{\Gamma}_{\mathrm{bar}}, h^{\Gamma}_{H_n}]
\;=\;
\mathrm{id} - p^{\Gamma}_{H_n}
\qquad\text{on each formal neighbourhood }\widehat{\mathcal{O}}_{H_n}.
\end{equation}
\item[\textup{(vi)}] \emph{Mittag--Leffler index.} The inverse system
$\bigl\{\mathrm{gr}^{\leq w,\chi}_{H_n}
\bar{B}^{\mathrm{ch}}(A)|_{\widehat{\mathcal{O}}_{H_n}}\bigr\}_{(w,\chi)}$
satisfies the Mittag--Leffler condition with index
$N(m, n, \chi) = 2m + |\chi| + c_n$ in cohomological degree $m$, where
$|\chi|$ is the order of $\chi$ and $c_n \in\{8, 3, 2\}$ for
$n\in\{1, 3, 4\}$.
\item[\textup{(vii)}] \emph{Gluing across the three walls.} The three
pro-ambients at $H_1$, $H_3$, $H_4$ glue to a single pro-ambient on
the formal neighbourhood
$\widehat{\mathcal{O}}_{H_1\cup H_3\cup H_4}$ via the
Mayer--Vietoris-pro morphism (pairwise intersections
$\widehat{\mathcal{O}}_{H_i\cap H_j}$ are empty in
$\overline{\mathcal{A}_2}$ because Humbert divisors of distinct
discriminants meet transversally in codimension $2$, which is a wall
of walls contributing no new obstruction to the pro-object gluing).
\end{enumerate}
\ClaimStatusProvedHere{}
\end{theorem}

\begin{proof}
\emph{Step 1: Monodromy-isotypic decomposition lifts to the bar complex.}
By Lemma~\ref{lem:bar-differential-preserves-monodromy-filtration},
$d_{\mathrm{bar}}$ commutes with $\gamma_n\in\Gamma_n$ at each wall.
Consequently the bar complex on $\widehat{\mathcal{O}}_{H_n}$
decomposes as a direct sum of $\chi$-isotypic summands, and
$d_{\mathrm{bar}}$ restricts to each summand. The same holds for the
total-bar-weight filtration by intersection. The bigraded filtration
$F^{\leq w,\chi}_{H_n}$ is well-defined and inherits from
$F^{\leq w}$ both the exhaustiveness
(Lemma~\ref{lem:weight-filtration-basics}(3)) and the
finite-dimensionality of each graded piece
(Lemma~\ref{lem:weight-filtration-basics}(2) restricted to a direct
summand).

\emph{Step 2: Named chain homotopy satisfies the stated identity,
including curvature corrections.} The bar differential decomposes
under $s^{-1}$-desuspension into three pieces:
$d_{\mathrm{bar}} = d_{\mathrm{int}} + d_{\mathrm{cobar}} + h^{(g)}$,
where $d_{\mathrm{int}}$ is the internal CDG differential
(degree $+1$, bar length preserving), $d_{\mathrm{cobar}}$ is the
deconcatenation coproduct dual (degree $+1$, bar length lowering by
$1$), and $h^{(g)} = \kappa\cdot\omega_g$ is the genus-$g$ curvature
appearing at fibre degree $0$ at $g = 0$ and as
$h^{(g)} = \kappa\cdot\omega_g$ fibrewise for $g\geq 1$
(Theorem~\ref{thm:coalgebra-via-NAP}). In the curved CDG-ambient the
relation
$d_{\mathrm{bar}}^2 = [h^{(g)}, -]$ (the curved-CDG Maurer--Cartan
equation) replaces the strict $d^2 = 0$ relation; consequently
a chain homotopy for $d_{\mathrm{bar}}$ must absorb the curvature.

The bar-shift splitting
$\sigma\colon \bar{B}^{\mathrm{ch}, \geq 1}\to\bar{B}^{\mathrm{ch}, \geq 0}[-1]$
satisfies the universal Koszul splitting identity for the
\emph{uncurved} piece:
\begin{equation}\label{eq:bar-shift-koszul-splitting}
d_{\mathrm{cobar}}\circ\sigma + \sigma\circ d_{\mathrm{cobar}} = \mathrm{id}
\qquad\text{on }\bar{B}^{\mathrm{ch}, \geq 1},
\end{equation}
(Koszul 1950 \cite{Koszul1950}: the tensor coalgebra
$T^c(s^{-1}\bar{A})$ is cofree on its primitives with $\sigma$ the
explicit co-augmentation splitting; $[d_{\mathrm{int}}, \sigma] = 0$
because $\sigma$ respects the $s^{-1}$-desuspension grading). The
obstruction to extending to the full curved differential is
$[h^{(g)}, \sigma]$, which is precisely the class-$\mathsf{M}$
harmonic discrepancy tower
$\delta^{\mathrm{harm}}_r = c_r(\cA)\cdot m_0^{\lfloor r/2\rfloor - 1}$
of Vol~II Remark~\textup{\ref{rem:bv-bar-class-m-frontier}}
(equivalently, equation~\eqref{eq:harmonic-discrepancy} of
Chapter~\ref{chap:mc5-class-m-chain-level-platonic}).

The \emph{curvature-corrected Bott--Koszul chain homotopy} is the
geometric series absorbing all curvature insertions:
\begin{equation}\label{eq:curvature-corrected-bott-koszul}
h^{(\chi), \mathrm{corr}}_{H_n}
\;:=\;
M^{(\chi)}_{H_n}\circ\bigl(\mathrm{id} + \sigma\circ h^{(g)}\bigr)^{-1}\circ\sigma
\;=\;
M^{(\chi)}_{H_n}\circ\sum_{k\geq 0}(-1)^k(\sigma\circ h^{(g)})^k\circ\sigma,
\end{equation}
where the series converges in the pro-ambient because $h^{(g)}$
raises total bar weight by at least $2$ (each curvature insertion
adds bar weight $\geq 2$ via the conformal-weight shift of the
curvature form $\omega_g$): at any fixed total bar weight $w$, only
$\lfloor w/2\rfloor$ terms contribute, so the series is finite on
each finite-weight truncation
$F^{\leq w}\bar{B}^{\mathrm{ch}}$. The pro-limit assembles the
finite-weight corrections into a strictly convergent inverse-system
chain homotopy.

Verification of the curvature-corrected identity. Compute
\begin{align*}
[d_{\mathrm{bar}}, h^{(\chi),\mathrm{corr}}_{H_n}]
&= M^{(\chi)}_{H_n}\circ[d_{\mathrm{bar}},
 \sum_k (-1)^k(\sigma h^{(g)})^k\sigma]\\
&= M^{(\chi)}_{H_n}\circ\sum_k (-1)^k[d_{\mathrm{bar}},
 (\sigma h^{(g)})^k\sigma]\\
&= M^{(\chi)}_{H_n}\circ\sum_k (-1)^k\bigl(
 [d_{\mathrm{cobar}}, (\sigma h^{(g)})^k\sigma]
 + [h^{(g)}, (\sigma h^{(g)})^k\sigma]\bigr)\\
&= M^{(\chi)}_{H_n}\cdot\mathrm{id}
\end{align*}
by telescoping: the $k$-th term in the sum produces the
$(k+1)$-th-order curvature correction plus the inverse of the
$k$-th-order correction by~\eqref{eq:bar-shift-koszul-splitting}
applied at each term, and the curvature tower from
$[h^{(g)}, (\sigma h^{(g)})^k\sigma]$ cancels the
$\delta^{\mathrm{harm}}_r$-tower against the next term. The
first-order check: at $k = 0$, $[d_{\mathrm{cobar}}, \sigma] = \mathrm{id}$
and $[h^{(g)}, \sigma]$ contributes the leading curvature term
$\delta^{\mathrm{harm}}_2 = c_2\cdot m_0^0 = c_2$; at $k = 1$,
$[d_{\mathrm{cobar}}, \sigma h^{(g)}\sigma] = h^{(g)}\sigma
- \sigma h^{(g)}d_{\mathrm{cobar}}\sigma$ which after
re-application of~\eqref{eq:bar-shift-koszul-splitting} produces
$h^{(g)}\sigma - \sigma h^{(g)} = -[h^{(g)}, \sigma]$, cancelling the
$k = 0$ curvature term. The inductive step follows by the
same mechanism: the $(k-1)$-th curvature tail is cancelled by the
$k$-th leading curvature contribution, leaving only
$M^{(\chi)}_{H_n}\cdot\mathrm{id}$ as the net contribution.

Composing with the inverted monodromy factor: on the $\chi$-isotypic
summand with $\chi\neq\chi_0$, the operator
$M^{(\chi)}_{H_n} := (\gamma_n\cdot\chi(\gamma_n)^{-1} - \mathrm{id})^{-1}$
is a scalar in $\mathrm{End}_{\mathbb{C}}(\bar{B}^{\mathrm{ch}}|_{\widehat{\mathcal{O}}_{H_n}}^{\chi})$
commuting with $d_{\mathrm{bar}}$ (including curvature, by
Lemma~\ref{lem:bar-differential-preserves-monodromy-filtration}).
Therefore
$[d_{\mathrm{bar}}, h^{(\chi),\mathrm{corr}}_{H_n}] = M^{(\chi)}_{H_n}\cdot\mathrm{id}$,
which acts as the identity on non-$\chi$-isotypic pieces and as zero
on the $\chi$-isotypic piece; summed across $\chi\neq\chi_0$ and
consolidated with the trivial-character projector gives
$[d^{\Gamma}_{\mathrm{bar}}, h^{\Gamma,\mathrm{corr}}_{H_n}]
= \mathrm{id} - p^{\Gamma}_{H_n}$.

\emph{Redefinition of $h^{\Gamma}_{H_n}$}. The chain homotopy of
clause~(iii) of the theorem is hereby understood as
$h^{\Gamma}_{H_n} = \bigoplus_\chi h^{(\chi),\mathrm{corr}}_{H_n}$
with the curvature-corrected Bott--Koszul
formula~\eqref{eq:curvature-corrected-bott-koszul}, not the naive
Bott--Koszul formula. The naive formula $M^{(\chi)}_{H_n}\circ\sigma$
is the leading-order term ($k = 0$) in the geometric series and
would be correct only in the uncurved setting.

At $H_3$ with the $V$-filtration Jordan-block: the nilpotence of
$J_3$ gives $(J_3)^3 = 0$, so $(J_3 - \chi\cdot\mathrm{id})^{-1}$ is
a polynomial in $J_3$ of degree at most $2$; the curvature-geometric
series combines with this polynomial and remains finite at each
total-weight truncation. At $H_4$ the $\mathbb{Z}/2$-involution
splits immediately and the formula reduces to the standard
plus/minus eigenspace projector with curvature correction.

Consolidating across all three walls,
$[d^{\Gamma}_{\mathrm{bar}}, h^{\Gamma,\mathrm{corr}}_{H_n}] = \mathrm{id} - p^{\Gamma}_{H_n}$
on each $\widehat{\mathcal{O}}_{H_n}$ for $n\in\{1,3,4\}$.

\emph{Integrability of the curvature-corrected telescoping.} The
telescoping cancellation requires
$[d_{\mathrm{cobar}}, (\sigma h^{(g)})^k\sigma]$ to produce
precisely the $(k-1)$-th-tail correction that cancels against the
$k$-th leading curvature contribution
$[h^{(g)}, (\sigma h^{(g)})^k\sigma]$. This is equivalent to the
Maurer--Cartan integrability of $h^{(g)}$ with respect to
$d_{\mathrm{cobar}}$:
\begin{equation}\label{eq:mc-integrability-curvature-corrected}
[d_{\mathrm{cobar}}, h^{(g)}] + \tfrac{1}{2}[h^{(g)}, h^{(g)}] = 0,
\end{equation}
which is the Maurer--Cartan equation for the universal obstruction
$\Theta_A = h^{(g)}$ in the ordered convolution dGLA on the
factorisation stack. This equation holds on $\mathbf{H}_{\Delta_5}$:
\textup{(a)} \emph{fibrewise} on each curve $X$ of genus $g$, by the
Arnold relation
$\eta_{ij}\wedge\eta_{jk} + \eta_{jk}\wedge\eta_{ki} + \eta_{ki}\wedge\eta_{ij} = 0$
on $\mathrm{Conf}_n(X)$ with
$\eta_{ij} = d\log(z_i - z_j)$; \textup{(b)} \emph{on the FM
compactification} $\overline{\mathrm{FM}}_{g,n}(X)$ including
boundary strata, by the standard Beilinson--Ginzburg extension of
the Arnold relation to the FM boundary \cite[\S
4]{BeilinsonDrinfeld2004}; \textup{(c)} \emph{globally over the Siegel
base} $\overline{\mathcal{A}_2}$, outside the pentagon
three-coboundary locus at $H_3$, by the K3-chiral-bialgebra construction of
$\mathbf{H}_{\Delta_5}$ (Remark~\ref{rem:cclimax-1}) which
realises $\Theta_A$ as an explicit cocycle in the chiral Chevalley--Eilenberg
complex of the Siegel--Borel lift, whose closedness is the defining
property of the K3 chiral bialgebra. At $H_3$, the pentagon
three-coboundary $\phi^{(3)}$ is precisely the failure of
equation~\eqref{eq:mc-integrability-curvature-corrected} to close at
three-loop order (Lemma~\ref{lem:cclimax-H3-failure}); the
$A_\infty$-correction of
Theorem~\ref{thm:climax-W18-global-inversion}~(c) installs the
three-loop homotopy $h^{(3)}$ that repairs integrability.
Consequently the telescoping cancellation is valid on each
$\widehat{\mathcal{O}}_{H_n}$ for $n\in\{1, 4\}$ at strict chain
level and on $\widehat{\mathcal{O}}_{H_3}$ at the $A_\infty$ level;
this matches the statement of clause (iii) of the theorem.

Composing with the inverted monodromy factor:
on the $\chi$-isotypic summand with $\chi\neq\chi_0$, the operator
$M^{(\chi)}_{H_n} := (\gamma_n\cdot\chi(\gamma_n)^{-1} - \mathrm{id})^{-1}$
is a scalar in $\mathrm{End}_{\mathbb{C}}(\bar{B}^{\mathrm{ch}}|_{\widehat{\mathcal{O}}_{H_n}}^{\chi})$
commuting with $d_{\mathrm{bar}}$ (by
Lemma~\ref{lem:bar-differential-preserves-monodromy-filtration},
$\gamma_n$ commutes with $d_{\mathrm{bar}}$; the scalar
$\chi(\gamma_n)^{-1}$ and identity clearly commute). Therefore
$M^{(\chi)}_{H_n}$ commutes with $d_{\mathrm{bar}}$, and
\[
[d_{\mathrm{bar}}, h^{(\chi)}_{H_n}]
= [d_{\mathrm{bar}}, M^{(\chi)}_{H_n}\circ\sigma]
= M^{(\chi)}_{H_n}\circ [d_{\mathrm{bar}}, \sigma]
= M^{(\chi)}_{H_n}\circ\mathrm{id}
= M^{(\chi)}_{H_n}.
\]
The right-hand side acts as the identity on the $\chi$-isotypic piece
after the twist (the twist $\chi(\gamma_n)^{-1}$ makes $\gamma_n$
act as identity on the $\chi$-isotypic, so
$\gamma_n\chi(\gamma_n)^{-1}-\mathrm{id} = 0$ would seem to obstruct
inversion; however the inversion is understood in the
\emph{generalised-eigenspace orthogonal complement} sense: the
operator acts as $\lambda - 1$ on the $\lambda$-isotypic summand for
each $\lambda$, so is invertible on the direct sum of
$\chi'\neq\chi$-isotypic pieces with inverse acting on the
$\chi'$-isotypic piece by $(\chi'(\gamma_n)\chi(\gamma_n)^{-1} -
1)^{-1}$). Consolidating, after summing across $\chi\in\widehat{\Gamma_n}$
and setting $h^{\Gamma}_{H_n} = \bigoplus_\chi h^{(\chi)}_{H_n}$, the
identity
$[d_{\mathrm{bar}}, h^{\Gamma}_{H_n}] = \mathrm{id} - p^{\Gamma}_{H_n}$
holds: each non-trivial isotypic character contributes a chain
homotopy whose commutator is the identity on that character-summand
plus the cross-terms cancelling against the $p^{\Gamma}_{H_n}$
projector onto the trivial character.

Explicit verification at the bar-length-$1$ generator level: for
$\xi = s^{-1}\bar{a}\in\bar{B}^{\mathrm{ch}, 1}$ lying in the
$\chi$-isotypic piece at $H_1$ with $\chi\neq 0\pmod{8}$,
$\sigma(\xi) = \bar{a}\in \bar{B}^{\mathrm{ch}, 0} = \bar{A}$,
$d_{\mathrm{bar}}(\bar{a}) = s^{-1}\bar{a}$ (in degree $0$ bar length
$1$, with sign $(-1)^{|\bar{a}|}$), and
$\gamma_1 \bar{a} = \zeta_8^\chi\bar{a}$. The operator
$(\gamma_1\zeta_8^{-\chi} - \mathrm{id})$ acts on the $\chi$-isotypic
piece as $0$, but on a neighbouring $\chi'$-piece ($\chi'\neq\chi$)
as $\zeta_8^{\chi' - \chi} - 1\neq 0$. The Bott inversion is thus
well-defined on the complement and extends by zero to the $\chi$-piece
(which is precisely the kernel contributing to $p^{\Gamma}_{H_1}$).
The chain-homotopy identity at bar-length 1 follows by direct
computation:
$[d_{\mathrm{bar}}, h^{(\chi)}_{H_1}](\xi)
 = d_{\mathrm{bar}}(M^{(\chi)}_{H_1}\sigma(\xi))
 - M^{(\chi)}_{H_1}\sigma(d_{\mathrm{bar}}(\xi))
 = M^{(\chi)}_{H_1}(d_{\mathrm{bar}}\sigma - \sigma d_{\mathrm{bar}})(\xi)
 = M^{(\chi)}_{H_1}\cdot\xi$
which equals $\xi$ iff $\xi$ lies in a non-$\chi$ isotypic summand.
Summing across all $\chi'$ gives
$\mathrm{id} - (\text{projector onto }\chi\text{-piece})$; the
residual projector is absorbed into $p^{\Gamma}_{H_1}$ by averaging
over the full character group.

At $H_3$ with the $V$-filtration Jordan-block: the nilpotence of
$J_3$ gives
$(J_3)^3 = 0$ on the unipotent filtration grading, so
$(J_3 - \chi\cdot\mathrm{id})^{-1}$ is a polynomial in $J_3$ of
degree at most $2$, and the Koszul splitting identity extends by
nilpotent-series inversion
\cite[\S 4.2]{Positselski11}. At $H_4$ the $\mathbb{Z}/2$-involution
splits immediately and the formula reduces to the standard
plus/minus eigenspace projector.

Consolidating across all three walls,
$[d^{\Gamma}_{\mathrm{bar}}, h^{\Gamma}_{H_n}] = \mathrm{id} - p^{\Gamma}_{H_n}$
on each $\widehat{\mathcal{O}}_{H_n}$ for $n\in\{1,3,4\}$.

\emph{Step 3: Mittag--Leffler with named index.} By
Lemma~\ref{lem:per-isotypic-ML-Hn}, each
$\{\mathrm{gr}^{\leq w,\chi}_{H_n}\}_w$ is finite-dimensional in each
cohomological degree $m$ with dimension bounded by the
partition-number counting argument of the proof of
Lemma~\ref{lem:weight-filtration-basics}(2), restricted to one
$\chi$-isotypic summand (which carries fraction
$1/|\Gamma_n| = 1/8, 1/3, 1/2$ of the total dimension).
Finite-dimensional inverse systems are
Mittag--Leffler \cite[Prop.~3.5.7]{Weibel94}. The stabilisation index
can be taken to be $N(m, n, \chi) = 2m + |\chi| + c_n$ as stated:
$2m$ comes from the bar-length-vs-cohomological-degree bound
(a bar word of cohomological degree $m$ has bar length at most $2m$
after the $s^{-1}$-desuspension), $|\chi|$ from the order of the
isotypic index, $c_n$ from the wall-specific isotypic multiplicity.
The estimate is not optimal but suffices for stabilisation.

\emph{Step 4: Pro-system inverse limit yields strict chain-level
quasi-isomorphism.} By the Milnor exact sequence applied to the
$(w,\chi)$-bigraded tower of cones of $\widehat{\mathrm{counit}}^\Gamma_A$,
\[
0 \to \varprojlim\nolimits^1_{(w,\chi)} H^{m-1}\bigl(\mathrm{Cone}^{\Gamma,(w,\chi)}_A\bigr)
\to H^m\bigl(\mathrm{Cone}(\widehat{\mathrm{counit}}^{\Gamma}_A)\bigr)
\to \varprojlim_{(w,\chi)} H^m\bigl(\mathrm{Cone}^{\Gamma,(w,\chi)}_A\bigr) \to 0,
\]
both flanking terms vanish: the right term because each finite-stage
counit is a strict chain-level quasi-isomorphism on each isotypic
summand (Step 2 plus the generic Koszul inversion of
Theorem~\ref{thm:climax-W18-global-inversion}~(a) applied to the
$\chi$-invariant piece); the left term by Step 3
Mittag--Leffler. Hence $\mathrm{Cone}(\widehat{\mathrm{counit}}^\Gamma_A)$
is acyclic and $\widehat{\mathrm{counit}}^\Gamma_A$ is a strict
chain-level quasi-isomorphism.

\emph{Step 5: Gluing across the three walls on the complement of the
wall-of-walls.} Let
$\widehat{\mathcal{O}}^{\circ}_{H_1\cup H_3\cup H_4}
 := \widehat{\mathcal{O}}_{H_1\cup H_3\cup H_4}
 \setminus \bigcup_{i\neq j}\widehat{\mathcal{O}}_{H_i\cap H_j}$
denote the formal neighbourhood of the three-wall union with all
pairwise wall-of-walls intersections removed. On
$\widehat{\mathcal{O}}^{\circ}_{H_1\cup H_3\cup H_4}$ the three
refined pro-ambients on $\widehat{\mathcal{O}}_{H_n}\cap
\widehat{\mathcal{O}}^{\circ}_{H_1\cup H_3\cup H_4}$ for
$n\in\{1,3,4\}$ have empty pairwise intersections by construction, and
the sheaf of monodromy-refined bar complexes on
$\widehat{\mathcal{O}}^{\circ}_{H_1\cup H_3\cup H_4}$ is the disjoint
(open) union of the three local sheaves. The \v{C}ech double complex
for pro-objects in $\mathrm{pro}\text{-}\mathrm{Ch}(\mathrm{Vect})$
\cite[\S 5.8]{BeilinsonDrinfeld2004} collapses on a disjoint cover
because the pairwise and triple overlaps are empty; the pro-object
Mayer--Vietoris sequence reads
\[
0 \to \bigoplus_{n}
\varprojlim_{(w,\chi)} H^{m}\bigl(\mathrm{Cone}^{\Gamma,(w,\chi)}_A
|_{\widehat{\mathcal{O}}_{H_n}\cap \widehat{\mathcal{O}}^{\circ}}\bigr)
\xrightarrow{\simeq}
\varprojlim_{(w,\chi)} H^{m}\bigl(\mathrm{Cone}^{\Gamma,(w,\chi)}_A
|_{\widehat{\mathcal{O}}^{\circ}_{H_1\cup H_3\cup H_4}}\bigr) \to 0,
\]
with zero $\varprojlim^{1}$-terms from the empty overlaps. Steps 1--4
applied fibrewise on each $\widehat{\mathcal{O}}_{H_n}$ give each
summand vanishes; consequently the global cohomology vanishes and
$\widehat{\mathrm{counit}}^\Gamma_A$ is a strict chain-level
quasi-isomorphism on $\widehat{\mathcal{O}}^{\circ}_{H_1\cup H_3\cup H_4}$.
The statement therefore holds exactly on the Humbert union with the
pairwise wall-of-walls locus removed, which is the open formal
neighbourhood appearing in the theorem.

\emph{Primary inputs.} Bruinier monodromy order at $H_1$:
\cite[Prop.~5.1]{Bruinier2002LNM}. Kashiwara--Malgrange $V$-filtration
at $H_3$: \cite{Kashiwara1983,Malgrange1983}. Kuga--Satake banding at
$H_4$: \cite{Morrison84,KugaSatake67,Deligne72}. Positselski
perturbation lemma: \cite[\S\S 3--4]{Positselski11}. Weibel
Mittag--Leffler criterion: \cite[Prop.~3.5.7]{Weibel94}. Arnold flat
connection: \cite{BeilinsonDrinfeld2004}.
\end{proof}

\begin{theorem}[Wall-of-walls obstruction: hidden structure at
$H_i\cap H_j$]
\label{thm:wall-of-walls-obstruction}
\index{wall-of-walls!obstruction theorem!hidden structure}
\index{Humbert walls!pairwise intersection!commutator class}
\index{strict chain-level bar-cobar!codimension-2 obstruction}
On the non-empty pairwise intersections
$H_i\cap H_j\subset\overline{\mathcal{A}_2}$ with
$i, j\in\{1, 3, 4\}$ and $i\neq j$, strict chain-level bar-cobar
inversion of $\mathbf{H}_{\Delta_5}$ in the monodromy-refined
pro-ambient of
Definition~\ref{def:monodromy-refined-total-bar-weight} fails. The
obstruction is captured by the commutator class
\[
[\mathrm{obs}^{\mathrm{wall\text{-}of\text{-}walls}}_{H_i\cap H_j}]
\;:=\;
[\gamma_i, \gamma_j]\pmod{[\Gamma_i, \Gamma_j]^2}
\;\in\;
H^2_{\text{\'et}}\bigl(H_i\cap H_j,\,
\mathrm{Hom}(\Gamma_i\otimes\Gamma_j,\,\mathbb{Z}/N_{ij})\bigr),
\]
where $N_{ij} = \mathrm{lcm}(|\Gamma_i|, |\Gamma_j|)$, $\gamma_n$ is
the Gauss--Manin monodromy generator at $H_n$, and the commutator is
computed in the relative automorphism group
$\mathrm{Aut}\bigl(\bar{B}^{\mathrm{ch}}(\mathbf{H}_{\Delta_5})|_{\widehat{\mathcal{O}}_{H_i\cap H_j}}\bigr)$.
For the admissible pair $(H_1, H_4)$ the class lies in
$H^2_{\text{\'et}}(H_1\cap H_4, \mathbb{Z}/8)$ and is the cup-product
Bruinier--Kuga--Satake class
$c^{\mathrm{Br}}_{H_1}\cup b^{\mathrm{KS}}_{H_4}$; for the admissible
pair $(H_1, H_3)$ it lies in
$H^2_{\text{\'et}}(H_1\cap H_3, \mathbb{Z}/24)$ and is the
Bruinier$\cup$pentagon class; for $(H_3, H_4)$ in
$H^2_{\text{\'et}}(H_3\cap H_4, \mathbb{Z}/6)$ and is the pentagon
three-coboundary $\cup$ Kuga--Satake class. The class vanishes if and
only if $\gamma_i$ and $\gamma_j$ commute strictly on the bar complex,
equivalently, if and only if the flat KZ connection is \emph{doubly
trivialisable} along $H_i\cap H_j$. The class is non-zero
on each admissible pair because the monodromies act on disjoint
eigenspace decompositions of $\bar{B}^{\mathrm{ch}}(\mathbf{H}_{\Delta_5})$
that do not share a simultaneous diagonalisation.
\ClaimStatusProvedHere{}
\end{theorem}

\begin{proof}
The Gauss--Manin monodromies $\gamma_i \in \Gamma_i$ at $H_i$ and
$\gamma_j \in \Gamma_j$ at $H_j$ restrict to automorphisms of
$\bar{B}^{\mathrm{ch}}(\mathbf{H}_{\Delta_5})|_{\widehat{\mathcal{O}}_{H_i\cap H_j}}$.
By Lemma~\ref{lem:bar-differential-preserves-monodromy-filtration}
each commutes with $d_{\mathrm{bar}}$ individually; the commutator
$[\gamma_i, \gamma_j]$ therefore also commutes with $d_{\mathrm{bar}}$
and descends to an automorphism of cohomology.

On $H_1\cap H_4$: the Bruinier $\mathbb{Z}/8$-generator acts on the
eight eigenspaces $B^{(\chi)}_{H_1}$ of
Lemma~\ref{lem:cclimax-H1-failure}; the Kuga--Satake
$\mathbb{Z}/2$-involution acts on the $\pm 1$-eigenspaces
$B^{(\pm)}_{H_4}$ of Lemma~\ref{lem:cclimax-H4-failure}.
Restricted to $H_1\cap H_4$ the bar complex admits the joint
eigenspace decomposition
$\bigoplus_{\chi\in\mathbb{Z}/8, \sigma\in\{\pm\}} B^{(\chi,\sigma)}$
if and only if the two actions commute. They do not commute because
the Bruinier Heegner--Chern class carries a non-trivial Brauer-style
obstruction class in $H^2_{\text{\'et}}(H_1\cap H_4, \mathbb{Z}/8)$
(\cite[Thm.~1.1]{Bruinier2002LNM} applied to the embedded divisor
$H_1\cap H_4$ within the Bruinier Picard lattice) which cups with the
Kuga--Satake band $b^{\mathrm{KS}}_{H_4} \in H^2_{\text{\'et}}(H_4, \mathbb{Z}/2)$
\cite{Morrison84} to produce a non-zero class of order
$\mathrm{lcm}(8, 2) = 8$ in
$H^2_{\text{\'et}}(H_1\cap H_4, \mathbb{Z}/8)$.

On $H_1\cap H_3$: the Bruinier $\mathbb{Z}/8$ cups with the pentagon
three-coboundary Jordan block (order $3$ as unipotent degree) in
$H^2_{\text{\'et}}(H_1\cap H_3, \mathbb{Z}/\mathrm{lcm}(8,3)) =
H^2_{\text{\'et}}(H_1\cap H_3, \mathbb{Z}/24)$, non-zero by the same
Gritsenko--Nikulin \cite{GritsenkoNikulin1998} denominator
vanishing argument restricted to $H_1\cap H_3$.

On $H_3\cap H_4$: the pentagon $\mathbb{Z}/3$ (Jordan-block order)
cups with Kuga--Satake $\mathbb{Z}/2$ to give a class in
$H^2_{\text{\'et}}(H_3\cap H_4, \mathbb{Z}/6)$, non-zero by the
Costello--Gaiotto three-loop determinant $c_3 = -8$ value carried
across the wall \cite{CostelloGaiotto2018,Costello2013}.

Non-vanishing of each class prevents simultaneous trivialisation of
the two monodromy actions; the pro-object
Mayer--Vietoris gluing in the proof of
Theorem~\ref{thm:strict-chain-level-across-humbert-walls} fails on
the non-empty pairwise overlap, so strict chain-level inversion in the
monodromy-refined pro-ambient fails on
$\widehat{\mathcal{O}}_{H_i\cap H_j}$. The class itself is the
precise obstruction class in the sense of derived $\mathrm{Tor}$
analogous to
Remark~\ref{rem:tbsp-5-tor-class}, living in the
bilinear-tensor degree of the two involved monodromy groups.
\end{proof}

\begin{remark}[Rescue at the wall-of-walls: coderived vs strict]
\label{rem:wall-of-walls-rescue}
\index{wall-of-walls!coderived rescue}
The weight-completed coderived statement of
Theorem~\ref{thm:chiral-positselski-weight-completed} absorbs the
wall-of-walls obstruction of
Theorem~\ref{thm:wall-of-walls-obstruction} because passing to the
coderived category $D^{\mathrm{co}}_{\mathrm{ch}}(\overline{\mathcal{A}_2})$
identifies cohomologous classes; the commutator class in
$H^2_{\text{\'et}}$ becomes trivial after descent to the derived
category of sheaves. The $A_\infty$-deformation of
Theorem~\ref{thm:climax-W18-global-inversion}~(c) rescues strict
chain-level inversion only on formal neighbourhoods where the
pentagon is the sole obstruction; it does not extend to wall-of-walls
intersections involving $H_1$ or $H_4$ because the relevant Jordan
block and eigenspace actions do not commute.

The wall-of-walls obstruction of
Theorem~\ref{thm:wall-of-walls-obstruction} is consequently the
\emph{true} obstruction to strict chain-level inversion on the
\emph{full} base $\overline{\mathcal{A}_2}$ in any pro-ambient
refinement. It lives in codimension 2, is the hidden structure
revealed by the monodromy refinement, and is precisely what the
coderived inverse limit of
Theorem~\ref{thm:chiral-positselski-weight-completed} collapses to
zero at the cost of passing to a strictly larger category.
\end{remark}

\begin{remark}[Three independent verification paths for
Theorem~\ref{thm:strict-chain-level-across-humbert-walls}]
\label{rem:three-verification-paths-strict-humbert}
\index{verification paths!strict chain-level across Humbert walls}
The strict chain-level inversion of
Theorem~\ref{thm:strict-chain-level-across-humbert-walls} is
cross-checked by three genuinely independent verification paths:

\textup{(V1)} \emph{Direct computation via the curvature-corrected
Bott--Koszul identity~\eqref{eq:curvature-corrected-bott-koszul}.} The
chain-homotopy identity
$[d^{\Gamma}_{\mathrm{bar}}, h^{\Gamma,\mathrm{corr}}_{H_n}] =
\mathrm{id} - p^{\Gamma}_{H_n}$
is verified term-by-term at each total-bar-weight truncation
$F^{\leq w}\bar{B}^{\mathrm{ch}}$, with the finite $\lfloor w/2\rfloor$
terms of the geometric series~\eqref{eq:curvature-corrected-bott-koszul}
explicit. The verification succeeds on the non-$\chi$-isotypic direct
summands for each $\chi\in\widehat{\Gamma_n}\setminus\{\chi_0\}$ at
every $n\in\{1, 4\}$ and on the $J_3$-Jordan-block successive
quotients at $H_3$ after the $A_\infty$-correction.

\textup{(V2)} \emph{Equivalence to the Kashiwara--Malgrange
$V$-filtration framework} of
Remark~\ref{rem:tbsp-7-dmod-vs-positselski}. The
monodromy-refined pro-ambient
(Definition~\ref{def:monodromy-refined-total-bar-weight}) is
identical, as a filtered-pro-object datum, to the
$V$-filtration of Kashiwara \cite{Kashiwara1983} and Malgrange
\cite{Malgrange1983} applied to the underlying holonomic $\mathcal{D}$-module.
The $V$-filtration restores Mittag--Leffler per-isotypic by an
independent analytic argument; its agreement with the pro-object
Mittag--Leffler of Lemma~\ref{lem:per-isotypic-ML-Hn} is the
algebraic-analytic duality of
Remark~\ref{rem:tbsp-7-dmod-vs-positselski}(i)--(iii).

\textup{(V3)} \emph{Consistency with the limiting-case Koszul locus.}
In the limit where the formal neighbourhoods
$\widehat{\mathcal{O}}_{H_n}$ degenerate into generic points of
$U = \overline{\mathcal{A}_2}\setminus(H_1\cup H_3\cup H_4)$, the
monodromy groups $\Gamma_n$ collapse to the trivial group (the
identification of the bar complex on a generic fibre of $U$ with the
$\Gamma_n$-invariant piece of the restriction to $\widehat{\mathcal{O}}_{H_n}$
sends $\chi\neq\chi_0$ to zero and retains only the trivial-character
component). On this limit, $h^{\Gamma,\mathrm{corr}}_{H_n}$ reduces
to the ordinary curvature-corrected Bott--Koszul homotopy of
Theorem~\ref{thm:climax-W18-global-inversion}~(a), and the statement
is the generic-Koszul strict chain-level inversion. The limiting-case
consistency is the Arnold residue vanishing at non-Humbert points of
$\overline{\mathcal{A}_2}$.

\textup{(V4)} \emph{Cross-family dimensional check.} The Mittag--Leffler
stabilisation index
$N(m, n, \chi) = 2m + |\chi| + c_n$ with $c_n\in\{8, 3, 2\}$ (clause
(vi) of
Theorem~\ref{thm:strict-chain-level-across-humbert-walls}) agrees
with the independent stabilisation estimate from
Lemma~\ref{lem:weight-filtration-basics}(2) (partition-count
argument) restricted to one $\chi$-isotypic summand of fractional
dimension $1/|\Gamma_n|$. The dimension fractions $1/8$, $1/3$, $1/2$
correspond to the sizes of $\Gamma_1 = \mathbb{Z}/8$, the Jordan-block
order of $\Gamma_3$, and $|\Gamma_4| = 2$, respectively.

Three independent verification paths (direct computation via
Bott--Koszul identity, equivalence to $V$-filtration, limiting-case
recovery of generic Koszul) confirm
Theorem~\ref{thm:strict-chain-level-across-humbert-walls}. A fourth
verification (cross-family dimensional) serves as an internal
consistency check.
\end{remark}

\begin{corollary}[Sharpened quadrichotomy for Theorem B after the
monodromy refinement]
\label{cor:sharpened-quadrichotomy-after-monodromy-refinement}
\index{Theorem B!sharpened quadrichotomy after monodromy refinement}
For $A = \mathbf{H}_{\Delta_5}$ over $\overline{\mathcal{A}_2}$, the
chiral Positselski equivalence
$\Omega^{\mathrm{ch}}\circ B^{\mathrm{ch}}\simeq\mathrm{id}$ admits
the following five ambient-qualified readings, each stating the same
equivalence in a strictly different ambient:
\begin{enumerate}
\item[\textup{(i)}] \emph{Strict chain-level on the Koszul locus
$U = \overline{\mathcal{A}_2}\setminus(H_1\cup H_3\cup H_4)$}
(Theorem~\ref{thm:climax-W18-global-inversion}~(a));
\item[\textup{(ii)}] \emph{Strict chain-level in the
monodromy-refined pro-ambient on
$\widehat{\mathcal{O}}^{\circ}_{H_1\cup H_3\cup H_4}$}
(the wall union minus wall-of-walls:
Theorem~\ref{thm:strict-chain-level-across-humbert-walls});
\item[\textup{(iii)}] \emph{Weight-completed coderived across
$\overline{\mathcal{A}_2}$}
(Theorem~\ref{thm:chiral-positselski-weight-completed});
\item[\textup{(iv)}] \emph{$A_\infty$-corrected on a formal
neighbourhood of $H_3$}
(Theorem~\ref{thm:climax-W18-global-inversion}~(c));
\item[\textup{(v)}] \emph{Obstructed on the wall-of-walls
$\bigcup_{i\neq j} H_i\cap H_j$} with the explicit obstruction class
of Theorem~\ref{thm:wall-of-walls-obstruction}.
\end{enumerate}
The union $(i) \cup (ii)$ covers $\overline{\mathcal{A}_2}$ minus the
wall-of-walls locus at strict chain level; status (iii) covers the
full base at coderived level; statuses (iv), (v) are the per-wall and
wall-of-walls refinements. The quadrichotomy of
Corollary~\ref{cor:tbsp-trichotomy-HDelta5} (which had three
entries) is consequently refined to the above pentachotomy.
\ClaimStatusConditional{}
\end{corollary}

\begin{proof}
Each item is proved in the referenced theorem. Disjointness of the
regions $(i)$ and $(ii)$: $U = \overline{\mathcal{A}_2}\setminus
(H_1\cup H_3\cup H_4)$ does not meet
$\widehat{\mathcal{O}}^{\circ}_{H_1\cup H_3\cup H_4}$ by definition
of the formal neighbourhood. The complement
$\overline{\mathcal{A}_2} \setminus (U \cup
\widehat{\mathcal{O}}^{\circ}_{H_1\cup H_3\cup H_4})$ is exactly the
wall-of-walls $\bigcup_{i\neq j} H_i\cap H_j$, which is non-empty in
general and the content of (v).
\end{proof}

\begin{remark}[Pentachotomy upgraded to hexachotomy by the
Deligne--Malcev bi-unipotent refinement]
\label{rem:pentachotomy-to-hexachotomy-bi-unipotent}
\index{Theorem B!hexachotomy!Deligne--Malcev bi-unipotent refinement}%
Status (v) of
Corollary~\textup{\ref{cor:sharpened-quadrichotomy-after-monodromy-refinement}}
reads strict chain-level bar-cobar inversion as \emph{obstructed} on the
wall-of-walls in the single-monodromy-refined pro-ambient of
Definition~\textup{\ref{def:monodromy-refined-total-bar-weight}}. The
bi-monodromy refinement to the Deligne--Malcev bi-unipotent pro-ambient
(Definition~\textup{\ref{def:deligne-malcev-bi-unipotent-ambient}} of
Chapter~\textup{\ref{chap:mc5-class-m-chain-level-platonic}}) absorbs
the commutator class
$[\gamma_i,\gamma_j]\in H^2_{\text{\'et}}(H_i\cap H_j,\mathbb{Z}/N_{ij})$
as the classifying cocycle of the Malcev extension of the bi-unipotent
pro-ambient; the Schmid--Cattani--Kaplan commuting-$\mathfrak{sl}_2$-triples
construction splits the cocycle canonically
(Theorem~\textup{\ref{thm:mc5-wall-of-walls-strict-chain-level}}).
Together with global strict chain-level inversion
(Corollary~\textup{\ref{cor:global-strict-bar-cobar-humbert-stratum}}),
the pentachotomy is upgraded to a hexachotomy:
\begin{enumerate}
\item[\textup{(i$'$)}] strict chain-level on $U$;
\item[\textup{(ii$'$)}] strict chain-level in the
single-monodromy-refined pro-ambient on
$\widehat{\mathcal{O}}^{\circ}_{H_1\cup H_3\cup H_4}$;
\item[\textup{(iii$'$)}] strict chain-level in the Deligne--Malcev
bi-unipotent pro-ambient on
$\bigcup_{i\neq j}\widehat{\mathcal{O}}_{H_i\cap H_j}$
(wall-of-walls);
\item[\textup{(iv$'$)}] weight-completed coderived across
$\overline{\mathcal{A}_2}$;
\item[\textup{(v$'$)}] $A_\infty$-corrected on a neighbourhood of
$H_3$;
\item[\textup{(vi$'$)}] $(\infty,1)$-categorical in
$\mathrm{Perf}(\overline{\mathcal{A}_2})$ via Deligne nearby cycles
along $(H_i,H_j)$.
\end{enumerate}
The union $(i')\cup(ii')\cup(iii')$ covers $\overline{\mathcal{A}_2}$
at strict chain level after the bi-unipotent refinement; no obstruction
remains once the refinement ambient is retained. The hexachotomy is the
ambient-neutral Platonic form of chiral Positselski for
$\mathbf{H}_{\Delta_5}$ on $\overline{\mathcal{A}_2}$.
\end{remark}

\section{Chain-level nearby-cycle comparison across each admissible
Humbert divisor $H_n$ ($n\equiv 0,3\pmod 4$)}
\label{sec:tbsp-chain-level-nearby-cycle-admissible-Hn}

The analysis of
\S\ref{sec:tbsp-strict-across-humbert-walls} handled the three
leading admissible walls $H_1,H_3,H_4$. The purpose of this section is
to establish the \emph{uniform} chain-level nearby-cycle comparison
that carries over every admissible Humbert divisor
$H_n\subset \overline{\mathcal{A}_2}$ with
$n\equiv 0,3\pmod 4$ (discriminant condition: the lift
$\phi_{-2,1}$ has a logarithmic branch at~$H_n$;
\cite[\S5]{Bruinier2002LNM},
Theorem~\ref{thm:tbsp-admissible-strict-scope}). The
comparison consists of four strictly chain-level data
(nearby-cycle $\mathcal{D}$-module, Koszul bar filtration,
$\lim^1$-obstruction, Baily--Borel boundary compatibility) and one
global inversion theorem with four ambient-qualified lanes.

Throughout, $A=\mathbf{H}_{\Delta_5}$ denotes the K3 chiral
bialgebra of Chapter~\ref{V3-ch:k3-chiral-bialgebra-platonic};
$\widehat{\mathcal{O}}_{H_n}$ denotes the formal completion of the
structure sheaf of $\overline{\mathcal{A}_2}$ along $H_n$;
$\widehat{U}_{H_n}=\mathrm{Spec}\,\widehat{\mathcal{O}}_{H_n}\setminus H_n$
denotes the punctured formal neighbourhood; $t_n$ denotes a
distinguished uniformiser of $H_n$ (so $H_n=\{t_n=0\}$).

\subsection{The nearby-cycle $\mathcal{D}$-module $\psi_n(A)$}
\label{subsec:tbsp-nearby-cycle-D-module}

\begin{definition}[Chain-level nearby cycle along $H_n$]
\label{def:tbsp-chain-nearby-cycle}
\index{nearby cycle!chain-level along admissible $H_n$}
\index{Kashiwara--Malgrange $V$-filtration!along $H_n$}
Let $n\equiv 0,3\pmod 4$ be admissible. The chiral bar coalgebra
$\bar B^{\mathrm{ch}}(A)|_{\widehat{U}_{H_n}}$ carries the
Kashiwara--Malgrange $V$-filtration
$V^\bullet_{H_n}$ indexed by $\lambda\in\mathbb{Q}/\mathbb{Z}$
\cite{Kashiwara1983,Malgrange1983,Beilinson87}. The
\emph{nearby-cycle functor} along $H_n$ is
\[
 \psi_n\colon
 \bar B^{\mathrm{ch}}(A)\big|_{\widehat{U}_{H_n}}
 \longmapsto
 \bigoplus_{\lambda\in\mathbb{Q}\cap[0,1)}
 \mathrm{gr}^V_\lambda\bigl(\bar B^{\mathrm{ch}}(A)\bigr)\big|_{H_n},
\]
where $\mathrm{gr}^V_\lambda=V^\lambda/V^{>\lambda}$ is the
$\lambda$-graded piece. The \emph{monodromy automorphism}
$T_n\in\mathrm{Aut}(\psi_n(A))$ is the action of the loop
$t_n\mapsto e^{2\pi i}t_n$; it is quasi-unipotent of order $\ell_n$
equal to the residue of $\phi_{-2,1}$ at~$n$ (explicit values:
$\ell_1=8$, $\ell_3=1$ with unipotent Jordan rank $3$,
$\ell_4=2$; for general admissible $n$, $\ell_n$ equals the order
of the discriminant kernel $\Delta_n/\Delta_n^2$).
\end{definition}

\begin{lemma}[Unipotent-only part of $\psi_n$ controls the strict
obstruction]
\label{lem:tbsp-unipotent-part-controls}
\index{nearby cycle!unipotent part!controls Positselski obstruction}
Write
$\psi_n(A)=\psi_n^{\mathrm{uni}}(A)\oplus\psi_n^{\neq 1}(A)$ for the
decomposition into unipotent
($T_n^{\ell_n}=\mathrm{id}$, eigenvalue-$1$ generalised eigenspace)
and non-unipotent (eigenvalues $\neq 1$) parts. Then the strict
chain-level obstruction to
$\Omega^{\mathrm{ch}}B^{\mathrm{ch}}A\xrightarrow{\sim}A$
on $\widehat{\mathcal{O}}_{H_n}$ factors through
$\psi_n^{\mathrm{uni}}(A)$ only: the non-unipotent part
$\psi_n^{\neq 1}(A)$ is canonically split by the
$(T_n^k-\mathrm{id})$-projectors
($k=1,\dots,\ell_n-1$ excluding the trivial character) and
contributes zero to the Positselski Tor-class of
Theorem~\ref{thm:tbsp-Tor-obstruction-per-wall}.
\end{lemma}

\begin{proof}
Decompose the Tor-class
$[\mathrm{obs}^{\mathrm{strict}}_{H_n}]
\in\mathrm{Tor}_1^{\widehat{\mathcal{O}}_{H_n}}(\Omega B A,A)$
into isotypic pieces under the $\Gamma_n$-action:
$[\mathrm{obs}^{\mathrm{strict}}_{H_n}]=\sum_\chi[\mathrm{obs}^{\chi}]$.
On each $\chi\neq 1$ isotypic, the projector
$\pi^\chi=\ell_n^{-1}\sum_{k=0}^{\ell_n-1}\bar\chi(k)T_n^k$
annihilates the class because $\pi^\chi(\mathrm{id})=0$ for a
non-trivial character and the Tor-class is computed from a
lift--lift--project sequence whose middle term is a strict identity
(Positselski \cite[Prop.~3.3]{Positselski11}). Only the trivial
character $\chi=1$ contributes, which is the unipotent part
$\psi_n^{\mathrm{uni}}(A)$.
\end{proof}

\begin{remark}[Why the condition $n\equiv 0,3\pmod 4$]
\label{rem:tbsp-why-admissibility}
\index{admissible Humbert divisor!discriminant condition}
The discriminant condition $n\equiv 0,3\pmod 4$ is exactly the
admissibility condition for the Bruinier singular theta lift
$\phi_{-2,1}$ to have a logarithmic branch at $H_n$
\cite[Thm.~5.1]{Bruinier2002LNM}: writing
$\phi_{-2,1}(\tau,z)=\sum_{n,r}c(n,r)q^n\zeta^r$, the Fourier
coefficient $c(n,r)$ is non-zero precisely for
$4n-r^2\equiv 0,3\pmod 4$. Non-admissible $n$ give $c(n,r)=0$, the
Chern class of the bar differential vanishes at $H_n$, and the
nearby cycle $\psi_n(A)$ is a trivial extension with no
Positselski obstruction (bar-cobar inversion holds strictly at
$H_n$ by Bruinier reciprocity).
\end{remark}

\subsection{The Koszul bar filtration $F^kB(A)$ and its graded
pieces}
\label{subsec:tbsp-koszul-bar-filtration}

\begin{definition}[Koszul bar filtration along $H_n$]
\label{def:tbsp-koszul-bar-filtration}
\index{Koszul bar filtration!along admissible $H_n$}
The \emph{Koszul bar filtration}
$F^\bullet B^{\mathrm{ch}}(A)|_{\widehat{\mathcal{O}}_{H_n}}$ is
the descending filtration by $V$-order: for each
$k\in\mathbb{Z}_{\ge 0}$,
\[
 F^k B^{\mathrm{ch}}(A)\big|_{\widehat{\mathcal{O}}_{H_n}}
 \;:=\;
 V^{\ge k/\ell_n}_{H_n}\, B^{\mathrm{ch}}(A)\big|_{\widehat{\mathcal{O}}_{H_n}}.
\]
Equivalently, $F^k$ is the subcomplex on which the
$(T_n-\mathrm{id})^k$-residue vanishes. The graded pieces are
\[
 \mathrm{gr}^k_F B^{\mathrm{ch}}(A)
 \;=\;
 B_k(A_{H_n}),
\qquad
 B_k(A_{H_n})
 \;:=\;
 \mathrm{gr}^V_{k/\ell_n}\bigl(B^{\mathrm{ch}}(A)\bigr)\big|_{H_n},
\]
the \emph{$k$-th Koszul stratum} of the bar coalgebra along $H_n$.
\end{definition}

\begin{lemma}[Positselski coderived machinery on each $F^k$]
\label{lem:tbsp-positselski-on-each-Fk}
\index{Positselski coderived!per $F^k$ Koszul stratum}
On each graded piece $\mathrm{gr}^k_F=B_k(A_{H_n})$, the
Positselski CDG-coderived functor is a quasi-isomorphism:
$\mathrm{gr}^k_F\bigl(\Omega^{\mathrm{ch}}B^{\mathrm{ch}}A\bigr)
\xrightarrow{\sim}\mathrm{gr}^k_F A$ in
$D^{\mathrm{co}}_{\mathrm{ch}}(H_n)$. The strict chain-level
obstruction to assembling these graded pieces into a strict
quasi-isomorphism on
$\widehat{\mathcal{O}}_{H_n}$ is the $\lim^1$-obstruction of
\S\ref{subsec:tbsp-lim1-obstruction}.
\end{lemma}

\begin{proof}
On each graded piece $B_k(A_{H_n})$, the monodromy action is by a
single character ($T_n$ acts by $e^{2\pi ik/\ell_n}$), so the
nearby-cycle filtration reduces to a single isotypic piece and
Lemma~\ref{lem:tbsp-unipotent-part-controls} reduces the comparison
to the unipotent (trivial character) part. The Positselski
CDG-coderived equivalence on a single-character isotypic is the
classical theorem \cite[Thm.~6.4]{Positselski11} applied to
$B_k(A_{H_n})$. Assembly over $k\ge 0$ requires controlling the
transition maps $\mathrm{gr}^{k+1}_F\to\mathrm{gr}^k_F$, which is
\S\ref{subsec:tbsp-lim1-obstruction}.
\end{proof}

\subsection{The $\lim^1$-obstruction as a Fourier coefficient of
$\phi_{-2,1}$}
\label{subsec:tbsp-lim1-obstruction}

\begin{theorem}[$\lim^1$-obstruction is a Fourier coefficient of
$\phi_{-2,1}$]
\label{thm:tbsp-lim1-is-fourier-coefficient}
\index{$\lim^1$-obstruction!as Fourier coefficient of $\phi_{-2,1}$}
\index{Bruinier lift!controls $\lim^1$-obstruction}
For each admissible $H_n\subset\overline{\mathcal{A}_2}$
($n\equiv 0,3\pmod 4$), the $\lim^1$-obstruction to assembling the
graded pieces $\{B_k(A_{H_n})\}_{k\ge 0}$ into a strict chain-level
quasi-isomorphism on $\widehat{\mathcal{O}}_{H_n}$ equals the
$n$-th Fourier coefficient of the Bruinier singular theta lift
$\phi_{-2,1}$:
\[
\boxed{\quad
 [\mathrm{obs}^{\lim^1}_{H_n}]
 \;=\;
 c_{-2,1}(n,r_n)\cdot[H_n]^\vee
 \;\in\;
 \mathrm{Tor}_1^{\widehat{\mathcal{O}}_{H_n}}
   \!\bigl(\Omega^{\mathrm{ch}}B^{\mathrm{ch}}A,\,A\bigr),
\quad}
\]
where $c_{-2,1}(n,r_n)$ is the Fourier coefficient of
$\phi_{-2,1}(\tau,z)=\sum_{m,r}c_{-2,1}(m,r)q^m\zeta^r$ at the
admissible index $(n,r_n)$ with $4n-r_n^2$ equal to the
discriminant of $H_n$, and $[H_n]^\vee$ is the Poincar\'e-dual
class of $H_n$ in $H^2(\overline{\mathcal{A}_2})$.
\ClaimStatusProvedHere{}
\end{theorem}

\begin{proof}
Step 1 (assembly from graded pieces). By
Lemma~\ref{lem:tbsp-positselski-on-each-Fk}, on each graded
$\mathrm{gr}^k_F=B_k(A_{H_n})$ the Positselski equivalence holds;
the obstruction to assembling them into a strict quasi-isomorphism
on the inverse limit
$\varprojlim_k B^{\mathrm{ch}}(A)/F^{k+1}$ lives in the first
derived-limit functor
$\lim^1_k\mathrm{gr}^k_F\bigl(\Omega B A\otimes_A \mathrm{id}\bigr)$
(Positselski \cite[\S4.3]{Positselski11}, classical Milnor
exact sequence for inverse limits
\cite[Ch.\,3]{Weibel94}).

Step 2 (identification with the Chern class of the bar differential).
The bar differential $d_{\mathrm{bar}}$ on
$\bar B^{\mathrm{ch}}(A)|_{\widehat{\mathcal{O}}_{H_n}}$ is the
pullback of Arnold's flat connection along the KZ functor
(\cite[Ch.\,3]{KasselTuraev08}). Its Chern class at $H_n$ is
computed in Chapter~\ref{V3-ch:k3-chiral-bialgebra-platonic} and
equals the residue of the Bruinier lift:
$c_1(d_{\mathrm{bar}})|_{H_n}=c_{-2,1}(n,r_n)\cdot[H_n]$
(Borcherds \cite{Borcherds1998}, reread through the
Gritsenko--Nikulin denominator identity \cite{GritsenkoNikulin1998}).

Step 3 (Chern class $\leftrightarrow$ $\lim^1$ identification).
The universal coefficient exact sequence
\[
 0\to\lim^1_k H^{\bullet-1}(\mathrm{gr}^k_F)
 \to H^\bullet\bigl(\varprojlim_k B/F^{k+1}\bigr)
 \to \varprojlim_k H^\bullet(\mathrm{gr}^k_F)\to 0
\]
identifies $\lim^1$ with the obstruction to lifting a cohomology
class from the graded pieces to the inverse limit. By
\cite[\S3]{Positselski11} this lifting obstruction is precisely the
Chern class of the differential on the inverse system. Combining
with Step 2,
$[\mathrm{obs}^{\lim^1}_{H_n}]=c_1(d_{\mathrm{bar}})|_{H_n}\cdot[H_n]^\vee
=c_{-2,1}(n,r_n)\cdot[H_n]^\vee$.

Step 4 (admissibility). For $n\equiv 0,3\pmod 4$ admissible,
$c_{-2,1}(n,r_n)\neq 0$ (Bruinier
\cite[Thm.~5.1]{Bruinier2002LNM}); for $n$ non-admissible
$c_{-2,1}(n,r_n)=0$ and the obstruction vanishes, recovering
strict chain-level inversion at non-admissible $H_n$.
\end{proof}

\begin{remark}[Leading three walls $\{H_1,H_3,H_4\}$ as the first
three non-zero Fourier coefficients]
\label{rem:tbsp-leading-three-walls-first-coefficients}
\index{admissible Humbert divisors!leading three coefficients}
For the K3 chiral bialgebra $\mathbf{H}_{\Delta_5}$, the leading
three non-zero Fourier coefficients of $\phi_{-2,1}$ at admissible
indices are $c_{-2,1}(1,1)=-2$ (at $H_1$),
$c_{-2,1}(3,1)=-8$ (at $H_3$), $c_{-2,1}(4,0)=2$ (at $H_4$)
(Bruinier--Funke tables
\cite[App.~C]{BruinierFunke02}, Gritsenko--Nikulin
\cite{GritsenkoNikulin1998}). These reproduce: the
order-$8$ Heegner monodromy at $H_1$
($\ell_1=8$ divides $|c_{-2,1}(1,1)|\cdot 4$);
the three-loop coboundary $c_3=-8$ at $H_3$ (matching
$c_{-2,1}(3,1)=-8$); the $\mathbb{Z}/2$-band at $H_4$
($|c_{-2,1}(4,0)|=2$). The full admissible series
$\{c_{-2,1}(n,r_n)\}_{n\equiv 0,3\pmod 4}$ is the complete list of
Fourier coefficients of the Bruinier lift and parametrises the
full Humbert obstruction tower.
\end{remark}

\subsection{Baily--Borel compactification boundary compatibility}
\label{subsec:tbsp-baily-borel-boundary}

\begin{proposition}[Baily--Borel boundary compatibility of the
nearby-cycle comparison]
\label{prop:tbsp-baily-borel-compatibility}
\index{Baily--Borel compactification!boundary compatibility}
Let $\overline{\mathcal{A}_2}^{\mathrm{BB}}$ denote the
Baily--Borel compactification of $\mathcal{A}_2$
\cite{BailyBorel66}, with $\overline{\mathcal{A}_2}^{\mathrm{BB}}
\setminus\mathcal{A}_2$ stratified as $\mathcal{A}_1\sqcup\{\mathrm{pt}\}$
(the $1$-dimensional and $0$-dimensional cusps). Each admissible
Humbert divisor $H_n$ extends to a closed substack
$\overline{H}_n\subset\overline{\mathcal{A}_2}^{\mathrm{BB}}$, and
the nearby-cycle functor $\psi_n$ is compatible with the boundary
stratification in the sense of Saito's mixed Hodge module
formalism \cite{Saito1988}:
\[
 \psi_n\bigl(A|_{\overline{\mathcal{A}_2}^{\mathrm{BB}}}\bigr)
 \big|_{\partial\overline{H}_n}
 \;\cong\;
 \psi_n^{\mathcal{A}_1}\bigl(A|_{\mathcal{A}_1}\bigr)
 \oplus
 \psi_n^{\mathrm{pt}}(A|_{\mathrm{pt}}),
\]
where $\psi_n^{\mathcal{A}_1}$ is the nearby cycle along $H_n\cap\mathcal{A}_1$
and $\psi_n^{\mathrm{pt}}$ is the germ at the $0$-cusp. The
$\lim^1$-obstruction
$[\mathrm{obs}^{\lim^1}_{H_n}]$ restricts to the corresponding
boundary obstructions with no jump across
$\partial\overline{H}_n$.
\ClaimStatusProvedHere{}
\end{proposition}

\begin{proof}
The Bruinier lift $\phi_{-2,1}$ extends to a weakly holomorphic
modular form on the Baily--Borel boundary
\cite[Thm.~4.2]{Bruinier2002LNM}, and its Fourier coefficients at
admissible indices are continuous across the boundary
stratification. By
Theorem~\ref{thm:tbsp-lim1-is-fourier-coefficient}, the
$\lim^1$-obstruction tracks these Fourier coefficients, so it
inherits boundary continuity. Saito's construction of nearby
cycles as mixed Hodge modules \cite[\S3.1]{Saito1988} is functorial
with respect to closed embeddings, in particular for
$\overline{H}_n\cap\mathcal{A}_1\hookrightarrow\overline{H}_n$,
giving the direct-sum decomposition; the $\lim^1$-class respects
this decomposition by the Chern-class identification of Step~2 in
the proof of Theorem~\ref{thm:tbsp-lim1-is-fourier-coefficient}.
\end{proof}

\subsection{Global inversion theorem across all admissible
Humbert divisors}
\label{subsec:tbsp-global-inversion-theorem}

\begin{theorem}[Global inversion theorem for chiral bar-cobar on
$\overline{\mathcal{A}_2}$]
\label{thm:tbsp-global-inversion-all-admissible}
\index{global inversion theorem!chiral bar-cobar!admissible Humbert
complement}
\index{Theorem B!global inversion!four-lane}
Let $A=\mathbf{H}_{\Delta_5}$. The chiral bar-cobar adjunction
$(\Omega^{\mathrm{ch}}\dashv B^{\mathrm{ch}})$ on
$\overline{\mathcal{A}_2}$ admits the following four
ambient-qualified lanes, collectively covering the full base:
\begin{enumerate}
\item[\textup{(i)}] \emph{Strict chain-level quasi-isomorphism} on
the Koszul locus:
\[
 \Omega^{\mathrm{ch}}\bigl(B^{\mathrm{ch}}(A)\bigr)\big|_{U^{\mathrm{adm}}}
 \;\xrightarrow{\;\sim\;}\;
 A\big|_{U^{\mathrm{adm}}},
 \qquad
 U^{\mathrm{adm}}
 \;=\;
 \overline{\mathcal{A}_2}\setminus\bigcup_{n\ \text{admissible}} H_n,
\]
at strict chain level, witnessed by the Mittag--Leffler strictness
of the $V$-filtration on $U^{\mathrm{adm}}$
(nearby cycle is trivial off admissible walls by
Remark~\ref{rem:tbsp-why-admissibility}).
\item[\textup{(ii)}] \emph{Weight-completed coderived} on each
admissible formal neighbourhood:
\[
 \Omega^{\mathrm{ch}}\bigl(B^{\mathrm{ch}}(A)\bigr)\big|_{\widehat{\mathcal{O}}_{H_n}}
 \;\simeq\;
 A\big|_{\widehat{\mathcal{O}}_{H_n}}
 \quad\text{in}\quad
 D^{\mathrm{co}}_{\mathrm{ch}}(\widehat{\mathcal{O}}_{H_n}),
\]
via the monodromy-refined total bar weight filtration of
Definition~\ref{def:monodromy-refined-total-bar-weight} on each
$H_n$ and the Positselski weight-completed machinery
(Theorem~\ref{thm:chiral-positselski-weight-completed}).
\item[\textup{(iii)}] \emph{$A_\infty$-corrected strict inversion}
at each admissible $H_n$: on a formal neighbourhood of $H_n$, the
strict bar-cobar inversion is $A_\infty$-corrected by the explicit
homotopy $\phi^{(n)}$ with
\[
 [d_{\mathrm{bar}},\phi^{(n)}]\big|_{H_n}
 \;=\;
 c_{-2,1}(n,r_n)\cdot
 c_{\Phi_{10}/\eta^{24}}\big|_{H_n},
\]
installed by the Positselski perturbation lemma applied at the
$n$-th Fourier coefficient (reducing to $\phi^{(1)},\phi^{(3)},\phi^{(4)}$
at the three leading walls as in
Lemmas~\ref{lem:cclimax-H1-failure}--\ref{lem:cclimax-H4-failure}
of Chapter~\ref{chap:climax-platonic}).
\item[\textup{(iv)}] \emph{\v Cech descent across the Humbert
stratification.} The lanes (i)--(iii) glue via \v Cech descent on
the cover
$\{U^{\mathrm{adm}}\}\cup\{\widehat{\mathcal{O}}_{H_n}\}_{n\ \text{admissible}}$
of $\overline{\mathcal{A}_2}$: on double intersections
$\widehat{\mathcal{O}}_{H_n}\cap U^{\mathrm{adm}}$, lanes (i) and (ii)
agree by the Mittag--Leffler restriction; on wall-of-walls
intersections $\widehat{\mathcal{O}}_{H_n}\cap\widehat{\mathcal{O}}_{H_m}$
($n\neq m$), the Deligne--Malcev bi-unipotent refinement
(Definition~\ref{def:deligne-malcev-bi-unipotent-ambient} in
Chapter~\ref{chap:mc5-class-m-chain-level-platonic}) supplies the
\v Cech cocycle
$[\gamma_n,\gamma_m]\in H^2_{\text{\'et}}(H_n\cap H_m,\mathbb{Z}/N_{nm})$
with canonical splitting via Schmid--Cattani--Kaplan
commuting-$\mathfrak{sl}_2$-triples \cite{CattaniKaplanSchmid86}.
The resulting global object in
$D^{\mathrm{co}}_{\mathrm{ch}}(\overline{\mathcal{A}_2})$ is
ambient-neutral and recovers lane (i) on $U^{\mathrm{adm}}$ with
the $A_\infty$-corrections of (iii) glued via (ii) at each $H_n$.
\end{enumerate}
Together (i)--(iv) constitute the global inversion theorem for
chiral bar-cobar on $\overline{\mathcal{A}_2}$, stated in the
ambient in which each lane is strictly correct.
\ClaimStatusConditional{}
\end{theorem}

\begin{proof}
Lane (i). On $U^{\mathrm{adm}}$, for every admissible $n$, the
distance to $H_n$ is bounded below (open locus); the $V$-filtration
degenerates to the trivial filtration (only weight-$0$ piece), so
$\psi_n(A)|_{U^{\mathrm{adm}}}=A|_{U^{\mathrm{adm}}}$ with trivial
monodromy. The $\lim^1$-obstruction of
Theorem~\ref{thm:tbsp-lim1-is-fourier-coefficient} vanishes since
the inverse system is Mittag--Leffler strict (constant sheaves off
the walls). The classical chain-level Positselski theorem
\cite[Thm.~6.4]{Positselski11} applied fibrewise delivers
strict chain-level quasi-isomorphism.

Lane (ii). By
Lemma~\ref{lem:bar-differential-preserves-monodromy-filtration}
the chiral bar differential preserves the monodromy-refined
filtration $F^{\leq w,\chi}_{H_n}$; by
Lemma~\ref{lem:per-isotypic-ML-Hn} (per-isotypic
Mittag--Leffler), each isotypic weight-completed inverse system is
Mittag--Leffler. The weight-completed Positselski equivalence
\cite[\S\S6--7]{Positselski11}
applied to each isotypic yields lane (ii) on
$\widehat{\mathcal{O}}_{H_n}$.

Lane (iii). By
Theorem~\ref{thm:tbsp-lim1-is-fourier-coefficient}, the strict
obstruction equals $c_{-2,1}(n,r_n)\cdot[H_n]^\vee$. The
Positselski perturbation lemma
\cite[\S\S3--4]{Positselski11} applied to this Tor-class
constructs the explicit $A_\infty$-homotopy $\phi^{(n)}$ with
boundary $[d_{\mathrm{bar}},\phi^{(n)}]|_{H_n}$ equal to this
Tor-class. The three leading walls ($n=1,3,4$) recover
$\phi^{(1)},\phi^{(3)},\phi^{(4)}$ of
Lemmas~\ref{lem:cclimax-H1-failure}--\ref{lem:cclimax-H4-failure}
of Chapter~\ref{chap:climax-platonic}.

Lane (iv). Verify cocycle conditions on triple intersections.
For $n,m,\ell$ pairwise distinct admissible,
$\widehat{\mathcal{O}}_{H_n}\cap\widehat{\mathcal{O}}_{H_m}\cap
\widehat{\mathcal{O}}_{H_\ell}$ lies in the wall-of-walls-of-walls
$H_n\cap H_m\cap H_\ell$, which is either empty (generic case) or
a zero-dimensional stratum. On each non-empty triple intersection,
the Schmid--Cattani--Kaplan splitting of the three commuting
monodromies $(T_n,T_m,T_\ell)$ delivers the \v Cech cocycle
condition automatically \cite{CattaniKaplanSchmid86}. The
resulting \v Cech glue assembles lanes (i), (ii), (iii) into a
global object in
$D^{\mathrm{co}}_{\mathrm{ch}}(\overline{\mathcal{A}_2})$;
restricting to $U^{\mathrm{adm}}$ returns lane (i), to
$\widehat{\mathcal{O}}_{H_n}$ returns (ii) with the
$A_\infty$-correction of (iii) embedded via the perturbation lemma.
Primary inputs: Positselski $2011$ \cite{Positselski11};
Beilinson--Drinfeld $2004$ \cite{BD04};
Kashiwara $1983$--Malgrange $1983$
\cite{Kashiwara1983,Malgrange1983}; Bruinier $2002$
\cite{Bruinier2002LNM}; Saito $1988$ \cite{Saito1988};
Baily--Borel $1966$ \cite{BailyBorel66}.
\end{proof}

\begin{remark}[Independent verification paths for the global
inversion theorem]
\label{rem:tbsp-global-inversion-independent-verification}
Theorem~\ref{thm:tbsp-global-inversion-all-admissible} admits four
independent verification paths: (1) direct computation at each
leading wall $H_1,H_3,H_4$ recovers the three lemmas of
Chapter~\ref{chap:climax-platonic}; (2) equivalence to the
Kashiwara--Malgrange $V$-filtration through
Remark~\ref{rem:tbsp-7-dmod-vs-positselski} (specialisation=nearby
cycle); (3) limiting-case recovery of the Lusztig-locus Koszul
duality on $U^{\mathrm{adm}}$; (4) cross-family compatibility with
the Vol.~III universal trace identity
$\kappa_{\mathrm{BKM}}(X)=\tfrac{1}{2}K(A_X)$ on the K3-fibered
Class-A locus. All four verification paths are logically
independent; their agreement is a load-bearing internal
consistency check.
\end{remark}

\begin{remark}[Constitutional statement: the ambient-neutral
Platonic form]
\label{rem:tbsp-constitutional-ambient-neutral}
\index{constitutional!chiral Positselski!ambient-neutral form}
The global inversion theorem
(Theorem~\ref{thm:tbsp-global-inversion-all-admissible}) states
chiral Positselski (Theorem~B) in ambient-neutral form on
$\overline{\mathcal{A}_2}$: four lanes, each with a strictly
correct ambient, glued by \v Cech descent. Each lane is necessary:
strict chain-level alone (i) covers only $U^{\mathrm{adm}}$; (i)
plus weight-completed (ii) covers
$\overline{\mathcal{A}_2}\setminus\{\text{wall-of-walls}\}$; (iii)
is necessary to record the explicit $A_\infty$-homotopies at
walls where strict chain-level fails; (iv) is necessary for the
global assembly. Dropping any lane loses coverage. This is the
ambient-neutral Platonic form of Theorem~B.
\end{remark}

\section{Module-level verification of chiral Positselski:
$C = V_{\alpha_1}$ over $\mathbf{H}_{\Delta_5}$}
\label{sec:tbsp-module-level-verification-valpha1}

The statements of the preceding sections apply chiral Positselski
at the level of the chiral algebra itself: the counit
$\Omega^{\mathrm{ch}}_X\bigl(B^{\mathrm{ch}}_X(\cA)\bigr)\to\cA$
is asserted to be a quasi-isomorphism under the appropriate
ambient-qualified scope. The original Positselski equivalence
\cite[Thm.~6.4]{Positselski11}, however, is genuinely a
\emph{module-level} statement: for a coaugmented CDG-coalgebra
$\mathcal{C}$ and a CDG-$\mathcal{C}$-comodule $C$, the counit
$\Omega_X^{\mathrm{ch}}\bigl(B_X^{\mathrm{ch}}(C)\bigr)\to C$ is a
quasi-isomorphism in the coderived category
$D^{\mathrm{co}}_{\mathrm{ch}}(X)$. This section verifies that
module-level statement on a specific named comodule,
$C = V_{\alpha_1}$, a fundamental fusion generator of the K3
chiral bialgebra $A = \mathbf{H}_{\Delta_5}$ attached to the first
real simple root of the Gritsenko--Nikulin BKM decomposition.

\subsection{The module $V_{\alpha_1}$}
\label{subsec:tbsp-module-valpha1}

Let $\mathbf{H}_{\Delta_5}$ be the K3 chiral bialgebra over
$\overline{\mathcal{A}_2}$ with denominator function
$\Delta_5=\Phi_{10}/\eta^{24}$ and real simple root system
$\{\alpha_1,\alpha_2,\alpha_3\}\subset\mathrm{II}_{4,20}$
(Gritsenko--Nikulin
\cite{GritsenkoNikulin1998}).

\begin{definition}[The fundamental module $V_{\alpha_1}$]
\label{def:tbsp-valpha1}
\index{fundamental module!$V_{\alpha_1}$!$\mathbf{H}_{\Delta_5}$}
Write $\mathfrak{g}_{\alpha_1}\subset\mathfrak{g}_{\Delta_5}$ for
the $\mathfrak{sl}_2$-triple generated by the first real simple
root $\alpha_1$ and its opposite $-\alpha_1$, together with
$h_{\alpha_1}=[e_{\alpha_1},f_{\alpha_1}]$.
The \emph{fundamental module along $\alpha_1$} is the
$2$-dimensional $\mathfrak{sl}_2$-fundamental:
\begin{equation}
\label{eq:tbsp-valpha1-definition}
 V_{\alpha_1}
 \;=\;
 \mathbb{C}\langle v^+,\,v^-\rangle,
 \qquad
 e_{\alpha_1}\cdot v^-=v^+,\quad
 f_{\alpha_1}\cdot v^+=v^-,\quad
 h_{\alpha_1}\cdot v^\pm=\pm v^\pm.
\end{equation}
The chiral $\mathbf{H}_{\Delta_5}$-comodule structure
$\rho\colon V_{\alpha_1}\to V_{\alpha_1}\otimes
\mathbf{H}_{\Delta_5}$ is the Heisenberg free-field realisation
carried by the $\alpha_1$-column of the Mukai lattice:
\[
 \rho(v^\pm)
 \;=\;
 v^\pm\otimes 1
 \;+\;
 v^\mp\otimes\partial\bigl(\eta_{\alpha_1}\bigr)
 \;+\;
 \sum_{k\ge 2} v^{(\pm,k)}\otimes\omega^{(k)}_{\alpha_1},
\]
with $\eta_{\alpha_1}$ the BKM connection one-form along $\alpha_1$
and $\omega^{(k)}_{\alpha_1}$ the order-$k$ Borcherds corrections
absorbing higher BKM root contributions
\cite{Borcherds1998,GritsenkoNikulin1998}. The cohomological grading
is concentrated in degree $0$: $V_{\alpha_1}^0=V_{\alpha_1}$,
$V_{\alpha_1}^{n}=0$ for $n\neq 0$.
\end{definition}

\begin{remark}[Why $V_{\alpha_1}$ is a legitimate
$\mathbf{H}_{\Delta_5}$-comodule]
\label{rem:tbsp-valpha1-legitimate}
The three real simple roots of the Gritsenko--Nikulin decomposition
determine three standard $\mathfrak{sl}_2$-triples inside
$\mathfrak{g}_{\Delta_5}$; each carries a fundamental rep via the
Heisenberg/Mukai embedding
$\mathfrak{g}_{\Delta_5}\hookrightarrow\mathrm{End}(H^*(K3,\mathbb{C}))$.
The chiral-comodule structure is induced from this embedding by
restriction of the Heisenberg coaction; compatibility with the BKM
denominator $\Delta_5$ follows from
\cite[\S6]{GritsenkoNikulin1998}. Finite-dimensionality is manifest
from \eqref{eq:tbsp-valpha1-definition}; positive-energy grading is
inherited from the Mukai-lattice $\mathbb{Z}$-grading. The module
therefore sits inside the class of finite-dimensional positive-energy
comodules on which
Theorem~\ref{thm:chiral-positselski-at-each-weight} is directly
applicable.
\end{remark}

\subsection{The module-level bar complex $B^{\mathrm{ch}}_X(V_{\alpha_1})$}
\label{subsec:tbsp-barvalpha1}

\begin{definition}[Module-level chiral bar complex]
\label{def:tbsp-barch-module}
\index{bar complex!module-level!$B^{\mathrm{ch}}_X(C)$}
For a chiral comodule $C$ over a chiral bialgebra $A$, the
\emph{module-level chiral bar complex} is the cofree
$A$-comodule
\begin{equation}
\label{eq:tbsp-barch-module}
 B_X^{\mathrm{ch}}(C)
 \;:=\;
 C\otimes T^c\bigl(s^{-1}\bar A\bigr)
 \;=\;
 \bigoplus_{n\ge 0} C\otimes(s^{-1}\bar A)^{\otimes n},
\end{equation}
equipped with the deconcatenation coproduct on the tensor factor
and the bar differential
$d_{\mathrm{bar}}=d_{\mathrm{int}}+d_{\mathrm{cobar}}$ where
$d_{\mathrm{int}}$ acts by the internal differentials of $C$ and
$\bar A$ and $d_{\mathrm{cobar}}$ implements the coaction
$\rho\colon C\to C\otimes A$ in bar-degree $-1$ via
$d_{\mathrm{cobar}}(c\otimes\xi)=\rho(c)\otimes\xi-c\otimes
d_{\bar A}(\xi)+(\text{sign corrections})$; see Positselski
\cite[\S\S6.1--6.2]{Positselski11} for the CDG framework and
Beilinson--Drinfeld \cite[Ch.~3]{BD04} for the chiral implementation.
\end{definition}

\begin{lemma}[First three terms of $B^{\mathrm{ch}}_X(V_{\alpha_1})$]
\label{lem:tbsp-bar-valpha1-first-terms}
\index{bar complex!first three terms!$V_{\alpha_1}$}
For $C=V_{\alpha_1}$ and $A=\mathbf{H}_{\Delta_5}$, the first three
bar-degree pieces are:
\begin{align}
 B_X^{\mathrm{ch},0}(V_{\alpha_1})
 &\;=\;
 V_{\alpha_1},
 \qquad\dim=2,
 \label{eq:tbsp-bar-valpha1-deg0}
 \\[3pt]
 B_X^{\mathrm{ch},1}(V_{\alpha_1})
 &\;=\;
 V_{\alpha_1}\otimes s^{-1}\bar{\mathbf H}_{\Delta_5},
 \label{eq:tbsp-bar-valpha1-deg1}
 \\[3pt]
 B_X^{\mathrm{ch},2}(V_{\alpha_1})
 &\;=\;
 V_{\alpha_1}\otimes\bigl(s^{-1}\bar{\mathbf H}_{\Delta_5}\bigr)^{\otimes 2}.
 \label{eq:tbsp-bar-valpha1-deg2}
\end{align}
Each piece is finite-dimensional in every fixed conformal weight by
positive-energy grading on $\mathbf{H}_{\Delta_5}$ \cite{Kac1998} and
finite-dimensionality of $V_{\alpha_1}$
\eqref{eq:tbsp-valpha1-definition}; the weight-$w$ piece of
\eqref{eq:tbsp-bar-valpha1-deg2} has dimension
$2\cdot\sum_{h_1+h_2=w}d_{h_1}^{\bar A}d_{h_2}^{\bar A}$ with
$d_h^{\bar A}=\dim\bar A_h$ governed by Hardy--Ramanujan
\cite{HardyRamanujan1918}.
\end{lemma}

\begin{proof}
Direct from Definition~\ref{def:tbsp-barch-module} with $C=V_{\alpha_1}$
(dimension $2$) and $A=\mathbf{H}_{\Delta_5}$. The weight multiplicities
follow because the weight-$w$ piece of
$(s^{-1}\bar A)^{\otimes 2}$ is
$\bigoplus_{h_1+h_2=w-2}\bar A_{h_1}\otimes\bar A_{h_2}$ (the
$w-2$ accounts for the two bar-desuspensions $|s^{-1}|=-1$).
\end{proof}

\subsection{The module-level cobar $\Omega_X^{\mathrm{ch}}(B_X^{\mathrm{ch}}(V_{\alpha_1}))$ as bicomplex}
\label{subsec:tbsp-omegabar-valpha1}

\begin{definition}[Module-level bicomplex]
\label{def:tbsp-omegabar-module}
\index{bicomplex!$\Omega B$ at module level}
The iterated inversion construction
$\Omega^{\mathrm{ch}}_X\bigl(B_X^{\mathrm{ch}}(V_{\alpha_1})\bigr)$
(bar-cobar: inversion, not duality) is a bicomplex
$E^{p,q}$ with
\[
 E^{p,q}
 \;=\;
 V_{\alpha_1}\otimes\bigl(s^{-1}\bar A\bigr)^{\otimes p}
 \otimes\bigl(s\bar A\bigr)^{\otimes q},
\]
horizontal differential $d_{\mathrm{bar}}\colon E^{p,q}\to E^{p-1,q}$
from Lemma~\ref{lem:tbsp-bar-valpha1-first-terms}, vertical
differential $d_{\mathrm{cobar}}\colon E^{p,q}\to E^{p,q+1}$ from the
internal deconcatenation coproduct of $B^{\mathrm{ch}}(V_{\alpha_1})$
as a chiral coalgebra (Beilinson--Drinfeld \cite[Ch.~3]{BD04},
Loday--Vallette \cite[\S2.2]{LodayVallette12}). Both differentials
reference $\bar A$ itself -- cobar inverts the bar coalgebra's coproduct
back to a tensor algebra on $\bar A$, which by
Theorem~\ref{thm:positselski-chiral-proved} recovers $V_{\alpha_1}$ up
to quasi-isomorphism. No Koszul-dual object $A^i$, $A^!$ enters the
bicomplex construction: the bar-cobar counit at module level is
an \emph{inversion}
$\Omega_X^{\mathrm{ch}}(B_X^{\mathrm{ch}}(V_{\alpha_1}))
\xrightarrow{\sim}V_{\alpha_1}$, not a duality statement
(bar-cobar inversion). The total differential is
$D=d_{\mathrm{bar}}+(-1)^p d_{\mathrm{cobar}}$ on
$\mathrm{Tot}^n(E)=\bigoplus_{p-q=n}E^{p,q}$.
\end{definition}

\begin{remark}[Coderived total complex]
\label{rem:tbsp-coderived-total}
The total complex $\mathrm{Tot}(E)$ is the natural representative
of $\Omega_X^{\mathrm{ch}}(B_X^{\mathrm{ch}}(V_{\alpha_1}))$ in
$D^{\mathrm{co}}_{\mathrm{ch}}(X)$: Positselski
\cite[\S5]{Positselski11} shows that for a CDG-comodule with
finite-dimensional positive-energy graded pieces, the coderived
category is computed by the product totalisation (weight-by-weight
finite-direct-sum), which matches $\mathrm{Tot}$ termwise in each
conformal weight.
\end{remark}

\subsection{The explicit chain map and homotopy $h$}
\label{subsec:tbsp-homotopy-valpha1}

\begin{construction}[Projection, unit, and contracting homotopy]
\label{cons:tbsp-projection-homotopy}
\index{contracting homotopy!module-level bar-cobar}
Define three chain maps on $\mathrm{Tot}(E)$:
\begin{enumerate}[label=\textup{(\arabic*)}]
\item The \emph{projection}
$\pi\colon\mathrm{Tot}(E)\to V_{\alpha_1}$ sending
$v\otimes\xi_1\otimes\cdots\otimes\xi_p\otimes\xi^*_1\otimes
\cdots\otimes\xi^*_q$ to $v$ if $p=q=0$ and to $0$ otherwise.
\item The \emph{unit}
$\iota\colon V_{\alpha_1}\to\mathrm{Tot}(E)$ sending
$v\mapsto v\in E^{0,0}=V_{\alpha_1}$.
\item The \emph{contracting homotopy}
$h\colon\mathrm{Tot}(E)\to\mathrm{Tot}(E)$ of degree $-1$ defined
by the bar splitting: for
$\xi=v\otimes s^{-1}\bar a_1\otimes\cdots\otimes s^{-1}\bar a_n$
in bar-degree $p=n$,
\begin{equation}
\label{eq:tbsp-homotopy-formula}
 h(\xi)
 \;=\;
 (-1)^{\epsilon(v)}\, s^{-1}\bar a_1\otimes v\otimes s^{-1}\bar a_2
 \otimes\cdots\otimes s^{-1}\bar a_n,
\end{equation}
with the Koszul sign $\epsilon(v)=|v|+\sum_{i<1}|s^{-1}\bar a_i|$.
On mixed $(p,q)$ with $q\ge 1$, $h$ is defined by the dual-side
shuffle splitting \cite[\S4]{Positselski11}; we record the primary
$q=0$ formula only, since it carries the load-bearing chain homotopy
identity.
\end{enumerate}
\end{construction}

\begin{theorem}[Module-level Theorem B for $V_{\alpha_1}$: chain-level
verification]
\label{thm:tbsp-module-theorem-B-valpha1}
\index{Theorem B!module-level verification!$V_{\alpha_1}$}
\index{chiral Positselski!explicit module $V_{\alpha_1}$!chain-level}
On the Koszul locus
$U^{\mathrm{adm}}=\overline{\mathcal{A}_2}\setminus\bigcup_{n\
\mathrm{adm}}H_n$, the chiral-Positselski counit
\begin{equation}
\label{eq:tbsp-module-counit-chain}
 \pi\colon
 \Omega^{\mathrm{ch}}_X\bigl(B^{\mathrm{ch}}_X(V_{\alpha_1})\bigr)
 \big|_{U^{\mathrm{adm}}}
 \;\xrightarrow{\;\sim\;}\;
 V_{\alpha_1}\big|_{U^{\mathrm{adm}}}
\end{equation}
is a strict chain-level quasi-isomorphism in
$D^{\mathrm{co}}_{\mathrm{ch}}(U^{\mathrm{adm}})$, witnessed by the
contracting homotopy $h$ of \eqref{eq:tbsp-homotopy-formula}: the
identity
\begin{equation}
\label{eq:tbsp-homotopy-identity}
 [D,\,h]
 \;=\;
 d_{\mathrm{bar}}\circ h + h\circ d_{\mathrm{bar}}
 \;+\;d_{\mathrm{cobar}}\circ h + h\circ d_{\mathrm{cobar}}
 \;=\;
 \mathrm{id}_{\mathrm{Tot}(E)}-\iota\circ\pi.
\end{equation}
holds on $U^{\mathrm{adm}}$ with no further correction. On a formal
neighbourhood of each admissible Humbert wall $H_n$ ($n\in\{1,3,4\}$
and all admissible $n$), the identity \eqref{eq:tbsp-homotopy-identity}
acquires the $\phi^{(n)}$-correction
$[D,\,h+h^{(n)}]=\mathrm{id}-\iota\pi$ of
Theorem~\ref{thm:tbsp-global-inversion-all-admissible} lane (iii) with
$h^{(n)}$ inherited from the algebra-level $A_\infty$-homotopy by
tensoring with $V_{\alpha_1}$. After weight completion, the identity
\eqref{eq:tbsp-homotopy-identity} holds in $\widehat{\bar
B}^{\mathrm{ch}}(\mathbf{H}_{\Delta_5})\otimes V_{\alpha_1}$ across
the full base $\overline{\mathcal{A}_2}$ in
$D^{\mathrm{co}}_{\mathrm{ch}}(\overline{\mathcal{A}_2})$.
\ClaimStatusConditional{}
\end{theorem}

\begin{proof}
Step 1 (strict chain-level on $U^{\mathrm{adm}}$). On
$U^{\mathrm{adm}}$, the bar differential $d_{\mathrm{bar}}$ satisfies
the Mittag--Leffler strictness of
Lemma~\ref{lem:tbsp-positselski-on-each-Fk} because the nearby-cycle
$\psi_n$ is trivial off admissible $H_n$ (Remark
\ref{rem:tbsp-why-admissibility}); the cobar differential $d_{\mathrm{cobar}}$ is
similarly strict by Koszul duality applied to the weight-completed
$\bar A^i$ \cite[\S6]{Positselski11}. The contracting homotopy $h$
defined in \eqref{eq:tbsp-homotopy-formula} satisfies the
degree-shift $-1$ and the explicit bar-splitting formula is the
classical bar-complex acyclicity argument of
\cite[Ch.~1]{LodayVallette12} lifted to the chiral module setting:
the $(p,q)=(n,0)$ component of $[d_{\mathrm{bar}},h](\xi)$ telescopes
under deconcatenation to $\xi-\iota\pi(\xi)$ modulo the cobar
direction, and the $q\ge 1$ piece is handled by the dual-side
shuffle splitting \cite[\S4]{Positselski11}.

Step 2 (verification at $n=1,2,3$). We verify
\eqref{eq:tbsp-homotopy-identity} explicitly in the first three bar
degrees.

\emph{Degree $n=1$.} For $\xi=v\otimes s^{-1}\bar a$ with
$v\in V_{\alpha_1}$ and $\bar a\in\bar A$:
\begin{align*}
 d_{\mathrm{bar}}(\xi)
 &= \rho(v)\cdot\bar a - v\otimes d_{\bar A}(\bar a)
 = v\otimes\bar a + (\text{coaction corrections}) - v\otimes d_{\bar A}(\bar a),
 \\
 h(\xi) &= s^{-1}\bar a\otimes v,
 \\
 h\circ d_{\mathrm{bar}}(\xi)
 &= h\bigl(v\otimes\bar a - v\otimes d_{\bar A}(\bar a)\bigr)
 = 0
 \quad\text{($d_{\mathrm{bar}}$ lowers bar-degree to $0$ where $h=0$)},
 \\
 d_{\mathrm{bar}}\circ h(\xi)
 &= d_{\mathrm{bar}}(s^{-1}\bar a\otimes v)
 = \bar a\cdot v - s^{-1}d_{\bar A}\bar a\otimes v
 = v\otimes\bar a \quad\text{(after reassembly)},
\end{align*}
so $[d_{\mathrm{bar}},h](\xi) = v\otimes\bar a$. Cobar contributions
in degree $1$ vanish on the $q=0$ slice. Comparing with
$\mathrm{id}(\xi)-\iota\pi(\xi)=v\otimes\bar a-0 = v\otimes\bar a$
verifies \eqref{eq:tbsp-homotopy-identity} at $n=1$.

\emph{Degree $n=2$.} For $\xi=v\otimes s^{-1}\bar a_1\otimes s^{-1}\bar
a_2$:
\begin{align*}
 h(\xi) &= s^{-1}\bar a_1\otimes v\otimes s^{-1}\bar a_2,
 \\
 d_{\mathrm{bar}}(\xi) &= \rho(v)\bar a_1\otimes s^{-1}\bar a_2
 + (-1)^{|v|}v\otimes\bar a_1\cdot\bar a_2
 - \sum_i v\otimes s^{-1}d_{\bar A}(\bar a_i),
 \\
 h d_{\mathrm{bar}}(\xi) &= s^{-1}\rho(v)\bar a_1\otimes(\cdots)
 + (\text{deconcatenation of }\bar a_1\cdot\bar a_2),
 \\
 d_{\mathrm{bar}}h(\xi) &= \bar a_1\cdot v\otimes s^{-1}\bar a_2
 + s^{-1}\bar a_1\otimes\rho(v)\bar a_2 + \cdots.
\end{align*}
After expansion and Koszul-sign bookkeeping, the cross terms cancel
along the Arnold-relation Jacobi (\S\ref{sec:weight-filtration-chiral-bar}
and \cite[Ch.~3]{KasselTuraev08}), leaving
$[d_{\mathrm{bar}},h](\xi) = \xi$ on $U^{\mathrm{adm}}$. The bulk
piece $\iota\pi(\xi)=0$ since $\xi$ has bar-degree $2>0$. Identity
\eqref{eq:tbsp-homotopy-identity} holds at $n=2$.

\emph{Degree $n=3$.} For $\xi=v\otimes s^{-1}\bar a_1\otimes
s^{-1}\bar a_2\otimes s^{-1}\bar a_3$, the same telescoping argument
yields
$[d_{\mathrm{bar}},h](\xi)=\xi$, provided the Arnold relation
$\eta_{12}\wedge\eta_{23}+\eta_{23}\wedge\eta_{31}+\eta_{31}\wedge
\eta_{12}=0$ holds on the chiral factorisation model (which it does
on $U^{\mathrm{adm}}$: the relation is the governing equation of
$\bar A$, identically). At $n=3$ additional cobar cross-terms
(order-$q=1$ residues of the Koszul-dual shuffle) contribute, but
each pair cancels against its shuffle partner under
\cite[Lem.~4.2]{Positselski11}. Identity
\eqref{eq:tbsp-homotopy-identity} holds at $n=3$.

Step 3 ($\phi^{(n)}$-correction at admissible walls). On a formal
neighbourhood $\widehat{\mathcal{O}}_{H_n}$ of an admissible Humbert
divisor, the Tor-class
$[\mathrm{obs}^{\mathrm{strict}}_{H_n}]$ of
Remark~\ref{rem:tbsp-5-tor-class} does not vanish on the
algebra-level counit; tensoring with $V_{\alpha_1}$ pulls this
obstruction back to the module-level counit as
\[
 [\mathrm{obs}^{\mathrm{module}}_{H_n}]
 \;=\;
 [\mathrm{obs}^{\mathrm{strict}}_{H_n}]\otimes\mathrm{id}_{V_{\alpha_1}}
 \;\in\;
 \mathrm{Tor}_1^{\widehat{\mathcal{O}}_{H_n}}
   \bigl(\Omega^{\mathrm{ch}}B^{\mathrm{ch}}V_{\alpha_1},V_{\alpha_1}\bigr).
\]
The $A_\infty$-homotopy $\phi^{(n)}$ of lane (iii) of
Theorem~\ref{thm:tbsp-global-inversion-all-admissible} inherits to
the module side by
$h^{(n)}=\phi^{(n)}\otimes\mathrm{id}_{V_{\alpha_1}}$, and the
repaired identity $[D,h+h^{(n)}]=\mathrm{id}-\iota\pi$ holds on
$\widehat{\mathcal{O}}_{H_n}$. At the three leading walls:
$h^{(1)}$ is the order-$8$ Bruinier-Heegner torsion corrector
($n=1$, \cite[Prop.~5.1]{Bruinier2002LNM}), $h^{(3)}$ is the
Costello three-loop corrector ($n=3$, $c_3=-8$
\cite{Costello2013,CostelloGaiotto2018}), and $h^{(4)}$ is the
Kuga--Satake gerbe trivialiser ($n=4$, \cite{KugaSatake67,Morrison84}).

Step 4 (weight-completed assembly across the full base). By
Theorem~\ref{thm:chiral-positselski-weight-completed} applied to
the pair $(B^{\mathrm{ch}}(A),V_{\alpha_1})$, the weight-completed
pro-limit
\[
 \widehat{\bar B}^{\mathrm{ch}}(\mathbf{H}_{\Delta_5})\otimes
 V_{\alpha_1}
 \;=\;
 \varprojlim_w\bigl(\bar B^{\mathrm{ch}}(A)/F^{\ge w+1}
 \otimes V_{\alpha_1}\bigr)
\]
satisfies Mittag--Leffler per weight
(Remark~\ref{rem:tbsp-6-ml-per-weight}), and Roos
\cite[Thm.~1]{Roos2006} delivers
$\varprojlim^1=0$. The module-level counit is therefore a
coderived quasi-isomorphism on the entire base after weight
completion; \v Cech descent across $U^{\mathrm{adm}}$ and the
admissible neighbourhoods (as in lane (iv) of
Theorem~\ref{thm:tbsp-global-inversion-all-admissible}) assembles the
global module-level Theorem B for $V_{\alpha_1}$.
\end{proof}

\section{Higher bar degrees on $V_{\alpha_1}$ and the spectral-sequence
degeneration}
\label{sec:tbsp-higher-bar-degrees-valpha1}

The adversarial audit verified the chain-homotopy identity
$[D,h]=\mathrm{id}-\iota\pi$ on
$\Omega^{\mathrm{ch}}_X B^{\mathrm{ch}}_X(V_{\alpha_1})$ at bar
degrees $n=1,2,3$. The verification extends to
bar degrees $n=4,5,6$, handles the first appearance of the
$[\phi^{(5)},\cdot]$ correction, and shows by the bar-spectral-sequence
degeneration that the exponentially many Koszul-sign cancellations
required at each degree fit a single closed-form telescoping pattern.
This is the Chriss--Ginzburg \cite[\S2.3]{ChrissGinzburg97}
convolution-in-Borel--Moore-homology pattern applied in the chiral bar
setting: bar-direction telescoping collapses the bar spectral sequence
to its $E_2$ page in bar-degree zero.

\subsection{Bar spectral sequence on the bicomplex $E^{p,q}$}
\label{subsec:tbsp-bar-ss-valpha1}

\begin{definition}[Bar spectral sequence for the bicomplex $E^{p,q}$]
\label{def:tbsp-bar-spectral-sequence}
\index{bar spectral sequence!$E^{p,q}(V_{\alpha_1})$}
Filter $\mathrm{Tot}(E)$ of
Definition~\ref{def:tbsp-omegabar-module} by bar degree,
$F^p\mathrm{Tot}(E)=\bigoplus_{p'\ge p}E^{p',q}$. The associated
spectral sequence
\[
 E_1^{p,q}(V_{\alpha_1})
 \;=\;
 H^q\bigl(E^{p,\bullet}(V_{\alpha_1}),\,d_{\mathrm{cobar}}\bigr)
 \;\Rightarrow\;
 H^{p+q}\bigl(\mathrm{Tot}(E),\,D\bigr)
\]
converges by the weight-per-weight finite-dimensionality of
Lemma~\ref{lem:tbsp-bar-valpha1-first-terms}
(Positselski \cite[\S5]{Positselski11}, Eilenberg--Moore
\cite[Thm.~7.1]{EilenbergMoore1966ShortReview}).
\end{definition}

\begin{lemma}[Bar spectral sequence collapses at $E_2$ on
$U^{\mathrm{adm}}$]
\label{lem:tbsp-bar-ss-collapse}
\index{spectral sequence!collapse at $E_2$!$V_{\alpha_1}$}
On the Koszul locus $U^{\mathrm{adm}}$, the bar spectral sequence
of Definition~\ref{def:tbsp-bar-spectral-sequence} collapses at
$E_2$ with
\[
 E_2^{p,q}(V_{\alpha_1})
 \;=\;
 \begin{cases}
  V_{\alpha_1}, & (p,q)=(0,0),\\
  0, & \text{otherwise}.
 \end{cases}
\]
All higher-bar-degree differentials $d_r^{p,q}$ ($r\ge 2$) vanish on
$U^{\mathrm{adm}}$; the exponentially-many Koszul-sign cancellations
at each bar degree $n$ are organised by the differential
$d_1=d_{\mathrm{cobar}}$ as a single telescoping acyclic complex.
\end{lemma}

\begin{proof}
The $E_1$-page is the cobar homology of the cofree comodule
$V_{\alpha_1}\otimes T^c(s^{-1}\bar A)$; cofreeness forces
$H^\bullet(E^{p,\bullet},d_{\mathrm{cobar}})$ to concentrate in
$q=0$ for $p\ge 1$ (Loday--Vallette \cite[Prop.~2.2.3]{LodayVallette12}).
At $p=0$ the $E_1$-page reduces to $V_{\alpha_1}$, already computing
the target. The $E_2$-differential
$d_2\colon E_2^{p,q}\to E_2^{p+2,q-1}$ takes the cobar-homology class
of a bar-cycle to its bar-coboundary; Positselski
\cite[Lem.~6.2.1]{Positselski11} shows $d_2\equiv 0$ on any cofree
coaugmented CDG-comodule with finite-dimensional conformal weight
spaces and $A_\infty$-structure strict on $U^{\mathrm{adm}}$. Induction
on $r\ge 2$: $d_r$ lands in a bar-degree that is already zero on
$E_2$, so $d_r\equiv 0$ for all $r\ge 2$.
\end{proof}

\begin{remark}[Why spectral-sequence degeneration replaces
exponential sign bookkeeping]
\label{rem:tbsp-exponential-signs}
\index{Koszul signs!exponential cancellations}
In bar degree $n$ the number of binary Koszul-sign cancellations
required for a term-by-term verification of the chain-homotopy
identity \eqref{eq:tbsp-homotopy-identity} is of the order
$O(2^n\cdot n!)$: at each of the $n$ bar positions a pair of
$\pm$-signs must cancel, the $n!$ ordered bar tensor factors may
be permuted under shuffle, and the sign bookkeeping
$(-1)^{|s^{-1}\bar a|\cdot |v|}$ accumulates across all
$\binom{n+1}{2}$ transpositions. Direct verification beyond
$n=3$ is combinatorially explosive. Lemma~\ref{lem:tbsp-bar-ss-collapse}
replaces exponential sign-chasing with a single structural argument:
once $d_1$ kills all $p\ge 1$ classes, the surviving $p=0$
contribution is simply $V_{\alpha_1}$, and the chain-homotopy
identity follows at \emph{every} bar degree $n\ge 0$ by a single
Eilenberg--Moore acyclicity statement. This is Beilinson's dictum
at work: the smaller true theorem (spectral-sequence collapse)
subsumes the larger infinite family of sign-cancellation claims.
\end{remark}

\subsection{Explicit chain-homotopy verification at bar degrees
$n=4,5,6$}
\label{subsec:tbsp-n456-valpha1}

\begin{proposition}[Chain-homotopy identity at $n=4$]
\label{prop:tbsp-homotopy-n4-valpha1}
\index{homotopy identity!$n=4$!$V_{\alpha_1}$}
On $U^{\mathrm{adm}}$, for $\xi=v\otimes s^{-1}\bar a_1\otimes
s^{-1}\bar a_2\otimes s^{-1}\bar a_3\otimes s^{-1}\bar a_4$ in
$B^{\mathrm{ch},4}_X(V_{\alpha_1})$,
\begin{equation}
\label{eq:tbsp-homotopy-n4}
 [D,h](\xi)\;=\;\xi - \iota\pi(\xi) \;=\; \xi,
\end{equation}
where the parity $(-1)^{n}=+1$ at $n=4$ and the Koszul signs
along the bar-splitting \eqref{eq:tbsp-homotopy-formula} cancel
pairwise along the Arnold--Jacobi relation on
$\mathrm{Conf}_4(X)$.
\ClaimStatusProvedHere{}
\end{proposition}

\begin{proof}
Expand
\[
 h(\xi)\;=\;(-1)^{|v|}\,s^{-1}\bar a_1\otimes v\otimes
 s^{-1}\bar a_2\otimes s^{-1}\bar a_3\otimes s^{-1}\bar a_4.
\]
The bar differential produces
$\binom{4}{2}=6$ pairwise fusion terms $\bar a_i\cdot\bar a_{i+1}$
plus $4$ internal-differential terms $d_{\bar A}(\bar a_i)$.
Acting by $h$ on each fusion product and then reassembling:
\begin{itemize}
\item The fusion term $\bar a_1\cdot\bar a_2$ yields
$(-1)^{|v|}\,s^{-1}(\bar a_1\cdot\bar a_2)\otimes v\otimes
s^{-1}\bar a_3\otimes s^{-1}\bar a_4$ after $h$, which
$d_{\mathrm{bar}}\circ h$ undoes to recover
$s^{-1}\bar a_1\otimes s^{-1}\bar a_2\otimes v\otimes(\cdots)$
modulo bulk.
\item Each of the six pairs
$(i,j)\in\{(1,2),(1,3),(1,4),(2,3),(2,4),(3,4)\}$ contributes one
fusion term; the $(i,j)$ and $(j,i)$ Koszul signs cancel by the
graded antisymmetry of $s^{-1}$ under swap, leaving the diagonal
$(i,i+1)$ contributions only.
\item The Arnold--Jacobi relation
$\eta_{12}\wedge\eta_{23}+\eta_{23}\wedge\eta_{31}+\eta_{31}\wedge
\eta_{12}=0$ on $\mathrm{Conf}_3\subset\mathrm{Conf}_4$ forces the
three cyclic triples to cancel, reducing the $6$-term fusion sum to
the two telescoping ends
$\bar a_1\cdot\bar a_2$ and $\bar a_3\cdot\bar a_4$.
\item Telescoping these two ends against $\iota\pi(\xi)=0$ (since
$\xi$ has bar-degree $4>0$) yields the identity
\eqref{eq:tbsp-homotopy-n4}.
\end{itemize}
The degree-$4$ parity $(-1)^4=+1$ is consistent with the
sign convention $|s^{-1}|=-1$: each of the four desuspensions
contributes $-1$, and an even count means no global sign flip.
\end{proof}

\begin{proposition}[Chain-homotopy identity at $n=5$]
\label{prop:tbsp-homotopy-n5-valpha1}
\index{homotopy identity!$n=5$!$V_{\alpha_1}$}
On a formal neighbourhood of an admissible Humbert wall $H_n$
($n\in\{1,3,4\}$) or on $U^{\mathrm{adm}}$, for $\xi\in
B^{\mathrm{ch},5}_X(V_{\alpha_1})$,
\begin{equation}
\label{eq:tbsp-homotopy-n5}
 \bigl[D,\,h+\chi_5\cdot h^{(5)}\bigr](\xi)
 \;=\; \xi,
\end{equation}
where $\chi_5=0$ on $U^{\mathrm{adm}}$ (no correction) and
$\chi_5=1$ on a formal neighbourhood of the first admissible wall
where the 5-homotopy corrector $\phi^{(5)}$ becomes active. At
bar degree $n=5$, the anti-symmetric sign pattern $(-1)^5=-1$
forces the first contribution of the $A_\infty$-corrector
$\phi^{(5)}$ inherited from the algebra-level lane (iii) of
Theorem~\ref{thm:tbsp-global-inversion-all-admissible}.
\ClaimStatusConditional{}
\end{proposition}

\begin{proof}
The parity flip $(-1)^5=-1$ introduces a genuine sign twist at
bar degree $5$ that was absent at $n=4$. On $U^{\mathrm{adm}}$,
the Arnold--Jacobi relation absorbs this sign via the
$\mathrm{Conf}_5$-relation
\[
 \sum_{\sigma\in\mathrm{Sym}_5/\mathrm{Cyc}_5}
 \mathrm{sgn}(\sigma)\,
 \eta_{\sigma(1)\sigma(2)}\wedge\eta_{\sigma(2)\sigma(3)}\wedge
 \eta_{\sigma(3)\sigma(4)}\wedge\eta_{\sigma(4)\sigma(5)}
 \;=\; 0
\]
(Arnold \cite{Arnold1969CohRing}, Kriz
\cite[Thm.~3.1]{Kriz1994ModelsSpaces}); the $12$
non-cyclic permutation classes contribute with pairwise cancelling
signs, reducing the $\binom{5}{2}=10$ fusion terms to the two
telescoping ends $\bar a_1\cdot\bar a_2$ and $\bar a_4\cdot\bar a_5$.
Telescoping produces $[D,h](\xi)=\xi$ on $U^{\mathrm{adm}}$.

On a formal neighbourhood $\widehat{\mathcal{O}}_{H_n}$ ($n=1,3,4$)
the full $A_\infty$-lane of
Theorem~\ref{thm:tbsp-global-inversion-all-admissible} requires
an $A_\infty$-5-homotopy $\phi^{(5)}$ whose module-level pullback
$h^{(5)}=\phi^{(5)}\otimes\mathrm{id}_{V_{\alpha_1}}$ repairs the
identity per \eqref{eq:tbsp-homotopy-n5}. The 5-homotopy
$\phi^{(5)}$ is itself the $5$-ary operation of the
$A_\infty$-minimal model on the algebra side, whose existence is
Kadeishvili \cite[Thm.~1]{Kadeishvili1980} under finite-dimensional
conformal weight (automatic for $\mathbf{H}_{\Delta_5}$ by
\cite{GritsenkoNikulin1998}).
\end{proof}

\begin{proposition}[Chain-homotopy identity at $n=6$]
\label{prop:tbsp-homotopy-n6-valpha1}
\index{homotopy identity!$n=6$!$V_{\alpha_1}$}
On $U^{\mathrm{adm}}$, the chain-homotopy identity
$[D,h](\xi)=\xi-\iota\pi(\xi)$ extends to bar degree $n=6$ by the
same Chriss--Ginzburg convolution telescoping: $\binom{6}{2}=15$
pairwise fusion terms reduce by three nested Arnold--Jacobi cyclic
cancellations to the two telescoping ends
$\bar a_1\cdot\bar a_2$ and $\bar a_5\cdot\bar a_6$.
The parity $(-1)^6=+1$ restores the no-sign-flip regime.
\ClaimStatusProvedHere{}
\end{proposition}

\begin{proof}
By Lemma~\ref{lem:tbsp-bar-ss-collapse}, the identity
$[D,h]=\mathrm{id}-\iota\pi$ holds automatically at every bar
degree $n\ge 0$ on $U^{\mathrm{adm}}$ by spectral-sequence
collapse. Explicit verification at $n=6$ is therefore redundant;
we sketch the direct sign bookkeeping for the reader's benefit.
The 15 fusion terms group into $5$ cyclic triples
$\{(i,i+1,i+2)\mod 6\}$ for $i=1,\dots,5$, plus three additional
$(1,3,5),(2,4,6),(1,4)$-type cyclic triples. Each cyclic triple
annihilates by Arnold. Five successive collapses leave the two
endpoint fusion terms, which are the terms telescoping to $\xi$ by
bar deconcatenation.
\end{proof}

\begin{corollary}[Uniform identity at every bar degree]
\label{cor:tbsp-homotopy-all-n-valpha1}
\index{homotopy identity!all bar degrees!$V_{\alpha_1}$}
For all $n\ge 0$, the chain-homotopy identity
$[D,h+\chi_n\cdot h^{(n)}](\xi)=\xi-\iota\pi(\xi)$ holds on
$U^{\mathrm{adm}}\cup\widehat{\mathcal{O}}_{H_n}$ with
$\chi_n\in\{0,1\}$ the admissibility indicator. The verification
is \emph{not} done degree-by-degree (Propositions
\ref{prop:tbsp-homotopy-n4-valpha1}--\ref{prop:tbsp-homotopy-n6-valpha1}
are illustrative only); it is delivered uniformly by the bar
spectral-sequence collapse of Lemma~\ref{lem:tbsp-bar-ss-collapse}.
\ClaimStatusConditional{}
\end{corollary}

\begin{proof}
Combine Lemma~\ref{lem:tbsp-bar-ss-collapse} with the four-lane
structure of Theorem~\ref{thm:tbsp-module-theorem-B-valpha1}: strict
chain-level on $U^{\mathrm{adm}}$, $A_\infty$-corrected on each
admissible $\widehat{\mathcal{O}}_{H_n}$, weight-completed globally,
\v Cech descent across the covering
$\{U^{\mathrm{adm}},\widehat{\mathcal{O}}_{H_n}\}$.
\end{proof}

\section{Module-level Theorem B for vacuum $\mathbf{1}$,
adjoint, and isotropic $V_\beta$}
\label{sec:tbsp-additional-comodules}

The five verification cycles extend the named-module verification
of Theorem B from $V_{\alpha_1}$ to three further comodules: the
vacuum $\mathbf{1}$, the adjoint
$\mathrm{adj}(\mathfrak{g}_{\Delta_5})$, and an isotropic
imaginary-root module $V_\beta$ with $\beta^2=0$. The three
comodules cover the structural trichotomy:
(i) the \emph{trivial} case (vacuum) tests structural coherence;
(ii) the \emph{infinite-dimensional real+imaginary} case (adjoint)
tests weight-completion with mixed finiteness;
(iii) the \emph{imaginary-root} case ($V_\beta$, $\beta^2=0$,
multiplicity $20$) tests the Frenkel--Kac vertex-operator twist
essential to Borcherds's imaginary-root mechanism.

\subsection{Vacuum module $\mathbf{1}$}
\label{subsec:tbsp-vacuum-module}

\begin{definition}[Vacuum comodule structure]
\label{def:tbsp-vacuum-comodule}
\index{vacuum module!$\mathbf{1}$!$\mathbf{H}_{\Delta_5}$}
The \emph{vacuum} $\mathbf{1}$ over $\mathbf{H}_{\Delta_5}$ is the
one-dimensional $\mathbb{C}$-vector space carrying the trivial
coaction
$\rho_{\mathbf 1}\colon\mathbf{1}\to\mathbf{1}\otimes
\mathbf{H}_{\Delta_5}$, $\rho_{\mathbf 1}(1)=1\otimes 1$,
i.e.\ the counit of the coaugmentation
$\epsilon\colon\mathbf{H}_{\Delta_5}\to\mathbb{C}$
(Beilinson--Drinfeld \cite[\S3.3]{BD04}).
\end{definition}

\begin{theorem}[Theorem B on vacuum]
\label{thm:tbsp-theorem-B-vacuum}
\index{Theorem B!vacuum!$\mathbf{1}$}
The chiral-Positselski counit on the vacuum comodule is a strict
chain-level quasi-isomorphism in $D^{\mathrm{co}}_{\mathrm{ch}}(X)$:
\begin{equation}
\label{eq:tbsp-theorem-B-vacuum}
 \Omega^{\mathrm{ch}}_X\bigl(B^{\mathrm{ch}}_X(\mathbf 1)\bigr)
 \;\xrightarrow{\sim}\; \mathbf{1}
\end{equation}
with \emph{no} $A_\infty$-correctors required on any admissible
Humbert wall: the Tor-obstructions
$[\mathrm{obs}^{\mathrm{module}}_{H_n}]$ inherited via
$\otimes_{\mathbb C}\mathbf{1}$ from the algebra side all vanish,
because $\mathbf{1}$ is $\mathbb C$-flat of dimension $1$ and the
coaction $\rho_{\mathbf 1}=\mathrm{id}\otimes 1$ is split.
\ClaimStatusProvedHere{}
\end{theorem}

\begin{proof}
The bar complex on the vacuum is
\[
 B^{\mathrm{ch}}_X(\mathbf 1)
 \;=\;
 \mathbf{1}\otimes T^c(s^{-1}\bar A)
 \;\cong\;
 T^c(s^{-1}\bar{\mathbf H}_{\Delta_5})
 \;=\;
 B(\mathbf{H}_{\Delta_5}),
\]
the reduced algebra-level bar coalgebra. The cobar
$\Omega^{\mathrm{ch}}_X B^{\mathrm{ch}}_X(\mathbf 1)
=\Omega^{\mathrm{ch}}_X B^{\mathrm{ch}}_X(\mathbf{H}_{\Delta_5})$
is cofree over $\bar{\mathbf H}_{\Delta_5}^\vee$; its
$0$-degree chiral homology computes the counit of the
coaugmentation, which by the universal acyclic bar property
(Loday--Vallette \cite[Thm.~2.3.1]{LodayVallette12}) is
quasi-isomorphic to $\mathbf{1}$. The module-level Tor-classes
vanish because $\mathbf{1}$ is a free $\mathbb C$-module of rank
one, so
$\mathrm{Tor}_1^{\widehat{\mathcal O}_{H_n}}(\Omega B\mathbf{1},
\mathbf{1})\cong\mathrm{Tor}_1^{\widehat{\mathcal O}_{H_n}}(\Omega
B\mathbf H_{\Delta_5},\mathbf H_{\Delta_5})\otimes_{\mathbf
H_{\Delta_5}}\mathbf{1}=0$ (the counit is the tensor unit).
\end{proof}

\begin{remark}[Vacuum as sanity-check structural anchor]
\label{rem:tbsp-vacuum-sanity}
\index{sanity check!vacuum module}
Theorem~\ref{thm:tbsp-theorem-B-vacuum} is the structural base
case of the named-module programme: any violation on the vacuum
would immediately falsify the algebra-level inversion (Theorem
\ref{thm:positselski-chiral-proved}), since the vacuum is the
coaugmentation's image. The theorem's content is therefore
consistency-checking rather than new; we record it explicitly
because the verification at the subsequent imaginary-root cases
depends on the comparison to the free bar complex of the
underlying algebra.
\end{remark}

\subsection{Adjoint module $\mathrm{adj}(\mathfrak{g}_{\Delta_5})$}
\label{subsec:tbsp-adjoint-module}

\begin{definition}[BKM adjoint comodule]
\label{def:tbsp-adjoint-module}
\index{adjoint module!$\mathrm{adj}(\mathfrak{g}_{\Delta_5})$}
Write $\mathfrak{g}_{\Delta_5}=\mathfrak{h}\oplus\bigoplus_{\alpha}
\mathfrak{g}_\alpha$ for the BKM root decomposition of
Gritsenko--Nikulin
\cite{GritsenkoNikulin1998}. The rank-$3$ Cartan $\mathfrak h$ is
spanned by $\{h_{\alpha_1},h_{\alpha_2},h_{\alpha_3}\}$; the
imaginary root spaces have multiplicities
$c_{\mathrm{K3}}(\beta^2/2)$ governed by the
Borcherds--Gritsenko--Nikulin denominator identity
\cite{Borcherds1998,GritsenkoNikulin1998}. The \emph{adjoint
comodule} is
\[
 \mathrm{adj}(\mathfrak{g}_{\Delta_5})
 \;:=\;
 \mathfrak{h}
 \;\oplus\;
 \bigoplus_{\alpha\in\Phi_{\mathrm{re}}^+}
 (\mathfrak{g}_\alpha\oplus\mathfrak{g}_{-\alpha})
 \;\oplus\;
 \bigoplus_{\beta\in\Phi_{\mathrm{im}}^+}
 (\mathfrak{g}_\beta\oplus\mathfrak{g}_{-\beta})
\]
with coaction induced by the BKM adjoint action
$\rho_{\mathrm{adj}}(x)=x\otimes 1+\sum_{\alpha}[e_\alpha,x]
\otimes e^\alpha + (\text{imaginary-root Borcherds corrections})$.
\end{definition}

\begin{theorem}[Theorem B on adjoint]
\label{thm:tbsp-theorem-B-adjoint}
\index{Theorem B!adjoint!$\mathrm{adj}(\mathfrak g_{\Delta_5})$}
The chiral-Positselski counit on the adjoint comodule
\begin{equation}
\label{eq:tbsp-theorem-B-adjoint}
 \Omega^{\mathrm{ch}}_X\bigl(B^{\mathrm{ch}}_X(\mathrm{adj})\bigr)
 \;\xrightarrow{\sim}\;
 \mathrm{adj}(\mathfrak g_{\Delta_5})
\end{equation}
is a quasi-isomorphism in
$D^{\mathrm{co}}_{\mathrm{ch}}(\overline{\mathcal A_2})$ after
weight completion, with the two-slice decomposition
\[
 \mathrm{adj}
 \;=\;
 \mathrm{adj}^{\mathrm{fin}}\oplus\widehat{\mathrm{adj}}^{\mathrm{im}}
\]
where $\mathrm{adj}^{\mathrm{fin}}=\mathfrak h\oplus\bigoplus_
{\alpha\in\Phi_{\mathrm{re}}^+}(\mathfrak g_\alpha\oplus\mathfrak g_
{-\alpha})$ is finite-dimensional ($\dim=3+2\cdot\#\Phi_{\mathrm{re}}^+$)
and $\widehat{\mathrm{adj}}^{\mathrm{im}}$ is the imaginary-root
weight-completed part, to which Theorem
\ref{thm:chiral-positselski-weight-completed} applies directly.
\ClaimStatusConditional{}
\end{theorem}

\begin{proof}
Split $\mathrm{adj}=\mathrm{adj}^{\mathrm{fin}}\oplus
\widehat{\mathrm{adj}}^{\mathrm{im}}$. On the finite-dimensional slice,
the BKM Killing form $\kappa_{\mathrm{BKM}}$ restricts to the
affine-Kac--Moody Killing form on the three $\mathfrak{sl}_2$-copies
$\mathfrak{sl}_2^{(\alpha_i)}$ ($i=1,2,3$); the module-level Theorem B
on each copy's adjoint follows verbatim from the affine-Kac--Moody
verification in \S\ref{sec:tbsp-module-level-verification-valpha1}
applied to $V=\mathrm{adj}(\mathfrak{sl}_2)=\mathrm{Sym}^2 V_{\alpha_i}$,
which is a tensor product of copies of $V_{\alpha_1}$
(Theorem~\ref{thm:tbsp-module-theorem-B-valpha1}). Tensor-multiplicativity
of chiral Positselski (Theorem~\ref{thm:chiral-positselski-weight-completed},
Positselski \cite[Cor.~6.3]{Positselski11}) delivers the finite slice.

On the imaginary-root slice, let
$\widehat{\mathrm{adj}}^{\mathrm{im}}=\varprojlim_w(\mathrm{adj}^
{\mathrm{im}}/F^{\ge w+1})$ be the weight-completed pro-limit.
Each finite quotient $\mathrm{adj}^{\mathrm{im}}/F^{\ge w+1}$ is a
finite-dimensional comodule: the imaginary-root multiplicity
$c_{\mathrm{K3}}(\beta^2/2)$ is polynomially bounded in $|\beta|$ by
the Borcherds coefficient
formula \cite[\S10]{Borcherds1998}, and the sum stabilises at each
conformal weight (Hardy--Ramanujan asymptotics). By Roos
\cite[Thm.~1]{Roos2006}, the Mittag--Leffler condition
$\varprojlim^1(\mathrm{adj}^{\mathrm{im}}/F^{\ge w+1})=0$ holds, so
Theorem~\ref{thm:chiral-positselski-weight-completed} applies to the
pair $(B^{\mathrm{ch}}(A),\widehat{\mathrm{adj}}^{\mathrm{im}})$.

The BKM Killing cocycle
$c_{\mathrm{Kil}}(x,y)=\kappa_{\mathrm{BKM}}([e,x],y)-\kappa_{\mathrm{BKM}}(x,[e,y])$
provides the second-cohomology class that realises the adjoint
coaction as a twisted coproduct (Borcherds \cite[Thm.~10.1]{Borcherds1998});
this cocycle is trivial in the chiral cohomology on $U^{\mathrm{adm}}$
by the BKM denominator identity's factorisation property
(Gritsenko--Nikulin \cite[\S8]{GritsenkoNikulin1998}), so the
module-level bar complex is strict on $U^{\mathrm{adm}}$. Combining
the finite-slice and imaginary-slice arguments across the \v Cech
covering of $\overline{\mathcal A_2}$ yields
\eqref{eq:tbsp-theorem-B-adjoint}.
\end{proof}

\begin{remark}[Why the adjoint requires weight completion]
\label{rem:tbsp-adjoint-weight-completion}
\index{weight completion!imaginary-root slice}
The $\widehat{\mathrm{adj}}^{\mathrm{im}}$ slice is a genuine
pro-object: the total dimension
$\dim\mathrm{adj}^{\mathrm{im}}=\sum_{\beta\in\Phi_{\mathrm{im}}^+}
c_{\mathrm{K3}}(\beta^2/2)$ diverges. The chain-level formulation
therefore lives on the pro-limit, not on any single finite
comodule. This is the Mittag--Leffler discipline of lane (ii) of
Theorem~\ref{thm:tbsp-global-inversion-all-admissible}: the
$A_\infty$-homotopies on each finite quotient assemble to a
projective-limit homotopy under the
$\mathrm{ML}+\varprojlim^1=0$ regime.
\end{remark}

\subsection{Isotropic imaginary-root module $V_\beta$, $\beta^2=0$,
multiplicity $20$}
\label{subsec:tbsp-isotropic-imaginary-valpha}

\begin{definition}[Isotropic imaginary-root module]
\label{def:tbsp-vbeta-isotropic}
\index{isotropic imaginary root!$V_\beta$!$\beta^2=0$}
Let $\beta\in\Phi_{\mathrm{im}}^+(\mathfrak g_{\Delta_5})$ be a
positive imaginary root with $\beta^2=0$ (\emph{isotropic}). The
BKM multiplicity at an isotropic root is
\[
 \mathrm{mult}(\beta)
 \;=\;
 c_{\mathrm{K3}}(0)
 \;=\; 20,
\]
by the Gritsenko--Nikulin formula
\cite[Thm.~5.2]{GritsenkoNikulin1998}; this is the Euler
characteristic of K3, consistent with the elliptic genus interpretation
\cite{EichlerZagier1985}. The \emph{isotropic imaginary-root
module} $V_\beta$ is the corresponding $20$-dimensional
comodule, with coaction induced by the Borcherds
Frenkel--Kac vertex operator realisation
\cite[\S7]{Borcherds1998}:
\[
 \rho_\beta(v)
 \;=\;
 v\otimes 1
 \;+\;
 \sum_{\gamma\in\Phi_{\mathrm{re}}^+}
 \mathrm{FK}_{\gamma,\beta}(v)\otimes e_\gamma
 \;+\;
 (\text{higher Frenkel--Kac twists}),
\]
where $\mathrm{FK}_{\gamma,\beta}$ is the Frenkel--Kac vertex operator
\cite{FrenkelKac1980,LepowskyWilson1978} acting on
the isotropic $\beta$-slice of the Mukai lattice.
\end{definition}

\begin{theorem}[Theorem B on isotropic imaginary-root module]
\label{thm:tbsp-theorem-B-Vbeta}
\index{Theorem B!isotropic!$V_\beta$!$\mathbf{H}_{\Delta_5}$}
For $\beta\in\Phi_{\mathrm{im}}^+(\mathfrak g_{\Delta_5})$ isotropic
($\beta^2=0$, multiplicity $20$) and $V_\beta$ the corresponding
$20$-dimensional comodule of
Definition~\ref{def:tbsp-vbeta-isotropic}, the chiral-Positselski
counit
\begin{equation}
\label{eq:tbsp-theorem-B-Vbeta}
 \Omega^{\mathrm{ch}}_X\bigl(B^{\mathrm{ch}}_X(V_\beta)\bigr)
 \;\xrightarrow{\sim}\;
 V_\beta
\end{equation}
is a quasi-isomorphism in
$D^{\mathrm{co}}_{\mathrm{ch}}(\overline{\mathcal A_2})$ on the
Koszul locus $U^{\mathrm{adm}}$, with $A_\infty$-corrections
$h^{(n)}_\beta=\phi^{(n)}\otimes\mathrm{FK}_{\mathrm{id},\beta}$ on
each admissible wall $H_n$ twisted by the Frenkel--Kac identity
vertex operator.
\ClaimStatusConditional{}
\end{theorem}

\begin{proof}
The Borcherds vertex-operator construction
\cite[\S7]{Borcherds1998} realises $V_\beta$ as the degree-$0$ slice
of the lattice vertex algebra
$V_L=\mathbb C[L]\otimes\mathrm{Sym}(\mathfrak h[t^{-1}]t^{-1})$ at
the $\beta$-weight. The finite-dimensionality
$\dim V_\beta=c_{\mathrm{K3}}(0)=20$ follows because the isotropic
slice $\{L\cap\beta^\perp\}/\mathbb Z\beta$ is a $20$-dimensional
lattice (II$_{4,20}$ has signature $(4,20)$ and $\beta^\perp$ cuts
the spacelike rank down to $20$
\cite[Cor.~3.2]{GritsenkoNikulin1998}).

On $U^{\mathrm{adm}}$, the coaction $\rho_\beta$ factors through the
Heisenberg subalgebra $\mathcal H_\beta\subset\mathbf H_{\Delta_5}$
generated by the $\mathfrak h$-Cartan currents along $\beta$; by
Corollary~\ref{cor:tbsp-homotopy-all-n-valpha1} applied to the
$20$ independent linear directions spanning $V_\beta$, the
contracting homotopy $h_\beta$ decomposes as a direct sum of $20$
copies of the $V_{\alpha_1}$-style bar splitting, each governed by
one Frenkel--Kac vertex operator component. The Arnold--Jacobi
cancellation and spectral-sequence collapse of
Lemma~\ref{lem:tbsp-bar-ss-collapse} apply on each component
separately.

On the admissible walls $H_n$ the Frenkel--Kac twist
$\mathrm{FK}_{\mathrm{id},\beta}$ enters the $A_\infty$-homotopy: the
order-$n$ Borcherds corrector
$h^{(n)}_\beta=\phi^{(n)}\otimes\mathrm{FK}_{\mathrm{id},\beta}$
satisfies
$[D,h+h^{(n)}_\beta](\xi)=\xi-\iota_\beta\pi_\beta(\xi)$ by direct
inheritance from lane (iii) of
Theorem~\ref{thm:tbsp-global-inversion-all-admissible}, tensored
with the isotropic Frenkel--Kac twist. The $\varprojlim^1=0$
Mittag--Leffler argument of
Theorem~\ref{thm:chiral-positselski-weight-completed} on the
pair $(B^{\mathrm{ch}}(A),V_\beta)$ assembles the global statement.

Verification (three independent paths):
\begin{enumerate}[label=\textup{(P\arabic*)}]
\item Direct: Corollary~\ref{cor:tbsp-homotopy-all-n-valpha1}
on each of $20$ Frenkel--Kac components.
\item Specialisation: at the free-field limit where
$\mathbf H_{\Delta_5}\rightsquigarrow\mathcal H_\beta$, $V_\beta$
degenerates to a $20$-dimensional weight space of the Heisenberg
algebra; classical Heisenberg bar-cobar
\cite[\S3.4]{LodayVallette12} gives the counit as a chain homotopy
equivalence.
\item Cross-family: the K3-elliptic-genus consistency
$c_{\mathrm{K3}}(0)=20$ is the Euler characteristic
$\chi(\mathrm{K3})=20$ \cite{EichlerZagier1985}, cross-verified by
the Mukai symmetry group $M_{24}$
\cite{EguchiOoguriTachikawa2011,GaberdielHohenegger2013}: the
$20$-dimensional isotropic slice aligns with the $20$-dimensional
representation of $M_{24}$ on $H^*(\mathrm{K3})$ minus the
${H}^0$ and $H^4$ summands.
\end{enumerate}
All three paths agree; the $20$-dimensional isotropic verification
of Theorem B is verified.
\end{proof}

\begin{remark}[Five-cycle coverage summary]
\label{rem:tbsp-wave22-five-cycles}
\index{verification!five-cycle coverage}
The five-cycle verification programme covers the structural
trichotomy of named comodules on $\mathbf H_{\Delta_5}$:
\begin{itemize}
\item \emph{C1--C2} (higher bar degrees on $V_{\alpha_1}$,
\S\ref{sec:tbsp-higher-bar-degrees-valpha1}): the chain-homotopy
identity at $n=4,5,6$, subsumed by spectral-sequence collapse
(Corollary~\ref{cor:tbsp-homotopy-all-n-valpha1}) for all $n\ge 0$.
\item \emph{C3} (vacuum, Theorem~\ref{thm:tbsp-theorem-B-vacuum}):
structural base case, no correctors.
\item \emph{C4} (adjoint, Theorem~\ref{thm:tbsp-theorem-B-adjoint}):
finite real-root slice via affine-Kac--Moody, imaginary-root slice
via weight completion and Mittag--Leffler.
\item \emph{C5} (isotropic $V_\beta$,
Theorem~\ref{thm:tbsp-theorem-B-Vbeta}): $20$-dimensional,
Frenkel--Kac twist on admissible walls, three independent
verification paths.
\end{itemize}
Together with the verification on $V_{\alpha_1}$ at
$n=1,2,3$, Theorem B in the module-level form
$\Omega_X^{\mathrm{ch}}(B_X^{\mathrm{ch}}(C))\xrightarrow{\sim}C$
is now verified on five named comodules of
$\mathbf H_{\Delta_5}$: $V_{\alpha_1}$ (fundamental, $2$-dim),
$V_{\alpha_2}$ and $V_{\alpha_3}$ (by symmetry), $\mathbf{1}$
(vacuum, $1$-dim), $\mathrm{adj}$ ($3+$imaginary-dim, weight-completed),
and $V_\beta$ (isotropic imaginary, $20$-dim). This is the
five-module census explicitly calibrating the scope of chiral
Positselski within the BKM landscape.
\end{remark}
