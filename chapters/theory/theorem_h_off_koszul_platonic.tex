\chapter{Theorem H boundary strata: Koszul locus, singular quotients, and critical level}
\label{chap:theorem-h-off-koszul-platonic}

Theorem~H is a theorem about the curve-level chiral Hochschild complex
\[
 RHH_{\mathrm{ch}}(\cA)
 \;=\;
 R\mathrm{Hom}_{\cA^e}(\cA,\cA)
\]
of a chiral algebra on a fixed smooth projective reference curve~$X$.
It is not a theorem about topological Hochschild homology of an
$E_1$-ring spectrum, categorical Hochschild cochains of a dg or
Calabi--Yau category, the Hochschild theory of a formal-disk mode
algebra, or a product-ambient Hall Hochschild theory. The comparison
convention is
Convention~\ref{conv:three-hochschild}; the local construction is
Remark~\ref{rem:three-hochschild-discipline}.
The coHochschild statement attached to Theorem~H is not a second
ambient: it is the transport of this chiral Hochschild complex across
the complete bimodule--bicomodule equivalence of
Theorem~\ref{thm:ordered-HH-coHH-cohomology}, with
$C=\barB^{\mathrm{ch}}(\cA)$ and diagonal bicomodule~$C_\Delta$.
No concentration statement is asserted for an arbitrary coalgebra
outside that equivalence.

On the PBW chiral Koszul locus, with the $E_\infty$-chiral completion,
finite conformal-weight windows, Mittag--Leffler completion,
perfectness, and genericity hypotheses of
Theorem~\ref{thm:main-koszul-hoch}, the direct-sum totalization in bar
degree is formed before weight completion and the collision-depth
spectral sequence on ordered Fulton--MacPherson compactifications
collapses by the pure-braid Orlik--Solomon contraction. The result is
the sharp curve-level amplitude
\[
 \ChirHoch^n(\cA)=0\qquad(n\notin\{0,1,2\})
\]
and the Hilbert polynomial
\[
 P_{\cA}(t,q)
 \;=\;
 \HS_{Z(\cA)}(q)
 \;+\;
 \HS_{\ChirHoch^{1}(\cA)}(q)t
 \;+\;
 \HS_{Z(\cA^{!})}(q)t^{2}.
\]
This is a Hilbert polynomial in the cohomological variable~$t$ with
finite $q$-graded coefficients. It is not a Hilbert series for a
topological Hochschild spectrum, a categorical Hochschild complex of a
Calabi--Yau category, or a completed product totalization. Every
boundary in this chapter is measured against that theorem.

\section{Object firewall}
\label{sec:theorem-h-object-firewall}

The objects entering Theorem~H have different types:
\[
\begin{array}{c|c}
\text{object} & \text{role}\\
\hline
\cA & \text{input chiral algebra on }X\\
\barB^{\mathrm{ch}}(\cA) & \text{bar coalgebra}\\
\Omega^{\mathrm{ch}}\barB^{\mathrm{ch}}(\cA) &
 \text{bar--cobar resolution of }\cA\\
\cA^{i}=H^*(\barB^{\mathrm{ch}}(\cA)) &
 \text{Koszul coalgebra homology}\\
\cA^!=\mathbb D_X(\cA^i) &
 \text{Verdier-dual Koszul algebra}\\
RHH_{\mathrm{ch}}(\cA) &
 \text{chiral Hochschild cochain complex}\\
Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)=RHH_{\mathrm{ch}}(\cA) &
 \text{derived chiral centre, chain-level closed-sector object}\\
\mathrm{Coext}_{C^e}^{\bullet}(C_\Delta,C_\Delta) &
 \text{coHochschild self-extension, after }C=\barB^{\mathrm{ch}}(\cA)
\end{array}
\]
The bar--cobar counit
\[
 \Omega^{\mathrm{ch}}\barB^{\mathrm{ch}}(\cA)\longrightarrow \cA
\]
is an inversion or resolution map. It is not the definition of the
Koszul dual algebra, and its cone is not the Theorem~H defect. The
Hochschild amplitude bound is lost precisely when the positive
collision-depth part of the Fulton--MacPherson Hochschild complex is no
longer killed by the PBW/Arnold contraction.
All quotients and subcomplexes below are taken in the conformal-weight
completed holonomic $\mathcal D_X$-module derived category. Thus
$F^1_{\mathrm{FM}}$ denotes the closed positive-depth term in the
completed collision-depth filtration, and the depth-zero quotient is a
completed quotient. Without this Hausdorff completion the infinite
standard families have the wrong ambient topology.
The equality
$Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)=RHH_{\mathrm{ch}}(\cA)$ names the
chiral derived centre of the boundary chiral algebra. It is not the
topological bulk of a $3$-dimensional TQFT, not the categorical
Hochschild cochains of a dg or Calabi--Yau category, and not a
product-ambient Hall Hochschild theory unless a separate open--closed
comparison theorem changes both the input and the target category.

\begin{definition}[Hochschild Koszul-defect complex]
\label{def:koszul-defect-complex}
\index{Koszul defect complex}
\index{chiral Hochschild!Koszul defect}
Let $F^\bullet_{\mathrm{FM}}RHH_{\mathrm{ch}}(\cA)$ be the complete
collision-depth filtration of the bigraded chiral Hochschild complex
used in Lemma~\ref{lem:hochschild-shift-computation} and
Theorem~\ref{thm:hochschild-concentration-E1}. The
\emph{Hochschild Koszul-defect complex} is the closed positive-depth
subcomplex
\[
 \mathrm{KD}_{\mathrm H}^{\bullet}(\cA)
 \;:=\;
 F^1_{\mathrm{FM}}RHH_{\mathrm{ch}}(\cA).
\]
Equivalently, it is the part of the chiral Hochschild complex supported
on non-trivial collision strata. It measures failure of the
Fulton--MacPherson/PBW collapse used in Theorem~H, not failure of the
bar--cobar counit.
\end{definition}

\begin{definition}[Hochschild Koszul-defect measure]
\label{def:koszul-defect-measure}
\index{Koszul defect measure}
For finite conformal-weight pieces set
\[
 \delta_{\mathrm H}^{(n)}(\cA)(q)
 \;:=\;
 \dim_q H^{n}\!\bigl(\mathrm{KD}_{\mathrm H}^{\bullet}(\cA)\bigr).
\]
The series
\[
 \Delta_{\mathrm H}(\cA)(t,q)
 \;:=\;
 \sum_{n\geq 0}\delta_{\mathrm H}^{(n)}(\cA)(q)t^n
\]
is defined only when the displayed graded dimensions are finite. On the
PBW chiral Koszul locus the positive-depth complex is acyclic in the
range that can contribute to total degree $\geq3$.
\end{definition}

\begin{remark}[Why this is the right defect measure]
\label{rem:koszul-defect-measure-right}
Theorem~H is not a formal consequence of the existence of the
bar--cobar resolution. The proof uses the transported
Shelton--Yuzvinsky contraction on positive Arnold classes in the
collision-depth spectral sequence. The complex
$\mathrm{KD}_{\mathrm H}^{\bullet}(\cA)$ is exactly the part of
$RHH_{\mathrm{ch}}(\cA)$ on which that contraction must act. If it has
cohomology in high total degree, no curve-level amplitude bound follows
from Theorem~H.
\end{remark}

\begin{definition}[Collision obstruction pages]
\label{def:theorem-h-collision-obstruction-pages}
\index{Theorem H!collision obstruction}
Let
\[
 E_1^{r,s}(\cA)
 =
 H^{r+s}\!\left(
  \operatorname{gr}^{\,r}_{F_{\mathrm{FM}}}
  RHH_{\mathrm{ch}}(\cA)
 \right)
\]
be the collision-depth spectral sequence of
$RHH_{\mathrm{ch}}(\cA)$, with $r$ the collision depth. The first
Hochschild collision obstruction is
\[
 \mathfrak{o}_{\mathrm H,1}^{\ge3}(\cA)
 :=
 \bigoplus_{\substack{r\geq1\\ r+s\geq3}}
 E_2^{r,s}(\cA),
\]
where $d_1$ is the residue differential along the first collision
divisors. The permanent obstruction is
\[
 \mathfrak{o}_{\mathrm H,\infty}^{\ge3}(\cA)
 :=
 \bigoplus_{\substack{r\geq1\\ r+s\geq3}}
 E_\infty^{r,s}(\cA)
 \subset
 \operatorname{gr}_{F_{\mathrm{FM}}}
 H^{\geq3}\!\bigl(RHH_{\mathrm{ch}}(\cA)\bigr).
\]
On the PBW chiral Koszul locus the transported pure-braid
Shelton--Yuzvinsky contraction makes
$\mathfrak{o}_{\mathrm H,1}^{\ge3}(\cA)=0$. Off that locus, higher
differentials can kill some $E_2$ classes. The exact amplitude
obstruction is the permanent part together with the degree-$3$ boundary
described in Theorem~\ref{thm:theorem-h-off-koszul-explicit-correction}.
\end{definition}

\section{Theorem H on the Koszul locus}
\label{sec:theorem-h-on-koszul}

\begin{theorem}[Theorem~H on the Koszul locus; sharp bigraded Hilbert series;
\ClaimStatusConditional]
\label{thm:theorem-h-on-koszul-locus}
\index{Theorem H!on Koszul locus}
Let $\cA$ be a PBW chiral Koszul algebra on a smooth projective curve
$X$, satisfying the PBW criterion of
Theorem~\ref{thm:pbw-koszulness-criterion}, the
$E_\infty$-chiral completion hypothesis of
Corollary~\ref{cor:hochschild-averaging-symmetric}, and the
finite-type/perfectness hypotheses of
Theorem~\ref{thm:main-koszul-hoch}. Assume that $\cA$ lies away from
critical affine levels and away from the non-generic simple-quotient
points excluded in Theorem~\ref{thm:main-koszul-hoch}. Then
\[
 \ChirHoch^n(\cA)=0\qquad(n\notin\{0,1,2\})
\]
and
\begin{equation}
\label{eq:theorem-h-on-koszul}
 P_{\cA}(t,q)
 \;=\;
 \HS_{Z(\cA)}(q)
 \;+\;
 \HS_{\ChirHoch^{1}(\cA)}(q)t
 \;+\;
 \HS_{Z(\cA^{!})}(q)t^{2}.
\end{equation}
Here
\[
 P_{\cA}(t,q)=\sum_{n\geq0}\dim_q \ChirHoch^n(\cA)\,t^n
\]
is computed after passing through finite conformal-weight windows and
then taking the strict Mittag--Leffler completion. The polynomial
assertion is exactly the vanishing of the positive collision-depth tail
after the PBW/Arnold contraction.
The $t^2$ column is the Verdier-dual chiral centre
$Z(\cA^!)^\vee\otimes\omega_X$ and has the same finite
$q$-graded coefficients as $Z(\cA^!)$.
\end{theorem}

\begin{proof}
The ordered concentration is
Theorem~\ref{thm:hochschild-concentration-E1}: the ordered
Fulton--MacPherson collision-depth spectral sequence kills all
positive Arnold fibre classes by the pure-braid
Shelton--Yuzvinsky contraction transported through the chiral
quadratic-Koszul comparison. The $E_\infty$-chiral completion
hypothesis lets Corollary~\ref{cor:hochschild-averaging-symmetric}
descend the ordered vanishing to the symmetric chiral Hochschild
complex. Proposition~\ref{prop:chirhoch-sharp-hilbert} identifies the
three surviving columns, and Theorem~\ref{thm:main-koszul-hoch}
supplies the Verdier shift by~$2$.
\end{proof}

\begin{remark}[Amplitude source]
\label{rem:three-columns-verdier}
The amplitude $[0,2]$ is the sum of two facts: after the PBW/Arnold
collapse, every bar degree is reduced to a holonomic
$\mathcal D_X$-Ext computation on a curve, and the geometric
$[p+2]$ shift on $\overline C_{p+2}(X)$ cancels the totalization
$[-p]$ shift. No topological Hochschild, categorical Hochschild, or
CY product-ambient Hochschild statement is used.
\end{remark}

\section{Off the Koszul locus}
\label{sec:theorem-h-off-koszul}

Let
\[
 Q_{\mathrm H}(\cA)
 \;:=\;
 RHH_{\mathrm{ch}}(\cA)/
 F^1_{\mathrm{FM}}RHH_{\mathrm{ch}}(\cA)
\]
be the depth-zero quotient. It is the curve-level piece left after
discarding non-trivial collision strata. Under the holonomic
finite-type hypotheses stated in
Theorem~\ref{thm:theorem-h-off-koszul-explicit-correction}, it has
cohomological amplitude~$[0,2]$ exactly when the depth-zero calculation
is a curve-level $\mathcal D_X$-Ext calculation. Off the Koszul locus it
need not identify its $t^2$ column with
$Z(\cA^!)$; that identification is part of Theorem~H and uses the
Koszul/Verdier comparison.

\begin{theorem}[Off-Koszul high-degree exact sequence;
\ClaimStatusProvedHere]
\label{thm:theorem-h-off-koszul-explicit-correction}
\index{Theorem H!off Koszul locus}
Let $\cA$ be a conformal-weight-graded chiral algebra on a smooth
projective curve, with finite graded pieces, for which the
completed Fulton--MacPherson chiral Hochschild filtration is defined,
exhaustive, and Hausdorff in each conformal weight, and suppose that
$Q_{\mathrm H}(\cA)$ has amplitude~$[0,2]$. Then the exact triangle
\[
 \mathrm{KD}_{\mathrm H}^{\bullet}(\cA)
 \longrightarrow
 RHH_{\mathrm{ch}}(\cA)
 \longrightarrow
 Q_{\mathrm H}(\cA)
 \xrightarrow{+1}
\]
gives
\[
 H^2(Q_{\mathrm H}(\cA))
 \xrightarrow{\;\partial_{\mathrm H}^{3}(\cA)\;}
 H^3(\mathrm{KD}_{\mathrm H}^{\bullet}(\cA))
 \longrightarrow
 \ChirHoch^3(\cA)
 \longrightarrow 0
\]
and hence an exact degree-$3$ obstruction
\[
 \mathfrak{o}_{\mathrm H}^{3}(\cA)
 :=
 \operatorname{coker}\!\left(
  \partial_{\mathrm H}^{3}(\cA)\colon
  H^2(Q_{\mathrm H}(\cA))
  \longrightarrow
  H^3(\mathrm{KD}_{\mathrm H}^{\bullet}(\cA))
 \right)
 \cong
 \ChirHoch^3(\cA),
\]
and, for every $n\geq4$,
\[
 \ChirHoch^n(\cA)
 \;\cong\;
 H^n(\mathrm{KD}_{\mathrm H}^{\bullet}(\cA)).
\]
Consequently $\ChirHoch^\bullet(\cA)$ has amplitude contained in
$[0,N]$ for $N\geq3$ if
$H^n(\mathrm{KD}_{\mathrm H}^{\bullet}(\cA))=0$ for $n>N$. It is
concentrated in $\{0,1,2\}$ if and only if
$\mathfrak{o}_{\mathrm H}^{3}(\cA)=0$ and
$H^n(\mathrm{KD}_{\mathrm H}^{\bullet}(\cA))=0$ for every $n\geq4$.
\end{theorem}

\begin{proof}
The definition of $\mathrm{KD}_{\mathrm H}^{\bullet}$ gives the
displayed exact triangle. The long exact cohomology sequence contains
\[
H^2(Q_{\mathrm H})\to H^3(\mathrm{KD}_{\mathrm H})
\to H^3(RHH_{\mathrm{ch}})\to H^3(Q_{\mathrm H}),
\]
and $H^3(Q_{\mathrm H})=0$ by the amplitude hypothesis. For
$n\geq4$ both $H^{n-1}(Q_{\mathrm H})$ and $H^n(Q_{\mathrm H})$
vanish, so the map
$H^n(\mathrm{KD}_{\mathrm H})\to H^n(RHH_{\mathrm{ch}})$ is an
isomorphism. Since
$H^n(RHH_{\mathrm{ch}})=\ChirHoch^n(\cA)$, the amplitude statements
follow. The cokernel displayed in degree~$3$ is exactly the first
permanent collision class remaining after the depth-zero curve piece
has absorbed its boundary.
\end{proof}

\begin{lemma}[Degree-three boundary obstruction; \ClaimStatusConditional]
\label{lem:theorem-h-degree-three-boundary}
\index{Theorem H!degree-three boundary obstruction}
In the hypotheses of
Theorem~\ref{thm:theorem-h-off-koszul-explicit-correction}, every class
of $\ChirHoch^3(\cA)$ is the image of a positive collision-depth class
modulo the boundary
$\partial_{\mathrm H}^{3}(\cA)\colon H^2(Q_{\mathrm H}(\cA))\to
H^3(\mathrm{KD}_{\mathrm H}^{\bullet}(\cA))$. Hence a nonzero
$\ChirHoch^3(\cA)$ is not a fourth column of Theorem~H; it is the
off-Koszul obstruction
$\mathfrak{o}_{\mathrm H}^{3}(\cA)$.
If $\cA$ satisfies the hypotheses of
Theorem~\ref{thm:theorem-h-on-koszul-locus}, then
$\mathrm{KD}_{\mathrm H}^{\bullet}(\cA)$ is acyclic in total degree
$\geq3$, $\mathfrak{o}_{\mathrm H}^{3}(\cA)=0$, and
$\ChirHoch^3(\cA)=0$.
\end{lemma}

\begin{proof}
The first assertion is the cokernel identification in
Theorem~\ref{thm:theorem-h-off-koszul-explicit-correction}. The second
is Theorem~\ref{thm:theorem-h-on-koszul-locus}, equivalently the
vanishing of the positive-depth obstruction complex after the
PBW/Arnold contraction.
\end{proof}

\begin{corollary}[Exact defect Hilbert tail; \ClaimStatusProvedHere]
\label{cor:concentration-iff-defect-zero}
In the situation of
Theorem~\ref{thm:theorem-h-off-koszul-explicit-correction}, assume
that all displayed conformal-weight graded dimensions are finite and
that the depth-zero quotient identifies, in degrees $0,1,2$, with the
three Theorem~H columns $Z(\cA)$, $\ChirHoch^1(\cA)$, and
$Z(\cA^!)^\vee\otimes\omega_X$. Then
\[
 P_{\cA}(t,q)
 =
 \HS_{Z(\cA)}(q)
 +\HS_{\ChirHoch^1(\cA)}(q)t
 +\HS_{Z(\cA^!)}(q)t^2
 +\dim_q\mathfrak{o}_{\mathrm H}^{3}(\cA)t^3
 +\sum_{n\geq4}
  \dim_q H^n(\mathrm{KD}_{\mathrm H}^{\bullet}(\cA))\,t^n .
\]
If $\partial_{\mathrm H}^{3}(\cA)=0$, the $t^3$ coefficient is
$\dim_q H^3(\mathrm{KD}_{\mathrm H}^{\bullet}(\cA))$; otherwise it is
the cokernel dimension of the boundary map. Without the low-column
Verdier identification, the same exact statement gives only the tail
in degrees $\geq3$.
\end{corollary}

\begin{proof}
Theorem~\ref{thm:theorem-h-off-koszul-explicit-correction} identifies
$\ChirHoch^3(\cA)$ with the cokernel
$\mathfrak{o}_{\mathrm H}^{3}(\cA)$ and identifies
$\ChirHoch^n(\cA)$ with
$H^n(\mathrm{KD}_{\mathrm H}^{\bullet}(\cA))$ for $n\geq4$. The
low-column hypothesis supplies the coefficients in degrees $0,1,2$.
\end{proof}

\begin{corollary}[coHochschild transport; \ClaimStatusConditional]
\label{cor:theorem-h-cohochschild-transport}
\index{coHochschild!Theorem H transport}
Let $C=\barB^{\mathrm{ch}}(\cA)$, and assume the complete
bimodule--bicomodule equivalence hypotheses of
Theorem~\ref{thm:ordered-HH-coHH-cohomology}. If $\cA$ satisfies
Theorem~\ref{thm:theorem-h-on-koszul-locus}, then
\[
 \mathrm{Coext}_{C^e}^{n}(C_\Delta,C_\Delta)=0
 \qquad (n\notin\{0,1,2\}).
\]
Off the Koszul locus, the same identification transports the exact
sequence of
Theorem~\ref{thm:theorem-h-off-koszul-explicit-correction} to the
coHochschild self-extension complex. Without the equivalence, there is
no Theorem~H statement for coHochschild theory.
\end{corollary}

\begin{proof}
Theorem~\ref{thm:ordered-HH-coHH-cohomology} identifies
$\ChirHoch^\bullet(\cA)$ with
$\mathrm{Coext}_{C^e}^{\bullet}(C_\Delta,C_\Delta)$ in the complete
ordered bar ambient. The concentration and the off-locus exact
sequence are therefore transported statements, not new vanishing
theorems for a coalgebra outside the Koszul bar image.
\end{proof}

\section{Logarithmic and admissible singular quotients}
\label{sec:theorem-h-logarithmic-wp}

\begin{theorem}[Triplet logarithmic boundary; \ClaimStatusConditional]
\label{thm:theorem-h-logarithmic-w(p)}
\index{Theorem H!logarithmic W(p) boundary}
\index{triplet algebra!chiral Hochschild}
Let $p\geq2$ and let $\cW(p)$ be the triplet algebra of
Definition~\ref{def:wp-algebra}. The known local facts are:
\[
 c(\cW(p))=1-\frac{6(p-1)^2}{p},\qquad
 p_{\max}(\cW(p))=2(2p-1),\qquad
 r_{\max}(\cW(p))=\infty,
\]
with $p_{\max}$ and $r_{\max}$ as in
Remark~\ref{rem:wp-pole-depth}. The algebra is $C_2$-cofinite
\textup{(}Proposition~\ref{prop:wp-c2-cofinite}\textup{)}, is not
freely strongly generated
\textup{(}Proposition~\ref{prop:wp-not-free-strong}\textup{)}, and
its chiral Koszulness is Conjecture~\ref{conj:wp-koszulness}.

Therefore the PBW/free proof of Theorem~H does not apply to
$\cW(p)$. If a non-free route proves for $\cW(p)$ the PBW chiral
Koszul, $E_\infty$-completion, finite-window perfectness, and averaging
hypotheses of Theorem~\ref{thm:theorem-h-on-koszul-locus}, Theorem~H
applies and $\ChirHoch^\bullet(\cW(p))$ has amplitude~$[0,2]$.
Without those hypotheses, its high-degree chiral Hochschild groups are
governed by the exact sequence of
Theorem~\ref{thm:theorem-h-off-koszul-explicit-correction}; no
explicit $t^3$ coefficient follows from $C_2$-cofiniteness or from the
maximal $W$-$W$ pole order alone.

Let
$E_{2,W\text{-}W}^{r,s}(\cW(p))\subset E_2^{r,s}(\cW(p))$ be the
image on the $E_2$-page of the local span of collision strata whose
OPE input is a $W$-$W$ pole of order $2(2p-1)$. Let
$E_{\infty,W\text{-}W}^{r,s}(\cW(p))$ denote the subquotient of
$E_\infty^{r,s}(\cW(p))$ represented by classes in that $E_2$-subspace
which survive all higher differentials. Under the amplitude hypothesis
on $Q_{\mathrm H}(\cW(p))$, the exact triplet obstruction component is
\[
 \mathfrak{o}_{\mathrm H,W}^{3}(\cW(p))
 :=
 \operatorname{im}\!\left(
  \bigoplus_{\substack{r\geq1\\ r+s=3}}
  E_{\infty,W\text{-}W}^{r,s}(\cW(p))
  \longrightarrow
  \mathfrak{o}_{\mathrm H}^{3}(\cW(p))
 \right).
\]
The remaining condition is the differential calculation deciding the map
$E_{2,W\text{-}W}^{r,s}(\cW(p))\to
E_{\infty,W\text{-}W}^{r,s}(\cW(p))$.
\end{theorem}

\begin{proof}
The displayed central charge and pole data are the defining data and
pole-depth computation in Chapter~\ref{chap:logarithmic-w-algebras}.
$C_2$-cofiniteness and non-free strong generation are exactly
Propositions~\ref{prop:wp-c2-cofinite} and
\ref{prop:wp-not-free-strong}. Theorem~H uses the PBW chiral Koszul
input and the transported Arnold contraction; $C_2$-cofiniteness is a
finite associated-variety condition and does not supply that
contraction. The final assertion is then the high-degree exact
sequence of Theorem~\ref{thm:theorem-h-off-koszul-explicit-correction}.
The pole order fixes the local $W$-$W$ summand on the first collision
divisor; it does not compute the higher differentials. The definition
of $\mathfrak{o}_{\mathrm H,W}^{3}$ records the permanent component
which must vanish as part of the Theorem~H obstruction for the triplet
algebra.
\end{proof}

\begin{theorem}[Admissible and minimal simple-quotient firewall;
\ClaimStatusConditional]
\label{thm:theorem-h-singular-quotient-firewall}
\index{Theorem H!admissible quotient boundary}
\index{Theorem H!minimal model boundary}
Theorem~H does not pass from a universal affine or principal
$\mathcal W$-algebra to a non-generic simple quotient by rationality,
$C_2$-cofiniteness, or finite ordinary module category alone.
For affine admissible quotients $L_k(\fg)$, the ordinary admissible
category controls Zhu-theoretic and conformal-block finiteness, but
the chiral bar-complex Ext theory is a separate factorization
calculation
\textup{(}Remark~\ref{rem:admissible-rationality-finiteness}\textup{)}.
For $\fg=\mathfrak{sl}_2$ chiral Koszulness at admissible level is
recorded in Remark~\ref{rem:admissible-koszul-status}; the Theorem~H
conclusion still requires the completion, perfectness, and averaging
hypotheses of Theorem~\ref{thm:theorem-h-on-koszul-locus}. For
$\mathfrak{sl}_3$ at non-degenerate admissible level
$k=-3+p/q$ with coprime $p\geq3$, $q\geq3$, assume the finite Li--bar
package of Theorem~\ref{thm:admissible-sl3-non-koszul-qge3}, in which
the reduced bar-relevant model contains the Cartan factors
\[
 \mathbb C[H_1]/(H_1^{d_C})\otimes
 \mathbb C[H_2]/(H_2^{d_C}),
 \qquad d_C=(p-1)q,
\]
and the root factors have exponent $d_R=(p-2)q$. Assume also a
convergent completed FM Hochschild filtration, the amplitude hypothesis
$Q_{\mathrm H}(L_k(\mathfrak{sl}_3))\in \mathrm D^{[0,2]}$, and a comparison map
from the quotient Li--bar page to the FM Hochschild collision page,
\[
 \Phi_{\mathrm H}^{\mathrm{adm}}\colon
 E_2^{2,*}\bigl(\barB(L_k(\mathfrak{sl}_3))\bigr)
 \longrightarrow
 \bigoplus_{r+s\geq3}
 E_2^{r,s}\!\left(RHH_{\mathrm{ch}}(L_k(\mathfrak{sl}_3))\right),
\]
compatible with the residue differential $d_1$. The Cartan classes
\[
 [H_1], [H_2]\in E_2^{2,*}\bigl(\barB(L_k(\mathfrak{sl}_3))\bigr),
 \qquad [\mathfrak h,\mathfrak h]=0,
\]
then define the admissible Hochschild obstruction
\[
 \mathfrak{o}_{\mathrm H,\mathrm{Cartan}}^{\ge3}(L_k(\mathfrak{sl}_3))
 :=
 \operatorname{im}\!\left(
  \Phi_{\mathrm H}^{\mathrm{adm}}\!
  \left(\operatorname{span}\{[H_1],[H_2]\}\right)
  \longrightarrow
  \mathfrak{o}_{\mathrm H,1}^{\ge3}(L_k(\mathfrak{sl}_3))
 \right),
\]
and its degree-$3$ permanent component
$\mathfrak{o}_{\mathrm H,\mathrm{Cartan}}^{3}$ is the image after
survival to $E_\infty$ followed by the edge map of
Theorem~\ref{thm:theorem-h-off-koszul-explicit-correction}. This is a
named component of $\mathfrak{o}_{\mathrm H}^{3}$, not the whole
obstruction unless the comparison proves that no other positive-depth
classes survive. The exact high-degree Theorem~H obligation for this
admissible quotient is
\[
 \mathfrak{o}_{\mathrm H}^{3}(L_k(\mathfrak{sl}_3))=0,\qquad
 H^n(\mathrm{KD}_{\mathrm H}^{\bullet}(L_k(\mathfrak{sl}_3)))=0
 \quad(n\geq4),
\]
with the low-degree Verdier-centre identification added if one wants
the Hilbert polynomial \eqref{eq:theorem-h-on-koszul}. For
$\mathrm{rk}(\fg)\geq3$ the analogous rank obstruction is
Conjecture~\ref{conj:admissible-koszul-rank-obstruction}. For
Virasoro minimal models and principal $\mathcal W$-algebras at
admissible levels, the concentration problem is the non-generic
class-\(\mathsf M\) problem of
Conjecture~\ref{conj:hochschild-concentration-class-M-non-generic}.
\end{theorem}

\begin{proof}
The cited admissible-level analysis separates ordinary representation
theory from the chiral/factorization bar complex. Theorem~H requires
the latter: PBW chiral Koszulness, completion, perfectness, and
averaging. Rationality and $C_2$-cofiniteness imply finite ordinary
data but do not identify the positive collision-depth Hochschild
defect complex with zero. The rank-one and rank-two affine examples
show that the issue is mathematical rather than terminological:
rank~$1$ absorbs the admissible null vector in the quotient bar
spectral sequence, while the rank~$2$ Cartan classes survive the
$d_1$ page because the Cartan bracket vanishes. The map
$\Phi_{\mathrm H}^{\mathrm{adm}}$ is precisely the missing bridge from
bar non-Koszulness to a Hochschild amplitude obstruction. The Cartan
image is decisive only after the comparison excludes all other
positive-depth survivors. The
class-\(\mathsf M\)
statement in
Conjecture~\ref{conj:hochschild-concentration-class-M-non-generic}
records the remaining Virasoro and principal $\mathcal W$ simple
quotient problem.
\end{proof}

\section{Critical level}
\label{sec:theorem-h-critical-level}

\begin{theorem}[Critical-level centre and amplitude obstruction;
\ClaimStatusConditional]
\label{thm:theorem-h-at-critical-level}
\index{Theorem H!critical level}
\index{Feigin--Frenkel centre!Theorem H}
Let $V_{-h^\vee}(\fg)$ be the affine Kac--Moody vacuum module at
critical level for a simple Lie algebra $\fg$ of rank $r$ and
exponents $d_1,\ldots,d_r$. Its chiral centre is the Feigin--Frenkel
centre:
\[
 Z(V_{-h^\vee}(\fg))
 \;\cong\;
 \mathfrak z(\widehat{\fg})
 \;\cong\;
 \mathrm{Fun}(\mathrm{Op}_{{}^L\fg}(D)).
\]
Equivalently, as a graded commutative differential algebra it is freely
generated by the Segal--Sugawara generators $S_i$ of conformal weights
$d_i+1$ and their derivatives, so
\begin{equation}
\label{eq:ff-center-hilbert}
 \HS_{\mathfrak z(\widehat{\fg})}(q)
 =
 \prod_{i=1}^{r}\prod_{m\geq0}
 \frac{1}{1-q^{d_i+1+m}}.
\end{equation}
This is a degree-zero chiral-centre statement. A full bigraded Hilbert
series for $RHH_{\mathrm{ch}}(V_{-h^\vee}(\fg))$ requires a chain-level
comparison between the critical chiral Hochschild complex and the
Beilinson--Drinfeld/Feigin--Tsygan oper complex. The Feigin--Frenkel
centre alone does not supply such a formula.

The critical-first affine bar perturbation is
\[
 d_k=d_{\mathrm{crit}}+(k+h^\vee)\delta,\qquad
 d_{\mathrm{crit}}\delta+\delta d_{\mathrm{crit}}
 =
 \left[\frac{\mathsf C_{\fg}}{2h^\vee},-\right],
\]
so the obstruction to transporting the generic FM collapse across the
critical divisor is the curvature action
$[(k+h^\vee)\mathsf C_{\fg}/(2h^\vee),-]$ of
Corollary~\ref{cor:critical-level-spectral}. Equivalently, the residue
class
\[
 \mathfrak c_{\mathrm{crit}}(\fg)
 :=
 \left[\frac{\mathsf C_{\fg}}{2h^\vee},-\right]
\]
is the named obstruction to a pole-free continuation of the generic
contracting homotopy through $k=-h^\vee$. Under the separate
Beilinson--Drinfeld/Feigin--Tsygan comparison recorded in
Remark~\ref{rem:critical-level-lie-vs-chirhoch}, the continuous
Lie-cohomological factor in the critical chiral Hochschild comparison is
\[
 H^\bullet_{\mathrm{Lie,cont}}(\fg\otimes t\mathbb C[[t]];\mathbb C)
 \cong
 \Lambda(P_1,\ldots,P_r)\otimes
 \mathbb C[\Theta_1,\ldots,\Theta_r],
 \qquad
 \deg P_i=2d_i+1,\quad
 \deg\Theta_i=2(d_i+1),
\]
with the degree-zero Feigin--Frenkel centre acting as the chiral centre
when the central action is retained. Its cohomological Hilbert--Poincare
series is
\[
 \operatorname{HP}_{\mathrm{crit}}(u)
 =
 \prod_{i=1}^{r}
 \frac{1+u^{2d_i+1}}{1-u^{2(d_i+1)}}.
\]
For every $N\geq1$, the monomial $\Theta_i^N$ has cohomological degree
$2N(d_i+1)$. Thus the critical comparison has no finite cohomological
amplitude. This is a critical-level comparison theorem, not a
specialization of Theorem~H.
\end{theorem}

\begin{proof}
The Feigin--Frenkel theorem identifies the centre of the critical
affine vertex algebra with functions on local opers for the Langlands
dual Lie algebra; see Theorem~\ref{thm:feigin-frenkel-center} and the
formal-disc comparison in Chapter~\ref{chap:three-hochschild-unification}.
The generators have degrees $d_i+1$, and applying all derivatives
$\partial^m$ gives the product formula
\eqref{eq:ff-center-hilbert}. The final sentence is the object
firewall: a centre computation is the $0$th chiral Hochschild column,
whereas a bigraded Hochschild Hilbert series requires all cochain
degrees and all differentials. The displayed perturbation identity is
Corollary~\ref{cor:critical-level-spectral}; it is the local
calculation showing why the generic differential cannot be continued
through the critical divisor as the PBW/Arnold contraction used in
Theorem~H. The last displayed algebra is the Lie-cohomological factor
from Remark~\ref{rem:critical-level-lie-vs-chirhoch}, with the
exponents rewritten as $d_i=m_i$; it is not the degree-zero centre
computed in \eqref{eq:ff-center-hilbert}.
\end{proof}

\begin{remark}[Critical level and amplitude]
\label{rem:critical-level-separately-parametrised}
The critical level is outside
Theorem~\ref{thm:theorem-h-on-koszul-locus}. The obstruction has three
faces: the Sugawara denominator degenerates, the curvature-residue class
$\mathfrak c_{\mathrm{crit}}(\fg)$ blocks pole-free transport of the
generic contraction, and diagonal Ext-concentration used by the
FM-collapse proof fails. The Hilbert series
\eqref{eq:ff-center-hilbert} has finite coefficients in each
conformal weight, but finite coefficients in degree zero do not imply
Theorem~H's cohomological amplitude bound.
\end{remark}

\section{Boundary firewall for Theorem~H}
\label{sec:theorem-h-unified}

\begin{theorem}[Boundary firewall for Theorem~H; \ClaimStatusConditional]
\label{thm:theorem-h-platonic-unified}
\index{Theorem H!boundary firewall}
For conformal-weight-graded chiral algebras in the standard landscape
on a smooth projective curve, the following regimes are disjoint.
\begin{enumerate}[label=\textup{(\alph*)}]
\item \emph{PBW chiral Koszul, generic, completed.}
Theorem~\ref{thm:theorem-h-on-koszul-locus} applies:
$\ChirHoch^\bullet$ has amplitude~$[0,2]$ and Hilbert series
\eqref{eq:theorem-h-on-koszul}. The coHochschild self-extension of
$\barB^{\mathrm{ch}}(\cA)$ has the same amplitude only after
Corollary~\ref{cor:theorem-h-cohochschild-transport}.
\item \emph{Off the PBW/Koszul collapse.}
The high-degree groups are controlled by the positive collision-depth
defect exact sequence of
Theorem~\ref{thm:theorem-h-off-koszul-explicit-correction}. The first
page to inspect is
$\mathfrak{o}_{\mathrm H,1}^{\ge3}(\cA)\subset E_2$; the exact
degree-$3$ obstruction is
$\mathfrak{o}_{\mathrm H}^{3}(\cA)$, and degrees $\geq4$ are the
cohomology of $\mathrm{KD}_{\mathrm H}^{\bullet}(\cA)$. The lower
columns need not be the Theorem~H columns unless the low-degree
Verdier comparison is separately proved.
\item \emph{Logarithmic, admissible, and minimal simple quotients.}
One must prove the Theorem~H hypotheses for the quotient itself or use
the defect exact sequence. Rationality, $C_2$-cofiniteness, and a finite
ordinary module category do not replace chiral Koszulness. The
$\mathfrak{sl}_2$ admissible simple quotient is the rank-one Koszul
surface, still subject to the finite-window completion and perfectness
requirements of Theorem~\ref{thm:theorem-h-on-koszul-locus}; the
$\mathfrak{sl}_3$, $q\geq3$ surface carries the Cartan
bar obstruction
$\mathfrak{o}_{\mathrm H,\mathrm{Cartan}}^{\ge3}$, with degree-$3$
component $\mathfrak{o}_{\mathrm H,\mathrm{Cartan}}^{3}$, only after
the comparison map $\Phi_{\mathrm H}^{\mathrm{adm}}$ is supplied. That
component must be tested inside the full obstruction
$\mathfrak{o}_{\mathrm H}^{3}$. Triplet logarithmic algebras carry the
$W$-$W$ collision component $\mathfrak{o}_{\mathrm H,W}^{3}$.
For class-\(\mathsf M\) objects, raw direct-sum ordinary chains are not
a substitute for the completed or coderived ambient
\textup{(}Proposition~\ref{prop:chiral-positselski-raw-direct-sum-class-M-false}\textup{)}.
\item \emph{Critical affine level.}
The Feigin--Frenkel theorem gives the degree-zero centre
\eqref{eq:ff-center-hilbert}. The critical perturbation is governed
by $d_k=d_{\mathrm{crit}}+(k+h^\vee)\delta$ and the residue class
$\mathfrak c_{\mathrm{crit}}(\fg)$; under the separate
BD/Feigin--Tsygan comparison, the Lie-cohomological factor has
Hilbert--Poincare series $\operatorname{HP}_{\mathrm{crit}}(u)$ and is
unbounded.
\end{enumerate}
None of these statements concerns $\mathrm{THH}$ of spectra,
categorical Hochschild cochains of dg/CY categories, mode-algebra
Hochschild cochains, or product-ambient Hall Hochschild theories unless
a separate comparison theorem explicitly changes the input and output
categories. In particular, K3 or higher-dimensional Calabi--Yau
categorical Hochschild amplitudes are not imported into this curve-level
chiral Hochschild theorem.
\end{theorem}

\begin{proof}
Item~(a) is Theorem~\ref{thm:theorem-h-on-koszul-locus}. Item~(b) is
the exact triangle defining $\mathrm{KD}_{\mathrm H}$ and the
long-exact-sequence argument of
Theorem~\ref{thm:theorem-h-off-koszul-explicit-correction}. Item~(c)
is Theorem~\ref{thm:theorem-h-singular-quotient-firewall} and
Theorem~\ref{thm:theorem-h-logarithmic-w(p)}. Item~(d) is
Theorem~\ref{thm:theorem-h-at-critical-level}. The exclusion of other
Hochschild theories is the typing convention of
Convention~\ref{conv:three-hochschild} and
Remark~\ref{rem:three-hochschild-discipline}.
\end{proof}
