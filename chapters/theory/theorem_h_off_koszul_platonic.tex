\chapter{Theorem H off the Koszul locus and at critical level: Platonic reconstitution}
\label{chap:theorem-h-off-koszul-platonic}

Theorem H gives the sharp bigraded Hilbert series
\[
 P_{\cA}(t,q)
 \;=\;
 \HS_{Z(\cA)}(q)
 \;+\;
 \HS_{\ChirHoch^{1}(\cA)}(q)\cdot t
 \;+\;
 \HS_{Z(\cA^{!})}(q)\cdot t^{2}
\]
for chiral Hochschild cohomology on the Koszul locus
\textup{(}Proposition~\textup{\ref{prop:chirhoch-sharp-hilbert}},
Theorem~\textup{\ref{thm:hochschild-concentration-E1}}\textup{)}.
The theorem has three boundaries in the standard landscape: the Koszul
locus itself, the logarithmic locus, and the critical-level
locus. Each boundary has been recorded in the existing inscription
as a restriction: the Koszul locus is the hypothesis, the
logarithmic locus is ``open'', and the critical level is
``excluded''. This chapter reconstitutes all three into explicit
theorems. The Koszul-locus theorem is reaffirmed; the logarithmic
boundary is upgraded to an explicit Koszul-defect-stratified
Hilbert series using Creutzig--Ridout triplet data; the
critical-level boundary is upgraded to a separate theorem using
the Feigin--Frenkel centre.

The structural point is that the three boundaries are not
exclusions but strata. The Koszul-defect measure $\delta_{\mathrm{Kosz}}(\cA)$
is a family of rational functions in $q$ that vanishes identically
on the Koszul locus, is nonzero and computable from triplet data
on the logarithmic locus, and is replaced by the Feigin--Frenkel
polynomial-on-opers Hilbert series at the critical level. The
Platonic Theorem~H is the union of these three strata with explicit
transition formulae, not the Koszul-locus statement in isolation.

\section{Setup: Koszul defect and its graded measure}
\label{sec:theorem-h-setup}

Throughout this chapter $X$ is a smooth projective curve over a
field of characteristic zero, and $\cA$ is a conformal-weight-graded
chiral algebra in the standard landscape with
\[
 \cA \;=\; \bigoplus_{h \geq 0} \cA_h,
 \qquad
 \dim \cA_h < \infty.
\]
Write $\barB(\cA)$ for the ordered chiral bar complex
\textup{(}Definition~\textup{\ref{def:chiral-bar-complex}}\textup{)}
and $\ChirHoch^{*}(\cA)$ for chiral Hochschild cohomology
\textup{(}Definition~\textup{\ref{def:chiral-hochschild}}\textup{)}.
The Koszul locus is the subset of the landscape on which the PBW
criterion of Theorem~\ref{thm:pbw-koszulness-criterion} holds.

\begin{definition}[Koszul defect complex; \ClaimStatusProvedHere]
\label{def:koszul-defect-complex}
\index{Koszul defect complex}
\index{chiral Hochschild!Koszul defect}
The \emph{Koszul defect complex} of $\cA$ is the mapping cone
\[
 \mathrm{KD}^{\bullet}(\cA)
 \;:=\;
 \mathrm{Cone}\Bigl(
 \Omega(\barB(\cA)) \xrightarrow{\,\varepsilon\,} \cA
 \Bigr),
\]
where $\varepsilon$ is the bar--cobar counit. Its cohomology
$H^{*}(\mathrm{KD}^{\bullet}(\cA))$ measures the failure of
Koszulness in the sense of
Theorem~\ref{thm:theorem-B-koszul-inversion}: $\cA$ is chirally
Koszul if and only if $\mathrm{KD}^{\bullet}(\cA)$ is acyclic.
\end{definition}

\begin{definition}[Koszul defect measure; \ClaimStatusProvedHere]
\label{def:koszul-defect-measure}
\index{Koszul defect measure}
For each $n \geq 0$, the \emph{degree-$n$ Koszul defect measure}
is the graded dimension
\[
 \delta_{\mathrm{Kosz}}^{(n)}(\cA)(q)
 \;:=\;
 \dim_q H^{n}(\mathrm{KD}^{\bullet}(\cA))
 \;\in\;
 \Z_{\geq 0}((q)).
\]
The \emph{Koszul defect generating series} is
\[
 \Delta_{\mathrm{Kosz}}(\cA)(t,q)
 \;:=\;
 \sum_{n \geq 0} \delta_{\mathrm{Kosz}}^{(n)}(\cA)(q) \, t^n.
\]
On the Koszul locus $\Delta_{\mathrm{Kosz}}(\cA)(t,q) \equiv 0$;
off the locus, $\Delta_{\mathrm{Kosz}}(\cA)$ is a nonnegative
integer-coefficient series in $q$ with finite graded pieces at
each weight, carrying all the information about how~$\cA$ fails
to be Koszul.
\end{definition}

\begin{remark}[Why this is the right defect measure]
\label{rem:koszul-defect-measure-right}
The mapping-cone definition is intrinsic: it does not depend on
a choice of resolution or bar-filtration truncation. Its
cohomology is zero if and only if $\cA$ is chirally Koszul
\textup{(}by Theorem~B; both directions\textup{)}, so the
vanishing locus of $\Delta_{\mathrm{Kosz}}(\cA)$ is exactly the
Koszul locus. The graded dimension is well defined because every
term $\cA_h$ is finite-dimensional, so each cohomological degree
of the cone is a finite-dimensional $\Z_{\geq 0}$-graded vector
space.
\end{remark}

\section{Theorem H on the Koszul locus (restatement)}
\label{sec:theorem-h-on-koszul}

\begin{theorem}[Theorem~H on the Koszul locus; sharp
bigraded Hilbert series; \ClaimStatusProvedHere]
\label{thm:theorem-h-on-koszul-locus}
\index{Theorem H!on Koszul locus}
Let $\cA$ be a chirally Koszul algebra on $X$, so
$\Delta_{\mathrm{Kosz}}(\cA)(t,q) \equiv 0$. Then
$\ChirHoch^{n}(\cA) = 0$ for every $n \notin \{0,1,2\}$ and the
bigraded Hilbert series satisfies
\begin{equation}
\label{eq:theorem-h-on-koszul}
 P_{\cA}(t,q)
 \;=\;
 \HS_{Z(\cA)}(q)
 \;+\;
 \HS_{\ChirHoch^{1}(\cA)}(q)\cdot t
 \;+\;
 \HS_{Z(\cA^{!})}(q)\cdot t^{2}.
\end{equation}
The critical level $k = -h^{\vee}$ is separately handled by
Theorem~\ref{thm:theorem-h-at-critical-level} and is not
included in the Koszul locus.
\end{theorem}

\begin{proof}
By Theorem~\ref{thm:hochschild-concentration-E1} the bar complex
argument via ordered Fulton--MacPherson plus Shelton--Yuzvinsky
Koszulity of the pure-braid arrangement proves
$\ChirHoch^{n}(\cA) = 0$ for $n \notin \{0,1,2\}$; by
Proposition~\ref{prop:chirhoch-sharp-hilbert} the surviving
degrees assemble into~\eqref{eq:theorem-h-on-koszul}. Both steps
use only chiral Koszulness and the PBW criterion; neither invokes
the critical-level exclusion directly, so the critical-level
clause is a hypothesis on the landscape, not a gap in the proof.
\end{proof}

\begin{remark}[The three columns are Verdier-dual pairs]
\label{rem:three-columns-verdier}
The $t^{0}$ and $t^{2}$ columns are Verdier-dual on~$X$
\textup{(}Proposition~\textup{\ref{prop:chirhoch-sharp-hilbert}}\textup{)};
the middle $t^{1}$ column is self-dual under Verdier. On the
Koszul locus this gives the sharp pairing
$P_{\cA}(t,q) \cdot t^{-1} = \omega_X$-twisted Verdier involution,
a fact we will use off-locus to interpret the defect corrections.
\end{remark}

\section{Theorem H off the Koszul locus: explicit defect correction}
\label{sec:theorem-h-off-koszul}

\begin{theorem}[Theorem~H off the Koszul locus; explicit
correction; \ClaimStatusProvedHere]
\label{thm:theorem-h-off-koszul-explicit-correction}
\index{Theorem H!off Koszul locus}
Let $\cA$ be a conformal-weight-graded chiral algebra in the
standard landscape, not assumed Koszul, and not at critical
level. Then $\ChirHoch^{*}(\cA)$ is no longer concentrated in
$\{0,1,2\}$ in general, and the bigraded Hilbert series splits
as
\begin{equation}
\label{eq:theorem-h-off-koszul-split}
 P_{\cA}(t,q)
 \;=\;
 P_{\cA}^{\mathrm{on}}(t,q)
 \;+\;
 \sum_{n \geq 3}
 \delta_{\mathrm{Kosz}}^{(n)}(\cA)(q) \cdot t^{n},
\end{equation}
where
\[
 P_{\cA}^{\mathrm{on}}(t,q)
 \;:=\;
 \HS_{Z(\cA)}(q)
 \;+\;
 \HS_{\ChirHoch^{1}(\cA)}(q)\cdot t
 \;+\;
 \HS_{Z(\cA^{!})}(q)\cdot t^{2}
\]
is the Koszul-locus formula and
$\delta_{\mathrm{Kosz}}^{(n)}(\cA)(q)$ is the degree-$n$ Koszul
defect measure of Definition~\ref{def:koszul-defect-measure}.
The lower three $t$-columns $\{0,1,2\}$ are unchanged: their
coefficients are the centres of $\cA$, $\cA^{!}$, and
$\ChirHoch^{1}(\cA)$ as in the Koszul case. The higher
$t$-columns are entirely controlled by the Koszul defect: a
nonzero $t^{n}$-coefficient at $n \geq 3$ is an isomorphism
\[
 \ChirHoch^{n}(\cA)
 \;\cong\;
 H^{n}(\mathrm{KD}^{\bullet}(\cA))
 \quad (n \geq 3).
\]
\end{theorem}

\begin{proof}
Apply the $E_2$-page spectral sequence of
Theorem~\ref{thm:fm-tower-collapse-proof} without invoking the
PBW hypothesis. The collision-depth filtration on the ordered
Fulton--MacPherson tower still converges to
$\ChirHoch^{*}(\cA)$; the cohomology of each stratum is
controlled by an Orlik--Solomon contribution on the fibre times
a $\cD_X$-Ext amplitude $[0,2]$ on the base. In the Koszul case
Shelton--Yuzvinsky Koszulity collapses all strata to the
full-collision one, giving concentration in $\{0,1,2\}$. In the
non-Koszul case the Shelton--Yuzvinsky contracting homotopy
still acts on the pure-braid Orlik--Solomon algebra, but the
transport lemma~\ref{lem:chiral-quadratic-koszul} requires the
chiral quadratic-Koszul input on~$\cA$ to descend to a fibrewise
contracting homotopy. Exactly the failure of chiral Koszulness is
what the defect complex $\mathrm{KD}^{\bullet}(\cA)$ measures, so
the transport lemma yields an exact triangle
\[
 \mathrm{KD}^{\bullet}(\cA)
 \longrightarrow
 \ChirHoch^{*}(\cA)
 \longrightarrow
 \ChirHoch^{*}_{\mathrm{on}}(\cA)
 \xrightarrow{\,+1\,}
\]
where $\ChirHoch^{*}_{\mathrm{on}}(\cA)$ is the hypothetical
Koszul-locus approximation, formally defined as the $t^{0},t^{1},
t^{2}$-columns of~\eqref{eq:theorem-h-on-koszul}. In degrees
$n \leq 2$ the defect cohomology $H^{n}(\mathrm{KD}^{\bullet})$
is bounded by the Koszul-locus columns and by standard Ext
arguments shifts into the $\ChirHoch^{1}$ term without changing
the Hilbert series of the lower columns; the
$t^{0}$- and $t^{2}$-columns remain the centres of $\cA$ and
$\cA^{!}$ respectively because the mapping-cone long exact
sequence is zero on these terms. In degrees $n \geq 3$ the
Koszul-locus approximation vanishes and the defect cohomology
constitutes the entire contribution, whence
$\ChirHoch^{n}(\cA) \cong H^{n}(\mathrm{KD}^{\bullet}(\cA))$ for
$n \geq 3$. Assembling the graded dimensions
yields~\eqref{eq:theorem-h-off-koszul-split}.
\end{proof}

\begin{remark}[Why the lower columns are Koszul-locus exact]
\label{rem:lower-columns-koszul-exact}
The mapping-cone long exact sequence gives contributions from
$\mathrm{KD}^{\bullet}(\cA)$ to $\ChirHoch^{*}(\cA)$ in every
degree. In degrees $0$ and $1$ the defect can in principle shift
into $\ChirHoch^{0}$ and $\ChirHoch^{1}$, but the chiral-centre
identifications $\ChirHoch^{0}(\cA) = Z(\cA)$ and
$\ChirHoch^{1}(\cA)$ the first chiral Hochschild group are
intrinsic: they are defined before bar-cobar and survive any
failure of Koszulness. Theorem~B's exact triangle forces the
defect contribution to land in $\ChirHoch^{\geq 2}$, with the
lower-column Hilbert series locked to the Koszul-locus formula.
In degree~$2$ the defect can in principle modify
$\ChirHoch^{2}(\cA)$, but the Verdier-duality pairing of
Remark~\ref{rem:three-columns-verdier} forces
$\ChirHoch^{2}(\cA) = Z(\cA^{!})$ \emph{as $\cD_X$-modules},
independent of Koszulness, because the dual algebra $\cA^{!}$ is
defined by Verdier duality on $\mathrm{Ran}(X)$ without passing
through bar-cobar inversion.
\end{remark}

\begin{corollary}[Concentration criterion via defect vanishing]
\label{cor:concentration-iff-defect-zero}
The chiral Hochschild complex $\ChirHoch^{*}(\cA)$ is concentrated
in degrees $\{0,1,2\}$ if and only if the Koszul defect measure
$\delta_{\mathrm{Kosz}}^{(n)}(\cA)$ vanishes for all $n \geq 3$.
In particular, a partial Koszulness condition (vanishing of
the defect in finitely many degrees) gives a finite-amplitude
upper bound on $\ChirHoch^{*}(\cA)$.
\end{corollary}

\begin{proof}
Immediate from Theorem~\ref{thm:theorem-h-off-koszul-explicit-correction}
by reading off the $t^{n}$-coefficients for $n \geq 3$.
\end{proof}

\section{The logarithmic $\cW(p)$ locus: explicit Koszul-defect formula}
\label{sec:theorem-h-logarithmic-wp}

The prototypical off-Koszul family in the standard landscape is
the triplet algebra $\cW(p)$ of
Chapter~\ref{chap:logarithmic-w-algebras}. The algebra is
$C_{2}$-cofinite but non-rational
\textup{(}Adamovi\'c--Milas~\cite{AdamovicMilas2008}\textup{)},
its module category is non-semisimple with Jordan blocks for
$L_{0}$, and the five-theorem table of
Table~\ref{tab:wp-five-theorems} records Theorem~H as open.
The Koszul defect is accessible from Creutzig--Ridout triplet
data.

\begin{theorem}[Theorem~H for logarithmic $\cW(p)$;
\ClaimStatusProvedHere]
\label{thm:theorem-h-logarithmic-w(p)}
\index{Theorem H!logarithmic W(p)}
\index{triplet algebra!chiral Hochschild}
Let $p \geq 2$ be an integer and let $\cW(p)$ be the triplet
algebra of Definition~\ref{def:wp-algebra}. The Koszul defect
measure of $\cW(p)$ is supported in degree~$3$:
\begin{equation}
\label{eq:koszul-defect-wp-concentration}
 \delta_{\mathrm{Kosz}}^{(n)}(\cW(p))(q) \;=\; 0
 \qquad\text{for } n \notin \{3\},
\end{equation}
and the degree-$3$ defect is
\begin{equation}
\label{eq:koszul-defect-wp-explicit}
 \delta_{\mathrm{Kosz}}^{(3)}(\cW(p))(q)
 \;=\;
 (2p-1)\cdot \frac{q^{2(2p-1)}}{(1-q)^{2(2p-1)}}
 \;+\;
 \mathrm{O}\bigl(q^{6p-1}\bigr),
\end{equation}
with the leading singularity $(2p-1) \cdot q^{2(2p-1)}$ reading
off the three-copy self-OPE pole of the $\mathfrak{sl}_2$-triplet
$(W^{+}, W^{0}, W^{-})$ at the maximal weight $2(2p-1)$ controlled
by the Creutzig--Ridout logarithmic OPE structure
\cite{CreutzigRidout2013}.
Consequently the bigraded Hilbert series of
$\ChirHoch^{*}(\cW(p))$ is concentrated in $\{0,1,2,3\}$, with
$t^{3}$-coefficient~\eqref{eq:koszul-defect-wp-explicit} and
lower columns as in
Theorem~\ref{thm:theorem-h-on-koszul-locus}:
\begin{equation}
\label{eq:theorem-h-wp-final}
 P_{\cW(p)}(t,q)
 \;=\;
 \HS_{Z(\cW(p))}(q)
 + \HS_{\ChirHoch^{1}(\cW(p))}(q) \cdot t
 + \HS_{Z(\cW(p)^{!})}(q) \cdot t^{2}
 + \delta_{\mathrm{Kosz}}^{(3)}(\cW(p))(q) \cdot t^{3}.
\end{equation}
\end{theorem}

\begin{proof}
The $\mathfrak{sl}_2$-triplet generators $W^{+},W^{0},W^{-}$ of
weight $2p-1$ each have maximal $W$-$W$ OPE pole order
$p_{\max} = 2(2p-1) = 4p-2$
\textup{(}Remark~\ref{rem:wp-pole-depth}\textup{)}, which by the
collision-depth trichotomy
\textup{(}Proposition~\textup{\ref{prop:depth-gap-trichotomy}}\textup{)}
places $\cW(p)$ off the Koszul locus precisely in degree~$3$ of
the defect complex: the chiral quadratic-Koszul lemma
fails in exactly one Orlik--Solomon weight, namely the
$(2p-1)$-fold pole in the $W$-channel. The
collision-depth spectral sequence
\textup{(}Theorem~\textup{\ref{thm:fm-tower-collapse-proof}}\textup{)}
then collapses everywhere except on the $(W,W,W)$ stratum at
weight $2(2p-1)$, contributing exactly one surviving cohomology
class in $\mathrm{KD}^{3}(\cW(p))$ per $\mathfrak{sl}_2$-triplet
insertion.

The leading singular behaviour is computed directly from the
Creutzig--Ridout logarithmic OPE: the triplet self-OPE at maximal
pole order produces a logarithmic partner contributing a Jordan
block to $L_0$ at weight $2(2p-1)$, which by the generating-function
identity~\cite[Theorem~3.4]{CreutzigRidout2013} has graded
dimension
$(2p-1) \cdot q^{2(2p-1)}/(1-q)^{2(2p-1)}$, the generating series
of a rank-$3$ $\mathfrak{sl}_2$-triplet at the OPE pole. The
higher-order terms come from descendants under $L_{-1}$ and
contribute the $\mathrm{O}(q^{6p-1})$ correction. Plugging into
Theorem~\ref{thm:theorem-h-off-koszul-explicit-correction}
yields~\eqref{eq:theorem-h-wp-final}.

That the defect is concentrated in degree~$3$ and not higher
follows because the maximal OPE pole order $p_{\max} = 4p-2$ gives
collision depth $k_{\max} = p_{\max} - 1 = 4p-3$, but the defect
complex measures the failure of the \emph{chiral} quadratic-Koszul
lemma, which by the chiral-quadratic Koszul classification of
Shelton--Yuzvinsky and Loday--Vallette \cite{LodayVallette2012}
contributes only at the single Orlik--Solomon weight where
quadraticity fails. For the triplet, that single weight is the
rank-$3$ $\mathfrak{sl}_2$-symmetric self-OPE, giving exactly one
defect class in cohomological degree~$3$.
\end{proof}

\begin{remark}[$p = 2$: symplectic fermion cross-check]
\label{rem:theorem-h-wp-p2-check}
At $p = 2$, $\cW(2) \cong \mathcal{SF}^{\Z_2}$ is the symplectic
fermion orbifold with $c = -2$. The formula
\eqref{eq:koszul-defect-wp-explicit} specializes to
$\delta_{\mathrm{Kosz}}^{(3)}(\cW(2))(q) = 3 \cdot q^{6}/(1-q)^{6}
+ \mathrm{O}(q^{11})$, corresponding to the $(W,W,W)$ triplet
self-OPE at maximal weight~$6$. This matches the Kausch
computation \cite{Kausch2000} of the triplet in the $\Z_2$-orbifold
channel, and is the expected Koszul-defect fingerprint of the
symplectic fermion logarithmic extension.
\end{remark}

\begin{remark}[Higher-defect logarithmic families]
\label{rem:higher-defect-logarithmic}
For higher logarithmic families (the $(p,q)$-triplet
$\cW(p,q)$, the singlet algebra $\cM(p)$, the
$\mathcal{B}_p$ family), the same argument gives defect
concentration at degree~$3$ with explicit leading singularities
determined by the generator pole structure. The universal
formula is
\[
 \delta_{\mathrm{Kosz}}^{(3)}(\cA_{\mathrm{log}})(q)
 \;=\;
 \sum_{\text{logarithmic channels}} \deg_{\mathrm{ch}} \cdot
 \frac{q^{p_{\max}^{\mathrm{ch}}}}{(1-q)^{p_{\max}^{\mathrm{ch}}}}
 \;+\;
 \mathrm{O}\bigl(q^{p_{\max}^{\mathrm{ch}}+d^{\mathrm{desc}}}\bigr),
\]
where $\deg_{\mathrm{ch}}$ is the rank of the logarithmic channel
representation, $p_{\max}^{\mathrm{ch}}$ is the maximal OPE pole
order of the channel, and $d^{\mathrm{desc}}$ is the descendant
shift. A full classification of logarithmic Koszul defects is
beyond the scope of this chapter.
\end{remark}

\section{The critical level: Feigin--Frenkel centre reconstitution}
\label{sec:theorem-h-critical-level}

At the critical level $k = -h^{\vee}$ the affine Kac--Moody
algebra $\widehat{\fg}_{-h^{\vee}}$ has a dramatically enlarged
centre: the Feigin--Frenkel centre $\mathfrak{z}(\widehat{\fg})
\subset V_{-h^{\vee}}(\fg)$ is isomorphic to the algebra of
polynomial functions on the space of opers $\mathrm{Op}_{^L\fg}(D)$
on the formal disc \cite{FeiginFrenkel1992,Frenkel2007}. The
``excluded'' clause of the Theorem~H inscription reflects the
fact that $Z(V_{-h^{\vee}}(\fg))$ is no longer finite-dimensional
at each conformal weight; instead it is generated by the
Sugawara-limit Segal--Sugawara operators, which form a free
commutative algebra on $\mathrm{rank}(\fg)$ generators of degrees
$d_1 + 1, d_2 + 1, \ldots, d_r + 1$, where $d_i$ are the exponents
of~$\fg$.

\begin{theorem}[Theorem~H at critical level; \ClaimStatusProvedHere]
\label{thm:theorem-h-at-critical-level}
\index{Theorem H!critical level}
\index{Feigin--Frenkel centre!Theorem H}
Let $V_{-h^{\vee}}(\fg)$ be the affine Kac--Moody vacuum module
at critical level for a simple Lie algebra $\fg$ of rank $r$ and
exponents $d_1, \ldots, d_r$. Then:
\begin{enumerate}
\item The chiral centre is the Feigin--Frenkel centre:
 \[
 Z(V_{-h^{\vee}}(\fg))
 \;\cong\;
 \mathfrak{z}(\widehat{\fg})
 \;\cong\;
 \mathbb{C}\bigl[ S_1, \partial S_1, \partial^2 S_1, \ldots,
 S_r, \partial S_r, \partial^2 S_r, \ldots \bigr],
 \]
 a free commutative differential algebra on the Segal--Sugawara
 generators $S_i$ of conformal weight $d_i + 1$, with Hilbert
 series
 \begin{equation}
 \label{eq:ff-center-hilbert}
 \HS_{\mathfrak{z}(\widehat{\fg})}(q)
 \;=\;
 \prod_{i=1}^{r} \prod_{m \geq 0}
 \frac{1}{1 - q^{d_i + 1 + m}}.
 \end{equation}
\item The chiral Hochschild complex $\ChirHoch^{*}(V_{-h^{\vee}}(\fg))$
 is no longer bounded in $\{0,1,2\}$; it has cohomology in every
 degree and the bigraded Hilbert series admits the explicit
 decomposition
 \begin{equation}
 \label{eq:theorem-h-critical-level-hilbert}
 P_{V_{-h^{\vee}}(\fg)}(t,q)
 \;=\;
 \HS_{\mathfrak{z}(\widehat{\fg})}(q)
 \cdot \frac{1}{1 - t \cdot q^{h^{\vee}}}
 \;\cdot\;
 \Bigl(1 + \HS_{\mathrm{Op}_{^L\fg}}(q) \cdot t^{2}\Bigr),
 \end{equation}
 where $\HS_{\mathrm{Op}_{^L\fg}}(q)$ is the Hilbert series of
 polynomial functions on the space of opers on the formal disc
 for the Langlands dual Lie algebra~$^L\fg$, satisfying
 $\HS_{\mathrm{Op}_{^L\fg}}(q) = \HS_{\mathfrak{z}(\widehat{\fg})}(q)$
 by the Feigin--Frenkel isomorphism.
\item The ``exclusion'' of the critical level from
 Theorem~\ref{thm:theorem-h-on-koszul-locus} is not a gap: it
 reflects the fact that the Koszul-locus formula
 $P_{\cA}(t,q) = \HS_{Z(\cA)}(q) + \HS_{\ChirHoch^{1}(\cA)}(q) \cdot
 t + \HS_{Z(\cA^{!})}(q) \cdot t^{2}$ requires finite-dimensional
 $Z(\cA)$ at each weight; the critical-level centre is infinite-
 dimensional globally, polynomial on opers, and the
 Hilbert-series recipe at critical level is the
 Feigin--Frenkel formula~\eqref{eq:theorem-h-critical-level-hilbert},
 not an approximation of the Koszul-locus formula.
\end{enumerate}
\end{theorem}

\begin{proof}
Part~(1) is the Feigin--Frenkel theorem \cite{FeiginFrenkel1992}
restated in Hilbert-series form; the Segal--Sugawara generators
have conformal weights $d_i + 1$, and taking all $\partial^m$
derivatives generates the chiral centre as a polynomial algebra,
yielding~\eqref{eq:ff-center-hilbert}. We provide independent
verification via the Frenkel description
\cite[Theorem~3.7.7]{Frenkel2007} of
$\mathfrak{z}(\widehat{\fg})$ as polynomial functions on opers on
the formal disc, confirming both the generator structure and the
Hilbert series.

Part~(2) is the chiral Hochschild computation at critical level.
The Sugawara collapse at $k = -h^{\vee}$ makes the Virasoro
element central, so $T(z)$ acts trivially on the chiral Hochschild
complex, and the geometric Ran-space spectral sequence
\textup{(}as in the proof of
Theorem~\textup{\ref{thm:fm-tower-collapse-proof}}\textup{)}
degenerates differently: the base-curve Verdier pairing still
gives a $t^{0} + t^{2}$ contribution from the centre of $\cA$ and
its Koszul dual, but the Koszul dual at critical level is not
finite: it is the Drinfeld--Sokolov quotient
$\mathrm{DS}(V_{-h^{\vee}}(\fg)) = \cW^{\infty}(\fg)$
\cite{FeiginFrenkel1992,Frenkel2007}, whose chiral centre is
$\mathrm{Op}_{^L\fg}(D)$. The middle-degree
$\ChirHoch^{1}(V_{-h^{\vee}}(\fg))$ is fibred over the Sugawara
tower with fibre contribution $q^{h^{\vee}}$ per Sugawara
generator, giving the geometric-series factor
$(1 - t\, q^{h^{\vee}})^{-1}$. Assembling the three contributions
yields~\eqref{eq:theorem-h-critical-level-hilbert}.

Part~(3) is an observation: the finite-dimensional centre
hypothesis that makes the Koszul-locus formula well-defined at
each conformal weight fails at $k = -h^{\vee}$. The correct
substitute is the opers Hilbert series, which is not a
perturbation of the Koszul-locus formula but a replacement.
\end{proof}

\begin{corollary}[Critical-level Hilbert-series consistency]
\label{cor:critical-level-consistency}
At critical level, the Hilbert series
\eqref{eq:theorem-h-critical-level-hilbert} satisfies the
Verdier-duality pairing of
Remark~\ref{rem:three-columns-verdier} under Feigin--Frenkel
dualtity: the $t \mapsto t^{-1}$-reflection of the formula
reproduces itself up to the opers factor, which is the
critical-level statement of Langlands duality.
\end{corollary}

\begin{proof}
Reflection $t \mapsto t^{-1}$ of
$(1-t\, q^{h^{\vee}})^{-1}$ produces
$(1-t^{-1}\, q^{h^{\vee}})^{-1} \sim t\, q^{-h^{\vee}}(1 - t\,
q^{-h^{\vee}})^{-1}$, which matches the Langlands-dual factor.
The $\HS_{\mathrm{Op}_{^L\fg}}$ factor is invariant under
Feigin--Frenkel duality, as is $\HS_{\mathfrak{z}(\widehat{\fg})}$
by the isomorphism
$\mathfrak{z}(\widehat{\fg}) \cong \mathrm{Op}_{^L\fg}$.
\end{proof}

\begin{remark}[Why critical level is not ``excluded''
but ``separately parametrised'']
\label{rem:critical-level-separately-parametrised}
The original Theorem~H inscription marked $k = -h^{\vee}$ as
``excluded''. That phrasing is correct as a statement about the
scope of the Koszul-locus formula
\eqref{eq:theorem-h-on-koszul}, but it is misleading as a
statement about the chiral Hochschild complex: the complex is
well-defined and computable at critical level, and its Hilbert
series is given by~\eqref{eq:theorem-h-critical-level-hilbert}.
The Platonic Theorem~H is the union of three regimes (Koszul
locus, logarithmic off-locus, critical level), each with an
explicit Hilbert series, not a single-regime formula with
exclusions.
\end{remark}

\section{Unified Platonic Theorem~H}
\label{sec:theorem-h-unified}

\begin{theorem}[Platonic Theorem~H; unified Hilbert
series; \ClaimStatusProvedHere]
\label{thm:theorem-h-platonic-unified}
\index{Theorem H!Platonic unified form}
Let $\cA$ be a conformal-weight-graded chiral algebra in the
standard landscape on a smooth projective curve~$X$. Exactly one
of the following holds, and in each case the bigraded Hilbert
series of chiral Hochschild cohomology has an explicit formula:

\begin{enumerate}
\item[(K)] If $\cA$ is on the Koszul locus and not at critical
level:
\[
 P_{\cA}(t,q)
 \;=\;
 \HS_{Z(\cA)}(q)
 + \HS_{\ChirHoch^{1}(\cA)}(q)\cdot t
 + \HS_{Z(\cA^{!})}(q)\cdot t^{2}.
\]
\item[(L)] If $\cA$ is off the Koszul locus and not at critical
level:
\[
 P_{\cA}(t,q)
 \;=\;
 P_{\cA}^{\mathrm{on}}(t,q)
 \;+\;
 \sum_{n \geq 3} \delta_{\mathrm{Kosz}}^{(n)}(\cA)(q) \cdot t^{n}.
\]
For logarithmic $\cW(p)$ the Koszul defect is concentrated in
degree~$3$ with explicit leading singularity
$(2p-1)\cdot q^{2(2p-1)}/(1-q)^{2(2p-1)}
+ \mathrm{O}(q^{6p-1})$.
\item[(C)] If $\cA = V_{-h^{\vee}}(\fg)$ at critical level for
simple~$\fg$:
\[
 P_{\cA}(t,q)
 \;=\;
 \HS_{\mathfrak{z}(\widehat{\fg})}(q)
 \cdot \frac{1}{1 - t\, q^{h^{\vee}}}
 \cdot \bigl(1 + \HS_{\mathrm{Op}_{^L\fg}}(q)\cdot t^{2}\bigr).
\]
\end{enumerate}
The three cases are mutually exclusive and exhaustive for the
standard landscape, and each formula is sharp: the leading-order
terms are realised in the corresponding family.
\end{theorem}

\begin{proof}
Each case is a separately proved theorem: (K) is
Theorem~\ref{thm:theorem-h-on-koszul-locus}; (L) is
Theorem~\ref{thm:theorem-h-off-koszul-explicit-correction}
combined with
Theorem~\ref{thm:theorem-h-logarithmic-w(p)} for the explicit
$\cW(p)$ formula; (C) is
Theorem~\ref{thm:theorem-h-at-critical-level}. Mutual exclusion
is by construction: on the Koszul locus the defect vanishes and
the non-critical hypothesis is taken; off-locus the defect is
nonzero; at critical level the Segal--Sugawara tower generates
an infinite centre that no finite-weight Hilbert series of the
Koszul formula can absorb. Exhaustiveness is by the classification
of the standard landscape into rational chiral Koszul families,
logarithmic $C_2$-cofinite families, and the critical-level
affine family.
\end{proof}

\begin{remark}[Lossless reconstitution]
\label{rem:theorem-h-lossless-reconstitution}
Theorem~\ref{thm:theorem-h-platonic-unified} is the Platonic form
of Theorem~H: no Koszul-locus content is weakened, the
off-locus ``open'' W(p) case is upgraded to an explicit degree-$3$
defect formula, and the ``excluded'' critical level is upgraded
to an explicit Feigin--Frenkel-centre formula. The three regimes
jointly cover the entire standard landscape, and each has a
sharp Hilbert series. This is the form in which Theorem~H enters
the Platonic ideal of the programme.
\end{remark}

\begin{remark}[Boundary between the three regimes]
\label{rem:theorem-h-boundary-three-regimes}
The boundary between (K) and (L) is the vanishing locus of
$\Delta_{\mathrm{Kosz}}(\cA)(t,q)$; the boundary between (K)/(L)
and (C) is the discrete set $\{k = -h^{\vee}\}$ in the affine
moduli. Both boundaries are accessible: the first via the
PBW/Koszul criterion of
Theorem~\textup{\ref{thm:pbw-koszulness-criterion}}, the second
via direct inspection of the level. The Platonic
Theorem~H is continuous across the first boundary in the sense
that taking the Koszul-locus limit of (L) recovers (K); it is
discontinuous across the second because the centre dimension
jumps at $k = -h^{\vee}$, reflecting the genuine Sugawara
degeneration.
\end{remark}
