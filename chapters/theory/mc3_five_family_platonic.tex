\section{Evaluation core, finite-window completion, and full DK}
\label{sec:mc3-platonic-scope}

The third master conjecture of Volume~I concerns thick generation of
the Drinfeld--Kohno factorization category
$\mathrm{DK}_\mathfrak{g}$ of an $E_1$-chiral quantum group
$Y_\hbar(\mathfrak{g})^{\mathrm{ch}}$ by evaluation-type modules.
Read at full strength, MC3 asserts that the entire factorization
category is generated under thick idempotent-complete closure by a
finite or countable seed of evaluation-type modules and their
prefundamental companions. The assertion has three categorical
surfaces.

\begin{enumerate}[label=\textup{(D\arabic*)},nosep]
\item\label{it:mc3-eval-core}
 \emph{Evaluation-generated core}
 $\mathrm{DK}^{\mathrm{ev}}_\mathfrak{g} \subset
 \mathrm{DK}_\mathfrak{g}$:
 the thick idempotent-complete subcategory generated by the
 evaluation modules $V_{\omega_i}(a)$ and their shifted
 prefundamental companions $L^{\pm}_{\omega_i,\,b}$
 at generic spectral parameters.
 This is a genuine subcategory; every object is built from
 evaluation-type modules by iterated cones, retracts, and tensor
 products.
\item\label{it:mc3-finite-window-completion}
 \emph{Finite-window completion}
 $\mathrm{DK}^{\mathrm{completed}}_{\mathfrak g,\nu}$:
 after choosing a separated exhaustive finite-window parameter~$\nu$
 and exact quotients
 $\tau_{\le N}\colon\mathrm{DK}^{\mathrm{ev}}_{\mathfrak g}
 \to\mathrm{DK}^{\mathrm{ev},\le N}_{\mathfrak g,\nu}$, set
 \[
  \mathrm{DK}^{\mathrm{completed}}_{\mathfrak g,\nu}
  =
  \varprojlim_N
  \operatorname{Ind}
  \bigl(\mathrm{DK}^{\mathrm{ev},\le N}_{\mathfrak g,\nu}\bigr).
 \]
 A completed object is a compatible tower of finite-window objects.
 Generation here means windowwise compact thick generation followed by
 Ind-completion and inverse limit.  It is not ordinary thick closure
 under arbitrary filtered colimits.
\item\label{it:mc3-full-DK}
 \emph{Full $\mathrm{DK}_\mathfrak{g}$}: the intended
 Drinfeld--Kohno factorization category.  An equivalence
 $\mathrm{DK}^{\mathrm{completed}}_{\mathfrak g,\nu}\simeq
 \mathrm{DK}_\mathfrak{g}$ is an additional compact-object and
 topology-comparison theorem, not a formal consequence of
 \ref{it:mc3-eval-core} or~\ref{it:mc3-finite-window-completion}.
 \textup{(}This is not a bar-cobar inversion statement:
 $\Omega(B(A)) = A$ on the nose, with no completion involved;
 the category-level DK equivalence is derived from bar-cobar
 via the Koszul-pairing derived-center comparison of
 Chapter~\ref{chap:deformation-theory}, not by naming
 $\Omega(B(A))$ as the closed-sector object.\textup{)}
\end{enumerate}

\begin{principle}[Scope distinction for MC3]
\label{princ:mc3-scope-distinction}
\emph{The evaluation core, its finite-window completion, and full DK
are distinct.} The five-family mechanism proves thick generation of
\ref{it:mc3-eval-core}. The finite-window comparison chapter proves
generation of~\ref{it:mc3-finite-window-completion} only after the
chosen window tower satisfies the Mittag--Leffler and coacyclic-cone
hypotheses. The full category~\ref{it:mc3-full-DK} is reached only by
an additional compact-completion theorem. In type~$A$ this is the
single remaining packet
Conjecture~\ref{conj:dk-compacts-completion}; outside type~$A$ the
same full-DK upgrade also requires the lift-and-lower and pro-Weyl
packets of Conjectures~\ref{conj:rank-independence-step2} and
\ref{conj:mc3-automatic-generalization}.
\end{principle}

\begin{remark}[Scope-qualified MC3 statement]
\label{rem:mc3-scope-qualified}
For every family in the standard landscape (rational type~$A$,
twisted B/C/D, generic simple~$\mathfrak{g}$, elliptic
$\mathrm{sl}_N$, super $\mathrm{gl}_{m|n}$), the evaluation-generated
core of the relevant DK category is thickly generated by
evaluation-type modules via a family-specific module-theoretic
mechanism. The Baxter hyperplane is the coordinate signature of
the rational type~$A$ mechanism only; four other families have
different signatures \textup{(}none, none, theta divisor,
parity-balance hyperplane\textup{)}.
Extension to a finite-window completed category is
Theorem~\ref{thm:mc3-full-DK-in-completed-category} under the tower
hypotheses of Definition~\ref{def:finite-window-mc3-tower}.
Identifying that completed category with the full DK category is the
compact-completion problem.
\end{remark}

\begin{proposition}[Finite-window upgrade boundary;
\ClaimStatusProvedElsewhere]
\label{prop:mc3-finite-window-upgrade-boundary}
\index{MC3!finite-window upgrade boundary}%
Let $\mathcal S\subset\mathrm{DK}^{\mathrm{ev}}_{\mathfrak g}$ be the
seed of
Theorem~\ref{thm:mc3-evaluation-core-five-family}, and let~$\nu$ be a
finite-window parameter. If
\textup{(i)} every compact window is generated by
$\tau_{\le N}\mathcal S$,
\textup{(ii)} the transition functors preserve compact generated
subcategories, and
\textup{(iii)} mapping spectra of compact-window towers satisfy the
Mittag--Leffler condition, then
\[
 \widehat{\operatorname{Loc}}_\nu(\mathcal S)
 =
 \mathrm{DK}^{\mathrm{completed}}_{\mathfrak g,\nu}.
\]
If compact objects are detected windowwise, then
\[
 \widehat{\operatorname{Thick}}_\nu(\mathcal S)
 =
 \bigl(\mathrm{DK}^{\mathrm{completed}}_{\mathfrak g,\nu}\bigr)^\omega .
\]
In a curved CDG realization the comparison cone must be coacyclic in
the second-kind coderived category in every finite window and the tower
of cone cohomologies must be Mittag--Leffler. In a square-zero total
model the same sentence reads ``acyclic''.
\end{proposition}

\begin{proof}
This is the finite-window theorem
Theorem~\ref{thm:mc3-full-DK-in-completed-category} together with
Definition~\ref{def:finite-window-mc3-tower}. The cone clause is the
Milnor exact-sequence argument of
Proposition~\ref{prop:compact-approximation-exists}: the
$\varprojlim^1$ term vanishes under Mittag--Leffler, and Positselski's
second-kind ambient replaces ordinary acyclicity by coacyclicity.
\end{proof}

\begin{remark}[Derived centre and Theorem H are separate surfaces]
\label{rem:mc3-derived-centre-hochschild-boundary}
\index{derived center!MC3 boundary}%
\index{Theorem H!MC3 boundary}%
MC3 is a generation theorem for a DK module category. It does not
identify chiral derived centres. The derived centre is
\[
 Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)
 =
 \ChirHoch^\bullet(\cA,\cA),
\]
and the brace dg-algebra comparison
$Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)\simeq
Z^{\mathrm{der}}_{\mathrm{ch}}(\cA^!)$ is controlled by the
Morita-kernel criterion of
Proposition~\ref{prop:derived-center-morita-criterion}; its naive
$H^0$ shadow uses the completed central-pairing criterion
Lemma~\ref{lem:naive-center-koszul-identification}.

MC3 also does not prove Theorem~H. The chiral Hochschild amplitude
$[0,2]$ is Theorem~\ref{thm:main-koszul-hoch} on the PBW
E$_\infty$-chiral Koszul locus, with symmetric averaging supplied by
Corollary~\ref{cor:hochschild-averaging-symmetric}. For an
E$_1$-chiral Yangian input without that completion, the ordered
concentration theorem
Theorem~\ref{thm:hochschild-concentration-E1} survives, while the
symmetric bound is Conjecture~\ref{conj:hochschild-concentration-E1-only}.
\end{remark}

\begin{remark}[Objects separated by the MC3 boundary]
\label{rem:mc3-object-separation}
\index{MC3!object separation}%
MC3 lives in a DK module category.  For a chiral algebra~$\cA$ the
objects
\[
 \cA,\qquad
 \bar B_X(\cA),\qquad
 \cA^{\mathrm i}:=H^\bullet \bar B_X(\cA),\qquad
 \cA^!:=\mathbb D_{\Ran}(\cA^{\mathrm i}),\qquad
 Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)
\]
carry different functorial meanings.  The counit
$\Omega_X\bar B_X(\cA)\to\cA$ is bar--cobar inversion.  The passage
from $\cA^{\mathrm i}$ to $\cA^!$ is Verdier duality on the Ran
space.  The bulk
$Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)=\ChirHoch^\bullet(\cA,\cA)$ is
the chiral Hochschild cochain object.  A finite-window DK completion
changes the topology of a module category; it does not identify any
two of these five objects.
\end{remark}

\section{Central theorem: five-family thick generation on the evaluation-generated core}
\label{sec:mc3-platonic-central-theorem}

\begin{theorem}[MC3 on the evaluation-generated core, five-family
 mechanism; \ClaimStatusProvedHere]
\label{thm:mc3-evaluation-core-five-family}
\index{MC3!evaluation-generated core!five-family mechanism|textbf}
\index{thick generation!evaluation-generated core|textbf}
Let $\mathfrak{g}$ be a simple Lie algebra, or
$\mathfrak{g} = \mathfrak{gl}_{m|n}$ with $m,n \ge 0$, or
let $\mathcal{E}_\rho$ be an elliptic curve with deformation
parameter~$\eta$. Let $\mathrm{DK}^{\mathrm{ev}}_\mathfrak{g}$
denote the evaluation-generated core of the DK factorization
category as in~\ref{it:mc3-eval-core}.
Then $\mathrm{DK}^{\mathrm{ev}}_\mathfrak{g}$ is thickly generated
by a countable seed of evaluation-type modules at generic spectral
parameters. The generation is controlled family-by-family by a
single module-theoretic mechanism:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{lll}
\toprule
\textbf{Family} & \textbf{Mechanism} & \textbf{Coordinate signature} \\
\midrule
$Y_\hbar(\mathrm{sl}_N)$ rational
 & asymptotic prefundamentals \textup{(}QQ-system\textup{)}
 & none \\
$Y^{\mathrm{tw}}_\hbar(\mathrm{so}_n),Y^{\mathrm{tw}}_\hbar(\mathrm{sp}_n)$
 & reflection-equation Shapovalov
 & none \\
general simple $\mathfrak{g}$
 & multiplicity-free $\ell$-weights
 & none \textup{(}Zariski-open\textup{)} \\
$E_{\rho,\eta}(\mathrm{sl}_N)$ elliptic
 & elliptic Bethe / DYBE
 & theta divisor $\Theta \subset \mathcal{E}_\rho^{N-1}$ \\
$Y_\hbar(\mathrm{gl}_{m|n})$ super (parity-balance sub-case)
 & $\mathbb{Z}/2$-graded Berezinian
 & parity-balance hyperplane \\
\bottomrule
\end{tabular}
\end{center}
In each family, the mechanism provides a non-degenerate pairing
(Shapovalov, reflection-equation Shapovalov, elliptic Bethe, or
super-Shapovalov) on the relevant tensor products at generic
parameters, together with the family-specific witness that keeps
fusion inside the evaluation-generated core: the QQ-system product
expansion, the Sklyanin determinant, simple Bethe-subalgebra spectrum,
the theta determinant, or the quantum Berezinian.
\end{theorem}

\begin{remark}[Super-Yangian scope: gl-super and osp-super sub-cases]
\label{rem:super-yangian-scope}
The super-Yangian row of the mechanism table covers the
$Y_\hbar(\mathfrak{gl}_{m|n})$ series via the parity-balance
hyperplane mechanism with the Nazarov quantum Berezinian as
witness. The sibling orthosymplectic series $Y_\hbar(\mathfrak{osp}(m|2n))$
is outside the parity-balance scope: its reflection equation
(Arnaudon--Cramp\'e--Doikou--Frappat--Ragoucy 2003,
\texttt{arXiv:math-ph/0304188}; the orthosymplectic
reflection-equation presentations appear in the related series of
papers by the same group) is a $\mathbb{Z}/2$-graded analogue of
the twisted $B/C/D$ reflection equation, and its thick-generation
witness is the super-Sklyanin--Berezinian determinant
$\operatorname{sBer} K(u)$, not the Nazarov quantum Berezinian on
the Baxter operator. The osp-super series therefore belongs to
an orthosymplectic $\mathbb{Z}/2$-graded reflection-equation
mechanism sibling to
Proposition~\ref{prop:mc3-type-BCD-reflection-shapovalov-platonic},
inscribed as the separate
Conjecture~\ref{conj:mc3-osp-super-reflection-equation-platonic},
not as a specialisation of
Proposition~\ref{prop:mc3-type-BCD-reflection-shapovalov-platonic}
itself, and not to the parity-balance mechanism of
Proposition~\ref{prop:mc3-super-parity-balance-platonic}.
The K3 Mukai signature is not an example of this Kac
orthosymplectic row.  Vol~III separates the ungraded orthogonal
envelope $Y_\hbar(\mathfrak{so}(4,20))$ and the Hodge-parity
Hodge-parity extension $Y_\hbar(\mathfrak{so}(4\mid20))$ from
Kac's $\mathfrak{osp}(4\mid20)$
\textup{(}\texttt{chapters/examples/k3\_yangian\_chapter.tex},
\texttt{rem:k3y-stage-super-yangian-scope},
\texttt{rem:gl-to-osp-correction}\textup{)}: the Mukai form is
symmetric on both Hodge-parity summands, while Kac orthosymplectic
type requires a symplectic form on the odd summand.  K3/Mukai
material may enter the ordinary orthogonal reflection-equation row,
or a separate non-Kac Hodge-parity comparison if that comparison is
constructed; it is not covered by the Kac orthosymplectic conjecture
below.
\end{remark}

\begin{proof}
The mechanism rows are
Propositions~\ref{prop:mc3-type-A-asymptotic-prefundamentals-platonic},
\ref{prop:mc3-type-BCD-reflection-shapovalov-platonic},
\ref{prop:mc3-uniform-chari-moura-platonic},
\ref{prop:mc3-uniform-chari-moura-nonsimplylaced-platonic},
\ref{prop:mc3-elliptic-theta-divisor-platonic}, and
\ref{prop:mc3-super-parity-balance-platonic}.
The two Chari--Moura/Hernandez propositions form the single uniform
simple-Lie row: classical and simply-laced exceptional types are
covered uniformly, while $G_2$ and $F_4$ enter by finite per-case
verification of their fundamental $q$-characters. No case uses the
rank-$1$ Baxter singular vector as hypothesis or conclusion; each
provides its own family-specific witness.
\end{proof}

\begin{remark}[MC3 mechanisms and G/L/C/M shadows]
\label{rem:mc3-vs-shadow-archetypes}
\index{MC3!versus shadow classes}%
The five MC3 families are module-category generation mechanisms, not
the shadow-depth classes $\mathsf G/\mathsf L/\mathsf C/\mathsf M$.
The latter are computed in the landscape census from the shadow tower:
$r_{\max}=2$ for Gaussian class~$\mathsf G$, $r_{\max}=3$ for
Lie/tree class~$\mathsf L$, $r_{\max}=4$ for contact/quartic
class~$\mathsf C$, and $r_{\max}=\infty$ for mixed class~$\mathsf M$.
The Theorem~C scalar sums are read from
Proposition~\ref{prop:archetype-complementarity-bridge}, not from
MC3.  The self-contained Vol~I ceiling is
$\{0,13,250/3,98/3\}$: the value~$13$ is the Virasoro
$\mathsf M$ witness, while $250/3$ and $98/3$ are the
$\mathcal W_3$ and Bershadsky--Polyakov $\mathsf M$-extension
values, not constants attached to all class~$\mathsf M$ algebras.
The Mukai-enhanced $\mathsf B$-row adds the external value~$8$ only
on the Vol~III K3/Mukai bridge, where
$c_+(\mathrm{Mukai}(K3))=4$ and
$K^{\kappa_{\mathrm{ch}}}=2c_+(\mathrm{Mukai}(K3))=8$.  This
$\mathsf B$-row is neither a sixth MC3 mechanism nor a Kac
orthosymplectic import.
\end{remark}

\begin{remark}[Assembly point]
\label{rem:mc3-assembly-point}
The theorem has one conclusion and five proofs. In each chart the
seed generates the same kind of object,
$\mathrm{DK}^{\mathrm{ev}}_\mathfrak{g}$, but the witness differs:
rational prefundamentals, reflection-equation pairing,
multiplicity-free $\ell$-weights, elliptic Bethe regularity, or
super-Berezinian parity balance. The common conclusion is ordinary
thick generation of the evaluation core.
\end{remark}

\section{The five family-specific propositions}
\label{sec:mc3-five-family-propositions}

\subsection{Type~$A$: asymptotic prefundamentals}

\begin{proposition}[Asymptotic prefundamentals: rational type~$A$;
 \ClaimStatusProvedHere]
\label{prop:mc3-type-A-asymptotic-prefundamentals-platonic}
\index{MC3!type~A!asymptotic prefundamentals}%
\index{Hernandez-Jimbo!QQ-system}%
\index{prefundamental!asymptotic limit}%
Let $\mathcal{C}_{\mathrm{pf}} \subset
\widehat{\mathcal{O}}(Y_\hbar(\mathfrak{sl}_N))$ be the monoidal
subcategory generated by the family of asymptotic prefundamental
modules $\{L^\pm_{\omega_i,a}\}_{1 \le i \le N-1,\; a \in \mathbb{C}}$
obtained as the Hernandez--Jimbo limit of Kirillov--Reshetikhin
modules. Then $\mathcal{C}_{\mathrm{pf}}$ is a thick tensor ideal of
the completed category $\widehat{\mathcal{O}}$, and no Baxter
hyperplane constraint on the spectral parameter~$a$ is required.
\end{proposition}

\begin{proof}
The Hernandez--Jimbo limit
$L^\pm_{\omega_i,a} = \lim_{k\to\infty} W^{(i)}_{k,a}$ exists in
$\widehat{\mathcal{O}}$ as a pro-object, and its $q$-character
is the Frenkel--Hernandez completed monomial expansion with dominant
$\ell$-weight $\Psi_{i,a}^{\pm1}$ and lower monomials generated by the
simple-root variables of the QQ-system. The proof uses only the two
properties supplied by that construction: triangularity with respect
to the $\ell$-weight order and multiplicity-freeness on the generic
tensor products under consideration. The Mukhin--Varchenko
Shapovalov determinant formula gives non-degeneracy on these
$\ell$-weight spaces, so the Bethe subalgebra has simple joint
spectrum and separates simples. Closure under subquotients follows
from multiplicity-freeness and the Frenkel--Mukhin algorithm; closure
under tensor products follows from the QQ-system fusion relations. The
Baxter singular-vector computation is not invoked: the negative
prefundamental arises as a limit of finite-dimensional
Kirillov--Reshetikhin modules, not as the simple cokernel of a
rank-$1$ Verma exact sequence on $b = a - \tfrac12$.
\end{proof}

\begin{remark}[Why this retires the Baxter hyperplane in type~$A$]
\label{rem:type-A-no-hyperplane}
The rank-$1$ Baxter singular vector is visible in the
rational-form presentation of $Y_\hbar(\mathrm{sl}_N)$ on
$b = a - \tfrac12$. In the QQ-system / asymptotic KR presentation
the same information is carried globally by the product
factorization of $\chi_q(L^-_{i,a})$: the negative prefundamental
is born with its full multiplicity-free $\ell$-character, so no
hyperplane is part of the generation hypothesis. The Baxter
hyperplane is a chart-specific witness, not a universal obstruction.
\end{remark}

\subsection{Types B/C/D: reflection-equation Shapovalov}

\begin{proposition}[Reflection-equation Shapovalov: twisted B/C/D;
 \ClaimStatusProvedHere]
\label{prop:mc3-type-BCD-reflection-shapovalov-platonic}
\index{MC3!twisted Yangian!reflection-equation Shapovalov}%
\index{twisted Yangian!Molev-Ragoucy}%
\index{Sklyanin determinant!thick-generation witness}%
Let $Y^{\mathrm{tw}}_\hbar(\mathfrak{so}_n)$ or
$Y^{\mathrm{tw}}_\hbar(\mathfrak{sp}_n)$ be the twisted Yangian of
Molev--Ragoucy type with reflection matrix $K(u)$ satisfying the
boundary Yang--Baxter equation
\[
 R_{12}(u-v)\,K_1(u)\,R_{21}(u+v)\,K_2(v) \;=\;
 K_2(v)\,R_{12}(u+v)\,K_1(u)\,R_{21}(u-v).
\]
The evaluation-generated core of
$\widehat{\mathcal{O}}(Y^{\mathrm{tw}}_\hbar)$ is thickly generated
by the fundamental evaluation family
$\{V_{\omega_i,a}\}_{i,\;a\in\mathbb{C}}$ at generic
parameters in the Zariski-open sense on the joint parameter space
$(q, a)$, excluding in particular the primitive-root-of-unity
locus of $q$; root-of-unity specializations are out of scope for
the five-family synthesis. The Sklyanin
determinant
$\operatorname{sdet}K(u) \in Z(Y^{\mathrm{tw}}_\hbar)[[u^{-1}]]$ is
central and supplies the generation witness.
\end{proposition}

\begin{proof}
The Manin-triple decomposition
$Y^{\mathrm{tw}}_\hbar = Y^+_\hbar \bowtie Y^-_\hbar$ and the
reflection-equation pairing
$\langle\cdot,\cdot\rangle_{\mathrm{RE}}$ is non-degenerate on
$\bigotimes_i V_{\omega_{j_i},a_i}$ at generic spectral parameters
(Letzter--Kolb coideal-subalgebra criterion; Molev--Ragoucy
non-degeneracy for twisted Yangians). Molev--Ragoucy singular
vectors in these tensor products supply explicit short exact
sequences whose kernels and cokernels are again evaluation-type
modules, closing the evaluation-generated core under extensions.
Centrality of $\operatorname{sdet}K(u)$ promotes character-level
closure to module-level closure by the standard composition thick
generation lemma. No Baxter hyperplane appears; the reflection
equation encodes the boundary spectral data uniformly.
\end{proof}

\subsection{Uniform: multiplicity-free $\ell$-weights}

\begin{proposition}[Chari--Moura multiplicity-free $\ell$-weights:
 classical and simply-laced exceptional types; \ClaimStatusProvedHere]
\label{prop:mc3-uniform-chari-moura-platonic}
\index{MC3!uniform!multiplicity-free $\ell$-weights}%
\index{Chari-Moura!$\ell$-weight theorem}%
\index{Frenkel-Mukhin!algorithm}%
Let $\mathfrak{g}$ be a simple Lie algebra of classical type
$A_n, B_n, C_n, D_n$ or simply-laced exceptional type
$E_6, E_7, E_8$. For every fundamental weight $\omega_i$ the
evaluation module $V_{\omega_i,a}$ of
$Y_\hbar(\mathfrak{g})$ has multiplicity-free $\ell$-weight
decomposition, and the evaluation-generated core of
$\mathcal{O}(Y_\hbar(\mathfrak{g}))$ is thick.
\end{proposition}

\begin{proof}
Multiplicity-freeness of $\chi_q(V_{\omega_i,a})$ is established
by two complementary inputs depending on type: quiver-variety
geometry for simply-laced types (Nakajima, arXiv:math/0105173)
and explicit $q$-character formulas for classical types
(Chari--Moura).
The Kirillov--Reshetikhin conjecture, proved in full generality,
closes the induction on the fusion products of KR modules;
combined with density of the simple joint spectrum, separation of
simples gives thickness of the evaluation-generated core.
No Baxter hyperplane is invoked: multiplicity-freeness is a
property of the full $\ell$-character, not of a rank-$1$ singular
vector.
\end{proof}

\begin{proposition}[Multiplicity-free $\ell$-weights:
 non-simply-laced exceptional types $G_2, F_4$; \ClaimStatusProvedElsewhere]
\label{prop:mc3-uniform-chari-moura-nonsimplylaced-platonic}
\index{MC3!non-simply-laced exceptional!multiplicity-free $\ell$-weights}%
Let $\mathfrak{g}$ be a simple Lie algebra of non-simply-laced
exceptional type $G_2$ or $F_4$. For every fundamental weight
$\omega_i$ the evaluation module $V_{\omega_i,a}$ of
$Y_\hbar(\mathfrak{g})$ has multiplicity-free $\ell$-weight
decomposition, and the evaluation-generated core of
$\mathcal{O}(Y_\hbar(\mathfrak{g}))$ is thick. The proof proceeds
per-case for the $2$ fundamentals of $G_2$ and the $4$ fundamentals
of $F_4$.
\end{proposition}

\begin{remark}[Frenkel--Mukhin algorithm is not uniformly valid
 at non-simply-laced exceptional types]
\label{rem:fm-nonsimplylaced-failure}
The Frenkel--Mukhin algorithm, which gives a uniform construction
of the true $q$-character of fundamental representations at
simply-laced types (where every fundamental is \emph{special} in
the sense of Frenkel--Mukhin), is not uniformly valid for
fundamentals of non-simply-laced types: at type $G_2$ and $F_4$
the algorithm can fail to terminate on a true $q$-character, or
can produce an output that does not coincide with the true
$q$-character of certain fundamentals, and the algorithm does
not by itself establish multiplicity-freeness of each
$\ell$-weight space. Case-by-case verification of the $2$
fundamental $q$-characters of $G_2$ and the $4$ fundamental
$q$-characters of $F_4$, together with their multiplicity-freeness,
is a finite computation carried out in
Hernandez~(arXiv:math/0606381) and Hernandez~(2010), building on
Hernandez--Jimbo and subsequent work. These references supply the
multiplicity-free $\ell$-character data used in
Proposition~\ref{prop:mc3-uniform-chari-moura-nonsimplylaced-platonic};
the thickness argument for the evaluation-generated core is the
same as in the simply-laced case, and depends only on
multiplicity-freeness of the $\ell$-character plus the
Kirillov--Reshetikhin conjecture.
\end{remark}

\begin{remark}[Relation to the asymptotic-prefundamental route]
\label{rem:chari-moura-vs-hernandez-jimbo}
At $\mathfrak{g} = \mathrm{sl}_N$ the two routes (asymptotic
prefundamentals,
Proposition~\ref{prop:mc3-type-A-asymptotic-prefundamentals-platonic};
multiplicity-free $\ell$-weights,
Proposition~\ref{prop:mc3-uniform-chari-moura-platonic}) produce
structurally distinct thick-generation statements at two distinct
category levels, not two proofs of a single theorem. The asymptotic
route proves thickness in the pro-completed category
$\widehat{\mathcal{O}}(Y_\hbar(\mathfrak{sl}_N))$: the
prefundamentals $L^\pm_{\omega_i,a}$ exist only as pro-objects
obtained as Hernandez--Jimbo limits of finite-dimensional
Kirillov--Reshetikhin modules, and the proof is global in the
spectral parameter via the product factorization of the
$q$-character. The multiplicity-free route proves thickness in the
uncompleted $\mathcal{O}(Y_\hbar(\mathfrak{sl}_N))$: the evaluation
modules $V_{\omega_i}(a)$ are finite-dimensional, and the proof is
pointwise via one-dimensional $\ell$-weight spaces. The two routes
are complementary at two category levels, not equivalent proofs of
one statement; each theorem is the MC3 thick-generation assertion
in its own category. The Gautam--Toledano Laredo equivalence ties
the Yangian and quantum-loop versions of each route together within
each category level.
\end{remark}

\subsection{Elliptic: theta-divisor complement}

\begin{proposition}[Elliptic Bethe / DYBE: theta-divisor complement;
 \ClaimStatusProvedHere]
\label{prop:mc3-elliptic-theta-divisor-platonic}
\index{MC3!elliptic Yangian!theta divisor}%
\index{elliptic Bethe ansatz!Felder-Varchenko}%
\index{DYBE!dynamical Yang-Baxter equation}%
Let $E_{\rho,\eta}(\mathfrak{sl}_N)$ be the elliptic Yangian over
the elliptic curve
$\mathcal{E}_\rho = \mathbb{C}/(\mathbb{Z} + \rho\mathbb{Z})$ with
deformation parameter~$\eta$. The evaluation-generated core is
thickly generated by evaluation modules at generic parameters on
the complement of the theta divisor
$\Theta \subset \mathcal{E}_\rho^{N-1}$.
\end{proposition}

\begin{proof}
The Felder--Varchenko elliptic Bethe ansatz produces joint
eigenvectors of the elliptic Bethe subalgebra
$B_{\mathrm{ell}}(u) \subset
E_{\rho,\eta}(\mathrm{sl}_N)[[u^{-1}]]$ parametrized by solutions
of the elliptic Bethe equations. The Bethe-regular locus is the
complement of the failure divisor: the zero locus of the elliptic
Shapovalov determinant, equivalently the zero locus of the
theta-Jacobian of the Bethe equations, which is a theta divisor
$\Theta \subset \mathcal{E}_\rho^{N-1}$. Separation of simples
gives thickness on the theta-divisor complement. The
codimension-$1$ failure locus replaces the codimension-$1$ Baxter
hyperplane of the rational type~$A$ case; both are divisors, but
the elliptic divisor is intrinsic to the elliptic curve, whereas
the Baxter hyperplane is a coordinate of the rational
presentation.
\end{proof}

\subsection{Super: parity-balance hyperplane}

\begin{proposition}[Super-Yangian parity-balance:
 $Y_\hbar(\mathfrak{gl}_{m|n})$; \ClaimStatusProvedHere]
\label{prop:mc3-super-parity-balance-platonic}
\index{MC3!super-Yangian!parity-balance}%
\index{quantum Berezinian!Nazarov}%
\index{super-Yangian!evaluation-generated core}%
For the super-Yangian $Y_\hbar(\mathfrak{gl}_{m|n})$ with
$\mathbb{Z}/2$-grading $A = A_{\bar0} \oplus A_{\bar1}$, the
Nazarov quantum Berezinian factorizes the Baxter operator as
$Q(u) = Q^+(u) \otimes Q^-(u)$ with $Q^\pm(u)$ acting on the
even/odd sectors. Thick generation of the evaluation-generated
core of $\mathcal{O}(Y_\hbar(\mathfrak{gl}_{m|n}))$ holds on the
complement of the parity-balance hyperplane
\[
 b^+ - b^- \;=\; a - m + n,
\]
which has codimension~$1$ in spectral parameter space.
\end{proposition}

\begin{proof}
The super-Shapovalov form on tensor products of evaluation modules
is non-degenerate away from the parity-balance hyperplane (the
$\mathbb{Z}/2$-graded Mukhin--Varchenko determinant formula, with
the Berezinian $\operatorname{Ber}(u)$ of Nazarov replacing the
ordinary determinant). On the hyperplane the form degenerates
along a single direction; off the hyperplane,
multiplicity-freeness of the $\ell$-character combined with the
parity-balance separation gives thickness by the argument of
Proposition~\ref{prop:mc3-uniform-chari-moura-platonic}. The factorization
$Q(u) = Q^+(u) \otimes Q^-(u)$ ensures that the parity-balance
constraint is a single codimension-$1$ condition, not a higher
intersection of even and odd Baxter loci.
\end{proof}

\begin{remark}[Scope is Y(gl(m|n)) only; osp-super sibling covered
 elsewhere]
\label{rem:parity-balance-scope-gl-only}
Proposition~\ref{prop:mc3-super-parity-balance-platonic} covers the
general-linear super-Yangian $Y_\hbar(\mathfrak{gl}_{m|n})$ only.
The orthosymplectic super-Yangian $Y_\hbar(\mathfrak{osp}(m|2n))$,
constructed via the $\mathbb{Z}/2$-graded reflection equation of
Arnaudon--Cramp\'e--Doikou--Frappat--Ragoucy (\texttt{arXiv:math-ph/0304188}
for the $\mathfrak{sl}(M|N)$ parent; the $\mathfrak{osp}(m|n)$
reflection-equation presentations are in the related series of
papers by the same group),
is outside the parity-balance mechanism scope. Its thick-generation
witness is a super-Sklyanin--Berezinian determinant
$\operatorname{sBer}K(u)$ defined off the reflection-equation
fixed locus, not the Nazarov quantum Berezinian
$\operatorname{Ber}(u)$ on the Baxter operator. The osp-super
series is covered by
Conjecture~\ref{conj:mc3-osp-super-reflection-equation-platonic},
a sibling statement to the ordinary
twisted $B/C/D$ reflection-equation Proposition~\ref
{prop:mc3-type-BCD-reflection-shapovalov-platonic}, rather than
as a specialisation of the latter; the two share the reflection-
equation, but their $R$-matrix ($R^{\mathrm{osp}}$
with $Q$-projector vs $R^{\mathrm{so}}$ / $R^{\mathrm{sp}}$),
their pairing ($\mathbb{Z}/2$-graded vs ordinary), and their
central witness ($\operatorname{sBer}K(u)$ vs $\operatorname{sdet}K(u)$)
differ, so the statements are genuinely sibling rather than
parent--specialisation. See
Remark~\ref{rem:super-yangian-scope} for the cross-volume context.
\end{remark}

\begin{conjecture}[Orthosymplectic super-Yangian reflection-equation
 Shapovalov; \ClaimStatusConjectured]
\label{conj:mc3-osp-super-reflection-equation-platonic}
\index{MC3!orthosymplectic super-Yangian!reflection-equation Shapovalov}%
\index{super-Sklyanin determinant!sBer}%
\index{Z/2-graded reflection equation!orthosymplectic}%
Let $Y_\hbar(\mathfrak{osp}(m|2n))$ be the orthosymplectic
super-Yangian constructed from the Kulish--Reshetikhin rational
$R^{\mathrm{osp}}$-matrix
\[
 R^{\mathrm{osp}}(u)
 \;=\; \operatorname{Id} \;-\; \frac{\hbar}{u}\, P_s
 \;-\; \frac{\hbar}{u + \hbar\,\kappa_{\mathfrak{osp}}/2}\, Q,
 \qquad \kappa_{\mathfrak{osp}} = m - 2n - 2,
\]
with $P_s$ the graded permutation and $Q$ the
$\mathfrak{osp}$-invariant trace-like projector, subject to a
reflection matrix $K(u)$ satisfying the $\mathbb{Z}/2$-graded
boundary Yang--Baxter equation in the Arnaudon--Cramp\'e--Doikou--
Frappat--Ragoucy series. Then the evaluation-generated core of
$\widehat{\mathcal{O}}(Y_\hbar(\mathfrak{osp}(m|2n)))$ is
conjecturally thickly generated by the family
$\{V_{\omega_1,a}\}_{a\in\mathbb{C}}$ at generic parameters, with
the super-Sklyanin--Berezinian determinant
\[
 \operatorname{sBer} K(u) \;\in\; Z(Y_\hbar(\mathfrak{osp}(m|2n)))[[u^{-1}]]
\]
central and supplying the generation witness off the
reflection-equation fixed locus.
\end{conjecture}

\begin{remark}[Scope and residual frontier of the osp-super
 reflection-equation conjecture]
\label{rem:mc3-osp-super-reflection-equation-scope}
Conjecture~\ref{conj:mc3-osp-super-reflection-equation-platonic}
is stated conjecturally because three inputs used in the ordinary
twisted $B/C/D$ proof
(Proposition~\ref{prop:mc3-type-BCD-reflection-shapovalov-platonic})
are not yet established in the $\mathbb{Z}/2$-graded setting at
the level needed for the evaluation-generated-core thickness:
(a) non-degeneracy of the $\mathbb{Z}/2$-graded reflection-equation
pairing on $\bigotimes_i V_{\omega_1,a_i}$, the super analogue of
the Molev--Ragoucy non-degeneracy for ordinary twisted Yangians;
(b) existence of singular-vector short exact sequences for
orthosymplectic Lax operators closing the core under extensions,
the super analogue of the Molev--Ragoucy singular-vector
construction; and (c) centrality of $\operatorname{sBer}K(u)$
promoting character-level closure to module-level closure in the
$\mathbb{Z}/2$-graded composition thick-generation lemma. Each
of (a), (b), (c) has partial evidence in the ACDFR series at rank
$\mathfrak{osp}(1|2)$ and $\mathfrak{osp}(2|2)$ (graded braid
identities, reflection equation verified pointwise), but the
rank-$(m,2n)$ statement at arbitrary Kac-orthosymplectic rank is the
residual frontier.  This is a Kac orthosymplectic problem, not a
K3/Mukai import: Vol~III's Mukai branch uses
$Y_\hbar(\mathfrak{so}(4,20))$ and a possible non-Kac Hodge-parity
$Y_\hbar(\mathfrak{so}(4\mid20))$, with the form symmetric on both
graded parts.  A proof of the present Kac orthosymplectic conjecture
would not by itself identify the K3 Hall--Drinfeld target;
conversely, a K3 orthogonal/Hodge-parity comparison would not settle
the arbitrary-rank Kac orthosymplectic reflection-equation theorem.
\end{remark}

\section{Baxter retraction: a type-A rational artifact}
\label{sec:mc3-platonic-baxter-retraction}

\begin{proposition}[Baxter hyperplane as a type-$A$ rational
 artifact; \ClaimStatusProvedHere]
\label{prop:baxter-retraction-type-A-artifact}
\index{Baxter constraint!retraction!type-$A$ rational artifact}%
\index{MC3!Baxter retraction}%
The rank-$1$ Baxter hyperplane $b = a - \tfrac12$ is a rational
specialization of the Shapovalov normalization specific to the
type-$A$ rational form of $Y_\hbar(\mathrm{sl}_N)$. It appears as
the codimension-$1$ locus at which the short exact sequence
$0 \to L^-_{i,a-1/2} \to V_1(a) \otimes L^-_{i,a}
\to L^-_{i,a+1/2} \to 0$
is witnessed by a rank-$1$ singular vector in the evaluation
tensor product. For the other four families of
Theorem~\ref{thm:mc3-evaluation-core-five-family} the Shapovalov
normalization differs:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item twisted B/C/D: reflection-equation Shapovalov pairing with
 $\operatorname{sdet}K(u)$ witness; no hyperplane,
\item uniform simple $\mathfrak{g}$: multiplicity-free
 $\ell$-weights with Zariski-open genericity; no hyperplane,
\item elliptic $\mathrm{sl}_N$: elliptic Shapovalov with theta-
 divisor complement; the codim-$1$ locus is the theta divisor,
\item super $\mathrm{gl}_{m|n}$: super-Shapovalov with
 $\mathbb{Z}/2$-graded Berezinian; the codim-$1$ locus is the
 parity-balance hyperplane.
\end{enumerate}
The Baxter hyperplane survives in the rational type~$A$ chart as a
coordinate-dependent witness of the type-$A$ mechanism; it is not a
universal MC3 hypothesis.
\end{proposition}

\begin{proof}
In type~$A$ rational, the singular vector
$w_0 = -v_+ \otimes f \cdot v_0 \in V_1(a) \otimes L^-(b)$ is
Yangian highest-weight precisely at $b = a - \tfrac12$, where the
Shapovalov form on the rank-$1$ $\ell$-weight subspace vanishes
along a single direction. Under degeneration
elliptic~$\to$~trigonometric~$\to$~rational
($\operatorname{Im}\rho \to\infty$, $q \to 1$), the theta-divisor
failure locus degenerates to rational discriminant components; in the
rank-$1$ type-$A$ chart one component is the Baxter hyperplane. The
uniform Bethe-subalgebra simplicity survives degeneration as a
Zariski-open condition, which is the
multiplicity-free $\ell$-weight route of
Proposition~\ref{prop:mc3-uniform-chari-moura-platonic}. Under the twisted
B/C/D reflection-equation route, the Sklyanin determinant is
central and the generation witness carries no hyperplane. In the
super case the parity-balance hyperplane is the
$\mathbb{Z}/2$-graded analogue; it is not the rational Baxter
hyperplane but an independent codim-$1$ locus.
Each of these claims is explicitly verified by the proof of the
respective proposition in
Section~\ref{sec:mc3-five-family-propositions}.
\end{proof}

\begin{remark}[Meaning of Baxter retraction]
\label{rem:baxter-retraction-meaning}
As a universal hypothesis of MC3, the rank-$1$ Baxter hyperplane is
absent: four mechanism rows do not use it, and the non-trivial
codimension-$1$ loci (theta divisor and parity-balance hyperplane) are
different divisors. As a type-$A$ computation, the rank-$1$ Baxter
singular vector remains a valid witness of the rational Shapovalov
normalization. MC3 uses the first statement; the type-$A$ calculation
supplies only a local witness.
\end{remark}

\section{The full-DK boundary}
\label{sec:mc3-platonic-full-DK-conjectural}

The evaluation-generated core maps to its finite-window completion and
to the intended full DK factorization category. Neither map is an
identity by definition. The full-DK assertion requires a compact-object
comparison after the finite-window theorem has been proved.

\begin{theorem}[Full $\mathrm{DK}_\mathfrak{g}$ boundary;
 \ClaimStatusConditional]
\label{thm:mc3-full-DK-conjectural}
\index{MC3!full DK!conjectural}%
\index{Drinfeld-Kohno!full category!generation}%
Let $\mathfrak{g}$ be a simple Lie algebra in the range of
Theorem~\ref{thm:mc3-evaluation-core-five-family}. The following are
the exact implications beyond the evaluation core.
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item \emph{Finite-window completion.} If a finite-window parameter
 $\nu$ satisfies the tower hypotheses of
 Definition~\ref{def:finite-window-mc3-tower}, then the seed of
 Theorem~\ref{thm:mc3-evaluation-core-five-family} generates
 $\mathrm{DK}^{\mathrm{completed}}_{\mathfrak g,\nu}$ in the completed
 sense of Definition~\ref{def:completed-thick-closure}. In the curved
 lane the comparison is a coacyclic-cone statement in the second-kind
 coderived category.
\item \emph{Type~$A$ \textup{(}$\mathfrak{g} = \mathrm{sl}_N$,
 $N \ge 2$\textup{)}}. Theorem~\ref{thm:mc3-type-a-resolution}
 proves the prefundamental CG, shifted-prefundamental generation, and
 pro-Weyl recovery packets. The upgrade from the completed shifted
 envelope to full $\mathrm{DK}_{\mathfrak{sl}_N}$ is exactly
 Conjecture~\ref{conj:dk-compacts-completion}.
\item \emph{Non-type-$A$ simple $\mathfrak{g}$}: shifted-envelope
 generation is conditional on
 Conjecture~\ref{conj:rank-independence-step2} \textup{(}Lemma~L:
 lift-and-lower\textup{)};
 the pro-Weyl and completion packets are conjectural and form
 Conjecture~\ref{conj:mc3-automatic-generalization}.
\item \emph{Sectorwise extension}: the sectorwise root-weight
 decomposition proved on the evaluation-generated core extends to
 the completed/coderived enlargement conjecturally
 \textup{(}Conjecture~\ref{conj:mc3-sectorwise-all-types}\textup{)}.
\item \emph{Non-coverage sectors}: the following five
 families lie outside the five-family union of
 Corollary~\ref{cor:five-family-union-coverage} and carry each
 their own conjectural MC3 reduction:
\begin{enumerate}[label=\textup{(v.\alph*)},nosep]
\item logarithmic $W(p)$ algebras in the Adamovi\'c--Milas sense
 \textup{(}obstruction: non-semisimple Virasoro with
 indecomposable reducibles\textup{)};
\item $\mathcal{N}=2$ superconformal algebras
 \textup{(}obstruction: mixed bosonic/fermionic strong generators
 without an $R$-matrix presentation; conjectural reduction via
 Kazama--Suzuki coset, Conjecture~\ref{conj:mc3-n2-sca-kazama-suzuki}\textup{)};
\item general cosets $G/H$ beyond the GKO and KRW constructions
 \textup{(}obstruction: affine-quotient geometry, non-principal
 reduction\textup{)};
\item indefinite non-rational lattice VOAs
 \textup{(}obstruction: non-rational weight lattices, theta-series
 modularity\textup{)};
\item root-of-unity admissible-level specializations of the above
 families \textup{(}obstruction: finite-dimensional simples,
 modular tensor category rigidity\textup{)}.
\end{enumerate}
 None of (v.a)--(v.e) admits the evaluation-generated core
 mechanism verbatim; each requires a separate investigation.
\end{enumerate}
In particular, MC3 in the strong sense of
``thick generation of $\mathrm{DK}_\mathfrak{g}$ by evaluation-type
modules'' is not yet proved for any non-type-$A$ simple $\mathfrak{g}$,
and remains conditional on a single compact-completion conjecture
in type~$A$. The non-coverage sectors of clause~(v) are
outside the $\mathfrak{g}$-indexed simple-Lie-algebra framework
altogether and each carries its own conjectural reduction.
\end{theorem}

\begin{remark}[Why the full-DK statement is harder]
\label{rem:full-DK-harder}
\ref{it:mc3-full-DK} demands control of the passage from the
finite-window category
$\mathrm{DK}^{\mathrm{completed}}_{\mathfrak g,\nu}$ to the intended
factorization category $\mathrm{DK}_\mathfrak{g}$. The formal
finite-window theorem supplies density and Mittag--Leffler cone
control inside the chosen topology. It does not prove that every
compact object of full DK is detected by that topology. In type~$A$
the remaining compact-object problem is isolated by
Conjecture~\ref{conj:dk-compacts-completion}; in other types the
analogous statement also depends on Lemma~L.
\end{remark}

\section{Cross-family coverage and residual sectors}
\label{sec:mc3-platonic-coverage}

\begin{corollary}[Five-family coverage with osp sibling;
\ClaimStatusConditional]
\label{cor:five-family-union-coverage}
\index{MC3!five-family coverage}%
The proved union of the five families of
Theorem~\ref{thm:mc3-evaluation-core-five-family} covers:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item every simple Lie algebra $\mathfrak{g}$ of classical type
 $A_n, B_n, C_n, D_n$ and simply-laced exceptional type $E_6, E_7,
 E_8$ via either
 Propositions~\ref{prop:mc3-type-A-asymptotic-prefundamentals-platonic},
 \ref{prop:mc3-type-BCD-reflection-shapovalov-platonic}, or
 \ref{prop:mc3-uniform-chari-moura-platonic}; non-simply-laced
 exceptional type $G_2, F_4$ via
 Proposition~\ref{prop:mc3-uniform-chari-moura-nonsimplylaced-platonic}
 (per-case verification of fundamental $q$-characters,
 Remark~\ref{rem:fm-nonsimplylaced-failure});
\item elliptic Yangians $E_{\rho,\eta}(\mathrm{sl}_N)$ over an
 elliptic curve, via Proposition~\ref{prop:mc3-elliptic-theta-divisor-platonic};
\item the general-linear super-Yangians
 $Y_\hbar(\mathfrak{gl}_{m|n})$ via the parity-balance mechanism of
 Proposition~\ref{prop:mc3-super-parity-balance-platonic}.
\end{enumerate}
The orthosymplectic super-Yangians
$Y_\hbar(\mathfrak{osp}(m|2n))$ are the conjectural sibling of the
twisted $B/C/D$ reflection-equation row:
Conjecture~\ref{conj:mc3-osp-super-reflection-equation-platonic}.
The union does not cover: logarithmic W-algebras $W(p)$ in the
Adamovi\'c--Milas sense, $N=2$ superconformal vertex algebras,
general coset constructions, lattice VOAs at non-rational weights,
and root-of-unity specializations of the above families. These
are separate investigations with their own generation mechanisms,
not covered by the five-family synthesis.
\end{corollary}

\begin{proof}
Classical and simply-laced exceptional simple types are covered by
Proposition~\ref{prop:mc3-uniform-chari-moura-platonic};
non-simply-laced exceptional types $G_2, F_4$ are covered by
Proposition~\ref{prop:mc3-uniform-chari-moura-nonsimplylaced-platonic}
per-case. Type
$A$ admits the additional route of
Proposition~\ref{prop:mc3-type-A-asymptotic-prefundamentals-platonic}; the
twisted forms B/C/D admit the reflection-equation route of
Proposition~\ref{prop:mc3-type-BCD-reflection-shapovalov-platonic}.
Propositions~\ref{prop:mc3-elliptic-theta-divisor-platonic} and
\ref{prop:mc3-super-parity-balance-platonic} supply the elliptic and
general-linear super cases. The orthosymplectic super case is
Conjecture~\ref{conj:mc3-osp-super-reflection-equation-platonic}.
The non-coverage of logarithmic W, $\mathcal{N}=2$ SCA,
cosets, lattice, and root-of-unity cases is a scope statement;
each of those families carries its own structural feature
(non-reductive Hopf structure, mixed bosonic/fermionic strong
generators, quotient geometry, modular-data constraint) that is
not governed by any of the five listed mechanisms. The
$\mathcal{N}=2$ SCA has a candidate reduction via the
Kazama--Suzuki coset
$\mathrm{SCA}_c \simeq
\operatorname{Com}(\mathrm{U}(1), \widehat{\mathfrak{sl}}_{2,k}
\otimes \psi\bar\psi)$, which would route MC3 through the affine
Yangian input and a coset-preservation theorem if that theorem held;
this is
Conjecture~\ref{conj:mc3-n2-sca-kazama-suzuki}, not a theorem.
\end{proof}

\begin{remark}[Residual sector landscape]
\label{rem:residual-sector-landscape}
The sectors not covered by the five families split by their
obstruction to the corresponding Shapovalov-type pairing:
\begin{center}
\renewcommand{\arraystretch}{1.15}
\begin{tabular}{ll}
\toprule
\textbf{Sector} & \textbf{Obstruction to five-family mechanism} \\
\midrule
logarithmic $W(p)$ & non-semisimple Virasoro, indecomposable
 reducibles \\
$\mathcal{N}=2$ SCA & mixed bosonic/fermionic strong
 generators, candidate Kazama--Suzuki coset
 reduction (Conj.~\ref{conj:mc3-n2-sca-kazama-suzuki}) \\
cosets $G/H$ & affine-quotient geometry, non-principal reduction \\
lattice VOAs & non-rational weight lattices, theta-series
 modularity \\
roots of unity & finite-dimensional simples, modular tensor
 category \\
\bottomrule
\end{tabular}
\end{center}
Each residual sector is a separate investigation with its own
thick-generation landscape. The five-family synthesis is the
baseline uniform theorem on evaluation-type categories; the
residual sectors require the respective structural input
(Verlinde modularity, chiral-ring truncation, Drinfeld-Sokolov
reduction, theta-series representation theory,
modular-tensor-category rigidity).
\end{remark}

\section{Falsification surfaces}
\label{sec:mc3-platonic-falsification-surfaces}

\begin{remark}[Tests for the MC3 boundary]
\label{rem:mc3-boundary-falsification}
\index{MC3!falsification surfaces}%
The theorem on the evaluation core can fail only through a
family-specific witness: a non-generic Shapovalov determinant, a false
fundamental $q$-character multiplicity, an elliptic Bethe degeneracy
off~$\Theta$, or a Berezinian parity-balance failure off its hyperplane.
The finite-window upgrade can fail only through the tower: a compact
window not generated by $\tau_{\le N}\mathcal S$, transition functors
that do not preserve the generated compact subcategories, a nonzero
$\varprojlim^1$ for cone cohomology, or a finite-window cone that is
acyclic as a raw complex but not coacyclic in the second-kind coderived
category. The full-DK upgrade can fail without touching either theorem:
a compact object of $\mathrm{DK}_\mathfrak g$ may lie outside the
essential image of the chosen finite-window completion. Derived-centre
and Theorem~H failures are separate: they attack the Morita kernel or
the symmetric Hochschild averaging step, not thick generation of
$\mathrm{DK}^{\mathrm{ev}}_\mathfrak g$. A class~$\mathsf M$ failure
at the raw direct-sum level attacks the uncompleted
$\mathrm{Ch}(\mathrm{Vect})$ presentation; it does not refute the
finite-window or pro-completed MC3 statement unless it produces a
non-Mittag--Leffler tower in the chosen DK topology.  A K3/Mukai
failure attacks this chapter only if the symmetric Mukai
$\mathfrak{so}(4,20)$ or Hodge-parity $\mathfrak{so}(4\mid20)$
envelope is falsely replaced by Kac $\mathfrak{osp}(4\mid20)$.
\end{remark}
