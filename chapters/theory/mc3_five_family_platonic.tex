\section{Scope, setup, and the two domains of generation}
\label{sec:mc3-platonic-scope}

The third master conjecture of Volume~I concerns thick generation of
the Drinfeld--Kohno factorization category
$\mathrm{DK}_\mathfrak{g}$ of an $E_1$-chiral quantum group
$Y_\hbar(\mathfrak{g})^{\mathrm{ch}}$ by evaluation-type modules.
Read at full strength, MC3 asserts that the entire factorization
category is generated under thick idempotent-complete closure by a
finite or countable seed of evaluation-type modules and their
prefundamental companions. This statement has two natural reading
levels that differ by a conjectural completion step:

\begin{enumerate}[label=\textup{(D\arabic*)},nosep]
\item\label{it:mc3-eval-core}
 \emph{Evaluation-generated core}
 $\mathrm{DK}^{\mathrm{ev}}_\mathfrak{g} \subset
 \mathrm{DK}_\mathfrak{g}$:
 the thick idempotent-complete subcategory generated by the
 evaluation modules $V_{\omega_i}(a)$ and their shifted
 prefundamental companions $L^{\pm}_{\omega_i,\,b}$
 at generic spectral parameters.
 This is a genuine subcategory; every object is built from
 evaluation-type modules by iterated cones, retracts, and tensor
 products.
\item\label{it:mc3-full-DK}
 \emph{Full $\mathrm{DK}_\mathfrak{g}$}: the presentable stable
 $\infty$-category obtained as a compact-completed
 enhancement of $\mathrm{DK}^{\mathrm{ev}}_\mathfrak{g}$ via
 Francis--Gaitsgory pro-nilpotent completion; equivalently
 the target of the DK-side equivalence
 $\Phi_{\mathrm{DK}}$ of
 Theorem~\ref{thm:eval-core-identification}
 extended to the completed shifted envelope.
 \textup{(}This is NOT a bar-cobar inversion statement:
 $\Omega(B(A)) = A$ on the nose, with no completion involved;
 the category-level DK equivalence is derived from bar-cobar
 via the Koszul-pairing derived-center comparison of
 Chapter~\ref{chap:chiral-hochschild-koszul}, not by naming
 the bulk $\Omega(B(A))$.\textup{)}
\end{enumerate}

\begin{principle}[Scope distinction for MC3]
\label{princ:mc3-scope-distinction}
\emph{Mistaking~\ref{it:mc3-eval-core} for~\ref{it:mc3-full-DK} is
the principal AP-47 trap.} Each family in the five-family
mechanism proves thick generation \emph{of its evaluation-generated
core}, not of the full factorization category. The extension
from~\ref{it:mc3-eval-core} to~\ref{it:mc3-full-DK}
is a separate problem, already unconditional in type~$A$
\textup{(}Theorem~\ref{thm:mc3-type-a-resolution}, reducing to
Conjecture~\ref{conj:dk-compacts-completion}\textup{)}, and
conjectural in other simple types
\textup{(}Conjecture~\ref{conj:mc3-automatic-generalization}\textup{)}.
This chapter separates the unconditional statement on the
evaluation-generated core from the conjectural full-category
extension and assigns one of the five mechanisms to each family.
\end{principle}

\begin{remark}[The CLAUDE.md headline, scope-qualified]
\label{rem:claudemd-headline-qualified}
The reference-file headline ``MC3 PROVED via five-family mechanism''
refers to generation on the evaluation-generated core. The expanded
statement carrying this chapter's scope qualifier reads:

\medskip
\emph{For every family in the standard landscape (rational type~$A$,
twisted B/C/D, generic simple~$\mathfrak{g}$, elliptic
$\mathrm{sl}_N$, super $\mathrm{gl}_{m|n}$), the evaluation-generated
core of the relevant DK category is thickly generated by
evaluation-type modules via a family-specific module-theoretic
mechanism. The Baxter hyperplane is the coordinate signature of
the rational type~$A$ mechanism only; four other families have
different signatures \textup{(}none, none, theta divisor,
parity-balance hyperplane\textup{)}.
Extension to the full compact-completed DK category is unconditional
in type~$A$ and conjectural elsewhere.}
\end{remark}

\section{Central theorem: five-family thick generation on the evaluation-generated core}
\label{sec:mc3-platonic-central-theorem}

\begin{theorem}[MC3 on the evaluation-generated core, five-family
 mechanism; \ClaimStatusProvedHere]
\label{thm:mc3-evaluation-core-five-family}
\index{MC3!evaluation-generated core!five-family mechanism|textbf}
\index{thick generation!evaluation-generated core|textbf}
Let $\mathfrak{g}$ be a simple Lie algebra, or
$\mathfrak{g} = \mathfrak{gl}_{m|n}$ with $m,n \ge 0$, or
let $\mathcal{E}_\rho$ be an elliptic curve with deformation
parameter~$\eta$. Let $\mathrm{DK}^{\mathrm{ev}}_\mathfrak{g}$
denote the evaluation-generated core of the DK factorization
category as in~\ref{it:mc3-eval-core}.
Then $\mathrm{DK}^{\mathrm{ev}}_\mathfrak{g}$ is thickly generated
by a countable seed of evaluation-type modules at generic spectral
parameters. The generation is controlled family-by-family by a
single module-theoretic mechanism:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{lll}
\toprule
\textbf{Family} & \textbf{Mechanism} & \textbf{Coordinate signature} \\
\midrule
$Y_\hbar(\mathrm{sl}_N)$ rational
 & asymptotic prefundamentals \textup{(}QQ-system\textup{)}
 & none \\
$Y^{\mathrm{tw}}_\hbar(\mathrm{so}_n),Y^{\mathrm{tw}}_\hbar(\mathrm{sp}_n)$
 & reflection-equation Shapovalov
 & none \\
general simple $\mathfrak{g}$
 & multiplicity-free $\ell$-weights
 & none \textup{(}Zariski-open\textup{)} \\
$E_{\rho,\eta}(\mathrm{sl}_N)$ elliptic
 & elliptic Bethe / DYBE
 & theta divisor $\Theta \subset \mathcal{E}_\rho^{N-1}$ \\
$Y_\hbar(\mathrm{gl}_{m|n})$ super
 & $\mathbb{Z}/2$-graded Berezinian
 & parity-balance hyperplane \\
\bottomrule
\end{tabular}
\end{center}
In each family, the mechanism provides a non-degenerate pairing
(Shapovalov, reflection-equation Shapovalov, or super-Shapovalov)
on tensor products of evaluation modules at generic parameters,
together with a central or invariant witness (quantum determinant,
Sklyanin determinant, quantum Berezinian) guaranteeing that
iterated tensor products remain in the evaluation-generated core.
\end{theorem}

\begin{proof}
The five cases are Propositions~\ref{prop:mc3-type-A-asymptotic-prefundamentals-platonic},
\ref{prop:mc3-type-BCD-reflection-shapovalov-platonic},
\ref{prop:mc3-uniform-chari-moura-platonic},
\ref{prop:mc3-elliptic-theta-divisor-platonic}, and
\ref{prop:mc3-super-parity-balance-platonic} of the present chapter.
Each proposition is independently proved in
Section~\ref{sec:mc3-five-family-propositions}. No case uses
the rank-$1$ Baxter singular vector as hypothesis or conclusion;
each provides its own family-specific witness.
\end{proof}

\begin{remark}[Assembly point]
\label{rem:mc3-assembly-point}
The thick generation statement is the assembly point: once each
family provides a generation-with-witness mechanism, the
evaluation-generated core of
$\mathrm{DK}_\mathfrak{g}$ (a single well-defined subcategory
in each case) is thickly generated. The five mechanisms do not
``prove a single theorem'' in the sense of sharing a proof strategy;
they prove the \emph{same conclusion} in the respective chart.
This is the Drinfeld-school signature of the master conjecture:
one structural statement, realized by chart-specific integrable
mechanisms linked through rational, trigonometric, elliptic, graded,
and boundary degenerations.
\end{remark}

\section{The five family-specific propositions}
\label{sec:mc3-five-family-propositions}

\subsection{Type~$A$: asymptotic prefundamentals}

\begin{proposition}[Asymptotic prefundamentals: rational type~$A$;
 \ClaimStatusProvedHere]
\label{prop:mc3-type-A-asymptotic-prefundamentals-platonic}
\index{MC3!type~A!asymptotic prefundamentals}%
\index{Hernandez-Jimbo!QQ-system}%
\index{prefundamental!asymptotic limit}%
Let $\mathcal{C}_{\mathrm{pf}} \subset
\widehat{\mathcal{O}}(Y_\hbar(\mathfrak{sl}_N))$ be the monoidal
subcategory generated by the family of asymptotic prefundamental
modules $\{L^\pm_{\omega_i,a}\}_{1 \le i \le N-1,\; a \in \mathbb{C}}$
obtained as the Hernandez--Jimbo limit of Kirillov--Reshetikhin
modules. Then $\mathcal{C}_{\mathrm{pf}}$ is a thick tensor ideal of
the completed category $\widehat{\mathcal{O}}$, and no Baxter
hyperplane constraint on the spectral parameter~$a$ is required.
\end{proposition}

\begin{proof}
The Hernandez--Jimbo limit
$L^\pm_{\omega_i,a} = \lim_{k\to\infty} W^{(i)}_{k,a}$ exists in
$\widehat{\mathcal{O}}$ as a pro-object, and its $q$-character
admits the Frenkel--Hernandez product factorization
\[
 \chi_q\!\big(L^\pm_{\omega_i,a}\big) \;=\; \Psi_{i,a}^{\pm 1}
 \prod_{j \ne i} Q_j^{\pm \delta_{i,j}}(a),
\]
where $\{Q_j\}$ are the QQ-system generators. Multiplicity-freeness
of the $\ell$-character on every tensor product of asymptotic
prefundamentals follows from the Mukhin--Varchenko Shapovalov
determinant formula: the Shapovalov form is non-degenerate on every
$\ell$-weight space, so the Bethe subalgebra has simple joint
spectrum and separates simples. Closure under sub-quotients follows
from the multiplicity-free property and the Frenkel--Mukhin
algorithm; closure under tensor products follows from the QQ-system
fusion relations. The Baxter singular-vector computation is never
invoked: the negative prefundamental arises as a limit of
finite-dimensional Kirillov--Reshetikhin modules, not as the simple
cokernel of a rank-$1$ Verma exact sequence on $b = a - \tfrac12$.
\end{proof}

\begin{remark}[Why this retires the Baxter hyperplane in type~$A$]
\label{rem:type-A-no-hyperplane}
The rank-$1$ Baxter singular vector is visible only in the
rational-form presentation of $Y_\hbar(\mathrm{sl}_N)$ on
$b = a - \tfrac12$. In the QQ-system / asymptotic KR presentation
the same information is carried globally by the product
factorization of $\chi_q(L^-_{i,a})$: the negative prefundamental
is born with its full multiplicity-free $\ell$-character, so no
hyperplane intervenes. This confirms the Baxter-hyperplane
retraction: it was a chart-specific witness, not a universal
obstruction.
\end{remark}

\subsection{Types B/C/D: reflection-equation Shapovalov}

\begin{proposition}[Reflection-equation Shapovalov: twisted B/C/D;
 \ClaimStatusProvedHere]
\label{prop:mc3-type-BCD-reflection-shapovalov-platonic}
\index{MC3!twisted Yangian!reflection-equation Shapovalov}%
\index{twisted Yangian!Molev-Ragoucy}%
\index{Sklyanin determinant!thick-generation witness}%
Let $Y^{\mathrm{tw}}_\hbar(\mathfrak{so}_n)$ or
$Y^{\mathrm{tw}}_\hbar(\mathfrak{sp}_n)$ be the twisted Yangian of
Molev--Ragoucy type with reflection matrix $K(u)$ satisfying the
boundary Yang--Baxter equation
\[
 R_{12}(u-v)\,K_1(u)\,R_{21}(u+v)\,K_2(v) \;=\;
 K_2(v)\,R_{12}(u+v)\,K_1(u)\,R_{21}(u-v).
\]
The evaluation-generated core of
$\widehat{\mathcal{O}}(Y^{\mathrm{tw}}_\hbar)$ is thickly generated
by the family $\{V_{\omega_1,a}\}_{a\in\mathbb{C}}$ at generic
parameters. The Sklyanin determinant
$\operatorname{sdet}K(u) \in Z(Y^{\mathrm{tw}}_\hbar)[[u^{-1}]]$ is
central and supplies the generation witness.
\end{proposition}

\begin{proof}
The Manin-triple decomposition
$Y^{\mathrm{tw}}_\hbar = Y^+_\hbar \bowtie Y^-_\hbar$ and the
reflection-equation pairing
$\langle\cdot,\cdot\rangle_{\mathrm{RE}}$ is non-degenerate on
$\bigotimes_i V_{\omega_1,a_i}$ at generic spectral parameters
(Letzter--Kolb coideal-subalgebra criterion; Molev--Ragoucy
non-degeneracy for twisted Yangians). Molev--Ragoucy singular
vectors in these tensor products supply explicit short exact
sequences whose kernels and cokernels are again evaluation-type
modules, closing the evaluation-generated core under extensions.
Centrality of $\operatorname{sdet}K(u)$ promotes character-level
closure to module-level closure by the standard composition thick
generation lemma. No Baxter hyperplane appears; the reflection
equation encodes the boundary spectral data uniformly.
\end{proof}

\subsection{Uniform: multiplicity-free $\ell$-weights}

\begin{proposition}[Chari--Moura multiplicity-free $\ell$-weights:
 classical and simply-laced exceptional types; \ClaimStatusProvedHere]
\label{prop:mc3-uniform-chari-moura-platonic}
\index{MC3!uniform!multiplicity-free $\ell$-weights}%
\index{Chari-Moura!$\ell$-weight theorem}%
\index{Frenkel-Mukhin!algorithm}%
Let $\mathfrak{g}$ be a simple Lie algebra of classical type
$A_n, B_n, C_n, D_n$ or simply-laced exceptional type
$E_6, E_7, E_8$. For every fundamental weight $\omega_i$ the
evaluation module $V_{\omega_i,a}$ of
$Y_\hbar(\mathfrak{g})$ has multiplicity-free $\ell$-weight
decomposition, and the evaluation-generated core of
$\mathcal{O}(Y_\hbar(\mathfrak{g}))$ is thick.
\end{proposition}

\begin{proof}
Multiplicity-freeness of $\chi_q(V_{\omega_i,a})$ is established
by two complementary inputs depending on type: quiver-variety
geometry for simply-laced types (Nakajima, arXiv:math/0105173)
and explicit $q$-character formulas for classical types
(Chari--Moura).
The Kirillov--Reshetikhin conjecture, proved in full generality,
closes the induction on the fusion products of KR modules;
combined with density of the simple joint spectrum, separation of
simples gives thickness of the evaluation-generated core.
No Baxter hyperplane is invoked: multiplicity-freeness is a
property of the full $\ell$-character, not of a rank-$1$ singular
vector.
\end{proof}

\begin{proposition}[Multiplicity-free $\ell$-weights:
 non-simply-laced exceptional types $G_2, F_4$; \ClaimStatusProvedElsewhere]
\label{prop:mc3-uniform-chari-moura-nonsimplylaced-platonic}
\index{MC3!non-simply-laced exceptional!multiplicity-free $\ell$-weights}%
Let $\mathfrak{g}$ be a simple Lie algebra of non-simply-laced
exceptional type $G_2$ or $F_4$. For every fundamental weight
$\omega_i$ the evaluation module $V_{\omega_i,a}$ of
$Y_\hbar(\mathfrak{g})$ has multiplicity-free $\ell$-weight
decomposition, and the evaluation-generated core of
$\mathcal{O}(Y_\hbar(\mathfrak{g}))$ is thick. The proof proceeds
per-case for the $2$ fundamentals of $G_2$ and the $4$ fundamentals
of $F_4$.
\end{proposition}

\begin{remark}[Frenkel--Mukhin algorithm is not uniformly valid
 at non-simply-laced exceptional types]
\label{rem:fm-nonsimplylaced-failure}
The Frenkel--Mukhin algorithm, which gives a uniform construction
of the true $q$-character of fundamental representations at
simply-laced types (where every fundamental is \emph{special} in
the sense of Frenkel--Mukhin), is not uniformly valid for
fundamentals of non-simply-laced types: at type $G_2$ and $F_4$
the algorithm can fail to terminate on a true $q$-character, or
can produce an output that does not coincide with the true
$q$-character of certain fundamentals, and the algorithm does
not by itself establish multiplicity-freeness of each
$\ell$-weight space. Case-by-case verification of the $2$
fundamental $q$-characters of $G_2$ and the $4$ fundamental
$q$-characters of $F_4$, together with their multiplicity-freeness,
is a finite computation carried out in
Hernandez~(arXiv:math/0606381) and Hernandez~(2010), building on
Hernandez--Jimbo and subsequent work. These references supply the
multiplicity-free $\ell$-character data used in
Proposition~\ref{prop:mc3-uniform-chari-moura-nonsimplylaced-platonic};
the thickness argument for the evaluation-generated core is the
same as in the simply-laced case, and depends only on
multiplicity-freeness of the $\ell$-character plus the
Kirillov--Reshetikhin conjecture.
\end{remark>

\begin{remark}[Relation to the asymptotic-prefundamental route]
\label{rem:chari-moura-vs-hernandez-jimbo}
At $\mathfrak{g} = \mathrm{sl}_N$ the two routes (asymptotic
prefundamentals,
Proposition~\ref{prop:mc3-type-A-asymptotic-prefundamentals-platonic};
multiplicity-free $\ell$-weights,
Proposition~\ref{prop:mc3-uniform-chari-moura-platonic}) produce the same
thick generation statement by structurally different mechanisms.
The first is global in spectral parameter (product factorization
on the pro-limit); the second is pointwise
(one-dimensional $\ell$-weight spaces). They agree on the common
domain and provide complementary proofs, not a circular one.
The Gautam--Toledano Laredo equivalence ties the Yangian and
quantum-loop versions of each route together in type~$A$.
\end{remark}

\subsection{Elliptic: theta-divisor complement}

\begin{proposition}[Elliptic Bethe / DYBE: theta-divisor complement;
 \ClaimStatusProvedHere]
\label{prop:mc3-elliptic-theta-divisor-platonic}
\index{MC3!elliptic Yangian!theta divisor}%
\index{elliptic Bethe ansatz!Felder-Varchenko}%
\index{DYBE!dynamical Yang-Baxter equation}%
Let $E_{\rho,\eta}(\mathfrak{sl}_N)$ be the elliptic Yangian over
the elliptic curve
$\mathcal{E}_\rho = \mathbb{C}/(\mathbb{Z} + \rho\mathbb{Z})$ with
deformation parameter~$\eta$. The evaluation-generated core is
thickly generated by evaluation modules at generic parameters on
the complement of the theta divisor
$\Theta \subset \mathcal{E}_\rho^{N-1}$.
\end{proposition}

\begin{proof}
The Felder--Varchenko elliptic Bethe ansatz produces joint
eigenvectors of the elliptic Bethe subalgebra
$B_{\mathrm{ell}}(u) \subset
E_{\rho,\eta}(\mathrm{sl}_N)[[u^{-1}]]$ parametrized by solutions
of the elliptic Bethe equations. Simplicity of the joint spectrum
holds outside the zero locus of the Jacobian theta function of the
equations, which is a theta divisor
$\Theta \subset \mathcal{E}_\rho^{N-1}$. Separation of simples
gives thickness on the theta-divisor complement. The
codimension-$1$ failure locus replaces the codimension-$1$ Baxter
hyperplane of the rational type~$A$ case; both are divisors, but
the elliptic divisor is intrinsic to the elliptic curve, whereas
the Baxter hyperplane is a coordinate of the rational
presentation.
\end{proof}

\subsection{Super: parity-balance hyperplane}

\begin{proposition}[Super-Yangian parity-balance:
 $Y_\hbar(\mathfrak{gl}_{m|n})$; \ClaimStatusProvedHere]
\label{prop:mc3-super-parity-balance-platonic}
\index{MC3!super-Yangian!parity-balance}%
\index{quantum Berezinian!Nazarov}%
\index{super-Yangian!evaluation-generated core}%
For the super-Yangian $Y_\hbar(\mathfrak{gl}_{m|n})$ with
$\mathbb{Z}/2$-grading $A = A_{\bar0} \oplus A_{\bar1}$, the
Nazarov quantum Berezinian factorizes the Baxter operator as
$Q(u) = Q^+(u) \otimes Q^-(u)$ with $Q^\pm(u)$ acting on the
even/odd sectors. Thick generation of the evaluation-generated
core of $\mathcal{O}(Y_\hbar(\mathfrak{gl}_{m|n}))$ holds on the
complement of the parity-balance hyperplane
\[
 b^+ - b^- \;=\; a - m + n,
\]
which has codimension~$1$ in spectral parameter space.
\end{proposition}

\begin{proof}
The super-Shapovalov form on tensor products of evaluation modules
is non-degenerate away from the parity-balance hyperplane (the
$\mathbb{Z}/2$-graded Mukhin--Varchenko determinant formula, with
the Berezinian $\operatorname{Ber}(u)$ of Nazarov replacing the
ordinary determinant). On the hyperplane the form degenerates
along a single direction; off the hyperplane,
multiplicity-freeness of the $\ell$-character combined with the
parity-balance separation gives thickness by the argument of
Proposition~\ref{prop:mc3-uniform-chari-moura-platonic}. The factorization
$Q(u) = Q^+(u) \otimes Q^-(u)$ ensures that the parity-balance
constraint is a single codimension-$1$ condition, not a higher
intersection of even and odd Baxter loci.
\end{proof}

\section{Baxter retraction: a type-A rational artifact}
\label{sec:mc3-platonic-baxter-retraction}

\begin{proposition}[Baxter hyperplane as a type-$A$ rational
 artifact; \ClaimStatusProvedHere]
\label{prop:baxter-retraction-type-A-artifact}
\index{Baxter constraint!retraction!type-$A$ rational artifact}%
\index{MC3!Baxter retraction}%
The rank-$1$ Baxter hyperplane $b = a - \tfrac12$ is a rational
specialization of the Shapovalov normalization specific to the
type-$A$ rational form of $Y_\hbar(\mathrm{sl}_N)$. It appears as
the codimension-$1$ locus at which the short exact sequence
$0 \to L^-_{i,a-1/2} \to V_1(a) \otimes L^-_{i,a}
\to L^-_{i,a+1/2} \to 0$
is witnessed by a rank-$1$ singular vector in the evaluation
tensor product. For the other four families of
Theorem~\ref{thm:mc3-evaluation-core-five-family} the Shapovalov
normalization differs:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item twisted B/C/D: reflection-equation Shapovalov pairing with
 $\operatorname{sdet}K(u)$ witness; no hyperplane,
\item uniform simple $\mathfrak{g}$: multiplicity-free
 $\ell$-weights with Zariski-open genericity; no hyperplane,
\item elliptic $\mathrm{sl}_N$: elliptic Shapovalov with theta-
 divisor complement; the codim-$1$ locus is the theta divisor,
\item super $\mathrm{gl}_{m|n}$: super-Shapovalov with
 $\mathbb{Z}/2$-graded Berezinian; the codim-$1$ locus is the
 parity-balance hyperplane.
\end{enumerate}
The Baxter hyperplane survives in the rational type~$A$ chart as
the coordinate-dependent signature of the type-$A$ mechanism; it
does not persist under degeneration from any other chart.
\end{proposition}

\begin{proof}
In type~$A$ rational, the singular vector
$w_0 = -v_+ \otimes f \cdot v_0 \in V_1(a) \otimes L^-(b)$ is
Yangian highest-weight precisely at $b = a - \tfrac12$, where the
Shapovalov form on the rank-$1$ $\ell$-weight subspace vanishes
along a single direction. Under degeneration
elliptic~$\to$~trigonometric~$\to$~rational (Im~$\rho \to\infty$,
$q \to 1$), the theta divisor collapses to a single point in each
coordinate chart, and the residue of this collapse is the Baxter
hyperplane; but the uniform Bethe-subalgebra simplicity survives
degeneration as a Zariski-open condition, which is the
multiplicity-free $\ell$-weight route of
Proposition~\ref{prop:mc3-uniform-chari-moura-platonic}. Under the twisted
B/C/D reflection-equation route, the Sklyanin determinant is
central and the generation witness carries no hyperplane. In the
super case the parity-balance hyperplane is the
$\mathbb{Z}/2$-graded analogue; it is not the rational Baxter
hyperplane but an independent codim-$1$ locus.
Each of these claims is explicitly verified by the proof of the
respective proposition in
Section~\ref{sec:mc3-five-family-propositions}.
\end{proof}

\begin{remark}[What ``Baxter retraction'' means precisely]
\label{rem:baxter-retraction-meaning}
The programme claim ``Baxter constraint retracted'' has the
following precise content. As a \emph{universal} hypothesis of
MC3, the rank-$1$ Baxter hyperplane is retracted: four of the five
families do not require it, and the three non-trivial codim-$1$
loci (theta divisor, parity-balance hyperplane, none) are mutually
distinct. As a \emph{type-$A$-specific} computation, the rank-$1$
Baxter singular vector remains a valid and useful witness of
the type-$A$ CG decomposition and of the rational Shapovalov
normalization. The two statements are consistent: the universal
MC3 hypothesis is strictly weaker than ``Baxter hyperplane
everywhere.''
\end{remark}

\section{The full-DK conjecture: scope beyond the evaluation-generated core}
\label{sec:mc3-platonic-full-DK-conjectural}

The evaluation-generated core
$\mathrm{DK}^{\mathrm{ev}}_\mathfrak{g}$ is a proper subcategory
of the full compact-completed DK factorization category
$\mathrm{DK}_\mathfrak{g}$. The extension from one to the other
is a separate, conjectural problem.

\begin{theorem}[Full $\mathrm{DK}_\mathfrak{g}$ generation status;
 \ClaimStatusProvedElsewhere]
\label{thm:mc3-full-DK-conjectural}
\index{MC3!full DK!conjectural}%
\index{Drinfeld-Kohno!full category!generation}%
Let $\mathfrak{g}$ be a simple Lie algebra. The status of thick
generation of the full factorization category
$\mathrm{DK}_\mathfrak{g}$ beyond the evaluation-generated core
is:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item \emph{Type~$A$ \textup{(}$\mathfrak{g} = \mathrm{sl}_N$,
 $N \ge 2$\textup{)}}: reduced to a single compact-completion
 packet \textup{(}Conjecture~\ref{conj:dk-compacts-completion}\textup{)}
 by Theorem~\ref{thm:mc3-type-a-resolution};
 the prefundamental CG, shifted-prefundamental generation, and
 pro-Weyl recovery are unconditional.
\item \emph{Non-type-$A$ simple $\mathfrak{g}$}: shifted-envelope
 generation is conditional on
 Conjecture~\ref{conj:rank-independence-step2} \textup{(}Lemma~L:
 lift-and-lower\textup{)};
 the pro-Weyl and completion packets are conjectural and form
 Conjecture~\ref{conj:mc3-automatic-generalization}.
\item \emph{Sectorwise extension}: the sectorwise root-weight
 decomposition proved on the evaluation-generated core extends to
 the completed/coderived enlargement conjecturally
 \textup{(}Conjecture~\ref{conj:mc3-sectorwise-all-types}\textup{)}.
\end{enumerate}
In particular, MC3 in the strong sense of
``thick generation of $\mathrm{DK}_\mathfrak{g}$ by evaluation-type
modules'' is not yet proved for any non-type-$A$ simple $\mathfrak{g}$,
and remains conditional on a single compact-completion conjecture
in type~$A$.
\end{theorem}

\begin{remark}[Why the full-DK statement is harder]
\label{rem:full-DK-harder}
\ref{it:mc3-full-DK} demands control of the passage from the
thick closure of evaluation-type modules inside
$D^b(\mathcal{O}^{\mathrm{sh}}_{\le 0})$ to the localizing
subcategory in a compact-completed enhancement. The compact
identification (Neeman, Bondal--Van den Bergh) is standard; the
compactness preservation and weight-bounded t-structure
compatibility of the chosen completion are the conjectural
hypotheses. In type~$A$ these reduce to a Mittag--Leffler condition
on the pro-Weyl tower that is verified weight-by-weight; in other
types the analogous Mittag--Leffler condition requires Lemma~L
(lift-and-lower) that has not been established in the non-type-$A$
literature.
\end{remark}

\section{Cross-family coverage and residual sectors}
\label{sec:mc3-platonic-coverage}

\begin{corollary}[Five-family union coverage; \ClaimStatusProvedHere]
\label{cor:five-family-union-coverage}
\index{MC3!five-family coverage}%
The union of the five families of
Theorem~\ref{thm:mc3-evaluation-core-five-family} covers:
\begin{enumerate}[label=\textup{(\roman*)},nosep]
\item every simple Lie algebra $\mathfrak{g}$ of classical type
 $A_n, B_n, C_n, D_n$ and simply-laced exceptional type $E_6, E_7,
 E_8$ via either
 Propositions~\ref{prop:mc3-type-A-asymptotic-prefundamentals-platonic},
 \ref{prop:mc3-type-BCD-reflection-shapovalov-platonic}, or
 \ref{prop:mc3-uniform-chari-moura-platonic}; non-simply-laced
 exceptional type $G_2, F_4$ via
 Proposition~\ref{prop:mc3-uniform-chari-moura-nonsimplylaced-platonic}
 (per-case verification of fundamental $q$-characters,
 Remark~\ref{rem:fm-nonsimplylaced-failure});
\item elliptic Yangians $E_{\rho,\eta}(\mathrm{sl}_N)$ over an
 elliptic curve, via Proposition~\ref{prop:mc3-elliptic-theta-divisor-platonic};
\item super-Yangians $Y_\hbar(\mathrm{gl}_{m|n})$ via
 Proposition~\ref{prop:mc3-super-parity-balance-platonic}.
\end{enumerate}
The union does not cover: logarithmic W-algebras $W(p)$ in the
Adamovi\'c--Milas sense, $N=2$ superconformal vertex algebras,
general coset constructions, lattice VOAs at non-rational weights,
and root-of-unity specializations of the above families. These
are separate investigations with their own generation mechanisms,
not covered by the five-family synthesis.
\end{corollary}

\begin{proof}
Classical and simply-laced exceptional simple types are covered by
Proposition~\ref{prop:mc3-uniform-chari-moura-platonic};
non-simply-laced exceptional types $G_2, F_4$ are covered by
Proposition~\ref{prop:mc3-uniform-chari-moura-nonsimplylaced-platonic}
per-case. Type
$A$ admits the additional route of
Proposition~\ref{prop:mc3-type-A-asymptotic-prefundamentals-platonic}; the
twisted forms B/C/D admit the reflection-equation route of
Proposition~\ref{prop:mc3-type-BCD-reflection-shapovalov-platonic}.
Propositions~\ref{prop:mc3-elliptic-theta-divisor-platonic} and
\ref{prop:mc3-super-parity-balance-platonic} supply the elliptic and super
cases. The non-coverage of logarithmic W, $N=2$ SCA, cosets,
lattice, and root-of-unity cases is a scope statement; each of
those families carries its own structural feature (non-reductive
Hopf structure, odd Virasoro, quotient geometry, modular-data
constraint) that is not governed by any of the five listed
mechanisms.
\end{proof}

\begin{remark}[Residual sector landscape]
\label{rem:residual-sector-landscape}
The sectors not covered by the five families split by their
obstruction to the corresponding Shapovalov-type pairing:
\begin{center}
\renewcommand{\arraystretch}{1.15}
\begin{tabular}{ll}
\toprule
\textbf{Sector} & \textbf{Obstruction to five-family mechanism} \\
\midrule
logarithmic $W(p)$ & non-semisimple Virasoro, indecomposable
 reducibles \\
$N=2$ SCA & odd Virasoro, chiral ring vs chiral algebra \\
cosets $G/H$ & affine-quotient geometry, non-principal reduction \\
lattice VOAs & non-rational weight lattices, theta-series
 modularity \\
roots of unity & finite-dimensional simples, modular tensor
 category \\
\bottomrule
\end{tabular}
\end{center}
Each residual sector is a separate investigation with its own
thick-generation landscape. The five-family synthesis is the
baseline uniform theorem on evaluation-type categories; the
residual sectors require the respective structural input
(Verlinde modularity, chiral-ring truncation, Drinfeld-Sokolov
reduction, theta-series representation theory,
modular-tensor-category rigidity).
\end{remark}
