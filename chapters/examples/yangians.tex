\chapter{Yangians and shifted Yangians}
\label{chap:yangians}

The Yangian $Y(\mathfrak{g})$ is an $\Eone$-chiral algebra: associative but not commutative, local but not fully local.  Its bar complex uses ordered configurations on the real line rather than on a complex curve, Fulton--MacPherson compactifications of real configuration spaces replace their complex counterparts, and Verdier duality inverts the $R$-matrix.  The resulting Koszul duals involve quantum groups $U_q(\mathfrak{g})$ rather than vertex algebras at shifted levels.

\begin{remark}[Yangian as atom: the braided face]
\label{rem:yangian-atom}
\index{Yangian!as modular Koszul atom (conditional on $\Eone$-Koszulness)}
In the ``atoms'' framing of Remark~\ref{rem:free-field-atoms},
the Yangian reveals the fourth irreducible face of modular Koszul
duality (conditional on $\Eone$-Koszulness; see
Proposition~\ref{prop:yangian-koszul}): \emph{braiding}.  The free fields, Kac--Moody, and
$\mathcal{W}$-algebras are all $\Einf$-chiral
(commutative/local), where Verdier duality acts by level
inversion.  The Yangian is $\Eone$-chiral (associative/ordered),
and Verdier duality acts by $R$-matrix inversion
$R \mapsto R^{-1}$ --- the homotopy-coherent shadow of the
master diagram~\eqref{eq:master-square} in the braided setting.
The derived Drinfeld--Kohno conjecture
(Conjecture~\ref{conj:derived-drinfeld-kohno}) articulates the
horizon: the $\Eone$-factorization categories of $Y(\mathfrak{g})$
and $U_q(\mathfrak{g})^{\mathrm{op}}$ should be equivalent,
completing the commutative theory with its noncommutative partner.
\end{remark}

\section{Yangian vertex algebra}
\label{sec:yangian-vertex}

\subsection{Yangian definition}

\begin{definition}[Yangian]\label{def:yangian}
\index{Yangian|textbf}
The \emph{Yangian} $Y(\mathfrak{g})$ associated to a simple Lie algebra $\mathfrak{g}$ 
is the associative algebra generated by $J^{(r)}_a$ for $a = 1, \ldots, \dim\mathfrak{g}$ 
and $r \geq 0$, with relations:
\begin{align*}
[J^{(0)}_a, J^{(0)}_b] &= f_{ab}^c J^{(0)}_c \\
[J^{(0)}_a, J^{(r)}_b] &= f_{ab}^c J^{(r)}_c \\
[J^{(1)}_a, [J^{(1)}_b, J^{(0)}_c]] - [J^{(0)}_a, [J^{(1)}_b, J^{(1)}_c]] &= \hbar^2 A_{abc}^{def} \{J^{(0)}_d, J^{(0)}_e, J^{(0)}_f\}
\end{align*}
where $\{x,y,z\} = \frac{1}{6}\sum_{\sigma \in S_3} x_{\sigma(1)}y_{\sigma(2)}z_{\sigma(3)}$ is the symmetrized triple product ($|S_3| = 6$), and $A_{abc}^{def}$ are structure constants determined by $\mathfrak{g}$ (Drinfeld's ``terrific'' relation).
\end{definition}

\begin{definition}[Yangian current]\label{def:yangian-current}
The \emph{Yangian current} is the generating function:
\[
J_a(u) = \sum_{r \geq 0} J^{(r)}_a u^{-r-1}
\]
\end{definition}

\begin{definition}[RTT presentation]\label{def:yangian-rtt}
\index{Yangian!RTT presentation}
\index{R-matrix}
The \emph{RTT presentation} (Yang--Baxter presentation) of
$Y(\mathfrak{g})$ is generated by a \emph{transfer matrix}
$T(u) = (T_{ij}(u))_{i,j=1}^{\dim V}
\in \mathrm{End}(V) \otimes Y(\mathfrak{g})$
(valued in $\mathrm{End}(V)$ for a fundamental representation~$V$,
not in~$\mathfrak{g}$), satisfying the \emph{RTT relation}:
\begin{equation}\label{eq:rtt-relation}
R_{12}(u-v)\, T_1(u)\, T_2(v)
\;=\; T_2(v)\, T_1(u)\, R_{12}(u-v)
\end{equation}
where $R_{12}(u) = 1 - \hbar P_{12}/u + O(u^{-2})$ is the
Yang $R$-matrix, $P_{12}$ is the permutation, and the subscripts
$1, 2$ indicate the tensor factor in
$\mathrm{End}(V) \otimes \mathrm{End}(V) \otimes Y(\mathfrak{g})$.
Expanding in modes
$T_{ij}(u) = \delta_{ij} + \hbar \sum_{r \geq 0} T_{ij}^{(r)} u^{-r-1}$,
the RTT relation \eqref{eq:rtt-relation} is equivalent to:
\begin{equation}\label{eq:rtt-modes}
[T_{ij}^{(r+1)}, T_{kl}^{(s)}]
- [T_{ij}^{(r)}, T_{kl}^{(s+1)}]
\;=\; T_{kj}^{(r)} T_{il}^{(s)}
- T_{kj}^{(s)} T_{il}^{(r)}.
\end{equation}
The RTT relation is \emph{quadratic} in the entries of $T$, making
this presentation directly amenable to quadratic Koszul duality
(mechanism~(iii) of Remark~\ref{rem:three-koszul-mechanisms}).
The isomorphism with the Drinfeld presentation
(Definition~\ref{def:yangian}) is given by
$J_a^{(0)} = \sum_{i,j} \rho(e_a)_{ij}\, T_{ij}^{(0)}$
and higher modes from the coproduct structure.
\end{definition}


\subsection{Yangian chiral algebra}

\begin{construction}[Yangian vertex algebra]\label{constr:yangian-va}
The \emph{Yangian chiral algebra} $Y(\mathfrak{g})^{\mathrm{ch}}$ is constructed as follows:

\emph{State space.} Completion of $Y(\mathfrak{g})$-modules.

\emph{Vertex operators.} $Y(J^{(r)}_a, z) = \sum_n J^{(r)}_{a,n} z^{-n-r-1}$.

\emph{OPE.} The RTT relation translates to:
\[
J_a(z) J_b(w) = \frac{f_{ab}^c J_c(w)}{z - w} + \frac{\alpha_{ab}^{cd} {:}J_c(w) J_d(w){:}}{(z-w)^2} + \text{regular}
\]
with higher poles from the quadratic Serre relations.
\end{construction}

\begin{theorem}[Yangian as $\Eone$-chiral; \ClaimStatusProvedHere]\label{thm:yangian-e1}
In the RTT presentation, the Yangian $Y(\mathfrak{g})^{\mathrm{ch}}$ carries the structure of a quadratic $\Eone$-chiral algebra (Definition~\ref{def:e1-chiral}; cf.\ \cite[§4.1]{HA}).  The $\Eone$-locality axiom is $R$-twisted commutativity:
\begin{equation}\label{eq:r-twisted-locality}
J_a(z) J_b(w) - R^{ab}_{cd}(z-w)\, J_d(w) J_c(z) = 0.
\end{equation}
For nontrivial $R \neq P$ (the permutation), this is strictly $\Eone$,
not $\Einf$.
\end{theorem}

\begin{proof}
\emph{Step~1: Quadraticity.}  The RTT relation~\eqref{eq:rtt-relation} is manifestly quadratic in the entries $T_{ij}(u)$.  More precisely, expanding $T_1(u) T_2(v)$ and $T_2(v) T_1(u)$ in the mode expansion $T_{ij}(u) = \delta_{ij} + \hbar \sum_r T_{ij}^{(r)} u^{-r-1}$, the relation~\eqref{eq:rtt-modes} expresses a quadratic relation among the generators $\{T_{ij}^{(r)}\}$.

\emph{Step~2: $\Eone$-chiral structure.}
The vertex operators $Y(T_{ij}^{(r)}, z) = \sum_n T_{ij,n}^{(r)} z^{-n-r-1}$ define fields on a curve $X$.  The OPE:
\[
T_{ij}(z,u)\, T_{kl}(w,v) \sim \frac{R^{ij,kl}(u-v)}{z-w} \cdot (\text{fusion})
\]
is $R$-matrix-valued, with simple poles.  The $R$-matrix $R(u) = 1 + \hbar r/u + O(u^{-2})$ provides the braiding: $\sigma_{12} \circ \mu = R_{12} \cdot \mu \circ \sigma_{12}$, which is the $\Eone$-locality axiom.  Associativity follows from the Yang--Baxter equation $R_{12}R_{13}R_{23} = R_{23}R_{13}R_{12}$.

\emph{Step~3: Non-commutativity.}
For $\mathfrak{g} = \mathfrak{sl}_2$, the classical $r$-matrix $r = \frac{1}{2}(e \otimes f + f \otimes e) + \frac{1}{4} h \otimes h$ (the split Casimir) satisfies $r \neq 0$, so the $R$-matrix $R(u) = 1 + \hbar r/u + O(u^{-2})$ is non-trivial: $R(u) \neq P$ for finite~$u$, and the RTT relation does not reduce to commutativity.  Thus the algebra is strictly $\Eone$.
\end{proof}

\begin{remark}[Presentations and quadraticity]\label{rem:yangian-presentations}
Whether the Yangian is ``quadratic'' depends on the presentation.  In the \emph{RTT presentation} (Definition~\ref{def:yangian-rtt}), the defining relation \eqref{eq:rtt-relation} is quadratic in $T$, so standard quadratic Koszul duality (mechanism~(iii) of Remark~\ref{rem:three-koszul-mechanisms}) applies directly.  In the \emph{Drinfeld presentation} (Definition~\ref{def:yangian}), the ``terrific relation'' is cubic in the generators, so Koszul duality requires nilpotent completion (Appendix~\ref{app:nilpotent-completion}).
The two presentations yield isomorphic algebras but different Koszul dual
computations.  A rigorous construction of the $\Eone$-chiral structure and its
Koszul dual should specify which presentation is used.
\end{remark}


\section{Koszul dual of the Yangian}\label{sec:yangian-koszul}

\subsection{RTT bar complex computation}

\begin{theorem}[Yangian bar complex via RTT; \ClaimStatusProvedHere]\label{thm:yangian-bar-rtt}
\textup{[Regime: quadratic \textup{(}$\Eone$-chiral;
Convention~\textup{\ref{conv:regime-tags})}.]}

For $\mathfrak{g} = \mathfrak{sl}_2$, in the RTT presentation, the bar complex $\bar{B}(Y(\mathfrak{sl}_2)^{\mathrm{ch}})$ is computed as follows.

\emph{Generators.} In bar degree~1, the generators are the duals $T_{ij}^{(r)*}$ of the RTT generators, placed on $\overline{C}_2(X)$ with log forms.

\emph{Differential.} The bar differential extracts the RTT relation via double residues:
\begin{equation}\label{eq:yangian-bar-diff}
d(T_{ij}^{(r)*} \otimes \eta_{12}) = \sum_{k} \bigl(T_{ik}^{(r)*} \otimes T_{kj}^{(0)*} - T_{ik}^{(0)*} \otimes T_{kj}^{(r)*}\bigr) \otimes \eta_{12} \wedge \eta_{23}
\end{equation}
This is the quadratic dual of the RTT relation~\eqref{eq:rtt-modes}: the bar differential encodes the orthogonal complement of the relations under the chiral pairing.

\emph{Bar complex structure.} Since the RTT relation is quadratic, the Koszul dual $A^!$ is determined by the data in bar degrees $\leq 2$ alone: generators $V^*$ and relations $R^\perp$.  If the algebra is Koszul (i.e., $\mathrm{Ext}^{i,j}_A(\mathbb{C}, \mathbb{C}) = 0$ for $i \neq j$), then the bar-cobar quasi-isomorphism recovers $A$ from $A^!$.
\end{theorem}

\begin{proof}
The RTT relation~\eqref{eq:rtt-modes} is:
$[T_{ij}^{(r+1)}, T_{kl}^{(s)}] - [T_{ij}^{(r)}, T_{kl}^{(s+1)}] = T_{kj}^{(r)} T_{il}^{(s)} - T_{kj}^{(s)} T_{il}^{(r)}$.
This is quadratic in $T$, so the bar complex of the quadratic algebra has:
\begin{itemize}
\item $\bar{B}^0 = \mathbb{C}$ (vacuum),
\item $\bar{B}^1 = V^* = \mathrm{span}\{T_{ij}^{(r)*}\}$ (dual generators on $\overline{C}_2(X)$),
\item $\bar{B}^2 = R^\perp \subset V^* \otimes V^*$ (orthogonal complement of relations on $\overline{C}_3(X)$).
\end{itemize}
The differential $d\colon \bar{B}^2 \to \bar{B}^1$ is the dual of the inclusion $R \hookrightarrow V \otimes V$, which gives~\eqref{eq:yangian-bar-diff}.
\end{proof}

\subsection{Koszul dual identification}

For the Heisenberg algebra, Koszul duality exchanged Lie-type and Com-type: $\mathcal{H}_k^! = \mathrm{Sym}^{\mathrm{ch}}(V^*)$ (\S\ref{sec:frame-koszul-dual}).  In the Yangian setting, both sides are associative ($\Eone$), and Verdier duality inverts the $R$-matrix instead of exchanging operadic types.

\begin{theorem}[Yangian Koszul dual; \ClaimStatusProvedHere]\label{thm:yangian-koszul-dual}
\index{Yangian!Koszul dual}
The Koszul dual of the Yangian $Y(\mathfrak{sl}_2)^{\mathrm{ch}}$ in the RTT presentation is the Yangian with inverse $R$-matrix:
\begin{equation}\label{eq:yangian-koszul-dual}
Y(\mathfrak{sl}_2)^{!,\mathrm{ch}} \;\cong\; Y_{R^{-1}}(\mathfrak{sl}_2)^{\mathrm{ch}}
\end{equation}
where $Y_{R^{-1}}$ is the RTT algebra defined by the relation:
\begin{equation}
R^{-1}_{12}(u-v)\, T_1^!(u)\, T_2^!(v) = T_2^!(v)\, T_1^!(u)\, R^{-1}_{12}(u-v)
\end{equation}
with $R^{-1}(u) = 1 + \hbar P/u + O(u^{-2})$ (note the sign change from $R(u) = 1 - \hbar P/u + \cdots$).

For the Yang $R$-matrix $R(u) = 1 - \hbar P/u$, the inverse is
$R^{-1}(u) = u/(u - \hbar P) = \sum_{n \geq 0} (\hbar P/u)^n$;
since $P^2 = 1$,
\[
R^{-1}(u) = 1 + \hbar P/u + \hbar^2/u^2 + \hbar^3 P/u^3 + \cdots.
\]
The Koszul dual RTT relation is the same quadratic
relation with $\hbar \to -\hbar$.
\end{theorem}

\begin{proof}
\emph{Step~1: Quadratic dual.}
The RTT algebra is quadratic, with
$V = \mathrm{span}\{T_{ij}^{(r)}\}$ and
$R = \mathrm{span}\{[\text{RTT relations}]\}
\subset V \otimes V$.
The quadratic Koszul dual has generators~$V^*$
and relations~$R^\perp$, the orthogonal complement
under the pairing
$V^* \otimes V^* \times V \otimes V \to \mathbb{C}$.

\emph{Step~2: Orthogonal complement computation.}
The RTT relation $R_{12}(u-v) T_1 T_2 = T_2 T_1 R_{12}(u-v)$ can be written as:
\[
(1 - \hbar P/(u-v)) T_1 T_2 = T_2 T_1 (1 - \hbar P/(u-v))
\]
Expanding: $T_1 T_2 - T_2 T_1 = \frac{\hbar}{u-v}(P T_1 T_2 - T_2 T_1 P)$.

The orthogonal complement $R^\perp$ consists of quadratic expressions $Q(T^*, T^*)$ that pair to zero with all elements of $R$.  For $\mathfrak{sl}_2$ (where $V = \mathbb{C}^2$ and $P$ is the $4 \times 4$ permutation matrix), $R^\perp$ is computed by direct matrix calculation.  The RTT relation at fixed spectral parameters defines a linear map $\Phi_R\colon \mathrm{End}(V)^{\otimes 2} \to \mathrm{End}(V)^{\otimes 2}$ by $X \mapsto R_{12} X - \sigma(X) R_{12}$ where $\sigma$ is the flip; its image is the space of relations~$R$.  Writing out the $16 \times 16$ matrix of $\Phi_R$ in the standard basis of $\mathrm{End}(\mathbb{C}^2)^{\otimes 2}$ and computing its orthogonal complement with respect to the trace pairing, one verifies that $R^\perp$ coincides with the image of $\Phi_{R^{-1}}$, i.e., the space of $R^{-1}$-RTT relations.

\emph{Step~2a: Local finiteness.}
The Yangian has infinitely many generators $\{T_{ij}^{(r)}\}_{r \geq 0}$ graded by level~$r$, but each graded component $V_r = \mathrm{span}\{T_{ij}^{(r)} : 1 \leq i,j \leq \dim V\}$ is finite-dimensional: $\dim V_r = (\dim V)^2 = 4$ for $\mathfrak{sl}_2$.  The RTT relation~\eqref{eq:rtt-modes} is homogeneous of degree $r + s$ in the level grading (relating generators of levels $r, s$ on both sides), so the quadratic algebra $A = T(V)/\langle R \rangle$ is locally finite: each graded component $A_n = \bigoplus_{r_1 + \cdots + r_k = n} V_{r_1} \cdots V_{r_k} / R_n$ is a quotient of a finite-dimensional space.

\emph{Step~2b: Graded Koszul duality.}
Since each graded component is finite-dimensional, the Positselski--Polishchuk framework \cite{PP05} for Koszul duality of locally finite graded algebras applies directly.  Their Theorem~2.1 establishes the bar-cobar adjunction and the Koszul property for quadratic algebras that are locally finite in a non-negative grading (which is satisfied here, with the level grading $r \geq 0$).  The key hypothesis --- that the grading is connected ($A_0 = \mathbb{C}$) --- holds after passing to the augmentation ideal.

\emph{Step~3: $R$-matrix inversion.}
For the Yang $R$-matrix, $R(u) = (u - \hbar P)/u$ (in the defining representation of $\mathfrak{sl}_2$).  The inverse is $R^{-1}(u) = u/(u - \hbar P) = 1 + \hbar P/u + (\hbar P)^2/u^2 + \cdots$.  Since $P^2 = 1$ on $V \otimes V$, this simplifies to $R^{-1}(u) = u/(u-\hbar)$ on the symmetric subspace and $R^{-1}(u) = u/(u+\hbar)$ on the antisymmetric subspace.

\emph{Step~4: Identification.}
The RTT algebra for $R^{-1}(u)$ is isomorphic to the RTT algebra for $R(u)$ with $\hbar \to -\hbar$.  To see this, note that $R^{-1}(u; \hbar) = 1 + \hbar P/u + O(u^{-2})$ agrees with $R(u; -\hbar) = 1 + \hbar P/u$ at leading order in $1/u$; the higher-order terms $\hbar^{2k}/u^{2k}$ and $\hbar^{2k+1}P/u^{2k+1}$ in $R^{-1}$ do \emph{not} appear in $R(u;-\hbar)$, but they are irrelevant: the mode expansion~\eqref{eq:rtt-modes} extracts only the leading-order data, and the higher-order terms are automatic consequences of the quadratic RTT relation via the Yang--Baxter equation.  Therefore $Y(\mathfrak{sl}_2)^! \cong Y(\mathfrak{sl}_2)^{\hbar \to -\hbar}$, confirming that the Yangian is ``almost self-dual'' up to a sign flip of the deformation parameter.
\end{proof}

\begin{corollary}[Yangian classical limit; \ClaimStatusProvedHere]\label{cor:yangian-classical-self-dual}
At $\hbar = 0$ (the classical limit), $R(u) = R^{-1}(u) = 1$, and the RTT algebra becomes commutative: $[T_{ij}^{(r)}, T_{kl}^{(s)}] = 0$, i.e., $\mathrm{Sym}(V)$ where $V = \bigoplus_{r \geq 0} \mathfrak{gl}_2^{(r)}$.  The Koszul dual of the symmetric algebra is the exterior algebra: $\mathrm{Sym}(V)^! = \bigwedge(V^*)$.  Since $R^{-1}(u;0) = R(u;0) = 1$, the defining relations of $\mathrm{Sym}(V)$ are the antisymmetric tensors $v \otimes w - w \otimes v$, and the orthogonal complement $R^\perp$ consists of the symmetric tensors $v \otimes w + w \otimes v$, which are the defining relations of $\bigwedge(V^*)$.  This confirms that the Yangian duality reduces to $\mathrm{Sym} \leftrightarrow \bigwedge$ at $\hbar = 0$.  (In the Drinfeld presentation, the classical limit is the noncommutative algebra $U(\mathfrak{g}[t])$; the two presentations yield different classical limits.)
\end{corollary}

\begin{remark}[Connection to Langlands duality]\label{rem:yangian-langlands}
The sign flip $\hbar \to -\hbar$ in the Yangian Koszul dual is the analog of the Feigin--Frenkel level shift $k \to -k - 2h^\vee$ for affine algebras.  For simply-laced $\mathfrak{g}$, the Yangian $Y(\mathfrak{g})$ is isomorphic to $Y(\mathfrak{g}^\vee)$ (since $\mathfrak{g} \cong \mathfrak{g}^\vee$), and the Koszul duality becomes a form of self-duality.  For non-simply-laced $\mathfrak{g}$, one expects $Y(\mathfrak{g})^! \simeq Y(\mathfrak{g}^\vee)$, realizing Langlands duality at the Yangian level.  Latyntsev \cite{Latyntsev23} constructs factorisation quantum groups that provide a complementary approach to Yangian duality via factorisation homology.
\end{remark}

\subsection{Koszulness and bar-cobar recovery}

\begin{proposition}[RTT Yangian is Koszul; \ClaimStatusProvedHere]\label{prop:yangian-koszul}
\index{Yangian!Koszulness}
The Yangian $Y(\mathfrak{sl}_2)$ in the RTT presentation
(Definition~\ref{def:yangian-rtt}) is Koszul: the bar complex
$\bar{B}(Y(\mathfrak{sl}_2))$ has cohomology concentrated on the
diagonal, i.e., $\mathrm{Ext}^{i,j}_{Y}(\mathbb{C}, \mathbb{C}) = 0$
for $i \neq j$ in the $(i, j)$-bigrading by homological degree and
internal (level) degree.
\end{proposition}

\begin{proof}
\emph{Step~1: PBW basis.}
Molev \cite{molev-yangians} establishes a Poincar\'{e}--Birkhoff--Witt
basis for $Y(\mathfrak{g})$ in the RTT presentation: ordered monomials
in the generators $\{T_{ij}^{(r)}\}_{1 \leq i,j \leq N,\, r \geq 0}$
(with respect to a fixed total order on the index set $(i,j,r)$) form
a vector space basis for $Y(\mathfrak{g})$.  The
associated graded algebra with respect to the PBW filtration is the
polynomial algebra $\mathrm{Sym}(V)$, where
$V = \bigoplus_{r \geq 0} \mathrm{End}(\mathbb{C}^N)_r$.

\emph{Step~2: PBW implies Koszul.}
By the Polishchuk--Positselski PBW criterion \cite[Ch.~4]{PP05}, a
quadratic algebra possessing a PBW basis is Koszul.  The argument:
the PBW property gives a flat deformation from $Y(\mathfrak{sl}_2)$
to its associated graded $\mathrm{Sym}(V)$.  Since the Koszul complex
$K(A) = A \otimes A^{!\ast}$ is functorial in the quadratic data, and
acyclicity of $K(\mathrm{Sym}(V))$ is classical ($\mathrm{Sym}$--$\bigwedge$ duality),
flatness of the deformation implies acyclicity of $K(Y(\mathfrak{sl}_2))$.

\emph{Step~3: Graded local finiteness.}
The level grading $V = \bigoplus_{r \geq 0} V_r$ with
$\dim V_r = N^2 = 4$ (for $N = 2$) ensures that each graded component
of the quadratic algebra is finite-dimensional, satisfying the local
finiteness hypothesis required for the Polishchuk--Positselski
framework (cf.\ step~2b of the proof of
Theorem~\ref{thm:yangian-koszul-dual}).
\end{proof}

\begin{corollary}[Yangian bar-cobar recovery; \ClaimStatusProvedHere]
\label{cor:yangian-bar-cobar}
The bar-cobar adjunction provides a quasi-isomorphism of
$\Eone$-chiral algebras:
\[
\Omega^{\mathrm{ch}}(\bar{B}^{\mathrm{ch}}(Y(\mathfrak{sl}_2)^{\mathrm{ch}}))
\xrightarrow{\;\sim\;} Y(\mathfrak{sl}_2)^{\mathrm{ch}}.
\]
\end{corollary}

\begin{proof}
The RTT Yangian is quadratic (Definition~\ref{def:yangian-rtt}),
locally finite in the level grading (step~2b of the proof of
Theorem~\ref{thm:yangian-koszul-dual}), and Koszul
(Proposition~\ref{prop:yangian-koszul}).  The mode expansion
$T_{ij}(u) = \delta_{ij} + \hbar \sum_{r \geq 0} T_{ij}^{(r)}
u^{-r-1}$ defines the augmentation $\varepsilon(T_{ij}^{(r)}) = 0$
for all $r \geq 0$, with augmentation ideal $I = \langle
T_{ij}^{(r)} \rangle$.  The PBW basis ensures $\bigcap_n I^n = 0$,
so the algebra is pro-nilpotent.
Theorem~\ref{thm:e1-chiral-koszul-duality} then applies:
the counit~\eqref{eq:e1-counit} of the $\Eone$-chiral bar-cobar
adjunction is a quasi-isomorphism for augmented pro-nilpotent
$\Eone$-chiral algebras.
\end{proof}

\subsection{Yangian modules and Koszul duality}

\begin{definition}[Evaluation module]\label{def:yangian-evaluation}
\index{Yangian!evaluation module|textbf}
\index{evaluation module}
For $a \in \mathbb{C}$ (the \emph{evaluation point}) and a
finite-dimensional $\mathfrak{g}$-module $V$, the \emph{evaluation
module} $V(a)$ of the Yangian $Y(\mathfrak{g})$ is defined by
pulling back along the evaluation homomorphism
\[
\mathrm{ev}_a\colon Y(\mathfrak{g}) \to U(\mathfrak{g}),
\qquad
J^{(r)}_b \mapsto a^r\, J_b.
\]
The $R$-matrix acts on tensor products of evaluation modules by:
\begin{equation}\label{eq:yangian-r-matrix}
R_{V(a), W(b)}
= R_{V,W}\!\left(\frac{a - b}{\hbar}\right)
\in \mathrm{End}(V \otimes W)
\end{equation}
where $R_{V,W}(u) = 1 + r_{V,W}/u + O(u^{-2})$ is the rational
$R$-matrix in the representations~$V, W$.
\end{definition}

\begin{proposition}[Koszul duality on Yangian modules; \ClaimStatusProvedHere]
\label{prop:yangian-module-koszul}
\index{Yangian!module Koszul duality}
The Koszul duality $Y(\mathfrak{g})^! \cong Y_{-\hbar}(\mathfrak{g})$
\textup{(}Theorem~\textup{\ref{thm:yangian-koszul-dual})} extends to
modules as follows:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Evaluation modules:} The Koszul dual of the evaluation
  module $V(a)$ at parameter $\hbar$ is $V(-a)$ at parameter $-\hbar$:
  $\Phi(V(a)_\hbar) = V(-a)_{-\hbar}$.
\item \emph{Tensor products:} For evaluation modules $V(a)$ and
  $W(b)$, the Koszul dual of $V(a) \otimes W(b)$ is
  $V(-a) \otimes W(-b)$ with the $R$-matrix replaced by its inverse:
  \[
  R^!_{V(-a), W(-b)}
  = R_{V,W}^{-1}\!\left(\frac{a - b}{\hbar}\right).
  \]
\item \emph{Braiding reversal:} The braiding on the category of
  finite-dimensional Yangian modules is reversed:
  $\sigma^! = \sigma^{-1}$, parallel to the
  $q \leftrightarrow q^{-1}$ reversal for quantum groups
  \textup{(}Remark~\textup{\ref{rem:kl-koszul})}.
\end{enumerate}
\end{proposition}

\begin{proof}
(i) The evaluation homomorphism $\mathrm{ev}_a$ sends
$T_{ij}^{(r)} \mapsto a^r \rho(e_a)_{ij}$.  Under
$\hbar \to -\hbar$, the RTT relation retains the same form
$R'\, T_1 T_2 = T_2 T_1\, R'$ but with $R$-matrix replaced by
$R^{-1}$.  The evaluation at
parameter $a$ in the original algebra corresponds to evaluation
at $-a$ in the dual algebra, since the spectral parameter
enters the $R$-matrix via $R(u) = 1 - \hbar P/u$, and
replacing $\hbar \to -\hbar$ is equivalent to $u \to -u$, i.e.,
$a \to -a$.

(ii) follows from (i) applied to each tensor factor, combined
with the functoriality of the bar construction on tensor
products (Theorem~\ref{thm:monoidal-module-koszul}).

(iii) The braiding on Yangian modules is given by
$\sigma = P \circ R(u)$, where $P$ is the permutation and
$R(u)$ is the $R$-matrix.  Under $\hbar \to -\hbar$:
$R(u; -\hbar)$ and $R^{-1}(u; \hbar)$ define the same quadratic
dual algebra (Theorem~\ref{thm:yangian-koszul-dual}, Step~4), so
$\sigma^! = P \circ R^{-1}(u) = \sigma^{-1}$.
\end{proof}

\begin{example}[Yangian $Y(\mathfrak{sl}_2)$ fundamental modules]
\label{ex:yangian-sl2-modules}
\index{Yangian!sl2 modules@$\mathfrak{sl}_2$ modules}
The Yangian $Y(\mathfrak{sl}_2)$ has evaluation modules
$V(a) \cong \mathbb{C}^2$ for each $a \in \mathbb{C}$.  The
$R$-matrix on $V(a) \otimes V(b)$ is:
\[
R(u) = \frac{u - \hbar P}{u - \hbar}
= \begin{pmatrix}
1 & 0 & 0 & 0 \\
0 & \frac{u}{u-\hbar} & \frac{-\hbar}{u-\hbar} & 0 \\
0 & \frac{-\hbar}{u-\hbar} & \frac{u}{u-\hbar} & 0 \\
0 & 0 & 0 & 1
\end{pmatrix}
\quad\text{at } u = \tfrac{a-b}{\hbar}.
\]
The tensor product $V(a) \otimes V(b)$ decomposes as
$\mathrm{Sym}^2(V) \oplus \bigwedge^2(V)$ when $a \neq b$
(generic), and is indecomposable when $a = b$ (the $R$-matrix
has a pole).  Under Koszul duality,
$V(a) \otimes V(b) \mapsto V(-a) \otimes V(-b)$ with the
$R$-matrix $R(u) \mapsto R^{-1}(u)$: the $R$-matrix eigenvalues
on $\mathrm{Sym}^2(V)$ and $\bigwedge^2(V)$ are inverted.
\end{example}

\begin{remark}[Verification of the main theorems for Yangians]
\label{rem:yangian-three-theorems}
\index{three theorems!Yangian showcase}
We trace the three main theorems through the Yangian
$Y(\mathfrak{sl}_2)^{\mathrm{ch}}$, highlighting the differences
between the $\Eone$-chiral and $\Einf$-chiral (vertex algebra) settings.

\emph{Theorem~A (Geometric Bar-Cobar Duality).}
The bar construction $\bar{B}(Y(\mathfrak{sl}_2)^{\mathrm{ch}})$
is computed via the RTT presentation
(Theorem~\ref{thm:yangian-bar-rtt}).  The resulting Koszul dual
$Y_{R^{-1}}(\mathfrak{sl}_2)^{\mathrm{ch}}$
(Theorem~\ref{thm:yangian-koszul-dual}) is the RTT algebra with
$R$-matrix inverted, equivalently the parameter flip
$\hbar \to -\hbar$.  This is the $\Eone$ analog of the
Feigin--Frenkel involution $k \to -k - 2h^\vee$ for affine algebras
(Remark~\ref{rem:yangian-langlands}).  The bar-cobar adjunction is an
equivalence by Theorem~\ref{thm:e1-chiral-koszul-duality}, and the
cobar reconstruction recovers the Yangian
(Corollary~\ref{cor:yangian-bar-cobar}).

\emph{Theorem~B (Bar-Cobar Inversion).}
By Proposition~\ref{prop:yangian-koszul}, the RTT Yangian is Koszul,
so the spectral sequence of
Theorem~\ref{thm:bar-cobar-inversion-qi} collapses at $E_2$.
At the classical level ($\hbar = 0$), this reduces to the standard
$\mathrm{Sym} \leftrightarrow \bigwedge$ Koszul resolution
(Corollary~\ref{cor:yangian-classical-self-dual}).

\emph{Theorem~C (Complementarity).}
The deformation-obstruction complementarity
(Theorem~\ref{thm:quantum-complementarity-main}) is formulated for
$\Einf$-chiral algebras via the modular operad and Feynman transform.
The Yangian, being strictly $\Eone$, does not possess a Virasoro
subalgebra, so neither the central charge $c$ nor the obstruction
coefficient $\kappa$ is defined---this explains the ``---'' entries
in the Master Table (Table~\ref{tab:master-invariants}).  The correct
$\Eone$ analog of genus-$g$ complementarity would involve the moduli
of bordered Riemann surfaces and ribbon graph expansions, rather than
$\overline{\mathcal{M}}_g$ and tautological classes.  Developing this
$\Eone$ genus theory remains an important open direction.
\end{remark}


\begin{conjecture}[$\Eone$ genus theory;
\ClaimStatusConjectured]\label{conj:e1-genus-theory}
\index{E1-chiral algebra@$\Eone$-chiral algebra!genus theory}
For an $\Eone$-chiral algebra $\cA$ on $\bR$
(or on an oriented interval), the genus-$g$ obstruction
should be defined using the moduli of bordered Riemann
surfaces $\mathcal{M}_{g,1}^{\partial}$
(surfaces with one parameterized boundary component).
Specifically:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{$\Eone$ bar complex at genus~$g$.}
The genus-$g$ bar complex
$\bar{B}_g^{\Eone}(\cA) :=
\bigoplus_{n \geq 0} \Gamma\bigl(
\overline{C}_n(\Sigma_{g,1}),\,
\cA^{\boxtimes n} \otimes
\Omega^n_{\log}\bigr)$
uses configuration spaces on the genus-$g$
surface $\Sigma_{g,1}$ with one boundary component.
\item \emph{Nilpotency via ribbon graphs.}
The bar differential satisfies $d^2 = 0$ on
$\Sigma_{g,1}$, with the proof using the
Orlik--Solomon relations on the
compactified configuration space
$\overline{C}_n(\Sigma_{g,1})$
(generalizing the genus-$0$ case for
$\Sigma_{0,1} \cong \bR^2$).
\item \emph{Ribbon graph expansion.}
The obstruction $\mathrm{obs}_g^{\Eone}(\cA)$
should be computed by summing over ribbon graphs
$\Gamma$ with genus~$g$ and one boundary cycle:
\[
\mathrm{obs}_g^{\Eone}(\cA) =
\sum_{\Gamma \in \mathrm{Rib}_{g,1}}
\frac{1}{|\mathrm{Aut}(\Gamma)|}
\int_{\overline{C}_\Gamma(\Sigma_{g,1})}
\omega_\Gamma(\cA)
\]
where $\omega_\Gamma(\cA)$ is the form
associated to the OPE data of~$\cA$ on the
edges and vertices of~$\Gamma$.
\item \emph{Complementarity.}
$\mathrm{obs}_g^{\Eone}(\cA) +
\mathrm{obs}_g^{\Eone}(\cA^!) = 0$
in $H^*(\mathcal{M}_{g,1}^{\partial})$,
since the $\Eone$ Koszul duality inverts
the $R$-matrix ($R \to R^{-1}$), which
reverses the orientation of the ribbon
graphs.
\end{enumerate}
For the Yangian $Y(\mathfrak{sl}_2)^{\mathrm{ch}}$,
the ribbon graph expansion at genus~$1$ should recover
$\mathrm{obs}_1 \propto \hbar$ (the deformation parameter),
vanishing at $\hbar = 0$ (the classical limit,
Corollary~\textup{\ref{cor:yangian-classical-self-dual}}).
(Contributing to Conjecture~\ref{conj:master-dk-kl}.)
\end{conjecture}

\begin{remark}[Scope]\label{rem:e1-genus-scope}
This conjecture requires several foundational
developments: (a)~a theory of compactified configuration
spaces on bordered surfaces with the correct boundary
behavior (extending Fulton--MacPherson to the case
with boundary); (b)~an $\Eone$ analog of the Feynman
transform for the modular operad; (c)~an $\Eone$ analog
of the Faber--Pandharipande formula for tautological
integrals on $\mathcal{M}_{g,1}^{\partial}$.  The
closest existing work is the Swiss-cheese operad
(Voronov~\cite{Voronov99}), which governs open-closed
homotopy algebras.
The toroidal algebra
(Remark~\ref{rem:toroidal-three-theorems}) faces the
same obstruction.

\emph{Homotopy template} (Convention~\ref{conv:hms-levels}): Type~I (existence).
The conjecture posits the existence of an $\Eone$ genus theory on bordered Riemann surfaces, extending the chiral bar complex from closed to open surfaces.
\end{remark}


\section{Shifted Yangians}
\label{sec:shifted-yangians}

\subsection{Definition}

\begin{definition}[Shifted Yangian]\label{def:shifted-yangian}
For a coweight $\mu \in X_*(\mathfrak{h})$, the \emph{shifted Yangian} $Y_\mu(\mathfrak{g})$
(Brundan--Kleshchev, Finkelberg--Tsymbaliuk, KWWY) is generated by elements
$E_i^{(r)}, F_i^{(r)}, H_i^{(r)}$ for $i = 1, \ldots, \mathrm{rank}(\mathfrak{g})$
and $r \geq 0$, satisfying the same Yangian relations as $Y(\mathfrak{g})$ except that the
generating function $H_i(u) = \sum_{r \geq 0} H_i^{(r)} u^{-r}$ is constrained to be a rational
function whose numerator and denominator degrees are prescribed by $\mu$:
\[
H_i(u) = \frac{\prod_{j=1}^{\mu_i^+}(u - a_j^{(i)})}{\prod_{j=1}^{\mu_i^-}(u - b_j^{(i)})}
\]
where $\mu_i^\pm = \langle \alpha_i, \mu^\pm \rangle$ and $\mu = \mu^+ - \mu^-$ decomposes the coweight into dominant parts. The parameters $a_j^{(i)}, b_j^{(i)}$ are complex numbers.
\end{definition}

\begin{conjecture}[Shifted Yangian as $\Eone$-chiral; \ClaimStatusConjectured]\label{conj:shifted-yangian-e1}
The shifted Yangian admits a chiral algebra structure $Y_\mu(\mathfrak{g})^{\mathrm{ch}}$
that is $\Eone$-chiral.  The shift $\mu$ modifies the central terms in the OPE,
and the rational constraint on $H_i(u)$ introduces additional poles in the vertex
operators.
(Contributing to Conjecture~\ref{conj:master-dk-kl}.)
\end{conjecture}

\begin{remark}[Scope]
A proof would require: (1)~an explicit vertex operator construction for
the shifted Yangian generators $E_i^{(r)}, F_i^{(r)}, H_i^{(r)}$ with
the rational constraint on $H_i(u)$; (2)~verification of the Borcherds
identity (weak associativity of the OPE) in the presence of the additional
poles introduced by the shift.  For the unshifted case ($\mu = 0$), the
$\Eone$-chiral structure is established in Theorem~\ref{thm:yangian-e1}.
The shifted case is expected to follow by an analogous argument using the
BFN construction (Theorem~\ref{thm:bfn}), which realizes the shifted
Yangian as equivariant homology and thus provides a natural source of
vertex operators.  Dimofte--Niu--Py \cite{DNP25} construct dg-shifted Yangians
from line operators in 3d holomorphic QFT, providing further evidence for the
$\Eone$-chiral structure in the shifted setting.

\emph{Homotopy template} (Convention~\ref{conv:hms-levels}): Type~I (existence).
The conjecture asserts the existence of an $\Eone$-chiral algebra structure on the shifted Yangian $Y_\mu(\mathfrak{g})$, extending the unshifted construction to non-zero coweights.
\end{remark}

\subsection{Comparison with dg-shifted Yangians}
\label{sec:dg-shifted-comparison}

\begin{proposition}[Structural comparison; \ClaimStatusProvedHere]
\label{prop:dg-shifted-comparison}
\index{Yangian!dg-shifted}
The $\Eone$-chiral Yangian
$Y(\mathfrak{g})^{\mathrm{ch}}$
\textup{(}Theorem~\textup{\ref{thm:yangian-e1})}
and the dg-shifted Yangian
$Y^{\mathrm{dg}}(\mathfrak{g})$
of Dimofte--Niu--Py~\textup{\cite{DNP25}}
share the following structural features:
\begin{enumerate}[label=\textup{(\roman*)}]
\item Both are $\Eone$-deformations of the affine algebra
  $\widehat{\mathfrak{g}}_k$, controlled by the rational
  $R$-matrix $R(u) = 1 + r/u + O(u^{-2})$.
\item Both have generators in RTT form
  $T(u) = 1 + \sum_{r \geq 1} T^{(r)} u^{-r}$
  with quadratic relations
  $R_{12}(u{-}v)\,T_1(u)\,T_2(v) = T_2(v)\,T_1(u)\,R_{12}(u{-}v)$.
\item The Koszul dual of both is the $R^{-1}$-Yangian:
  replacing $R$ by $R^{-1}$ corresponds to
  $\hbar \mapsto -\hbar$.
\item Evaluation modules $V(a)$ and their tensor products
  behave identically in both constructions
  (Proposition~\ref{prop:eval-module-bar}).
\end{enumerate}
The constructions differ in their geometric origins:
$Y(\mathfrak{g})^{\mathrm{ch}}$ arises from FM configuration
space integrals on algebraic curves, while
$Y^{\mathrm{dg}}(\mathfrak{g})$ arises from line operators
in 3d holomorphic Chern--Simons theory~\textup{\cite{DNP25}}.
\end{proposition}

\begin{proof}
Part~(i) follows from comparing the deformation parameter:
both constructions deform the vertex algebra OPE by a
spectral parameter $u = z_1 - z_2$ that controls
$\Eone$-locality.  In the FM picture, $u$ is the
relative coordinate on $\overline{C}_2(X)$
(Theorem~\ref{thm:yangian-e1}).  In the DNP construction,
$u$ is the holomorphic coordinate along the topological
line (the ``spectral parameter'' of \cite[\S5]{DNP25}).

Part~(ii): the RTT presentation is a presentation of the
quadratic algebra, hence determined by the $R$-matrix.
Since both constructions use the same $R$-matrix, the
RTT relations agree.

Part~(iii) follows from Theorem~\ref{thm:yangian-koszul-dual}
(this manuscript) and \cite[Thm.~6.12]{DNP25}
(the Koszul dual of line operators is controlled by $R^{-1}$).

Part~(iv): the bar complex of the evaluation module
is the CE complex $C^*(\mathfrak{g}, V)$
(Proposition~\ref{prop:eval-module-bar}), which
depends only on the underlying $\mathfrak{g}$-action,
not on the geometric origin of the Yangian structure.
\end{proof}

\begin{remark}[Quasi-isomorphism question]
\label{rem:dg-shifted-qi}
The structural comparison above does not establish a
quasi-isomorphism $Y(\mathfrak{g})^{\mathrm{ch}}
\simeq Y^{\mathrm{dg}}(\mathfrak{g})$: this would require
matching the full $A_\infty$ structure (higher operations
$m_k$ for $k \geq 3$).  Since both arise as quadratic
$\Eone$-algebras with the same $R$-matrix, the $A_\infty$
structures are determined by the bar complex, and
a quasi-isomorphism would follow from the Koszul property
(Remark~\ref{rem:yangian-collapse-conj}).
\end{remark}

\begin{remark}[Maurer--Cartan element as twisting morphism]
\label{rem:dnp-mc-twisting}
\index{Maurer--Cartan element!twisting morphism}
\index{Yang--Baxter equation!A-infinity@$A_\infty$}
The dg-shifted Yangian of \cite{DNP25} carries a
Maurer--Cartan element $r(z) \in \cA^! \otimes \cA^!$
satisfying an $A_\infty$ Yang--Baxter equation.
In the bar-cobar framework, this element admits a precise
identification: $r(z)$ is the \emph{twisting morphism}
$\tau\colon \bar{B}(\cA) \to \cA^!$
of Theorem~\ref{thm:bar-cobar-isomorphism-main},
restricted to the degree-$2$ component and evaluated at
spectral parameter~$z$.

Explicitly:
\begin{enumerate}[label=\textup{(\roman*)}]
\item The MC equation $dr + r \star r = 0$ is the
  bar-cobar adjunction equation
  $\partial \tau + \tau \cup \tau = 0$,
  which is equivalent to $d_{\bar{B}}^2 = 0$
  (Theorem~\ref{thm:bar-nilpotency-complete}).
\item The $A_\infty$ Yang--Baxter equation of
  \cite[\S5]{DNP25} is the $\Eone$-chiral generalization:
  the bar complex differential encodes the RTT relation
  (Theorem~\ref{thm:yangian-bar-rtt}), and $d^2 = 0$
  is precisely the Yang--Baxter equation for $R(u)$.
\item The non-renormalization theorem of \cite[Thm.~4.1]{DNP25}
  (OPE of line operators is $1$-loop exact) corresponds to
  the Koszul property: the bar spectral sequence collapses
  at $E_2$, so the $A_\infty$ structure is determined by
  the quadratic ($R$-matrix) data.
\end{enumerate}

This identification suggests that the $\Eone$-chiral Yangian
and the dg-shifted Yangian are \emph{the same object}
viewed through two different lenses: configuration space
integrals (this manuscript) and holomorphic Chern--Simons
perturbation theory~(\cite{DNP25}).  A proof would follow
from establishing the Koszulness of the $\Eone$-chiral bar complex
(Remark~\ref{rem:yangian-collapse-conj}).
\end{remark}

\section{Coulomb branch algebras}
\label{sec:coulomb-branch}

\subsection{Definition from gauge theory}

\begin{definition}[Coulomb branch algebra]\label{def:coulomb-branch}
\index{Coulomb branch}
For a 3d $\mathcal{N} = 4$ gauge theory with gauge group $G$ and matter representation $N$, 
the \emph{Coulomb branch algebra} $\mathcal{A}(G, N)$ is the quantization of the 
Coulomb branch $\mathcal{M}_C(G, N)$.

For type A quiver gauge theories, $\mathcal{A}(G, N)$ is a shifted Yangian or 
truncated shifted Yangian.
\end{definition}

\begin{theorem}[BFN construction; \ClaimStatusProvedElsewhere]\label{thm:bfn}
(Braverman--Finkelberg--Nakajima \cite{BFN18}) The Coulomb branch algebra is:
\[
\mathcal{A}(G, N) = H^{G_{\mathcal{O}} \times \mathbb{C}^\times}_*(\mathcal{R}(G, N))
\]
the equivariant Borel--Moore homology of the space of triples $\mathcal{R}(G, N)$.
\end{theorem}

\begin{example}[Quiver of type $A_n$]\label{ex:coulomb-an}
For the $A_n$ quiver with dimension vector $(1, 2, \ldots, n)$:
\[
\mathcal{A} = Y_{(n-1, n-2, \ldots, 1, 0)}(\mathfrak{sl}_n)
\]
a shifted Yangian.
\end{example}


\section{Cohomological Hall algebras}
\label{sec:coha}

\subsection{Definition}

\begin{definition}[Cohomological Hall algebra]\label{def:coha}
For a quiver $Q$ with dimension vector $\mathbf{d}$, the \emph{cohomological Hall algebra} (CoHA) is:
\[
\mathcal{H}_Q = \bigoplus_{\mathbf{d}} H^{G_{\mathbf{d}}}_{\mathrm{BM},*}(\mathcal{M}_{\mathbf{d}}^{\mathrm{nil}})
\]
where $\mathcal{M}_{\mathbf{d}}^{\mathrm{nil}}$ is the moduli of nilpotent representations.

The product is defined via the Hecke correspondence:
\[
\mathcal{M}_{\mathbf{d}_1} \times \mathcal{M}_{\mathbf{d}_2} \xleftarrow{p} \mathcal{E} \xrightarrow{q} \mathcal{M}_{\mathbf{d}_1 + \mathbf{d}_2}
\]
as $a \star b = q_* p^*(a \boxtimes b)$.
\end{definition}

\begin{conjecture}[CoHA as $\Eone$-chiral; \ClaimStatusConjectured]\label{conj:coha-e1}
The CoHA $\mathcal{H}_Q$ carries an $\Eone$-chiral algebra structure.  The evidence:
\begin{enumerate}[label=(\roman*)]
\item The Hecke product is associative but not commutative.
\item The factorization structure comes from stratifying by dimension vectors.
\item The chiral enhancement should use configuration spaces of points on curves
parameterizing deformation directions.
\end{enumerate}
A rigorous construction requires specifying the vertex operators and verifying
weak associativity (the Borcherds identity).
(Contributing to Conjecture~\ref{conj:master-dk-kl}.)
\end{conjecture}

\begin{remark}[Scope]
The key obstacle is translating the Hecke product
$a \star b = q_* p^*(a \boxtimes b)$ into a vertex operator OPE.
The Hecke correspondence provides an associative product on
$\bigoplus_{\mathbf{d}} H^{G_{\mathbf{d}}}_{\mathrm{BM},*}(\mathcal{M}_{\mathbf{d}}^{\mathrm{nil}})$,
but this product is defined via correspondences (push-pull in
Borel--Moore homology), not via residues of meromorphic functions.
The expected translation uses the Schiffmann--Vasserot construction
\cite{SV13}, which realizes the CoHA action on the equivariant
cohomology of Hilbert schemes as a Heisenberg/W-algebra action.
Extending this from an \emph{action} to a full vertex algebra
\emph{structure on the CoHA itself} remains open.

\emph{Homotopy template} (Convention~\ref{conv:hms-levels}): Type~I (existence).
The conjecture asserts the existence of an $\Eone$-chiral algebra structure on the cohomological Hall algebra, lifting the Hecke product to a vertex operator OPE.
\end{remark}

\begin{example}[Jordan quiver CoHA]\label{ex:jordan-coha}
For the Jordan quiver (one vertex, one loop), the CoHA is:
\[
\mathcal{H}_{\text{Jordan}} = \bigoplus_{n \geq 0} H^{GL_n}_{\mathrm{BM},*}(\mathcal{N}_n) \;\curvearrowright\; \bigoplus_{n \geq 0} H^*_T(\mathrm{Hilb}^n(\bC^2))
\]
The CoHA acts on the equivariant cohomology of the Hilbert scheme via Nakajima-type operators (Schiffmann--Vasserot), realizing a Heisenberg/W-algebra structure on the target.  The CoHA itself carries a \emph{different} product structure (the shuffle product) from the OPE product on the chiral algebra side.
\end{example}


\subsection{Shifted Yangians and Coulomb branch algebras}
\label{sec:shifted-coulomb}
\index{shifted Yangian!Coulomb branch}
\index{Coulomb branch!shifted Yangian}

The BFN construction (Theorem~\ref{thm:bfn}) realizes Coulomb branch
algebras $\mathcal{A}_C(N,G)$ as equivariant Borel--Moore homology of
spaces of triples on the affine Grassmannian.  For type $A$ quiver gauge
theories, these algebras are shifted Yangians
$Y_\mu(\fg)$ (Definition~\ref{def:shifted-yangian}), and their
$\Eone$-chiral structure (Conjecture~\ref{conj:shifted-yangian-e1})
interacts with Koszul duality as follows.

\begin{conjecture}[Langlands-dual Koszul duality for shifted Yangians;
\ClaimStatusConjectured]
\label{conj:shifted-yangian-langlands}
\index{shifted Yangian!Koszul dual}
\index{symplectic duality}
For a coweight $\mu \in X_*(\mathfrak{h})$ and simple $\fg$:
\begin{equation}\label{eq:shifted-langlands-dual}
Y_\mu(\fg)^! \;\simeq\; Y_{-\mu}(\fg^\vee)
\end{equation}
as $\Eone$-chiral algebras, where $\fg^\vee$ is the Langlands dual and
$-\mu$ is the negated coweight (viewed as a weight of $\fg^\vee$ via
the canonical identification $X_*(\mathfrak{h}) \cong X^*(\mathfrak{h}^\vee)$).
For unshifted $\mu = 0$, this reduces to Yangian self-duality
(Theorem~\ref{thm:yangian-koszul-dual}) when $\fg$ is simply-laced.
(Contributing to Conjecture~\ref{conj:master-dk-kl}.)
\end{conjecture}

\begin{remark}[Scope]
The conjecture is an $\Eone$-chiral incarnation of symplectic duality
(Braden--Licata--Proudfoot--Webster \cite{BLPW16}).  In that framework,
the Coulomb branch $\mathcal{M}_C(N,G)$ and the Higgs branch
$\mathcal{M}_H(N,G)$ are exchanged: Koszul duality of the quantized
algebras corresponds to Coulomb--Higgs interchange.  For the RTT
presentation, the shift $\mu \mapsto -\mu$ reverses the asymptotic
boundary conditions of the monopole operators, which at the level of
the $R$-matrix is the replacement $R(u) \mapsto R(u)^{-1}$ together with
the Langlands involution on root data.  A proof would require
establishing the $\Eone$-chiral structure on $Y_\mu(\fg)$
(Conjecture~\ref{conj:shifted-yangian-e1}) and then verifying that
the bar complex with $R$-matrix data at coweight $\mu$ computes the
cobar at coweight $-\mu$ for $\fg^\vee$.

\emph{Homotopy template} (Convention~\ref{conv:hms-levels}): Type~II (equivalence).
The conjecture asserts a Koszul duality equivalence between the shifted Yangian at coweight $\mu$ and the Langlands-dual shifted Yangian at negated coweight, realizing symplectic duality at the $\Eone$-chiral level.
\end{remark}

\begin{example}[Type $A_1$ quiver with framing]
\label{ex:coulomb-a1-framing}
\index{Coulomb branch!type $A_1$}
For the $A_1$ quiver with gauge group $G = GL_1$ and framing
$\mathbf{w} = (n)$ (i.e., $n$ fundamental hypermultiplets), the
Coulomb branch is the Kleinian singularity:
\[
\mathcal{M}_C(n, GL_1) = \{(x,y,z) \in \bC^3 : xy = z^n\}
\]
The quantized Coulomb branch algebra
$A_\hbar = \bC\langle x, y\rangle / (xy - (z+\hbar)(z+2\hbar)
\cdots (z+n\hbar))$ is a shifted Yangian $Y_\mu(\mathfrak{sl}_2)$
with $\mu = (n-1)\omega^\vee$ (fundamental coweight).  Its bar complex
$\barB(A_\hbar)$ has:
\begin{align*}
\dim \barB^1 &= 3 \quad (x, y, z) \\
\dim \barB^2 &= 9 \quad (\text{all pairs}) \\
d([x|y]) &= xy = z^n + O(\hbar) \\
d([y|x]) &= yx = z^n + n\hbar z^{n-1} + O(\hbar^2)
\end{align*}
The bar cohomology $H^2(\barB(A_\hbar))$ detects the deformation
of the singularity: $\dim H^2 = n - 1$ (the Milnor number of $xy = z^n$).
The Koszul dual is the Higgs branch algebra, which for this quiver is
$A_{-\hbar} = \bC\langle x', y'\rangle / (x'y' = z^n - n\hbar z^{n-1}
+ \cdots)$, the same singularity with reversed quantization parameter.
\end{example}

\subsection{Cohomological Hall algebras as $\Eone$-chiral algebras}
\label{sec:coha-structure}
\index{CoHA!$\Eone$-chiral structure}
\index{CoHA!factorization structure}

The CoHA product (Definition~\ref{def:coha}) is defined via Hecke
correspondences on moduli of quiver representations.  Its factorization
structure---the assembly of $\mathcal{H}_Q = \bigoplus_{\mathbf{d}}
H^{G_{\mathbf{d}}}_{\mathrm{BM},*}(\mathcal{M}_{\mathbf{d}}^{\mathrm{nil}})$
over all dimension vectors---is reminiscent of the factorization algebras
underlying vertex algebras.

\begin{conjecture}[CoHA--Yangian Koszul duality; \ClaimStatusConjectured]
\label{conj:coha-koszul}
\index{CoHA!Koszul duality}
For a quiver $Q$ whose underlying graph is the Dynkin diagram of a
simple Lie algebra $\fg_Q$:
\begin{equation}\label{eq:coha-yangian-dual}
\mathcal{H}_Q^! \;\simeq\; Y(\fg_Q)
\end{equation}
as $\Eone$-algebras, where $\mathcal{H}_Q$ is the CoHA of $Q$
\textup{(}Definition~\textup{\ref{def:coha})} and $Y(\fg_Q)$ is the
Yangian \textup{(}Definition~\textup{\ref{def:yangian})}.
(Contributing to Conjecture~\ref{conj:master-dk-kl}.)
\end{conjecture}

\begin{remark}[Scope and evidence]
\begin{enumerate}[label=(\roman*)]

\section{Explicit Yangian bar complex computations}
\label{sec:yangian-bar-explicit}
\index{Yangian!bar complex!explicit}

\subsection{RTT bar complex for $Y(\mathfrak{sl}_2)$}

\begin{computation}[Yangian $Y(\mathfrak{sl}_2)$: bar complex
through degree 3]
\label{comp:yangian-sl2-bar}
\index{Yangian!sl2@$\mathfrak{sl}_2$!bar complex}

The Yangian $Y(\mathfrak{sl}_2)$ in Drinfeld's first realization
has generators $\{e_0, f_0, h_0, e_1, f_1, h_1\}$ with
relations encoding the $\mathfrak{sl}_2$ Lie algebra at level~0
and the Yangian deformation at level~1.

\emph{RTT presentation.}
In the RTT formalism (Definition~\ref{def:yangian-rtt}), the
algebra is generated by the matrix entries $t_{ij}^{(r)}$
($i,j \in \{1,2\}$, $r \geq 0$) of the generating matrix
$T(u) = \sum_{r \geq 0} T^{(r)} u^{-r}$ subject to
$R_{12}(u-v) T_1(u) T_2(v) = T_2(v) T_1(u) R_{12}(u-v)$,
where $R(u) = 1 - P/u$ is the rational $R$-matrix
($P$ is the permutation operator).

\emph{Degree 1.}
$\bar{B}^1(Y(\mathfrak{sl}_2))$ is spanned by the
generators $t_{ij}^{(r)}$ for all $r \geq 1$
(the augmentation ideal excludes $t_{ij}^{(0)} = \delta_{ij}$).
At filtration degree $\leq n$: $\dim \bar{B}^1_{\leq n}
= 4n$ (four matrix entries at each level $r = 1, \ldots, n$).

\emph{Degree 2.}
Elements $[t_{ij}^{(r)} | t_{kl}^{(s)}] \otimes \eta_{12}$
with $r, s \geq 1$.  The bar differential extracts the $R$-matrix
commutation relation:
\begin{align}
d\bigl([t_{ij}^{(r)} | t_{kl}^{(s)}] \otimes \eta_{12}\bigr)
&= \Res_{u = v}\left[
\frac{R_{12}(u-v)_{ij,kl} - \delta_{ij}\delta_{kl}}{u-v}
\right]_{r,s} \label{eq:yangian-bar-d2}
\end{align}
The $R$-matrix $R(u) = 1 - P/u$ gives:
\[
\frac{R(u)_{ij,kl} - \delta_{ij}\delta_{kl}}{u}
= -\frac{\delta_{il}\delta_{kj}}{u^2}
\quad (\text{the permutation term})
\]
The residue at $u = v$ extracts the $1/(u-v)$ coefficient:
\begin{align}
d([t_{ij}^{(r)} | t_{kl}^{(s)}]) &=
-t_{il}^{(r+s-1)} \delta_{kj} + t_{kj}^{(r+s-1)} \delta_{il}
\label{eq:yangian-d2-explicit}
\end{align}
(the RTT relation at the level of generating functions).

\emph{Explicit at lowest levels.}
At $r = s = 1$:
\begin{align*}
d([t_{11}^{(1)} | t_{22}^{(1)}])
&= -t_{12}^{(1)} + t_{21}^{(1)} \\
d([t_{12}^{(1)} | t_{21}^{(1)}])
&= -t_{11}^{(1)} + t_{22}^{(1)} = -h_0^{(1)}
\end{align*}
where $h_0^{(1)} = t_{11}^{(1)} - t_{22}^{(1)}$ is the Cartan
generator at level~1.

\emph{Degree 3: Serre-type relations.}
At degree~3, the bar differential encodes the Yangian Serre
relations.  The key element:
\begin{align*}
d([e_1 | e_1 | f_1] \otimes \eta_{12} \wedge \eta_{13})
&= [e_1 | h_1] \otimes \eta_{13}
+ \text{(other terms)}
\end{align*}
The Serre relation
$[e_1, [e_1, f_0]] - [e_0, [e_1, f_1]] = \frac{1}{4}\{h_0, e_0\}$
appears as a cohomological relation in $H^3(\bar{B}(Y))$.

\emph{Dimension table.}
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|cccc}
Filtration degree & $\dim \bar{B}^1$ & $\dim \bar{B}^2$ &
$\dim \bar{B}^3$ & $\dim H^2$ \\
\hline
$\leq 1$ & 4 & 16 & 64 & 6 \\
$\leq 2$ & 8 & 64 & 512 & 22 \\
$\leq 3$ & 12 & 144 & 1728 & 42
\end{tabular}
\end{center}
Note: the column ``$\dim H^2$'' reports the Koszul dual dimension
$(A^!)_2$ of the truncated RTT \emph{quadratic algebra}
$Y(\mathfrak{gl}_2)_{\leq L}$, which grows with the filtration
level~$L$.  This differs from the \emph{chiral} bar cohomology
$\dim H^2(\barBgeom(Y(\mathfrak{sl}_2)^{\mathrm{ch}})) = 10$
(Table~\ref{tab:bar-dimensions}), which uses
$4$ independent generators only (Remark~\ref{rem:yangian-gl2-kunneth}).
\end{computation}


\subsection{Yangian Koszul dual: explicit verification}

\begin{computation}[Yangian duality: $R(u) \mapsto R(u)^{-1}$]
\label{comp:yangian-dual-explicit}
\index{Yangian!Koszul dual!explicit}

The $\Eone$-chiral Koszul duality of
Theorem~\ref{thm:yangian-koszul-dual}
sends $R(u)$ to~$R(u)^{-1}$.
For the rational~$R$-matrix:
\[
R(u) = 1 - \frac{P}{u}, \qquad
R(u)^{-1} = 1 + \frac{P}{u - 1}
= \frac{u}{u-1} - \frac{P}{u-1}
\]

\emph{Interpretation.}
The dual $R$-matrix $R(u)^{-1}$ defines the dual Yangian
$Y(\mathfrak{sl}_2)^!$ via the RTT relation
$R^{-1}_{12}(u-v) T_1(u) T_2(v)
= T_2(v) T_1(u) R^{-1}_{12}(u-v)$.

The pole of $R(u)^{-1}$ at $u = 1$ (rather than $u = 0$
for $R(u)$) shifts the ``interaction point'' in the bar complex.
The degree-2 bar differential~\eqref{eq:yangian-d2-explicit}
becomes:
\[
d^!([t_{ij}^{(r)} | t_{kl}^{(s)}])
= t_{il}^{(r+s)} \delta_{kj} - t_{kj}^{(r+s)} \delta_{il}
\]
(the shift $r+s-1 \mapsto r+s$ reflects the pole shift
$u = 0 \mapsto u = 1$).

\emph{Self-duality at the level of bar complexes.}
The Yangian $Y(\fg)$ is conjectured to be \emph{self-dual}
(Theorem~\ref{thm:yangian-koszul-dual}): $Y(\fg)^! \simeq Y(\fg)$
after the spectral parameter shift $u \mapsto u + 1$.  At the level
of bar complexes, this is the statement that the bar complex with
$R$-matrix $R(u)^{-1}$ is quasi-isomorphic to that with
$R$-matrix $R(u-1)$, which by the translation invariance
of the RTT relation is isomorphic to the original bar complex.

\emph{Central charge.}
The Yangian has no central extension in the usual sense
(it is an $\Eone$-algebra, not $\Einf$), so the complementarity
theorem (Theorem~\ref{thm:quantum-complementarity-main}) does not
apply directly.  The $\Eone$ analog of complementarity
(Remark~\ref{rem:yangian-three-theorems}) involves the
\emph{framed} moduli space of curves, which is not yet
developed in full generality.
\end{computation}

\begin{remark}[K\"unneth decomposition for $\widehat{\mathfrak{gl}}_2$]
\label{rem:yangian-gl2-kunneth}
\index{Yangian!Kunneth decomposition@K\"unneth decomposition}
In the chiral bar complex, the level-$1$ generators
$T_{ij}^{(1)}$ ($1 \leq i,j \leq 2$) span
$\mathfrak{gl}_2 = \mathfrak{sl}_2 \oplus \mathfrak{gl}_1$, and
the OPE among them reproduces the $\widehat{\mathfrak{gl}}_2$
current algebra (double pole from the level, simple pole from
the Lie bracket).  Higher-level generators $T_{ij}^{(r)}$
($r \geq 2$) appear as composites
(normal-ordered products of level-$1$ fields).
By the K\"unneth formula for bar complexes
of direct products:
\[
H^n(\barBgeom(\widehat{\mathfrak{gl}}_2))
= \sum_{i+j = n}
H^i(\barBgeom(\widehat{\mathfrak{sl}}_2))
\otimes
H^j(\barBgeom(\widehat{\mathfrak{gl}}_1))
\]
with $H^i(\widehat{\mathfrak{sl}}_2) = R(i{+}3)$ (Riordan) and
$H^j(\widehat{\mathfrak{gl}}_1) = p(j{-}2)$ (partition function,
$p({-}1) = p(0) = 1$).  This gives
$H^n(\widehat{\mathfrak{gl}}_2) = 1, 4, 10, 25, 62, \ldots$
At degrees $1$ and $2$, the Yangian bar cohomology matches
$\widehat{\mathfrak{gl}}_2$ exactly.  At degree~$3$, the Yangian Serre
relations (degree-$3$ bar differential) contribute $3$ additional
classes, giving $H^3 = 25 + 3 = 28$.
\end{remark}


\subsection{Coulomb branch bar complex}

\begin{computation}[Coulomb branch and Yangian bar]
\label{comp:coulomb-bar}
\index{Coulomb branch!bar complex}

For the Jordan quiver with dimension vector $\mathbf{d} = (n)$
and framing vector $\mathbf{w} = (1)$, the Coulomb branch
$\mathcal{M}_C(n, 1)$ is:
\[
\mathcal{M}_C(n, 1)
= \operatorname{Spec}\bigl(
\bC[z_1, \ldots, z_n]^{S_n}[\hbar]\bigr)
\cong \bC^n/S_n \times \bC_\hbar
\]
(the symmetric product of the affine line, quantized by $\hbar$).

The quantized Coulomb branch algebra
$A_\hbar(n, 1)$ is generated by
$e_k = \sum_{i=1}^n z_i^k$ ($k \geq 1$) with
the shifted Yangian relation:
\[
[e_k, e_l] = \hbar \cdot k \delta_{k+l, 0} \cdot n
+ \text{(lower terms)}
\]

\emph{Bar complex.}
The bar complex $\bar{B}(A_\hbar(n,1))$ has degree-2
elements $[e_k | e_l] \otimes \eta_{12}$ with differential:
\[
d([e_k | e_l]) = [e_k, e_l]
= \hbar k n \delta_{k+l} + \sum_{j=1}^{k-1}
\binom{k}{j} e_{k-j} e_{l+j} - (k \leftrightarrow l)
\]
At $n = 1$ (the Heisenberg case): $A_\hbar(1,1) \cong
\mathcal{H}_\hbar$ (the Heisenberg algebra at level $\hbar$),
and the bar complex reduces to the Heisenberg bar complex
of Computation~\ref{comp:heisenberg-deg3-full}.

At $n = 2$: the algebra $A_\hbar(2,1)$ contains both the
Heisenberg subalgebra (generated by $e_1 = z_1 + z_2$) and
a relative subalgebra (generated by $e_2 - e_1^2/2
= (z_1-z_2)^2/2$).  The bar complex detects the interaction
between these two sectors via the residue at the diagonal
$z_1 = z_2$ in the symmetric product.
\end{computation}


\section{Higher-rank Yangians and quantum groups}
\label{sec:higher-rank-yangians}
\index{Yangian!higher rank}

\subsection{\texorpdfstring{Yangian $Y(\mathfrak{sl}_3)$: bar complex}{Yangian Y(sl3): bar complex}}

\begin{computation}[$Y(\mathfrak{sl}_3)$ bar complex through degree 2]
\label{comp:yangian-sl3-bar}
\index{Yangian!sl3@$\mathfrak{sl}_3$!bar complex}

The Yangian $Y(\mathfrak{sl}_3)$ is generated by the $3 \times 3$
matrix $T(u) = (t_{ij}(u))_{1 \leq i,j \leq 3}$ with
$t_{ij}(u) = \delta_{ij} + \sum_{r=1}^\infty t_{ij}^{(r)} u^{-r}$,
subject to the RTT relation with $R$-matrix
$R_{12}(u) = 1 - P_{12}/u$.

\emph{Degree 1.}
The generators $\{t_{ij}^{(r)} : 1 \leq i,j \leq 3, \; r \geq 1\}$
form the desuspended bar generators $s^{-1}t_{ij}^{(r)}$.
At filtration degree $\leq 1$: $\dim \bar{B}^1 = 9$
(the nine entries of the first-order matrix $T^{(1)}$).
At filtration degree $\leq n$: $\dim \bar{B}^1 = 9n$.

\emph{Degree 2.}
The bar differential encodes the RTT relation.
For generators at levels $r, s \geq 1$:
\begin{align*}
d([t_{ij}^{(r)} | t_{kl}^{(s)}])
&= t_{ij}^{(r)} t_{kl}^{(s)} - t_{kl}^{(s)} t_{ij}^{(r)} \\
&\quad - t_{kj}^{(r)} t_{il}^{(s-1)} + t_{il}^{(s)} t_{kj}^{(r-1)} \\
&= [t_{ij}^{(r)}, t_{kl}^{(s)}]
- t_{kj}^{(r)} t_{il}^{(s-1)} + t_{il}^{(s)} t_{kj}^{(r-1)}.
\end{align*}

\emph{Serre relations.}
For $\mathfrak{sl}_3$, the Serre relations are:
\[
[t_{12}^{(1)}, [t_{12}^{(1)}, t_{23}^{(1)}]] = 0, \qquad
[t_{23}^{(1)}, [t_{23}^{(1)}, t_{12}^{(1)}]] = 0.
\]
These appear as degree-3 bar cocycles: the element
$[t_{12}^{(1)} | t_{12}^{(1)} | t_{23}^{(1)}]
- 2[t_{12}^{(1)} | t_{23}^{(1)} | t_{12}^{(1)}]
+ [t_{23}^{(1)} | t_{12}^{(1)} | t_{12}^{(1)}]$
is exact in $\bar{B}^3$, reflecting the Serre relation.

\emph{Dimension table.}
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|ccc}
Filtration degree & $\dim \bar{B}^1$ & $\dim \bar{B}^2$ &
$\dim H^2$ \\
\hline
$\leq 1$ & 9 & 81 & 27 \\
$\leq 2$ & 18 & 324 & 99
\end{tabular}
\end{center}
The degree-2 cohomology satisfies $\dim H^2 = 3 \dim H^2(\mathfrak{sl}_2)$
at each filtration level, reflecting the rank-2 structure.
\end{computation}

\begin{proposition}[Rank dependence of Yangian bar complex; \ClaimStatusProvedHere]
\label{prop:yangian-rank-dependence}
\index{Yangian!bar complex!rank dependence}

For the Yangian $Y(\mathfrak{sl}_N)$ at filtration degree $\leq 1$:
\begin{enumerate}
\item $\dim \bar{B}^1 = N^2 - 1$ (the dimension of $\mathfrak{sl}_N$,
after removing the trace condition $\sum_i t_{ii}^{(r)} = 0$).
\item $\dim \bar{B}^2 = (N^2-1)^2$ (all pairs of generators).
\item $\dim H^2 = \dim \mathfrak{sl}_N \otimes \mathfrak{sl}_N / \mathrm{Im}(d_2)
= (N^2-1)^2 - \binom{N^2-1}{2} + \dim(\mathrm{Serre})$,
where $\mathrm{Serre}$ denotes the Serre relation space.
\item The growth rate is polynomial in $N$: $\dim \bar{B}^n = O(N^{2n})$.
\end{enumerate}
\end{proposition}

\begin{proof}
(i) At filtration degree 1, the generators are
$\{t_{ij}^{(1)} : 1 \leq i,j \leq N\}$ modulo the trace
condition, giving $N^2 - 1$ generators (matching $\dim \mathfrak{sl}_N$).

(ii) The degree-2 bar space $\bar{B}^2$ is spanned by
$\{[t_{ij}^{(1)} | t_{kl}^{(1)}]\}$, giving $(N^2-1)^2$ elements.

(iii) The image of $d_2$ consists of the RTT relations at
level $(1,1)$: $[t_{ij}^{(1)}, t_{kl}^{(1)}] - \delta_{kj} t_{il}^{(1)}
+ \delta_{il} t_{kj}^{(1)}$.  These span a subspace of
dimension $(N^2-1)^2 - \dim H^2$, and $H^2 \cong \mathfrak{sl}_N[u]/(u^2)$
at this filtration level.

(iv) At filtration degree $\leq n$, $\dim \bar{B}^1 = (N^2-1)n$,
so $\dim \bar{B}^k = O(N^{2k} n^k)$.  For fixed $n$, this is
polynomial in $N$.
\end{proof}


\subsection{Quantum group connection}

\begin{remark}[Yangian vs quantum group bar complexes]
\label{rem:yangian-quantum-group}
\index{quantum group!bar complex}

The Yangian $Y(\fg)$ and the quantum group $U_q(\fg)$ are
related by the ``rational vs trigonometric'' dichotomy:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{l|cc}
& \textbf{Yangian $Y(\fg)$} & \textbf{$U_q(\hat{\fg})$} \\
\hline
$R$-matrix type & Rational: $R(u) = 1 - P/u$ &
Trigonometric: $R(u) = q^{u} P + q^{-u}(1-P)$ \\
Spectral parameter & Additive: $u - v$ & Multiplicative: $u/v$ \\
Bar propagator & $\eta_{ij} = d\log(z_i - z_j)$ &
$\omega_{ij} = d\log(z_i/z_j)$ \\
FM space & $\overline{\mathrm{FM}}_n(\mathbb{C})$ &
$\overline{\mathrm{FM}}_n(\mathbb{C}^\times)$ \\
Koszul dual & $Y_{R^{-1}}(\fg)$ & $U_{q^{-1}}(\hat{\fg})$
\end{tabular}
\end{center}

The bar complex of $U_q(\hat{\fg})$ uses the trigonometric
$R$-matrix as the degree-2 differential.  The FM compactification
$\overline{\mathrm{FM}}_n(\mathbb{C}^\times)$ replaces the
rational propagator $d\log(z_i - z_j)$ with the trigonometric
propagator $d\log(z_i/z_j)$, and the boundary strata of the
FM space encode the coproduct structure of the quantum group.

At the critical value $q = 1$, the quantum group degenerates to
the universal enveloping algebra $U(\fg)$ and the trigonometric
$R$-matrix becomes the identity, reducing the bar complex to
the ordinary Chevalley--Eilenberg complex.  This mirrors the
genus-0 critical level phenomenon for affine KM algebras
(Proposition~\ref{prop:km-bar-curvature} at $k = -h^\vee$).
\end{remark}


\subsection{Transfer matrix and Baxter Q-operator}

\begin{computation}[Transfer matrix in the bar complex]
\label{comp:transfer-matrix-bar}
\index{transfer matrix!bar complex}

The transfer matrix of the Yangian $Y(\mathfrak{sl}_2)$ is:
\[
\mathcal{T}(u) = \operatorname{tr}_{\mathbb{C}^2} T(u)
= t_{11}(u) + t_{22}(u).
\]
This is a generating function for commuting elements:
$[\mathcal{T}(u), \mathcal{T}(v)] = 0$ for all $u, v$.

\emph{Bar complex interpretation.}
The commutativity $[\mathcal{T}(u), \mathcal{T}(v)] = 0$ means that
the degree-2 bar element
\[
[\mathcal{T}^{(r)} | \mathcal{T}^{(s)}] \in \bar{B}^2(Y(\mathfrak{sl}_2))
\]
is exact for all $r, s$:
$d[\mathcal{T}^{(r)} | \mathcal{T}^{(s)}] = [\mathcal{T}^{(r)},
\mathcal{T}^{(s)}] = 0$.
Thus the elements $[\mathcal{T}^{(r)} | \mathcal{T}^{(s)}]$
are \emph{bar cocycles}, defining classes in $H^2(\bar{B})$.
These classes are in fact trivial (exact) because
$\mathcal{T}^{(r)} = t_{11}^{(r)} + t_{22}^{(r)}$ is a sum
of generators, and:
\begin{align*}
d([t_{11}^{(r)} | t_{22}^{(s)}] + [t_{22}^{(s)} | t_{11}^{(r)}])
&= [t_{11}^{(r)}, t_{22}^{(s)}] + [t_{22}^{(s)}, t_{11}^{(r)}] = 0.
\end{align*}

The Baxter Q-operator $Q(u)$ satisfies
$\mathcal{T}(u) Q(u) = Q(u+1) + Q(u-1)$
(the TQ-relation).  In the bar complex, this relation
appears as a degree-2 cocycle condition with a shift
in the spectral parameter filtration.
\end{computation}


\section{Yangian representation theory and bar modules}
\label{sec:yangian-rep-bar}
\index{Yangian!representations}

\begin{definition}[Evaluation module]\label{def:evaluation-module}
\index{evaluation module|textbf}
For $a \in \mathbb{C}$ and $V$ a finite-dimensional
$\mathfrak{sl}_N$-representation, the \emph{evaluation module}
$V(a)$ of $Y(\mathfrak{sl}_N)$ is defined by:
\[
t_{ij}(u) \mapsto \delta_{ij} + \frac{\rho_{ij}}{u - a}
\]
where $\rho_{ij}$ is the $\mathfrak{sl}_N$-action on $V$.
The parameter $a$ is the \emph{evaluation point}.
\end{definition}

\begin{proposition}[Evaluation module bar complex; \ClaimStatusProvedHere]
\label{prop:eval-module-bar}

The bar complex of the evaluation module $V(a)$ is the
Chevalley--Eilenberg complex of $\mathfrak{sl}_N$ acting on $V$,
shifted by the evaluation point:
\[
\bar{B}(Y(\mathfrak{sl}_N); V(a))
\simeq C^*(\mathfrak{sl}_N, V)
\]
as a chain complex, where the differential uses the
$\mathfrak{sl}_N$-action $\rho$.
\end{proposition}

\begin{proof}
The evaluation homomorphism $\operatorname{ev}_a \colon Y(\mathfrak{sl}_N) \to U(\mathfrak{sl}_N)$
sends $t_{ij}^{(r)}$ to $\rho_{ij} \delta_{r,1}$ (all higher modes act trivially).
The bar complex of the module $V(a)$ is therefore determined by
the $\mathfrak{sl}_N$-action alone: the degree-$n$ bar differential
\[
d([v_1 | \cdots | v_n]) = \sum_{k} (-1)^k [\cdots | \rho(v_k) | \cdots]
+ \sum_{i < j} (-1)^{i+j} [\cdots | [v_i, v_j] | \cdots]
\]
is exactly the Chevalley--Eilenberg differential.  The evaluation
point $a$ appears only as a shift in the spectral parameter filtration
and does not affect the chain complex structure.
\end{proof}

\begin{remark}[Tensor product of evaluation modules]
\label{rem:tensor-eval}
\index{evaluation module!tensor product}

For two evaluation modules $V(a)$ and $W(b)$ with $a \neq b$,
the tensor product $V(a) \otimes W(b)$ is again a Yangian module.
The bar complex of the tensor product is related to the bar complexes
of the factors by a K\"unneth-type formula:
\[
\bar{B}(Y; V(a) \otimes W(b)) \simeq
\bar{B}(Y; V(a)) \otimes_{\bar{B}(Y)} \bar{B}(Y; W(b))
\]
where the tensor product is taken over the bar construction.

When $a = b$ (coincident evaluation points), the tensor product
develops a pole: the $R$-matrix $R(a-b) = R(0) = 1 - P/0$ diverges.
This singularity corresponds to the non-semisimplicity of the tensor
product at coincident points, mirroring the diagonal singularity
in the FM compactification.  The resolution of this singularity
(the FM blow-up at the diagonal) produces the Yangian analog of
the bar complex propagator.
\end{remark}

\subsection{Yangian category $\mathcal{O}$ from bar-cobar duality}
\label{sec:yangian-category-O}
\index{Yangian!category O}

The module Koszul duality of
Theorem~\ref{thm:e1-module-koszul-duality}, combined with
the evaluation module bar complex
(Proposition~\ref{prop:eval-module-bar}), yields a
BGG-type resolution theory for Yangian modules.

\begin{theorem}[Yangian BGG resolution; \ClaimStatusProvedHere]
\label{thm:yangian-bgg}
\index{BGG resolution!Yangian}
For $\mathfrak{g} = \mathfrak{sl}_N$ and a finite-dimensional
irreducible $\mathfrak{g}$-module $L(\lambda)$, the evaluation
module $L(\lambda)(a)$ of $Y(\mathfrak{sl}_N)$ admits a
bar resolution:
\begin{equation}\label{eq:yangian-bgg}
\cdots \to \bar{B}_2(Y; L(\lambda)(a))
  \to \bar{B}_1(Y; L(\lambda)(a))
  \to L(\lambda)(a) \to 0
\end{equation}
where $\bar{B}_n(Y; L(\lambda)(a))
\simeq C^n(\mathfrak{sl}_N, L(\lambda))$
is the Chevalley--Eilenberg complex.  In particular:
\begin{enumerate}[label=\textup{(\roman*)}]
\item The resolution length is $\dim \mathfrak{sl}_N
  = N^2 - 1$.
\item The terms are
  $\bar{B}_n \cong L(\lambda) \otimes
  \bigwedge^n \mathfrak{sl}_N^*$,
  with differentials given by the dual of the
  $\mathfrak{sl}_N$-action.
\item Under Koszul duality $\Phi\colon
  Y(\mathfrak{sl}_N)\text{-mod}
  \to Y_{-\hbar}(\mathfrak{sl}_N)\text{-comod}$,
  the resolution maps to the dual BGG resolution
  of $L(\lambda)(-a)$ for the Koszul dual Yangian
  \textup{(}Proposition~\textup{\ref{prop:yangian-module-koszul})}.
\end{enumerate}
\end{theorem}

\begin{proof}
Parts~(i) and~(ii) follow immediately from
Proposition~\ref{prop:eval-module-bar}: the bar complex
is the CE complex $C^*(\mathfrak{sl}_N, L(\lambda))$,
which has terms $L(\lambda) \otimes \bigwedge^n \mathfrak{sl}_N^*$
and length $\dim \mathfrak{sl}_N$.

For part~(iii): Proposition~\ref{prop:yangian-module-koszul}
establishes $\Phi(V(a)_\hbar) = V(-a)_{-\hbar}$.
The bar-cobar functor $\Phi$ applied termwise to the
resolution~\eqref{eq:yangian-bgg} gives:
\[
\Phi(\bar{B}_n(Y_\hbar; L(\lambda)(a)))
\simeq \bar{B}_n(Y_{-\hbar}; L(\lambda)(-a))
\]
because $\Phi$ acts on the CE complex by inverting
$\hbar$ (which replaces $R$ by $R^{-1}$) and negating
the evaluation point.  The CE differential is
$\hbar$-independent (it depends only on the
$\mathfrak{sl}_N$-action), so the dual resolution
is isomorphic as a chain complex.
\end{proof}

\begin{corollary}[Ext exchange for Yangian modules;
\ClaimStatusProvedHere]
\label{cor:yangian-ext-exchange}
\index{Ext!Yangian modules}
For evaluation modules $L(\lambda)(a)$ and $L(\mu)(b)$
of $Y(\mathfrak{sl}_N)$:
\[
\operatorname{Ext}^n_{Y_\hbar}(L(\lambda)(a), L(\mu)(b))
\;\cong\;
\operatorname{Ext}^n_{Y_{-\hbar}}(L(\lambda)(-a), L(\mu)(-b))
\]
via the module Koszul duality functor $\Phi$.
\end{corollary}

\begin{proof}
By Theorem~\ref{thm:e1-module-koszul-duality}(ii),
$\operatorname{Ext}^n_{\cA}(L, M)
\cong \operatorname{Coext}^n_{\cA^!}(\Phi(M), \Phi(L))$.
For Yangian evaluation modules, $\Phi$ sends
$(L(\lambda)(a), \hbar)$ to $(L(\lambda)(-a), -\hbar)$
(Proposition~\ref{prop:yangian-module-koszul}).
Since $Y_{-\hbar}(\mathfrak{sl}_N)$ is again a Yangian
(with inverted deformation parameter), the Coext
groups are ordinary Ext groups for $Y_{-\hbar}$, giving
the stated isomorphism.
\end{proof}

\begin{remark}[Conditional extension to full Yangian
category $\mathcal{O}$]
\label{rem:yangian-cat-O-conditional}
The results above hold unconditionally for
evaluation modules (which reduce to finite-dimensional
$\mathfrak{sl}_N$-representations).  Extending to
the \emph{full} Yangian category~$\mathcal{O}$---which
includes infinite-dimensional highest-weight modules,
Verma modules $M(\lambda, a)$, and their subquotients---requires
the Koszulness conjecture
(Remark~\ref{rem:yangian-collapse-conj}).  If $Y(\mathfrak{g})$
is Koszul, then the bar spectral sequence collapses,
the bar complex provides minimal resolutions of
\emph{all} modules in $\mathcal{O}$, and the module
Koszul duality extends to an equivalence
$D^b(\mathcal{O}_{Y_\hbar}) \simeq
D^b(\mathcal{O}_{Y_{-\hbar}})$
sending $M(\lambda, a) \mapsto M(\lambda, -a)$.
\end{remark}

\section{The derived Drinfeld--Kohno theorem}
\label{sec:derived-dk}
\index{Drinfeld--Kohno!derived|textbf}

The classical Drinfeld--Kohno theorem \cite{Kohno87, Drinfeld90}
identifies the monodromy representation of the KZ connection on
$\widehat{\mathfrak{g}}_k$ conformal blocks with the $R$-matrix
representation of $U_q(\mathfrak{g})$ at
$q = e^{\pi i/(k + h^\vee)}$.  The constructions of this chapter
and Part~I yield a \emph{derived} enhancement: $\Eone$-chiral Koszul
duality provides a chain-level equivalence that reverses braiding,
and the classical Kazhdan--Lusztig and Kazhdan equivalences
translate this into the $q \mapsto q^{-1}$ involution on quantum
group representations.

\subsection{The Drinfeld--Kohno square for affine algebras}

\begin{theorem}[Derived Drinfeld--Kohno for affine algebras;
\ClaimStatusProvedHere]\label{thm:derived-dk-affine}
\index{Drinfeld--Kohno!derived!affine algebras}
Let $k$ be a non-negative integer,
$q = e^{\pi i/(k + h^\vee)}$, and $k' = -k - 2h^\vee$
\textup{(}so $q' = e^{\pi i/(k' + h^\vee)} = q^{-1}$\textup{)}.
The module Koszul duality functor
$\Phi \colon D^b(\mathcal{O}_k^{\mathrm{int}}(\widehat{\mathfrak{g}}))
\to D^b(\mathcal{O}_{k'}(\widehat{\mathfrak{g}}))$
\textup{(}Theorem~\textup{\ref{thm:e1-module-koszul-duality})}
fits into a commutative square of braided monoidal functors:
\begin{equation}\label{eq:dk-square-affine}
\begin{tikzcd}
\mathcal{O}_k^{\mathrm{int}}(\widehat{\mathfrak{g}})
  \ar[r, "\Phi", "\sim"']
  \ar[d, "\Phi_{\mathrm{KL}}"', "\sim"]
& \mathcal{O}_{k'}(\widehat{\mathfrak{g}})
  \ar[d, "\Phi_{\mathrm{KL}}", "\sim"']
\\
\mathcal{C}(U_q(\mathfrak{g}))
  \ar[r, "q \mapsto q^{-1}"', "\sim"]
& \mathcal{C}(U_{q^{-1}}(\mathfrak{g}))
\end{tikzcd}
\end{equation}
where the vertical arrows are the Kazhdan--Lusztig equivalences
\textup{(}Theorem~\textup{\ref{thm:kazhdan-lusztig-equivalence})},
the top arrow is bar-cobar duality with braiding reversal
\textup{(}Theorem~\textup{\ref{thm:monoidal-module-koszul})}, and
the bottom arrow is the $R$-matrix inversion
$R_q \mapsto R_{q^{-1}} = R_q^{-1}$.  All four functors are
equivalences of braided monoidal categories.
\end{theorem}

\begin{proof}
We verify commutativity of the diagram on objects,
morphisms, tensor products, and braiding.

\emph{Step~1: Objects.}
The top arrow $\Phi$ sends the integrable module $L(\Lambda)$
at level~$k$ to the Koszul dual module at level $k' = -k - 2h^\vee$
(Theorem~\ref{thm:universal-kac-moody-koszul}).  The left vertical sends
$L(\Lambda)$ to the tilting module $T_q(\Lambda)$
(Theorem~\ref{thm:kazhdan-lusztig-equivalence}).  Traversing the
diagram in both directions: $\Phi_{\mathrm{KL}} \circ \Phi$
sends $L(\Lambda)$ to $T_{q^{-1}}(\Lambda)$ (since $q' = q^{-1}$),
which is $\Phi_{\mathrm{KL}}(L(\Lambda))$ in the
$q^{-1}$-category.  This equals the image under
$(q \mapsto q^{-1}) \circ \Phi_{\mathrm{KL}}$, since the
$q \mapsto q^{-1}$ functor sends $T_q(\Lambda)$ to
$T_{q^{-1}}(\Lambda)$ (the labeling by dominant weights $\Lambda$
is independent of the sign of~$q$).

\emph{Step~2: Braiding.}
The braiding on $\mathcal{O}_k^{\mathrm{int}}$ is the
monodromy of the KZ connection on conformal blocks.  By the
classical Kohno--Drinfeld theorem \cite{Kohno87, Drinfeld90},
$\Phi_{\mathrm{KL}}$ sends this monodromy to the quasi-$R$-matrix
$\mathcal{R}_q \in U_q(\mathfrak{g})^{\widehat{\otimes}\,2}$.
The top arrow $\Phi$ reverses the braiding:
$\sigma \mapsto \sigma^{-1}$
(Theorem~\ref{thm:monoidal-module-koszul}, step~3).
On the quantum group side, this corresponds to
$\mathcal{R}_q \mapsto \mathcal{R}_q^{-1} = \mathcal{R}_{q^{-1}}$,
which is precisely the bottom arrow.

\emph{Step~3: Tensor structure.}
The top arrow preserves the monoidal structure up to braiding reversal
(Theorem~\ref{thm:monoidal-module-koszul}, step~2): the bar complex
of the fusion product $M \boxtimes N$ decomposes via the Alexander--Whitney
map into the cotensor product of the individual bar complexes.
The KL equivalence identifies the fusion product with the quantum group
tensor product \cite{KL93}, and the $q \mapsto q^{-1}$ functor
preserves the tensor product structure (the underlying fusion rules
depend only on $\mathfrak{g}$, not on the sign of~$q$).
\end{proof}

\begin{remark}[The DK square as geometry]\label{rem:dk-square-geometry}
The four arrows of the DK square~\eqref{eq:dk-square-affine} have
distinct geometric origins:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{ll}
\textbf{Top}: & Verdier duality on $\overline{C}_n(\bP^1)$
  reverses collision cycles \\
\textbf{Left/Right}: & KZ monodromy $=$ quasi-$R$-matrix
  (analytic continuation) \\
\textbf{Bottom}: & $R_q \mapsto R_{q^{-1}} = R_q^{-1}$
  (Hopf algebra involution)
\end{tabular}
\end{center}
The commutativity of the square is the statement that
bar-cobar duality \emph{geometrically implements} the quantum
group involution: the algebraic operation $q \mapsto q^{-1}$
is realized by orientation reversal on configuration spaces.
\end{remark}

\subsection{Derived Drinfeld--Kohno for Yangians}

\begin{theorem}[Derived Drinfeld--Kohno for Yangians;
\ClaimStatusProvedHere]\label{thm:derived-dk-yangian}
\index{Drinfeld--Kohno!derived!Yangian}
%
\textup{[Category: full abelian
$\operatorname{Rep}_{\mathrm{fd}}(U_q(\mathfrak{g}))$,
not the semisimplified tilting category
$\mathcal{C}(U_q(\mathfrak{g}))$ of
Theorem~\textup{\ref{thm:derived-dk-affine}}---the
Kazhdan equivalence for Yangians targets the full
finite-dimensional category.]}
%
For $\mathfrak{g} = \mathfrak{sl}_N$, the $\Eone$-chiral Koszul
duality functor
\[
\Phi \colon D^b(\operatorname{Rep}_{\mathrm{fd}}(Y_\hbar(\mathfrak{g})))
\;\to\;
D^b(\operatorname{Rep}_{\mathrm{fd}}(Y_{-\hbar}(\mathfrak{g})))
\]
fits into a commutative diagram with the Kazhdan equivalence
$\operatorname{Rep}_{\mathrm{fd}}(Y_\hbar(\mathfrak{g}))
\simeq \operatorname{Rep}_{\mathrm{fd}}(U_q(\mathfrak{g}))
\big|_{q = e^\hbar}$ \cite{Drinfeld85}:
\begin{equation}\label{eq:dk-square-yangian}
\begin{tikzcd}
D^b(\operatorname{Rep}_{\mathrm{fd}}(Y_\hbar))
  \ar[r, "\Phi"]
  \ar[d, "\Phi_K"']
& D^b(\operatorname{Rep}_{\mathrm{fd}}(Y_{-\hbar}))
  \ar[d, "\Phi_K"]
\\
D^b(\operatorname{Rep}_{\mathrm{fd}}(U_q))
  \ar[r, "q \mapsto q^{-1}"']
& D^b(\operatorname{Rep}_{\mathrm{fd}}(U_{q^{-1}}))
\end{tikzcd}
\end{equation}
The top arrow reverses the braiding:
$\sigma_R \mapsto \sigma_{R^{-1}}$
\textup{(}Proposition~\textup{\ref{prop:yangian-module-koszul}(iii))}.
Under the Kazhdan identification $\Phi_K$, this becomes the
$q \mapsto q^{-1}$ involution on quantum group representations.
In particular, the $\Eone$-chiral bar-cobar duality of the Yangian
provides a chain-level realization of the Drinfeld--Kohno
correspondence.
\end{theorem}

\begin{proof}
\emph{Step~1: The Kazhdan equivalence.}
The Kazhdan functor~\cite{Drinfeld85}
$\Phi_K \colon \operatorname{Rep}_{\mathrm{fd}}(Y_\hbar(\mathfrak{g}))
\xrightarrow{\sim}
\operatorname{Rep}_{\mathrm{fd}}(U_q(\mathfrak{g}))$
at $q = e^\hbar$ is a monoidal equivalence that sends
the $R$-matrix $R_Y(u) = 1 - \hbar P/u$ of the Yangian
to the trigonometric $R$-matrix $R_q$ of the quantum group
via the exponential map
$R_Y(u) \mapsto R_q(e^u)$.  On evaluation modules,
$\Phi_K(V(a)_\hbar) = V_q(e^a)$, where $V_q(z)$ denotes
the evaluation module of
$U_q(\widehat{\mathfrak{g}})$ at spectral parameter
$z = e^a$.

\smallskip\noindent\emph{Step~2: Commutativity on evaluations.}
The top arrow~$\Phi$ sends $V(a)_\hbar$ to
$V(-a)_{-\hbar}$
(Proposition~\ref{prop:yangian-module-koszul}(i)).
Applying~$\Phi_K$ to the right column yields
$V(-a)_{-\hbar}$ $\mapsto$ $V_{q^{-1}}(e^{-a})$.
Applying~$\Phi_K$ first (left column) followed by
$q \mapsto q^{-1}$ gives
\[
V(a)_\hbar \,\mapsto\, V_q(e^a) \,\mapsto\, V_{q^{-1}}(e^a).
\]
These agree when
$V_{q^{-1}}(e^{-a}) \cong V_{q^{-1}}(e^a)$
as abstract $U_{q^{-1}}(\mathfrak{g})$-modules, which
holds because evaluation modules of a quantum loop
algebra at the same highest weight are isomorphic as
$U_{q^{-1}}(\mathfrak{g})$-modules
(the evaluation point only affects the loop algebra
action, not the underlying
$U_{q^{-1}}(\mathfrak{g})$-structure).

\emph{Step~3: Braiding.}
On the Yangian side, the braiding is
$\sigma_Y = P \circ R_Y(u)$, and $\Phi$ sends it to
$\sigma_Y^{-1} = P \circ R_Y^{-1}(u)$
(Proposition~\ref{prop:yangian-module-koszul}(iii)).
The Kazhdan functor identifies $R_Y(u)$ with $R_q(e^u)$,
so $R_Y^{-1}(u) \mapsto R_q^{-1}(e^u) = R_{q^{-1}}(e^u)$.
This is precisely the braiding on the $q^{-1}$ side.

\emph{Step~4: Ext exchange.}
Corollary~\ref{cor:yangian-ext-exchange} gives
\[
\operatorname{Ext}^n_{Y_\hbar}(V(a), W(b))
\cong \operatorname{Ext}^n_{Y_{-\hbar}}(V(-a), W(-b)).
\]
\textup{[Category: full abelian
$\operatorname{Rep}_{\mathrm{fd}}(U_q(\mathfrak{g}))$
via Kazhdan equivalence.]}
Via $\Phi_K$, this becomes
\[
\operatorname{Ext}^n_{U_q}(V_q(e^a), W_q(e^b))
\cong \operatorname{Ext}^n_{U_{q^{-1}}}(V_{q^{-1}}(e^{-a}),
W_{q^{-1}}(e^{-b})),
\]
the derived enhancement of the classical DK identification.
\end{proof}

\subsection{The factorization-level statement}

\begin{definition}[$\Eone$-factorization category]
\label{def:e1-factorization-category}
\index{E1-factorization category@$\Eone$-factorization category|textbf}
For an $\Eone$-chiral algebra $\cA$ on a curve $X$, the
\emph{$\Eone$-factorization category}
$\operatorname{Fact}_{\Eone}(\cA)$ consists of:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Objects:} $\cA$-modules on ordered configuration spaces
  $C_n^{\mathrm{ord}}(X) = \{(z_1, \ldots, z_n) : z_i \neq z_j\}$.
\item \emph{Factorization structure:} for disjoint intervals
  $I_1, \ldots, I_k \subset X$, isomorphisms
  $\cA\text{-mod}(C_{n_1}^{\mathrm{ord}}(I_1) \times \cdots
  \times C_{n_k}^{\mathrm{ord}}(I_k))
  \xrightarrow{\sim}
  \cA\text{-mod}(C_n^{\mathrm{ord}}(\bigcup I_j))$.
\item \emph{Braiding:} the braid group $B_n$ acts on
  $C_n^{\mathrm{ord}}(X)$ by monodromy around the diagonals.
  The $R$-matrix $R_\cA$ encodes this monodromy.
\item \emph{Monoidal structure:} the fusion product
  $\boxtimes_\cA$ on modules, given by colliding
  points along ordered intervals.
\end{enumerate}
\end{definition}

\begin{theorem}[Factorization DK on evaluation locus;
\ClaimStatusProvedHere]\label{thm:factorization-dk-eval}
\index{Drinfeld--Kohno!derived!factorization}
Restrict to the full subcategory
$\operatorname{Fact}_{\Eone}^{\mathrm{eval}}(Y(\mathfrak{sl}_N))$
generated by tensor products of evaluation modules.  The
$\Eone$-chiral bar-cobar functor induces an equivalence of
$\Eone$-factorization categories:
\begin{equation}\label{eq:factorization-dk-eval}
\Phi \colon
\operatorname{Fact}_{\Eone}^{\mathrm{eval}}(Y_\hbar(\mathfrak{sl}_N))
\;\xrightarrow{\;\sim\;}\;
\operatorname{Fact}_{\Eone}^{\mathrm{eval}}
  (Y_{-\hbar}(\mathfrak{sl}_N))^{\mathrm{op}}
\end{equation}
with the following properties:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{On objects:} $V(a_1) \otimes \cdots \otimes V(a_n)
  \;\mapsto\; V(-a_1) \otimes \cdots \otimes V(-a_n)$
  with $\hbar \mapsto -\hbar$.
\item \emph{On braiding:}
  $R(u; \hbar) \;\mapsto\; R^{-1}(u; \hbar) = R(u; -\hbar)$.
\item \emph{On factorization:} the factorization isomorphisms
  are preserved, with the $\Eone$-ordering reversed
  \textup{(}reflecting the ``op'' on the target\textup{)}.
\end{enumerate}
\end{theorem}

\begin{proof}
Part~(i) is Proposition~\ref{prop:yangian-module-koszul}(i)
applied to each tensor factor.  Part~(ii) is
Proposition~\ref{prop:yangian-module-koszul}(iii).  For
part~(iii): the factorization isomorphisms on the source are
encoded by the bar complex propagator $\eta_{ij} = d\log(z_i - z_j)$
on $\overline{C}_n^{\mathrm{ord}}(X)$.  The Verdier involution
(Theorem~\ref{thm:bar-cobar-verdier}) acts on the
compactified configuration space by reversing the orientation of
collision cycles, which reverses the ordering of the
$\Eone$-intervals.  The bar-cobar quasi-isomorphism
(Corollary~\ref{cor:yangian-bar-cobar}) ensures this
factorization structure is inherited by the dual, producing
the ``op'' monoidal structure on the target.
\end{proof}

\begin{remark}[Rational vs.\ trigonometric DK]
\label{rem:rational-trigonometric-dk}
\index{Drinfeld--Kohno!rational vs trigonometric}
The DK square takes different forms depending on the $R$-matrix type:

\emph{Rational DK}
\textup{(}Theorem~\textup{\ref{thm:derived-dk-yangian})}.
The Yangian $R$-matrix $R(u) = 1 - \hbar P/u$ is rational
in the additive spectral parameter $u$.  Koszul duality sends
$\hbar \to -\hbar$, i.e., $R(u) \to R^{-1}(u)$.  The bar
complex uses the propagator $d\log(z_i - z_j)$ on the
FM compactification of $\bC$.

\emph{Trigonometric DK}
\textup{(}Remark~\textup{\ref{rem:yangian-quantum-group})}.
The quantum loop algebra $U_q(\widehat{\mathfrak{g}})$ has
trigonometric $R$-matrix $R(u/v)$ in the multiplicative
parameter.  Koszul duality sends $q \to q^{-1}$.  The bar
complex uses the propagator $d\log(z_i/z_j)$ on the
FM compactification of $\bC^\times$.

The Kazhdan functor $\Phi_K$ intertwines the rational and
trigonometric pictures via the exponential map $u \mapsto e^u$,
$\bC \to \bC^\times$.  The DK
square~\eqref{eq:dk-square-yangian} encodes the commutativity
of this intertwining with bar-cobar duality.
\end{remark}

The Yangian chapter reveals the $\Eone$-chiral face of the theory.
The derived Drinfeld--Kohno theorem
(Theorems~\ref{thm:derived-dk-affine}
and~\ref{thm:derived-dk-yangian}) proves that bar-cobar duality
\emph{geometrically implements} the quantum group involution
$q \mapsto q^{-1}$, via orientation reversal on ordered
configuration spaces.  The $\Einf$-chiral examples of the preceding
chapters all have commutative factorization structure; the Yangian is
the first object whose Koszul duality intertwines genuinely
noncommutative (braided) monoidal data.  The extension from the
evaluation locus (Theorem~\ref{thm:factorization-dk-eval}) to the
full factorization categorical equivalence
(Conjecture~\ref{conj:full-derived-dk}) is the remaining frontier of
the $\Eone$-chiral theory, and the natural bridge to the modular
Koszul programme (\S\ref{sec:modular-koszul-programme}).
