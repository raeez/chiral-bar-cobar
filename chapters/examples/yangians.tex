\chapter{Yangians and shifted Yangians}
\label{chap:yangians}

This chapter treats Yangians and their shifted variants, with connections to 
Coulomb branches and cohomological Hall algebras.


\section{Yangian vertex algebra}
\label{sec:yangian-vertex}

\subsection{Yangian definition}

\begin{definition}[Yangian]\label{def:yangian}
\index{Yangian|textbf}
The \emph{Yangian} $Y(\mathfrak{g})$ associated to a simple Lie algebra $\mathfrak{g}$ 
is the associative algebra generated by $J^{(r)}_a$ for $a = 1, \ldots, \dim\mathfrak{g}$ 
and $r \geq 0$, with relations:
\begin{align*}
[J^{(0)}_a, J^{(0)}_b] &= f_{ab}^c J^{(0)}_c \\
[J^{(0)}_a, J^{(r)}_b] &= f_{ab}^c J^{(r)}_c \\
[J^{(1)}_a, [J^{(1)}_b, J^{(0)}_c]] - [J^{(0)}_a, [J^{(1)}_b, J^{(1)}_c]] &= \hbar^2 A_{abc}^{def} \{J^{(0)}_d, J^{(0)}_e, J^{(0)}_f\}
\end{align*}
where $\{x,y,z\} = \frac{1}{6}\sum_{\sigma \in S_3} x_{\sigma(1)}y_{\sigma(2)}z_{\sigma(3)}$ is the symmetrized triple product ($|S_3| = 6$), and $A_{abc}^{def}$ are structure constants determined by $\mathfrak{g}$ (Drinfeld's ``terrific'' relation).
\end{definition}

\begin{definition}[Yangian current]\label{def:yangian-current}
The \emph{Yangian current} is the generating function:
\[
J_a(u) = \sum_{r \geq 0} J^{(r)}_a u^{-r-1}
\]
\end{definition}

\begin{definition}[RTT presentation]\label{def:yangian-rtt}
\index{Yangian!RTT presentation}
\index{R-matrix}
The \emph{RTT presentation} (Yang--Baxter presentation) of
$Y(\mathfrak{g})$ is generated by a \emph{transfer matrix}
$T(u) = (T_{ij}(u))_{i,j=1}^{\dim V}
\in \mathrm{End}(V) \otimes Y(\mathfrak{g})$
(valued in $\mathrm{End}(V)$ for a fundamental representation~$V$,
not in~$\mathfrak{g}$), satisfying the \emph{RTT relation}:
\begin{equation}\label{eq:rtt-relation}
R_{12}(u-v)\, T_1(u)\, T_2(v)
\;=\; T_2(v)\, T_1(u)\, R_{12}(u-v)
\end{equation}
where $R_{12}(u) = 1 - \hbar P_{12}/u + O(u^{-2})$ is the
Yang $R$-matrix, $P_{12}$ is the permutation, and the subscripts
$1, 2$ indicate the tensor factor in
$\mathrm{End}(V) \otimes \mathrm{End}(V) \otimes Y(\mathfrak{g})$.
Expanding in modes
$T_{ij}(u) = \delta_{ij} + \hbar \sum_{r \geq 0} T_{ij}^{(r)} u^{-r-1}$,
the RTT relation \eqref{eq:rtt-relation} is equivalent to:
\begin{equation}\label{eq:rtt-modes}
[T_{ij}^{(r+1)}, T_{kl}^{(s)}]
- [T_{ij}^{(r)}, T_{kl}^{(s+1)}]
\;=\; T_{kj}^{(r)} T_{il}^{(s)}
- T_{kj}^{(s)} T_{il}^{(r)}.
\end{equation}
The RTT relation is \emph{quadratic} in the entries of $T$, making
this presentation directly amenable to quadratic Koszul duality
(mechanism~(iii) of Remark~\ref{rem:three-koszul-mechanisms}).
The isomorphism with the Drinfeld presentation
(Definition~\ref{def:yangian}) is given by
$J_a^{(0)} = \sum_{i,j} \rho(e_a)_{ij}\, T_{ij}^{(0)}$
and higher modes from the coproduct structure.
\end{definition}


\subsection{Yangian chiral algebra}

\begin{construction}[Yangian vertex algebra]\label{constr:yangian-va}
The \emph{Yangian chiral algebra} $Y(\mathfrak{g})^{\mathrm{ch}}$ is constructed as follows:

\emph{State space.} Completion of $Y(\mathfrak{g})$-modules.

\emph{Vertex operators.} $Y(J^{(r)}_a, z) = \sum_n J^{(r)}_{a,n} z^{-n-r-1}$.

\emph{OPE.} The RTT relation translates to:
\[
J_a(z) J_b(w) = \frac{f_{ab}^c J_c(w)}{z - w} + \frac{\alpha_{ab}^{cd} {:}J_c(w) J_d(w){:}}{(z-w)^2} + \text{regular}
\]
with higher poles from the quadratic Serre relations.
\end{construction}

\begin{theorem}[Yangian as $\Eone$-chiral; \ClaimStatusProvedHere]\label{thm:yangian-e1}
In the RTT presentation, the Yangian $Y(\mathfrak{g})^{\mathrm{ch}}$ carries the structure of a quadratic $\Eone$-chiral algebra (Definition~\ref{def:e1-chiral}).  The $\Eone$-locality axiom is $R$-twisted commutativity:
\begin{equation}\label{eq:r-twisted-locality}
J_a(z) J_b(w) - R^{ab}_{cd}(z-w)\, J_d(w) J_c(z) = 0.
\end{equation}
For nontrivial $R \neq P$ (the permutation), this is strictly $\Eone$,
not $\Einf$.
\end{theorem}

\begin{proof}
\emph{Step~1: Quadraticity.}  The RTT relation~\eqref{eq:rtt-relation} is manifestly quadratic in the entries $T_{ij}(u)$.  More precisely, expanding $T_1(u) T_2(v)$ and $T_2(v) T_1(u)$ in the mode expansion $T_{ij}(u) = \delta_{ij} + \hbar \sum_r T_{ij}^{(r)} u^{-r-1}$, the relation~\eqref{eq:rtt-modes} expresses a quadratic relation among the generators $\{T_{ij}^{(r)}\}$.

\emph{Step~2: $\Eone$-chiral structure.}
The vertex operators $Y(T_{ij}^{(r)}, z) = \sum_n T_{ij,n}^{(r)} z^{-n-r-1}$ define fields on a curve $X$.  The OPE:
\[
T_{ij}(z,u)\, T_{kl}(w,v) \sim \frac{R^{ij,kl}(u-v)}{z-w} \cdot (\text{fusion})
\]
is $R$-matrix-valued, with simple poles.  The $R$-matrix $R(u) = 1 + \hbar r/u + O(u^{-2})$ provides the braiding: $\sigma_{12} \circ \mu = R_{12} \cdot \mu \circ \sigma_{12}$, which is the $\Eone$-locality axiom.  Associativity follows from the Yang--Baxter equation $R_{12}R_{13}R_{23} = R_{23}R_{13}R_{12}$.

\emph{Step~3: Non-commutativity.}
For $\mathfrak{g} = \mathfrak{sl}_2$, the classical $r$-matrix $r = \frac{1}{2}(e \otimes f + f \otimes e) + \frac{1}{4} h \otimes h$ (the split Casimir) satisfies $r \neq 0$, so the $R$-matrix $R(u) = 1 + \hbar r/u + O(u^{-2})$ is non-trivial: $R(u) \neq P$ for finite~$u$, and the RTT relation does not reduce to commutativity.  Thus the algebra is strictly $\Eone$.
\end{proof}

\begin{remark}[Presentations and quadraticity]\label{rem:yangian-presentations}
Whether the Yangian is ``quadratic'' depends on the presentation:
\begin{itemize}
\item In the \emph{RTT presentation} (Definition~\ref{def:yangian-rtt}), the
defining relation \eqref{eq:rtt-relation} is
quadratic in $T$, so standard quadratic Koszul duality (mechanism~(iii) of
Remark~\ref{rem:three-koszul-mechanisms}) applies directly.
\item In the \emph{Drinfeld presentation} (Definition~\ref{def:yangian}), the
``terrific relation'' is cubic in the generators, so Koszul duality requires
nilpotent completion (Appendix~\ref{app:nilpotent-completion}).
\end{itemize}
The two presentations yield isomorphic algebras but different Koszul dual
computations.  A rigorous construction of the $\Eone$-chiral structure and its
Koszul dual should specify which presentation is used.
\end{remark}


\section{Koszul dual of the Yangian}\label{sec:yangian-koszul}

\subsection{RTT bar complex computation}

\begin{theorem}[Yangian bar complex via RTT; \ClaimStatusProvedHere]\label{thm:yangian-bar-rtt}
For $\mathfrak{g} = \mathfrak{sl}_2$, in the RTT presentation, the bar complex $\bar{B}(Y(\mathfrak{sl}_2)^{\mathrm{ch}})$ is computed as follows.

\emph{Generators.} In bar degree~1, the generators are the duals $T_{ij}^{(r)*}$ of the RTT generators, placed on $\overline{C}_2(X)$ with log forms.

\emph{Differential.} The bar differential extracts the RTT relation via double residues:
\begin{equation}\label{eq:yangian-bar-diff}
d(T_{ij}^{(r)*} \otimes \eta_{12}) = \sum_{k} \bigl(T_{ik}^{(r)*} \otimes T_{kj}^{(0)*} - T_{ik}^{(0)*} \otimes T_{kj}^{(r)*}\bigr) \otimes \eta_{12} \wedge \eta_{23}
\end{equation}
This is the quadratic dual of the RTT relation~\eqref{eq:rtt-modes}: the bar differential encodes the orthogonal complement of the relations under the chiral pairing.

\emph{Bar complex structure.} Since the RTT relation is quadratic, the Koszul dual $A^!$ is determined by the data in bar degrees $\leq 2$ alone: generators $V^*$ and relations $R^\perp$.  If the algebra is Koszul (i.e., $\mathrm{Ext}^{i,j}_A(\mathbb{C}, \mathbb{C}) = 0$ for $i \neq j$), then the bar-cobar quasi-isomorphism recovers $A$ from $A^!$.
\end{theorem}

\begin{proof}
The RTT relation~\eqref{eq:rtt-modes} is:
$[T_{ij}^{(r+1)}, T_{kl}^{(s)}] - [T_{ij}^{(r)}, T_{kl}^{(s+1)}] = T_{kj}^{(r)} T_{il}^{(s)} - T_{kj}^{(s)} T_{il}^{(r)}$.
This is quadratic in $T$, so the bar complex of the quadratic algebra has:
\begin{itemize}
\item $\bar{B}^0 = \mathbb{C}$ (vacuum),
\item $\bar{B}^1 = V^* = \mathrm{span}\{T_{ij}^{(r)*}\}$ (dual generators on $\overline{C}_2(X)$),
\item $\bar{B}^2 = R^\perp \subset V^* \otimes V^*$ (orthogonal complement of relations on $\overline{C}_3(X)$).
\end{itemize}
The differential $d\colon \bar{B}^2 \to \bar{B}^1$ is the dual of the inclusion $R \hookrightarrow V \otimes V$, which gives~\eqref{eq:yangian-bar-diff}.
\end{proof}

\subsection{Koszul dual identification}

\begin{theorem}[Yangian Koszul dual; \ClaimStatusProvedHere]\label{thm:yangian-koszul-dual}
\index{Yangian!Koszul dual}
The Koszul dual of the Yangian $Y(\mathfrak{sl}_2)^{\mathrm{ch}}$ in the RTT presentation is the Yangian with inverse $R$-matrix:
\begin{equation}\label{eq:yangian-koszul-dual}
Y(\mathfrak{sl}_2)^{!,\mathrm{ch}} \;\cong\; Y_{R^{-1}}(\mathfrak{sl}_2)^{\mathrm{ch}}
\end{equation}
where $Y_{R^{-1}}$ is the RTT algebra defined by the relation:
\begin{equation}
R^{-1}_{12}(u-v)\, T_1^!(u)\, T_2^!(v) = T_2^!(v)\, T_1^!(u)\, R^{-1}_{12}(u-v)
\end{equation}
with $R^{-1}(u) = 1 + \hbar P/u + O(u^{-2})$ (note the sign change from $R(u) = 1 - \hbar P/u + \cdots$).

For the Yang $R$-matrix $R(u) = 1 - \hbar P/u$, the inverse is $R^{-1}(u) = u/(u - \hbar P) = \sum_{n \geq 0} (\hbar P/u)^n$, where $P^2 = 1$ gives $R^{-1}(u) = 1 + \hbar P/u + \hbar^2/u^2 + \hbar^3 P/u^3 + \cdots$.  The Koszul dual RTT relation is the same quadratic relation with $\hbar \to -\hbar$.
\end{theorem}

\begin{proof}
\emph{Step~1: Quadratic dual.}
The RTT algebra is quadratic: $V = \mathrm{span}\{T_{ij}^{(r)}\}$, $R = \mathrm{span}\{[\text{RTT relations}]\} \subset V \otimes V$.  The quadratic Koszul dual has generators $V^*$ and relations $R^\perp$ (the orthogonal complement under the pairing $V^* \otimes V^* \times V \otimes V \to \mathbb{C}$).

\emph{Step~2: Orthogonal complement computation.}
The RTT relation $R_{12}(u-v) T_1 T_2 = T_2 T_1 R_{12}(u-v)$ can be written as:
\[
(1 - \hbar P/(u-v)) T_1 T_2 = T_2 T_1 (1 - \hbar P/(u-v))
\]
Expanding: $T_1 T_2 - T_2 T_1 = \frac{\hbar}{u-v}(P T_1 T_2 - T_2 T_1 P)$.

The orthogonal complement $R^\perp$ consists of quadratic expressions $Q(T^*, T^*)$ that pair to zero with all elements of $R$.  For $\mathfrak{sl}_2$ (where $V = \mathbb{C}^2$ and $P$ is the $4 \times 4$ permutation matrix), $R^\perp$ is computed by direct matrix calculation.  The RTT relation at fixed spectral parameters defines a linear map $\Phi_R\colon \mathrm{End}(V)^{\otimes 2} \to \mathrm{End}(V)^{\otimes 2}$ by $X \mapsto R_{12} X - \sigma(X) R_{12}$ where $\sigma$ is the flip; its image is the space of relations~$R$.  Writing out the $16 \times 16$ matrix of $\Phi_R$ in the standard basis of $\mathrm{End}(\mathbb{C}^2)^{\otimes 2}$ and computing its orthogonal complement with respect to the trace pairing, one verifies that $R^\perp$ coincides with the image of $\Phi_{R^{-1}}$, i.e., the space of $R^{-1}$-RTT relations.

\emph{Step~2a: Local finiteness.}
The Yangian has infinitely many generators $\{T_{ij}^{(r)}\}_{r \geq 0}$ graded by level~$r$, but each graded component $V_r = \mathrm{span}\{T_{ij}^{(r)} : 1 \leq i,j \leq \dim V\}$ is finite-dimensional: $\dim V_r = (\dim V)^2 = 4$ for $\mathfrak{sl}_2$.  The RTT relation~\eqref{eq:rtt-modes} is homogeneous of degree $r + s$ in the level grading (relating generators of levels $r, s$ on both sides), so the quadratic algebra $A = T(V)/\langle R \rangle$ is locally finite: each graded component $A_n = \bigoplus_{r_1 + \cdots + r_k = n} V_{r_1} \cdots V_{r_k} / R_n$ is a quotient of a finite-dimensional space.

\emph{Step~2b: Graded Koszul duality.}
Since each graded component is finite-dimensional, the Positselski--Polishchuk framework \cite{PP05} for Koszul duality of locally finite graded algebras applies directly.  Their Theorem~2.1 establishes the bar-cobar adjunction and the Koszul property for quadratic algebras that are locally finite in a non-negative grading (which is satisfied here, with the level grading $r \geq 0$).  The key hypothesis --- that the grading is connected ($A_0 = \mathbb{C}$) --- holds after passing to the augmentation ideal.

\emph{Step~3: $R$-matrix inversion.}
For the Yang $R$-matrix, $R(u) = (u - \hbar P)/u$ (in the defining representation of $\mathfrak{sl}_2$).  The inverse is $R^{-1}(u) = u/(u - \hbar P) = 1 + \hbar P/u + (\hbar P)^2/u^2 + \cdots$.  Since $P^2 = 1$ on $V \otimes V$, this simplifies to $R^{-1}(u) = u/(u-\hbar)$ on the symmetric subspace and $R^{-1}(u) = u/(u+\hbar)$ on the antisymmetric subspace.

\emph{Step~4: Identification.}
The RTT algebra for $R^{-1}(u)$ is isomorphic to the RTT algebra for $R(u)$ with $\hbar \to -\hbar$.  To see this, note that $R^{-1}(u; \hbar) = 1 + \hbar P/u + O(u^{-2})$ agrees with $R(u; -\hbar) = 1 + \hbar P/u$ at leading order in $1/u$; the higher-order terms $\hbar^{2k}/u^{2k}$ and $\hbar^{2k+1}P/u^{2k+1}$ in $R^{-1}$ do \emph{not} appear in $R(u;-\hbar)$, but they are irrelevant: the mode expansion~\eqref{eq:rtt-modes} extracts only the leading-order data, and the higher-order terms are automatic consequences of the quadratic RTT relation via the Yang--Baxter equation.  Therefore $Y(\mathfrak{sl}_2)^! \cong Y(\mathfrak{sl}_2)^{\hbar \to -\hbar}$, confirming that the Yangian is ``almost self-dual'' up to a sign flip of the deformation parameter.
\end{proof}

\begin{corollary}[Yangian classical limit; \ClaimStatusProvedHere]\label{cor:yangian-classical-self-dual}
At $\hbar = 0$ (the classical limit), $R(u) = R^{-1}(u) = 1$, and the RTT algebra becomes commutative: $[T_{ij}^{(r)}, T_{kl}^{(s)}] = 0$, i.e., $\mathrm{Sym}(V)$ where $V = \bigoplus_{r \geq 0} \mathfrak{gl}_2^{(r)}$.  The Koszul dual of the symmetric algebra is the exterior algebra: $\mathrm{Sym}(V)^! = \bigwedge(V^*)$.  Since $R^{-1}(u;0) = R(u;0) = 1$, the defining relations of $\mathrm{Sym}(V)$ are the antisymmetric tensors $v \otimes w - w \otimes v$, and the orthogonal complement $R^\perp$ consists of the symmetric tensors $v \otimes w + w \otimes v$, which are the defining relations of $\bigwedge(V^*)$.  This confirms that the Yangian duality reduces to $\mathrm{Sym} \leftrightarrow \bigwedge$ at $\hbar = 0$.  (In the Drinfeld presentation, the classical limit is the noncommutative algebra $U(\mathfrak{g}[t])$; the two presentations yield different classical limits.)
\end{corollary}

\begin{remark}[Connection to Langlands duality]\label{rem:yangian-langlands}
The sign flip $\hbar \to -\hbar$ in the Yangian Koszul dual is the analog of the Feigin--Frenkel level shift $k \to -k - 2h^\vee$ for affine algebras.  For simply-laced $\mathfrak{g}$, the Yangian $Y(\mathfrak{g})$ is isomorphic to $Y(\mathfrak{g}^\vee)$ (since $\mathfrak{g} \cong \mathfrak{g}^\vee$), and the Koszul duality becomes a form of self-duality.  For non-simply-laced $\mathfrak{g}$, one expects $Y(\mathfrak{g})^! \simeq Y(\mathfrak{g}^\vee)$, realizing Langlands duality at the Yangian level.  Latyntsev \cite{Latyntsev23} constructs factorisation quantum groups that provide a complementary approach to Yangian duality via factorisation homology.
\end{remark}

\subsection{Koszulness and bar-cobar recovery}

\begin{proposition}[RTT Yangian is Koszul; \ClaimStatusProvedHere]\label{prop:yangian-koszul}
\index{Yangian!Koszulness}
The Yangian $Y(\mathfrak{sl}_2)$ in the RTT presentation
(Definition~\ref{def:yangian-rtt}) is Koszul: the bar complex
$\bar{B}(Y(\mathfrak{sl}_2))$ has cohomology concentrated on the
diagonal, i.e., $\mathrm{Ext}^{i,j}_{Y}(\mathbb{C}, \mathbb{C}) = 0$
for $i \neq j$ in the $(i, j)$-bigrading by homological degree and
internal (level) degree.
\end{proposition}

\begin{proof}
\emph{Step~1: PBW basis.}
Molev \cite{molev-yangians} establishes a Poincar\'{e}--Birkhoff--Witt
basis for $Y(\mathfrak{g})$ in the RTT presentation: ordered monomials
in the generators $\{T_{ij}^{(r)}\}_{1 \leq i,j \leq N,\, r \geq 0}$
(with respect to a fixed total order on the index set $(i,j,r)$) form
a vector space basis for $Y(\mathfrak{g})$.  The
associated graded algebra with respect to the PBW filtration is the
polynomial algebra $\mathrm{Sym}(V)$, where
$V = \bigoplus_{r \geq 0} \mathrm{End}(\mathbb{C}^N)_r$.

\emph{Step~2: PBW implies Koszul.}
By the Polishchuk--Positselski PBW criterion \cite[Ch.~4]{PP05}, a
quadratic algebra possessing a PBW basis is Koszul.  The argument:
the PBW property gives a flat deformation from $Y(\mathfrak{sl}_2)$
to its associated graded $\mathrm{Sym}(V)$.  Since the Koszul complex
$K(A) = A \otimes A^{!\ast}$ is functorial in the quadratic data, and
acyclicity of $K(\mathrm{Sym}(V))$ is classical ($\mathrm{Sym}$--$\bigwedge$ duality),
flatness of the deformation implies acyclicity of $K(Y(\mathfrak{sl}_2))$.

\emph{Step~3: Graded local finiteness.}
The level grading $V = \bigoplus_{r \geq 0} V_r$ with
$\dim V_r = N^2 = 4$ (for $N = 2$) ensures that each graded component
of the quadratic algebra is finite-dimensional, satisfying the local
finiteness hypothesis required for the Polishchuk--Positselski
framework (cf.\ step~2b of the proof of
Theorem~\ref{thm:yangian-koszul-dual}).
\end{proof}

\begin{corollary}[Yangian bar-cobar recovery; \ClaimStatusProvedHere]
\label{cor:yangian-bar-cobar}
The bar-cobar adjunction provides a quasi-isomorphism of
$\Eone$-chiral algebras:
\[
\Omega^{\mathrm{ch}}(\bar{B}^{\mathrm{ch}}(Y(\mathfrak{sl}_2)^{\mathrm{ch}}))
\xrightarrow{\;\sim\;} Y(\mathfrak{sl}_2)^{\mathrm{ch}}.
\]
\end{corollary}

\begin{proof}
The RTT Yangian is quadratic (Definition~\ref{def:yangian-rtt}),
locally finite in the level grading (step~2b of the proof of
Theorem~\ref{thm:yangian-koszul-dual}), and Koszul
(Proposition~\ref{prop:yangian-koszul}).  The mode expansion
$T_{ij}(u) = \delta_{ij} + \hbar \sum_{r \geq 0} T_{ij}^{(r)}
u^{-r-1}$ defines the augmentation $\varepsilon(T_{ij}^{(r)}) = 0$
for all $r \geq 0$, with augmentation ideal $I = \langle
T_{ij}^{(r)} \rangle$.  The PBW basis ensures $\bigcap_n I^n = 0$,
so the algebra is pro-nilpotent.
Theorem~\ref{thm:e1-chiral-koszul-duality} then applies:
the counit~\eqref{eq:e1-counit} of the $\Eone$-chiral bar-cobar
adjunction is a quasi-isomorphism for augmented pro-nilpotent
$\Eone$-chiral algebras.
\end{proof}

\subsection{Yangian modules and Koszul duality}

\begin{definition}[Evaluation module]\label{def:yangian-evaluation}
\index{Yangian!evaluation module|textbf}
\index{evaluation module}
For $a \in \mathbb{C}$ (the \emph{evaluation point}) and a
finite-dimensional $\mathfrak{g}$-module $V$, the \emph{evaluation
module} $V(a)$ of the Yangian $Y(\mathfrak{g})$ is defined by
pulling back along the evaluation homomorphism
\[
\mathrm{ev}_a\colon Y(\mathfrak{g}) \to U(\mathfrak{g}),
\qquad
J^{(r)}_b \mapsto a^r\, J_b.
\]
The $R$-matrix acts on tensor products of evaluation modules by:
\begin{equation}\label{eq:yangian-r-matrix}
R_{V(a), W(b)}
= R_{V,W}\!\left(\frac{a - b}{\hbar}\right)
\in \mathrm{End}(V \otimes W)
\end{equation}
where $R_{V,W}(u) = 1 + r_{V,W}/u + O(u^{-2})$ is the rational
$R$-matrix in the representations~$V, W$.
\end{definition}

\begin{proposition}[Koszul duality on Yangian modules; \ClaimStatusProvedHere]
\label{prop:yangian-module-koszul}
\index{Yangian!module Koszul duality}
The Koszul duality $Y(\mathfrak{g})^! \cong Y_{-\hbar}(\mathfrak{g})$
\textup{(}Theorem~\textup{\ref{thm:yangian-koszul-dual})} extends to
modules as follows:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Evaluation modules:} The Koszul dual of the evaluation
  module $V(a)$ at parameter $\hbar$ is $V(-a)$ at parameter $-\hbar$:
  $\Phi(V(a)_\hbar) = V(-a)_{-\hbar}$.
\item \emph{Tensor products:} For evaluation modules $V(a)$ and
  $W(b)$, the Koszul dual of $V(a) \otimes W(b)$ is
  $V(-a) \otimes W(-b)$ with the $R$-matrix replaced by its inverse:
  \[
  R^!_{V(-a), W(-b)}
  = R_{V,W}^{-1}\!\left(\frac{a - b}{\hbar}\right).
  \]
\item \emph{Braiding reversal:} The braiding on the category of
  finite-dimensional Yangian modules is reversed:
  $\sigma^! = \sigma^{-1}$, parallel to the
  $q \leftrightarrow q^{-1}$ reversal for quantum groups
  \textup{(}Remark~\textup{\ref{rem:kl-koszul})}.
\end{enumerate}
\end{proposition}

\begin{proof}
(i) The evaluation homomorphism $\mathrm{ev}_a$ sends
$T_{ij}^{(r)} \mapsto a^r \rho(e_a)_{ij}$.  Under
$\hbar \to -\hbar$, the RTT relation retains the same form
$R'\, T_1 T_2 = T_2 T_1\, R'$ but with $R$-matrix replaced by
$R^{-1}$.  The evaluation at
parameter $a$ in the original algebra corresponds to evaluation
at $-a$ in the dual algebra, since the spectral parameter
enters the $R$-matrix via $R(u) = 1 - \hbar P/u$, and
replacing $\hbar \to -\hbar$ is equivalent to $u \to -u$, i.e.,
$a \to -a$.

(ii) follows from (i) applied to each tensor factor, combined
with the functoriality of the bar construction on tensor
products (Theorem~\ref{thm:monoidal-module-koszul}).

(iii) The braiding on Yangian modules is given by
$\sigma = P \circ R(u)$, where $P$ is the permutation and
$R(u)$ is the $R$-matrix.  Under $\hbar \to -\hbar$:
$R(u; -\hbar)$ and $R^{-1}(u; \hbar)$ define the same quadratic
dual algebra (Theorem~\ref{thm:yangian-koszul-dual}, Step~4), so
$\sigma^! = P \circ R^{-1}(u) = \sigma^{-1}$.
\end{proof}

\begin{example}[Yangian $Y(\mathfrak{sl}_2)$ fundamental modules]
\label{ex:yangian-sl2-modules}
\index{Yangian!sl2 modules@$\mathfrak{sl}_2$ modules}
The Yangian $Y(\mathfrak{sl}_2)$ has evaluation modules
$V(a) \cong \mathbb{C}^2$ for each $a \in \mathbb{C}$.  The
$R$-matrix on $V(a) \otimes V(b)$ is:
\[
R(u) = \frac{u - \hbar P}{u - \hbar}
= \begin{pmatrix}
1 & 0 & 0 & 0 \\
0 & \frac{u}{u-\hbar} & \frac{-\hbar}{u-\hbar} & 0 \\
0 & \frac{-\hbar}{u-\hbar} & \frac{u}{u-\hbar} & 0 \\
0 & 0 & 0 & 1
\end{pmatrix}
\quad\text{at } u = \tfrac{a-b}{\hbar}.
\]
The tensor product $V(a) \otimes V(b)$ decomposes as
$\mathrm{Sym}^2(V) \oplus \bigwedge^2(V)$ when $a \neq b$
(generic), and is indecomposable when $a = b$ (the $R$-matrix
has a pole).  Under Koszul duality,
$V(a) \otimes V(b) \mapsto V(-a) \otimes V(-b)$ with the
$R$-matrix $R(u) \mapsto R^{-1}(u)$: the $R$-matrix eigenvalues
on $\mathrm{Sym}^2(V)$ and $\bigwedge^2(V)$ are inverted.
\end{example}

\begin{remark}[Verification of the main theorems for Yangians]
\label{rem:yangian-three-theorems}
\index{three theorems!Yangian showcase}
We trace the three main theorems through the Yangian
$Y(\mathfrak{sl}_2)^{\mathrm{ch}}$, highlighting the differences
between the $\Eone$-chiral and $\Einf$-chiral (vertex algebra) settings.

\emph{Theorem~A (Geometric Bar-Cobar Duality).}
The bar construction $\bar{B}(Y(\mathfrak{sl}_2)^{\mathrm{ch}})$
is computed via the RTT presentation
(Theorem~\ref{thm:yangian-bar-rtt}).  The resulting Koszul dual
$Y_{R^{-1}}(\mathfrak{sl}_2)^{\mathrm{ch}}$
(Theorem~\ref{thm:yangian-koszul-dual}) is the RTT algebra with
$R$-matrix inverted, equivalently the parameter flip
$\hbar \to -\hbar$.  This is the $\Eone$ analog of the
Feigin--Frenkel involution $k \to -k - 2h^\vee$ for affine algebras
(Remark~\ref{rem:yangian-langlands}).  The bar-cobar adjunction is an
equivalence by Theorem~\ref{thm:e1-chiral-koszul-duality}, and the
cobar reconstruction recovers the Yangian
(Corollary~\ref{cor:yangian-bar-cobar}).

\emph{Theorem~B (Bar-Cobar Inversion).}
By Proposition~\ref{prop:yangian-koszul}, the RTT Yangian is Koszul,
so the spectral sequence of
Theorem~\ref{thm:bar-cobar-inversion-qi} collapses at $E_2$.
At the classical level ($\hbar = 0$), this reduces to the standard
$\mathrm{Sym} \leftrightarrow \bigwedge$ Koszul resolution
(Corollary~\ref{cor:yangian-classical-self-dual}).

\emph{Theorem~C (Complementarity).}
The deformation-obstruction complementarity
(Theorem~\ref{thm:quantum-complementarity-main}) is formulated for
$\Einf$-chiral algebras via the modular operad and Feynman transform.
The Yangian, being strictly $\Eone$, does not possess a Virasoro
subalgebra, so neither the central charge $c$ nor the obstruction
coefficient $\kappa$ is defined---this explains the ``---'' entries
in the Master Table (Table~\ref{tab:master-invariants}).  The correct
$\Eone$ analog of genus-$g$ complementarity would involve the moduli
of bordered Riemann surfaces and ribbon graph expansions, rather than
$\overline{\mathcal{M}}_g$ and tautological classes.  Developing this
$\Eone$ genus theory remains an important open direction.
\end{remark}


\begin{conjecture}[$\Eone$ genus theory;
\ClaimStatusConjectured]\label{conj:e1-genus-theory}
\index{E1-chiral algebra@$\Eone$-chiral algebra!genus theory}
For an $\Eone$-chiral algebra $\cA$ on $\bR$
(or on an oriented interval), the genus-$g$ obstruction
should be defined using the moduli of bordered Riemann
surfaces $\mathcal{M}_{g,1}^{\partial}$
(surfaces with one parameterized boundary component).
Specifically:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{$\Eone$ bar complex at genus~$g$.}
The genus-$g$ bar complex
$\bar{B}_g^{\Eone}(\cA) :=
\bigoplus_{n \geq 0} \Gamma\bigl(
\overline{C}_n(\Sigma_{g,1}),\,
\cA^{\boxtimes n} \otimes
\Omega^n_{\log}\bigr)$
uses configuration spaces on the genus-$g$
surface $\Sigma_{g,1}$ with one boundary component.
\item \emph{Nilpotency via ribbon graphs.}
The bar differential satisfies $d^2 = 0$ on
$\Sigma_{g,1}$, with the proof using the
Orlik--Solomon relations on the
compactified configuration space
$\overline{C}_n(\Sigma_{g,1})$
(generalizing the genus-$0$ case for
$\Sigma_{0,1} \cong \bR^2$).
\item \emph{Ribbon graph expansion.}
The obstruction $\mathrm{obs}_g^{\Eone}(\cA)$
should be computed by summing over ribbon graphs
$\Gamma$ with genus~$g$ and one boundary cycle:
\[
\mathrm{obs}_g^{\Eone}(\cA) =
\sum_{\Gamma \in \mathrm{Rib}_{g,1}}
\frac{1}{|\mathrm{Aut}(\Gamma)|}
\int_{\overline{C}_\Gamma(\Sigma_{g,1})}
\omega_\Gamma(\cA)
\]
where $\omega_\Gamma(\cA)$ is the form
associated to the OPE data of~$\cA$ on the
edges and vertices of~$\Gamma$.
\item \emph{Complementarity.}
$\mathrm{obs}_g^{\Eone}(\cA) +
\mathrm{obs}_g^{\Eone}(\cA^!) = 0$
in $H^*(\mathcal{M}_{g,1}^{\partial})$,
since the $\Eone$ Koszul duality inverts
the $R$-matrix ($R \to R^{-1}$), which
reverses the orientation of the ribbon
graphs.
\end{enumerate}
For the Yangian $Y(\mathfrak{sl}_2)^{\mathrm{ch}}$,
the ribbon graph expansion at genus~$1$ should recover
$\mathrm{obs}_1 \propto \hbar$ (the deformation parameter),
vanishing at $\hbar = 0$ (the classical limit,
Corollary~\textup{\ref{cor:yangian-classical-self-dual}}).
\end{conjecture}

\begin{remark}[Scope]\label{rem:e1-genus-scope}
This conjecture requires several foundational
developments: (a)~a theory of compactified configuration
spaces on bordered surfaces with the correct boundary
behavior (extending Fulton--MacPherson to the case
with boundary); (b)~an $\Eone$ analog of the Feynman
transform for the modular operad; (c)~an $\Eone$ analog
of the Faber--Pandharipande formula for tautological
integrals on $\mathcal{M}_{g,1}^{\partial}$.  The
closest existing work is the Swiss-cheese operad
(Voronov~\cite{Voronov99}), which governs open-closed
homotopy algebras.
The toroidal algebra
(Remark~\ref{rem:toroidal-three-theorems}) faces the
same obstruction.
\end{remark}


\section{Shifted Yangians}
\label{sec:shifted-yangians}

\subsection{Definition}

\begin{definition}[Shifted Yangian]\label{def:shifted-yangian}
For a coweight $\mu \in X_*(\mathfrak{h})$, the \emph{shifted Yangian} $Y_\mu(\mathfrak{g})$
(Brundan--Kleshchev, Finkelberg--Tsymbaliuk, KWWY) is generated by elements
$E_i^{(r)}, F_i^{(r)}, H_i^{(r)}$ for $i = 1, \ldots, \mathrm{rank}(\mathfrak{g})$
and $r \geq 0$, satisfying the same Yangian relations as $Y(\mathfrak{g})$ except that the
generating function $H_i(u) = \sum_{r \geq 0} H_i^{(r)} u^{-r}$ is constrained to be a rational
function whose numerator and denominator degrees are prescribed by $\mu$:
\[
H_i(u) = \frac{\prod_{j=1}^{\mu_i^+}(u - a_j^{(i)})}{\prod_{j=1}^{\mu_i^-}(u - b_j^{(i)})}
\]
where $\mu_i^\pm = \langle \alpha_i, \mu^\pm \rangle$ and $\mu = \mu^+ - \mu^-$ decomposes the coweight into dominant parts. The parameters $a_j^{(i)}, b_j^{(i)}$ are complex numbers.
\end{definition}

\begin{conjecture}[Shifted Yangian as $\Eone$-chiral; \ClaimStatusConjectured]\label{conj:shifted-yangian-e1}
The shifted Yangian admits a chiral algebra structure $Y_\mu(\mathfrak{g})^{\mathrm{ch}}$
that is $\Eone$-chiral.  The shift $\mu$ modifies the central terms in the OPE,
and the rational constraint on $H_i(u)$ introduces additional poles in the vertex
operators.
\end{conjecture}

\begin{remark}[Scope]
A proof would require: (1)~an explicit vertex operator construction for
the shifted Yangian generators $E_i^{(r)}, F_i^{(r)}, H_i^{(r)}$ with
the rational constraint on $H_i(u)$; (2)~verification of the Borcherds
identity (weak associativity of the OPE) in the presence of the additional
poles introduced by the shift.  For the unshifted case ($\mu = 0$), the
$\Eone$-chiral structure is established in Theorem~\ref{thm:yangian-e1}.
The shifted case is expected to follow by an analogous argument using the
BFN construction (Theorem~\ref{thm:bfn}), which realizes the shifted
Yangian as equivariant homology and thus provides a natural source of
vertex operators.  Dimofte--Niu--Py \cite{DNP25} construct dg-shifted Yangians
from line operators in 3d holomorphic QFT, providing further evidence for the
$\Eone$-chiral structure in the shifted setting.
\end{remark}


\section{Coulomb branch algebras}
\label{sec:coulomb-branch}

\subsection{Definition from gauge theory}

\begin{definition}[Coulomb branch algebra]\label{def:coulomb-branch}
\index{Coulomb branch}
For a 3d $\mathcal{N} = 4$ gauge theory with gauge group $G$ and matter representation $N$, 
the \emph{Coulomb branch algebra} $\mathcal{A}(G, N)$ is the quantization of the 
Coulomb branch $\mathcal{M}_C(G, N)$.

For type A quiver gauge theories, $\mathcal{A}(G, N)$ is a shifted Yangian or 
truncated shifted Yangian.
\end{definition}

\begin{theorem}[BFN construction; \ClaimStatusProvedElsewhere]\label{thm:bfn}
(Braverman--Finkelberg--Nakajima \cite{BFN18}) The Coulomb branch algebra is:
\[
\mathcal{A}(G, N) = H^{G_{\mathcal{O}} \times \mathbb{C}^\times}_*(\mathcal{R}(G, N))
\]
the equivariant Borel--Moore homology of the space of triples $\mathcal{R}(G, N)$.
\end{theorem}

\begin{example}[Quiver of type $A_n$]\label{ex:coulomb-an}
For the $A_n$ quiver with dimension vector $(1, 2, \ldots, n)$:
\[
\mathcal{A} = Y_{(n-1, n-2, \ldots, 1, 0)}(\mathfrak{sl}_n)
\]
a shifted Yangian.
\end{example}


\section{Cohomological Hall algebras}
\label{sec:coha}

\subsection{Definition}

\begin{definition}[Cohomological Hall algebra]\label{def:coha}
For a quiver $Q$ with dimension vector $\mathbf{d}$, the \emph{cohomological Hall algebra} (CoHA) is:
\[
\mathcal{H}_Q = \bigoplus_{\mathbf{d}} H^{G_{\mathbf{d}}}_{\mathrm{BM},*}(\mathcal{M}_{\mathbf{d}}^{\mathrm{nil}})
\]
where $\mathcal{M}_{\mathbf{d}}^{\mathrm{nil}}$ is the moduli of nilpotent representations.

The product is defined via the Hecke correspondence:
\[
\mathcal{M}_{\mathbf{d}_1} \times \mathcal{M}_{\mathbf{d}_2} \xleftarrow{p} \mathcal{E} \xrightarrow{q} \mathcal{M}_{\mathbf{d}_1 + \mathbf{d}_2}
\]
as $a \star b = q_* p^*(a \boxtimes b)$.
\end{definition}

\begin{conjecture}[CoHA as $\Eone$-chiral; \ClaimStatusConjectured]\label{conj:coha-e1}
The CoHA $\mathcal{H}_Q$ carries an $\Eone$-chiral algebra structure.  The evidence:
\begin{enumerate}[label=(\roman*)]
\item The Hecke product is associative but not commutative.
\item The factorization structure comes from stratifying by dimension vectors.
\item The chiral enhancement should use configuration spaces of points on curves
parameterizing deformation directions.
\end{enumerate}
A rigorous construction requires specifying the vertex operators and verifying
weak associativity (the Borcherds identity).
\end{conjecture}

\begin{remark}[Scope]
The key obstacle is translating the Hecke product
$a \star b = q_* p^*(a \boxtimes b)$ into a vertex operator OPE.
The Hecke correspondence provides an associative product on
$\bigoplus_{\mathbf{d}} H^{G_{\mathbf{d}}}_{\mathrm{BM},*}(\mathcal{M}_{\mathbf{d}}^{\mathrm{nil}})$,
but this product is defined via correspondences (push-pull in
Borel--Moore homology), not via residues of meromorphic functions.
The expected translation uses the Schiffmann--Vasserot construction
\cite{SV13}, which realizes the CoHA action on the equivariant
cohomology of Hilbert schemes as a Heisenberg/W-algebra action.
Extending this from an \emph{action} to a full vertex algebra
\emph{structure on the CoHA itself} remains open.
\end{remark}

\begin{example}[Jordan quiver CoHA]\label{ex:jordan-coha}
For the Jordan quiver (one vertex, one loop), the CoHA is:
\[
\mathcal{H}_{\text{Jordan}} = \bigoplus_{n \geq 0} H^{GL_n}_{\mathrm{BM},*}(\mathcal{N}_n) \;\curvearrowright\; \bigoplus_{n \geq 0} H^*_T(\mathrm{Hilb}^n(\bC^2))
\]
The CoHA acts on the equivariant cohomology of the Hilbert scheme via Nakajima-type operators (Schiffmann--Vasserot), realizing a Heisenberg/W-algebra structure on the target.  The CoHA itself carries a \emph{different} product structure (the shuffle product) from the OPE product on the chiral algebra side.
\end{example}
