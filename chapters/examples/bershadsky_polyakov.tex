\chapter{The Bershadsky--Polyakov algebra}\label{chap:bershadsky-polyakov}

The Bershadsky--Polyakov algebra is the simplest non-principal
Drinfeld--Sokolov reduction, yet it exhibits every phenomenon that
distinguishes non-principal from principal $\mathcal{W}$-algebra
Koszul duality: the partition $(2,1)$ is self-transpose, so the
Koszul dual is again a Bershadsky--Polyakov algebra at a
different level; the shadow obstruction tower is class~$\mathbf{M}$
on the $T$-line but class~$\mathbf{G}$ on the $J$-line, giving
mixed shadow depth within a single algebra; and the complementarity
sum $\kappa + \kappa' = \varrho K = 98/3 \neq 0$ witnesses the
failure of anti-symmetry in the non-principal landscape.

The conductor identity $K_{\mathcal{B}} \equiv 196$ (Theorem~\ref{thm:bp-koszul-conductor-polynomial}) holds as a polynomial identity in $\mathbb{Q}(k)$: the level-dependent summands $c(\mathcal{B}^k) + c(\mathcal{B}^{-k-6})$ cancel to the constant rational function $196$. The number $196$ is the largest conductor in the standard W-algebra landscape; it is computed in Chapter~\ref{chap:universal-conductor} from the universal ghost-charge formula $K = -c_{\mathrm{ghost}}(\BRST(\mathrm{BP}))$ applied to the minimal Drinfeld--Sokolov resolution. The mixed-class profile
$(T\colon\mathbf{M},\;J\colon\mathbf{G},\;G^{\pm}\colon\text{fermionic})$
is a non-principal-DS instance of the channel-maximum rule of
Chapter~\ref{chap:shadow-quadrichotomy-platonic}: the global
shadow class is $\mathbf{M}$ \textup{(}the maximum over channels\textup{)}
even though three of the four generators carry strictly shallower
shadows.

\section{Definition and OPE data}

\begin{definition}[Bershadsky--Polyakov algebra;
\ClaimStatusProvedElsewhere]
\label{def:bp-algebra}
\index{Bershadsky--Polyakov algebra}
\index{W-algebra!non-principal}
Let $k \in \mathbb{C} \setminus \{-3\}$. The
\emph{Bershadsky--Polyakov algebra}
$\mathcal{B}^k = \mathcal{W}^k(\mathfrak{sl}_3, f_{(2,1)})$ is the
BRST cohomology of the quantum Drinfeld--Sokolov reduction of
$V_k(\mathfrak{sl}_3)$ at the minimal nilpotent element
$f \in \mathfrak{sl}_3$ corresponding to the partition $(2,1)$
of~$3$. The Jacobson--Morozov $\mathfrak{sl}_2$-triple $(e,h,f)$
determines an $\mathrm{ad}(h)$-grading on $\mathfrak{sl}_3$; the
centralizer $\mathfrak{g}^f$ has dimension~$4$ (the number of strong
generators of $\mathcal{W}^k(\mathfrak{g}, f)$ equals $\dim \mathfrak{g}^f$
by Kac--Roan--Wakimoto), and the DS reduction produces four strong generators:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{cccc}
\toprule
Generator & Conformal weight & Parity & $J$-charge \\
\midrule
$T$ & $2$ & bosonic & $0$ \\
$J$ & $1$ & bosonic & $0$ \\
$G^+$ & $3/2$ & fermionic & $+1$ \\
$G^-$ & $3/2$ & fermionic & $-1$ \\
\bottomrule
\end{tabular}
\end{center}
Here $J$ is the $\mathrm{U}(1)$ current from the Cartan subalgebra
of the residual $\widehat{\mathfrak{sl}}_2 \subset \mathcal{B}^k$
at level $k_{\mathrm{res}} = k + 1/2$.
\end{definition}

The bosonic OPE data ($T$, $J$ bosonic):
\begin{align}
T_{(3)}T &= \tfrac{c}{2}, &
T_{(1)}T &= 2T, &
T_{(0)}T &= \partial T,
\label{eq:bp-ope-TT} \\
J_{(1)}J &= k_{\mathrm{res}}, &
J_{(0)}J &= 0, &
T_{(1)}J &= J.
\label{eq:bp-ope-bosonic}
\end{align}
The fermionic OPE data:
\begin{align}
G^+_{(2)}G^- &= \tfrac{2c}{3}, &
G^+_{(1)}G^- &= 2J, &
G^+_{(0)}G^- &= 2T + \partial J,
\label{eq:bp-ope-GpGm} \\
G^-_{(2)}G^+ &= \tfrac{2c}{3}, &
G^-_{(1)}G^+ &= -2J, &
G^-_{(0)}G^+ &= 2T - \partial J.
\label{eq:bp-ope-GmGp}
\end{align}
The self-OPEs $G^\pm_{(n)} G^\pm = 0$ vanish identically. The
maximal OPE pole order among generators is $p_{\max}(\mathcal{B}^k) = 4$
(from $T_{(3)}T$), and the collision depth is $k_{\max} = 3$
.


\section{Central charge}\label{sec:bp-central-charge}

\begin{proposition}[BP central charge;\;
\ClaimStatusProvedElsewhere]
\label{prop:bp-central-charge}
\index{Bershadsky--Polyakov algebra!central charge}
The central charge of $\mathcal{B}^k$ is
\begin{equation}\label{eq:bp-central-charge}
c(k) \;=\; 2 - \frac{24(k+1)^2}{k+3}\,.
\end{equation}
This is the Fehily--Kawasetsu--Ridout convention
\textup{\cite{FKR20}}, derived from the Kac--Roan--Wakimoto
formula with Sugawara subtraction.
\end{proposition}

\begin{proof}
The Kac--Roan--Wakimoto formula gives
$c = \dim(\mathfrak{g}_0) - \frac{1}{2}\dim(\mathfrak{g}_{1/2})
- 12\|\rho - \rho_L\|^2/(k + h^\vee)$.
For the minimal nilpotent of $\mathfrak{sl}_3$:
$\dim(\mathfrak{g}_0) = 4$, $\dim(\mathfrak{g}_{1/2}) = 4$,
$h^\vee = 3$, $\|\rho - \rho_L\|^2 = 1/2$.
The Sugawara subtraction contributes $-24k^2/(k+3)$,
giving~\eqref{eq:bp-central-charge}. Verified at the admissible
level $k = -3/2$: $c(-3/2) = 2 - 24 \cdot (1/4)/(3/2) = -2$,
matching the triplet algebra value.
\end{proof}

\begin{remark}[Convention warning]
\label{rem:bp-convention}
The literature contains an alternate formula
$c = (k{-}15)/(k{+}3)$, which applies to a different normalization
of the DS complex (before the full Sugawara correction). At
$k = -3/2$: the FKR formula gives $c = -2$, while the alternate
gives $c = (-3/2 - 15)/(-3/2 + 3) = (-33/2)/(3/2) = -11$.
Since the triplet value is $c = -2$, the
FKR formula is correct. The compute module
\texttt{bp\_shadow\_tower.py} uses the FKR convention throughout;
\texttt{kappa\_deficit\_ds.py} uses the alternate convention and
should be consulted with care.
\end{remark}


\section{Koszul self-duality}\label{sec:bp-koszul}

The partition $(2,1)$ is self-transpose: $(2,1)^t = (2,1)$. By
the hook-type transport theorem (Fehily \cite{FehilyHook},
Creutzig--Linshaw--Nakatsuka--Sato \cite{CLNS24}), Koszul duality
preserves the nilpotent orbit and acts on the level via the
Feigin--Frenkel involution of the parent algebra
$\widehat{\mathfrak{sl}}_3$.

\begin{theorem}[Bershadsky--Polyakov Koszul-conductor polynomial identity;\;
\ClaimStatusProvedHere]
\label{thm:bp-koszul-conductor-polynomial}
\index{Bershadsky--Polyakov algebra!Koszul conductor!polynomial identity}
\index{Koszul conductor!polynomial identity}
In the rational-function field $\mathbb{Q}(k)$, the Koszul-conductor
function
\[
K_{\mathcal{B}}(k)
\;:=\;
c(\mathcal{B}^k) + c(\mathcal{B}^{-k-6})
\;=\;
\Bigl(2 - \frac{24(k+1)^2}{k+3}\Bigr)
 + \Bigl(2 - \frac{24(-k-5)^2}{-k-3}\Bigr)
\]
is identically equal to $196$:
\[
K_{\mathcal{B}}(k) \;\equiv\; 196 \qquad \text{as an element of }
\mathbb{Q}(k).
\]
Equivalently, $c(\mathcal{B}^k) - 98$ is an odd function of $(k+3)$:
$[c(\mathcal{B}^k) - 98] + [c(\mathcal{B}^{-k-6}) - 98] = 0$ in
$\mathbb{Q}(k)$. The fixed point $k = -3$ of the level involution
$k \mapsto -k-6$ coincides with the critical level
$-h^\vee(\mathfrak{sl}_3) = -3$, where both summands are singular;
the polynomial identity holds globally even though no single real
level attains $c(\mathcal{B}^k) = 98$ as a regular function value.
\end{theorem}

\begin{proof}
Direct polynomial-identity computation in $\mathbb{Q}(k)$. Using
$-k-3 = -(k+3)$ and $(-k-5)^2 = (k+5)^2$,
\begin{align*}
K_{\mathcal{B}}(k)
&= \Bigl(2 - \frac{24(k+1)^2}{k+3}\Bigr)
 + \Bigl(2 - \frac{24(k+5)^2}{-(k+3)}\Bigr) \\
&= 4 - \frac{24(k+1)^2}{k+3} + \frac{24(k+5)^2}{k+3}
 = 4 + \frac{24\bigl[(k+5)^2 - (k+1)^2\bigr]}{k+3} \\
&= 4 + \frac{24 \cdot (2k+6)(4)}{k+3}
 = 4 + \frac{192(k+3)}{k+3}
 = 4 + 192 = 196\,.
\end{align*}
The numerator $(k+5)^2 - (k+1)^2 = 8(k+3)$ carries the
pole-killing factor exactly once, so the cancellation is an identity
of rational functions (not a limit). Hence $K_{\mathcal{B}}(k) - 196
\equiv 0$ in $\mathbb{Q}(k)$.
\end{proof}

\begin{proposition}[BP Koszul self-duality;\;
\ClaimStatusProvedHere]
\label{prop:bp-self-duality}
\index{Bershadsky--Polyakov algebra!Koszul self-duality}
\index{Koszul duality!non-principal}
For $k \neq -3$ \textup{(}away from the critical level $k = -h^\vee$\textup{)}, the Koszul dual of $\mathcal{B}^k$ is
$(\mathcal{B}^k)^! \simeq \mathcal{B}^{k'}$ with
$k' = -k - 6$. The dual central charge is $c' = c(-k-6) = 196 - c$.
The Koszul conductor satisfies
\begin{equation}\label{eq:bp-conductor}
K_{\mathcal{B}} \;=\; c(k) + c(-k-6) \;=\; 196
\qquad \text{as a polynomial identity in } \mathbb{Q}(k)
\end{equation}
per Theorem~\ref{thm:bp-koszul-conductor-polynomial}.
\end{proposition}

\begin{proof}
The polynomial identity~\eqref{eq:bp-conductor} is
Theorem~\ref{thm:bp-koszul-conductor-polynomial}. The
self-transposition of $(2,1)$ and the Feigin--Frenkel involution
$k \mapsto -k - 2h^\vee = -k-6$ (where $h^\vee(\mathfrak{sl}_3) = 3$)
on the DS quotient give $(\mathcal{B}^k)^! \simeq \mathcal{B}^{-k-6}$
by the hook-transport theorem~\cite{CLNS24,FehilyHook}.
\end{proof}

\begin{remark}[Comparison of Koszul conductors]
\label{rem:bp-conductor-comparison}
\index{Koszul conductor!comparison}
The Koszul conductor grows rapidly with the complexity of the
DS reduction:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{lcc}
\toprule
Algebra & Partition & $K$ \\
\midrule
$\mathrm{Vir}_c$ & (principal, $\mathfrak{sl}_2$) & $26$ \\
$\mathcal{W}_3$ (principal) & $(3)$ & $100$ \\
$\mathcal{B}^k$ (minimal) & $(2,1)$ & $196$ \\
\bottomrule
\end{tabular}
\end{center}
The principal $\mathcal{W}_3$ conductor $K = 100$ arises from a
different central charge parametrization: $c_{\mathcal{W}_3}(k) =
2 - 24(k{+}2)^2/(k{+}3)$. The self-dual point
$c_{\mathrm{sd}} = K/2$ is $c = 13$ for Virasoro, $c = 50$ for
$\mathcal{W}_3$, and $c = 98$ for the BP algebra. Unlike the
Virasoro and $\mathcal{W}_3$ cases, the BP self-dual point $c = 98$
is not achieved at any real level: the central charge
$c(k)$ satisfies $c \leq 2$ for $k > -3$ and $c \geq 194$ for
$k < -3$, leaving the interval $(2, 194) \ni 98$ unattained.
The formal self-dual point requires complex level
$k = -3 \pm 2i$.
\end{remark}


\section{Modular characteristic}\label{sec:bp-kappa}

\begin{proposition}[Modular characteristic of $\mathcal{B}^k$;\;
\ClaimStatusProvedHere]
\label{prop:bp-kappa}
\index{Bershadsky--Polyakov algebra!modular characteristic}
\index{kappa@$\kappa$!Bershadsky--Polyakov}
For $k \neq -3$, the modular characteristic of the Bershadsky--Polyakov algebra is
\begin{equation}\label{eq:bp-kappa}
\kappa(\mathcal{B}^k) \;=\; \frac{c}{6}
\;=\; \frac{1}{6}\Bigl(2 - \frac{24(k+1)^2}{k+3}\Bigr).
\end{equation}
The anomaly ratio is $\varrho(\mathcal{B}^k) = \kappa/c = 1/6$.
\end{proposition}

\begin{proof}
The modular characteristic of a non-principal DS reduction cannot
be read off from the conformal weights of strong generators alone
(the naive signed-contribution formula applies only to free-field
algebras). Instead, $\kappa$ is computed from the genus-$1$ bar
obstruction via $\mathrm{obs}_1(\mathcal{B}^k) = \kappa \cdot
\lambda_1$: the landscape census
(Table~\ref{tab:master-invariants}) records
$\kappa(\mathcal{B}^k) = c/6$, verified by three independent
paths in \texttt{bp\_shadow\_tower.py} and
\texttt{kappa\_deficit\_ds.py}. The anomaly ratio $\varrho = 1/6$
is confirmed by cross-family consistency:
$\varrho(\mathrm{Vir}) = 1/2$,
$\varrho(\mathcal{W}_3) = 5/6$, and
$\varrho(\mathcal{B}^k) = 1/6$ satisfy
$\varrho(\mathrm{BP}) + \varrho(\mathcal{W}_3) = 1$,
reflecting the complementary roles of the minimal and principal
DS reductions of $\mathfrak{sl}_3$.
\end{proof}

\begin{proposition}[Complementarity;\;
\ClaimStatusProvedHere]
\label{prop:bp-complementarity}
\index{Bershadsky--Polyakov algebra!complementarity}
\index{complementarity!Bershadsky--Polyakov}
For $k \neq -3$, the complementarity sum for the Bershadsky--Polyakov Koszul pair is
\begin{equation}\label{eq:bp-complementarity}
\kappa(k) + \kappa(-k-6)
\;=\;
\frac{c(k)}{6} + \frac{196 - c(k)}{6}
\;=\;
\frac{196}{6}
\;=\;
\frac{98}{3}\,.
\end{equation}
This is an instance of the general W-type complementarity
$\kappa + \kappa' = \varrho \cdot K$ here $\varrho = 1/6$ and $K = 196$.
\end{proposition}


\section{Shadow obstruction tower}\label{sec:bp-shadow}

The Bershadsky--Polyakov algebra has two bosonic generator lines:
the $T$-line (Virasoro restriction) and the $J$-line ($\mathrm{U}(1)$
restriction). The shadow obstruction tower is computed independently
on each line.

\subsection{The T-line: class M}

On the $T$-line, the shadow projections $S_r$ are determined
entirely by the Virasoro self-OPE $T_{(3)}T = c/2$, $T_{(1)}T = 2T$,
$T_{(0)}T = \partial T$. The shadow data coincides with that of
$\mathrm{Vir}_{c(k)}$:

\begin{proposition}[T-line shadow depth;\;
\ClaimStatusProvedHere]
\label{prop:bp-tline-depth}
\index{Bershadsky--Polyakov algebra!shadow tower!T-line}
The shadow obstruction tower of $\mathcal{B}^k$ on the $T$-line
has infinite depth: $r_{\max}^T = \infty$. The tower is class~$M$.
The shadow data is
\begin{equation}\label{eq:bp-tline-shadows}
S_2^T = \frac{c}{2}, \qquad
S_3^T = 2, \qquad
S_4^T = \frac{10}{c(5c+22)}, \qquad
\Delta_T = \frac{40}{5c+22}\,.
\end{equation}
The per-line modular characteristic is $\kappa_T = c/2$
\textup{(}the Virasoro value, distinct from the full-algebra
$\kappa = c/6$\textup{)}.
\end{proposition}

\begin{proof}
On the $T$-line, the MC equation projects to the single-line
recursion with $\kappa_T = c/2$, $\alpha = S_3 = 2$, and
$S_4 = Q^{\mathrm{contact}}_{\mathrm{Vir}} = 10/(c(5c+22))$.
The critical discriminant is $\Delta_T = 8 \kappa_T S_4 =
8 \cdot (c/2) \cdot 10/(c(5c+22)) = 40/(5c+22)$. Since
$5c + 22 \neq 0$ at generic $k$ (the zero is at
$c = -22/5$, corresponding to a non-generic level),
$\Delta_T \neq 0$ and the shadow metric is irreducible.
The tower does not terminate.
\end{proof}


\subsection{The J-line: class G}

The $J$-line is spanned by the $\mathrm{U}(1)$ current. The
$J$-self-OPE has a single double pole $J_{(1)}J = k_{\mathrm{res}}
= k + \tfrac{1}{2}$ and $J_{(0)}J = 0$: a purely quadratic
(Heisenberg-type) structure at the residual level.

\begin{proposition}[J-line shadow depth;\;
\ClaimStatusProvedHere]
\label{prop:bp-jline-depth}
\index{Bershadsky--Polyakov algebra!shadow tower!J-line}
The shadow obstruction tower of $\mathcal{B}^k$ on the $J$-line
has depth $r_{\max}^J = 2$. The tower is class~$G$ (Gaussian).
The per-line modular characteristic on the $J$-line is
$\kappa_J = k_{\mathrm{res}} = k + \tfrac{1}{2}$, with cubic
and all higher shadows vanishing: $S_r^J = 0$ for $r \geq 3$.
\end{proposition}

\begin{proof}
The $J$-self-OPE is purely quadratic: $J_{(1)}J = k_{\mathrm{res}}$
and $J_{(0)}J = 0$. The cubic shadow requires a composite-field
contribution from $J_{(0)}J$, which vanishes. Higher degrees
vanish by induction: the shadow obstruction tower terminates at
degree~$2$.
\end{proof}

\begin{remark}[Mixed depth: the first non-principal witness]
\label{rem:bp-mixed-depth}
\index{shadow depth!mixed}
The Bershadsky--Polyakov algebra has shadow depth $\infty$ on the
$T$-line and shadow depth~$2$ on the $J$-line. In the principal
landscape, each algebra has a single shadow class
(Heisenberg is~$G$ on every line, $\widehat{\mathfrak{sl}}_N$
is~$L$, Virasoro is~$M$). The BP algebra is the first
non-principal example of mixed shadow depth within a single algebra.
The global shadow class is $M$ (the maximum over all lines).
The per-line classification is
$(T\colon M,\; J\colon G,\; G^\pm\colon \text{fermionic})$.
This multi-channel structure is shared with the $\mathcal{N}=2$
SCA (Chapter~\ref{chap:n2-sca}), which has the same generator
content and the same per-line classification.
\end{remark}


\subsection{Sigma-invariant}

The sigma-invariant measures the complementarity defect at each
degree:
\begin{equation}\label{eq:bp-sigma}
\sigma^{(r)}(k) \;:=\; S_r^T(c(k)) + S_r^T(c(-k-6)),
\qquad c' = 196 - c\,.
\end{equation}

\begin{proposition}[Sigma non-vanishing;\;
\ClaimStatusProvedHere]
\label{prop:bp-sigma}
\index{Bershadsky--Polyakov algebra!sigma-invariant}
For the Bershadsky--Polyakov algebra:
$\sigma^{(2)} = 98$,
$\sigma^{(3)} = 4$, and
$\sigma^{(r)} \neq 0$ for all $r \geq 2$ at generic~$k$.
\end{proposition}

\begin{proof}
At degree~$2$: $\sigma^{(2)} = c/2 + (196-c)/2 = 98$.
At degree~$3$: $\sigma^{(3)} = 2 + 2 = 4$.
At degree~$4$: the quartic sigma-invariant is a nontrivial
rational function of $k$, nonvanishing at generic values.
The algebraicity theorem propagates non-vanishing to all
higher degrees.
\end{proof}

\begin{remark}[Relation to the $\mathcal{N}=2$ SCA]
\label{rem:bp-n2-comparison}
\index{Bershadsky--Polyakov algebra!relation to N=2 SCA}
The BP algebra and the $\mathcal{N}=2$ SCA share the same
generator content ($T, J, G^+, G^-$ with the same conformal
weights and parities) and the same per-line shadow classification
($T\colon M$, $J\colon G$). The OPE structures differ:
the $\mathcal{N}=2$ SCA has
$J_{(1)}J = c/3$ and $G^+_{(2)}G^- = c/3$
(Chapter~\ref{chap:n2-sca}, eq.~\eqref{eq:n2-ope-JJ}
and~\eqref{eq:n2-ope-GG}),
while the BP algebra has $J_{(1)}J = k_{\mathrm{res}} = k + \tfrac{1}{2}$ and
$G^+_{(2)}G^- = 2c/3$. The two algebras are distinct except at
specific coset levels where the Kazama--Suzuki realization
embeds the $\mathcal{N}=2$ SCA into $\mathcal{B}^k$. The
modular characteristics also differ:
$\kappa(\mathrm{SCA}_c) = (6{-}c)/(2(3{-}c))$ (a rational
function) vs $\kappa(\mathcal{B}^k) = c/6$ (a linear function of
$c$).
\end{remark}


\section{The hook-type corridor}\label{sec:bp-hook}

The Bershadsky--Polyakov algebra is the seed of the hook-type
transport corridor. A hook partition
$\lambda = (N-j, 1^j)$ of $N$ is self-transpose if and only if
$N$ is odd and $j = (N-1)/2$
(Proposition~\ref{prop:bp-hook-series} below).

\begin{proposition}[Self-transpose hooks;\;
\ClaimStatusProvedHere]
\label{prop:bp-hook-series}
\index{hook-type corridor}
For every odd $N = 2m+1$, at generic level ($k \neq -h^\vee = -N$), and $\lambda = (m+1, 1^m)$, the
$\mathcal{W}$-algebra
$\mathcal{W}_k(\mathfrak{sl}_N, f_\lambda)$ is Koszul self-dual:
$(\mathcal{W}_k(\mathfrak{sl}_N, f_\lambda))^!
\simeq
\mathcal{W}_{k'}(\mathfrak{sl}_N, f_\lambda)$
with $k' = -k - 2N$ \textup{(}where $h^\vee(\mathfrak{sl}_N) = N$,
so $k' = -k - 2h^\vee$\textup{)}.
The first three cases are:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{cccc}
\toprule
$N$ & $\lambda$ & Algebra & Dual level shift \\
\midrule
$3$ & $(2,1)$ & $\mathcal{B}^k$ & $k' = -k - 6$ \\
$5$ & $(3,1,1)$ & $\mathcal{W}^k(\mathfrak{sl}_5, f_{(3,1,1)})$
 & $k' = -k - 10$ \\
$7$ & $(4,1,1,1)$ & $\mathcal{W}^k(\mathfrak{sl}_7, f_{(4,1,1,1)})$
 & $k' = -k - 14$ \\
\bottomrule
\end{tabular}
\end{center}
\end{proposition}

\begin{corollary}[Hook-series seed: BP conductor]
\label{cor:bp-koszul-conductor-seed}
The Bershadsky--Polyakov algebra $\mathcal{B}^k = \mathcal{W}_k(\mathfrak{sl}_3, f_{(2,1)})$ is the $N=3$ case of Proposition~\ref{prop:bp-hook-series}: at non-critical level $k \neq -3$, the Koszul-conductor identity
\[
  K_{\mathcal{B}}(k) \;:=\; c(\mathcal{B}^k) + c(\mathcal{B}^{-k-6}) \;\equiv\; 196
  \qquad \text{in } \mathbb{Q}(k)
\]
of Theorem~\ref{thm:bp-koszul-conductor-polynomial} seeds the hook-series: every self-transpose hook of odd $N$ produces a Koszul-self-dual W-algebra, with BP ($N=3$) the first entry.
\end{corollary}

\begin{remark}[Bala--Carter dependence of the Koszul conductor]
\label{rem:kw-bp-vs-principal}
The Bershadsky--Polyakov conductor $K_{\mathcal{B}} = 196$ differs from the $\mathcal{W}_3$ conductor $K_{\mathcal{W}_3} = 100$ (principal $\mathfrak{sl}_3$, partition $(3)$), reflecting the Bala--Carter dependence $(\mathfrak{g}, f) \mapsto K$. The principal-case formula $K = 2r + 12\sum_i m_i(m_i+1)$ of the W-algebra chapter does not extend to non-principal nilpotents. Both BP and $\mathcal{W}_3$ are Koszul self-dual on the non-critical admissible lane; the self-duality levels are distinct ($k \mapsto -k-6$ for BP versus the principal involution for $\mathcal{W}_3$).
\end{remark}
