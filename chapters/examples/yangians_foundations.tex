\chapter{Yangians and shifted Yangians}
\label{chap:yangians}
\label{rem:yangian-status}

\index{Yangian!as modular Koszul atom}

The Kac--Moody and $\mathcal{W}$-algebra families are
$\Einf$-chiral: commutative, fully local, with unordered
configuration spaces $C_n(X)$.  The Kazhdan--Lusztig equivalence
\cite{KL93}
sends $\mathcal{O}_k^{\mathrm{int}}(\widehat{\mathfrak{g}})$ to the
braided monoidal category $\mathcal{C}(U_q(\mathfrak{g}))$, whose
$R$-matrix structure is not commutative.  The bar-cobar machine must
therefore extend to \emph{ordered} configurations --- and the Yangian
$Y(\mathfrak{g})$ provides the first $\Eone$-chiral algebra for which
this extension is explicit.

On ordered configurations, the bar complex $\barB^{\Eone}(\cA)$
is no longer symmetric, and Verdier duality no longer acts by level
inversion on the algebra.  Instead, it inverts the $R$-matrix on
the module category: $\mathbb{D} R(u) = R^{-1}(u)$.

\begin{remark}[Three levels of Yangian structure]
\label{rem:yangian-three-levels}
\index{Yangian!three levels of structure}
The word ``Yangian'' carries three distinct meanings in the
literature, corresponding to three levels of structure that this
chapter carefully separates:
\begin{enumerate}[label=\textup{(\Alph*)}]
\item \emph{Algebraic Yangian $Y(\fg)$.}
  The associative algebra in the RTT presentation
  (Definition~\ref{def:yangian-rtt}) or the Drinfeld presentation
  (Definition~\ref{def:yangian}).  The spectral parameter~$u$ is a
  formal variable.  The defining relation is
  quadratic-linear (filtered quadratic via RTT);
  the associated graded
  $\operatorname{gr} Y(\fg) \cong U(\fg[z])$ is
  quadratic~\cite{Drinfeld85,molev-yangians}.
\item \emph{$\Eone$-chiral Yangian on a curve~$X$.}
  The algebraic Yangian equipped with $\Eone$-factorization structure
  (Theorem~\ref{thm:yangian-e1}): vertex operators on~$X$ with
  the spectral parameter identified as the coordinate difference
  $u = z_1 - z_2$ on the curve, and the $R$-matrix providing the
  braided locality constraint~\eqref{eq:r-twisted-locality}.
\item \emph{QFT realization.}
  In 4d Chern--Simons theory (Costello--Witten--Yamazaki~\cite{CWY18}),
  boundary operators form a chiral algebra and line operators form
  a monoidal category controlled by the Yangian.  The spectral
  parameter is the holomorphic coordinate of the 4d theory,
  restricted to the 2d boundary.
\end{enumerate}
Level~(A) is purely algebraic and level-free.
Level~(B), developed in this chapter, geometrizes the spectral
parameter and produces the $\Eone$-bar-cobar duality and the DK
ladder.  Level~(C) is a physical realization of~(B) and is
discussed in Chapter~\ref{ch:bv-brst}.
\end{remark}

\section{Yangian vertex algebra}
\label{sec:yangian-vertex}

\subsection{Yangian definition}

\begin{definition}[Yangian]\label{def:yangian}
\index{Yangian|textbf}
The \emph{Yangian} $Y(\mathfrak{g})$ associated to a simple Lie algebra $\mathfrak{g}$ 
is the associative algebra generated by $J^{(r)}_a$ for $a = 1, \ldots, \dim\mathfrak{g}$ 
and $r \geq 0$, with relations:
\begin{align*}
[J^{(0)}_a, J^{(0)}_b] &= f_{ab}^c J^{(0)}_c \\
[J^{(0)}_a, J^{(r)}_b] &= f_{ab}^c J^{(r)}_c \\
[J^{(1)}_a, [J^{(1)}_b, J^{(0)}_c]] - [J^{(0)}_a, [J^{(1)}_b, J^{(1)}_c]] &= \hbar^2 A_{abc}^{def} \{J^{(0)}_d, J^{(0)}_e, J^{(0)}_f\}
\end{align*}
where $\{x,y,z\} = \frac{1}{6}\sum_{\sigma \in S_3} x_{\sigma(1)}y_{\sigma(2)}z_{\sigma(3)}$ is the symmetrized triple product ($|S_3| = 6$), and $A_{abc}^{def}$ are structure constants determined by $\mathfrak{g}$ (Drinfeld's ``terrific'' relation).
\end{definition}

\begin{definition}[Yangian current]\label{def:yangian-current}
The \emph{Yangian current} is the generating function:
\[
J_a(u) = \sum_{r \geq 0} J^{(r)}_a u^{-r-1}
\]
\end{definition}

\begin{definition}[RTT presentation]\label{def:yangian-rtt}
\index{Yangian!RTT presentation}
\index{R-matrix}
The \emph{RTT presentation} (Yang--Baxter presentation) of
$Y(\mathfrak{g})$ is generated by a \emph{transfer matrix}
$T(u) = (T_{ij}(u))_{i,j=1}^{\dim V}
\in \mathrm{End}(V) \otimes Y(\mathfrak{g})$
(valued in $\mathrm{End}(V)$ for a fundamental representation~$V$,
not in~$\mathfrak{g}$), satisfying the \emph{RTT relation}:\index{RTT relation}
\begin{equation}\label{eq:rtt-relation}
R_{12}(u-v)\, T_1(u)\, T_2(v)
\;=\; T_2(v)\, T_1(u)\, R_{12}(u-v)
\end{equation}
where $R_{12}(u) = 1 - \hbar P_{12}/u + O(u^{-2})$ is the
Yang $R$-matrix, $P_{12}$ is the permutation, and the subscripts
$1, 2$ indicate the tensor factor in
$\mathrm{End}(V) \otimes \mathrm{End}(V) \otimes Y(\mathfrak{g})$.
Expanding in modes
$T_{ij}(u) = \delta_{ij} + \hbar \sum_{r \geq 0} T_{ij}^{(r)} u^{-r-1}$,
the RTT relation \eqref{eq:rtt-relation} is equivalent to:
\begin{equation}\label{eq:rtt-modes}
[T_{ij}^{(r+1)}, T_{kl}^{(s)}]
- [T_{ij}^{(r)}, T_{kl}^{(s+1)}]
\;=\; T_{kj}^{(r)} T_{il}^{(s)}
- T_{kj}^{(s)} T_{il}^{(r)}.
\end{equation}
The RTT relation is \emph{quadratic} in the entries of $T$, making
this presentation directly amenable to quadratic Koszul duality
(mechanism~(iii) of Remark~\ref{rem:three-koszul-mechanisms}).
The isomorphism with the Drinfeld presentation
(Definition~\ref{def:yangian}) is given by
$J_a^{(0)} = \sum_{i,j} \rho(e_a)_{ij}\, T_{ij}^{(0)}$
and higher modes from the coproduct structure.
\end{definition}


\subsection{Yangian chiral algebra}

\begin{construction}[Yangian vertex algebra]\label{constr:yangian-va}
The \emph{Yangian chiral algebra} $Y(\mathfrak{g})^{\mathrm{ch}}$ is constructed as follows:

\emph{State space.} Completion of $Y(\mathfrak{g})$-modules.

\emph{Vertex operators.} $Y(J^{(r)}_a, z) = \sum_n J^{(r)}_{a,n} z^{-n-r-1}$.

\emph{OPE.} The RTT relation translates to:
\[
J_a(z) J_b(w) = \frac{f_{ab}^c J_c(w)}{z - w} + \frac{\alpha_{ab}^{cd} {:}J_c(w) J_d(w){:}}{(z-w)^2} + \text{regular}
\]
with higher poles from the quadratic Serre relations.
\end{construction}

\begin{theorem}[Yangian as \texorpdfstring{$\Eone$}{E1}-chiral; \ClaimStatusProvedHere]\label{thm:yangian-e1}
In the RTT presentation, the Yangian $Y(\mathfrak{g})^{\mathrm{ch}}$ carries the structure of a quadratic $\Eone$-chiral algebra (Definition~\ref{def:e1-chiral}; cf.\ \cite[§4.1]{HA}).  The $\Eone$-locality axiom is $R$-twisted commutativity:
\begin{equation}\label{eq:r-twisted-locality}
J_a(z) J_b(w) - R^{ab}_{cd}(z-w)\, J_d(w) J_c(z) = 0.
\end{equation}
For nontrivial $R \neq P$ (the permutation), this is strictly $\Eone$,
not $\Einf$.
\end{theorem}

\begin{proof}
\emph{Step~1: Quadraticity.}  The RTT relation~\eqref{eq:rtt-relation} is manifestly quadratic in the entries $T_{ij}(u)$.  More precisely, expanding $T_1(u) T_2(v)$ and $T_2(v) T_1(u)$ in the mode expansion $T_{ij}(u) = \delta_{ij} + \hbar \sum_r T_{ij}^{(r)} u^{-r-1}$, the relation~\eqref{eq:rtt-modes} expresses a quadratic relation among the generators $\{T_{ij}^{(r)}\}$.

\emph{Step~2: $\Eone$-chiral structure.}
The vertex operators $Y(T_{ij}^{(r)}, z) = \sum_n T_{ij,n}^{(r)} z^{-n-r-1}$ define fields on a curve $X$.  The OPE:
\[
T_{ij}(z,u)\, T_{kl}(w,v) \sim \frac{R^{ij,kl}(u-v)}{z-w} \cdot (\text{fusion})
\]
is $R$-matrix-valued, with simple poles.  The $R$-matrix $R(u) = 1 + \hbar r/u + O(u^{-2})$ provides the braiding: $\sigma_{12} \circ \mu = R_{12} \cdot \mu \circ \sigma_{12}$, which is the $\Eone$-locality axiom.  Associativity follows from the Yang--Baxter equation $R_{12}R_{13}R_{23} = R_{23}R_{13}R_{12}$.

\emph{Step~3: Non-commutativity.}
For $\mathfrak{g} = \mathfrak{sl}_2$, the classical $r$-matrix $r = \frac{1}{2}(e \otimes f + f \otimes e) + \frac{1}{4} h \otimes h$ (the split Casimir) satisfies $r \neq 0$, so the $R$-matrix $R(u) = 1 + \hbar r/u + O(u^{-2})$ is non-trivial: $R(u) \neq P$ for finite~$u$, and the RTT relation does not reduce to commutativity.  Thus the algebra is strictly $\Eone$.
\end{proof}

\begin{remark}[Presentations and quadraticity]\label{rem:yangian-presentations}
Whether the Yangian is ``quadratic'' depends on the presentation.  In the \emph{RTT presentation} (Definition~\ref{def:yangian-rtt}), the defining relation \eqref{eq:rtt-relation} is quadratic in $T$, so standard quadratic Koszul duality (mechanism~(iii) of Remark~\ref{rem:three-koszul-mechanisms}) applies directly.  In the \emph{Drinfeld presentation} (Definition~\ref{def:yangian}), the ``terrific relation'' is cubic in the generators, so Koszul duality requires nilpotent completion (Appendix~\ref{app:nilpotent-completion}).
The two presentations yield isomorphic algebras but different Koszul dual
computations.  A rigorous construction of the $\Eone$-chiral structure and its
Koszul dual should specify which presentation is used.
\end{remark}

\begin{definition}[\texorpdfstring{$\Eone$}{E1}-chiral Yangian]
\label{def:e1-chiral-yangian}
\index{Yangian!$\Eone$-chiral|textbf}
An \emph{$\Eone$-chiral Yangian} on a smooth algebraic curve~$X$
is an $\Eone$-factorization algebra on $\operatorname{Ran}(X)$
(Definition~\ref{def:e1-chiral}) whose local operator algebra, in
the RTT presentation, reproduces the Yangian $R(u)$-matrix
relations~\eqref{eq:rtt-relation} with the spectral parameter
identified as the coordinate difference of ordered insertion points:
$u = z_1 - z_2$.  Concretely, its bar complex
$\barB^{\Eone}_X(Y(\fg)^{\mathrm{ch}})$ uses \emph{ordered}
configuration spaces $\operatorname{Conf}^{\mathrm{ord}}_n(X)$,
and Verdier duality acts by $R$-matrix inversion
$R(u) \mapsto R(u)^{-1}$
(Theorem~\ref{thm:yangian-koszul-dual}), not by type exchange.
\end{definition}

\begin{remark}[Why ordered configurations are forced]
\label{rem:ordered-forced}
\index{ordered configurations!mathematical necessity}
The $\Eone$-chiral structure of Theorem~\ref{thm:yangian-e1} is
not an optional generalization of the $\Einf$-chiral theory.  It is
forced by the following chain of mathematical necessities.
\begin{enumerate}[label=\textup{(\arabic*)}]
\item The Kazhdan--Lusztig equivalence
  (Theorem~\ref{thm:kazhdan-lusztig-equivalence}) sends
  $\mathcal{O}_k^{\mathrm{int}}(\widehat{\fg})$ to
  $\mathcal{C}(U_q(\fg))$.  The target is a braided monoidal
  category whose braiding $\sigma$ is \emph{not} the symmetric
  flip: $R_{12} \neq P_{12}$.
\item Bar-cobar duality on the affine KM side reverses the braiding:
  $\sigma \mapsto \sigma^{-1}$
  (Theorem~\ref{thm:monoidal-module-koszul}).  Under the KL
  equivalence, this becomes $R_q \mapsto R_q^{-1}$, i.e.,
  $q \mapsto q^{-1}$.
\item On the quantum-group side, $R$-matrix inversion is not
  level inversion.  It is a Hopf algebra involution that
  reverses the \emph{ordering} of tensor factors.  To realize
  this geometrically on a curve, the bar complex must track
  which insertions are ``to the left'' and ``to the right'' ---
  hence ordered configurations $C_n^{\mathrm{ord}}(X)$.
\item Without ordering, Verdier duality on
  $\overline{C}_n(X)$ acts by level inversion $k \mapsto k'$
  (the $\Einf$-chiral story of Part~I).  With ordering, it
  acts by $R$-matrix inversion.  The two actions are related
  by the DK square~\eqref{eq:dk-square-affine}.
\end{enumerate}
In short: the KL equivalence exists; bar-cobar duality must be
compatible with it; compatibility forces ordered configurations.
\end{remark}

\section{Koszul dual of the Yangian}\label{sec:yangian-koszul}

\subsection{RTT bar complex computation}

\begin{theorem}[Yangian bar complex via RTT; \ClaimStatusProvedHere]\label{thm:yangian-bar-rtt}
\textup{[Regime: quadratic \textup{(}$\Eone$-chiral;
Convention~\textup{\ref{conv:regime-tags})}.]}

For $\mathfrak{g} = \mathfrak{sl}_2$, in the RTT presentation, the bar complex $\bar{B}(Y(\mathfrak{sl}_2)^{\mathrm{ch}})$ is computed as follows.

\emph{Generators.} In bar degree~1, the generators are the duals $T_{ij}^{(r)*}$ of the RTT generators, placed on $\overline{C}_2(X)$ with log forms.

\emph{Differential.} The bar differential extracts the RTT relation via double residues:
\begin{equation}\label{eq:yangian-bar-diff}
d(T_{ij}^{(r)*} \otimes \eta_{12}) = \sum_{k} \bigl(T_{ik}^{(r)*} \otimes T_{kj}^{(0)*} - T_{ik}^{(0)*} \otimes T_{kj}^{(r)*}\bigr) \otimes \eta_{12} \wedge \eta_{23}
\end{equation}
This is the quadratic dual of the RTT relation~\eqref{eq:rtt-modes}: the bar differential encodes the orthogonal complement of the relations under the chiral pairing.

\emph{Bar complex structure.} Since the RTT relation is quadratic, the Koszul dual $A^!$ is determined by the data in bar degrees $\leq 2$ alone: generators $V^*$ and relations $R^\perp$.  If the algebra is Koszul (i.e., $\mathrm{Ext}^{i,j}_A(\mathbb{C}, \mathbb{C}) = 0$ for $i \neq j$), then the bar-cobar quasi-isomorphism recovers $A$ from $A^!$.
\end{theorem}

\begin{proof}
The RTT relation~\eqref{eq:rtt-modes} is:
$[T_{ij}^{(r+1)}, T_{kl}^{(s)}] - [T_{ij}^{(r)}, T_{kl}^{(s+1)}] = T_{kj}^{(r)} T_{il}^{(s)} - T_{kj}^{(s)} T_{il}^{(r)}$.
This is quadratic in $T$, so the bar complex of the quadratic algebra has:
\begin{itemize}
\item $\bar{B}^0 = \mathbb{C}$ (vacuum),
\item $\bar{B}^1 = V^* = \mathrm{span}\{T_{ij}^{(r)*}\}$ (dual generators on $\overline{C}_2(X)$),
\item $\bar{B}^2 = R^\perp \subset V^* \otimes V^*$ (orthogonal complement of relations on $\overline{C}_3(X)$).
\end{itemize}
The differential $d\colon \bar{B}^2 \to \bar{B}^1$ is the dual of the inclusion $R \hookrightarrow V \otimes V$, which gives~\eqref{eq:yangian-bar-diff}.
\end{proof}

\subsection{Koszul dual identification}

For the Heisenberg algebra, Koszul duality exchanged Lie-type and Com-type: $\mathcal{H}_k^! = \mathrm{Sym}^{\mathrm{ch}}(V^*)$ (\S\ref{sec:frame-koszul-dual}).  In the Yangian setting, both sides are associative ($\Eone$), and Verdier duality inverts the $R$-matrix instead of exchanging operadic types.

\begin{theorem}[Yangian Koszul dual; \ClaimStatusProvedHere]\label{thm:yangian-koszul-dual}
\index{Yangian!Koszul dual}
The Koszul dual of the Yangian $Y(\mathfrak{sl}_2)^{\mathrm{ch}}$ in the RTT presentation is the Yangian with inverse $R$-matrix:
\begin{equation}\label{eq:yangian-koszul-dual}
Y(\mathfrak{sl}_2)^{!,\mathrm{ch}} \;\cong\; Y_{R^{-1}}(\mathfrak{sl}_2)^{\mathrm{ch}}
\end{equation}
where $Y_{R^{-1}}$ is the RTT algebra defined by the relation:
\[
R^{-1}_{12}(u-v)\, T_1^!(u)\, T_2^!(v) = T_2^!(v)\, T_1^!(u)\, R^{-1}_{12}(u-v)
\]
with
\[
R^{-1}(u) = 1 + \hbar P/u + O(u^{-2}),
\]
so the linear term changes sign from $R(u) = 1 - \hbar P/u + \cdots$.

For the Yang $R$-matrix $R(u) = 1 - \hbar P/u$, the inverse is
$R^{-1}(u) = u/(u - \hbar P) = \sum_{n \geq 0} (\hbar P/u)^n$;
since $P^2 = 1$,
\[
R^{-1}(u) = 1 + \hbar P/u + \hbar^2/u^2 + \hbar^3 P/u^3 + \cdots.
\]
The Koszul dual RTT relation is the same quadratic
relation with $\hbar \to -\hbar$.
\end{theorem}

\begin{proof}
\emph{Step~1: Quadratic dual.}
The RTT algebra is quadratic, with
$V = \mathrm{span}\{T_{ij}^{(r)}\}$ and
$R = \mathrm{span}\{[\text{RTT relations}]\}
\subset V \otimes V$.
The quadratic Koszul dual has generators~$V^*$
and relations~$R^\perp$, the orthogonal complement
under the pairing
$V^* \otimes V^* \times V \otimes V \to \mathbb{C}$.

\emph{Step~2: Orthogonal complement computation.}
The RTT relation $R_{12}(u-v) T_1 T_2 = T_2 T_1 R_{12}(u-v)$ can be written as:
\[
(1 - \hbar P/(u-v)) T_1 T_2 = T_2 T_1 (1 - \hbar P/(u-v))
\]
Expanding: $T_1 T_2 - T_2 T_1 = \frac{\hbar}{u-v}(P T_1 T_2 - T_2 T_1 P)$.

The orthogonal complement $R^\perp$ consists of quadratic expressions $Q(T^*, T^*)$ that pair to zero with all elements of $R$.  For $\mathfrak{sl}_2$ (where $V = \mathbb{C}^2$ and $P$ is the $4 \times 4$ permutation matrix), $R^\perp$ is computed by direct matrix calculation.  The RTT relation at fixed spectral parameters defines a linear map $\Phi_R\colon \mathrm{End}(V)^{\otimes 2} \to \mathrm{End}(V)^{\otimes 2}$ by $X \mapsto R_{12} X - \sigma(X) R_{12}$ where $\sigma$ is the flip; its image is the space of relations~$R$.  Writing out the $16 \times 16$ matrix of $\Phi_R$ in the standard basis of $\mathrm{End}(\mathbb{C}^2)^{\otimes 2}$ and computing its orthogonal complement with respect to the trace pairing, $R^\perp$ coincides with the image of $\Phi_{R^{-1}}$, i.e., the space of $R^{-1}$-RTT relations (verified by explicit $16\times 16$ matrix computation; see \texttt{test\_koszul\_dual\_dims.py}).

\emph{Step~2a: Local finiteness.}
The Yangian has infinitely many generators $\{T_{ij}^{(r)}\}_{r \geq 0}$ graded by level~$r$, but each graded component $V_r = \mathrm{span}\{T_{ij}^{(r)} : 1 \leq i,j \leq \dim V\}$ is finite-dimensional: $\dim V_r = (\dim V)^2 = 4$ for $\mathfrak{sl}_2$.  The RTT relation~\eqref{eq:rtt-modes} is homogeneous of degree $r + s$ in the level grading (relating generators of levels $r, s$ on both sides), so the quadratic algebra $A = T(V)/\langle R \rangle$ is locally finite: each graded component $A_n = \bigoplus_{r_1 + \cdots + r_k = n} V_{r_1} \cdots V_{r_k} / R_n$ is a quotient of a finite-dimensional space.

\emph{Step~2b: Graded Koszul duality.}
Since each graded component is finite-dimensional, the Positselski--Polishchuk framework \cite{PP05} for Koszul duality of locally finite graded algebras applies directly.  Their Theorem~2.1 establishes the bar-cobar adjunction and the Koszul property for quadratic algebras that are locally finite in a non-negative grading (which is satisfied here, with the level grading $r \geq 0$).  The key hypothesis --- that the grading is connected ($A_0 = \mathbb{C}$) --- holds after passing to the augmentation ideal.

\emph{Step~3: $R$-matrix inversion.}
For the Yang $R$-matrix, $R(u) = (u - \hbar P)/u$ (in the defining representation of $\mathfrak{sl}_2$).  The inverse is $R^{-1}(u) = u/(u - \hbar P) = 1 + \hbar P/u + (\hbar P)^2/u^2 + \cdots$.  Since $P^2 = 1$ on $V \otimes V$, this simplifies to $R^{-1}(u) = u/(u-\hbar)$ on the symmetric subspace and $R^{-1}(u) = u/(u+\hbar)$ on the antisymmetric subspace.

\emph{Step~4: Identification.}
The RTT algebra for $R^{-1}(u)$ is isomorphic to the RTT algebra for $R(u)$ with $\hbar \to -\hbar$.  To see this, note that $R^{-1}(u; \hbar) = 1 + \hbar P/u + O(u^{-2})$ agrees with $R(u; -\hbar) = 1 + \hbar P/u$ at leading order in $1/u$; the higher-order terms $\hbar^{2k}/u^{2k}$ and $\hbar^{2k+1}P/u^{2k+1}$ in $R^{-1}$ do \emph{not} appear in $R(u;-\hbar)$, but they are irrelevant: the mode expansion~\eqref{eq:rtt-modes} extracts only the leading-order data, and the higher-order terms are automatic consequences of the quadratic RTT relation via the Yang--Baxter equation.  Therefore $Y(\mathfrak{sl}_2)^! \cong Y(\mathfrak{sl}_2)^{\hbar \to -\hbar}$, confirming that the Yangian is ``almost self-dual'' up to a sign flip of the deformation parameter.
\end{proof}

\begin{corollary}[Yangian classical limit; \ClaimStatusProvedHere]\label{cor:yangian-classical-self-dual}
At $\hbar = 0$ (the classical limit), $R(u) = R^{-1}(u) = 1$, and the RTT algebra becomes commutative: $[T_{ij}^{(r)}, T_{kl}^{(s)}] = 0$, i.e., $\mathrm{Sym}(V)$ where $V = \bigoplus_{r \geq 0} \mathfrak{gl}_2^{(r)}$.  The Koszul dual of the symmetric algebra is the exterior algebra: $\mathrm{Sym}(V)^! = \bigwedge(V^*)$.  Since $R^{-1}(u;0) = R(u;0) = 1$, the defining relations of $\mathrm{Sym}(V)$ are the antisymmetric tensors $v \otimes w - w \otimes v$, and the orthogonal complement $R^\perp$ consists of the symmetric tensors $v \otimes w + w \otimes v$, which are the defining relations of $\bigwedge(V^*)$.  This confirms that the Yangian duality reduces to $\mathrm{Sym} \leftrightarrow \bigwedge$ at $\hbar = 0$.  (In the Drinfeld presentation, the classical limit is the noncommutative algebra $U(\mathfrak{g}[t])$; the two presentations yield different classical limits.)
\end{corollary}

\begin{remark}[Connection to Langlands duality]\label{rem:yangian-langlands}
The sign flip $\hbar \to -\hbar$ in the Yangian Koszul dual is the analog of the Feigin--Frenkel level shift $k \to -k - 2h^\vee$ for affine algebras.  For simply-laced $\mathfrak{g}$, the Yangian $Y(\mathfrak{g})$ is isomorphic to $Y(\mathfrak{g}^\vee)$ (since $\mathfrak{g} \cong \mathfrak{g}^\vee$), and the Koszul duality becomes a form of self-duality.  For non-simply-laced $\mathfrak{g}$, one expects $Y(\mathfrak{g})^! \simeq Y(\mathfrak{g}^\vee)$, realizing Langlands duality at the Yangian level.  Latyntsev \cite{Latyntsev23} constructs factorisation quantum groups that provide a complementary approach to Yangian duality via factorisation homology.
\end{remark}

\begin{remark}[The \texorpdfstring{$\Eone$}{E1} inversion principle and the nilpotence-periodicity correspondence]
\label{rem:yangian-e1-inversion}
\index{E1 inversion principle@$\Eone$ inversion principle!Yangian instance}
\index{nilpotence-periodicity correspondence!Yangian instance}
The $R$-matrix inversion $R \mapsto R^{-1}$ in
Theorem~\ref{thm:yangian-koszul-dual} is one instance of a
general principle for $\Eone$-chiral Koszul duality: the
non-commutativity datum is always inverted.  For lattice
vertex algebras with non-symmetric cocycles
(\S\ref{sec:lattice:e1-families}), the cocycle is inverted:
$\varepsilon \mapsto \varepsilon^{-1}$.  Both instances arise from
the self-duality ${\chirAss}^! \cong \chirAss \otimes \mathrm{sgn}$
of the chiral associative operad, with the sign twist realized as
orientation reversal on ordered configurations.  See
Theorem~\ref{thm:e1-inversion-principle} for the unified
statement.

In the nilpotence-periodicity framework
(Remark~\ref{rem:nilpotence-periodicity}), the inversion
$R \mapsto R^{-1}$ is the statement that the categorical logarithm
changes sign under Koszul duality: $\log(\cA^!) = -\log(\cA)$.
Exponentiating: if $\cA$ has $R$-matrix $R(u) = 1 + \hbar/u + \cdots$,
then $\cA^!$ has $R$-matrix $R(u)^{-1} = 1 - \hbar/u + \cdots$; the
nilpotent datum~$\hbar$ flips sign, and the periodic datum
($R$-matrix eigenvalues) is inverted.  For the Yangian $Y(\fg)$, the
bar complex is nilpotent ($d^2 = 0$), while the $R$-matrix
$R(u) = 1 + \hbar P/u$ generates the periodic structure of quantum
group representations; the bar construction extracts the former from
the latter.
\end{remark}

\subsection{Koszulness and bar-cobar recovery}

\begin{proposition}[RTT Yangian is Koszul; \ClaimStatusProvedHere]\label{prop:yangian-koszul}
\index{Yangian!Koszulness}
For any simple Lie algebra $\mathfrak{g}$ with fundamental
representation~$V_0$ of dimension~$N = \dim V_0$, assume that
the RTT Yangian $Y_{\hbar}(\fg)$ satisfies the PBW property
\textup{(}Molev \cite{molev-yangians}\textup{)}.
Then $Y(\mathfrak{g})$ in the RTT presentation
(Definition~\ref{def:yangian-rtt}) is Koszul: the bar complex
$\bar{B}(Y(\mathfrak{g}))$ has cohomology concentrated on the
diagonal, i.e., $\mathrm{Ext}^{i,j}_{Y}(\mathbb{C}, \mathbb{C}) = 0$
for $i \neq j$ in the $(i, j)$-bigrading by homological degree and
internal (level) degree.
\end{proposition}

\begin{proof}
\emph{Step~1: PBW basis.}
Molev \cite{molev-yangians} establishes a Poincar\'{e}--Birkhoff--Witt
basis for $Y(\mathfrak{g})$ in the RTT presentation for every simple
$\mathfrak{g}$: ordered monomials
in the generators $\{T_{ij}^{(r)}\}_{1 \leq i,j \leq N,\, r \geq 0}$
(with respect to a fixed total order on the index set $(i,j,r)$) form
a vector space basis for $Y(\mathfrak{g})$.  The
associated graded algebra with respect to the PBW filtration is the
polynomial algebra $\mathrm{Sym}(V)$, where
$V = \bigoplus_{r \geq 0} \mathrm{End}(\mathbb{C}^N)_r$.

\emph{Step~2: Graded local finiteness.}
The level grading $V = \bigoplus_{r \geq 0} V_r$ with
$\dim V_r = N^2 = (\dim V_0)^2$ ensures that each graded component
of the quadratic algebra is finite-dimensional.  The RTT
relation~\eqref{eq:rtt-modes} is homogeneous of degree~$r+s$ in the
level grading, so the quadratic algebra
$A = T(V)/\langle R \rangle$ is locally finite: the graded
component~$A_n$ is a quotient of the finite-dimensional space
$\bigoplus_{r_1 + \cdots + r_k = n} V_{r_1} \otimes \cdots
\otimes V_{r_k}$.  This satisfies the local finiteness hypothesis
of the Polishchuk--Positselski framework
(cf.\ step~2b of the proof of
Theorem~\ref{thm:yangian-koszul-dual}).  The key point is
that $N^2$ is finite and independent of~$r$, so local finiteness
holds uniformly for all simple~$\mathfrak{g}$.

\emph{Step~3: PBW implies Koszul.}
By the Polishchuk--Positselski PBW criterion \cite[Ch.~4]{PP05}, a
quadratic algebra possessing a PBW basis is Koszul, provided
it is locally finite in a non-negative connected grading (verified
in Step~2).  The argument:
the PBW property gives a flat deformation from $Y(\mathfrak{g})$
to its associated graded $\mathrm{Sym}(V)$.  Since the Koszul complex
$K(A) = A \otimes A^{!\ast}$ is functorial in the quadratic data, and
acyclicity of $K(\mathrm{Sym}(V))$ is classical ($\mathrm{Sym}$--$\bigwedge$ duality),
flatness of the deformation implies acyclicity of $K(Y(\mathfrak{g}))$.
\end{proof}

\begin{corollary}[Yangian bar-cobar recovery; \ClaimStatusProvedHere]
\label{cor:yangian-bar-cobar}
For any simple~$\mathfrak{g}$, the bar-cobar adjunction provides a
quasi-isomorphism of $\Eone$-chiral algebras:
\[
\Omega^{\mathrm{ch}}(\bar{B}^{\mathrm{ch}}(Y(\mathfrak{g})^{\mathrm{ch}}))
\xrightarrow{\;\sim\;} Y(\mathfrak{g})^{\mathrm{ch}}.
\]
\end{corollary}

\begin{proof}
The RTT Yangian is quadratic (Definition~\ref{def:yangian-rtt}),
locally finite in the level grading (step~2b of the proof of
Theorem~\ref{thm:yangian-koszul-dual}), and Koszul
(Proposition~\ref{prop:yangian-koszul}).  The mode expansion
$T_{ij}(u) = \delta_{ij} + \hbar \sum_{r \geq 0} T_{ij}^{(r)}
u^{-r-1}$ defines the augmentation $\varepsilon(T_{ij}^{(r)}) = 0$
for all $r \geq 0$, with augmentation ideal $I = \langle
T_{ij}^{(r)} \rangle$.  The PBW basis ensures $\bigcap_n I^n = 0$,
so the algebra is pro-nilpotent.
Theorem~\ref{thm:e1-chiral-koszul-duality} then applies:
the counit~\eqref{eq:e1-counit} of the $\Eone$-chiral bar-cobar
adjunction is a quasi-isomorphism for augmented pro-nilpotent
$\Eone$-chiral algebras.
\end{proof}

\begin{remark}[Canonical twisting morphism for Yangians]
\label{rem:yangian-twisting-morphism}
The canonical twisting morphism for the RTT Yangian
$\tau_Y \colon \barB(Y_\hbar(\fg)) \to Y_\hbar(\fg)$
projects bar generators $s^{-1}T_{ij}^{(r)*}$ to the RTT
generators~$T_{ij}^{(r)}$.  The Maurer--Cartan equation encodes the
RTT relation $R_{12}(u-v)\, T_1(u)\, T_2(v) = T_2(v)\, T_1(u)\, R_{12}(u-v)$:
the convolution square $\tau_Y \star \tau_Y$ extracts the $R$-matrix
coefficients.  Koszulness (Proposition~\ref{prop:yangian-koszul})
says that the twisted tensor product
$Y_\hbar(\fg) \otimes_{\tau_Y} \barB(Y_\hbar(\fg))$
is acyclic---the RTT relations generate all syzygies.
\end{remark}

\subsection{Yangian modules and Koszul duality}

\begin{definition}[Evaluation module]\label{def:yangian-evaluation}
\index{Yangian!evaluation module|textbf}
\index{evaluation module}
For $a \in \mathbb{C}$ (the \emph{evaluation point}) and a
finite-dimensional $\mathfrak{g}$-module $V$, the \emph{evaluation
module} $V(a)$ of the Yangian $Y(\mathfrak{g})$ is defined by
pulling back along the evaluation homomorphism
\[
\mathrm{ev}_a\colon Y(\mathfrak{g}) \to U(\mathfrak{g}),
\qquad
J^{(r)}_b \mapsto a^r\, J_b.
\]
The $R$-matrix acts on tensor products of evaluation modules by:
\begin{equation}\label{eq:yangian-r-matrix}
R_{V(a), W(b)}
= R_{V,W}\!\left(\frac{a - b}{\hbar}\right)
\in \mathrm{End}(V \otimes W)
\end{equation}
where $R_{V,W}(u) = 1 + r_{V,W}/u + O(u^{-2})$ is the rational
$R$-matrix in the representations~$V, W$.
\end{definition}

\begin{proposition}[Koszul duality on Yangian modules; \ClaimStatusProvedHere]
\label{prop:yangian-module-koszul}
\index{Yangian!module Koszul duality}
The Koszul duality $Y(\mathfrak{g})^! \cong Y_{-\hbar}(\mathfrak{g})$
\textup{(}Theorem~\textup{\ref{thm:yangian-koszul-dual})} extends to
modules as follows:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Evaluation modules:} The Koszul dual of the evaluation
  module $V(a)$ at parameter $\hbar$ is $V(-a)$ at parameter $-\hbar$:
  $\Phi(V(a)_\hbar) = V(-a)_{-\hbar}$.
\item \emph{Tensor products:} For evaluation modules $V(a)$ and
  $W(b)$, the Koszul dual of $V(a) \otimes W(b)$ is
  $V(-a) \otimes W(-b)$ with the $R$-matrix replaced by its inverse:
  \[
  R^!_{V(-a), W(-b)}
  = R_{V,W}^{-1}\!\left(\frac{a - b}{\hbar}\right).
  \]
\item \emph{Braiding reversal:} The braiding on the category of
  finite-dimensional Yangian modules is reversed:
  $\sigma^! = \sigma^{-1}$, parallel to the
  $q \leftrightarrow q^{-1}$ reversal for quantum groups
  \textup{(}Remark~\textup{\ref{rem:kl-koszul})}.
\end{enumerate}
\end{proposition}

\begin{proof}
(i) The evaluation homomorphism $\mathrm{ev}_a$ sends
$T_{ij}^{(r)} \mapsto a^r \rho(e_a)_{ij}$.  Under
$\hbar \to -\hbar$, the RTT relation retains the same form
$R'\, T_1 T_2 = T_2 T_1\, R'$ but with $R$-matrix replaced by
$R^{-1}$.  The evaluation at
parameter $a$ in the original algebra corresponds to evaluation
at $-a$ in the dual algebra, since the spectral parameter
enters the $R$-matrix via $R(u) = 1 - \hbar P/u$, and
replacing $\hbar \to -\hbar$ is equivalent to $u \to -u$, i.e.,
$a \to -a$.

(ii) follows from (i) applied to each tensor factor, combined
with the functoriality of the bar construction on tensor
products (Theorem~\ref{thm:monoidal-module-koszul}).

(iii) The braiding on Yangian modules is given by
$\sigma = P \circ R(u)$, where $P$ is the permutation and
$R(u)$ is the $R$-matrix.  Under $\hbar \to -\hbar$:
$R(u; -\hbar)$ and $R^{-1}(u; \hbar)$ define the same quadratic
dual algebra (Theorem~\ref{thm:yangian-koszul-dual}, Step~4), so
$\sigma^! = P \circ R^{-1}(u) = \sigma^{-1}$.
\end{proof}

\begin{example}[Yangian \texorpdfstring{$Y(\mathfrak{sl}_2)$}{Y(sl_2)} fundamental modules]
\label{ex:yangian-sl2-modules}
\index{Yangian!sl2 modules@sl2 mods}
The Yangian $Y(\mathfrak{sl}_2)$ has evaluation modules
$V(a) \cong \mathbb{C}^2$ for each $a \in \mathbb{C}$.  The
$R$-matrix on $V(a) \otimes V(b)$ is:
\[
R(u) = \frac{u - \hbar P}{u - \hbar}
= \begin{pmatrix}
1 & 0 & 0 & 0 \\
0 & \frac{u}{u-\hbar} & \frac{-\hbar}{u-\hbar} & 0 \\
0 & \frac{-\hbar}{u-\hbar} & \frac{u}{u-\hbar} & 0 \\
0 & 0 & 0 & 1
\end{pmatrix}
\quad\text{at } u = \tfrac{a-b}{\hbar}.
\]
The tensor product $V(a) \otimes V(b)$ decomposes as
$\mathrm{Sym}^2(V) \oplus \bigwedge^2(V)$ when $a \neq b$
(generic), and is indecomposable when $a = b$ (the $R$-matrix
has a pole).  Under Koszul duality,
$V(a) \otimes V(b) \mapsto V(-a) \otimes V(-b)$ with the
$R$-matrix $R(u) \mapsto R^{-1}(u)$: the $R$-matrix eigenvalues
on $\mathrm{Sym}^2(V)$ and $\bigwedge^2(V)$ are inverted.
\end{example}

\begin{remark}[Verification of the main theorems for Yangians]
\label{rem:yangian-three-theorems}
\index{three theorems!Yangian showcase}
We trace the three main theorems through the Yangian
$Y(\mathfrak{sl}_2)^{\mathrm{ch}}$, highlighting the differences
between the $\Eone$-chiral and $\Einf$-chiral (vertex algebra) settings.

\emph{Theorem~A (Geometric Bar-Cobar Duality).}
The bar construction $\bar{B}(Y(\mathfrak{sl}_2)^{\mathrm{ch}})$
is computed via the RTT presentation
(Theorem~\ref{thm:yangian-bar-rtt}).  The resulting Koszul dual
$Y_{R^{-1}}(\mathfrak{sl}_2)^{\mathrm{ch}}$
(Theorem~\ref{thm:yangian-koszul-dual}) is the RTT algebra with
$R$-matrix inverted, equivalently the parameter flip
$\hbar \to -\hbar$.  This is the $\Eone$ analog of the
Feigin--Frenkel involution $k \to -k - 2h^\vee$ for affine algebras
(Remark~\ref{rem:yangian-langlands}).  The bar-cobar adjunction is an
equivalence by Theorem~\ref{thm:e1-chiral-koszul-duality}, and the
cobar reconstruction recovers the Yangian
(Corollary~\ref{cor:yangian-bar-cobar}).

\emph{Theorem~B (Bar-Cobar Inversion).}
By Proposition~\ref{prop:yangian-koszul}, the RTT Yangian is Koszul,
so the spectral sequence of
Theorem~\ref{thm:bar-cobar-inversion-qi} collapses at $E_2$.
At the classical level ($\hbar = 0$), this reduces to the standard
$\mathrm{Sym} \leftrightarrow \bigwedge$ Koszul resolution
(Corollary~\ref{cor:yangian-classical-self-dual}).

\emph{Theorem~C (Complementarity).}
The deformation-obstruction complementarity
(Theorem~\ref{thm:quantum-complementarity-main}) is formulated for
$\Einf$-chiral algebras via the modular operad and Feynman transform.
The Yangian, being strictly $\Eone$, does not possess a Virasoro
subalgebra, so neither the central charge $c$ nor the obstruction
coefficient $\kappa$ is defined---this explains the ``---'' entries
in the Master Table (Table~\ref{tab:master-invariants}).  The correct
$\Eone$ analog of genus-$g$ complementarity would involve the moduli
of bordered Riemann surfaces and ribbon graph expansions, rather than
$\overline{\mathcal{M}}_g$ and tautological classes.  Developing this
$\Eone$ genus theory remains an important open direction.
\end{remark}


\begin{conjecture}[\texorpdfstring{$\Eone$}{E1} genus theory;
\ClaimStatusConjectured]\label{conj:e1-genus-theory}
\index{E1-chiral algebra@$\Eone$-chiral algebra!genus theory}
For an $\Eone$-chiral algebra $\cA$ on $\bR$
(or on an oriented interval), the genus-$g$ obstruction
should be defined using the moduli of bordered Riemann
surfaces $\mathcal{M}_{g,1}^{\partial}$
(surfaces with one parameterized boundary component).
Specifically:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{$\Eone$ bar complex at genus~$g$.}
The genus-$g$ bar complex
$\bar{B}_g^{\Eone}(\cA) :=
\bigoplus_{n \geq 0} \Gamma\bigl(
\overline{C}_n(\Sigma_{g,1}),\,
\cA^{\boxtimes n} \otimes
\Omega^n_{\log}\bigr)$
uses configuration spaces on the genus-$g$
surface $\Sigma_{g,1}$ with one boundary component.
\item \emph{Nilpotency via ribbon graphs.}
The bar differential satisfies $d^2 = 0$ on
$\Sigma_{g,1}$, with the proof using the
Orlik--Solomon relations on the
compactified configuration space
$\overline{C}_n(\Sigma_{g,1})$
(generalizing the genus-$0$ case for
$\Sigma_{0,1} \cong \bR^2$).
\item \emph{Ribbon graph expansion.}
The obstruction $\mathrm{obs}_g^{\Eone}(\cA)$
should be computed by summing over ribbon graphs
$\Gamma$ with genus~$g$ and one boundary cycle:
\[
\mathrm{obs}_g^{\Eone}(\cA) =
\sum_{\Gamma \in \mathrm{Rib}_{g,1}}
\frac{1}{|\mathrm{Aut}(\Gamma)|}
\int_{\overline{C}_\Gamma(\Sigma_{g,1})}
\omega_\Gamma(\cA)
\]
where $\omega_\Gamma(\cA)$ is the form
associated to the OPE data of~$\cA$ on the
edges and vertices of~$\Gamma$.
\item \emph{Complementarity.}
$\mathrm{obs}_g^{\Eone}(\cA) +
\mathrm{obs}_g^{\Eone}(\cA^!) = 0$
in $H^*(\mathcal{M}_{g,1}^{\partial})$,
since the $\Eone$ Koszul duality inverts
the $R$-matrix ($R \to R^{-1}$), which
reverses the orientation of the ribbon
graphs.
\end{enumerate}
For the Yangian $Y(\mathfrak{sl}_2)^{\mathrm{ch}}$,
the ribbon graph expansion at genus~$1$ should recover
$\mathrm{obs}_1 \propto \hbar$ (the deformation parameter),
vanishing at $\hbar = 0$ (the classical limit,
Corollary~\textup{\ref{cor:yangian-classical-self-dual}}).
(Contributing to Conjecture~\ref{conj:master-dk-kl}.)
\end{conjecture}

\begin{remark}[Scope]\label{rem:e1-genus-scope}
This conjecture requires several foundational
developments: (a)~a theory of compactified configuration
spaces on bordered surfaces with the correct boundary
behavior (extending Fulton--MacPherson to the case
with boundary); (b)~an $\Eone$ analog of the Feynman
transform for the modular operad; (c)~an $\Eone$ analog
of the Faber--Pandharipande formula for tautological
integrals on $\mathcal{M}_{g,1}^{\partial}$.  The
closest existing work is the Swiss-cheese operad
(Voronov~\cite{Voronov99}), which governs open-closed
homotopy algebras.
The toroidal algebra
(Remark~\ref{rem:toroidal-three-theorems}) faces the
same obstruction.

\emph{Homotopy template} (Convention~\ref{conv:hms-levels}): Type~I (existence).
The conjecture posits the existence of an $\Eone$ genus theory on bordered Riemann surfaces, extending the chiral bar complex from closed to open surfaces.
\end{remark}


\section{Shifted Yangians}
\label{sec:shifted-yangians}

\subsection{Definition}

\begin{definition}[Shifted Yangian]\label{def:shifted-yangian}
For a coweight $\mu \in X_*(\mathfrak{h})$, the \emph{shifted Yangian} $Y_\mu(\mathfrak{g})$
(Brundan--Kleshchev, Finkelberg--Tsymbaliuk, KWWY) is generated by elements
$E_i^{(r)}, F_i^{(r)}, H_i^{(r)}$ for $i = 1, \ldots, \mathrm{rank}(\mathfrak{g})$
and $r \geq 0$, satisfying the same Yangian relations as $Y(\mathfrak{g})$ except that the
generating function $H_i(u) = \sum_{r \geq 0} H_i^{(r)} u^{-r}$ is constrained to be a rational
function whose numerator and denominator degrees are prescribed by $\mu$:
\[
H_i(u) = \frac{\prod_{j=1}^{\mu_i^+}(u - a_j^{(i)})}{\prod_{j=1}^{\mu_i^-}(u - b_j^{(i)})}
\]
where $\mu_i^\pm = \langle \alpha_i, \mu^\pm \rangle$ and $\mu = \mu^+ - \mu^-$ decomposes the coweight into dominant parts. The parameters $a_j^{(i)}, b_j^{(i)}$ are complex numbers.
\end{definition}

\begin{conjecture}[Shifted Yangian as \texorpdfstring{$\Eone$}{E1}-chiral; \ClaimStatusConjectured]\label{conj:shifted-yangian-e1}
The shifted Yangian admits a chiral algebra structure $Y_\mu(\mathfrak{g})^{\mathrm{ch}}$
that is $\Eone$-chiral.  The shift $\mu$ modifies the central terms in the OPE,
and the rational constraint on $H_i(u)$ introduces additional poles in the vertex
operators.
(Contributing to Conjecture~\ref{conj:master-dk-kl}.)
\end{conjecture}

\begin{remark}[Scope]
A proof would require: (1)~an explicit vertex operator construction for
the shifted Yangian generators $E_i^{(r)}, F_i^{(r)}, H_i^{(r)}$ with
the rational constraint on $H_i(u)$; (2)~verification of the Borcherds
identity (weak associativity of the OPE) in the presence of the additional
poles introduced by the shift.  For the unshifted case ($\mu = 0$), the
$\Eone$-chiral structure is established in Theorem~\ref{thm:yangian-e1}.
A conjectural extension is that the shifted case follows by an analogous argument using the
BFN construction (Theorem~\ref{thm:bfn}), which realizes the shifted
Yangian as equivariant homology and thus provides a natural source of
vertex operators.  Dimofte--Niu--Py \cite{DNP25} construct dg-shifted Yangians
from line operators in 3d holomorphic QFT, providing further evidence for the
$\Eone$-chiral structure in the shifted setting.

\emph{Homotopy template} (Convention~\ref{conv:hms-levels}): Type~I (existence).
The conjecture asserts the existence of an $\Eone$-chiral algebra structure on the shifted Yangian $Y_\mu(\mathfrak{g})$, extending the unshifted construction to non-zero coweights.
\end{remark}

\subsection{Comparison with dg-shifted Yangians}
\label{sec:dg-shifted-comparison}

\begin{proposition}[Structural comparison; \ClaimStatusProvedHere]
\label{prop:dg-shifted-comparison}
\index{Yangian!dg-shifted}
The $\Eone$-chiral Yangian
$Y(\mathfrak{g})^{\mathrm{ch}}$
\textup{(}Theorem~\textup{\ref{thm:yangian-e1})}
and the dg-shifted Yangian
$Y^{\mathrm{dg}}(\mathfrak{g})$
of Dimofte--Niu--Py~\textup{\cite{DNP25}}
share the following structural features:
\begin{enumerate}[label=\textup{(\roman*)}]
\item Both are $\Eone$-deformations of the affine algebra
  $\widehat{\mathfrak{g}}_k$, controlled by the rational
  $R$-matrix $R(u) = 1 + r/u + O(u^{-2})$.
\item Both have generators in RTT form
  $T(u) = 1 + \sum_{r \geq 1} T^{(r)} u^{-r}$
  with quadratic relations
  $R_{12}(u{-}v)\,T_1(u)\,T_2(v) = T_2(v)\,T_1(u)\,R_{12}(u{-}v)$.
\item The Koszul dual of both is the $R^{-1}$-Yangian:
  replacing $R$ by $R^{-1}$ corresponds to
  $\hbar \mapsto -\hbar$.
\item Evaluation modules $V(a)$ and their tensor products
  behave identically in both constructions
  (Proposition~\ref{prop:eval-module-bar}).
\end{enumerate}
The constructions differ in their geometric origins:
$Y(\mathfrak{g})^{\mathrm{ch}}$ arises from FM configuration
space integrals on algebraic curves, while
$Y^{\mathrm{dg}}(\mathfrak{g})$ arises from line operators
in 3d holomorphic Chern--Simons theory~\textup{\cite{DNP25}}.
\end{proposition}

\begin{proof}
Part~(i) follows from comparing the deformation parameter:
both constructions deform the vertex algebra OPE by a
spectral parameter $u = z_1 - z_2$ that controls
$\Eone$-locality.  In the FM picture, $u$ is the
relative coordinate on $\overline{C}_2(X)$
(Theorem~\ref{thm:yangian-e1}).  In the DNP construction,
$u$ is the holomorphic coordinate along the topological
line (the ``spectral parameter'' of \cite[\S5]{DNP25}).

Part~(ii): the RTT presentation is a presentation of the
quadratic algebra, hence determined by the $R$-matrix.
Since both constructions use the same $R$-matrix, the
RTT relations agree.

Part~(iii) follows from Theorem~\ref{thm:yangian-koszul-dual}
(this manuscript) and \cite[Thm.~6.12]{DNP25}
(the Koszul dual of line operators is controlled by $R^{-1}$).

Part~(iv): the bar complex of the evaluation module
is the CE complex $C^*(\mathfrak{g}, V)$
(Proposition~\ref{prop:eval-module-bar}), which
depends only on the underlying $\mathfrak{g}$-action,
not on the geometric origin of the Yangian structure.
\end{proof}

\begin{remark}[Quasi-isomorphism question]
\label{rem:dg-shifted-qi}
The structural comparison above does not establish a
quasi-isomorphism $Y(\mathfrak{g})^{\mathrm{ch}}
\simeq Y^{\mathrm{dg}}(\mathfrak{g})$: this would require
constructing a separated complete RTT-level filtration on the
dg-shifted target whose finite quotients recover the theorematic finite
RTT stages.  Corollary~\ref{cor:yangian-hlevel-comparison-criterion}
shows that once such compatible finite quotients are in place, the
comparison with the coefficientwise RTT completion is formal.
Koszulness (Remark~\ref{rem:yangian-collapse-conj}) remains one natural
route to producing those compatible quotient identifications.
\end{remark}

\begin{remark}[Maurer--Cartan element as twisting morphism]
\label{rem:dnp-mc-twisting}
\index{Maurer--Cartan element!twisting morphism}
\index{Yang--Baxter equation!A-infinity@$A_\infty$}
The dg-shifted Yangian of \cite{DNP25} carries a
Maurer--Cartan element $r(z) \in \cA^! \otimes \cA^!$
satisfying an $A_\infty$ Yang--Baxter equation.
In the bar-cobar framework, this element admits a precise
identification: $r(z)$ is the \emph{twisting morphism}
$\tau\colon \bar{B}(\cA) \to \cA^!$
of Theorem~\ref{thm:bar-cobar-isomorphism-main},
restricted to the degree-$2$ component and evaluated at
spectral parameter~$z$.

Explicitly:
\begin{enumerate}[label=\textup{(\roman*)}]
\item The MC equation $dr + r \star r = 0$ is the
  bar-cobar adjunction equation
  $\partial \tau + \tau \cup \tau = 0$,
  which is equivalent to $d_{\bar{B}}^2 = 0$
  (Theorem~\ref{thm:bar-nilpotency-complete}).
\item The $A_\infty$ Yang--Baxter equation of
  \cite[\S5]{DNP25} is the $\Eone$-chiral generalization:
  the bar complex differential encodes the RTT relation
  (Theorem~\ref{thm:yangian-bar-rtt}), and $d^2 = 0$
  is precisely the Yang--Baxter equation for $R(u)$.
\item The non-renormalization theorem of \cite[Thm.~4.1]{DNP25}
  (OPE of line operators is $1$-loop exact) corresponds to
  the Koszul property: the bar spectral sequence collapses
  at $E_2$, so the $A_\infty$ structure is determined by
  the quadratic ($R$-matrix) data.
\end{enumerate}

The bridge claim (Conjecture~\ref{conj:chiral-dg-shifted-qi}) is:
the $\Eone$-chiral Yangian and the dg-shifted Yangian are two
presentations of one object, one via configuration-space
integrals (this manuscript) and one via holomorphic
Chern--Simons perturbation theory~(\cite{DNP25}).  The missing
input is the Koszulness of the $\Eone$-chiral bar complex
(Remark~\ref{rem:yangian-collapse-conj}).
\end{remark}

\begin{conjecture}[Chiral--dg-shifted quasi-isomorphism; \ClaimStatusConjectured]\label{conj:chiral-dg-shifted-qi}
The $\Eone$-chiral Yangian $Y(\fg)^{\mathrm{ch}}$ of this monograph and the
dg-shifted Yangian $Y^{\mathrm{dg}}(\fg)$ of Dimofte--Niu--Py~\cite{DNP25}
are quasi-isomorphic as $\Eone$-algebras.
\end{conjecture}

\begin{proposition}[Degree-2 mixed-tensor criterion for dg-shifted
local transport; \ClaimStatusProvedHere]
\label{prop:dg-shifted-rtt-degree2-mixed-tensor-criterion}
Let $\Ydg_{\cA}$ be a dg-shifted or factorization-theoretic Yangian
candidate with one-factor fundamental line operator
$L_a^{\mathrm{dg}}(u)\in \End(V)\otimes \End(V(a))$, $V=\mathbb{C}^M$,
and residue operator
$\Xi_a^{\mathrm{dg}}:=\operatorname*{Res}_{u=a}L_a^{\mathrm{dg}}(u)$.
Assume:
\begin{enumerate}[label=\textup{(\roman*)}]
\item the one-factor line operator is functorial for the diagonal
  $\mathfrak{sl}_M$-action and normalized by
  $L_a^{\mathrm{dg}}(u)\to 1$ as $u\to\infty$;
\item the local OPE on the fundamental line is one-loop exact, with
  simple pole computed by the degree-$2$ Maurer--Cartan element
  $r(z)=\tau|_{\deg 2}$ from Remark~\ref{rem:dnp-mc-twisting};
\item $\Xi_a^{\mathrm{dg}}(e_1\otimes e_2)=-\hbar(e_2\otimes e_1)$.
\end{enumerate}
Then $\Xi_a^{\mathrm{dg}}=-\hbar P$ and
$L_a^{\mathrm{dg}}(u)=R_{0a}(u-a)$.
If, moreover, the twisted-coproduct hypotheses of
Corollary~\ref{cor:dg-shifted-rtt-typea-single-line-bootstrap} hold on
universal packets and generic tensor powers, then the dg-shifted local
Yangian packet closes: every universal packet carries the standard
finite-stage evaluation action and $\Delta_{r,0}(N)=0$ for $0\le r\le N$
on the evaluation-generated core.

Equivalently, once the standard RTT side is fixed, the remaining local
transport problem is reduced to transporting the degree-$2$ twisting
coefficient on the ordered mixed-tensor line $e_1\otimes e_2$ to the
standard normalized residue.
\end{proposition}

\begin{proof}
Assumption~\textup{(iii)} is exactly the ordered mixed-tensor formula of
Proposition~\ref{prop:dg-shifted-rtt-typea-uniform-single-line}, while
\textup{(i)} supplies the functoriality and normalization hypotheses
used there and in
Corollary~\ref{cor:dg-shifted-rtt-typea-residue-reduction}, giving
$\Xi_a^{\mathrm{dg}}=-\hbar P$ and $L_a^{\mathrm{dg}}(u)=R_{0a}(u-a)$.
The twisted-coproduct statement follows verbatim from
Corollary~\ref{cor:dg-shifted-rtt-typea-single-line-bootstrap}.
\end{proof}

\begin{corollary}[Seed-normalized extraction of the dg-shifted local
coefficient; \ClaimStatusProvedHere]
\label{cor:dg-shifted-rtt-seed-normalized-coefficient}
In the setting of
Proposition~\ref{prop:dg-shifted-rtt-degree2-mixed-tensor-criterion},
write the one-loop singular part in the standard logarithmic
normalization $r_a^{\mathrm{fund}}(z)\,\eta_{12}$,
$\eta_{12}=d\log(z_1-z_2)$,
with $r_a^{\mathrm{fund}}(z)$ the operator-valued degree-$2$ twisting
morphism coefficient induced by $r(z)=\tau|_{\deg 2}$ on
$V\otimes V(a)$.  Then

The following chain of propositions establishes successively weaker

\begin{proposition}[Stepwise reduction to evaluation detection;
\ClaimStatusProvedHere]
\label{prop:dg-shifted-rtt-presentation-criterion}
\label{prop:dg-shifted-rtt-locality-criterion}
\label{prop:dg-shifted-rtt-formula-preservation}
\label{prop:dg-shifted-rtt-coefficient-criterion}
\label{prop:dg-shifted-rtt-kernel-coefficient-criterion}
\label{prop:dg-shifted-rtt-oneloop-kernel-criterion}
\label{prop:dg-shifted-rtt-evaluation-detection}
The following conditions on the quotient
$Q_N:=\Ydg_{\cA}/F^{N+1}\Ydg_{\cA}$ are each sufficient for
$Q_N \cong Y_{\le N}$, listed from strongest to weakest.
Each step implies its predecessor by direct verification.
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Presentation level.}
  $Q_N$ is generated by $T_{ij}^{(r)}$ with $r \le N$ satisfying the
  truncated RTT relations and no further; evaluation modules agree with
  the finite RTT stage.  Then $Q_N\cong Y_{\le N}$ by the universal
  property of the truncated RTT presentation.

\item \emph{Pole-order locality.}
  The structure formulas are RTT-triangular: every term in
  $d(T_{ij}^{(r)})$, $\partial(T_{ij}^{(r)})$, the MC twisting, and
  $\Delta^{\mathrm{tw}}(T_{ij}^{(r)})$ contains a factor $T_{ab}^{(s)}$
  with $s \ge r$, and the level-$\ell$ part depends only on modes with
  $s \le \ell$.  Then $F^{N+1}\Ydg_{\cA}$ is preserved and $Q_N$ depends
  only on the first $N$ modes, reducing to~\textup{(i)}.

\item \emph{Rational line-operator formulas.}
  $T(z,u)=1+\hbar\sum_{r \ge 0} T^{(r)}(z)\,u^{-r-1}$; differential
  and translation act coefficientwise; MC correction from rational
  convolution; $\Delta^{\mathrm{tw}}T^{(\ell)}=\sum_{a+b=\ell}
  T^{(a)}\otimes T^{(b)}$.  Implies the locality hypotheses
  of~\textup{(ii)}.

\item \emph{Coefficient-level RTT.}
  The truncated monodromy matrix
  $T_N(u)=1+\hbar\sum_{0\le r\le N}T^{(r)}u^{-r-1}$ satisfies the
  standard truncated RTT relation in~$Q_N$; the $T^{(r)}_{ij}$ generate
  $Q_N$ with no further relations; evaluation modules match.
  Reduces to~\textup{(i)}.

\item \emph{Kernel-coefficient identity.}
  The OPE kernel coefficients
  $K_{a,b}=\operatorname{Coeff}_{u^{-a-1}v^{-b-1}}$ agree with the
  standard finite RTT coefficients and generate all relations.
  Reduces to~\textup{(iv)}.

\item \emph{One-loop kernel identity.}
  Define
  $\mathsf{K}^{\mathrm{line}}_{a,b}(N):=
  \operatorname{Coeff}_{u^{-a-1}v^{-b-1}}
  (R(u\!-\!v)\,T_{N,1}(u)\,T_{N,2}(v)
  -T_{N,2}(v)\,T_{N,1}(u)\,R(u\!-\!v))$
  and let $\mathsf{K}^{\mathrm{RTT}}_{a,b}(N)$ denote the standard
  coefficient.  If the OPE is one-loop exact and
  $\mathsf{K}^{\mathrm{line}}_{a,b}(N)
  =\mathsf{K}^{\mathrm{RTT}}_{a,b}(N)$ for all $a,b \le N$, then
  step~\textup{(v)} is satisfied.

\item \emph{Evaluation detection.}
  Let $\Delta_{a,b}(N):=
  \mathsf{K}^{\mathrm{line}}_{a,b}(N)
  -\mathsf{K}^{\mathrm{RTT}}_{a,b}(N)$.  If $Q_N$ is represented
  faithfully on tensor products of fundamental evaluation modules and
  $\Delta_{a,b}(N)$ acts by zero on that subcategory for all
  $a,b \le N$, then faithfulness forces $\Delta_{a,b}(N)=0$ and
  step~\textup{(vi)} is satisfied.
\end{enumerate}
\end{proposition}

\begin{proof}
Each step is verified by checking that the stated hypotheses imply
those of the preceding step.  Step~\textup{(i)} uses the universal
property; \textup{(ii)} uses RTT-triangularity to preserve ideals;
\textup{(iii)} verifies triangularity from the rational structure;
\textup{(iv)}-\textup{(vi)} match kernel coefficients with RTT
coefficients; \textup{(vii)} uses faithfulness of the evaluation
subcategory.
\end{proof}

\begin{proposition}[Boundary-seed criterion for truncated RTT
defects; \ClaimStatusProvedHere]
\label{prop:dg-shifted-rtt-boundary-seed}
Fix $N \ge 0$ and let $\Delta_{a,b}(N)$ be the coefficient defect of
Proposition~\ref{prop:dg-shifted-rtt-evaluation-detection}.  If the
spectral recursion $\Delta_{a+1,b}(N)=\Delta_{a,b+1}(N)$ holds for
$a+b+1\le N$, the boundary defects $\Delta_{a,0}(N)$ act by zero on the
faithful evaluation family for $0\le a\le N$, and the evaluation action
is standard, then $\Delta_{a,b}(N)=0$ for $a+b\le N$, so
Proposition~\ref{prop:dg-shifted-rtt-evaluation-detection} is satisfied
after checking only $\{\Delta_{a,0}(N)\}_{0 \le a \le N}$.
\end{proposition}

\begin{proof}
The recursion gives $\Delta_{a,b}(N)=\Delta_{a+b,0}(N)$; faithfulness
and boundary vanishing force all defects to vanish.
\end{proof}

\begin{proposition}[Boundary-strip coefficient formula on the
evaluation packet; \ClaimStatusProvedHere]
\label{prop:dg-shifted-rtt-boundary-coefficient-formula}
Let $T_1(u)=\sum_{r\ge 0} T_1^{(r)}u^{-r}$,
$T_2(v)=\sum_{s\ge 0} T_2^{(s)}v^{-s}$ be the normalized fundamental
monodromy series, and expand
$R(u-v)=1-\hbar\sum_{t\ge 0} P_{12}\,u^{-t-1}v^t$ for $|u|>|v|$.
Then
\begin{multline*}
\operatorname{Coeff}_{u^{-a-1}v^{-1}}
\bigl(
R(u\!-\!v)\,T_1(u)\,T_2(v)-T_2(v)\,T_1(u)\,R(u\!-\!v)
\bigr)\\
=\;
T_1^{(a+1)}T_2^{(1)}-T_2^{(1)}T_1^{(a+1)}
-\hbar\!\sum_{t=0}^{a}\!
\bigl(
P_{12}\,T_1^{(a-t)}T_2^{(t+1)}
-T_2^{(t+1)}T_1^{(a-t)}P_{12}
\bigr).
\end{multline*}
\end{proposition}

\begin{proof}
Write $R^{(t)}=-\hbar P_{12}$.  The identity part of $R$ gives the
commutator term; matching $u$- and $v$-powers in the $R$-series forces
$T_1^{(a-t)}$ and $T_2^{(t+1)}$, and summing over $0\le t\le a$ yields
the result.
\end{proof}

\begin{proposition}[Explicit coefficient formula for the fundamental
monodromy series; \ClaimStatusProvedHere]
\label{prop:dg-shifted-rtt-fundamental-coefficient-formula}
Fix generic evaluation points $a_1,\dots,a_m$ and write
\[
T^{(m)}(u)=L_{a_1}(u)\cdots L_{a_m}(u)
=\sum_{r\ge0}T^{(m,r)}u^{-r},
\qquad
L_{a_j}(u)=1-\frac{\hbar P_{0j}}{u-a_j},
\]
on the tensor product
\[
M_m:=V_{\mathrm{fund}}(a_1)\otimes\cdots\otimes
V_{\mathrm{fund}}(a_m).
\]
Then
\[
T^{(m,0)}=1,
\]
and for every $r\ge1$ one has
\[
T^{(m,r)}
=
\sum_{s=1}^{\min(r,m)}(-\hbar)^s
\sum_{1\le j_1<\cdots<j_s\le m}
\sum_{\substack{n_1,\dots,n_s\ge0\\ n_1+\cdots+n_s=r-s}}
a_{j_1}^{n_1}\cdots a_{j_s}^{n_s}\,
P_{0j_1}\cdots P_{0j_s}.
\]
In particular, every summand of $T^{(m,r)}$ is supported on at most
$r$ quantum slots.
\end{proposition}

\begin{proof}
For each factor one has the geometric-series expansion
\[
L_{a_j}(u)
=
1-\hbar\sum_{n\ge0}a_j^n u^{-n-1}P_{0j}
\]
in the standard region $|u|>|a_j|$.  Multiplying the $m$ factors in the
fixed order $j=1,\dots,m$, the coefficient of $u^{-r}$ is obtained by
choosing the non-identity term from exactly $s$ factors
$j_1<\cdots<j_s$ and the identity term from the rest.  If the chosen
terms have degrees $n_1+1,\dots,n_s+1$, then the total $u$-degree
condition is
\[
(n_1+1)+\cdots+(n_s+1)=r,
\]
equivalently $n_1+\cdots+n_s=r-s$.  The contribution of that choice is
\[
(-\hbar)^s a_{j_1}^{n_1}\cdots a_{j_s}^{n_s}
P_{0j_1}\cdots P_{0j_s},
\]
and summing over all subsets and all nonnegative tuples
$(n_1,\dots,n_s)$ gives the displayed formula.

The final sentence is immediate: a summand with $s$ chosen factors
acts nontrivially only on the corresponding $s$ quantum slots, and
$s\le r$.
\end{proof}

\begin{corollary}[Line-side boundary-strip support bound on generic
tensor powers; \ClaimStatusProvedHere]
\label{cor:dg-shifted-rtt-boundary-support-bound}
In the setting of
Proposition~\ref{prop:dg-shifted-rtt-fundamental-coefficient-formula},
let
\[
\mathsf{K}^{\mathrm{line},(m)}_{a,0}
:=
\operatorname{Coeff}_{u^{-a-1}v^{-1}}
\Bigl(
R(u-v)\,T_1(u)\,T_2(v)-T_2(v)\,T_1(u)\,R(u-v)
\Bigr)
\]
be the line-side boundary-strip coefficient on $M_m$, with
$T_i$ written on the $i$th auxiliary copy of~$V$.  Then every summand
of $\mathsf{K}^{\mathrm{line},(m)}_{a,0}$ is supported on at most
$a+1$ quantum slots.

Equivalently, on the standard type-$A$ fundamental evaluation packet
the line-side half of the finite-detection reduction is already
compatible with the slot bound appearing in
Proposition~\ref{prop:dg-shifted-rtt-finite-tensor-detection}.
\end{corollary}

\begin{proof}
By
Proposition~\ref{prop:dg-shifted-rtt-boundary-coefficient-formula},
\[
\mathsf{K}^{\mathrm{line},(m)}_{a,0}
=
T_1^{(a+1)}T_2^{(1)}-T_2^{(1)}T_1^{(a+1)}
-\hbar\sum_{t=0}^{a}
\Bigl(
P_{12}\,T_1^{(a-t)}T_2^{(t+1)}
-T_2^{(t+1)}T_1^{(a-t)}P_{12}
\Bigr).
\]
By
Proposition~\ref{prop:dg-shifted-rtt-fundamental-coefficient-formula},
every summand of $T_1^{(a+1)}$ is supported on a subset
$S\subset\{1,\dots,m\}$ with $|S|\le a+1$, while every summand of
$T_2^{(1)}$ is supported on a single slot $\{k\}$.  If $k\notin S$,
the two summands commute because they act on distinct quantum slots,
so their contribution cancels in the commutator
$T_1^{(a+1)}T_2^{(1)}-T_2^{(1)}T_1^{(a+1)}$.  Thus every surviving
summand of the commutator is supported on at most $|S|\le a+1$ quantum
slots.

For the $t$th summand of the remaining series, every summand of
$T_1^{(a-t)}$ is supported on at most $a-t$ quantum slots and every
summand of $T_2^{(t+1)}$ is supported on at most $t+1$ quantum slots.
Since $P_{12}$ acts only on auxiliary spaces, it does not enlarge the
quantum support.  Therefore every summand of
$P_{12}\,T_1^{(a-t)}T_2^{(t+1)}$ and of
$T_2^{(t+1)}T_1^{(a-t)}P_{12}$ is supported on at most
$(a-t)+(t+1)=a+1$ quantum slots.

Combining the two cases proves the claim.
\end{proof}

\begin{proposition}[Defect-side support mechanism from RTT degree;
\ClaimStatusProvedHere]
\label{prop:dg-shifted-rtt-defect-support-mechanism}
Fix $N \ge 0$, $0\le a\le N$, and a generic tensor product
\[
M_m:=V_{\mathrm{fund}}(a_1)\otimes\cdots\otimes
V_{\mathrm{fund}}(a_m).
\]
Assume:
\begin{enumerate}[label=\textup{(\roman*)}]
\item the action of the truncated monodromy coefficients on $M_m$ is
  induced by the coefficientwise twisted-coproduct multiplication law
  of Proposition~\ref{prop:dg-shifted-rtt-formula-preservation}; and
\item the boundary-strip defect $\Delta_{a,0}(N)$ is homogeneous of RTT
  degree $a+1$.
\end{enumerate}
Then every summand in the action of $\Delta_{a,0}(N)$ on $M_m$ is
supported on at most $a+1$ quantum slots.  Consequently the action of
$\Delta_{a,0}(N)$ on a tensor product of arbitrary length is
determined by its restrictions to the generic tensor subproducts of
length at most $a+1$.

In the standard type-$A$ fundamental packet, this is the defect-side
counterpart of
Corollary~\ref{cor:dg-shifted-rtt-boundary-support-bound}.
\end{proposition}

\begin{proof}
By
Proposition~\ref{prop:dg-shifted-rtt-formula-preservation},
the iterated twisted coproduct on a tensor power is obtained by
coefficientwise multiplication of monodromy matrices.  Therefore every
tensor summand occurring in the action of a homogeneous degree-$(a+1)$
monomial in the coefficients $T^{(r)}$ is indexed by a decomposition
\[
d_1+\cdots+d_s=a+1,
\qquad d_i\ge1,
\]
over the active tensor factors.  Since each active factor contributes
positive RTT degree, one necessarily has $s\le a+1$.  Thus every such
tensor summand is supported on at most $a+1$ quantum slots.

The defect $\Delta_{a,0}(N)$ is a finite linear combination of such
homogeneous monomials by assumption~\textup{(ii)}, so the same support
bound holds for every summand in its action on $M_m$.  The final
sentence is immediate: on a longer tensor product, each summand factors
through one of the generic subproducts supported on at most $a+1$
slots.
\end{proof}

\begin{proposition}[Universal generic packet reduction for the
boundary strip; \ClaimStatusProvedHere]
\label{prop:dg-shifted-rtt-universal-generic-packets}
Fix $m\ge1$ and algebraically independent parameters
$x_1,\dots,x_m$.  Write
\[
K_m:=\mathbb{C}(x_1,\dots,x_m),
\qquad
M_m^{\mathrm{gen}}

V_{\mathrm{fund}}(x_1)\otimes_{K_m}\cdots\otimes_{K_m}

\qquad
L_{x_j}(u)=1-\frac{\hbar P_{0j}}{u-x_j},

and that the truncated quotient acts by the standard evaluation

Then for every $a\le N$, the matrix entries of the operators
\[
\mathsf{K}^{\mathrm{line},(m)}_{a,0},
\qquad
\mathsf{K}^{\mathrm{RTT},(m)}_{a,0},
\qquad
\Delta^{(m)}_{a,0}(N)
:=
\mathsf{K}^{\mathrm{line},(m)}_{a,0}
-\mathsf{K}^{\mathrm{RTT},(m)}_{a,0}

on $M_m^{\mathrm{gen}}$ lie in the polynomial ring
$\mathbb{C}[x_1,\dots,x_m]$.

\[
x_i\mapsto \alpha_i
\qquad (\alpha_i\neq \alpha_j \text{ for } i\neq j),
\]
the specialized operator is exactly the action of the same boundary

\[

V_{\mathrm{fund}}(\alpha_m).

generic packet $M_m^{\mathrm{gen}}$.
\end{proposition}

\begin{proof}
By
Proposition~\ref{prop:dg-shifted-rtt-fundamental-coefficient-formula},
each monodromy coefficient on the line side is a finite sum of terms
\[
(-\hbar)^s x_{j_1}^{n_1}\cdots x_{j_s}^{n_s} P_{0j_1}\cdots P_{0j_s},
\]
so its matrix entries lie in $\mathbb{C}[x_1,\dots,x_m]$.  The
boundary-strip formula of
Proposition~\ref{prop:dg-shifted-rtt-boundary-coefficient-formula}
expresses $\mathsf{K}^{\mathrm{line},(m)}_{a,0}$ as a finite
polynomial combination of those coefficients, hence its matrix entries
also lie in $\mathbb{C}[x_1,\dots,x_m]$.

On the RTT side, the standard evaluation action uses the same
coefficientwise monodromy formulas at finite stage, so the matrix
entries of $\mathsf{K}^{\mathrm{RTT},(m)}_{a,0}$ are polynomial in the
same parameters.  Therefore the same is true for the defect
$\Delta^{(m)}_{a,0}(N)$.

Specializing $x_i\mapsto\alpha_i$ simply evaluates those polynomial
matrix entries at the chosen distinct parameters, which is exactly the
generic tensor product
$V_{\mathrm{fund}}(\alpha_1)\otimes\cdots\otimes
V_{\mathrm{fund}}(\alpha_m)$.  If the operator on
$M_m^{\mathrm{gen}}$ is zero, then all of its polynomial matrix entries
are zero, so every generic specialization is zero as well.  The
converse follows because a polynomial matrix entry that vanishes on
every point of the open set $\{\alpha_i\neq\alpha_j\}$ must vanish
identically.
\end{proof}

\begin{corollary}[Minimal canonical family from the boundary-strip
induction; \ClaimStatusProvedHere]
\label{cor:dg-shifted-rtt-minimal-canonical-family}
Fix $0\le a\le N$.  Under the hypotheses of
Proposition~\ref{prop:dg-shifted-rtt-defect-support-mechanism} and
Proposition~\ref{prop:dg-shifted-rtt-universal-generic-packets}, to
prove
\[
\Delta_{a,0}(N)=0
\]
it is enough to verify vanishing on the canonical universal packets
\[
M_m^{\mathrm{gen}}
\qquad (1\le m\le a+1).
\]
Conversely, any all-rank induction proving $\Delta_{a,0}(N)=0$ must in
particular kill these packets.  Hence this family is the minimal
canonical subproduct family furnished by the present boundary-strip
reduction.
\end{corollary}

\begin{proof}
By
Proposition~\ref{prop:dg-shifted-rtt-defect-support-mechanism},
the action of $\Delta_{a,0}(N)$ on an arbitrary generic tensor product
is determined by its restrictions to the subproducts of lengths at most
$a+1$.  By
Proposition~\ref{prop:dg-shifted-rtt-universal-generic-packets}, each
fixed tensor length contributes only one universal generic packet up to
specialization.  Therefore vanishing on the displayed family forces
vanishing on every generic tensor product relevant to the
finite-detection argument.

The converse is tautological: if $\Delta_{a,0}(N)$ vanishes on every
generic tensor product, then it vanishes on these particular universal
ones.  This is the smallest family produced by combining the
length-$(a+1)$ support reduction with the collapse of each fixed tensor
length to one universal generic packet.
\end{proof}

  \]
\item vanishing of a coefficient defect on those generic tensor products

\item for every $0 \le a \le N$, the defect $\Delta_{a,0}(N)$ acts by

  M_m^{\mathrm{gen}}

  \qquad (1\le m\le a+1).

Corollary~\ref{cor:dg-shifted-rtt-minimal-canonical-family},
assumptions~\textup{(i)}, \textup{(ii)}, \textup{(iii)},
and~\textup{(vi)} imply that all generic tensor-product actions of

length.  Assumptions~\textup{(iv)} and~\textup{(v)} then propagate

Proposition~\ref{prop:dg-shifted-rtt-evaluation-detection}.  That

\[
\Delta_{a,0}(N)=0.
\]

Doing this for all $a\le N$ kills the entire boundary strip, so
Proposition~\ref{prop:dg-shifted-rtt-boundary-seed} reduces the full

packets.
\end{proof}

\begin{proposition}[Top-packet induction step on the universal family;
\ClaimStatusProvedHere]
\label{prop:dg-shifted-rtt-top-packet-induction}
Set $m:=a+1$ and let
\[
M_m^{\mathrm{gen}}
=
V_{\mathrm{fund}}(x_1)\otimes_{\mathbb{C}(x_1,\dots,x_m)}\cdots
\otimes_{\mathbb{C}(x_1,\dots,x_m)}V_{\mathrm{fund}}(x_m)
\]
be the top universal packet for the boundary index~$a$.  Let
\[
\mathsf{F}_{<m}\End(V^{\otimes 2}\otimes M_m^{\mathrm{gen}})
\]
be the linear span of operators supported on proper subsets of the
quantum slots $\{1,\dots,m\}$.  Assume the hypotheses of
Proposition~\ref{prop:dg-shifted-rtt-defect-support-mechanism} and that
\[
\Delta_{a,0}(N)
\]
acts by zero on the universal packets $M_r^{\mathrm{gen}}$ for every
$1\le r\le a$.  Then
\[
\Delta^{(m)}_{a,0}(N)=0
\quad\Longleftrightarrow\quad
\bigl[\Delta^{(m)}_{a,0}(N)\bigr]=0
\ \text{in}\ 
\End(V^{\otimes 2}\otimes M_m^{\mathrm{gen}})/\mathsf{F}_{<m}.
\]
Equivalently, once the shorter universal packets are killed, the
induction from packet lengths $\le a$ to the top packet $m=a+1$
closes exactly on the top-support class.
\end{proposition}

\begin{proof}
By
Proposition~\ref{prop:dg-shifted-rtt-defect-support-mechanism}, every
summand in the action of $\Delta_{a,0}(N)$ on $M_m^{\mathrm{gen}}$ is
supported on at most $m=a+1$ quantum slots.  Therefore modulo
$\mathsf{F}_{<m}$ only the full-support part can survive.

If a summand is supported on a proper subset
$S\subsetneq\{1,\dots,m\}$, then it factors through the corresponding
universal packet $M_{|S|}^{\mathrm{gen}}$ with $|S|\le a$, so it
vanishes by hypothesis.  Hence
$\Delta^{(m)}_{a,0}(N)$ vanishes if and only if its class in the
quotient by $\mathsf{F}_{<m}$ vanishes.
\end{proof}

\begin{proposition}[Closed-form line-side top-support class on the top
packet; \ClaimStatusProvedHere]
\label{prop:dg-shifted-rtt-top-packet-line-formula}
In the setting of
Proposition~\ref{prop:dg-shifted-rtt-top-packet-induction}, relabel the
two auxiliary copies by $\alpha,\beta$ and the $m=a+1$ quantum slots by
$1,\dots,m$.  For an ordered subset
$S=\{s_1<\cdots<s_r\}\subseteq [m]$, write
\[
P_{\alpha,S}:=P_{\alpha s_1}\cdots P_{\alpha s_r},
\qquad
P_{\beta,S}:=P_{\beta s_1}\cdots P_{\beta s_r},
\]
and let $S^c=[m]\setminus S$.  Then the image of the line-side boundary
coefficient
\[
\mathsf{K}^{\mathrm{line},(m)}_{a,0}
\]
in the quotient
\[
\End(V^{\otimes 2}\otimes M_m^{\mathrm{gen}})/\mathsf{F}_{<m}
\]
is represented by the parameter-free operator
\[
\Omega_a^{\mathrm{line}}
:=
(-\hbar)^{a+2}
\left[
\sum_{j=1}^{m}
\bigl(
P_{\alpha,[m]}P_{\beta,j}-P_{\beta,j}P_{\alpha,[m]}
\bigr)
+\sum_{S\subsetneq [m]}
\bigl(
P_{\alpha\beta}P_{\alpha,S}P_{\beta,S^c}
-P_{\beta,S^c}P_{\alpha,S}P_{\alpha\beta}
\bigr)
\right].
\]
In particular, the line-side top-support class on the top packet is
independent of the universal evaluation parameters $x_1,\dots,x_m$.
\end{proposition}

\begin{proof}
Work modulo $\mathsf{F}_{<m}$, so only full-support terms on the
quantum slots $[m]$ survive.  By
Proposition~\ref{prop:dg-shifted-rtt-fundamental-coefficient-formula},
the full-support part of $T_\alpha^{(m)}$ is the unique term obtained
by choosing the non-identity contribution from every factor and taking
zero extra degree in each one:
\[
T_\alpha^{(m)}\equiv (-\hbar)^m P_{\alpha,[m]}
\qquad \mod \mathsf{F}_{<m}.
\]
Likewise
\[
T_\beta^{(1)}=-\hbar\sum_{j=1}^{m}P_{\beta,j}.
\]
Substituting these into the commutator part of
Proposition~\ref{prop:dg-shifted-rtt-boundary-coefficient-formula}
gives the first displayed sum.

For the $t$th term of the remaining series, a contribution survives
modulo $\mathsf{F}_{<m}$ only when the support of $T_\alpha^{(a-t)}$
and the support of $T_\beta^{(t+1)}$ are complementary subsets of
$[m]$.  Since $|[m]|=a+1$, this forces the first support to have size
$a-t$ and the second support to have size $t+1$, with zero extra
degree on both sides.  Therefore the only surviving terms are exactly
\[
(-\hbar)^{a-t}P_{\alpha,S},
\qquad
(-\hbar)^{t+1}P_{\beta,S^c},
\]
for proper subsets $S\subsetneq [m]$, and substituting them into the
boundary-strip coefficient formula yields the second displayed sum with
overall factor $(-\hbar)^{a+2}$.

All other terms are supported on proper subsets and therefore die in
the quotient.  The final sentence follows because the surviving
representative involves only permutation operators and~$\hbar$.
\end{proof}

\begin{corollary}[Abstract top-packet comparison;
\ClaimStatusProvedHere]
\label{cor:dg-shifted-rtt-top-packet-comparison}
Assume the hypotheses of
Proposition~\ref{prop:dg-shifted-rtt-top-packet-induction}.  Let

This is the exact remaining all-rank comparison on the universal

\begin{proof}
The displayed identity is just the definition of the defect class in

\end{proof}

R(u-v)\,T_{N,1}^{\mathrm{std}}(u)\,T_{N,2}^{\mathrm{std}}(v)

\begin{proposition}[Fundamental propagation criterion for boundary-strip
RTT defects; \ClaimStatusProvedHere]
\label{prop:dg-shifted-rtt-fundamental-propagation}
Fix $N \ge 0$, and for generic evaluation points
$a_1,\dots,a_m$ write
\[
M_m:=V_{\mathrm{fund}}(a_1)\otimes\cdots\otimes V_{\mathrm{fund}}(a_m).
\]
Let
\[
\mathcal{D}_N(u,v):=
R(u-v)\,T_{N,1}(u)\,T_{N,2}(v)
-T_{N,2}(v)\,T_{N,1}(u)\,R(u-v)
\]
be the truncated RTT defect operator whose coefficients are the
boundary-strip defects of
Proposition~\ref{prop:dg-shifted-rtt-boundary-seed}.  Assume:
\begin{enumerate}[label=\textup{(\roman*)}]
\item on every generic tensor power $M_m$, the truncated monodromy
  matrix is obtained by iterating the twisted coproduct, so that
  \[
  T_N^{(m+1)}(u)=T_N^{(m)}(u)\,L_{m+1}(u)
  \]
  with $L_{m+1}(u)$ the one-factor fundamental monodromy matrix acting
  on the last tensor factor, and operators supported on disjoint
  quantum slots commute;
\item for every generic fundamental evaluation module
  $V_{\mathrm{fund}}(a)$, the one-factor monodromy matrix $L(u)$
  satisfies the standard truncated RTT relation, equivalently
  \[
  \mathcal{D}_N(u,v)\big|_{V_{\mathrm{fund}}(a)}=0;
  \]
\item the faithful evaluation subcategory of
  Proposition~\ref{prop:dg-shifted-rtt-evaluation-detection} is
  generated from the generic tensor powers $M_m$ by finite direct sums
  and subquotients; and
\item the standard evaluation action of $T_N(u)$ is compatible with the
  same twisted-coproduct multiplication law.
\end{enumerate}
Then $\mathcal{D}_N(u,v)$ acts by zero on every object of the faithful
evaluation subcategory.  Consequently
\[
\Delta_{a,0}(N)=0
\qquad (0\le a\le N),
\]
so
Propositions~\ref{prop:dg-shifted-rtt-boundary-seed}
and~\ref{prop:dg-shifted-rtt-evaluation-detection}
identify the quotient with $Y_{\le N}$ once the degree-$2$ twisting
morphism matches $\tau|_{\deg 2}$.
\end{proposition}

\begin{proof}
We first show by induction on $m$ that
\[
\mathcal{D}_N(u,v)\big|_{M_m}=0
\qquad\text{for every } m\ge 1.
\]
For $m=1$ this is assumption~\textup{(ii)}.

Assume the statement for $M_m$.  By assumption~\textup{(i)}, on
$M_{m+1}=M_m\otimes V_{\mathrm{fund}}(a_{m+1})$ one has
\[
T_N^{(m+1)}(u)=T_N^{(m)}(u)\,L_{m+1}(u),
\qquad
T_N^{(m+1)}(v)=T_N^{(m)}(v)\,L_{m+1}(v).
\]
Since the last tensor factor commutes with the earlier quantum slots,
we compute
\begin{align*}
R\,T^{(m+1)}_{N,1}(u)\,T^{(m+1)}_{N,2}(v)
&=
R\,T^{(m)}_{N,1}(u)\,T^{(m)}_{N,2}(v)\,
L_{m+1,1}(u)\,L_{m+1,2}(v) \\
&=
T^{(m)}_{N,2}(v)\,T^{(m)}_{N,1}(u)\,
R\,L_{m+1,1}(u)\,L_{m+1,2}(v) \\
&=
T^{(m)}_{N,2}(v)\,T^{(m)}_{N,1}(u)\,
L_{m+1,2}(v)\,L_{m+1,1}(u)\,R \\
&=
T^{(m+1)}_{N,2}(v)\,T^{(m+1)}_{N,1}(u)\,R.
\end{align*}
The second equality uses the induction hypothesis on $M_m$, and the
third uses the one-factor RTT relation from assumption~\textup{(ii)} on
the last tensor factor.  Hence the defect vanishes on $M_{m+1}$.

By assumption~\textup{(iii)}, every object of the faithful evaluation
subcategory is built from the generic tensor powers $M_m$ by finite
direct sums and subquotients.  Vanishing of the defect operator is
preserved under those operations, so $\mathcal{D}_N(u,v)$ acts by zero
on the whole faithful evaluation subcategory.  Assumption~\textup{(iv)}
supplies the standard evaluation compatibility required by
Proposition~\ref{prop:dg-shifted-rtt-evaluation-detection}, which then
forces all boundary-strip coefficients $\Delta_{a,0}(N)$ to vanish.
\end{proof}

\begin{proposition}[Auxiliary-kernel criterion for fundamental RTT
propagation; \ClaimStatusProvedHere]
\label{prop:dg-shifted-rtt-auxiliary-kernel-criterion}
Fix $N \ge 0$ and let $R(u)$ be the rational Yang $R$-matrix on
$V_{\mathrm{fund}}\otimes V_{\mathrm{fund}}$, satisfying the
Yang--Baxter equation
\[
R_{12}(u-v)\,R_{13}(u-a)\,R_{23}(v-a)
=
R_{23}(v-a)\,R_{13}(u-a)\,R_{12}(u-v).
\]
Assume:
\begin{enumerate}[label=\textup{(\roman*)}]
\item for every generic evaluation point $a$, the one-factor
  fundamental monodromy matrix induced by the line operator is the
  truncated auxiliary kernel
  \[
  L_a(u)=R_{0a}(u-a)
  \]
  on the fundamental evaluation module
  $V_{\mathrm{fund}}(a)$;
\item on generic tensor products of fundamental evaluation modules, the
  truncated monodromy matrix is obtained by the same twisted-coproduct
  multiplication law used in
  Proposition~\ref{prop:dg-shifted-rtt-fundamental-propagation}.
\end{enumerate}
Then the fundamental one-factor matrices satisfy the truncated RTT
relation
\[
R_{12}(u-v)\,L_{1,a}(u)\,L_{2,a}(v)
=
L_{2,a}(v)\,L_{1,a}(u)\,R_{12}(u-v)
\]
on every fundamental evaluation module.  Consequently the hypotheses of
Proposition~\ref{prop:dg-shifted-rtt-fundamental-propagation} are
satisfied, so the Yangian finite-detection frontier reduces to the
single pairwise kernel identity
$L_a(u)=R_{0a}(u-a)$ on the fundamental line operator, together with
the standard twisted-coproduct multiplication law.
\end{proposition}

\begin{proof}
By assumption~\textup{(i)},
\[
L_{1,a}(u)=R_{1a}(u-a),
\qquad
L_{2,a}(v)=R_{2a}(v-a),
\]
where indices $1,2$ denote the two auxiliary copies of the fundamental
representation and $a$ denotes the quantum space
$V_{\mathrm{fund}}(a)$.  Substituting these expressions into the RTT
relation gives
\[
R_{12}(u-v)\,R_{1a}(u-a)\,R_{2a}(v-a)
=
R_{2a}(v-a)\,R_{1a}(u-a)\,R_{12}(u-v),
\]
which is exactly the Yang--Baxter equation for the rational Yang
$R$-matrix.  Therefore the one-factor fundamental monodromy satisfies
the RTT relation.

Assumptions~\textup{(i)} and~\textup{(ii)} are exactly the hypotheses of
Proposition~\ref{prop:dg-shifted-rtt-universal-evaluation-rigidity}, so
every universal packet carries the standard finite-stage evaluation
action.  In particular, the standard evaluation compatibility required
in Proposition~\ref{prop:dg-shifted-rtt-fundamental-propagation} is
automatic.  Hence that proposition applies and kills the entire
boundary strip.
\end{proof}

\begin{proposition}[Type-A uniqueness of the auxiliary kernel on the
fundamental line; \ClaimStatusProvedHere]
\label{prop:dg-shifted-rtt-typea-auxiliary-uniqueness}
Let $V=\mathbb{C}^M$ be the defining representation of
$\mathfrak{sl}_M$, and let
\[
L_a(u)\in \End(V)\otimes\End(V(a))
\]
be a rational one-factor monodromy matrix on the fundamental
evaluation module $V(a)$.  Assume:
\begin{enumerate}[label=\textup{(\roman*)}]
\item $L_a(u)$ is $\mathfrak{sl}_M$-equivariant on
  $V\otimes V(a)$;
\item $L_a(u)$ is regular away from $u=a$ and has unit asymptotic
  \[
  L_a(u)\to 1
  \qquad (u\to\infty);
  \]
\item the only pole is simple at $u=a$, with residue
  \[
  \operatorname*{Res}_{u=a}L_a(u)=-\hbar P,
  \]
  where $P$ is the permutation operator on
  $V\otimes V(a)\cong V\otimes V$.
\end{enumerate}
Then
\[
L_a(u)=1-\frac{\hbar P}{u-a}=R_{0a}(u-a),
\]
the standard rational Yang $R$-matrix on the defining fundamental
representation.  Consequently, for the standard type-A RTT tower the
pairwise frontier of
Proposition~\ref{prop:dg-shifted-rtt-auxiliary-kernel-criterion}
reduces to these three local checks.
\end{proposition}

\begin{proof}
Because
\[
V\otimes V \cong \operatorname{Sym}^2(V)\oplus \Lambda^2(V),
\]
and both summands are irreducible $\mathfrak{sl}_M$-modules, every
$\mathfrak{sl}_M$-equivariant endomorphism of $V\otimes V$ acts by a
scalar on each summand.  Equivalently,
\[
L_a(u)=\alpha(u)\,1+\beta(u)\,P
\]
for rational scalar functions $\alpha(u),\beta(u)$.

The unit asymptotic implies
\[
\alpha(u)\to 1,\qquad \beta(u)\to 0
\qquad (u\to\infty).
\]
The residue condition says that the only pole is simple at $u=a$ and
its residue is exactly $-\hbar P$.  Hence $\alpha(u)$ has no pole and
must therefore be the constant function $1$, while $\beta(u)$ has the
unique form
\[
\beta(u)=-\frac{\hbar}{u-a}.
\]
Substituting these two functions gives
\[
L_a(u)=1-\frac{\hbar P}{u-a},
\]
which is precisely the standard rational Yang $R$-matrix kernel
$R_{0a}(u-a)$.
\end{proof}

\begin{proposition}[Type-A residue detection on the symmetric and
antisymmetric channels; \ClaimStatusProvedHere]
\label{prop:dg-shifted-rtt-typea-residue-channels}
In the setting of
Proposition~\ref{prop:dg-shifted-rtt-typea-auxiliary-uniqueness}, let
\[
\Xi_a:=\operatorname*{Res}_{u=a}L_a(u)\in \End(V\otimes V(a)).
\]
Then the residue identity
\[
\Xi_a=-\hbar P
\]
holds if and only if $\Xi_a$ acts by the scalar $-\hbar$ on the
symmetric summand $\operatorname{Sym}^2(V)$ and by the scalar $+\hbar$ on the
antisymmetric summand $\Lambda^2(V)$.

Equivalently, after identifying $V(a)\cong V$, it is enough to check
the two highest-weight channel formulas
\[
\Xi_a(e_1\otimes e_1)=-\hbar(e_1\otimes e_1),
\qquad
\Xi_a(e_1\otimes e_2-e_2\otimes e_1)
=+\hbar(e_1\otimes e_2-e_2\otimes e_1).
\]
Consequently, for the standard type-A RTT tower the remaining analytic
Yangian frontier on the fundamental line is a two-channel scalar
residue computation.
\end{proposition}

\begin{proof}
By the $\mathfrak{sl}_M$-equivariance hypothesis of
Proposition~\ref{prop:dg-shifted-rtt-typea-auxiliary-uniqueness}, the
operator $\Xi_a$ preserves the irreducible decomposition
\[
V\otimes V \cong \operatorname{Sym}^2(V)\oplus \Lambda^2(V).
\]
Schur's lemma therefore gives scalars $\lambda_{\mathrm{sym}}$ and
$\lambda_{\mathrm{alt}}$ such that
\[
\Xi_a\big|_{\operatorname{Sym}^2(V)}=\lambda_{\mathrm{sym}}\id,
\qquad
\Xi_a\big|_{\Lambda^2(V)}=\lambda_{\mathrm{alt}}\id.
\]
The permutation operator acts by $+1$ on $\operatorname{Sym}^2(V)$ and by $-1$ on
$\Lambda^2(V)$.  Hence $\Xi_a=-\hbar P$ is equivalent to the two scalar
equalities
\[
\lambda_{\mathrm{sym}}=-\hbar,
\qquad
\lambda_{\mathrm{alt}}=+\hbar.
\]
The displayed highest-weight vectors generate the two irreducible
summands, so checking the action there is equivalent to checking the two
channel scalars themselves.
\end{proof}

\begin{corollary}[Type-A residue detection on one mixed tensor line;
\ClaimStatusProvedHere]
\label{cor:dg-shifted-rtt-typea-single-line}
Assume the hypotheses of
Proposition~\ref{prop:dg-shifted-rtt-typea-residue-channels}, and
assume $M\ge 2$.  Then the residue identity
\[
\Xi_a=-\hbar P
\]
is equivalent to the single mixed-tensor formula
\[
\Xi_a(e_1\otimes e_2)=-\hbar(e_2\otimes e_1).
\]

Equivalently, for the standard type-A RTT tower the remaining analytic
Yangian frontier on the fundamental line may be checked on one ordered
off-diagonal tensor line.
\end{corollary}

\begin{proof}
Write
\[
s:=e_1\otimes e_2+e_2\otimes e_1\in \operatorname{Sym}^2(V),
\qquad
a:=e_1\otimes e_2-e_2\otimes e_1\in \Lambda^2(V),
\]
so that
\[
e_1\otimes e_2=\frac{1}{2}(s+a).
\]
If $\Xi_a=-\hbar P$, then the displayed formula is immediate.

Conversely, by
Proposition~\ref{prop:dg-shifted-rtt-typea-residue-channels}, the
operator $\Xi_a$ acts by scalars
$\lambda_{\mathrm{sym}},\lambda_{\mathrm{alt}}$ on
$\operatorname{Sym}^2(V)$ and $\Lambda^2(V)$, so
\[
\Xi_a(e_1\otimes e_2)
=\frac{1}{2}\lambda_{\mathrm{sym}}\,s
+\frac{1}{2}\lambda_{\mathrm{alt}}\,a
=\frac{\lambda_{\mathrm{sym}}+\lambda_{\mathrm{alt}}}{2}\,
e_1\otimes e_2
+\frac{\lambda_{\mathrm{sym}}-\lambda_{\mathrm{alt}}}{2}\,
e_2\otimes e_1.
\]
If this equals $-\hbar(e_2\otimes e_1)$, then comparing the
coefficients of the linearly independent tensors
$e_1\otimes e_2$ and $e_2\otimes e_1$ gives
\[
\lambda_{\mathrm{sym}}+\lambda_{\mathrm{alt}}=0,
\qquad
\lambda_{\mathrm{sym}}-\lambda_{\mathrm{alt}}=-2\hbar.
\]
Hence $\lambda_{\mathrm{sym}}=-\hbar$ and
$\lambda_{\mathrm{alt}}=+\hbar$, so
Proposition~\ref{prop:dg-shifted-rtt-typea-residue-channels} yields
$\Xi_a=-\hbar P$.
\end{proof}

\begin{proposition}[Uniform residue extraction from one ordered tensor
line; \ClaimStatusProvedHere]
\label{prop:dg-shifted-rtt-typea-uniform-single-line}
Assume the hypotheses of
Corollary~\ref{cor:dg-shifted-rtt-typea-residue-reduction}, and assume

\begin{proof}

=

\end{proof}
