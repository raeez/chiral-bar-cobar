\chapter{The first rank-2 holographic datum: \texorpdfstring{$\mathcal H(\Walg_3)$}{H(W3)}}
\label{ch:w3-holographic-datum}

Heisenberg is the rank-$1$ Gaussian archetype, affine
$\widehat{\mathfrak{sl}}_2$ is the rank-$1$ Lie/tree archetype,
and Virasoro is the rank-$1$ mixed class~M prototype. Each is
single-generator or scalar. The first standard-landscape algebra with
genuine multi-channel structure is the principal algebra
$\Walg_3^k$. Its holographic modular Koszul datum
\[
\mathcal H(\Walg_3)=
(\cA,\cA^i,\cA^!,\cC,r_\cA(z),\Theta_\cA,\nabla^{\mathrm{hol}})
\]
has a computed algebraic surface on the principal Drinfeld--Sokolov
locus. The bar-dual coalgebra $\cA^i$, the Verdier/Koszul branch
$\cA^!$, the chiral derived-centre bulk
$\cC=Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)$, and the line/evaluation
comparison category remain separate throughout. The exact-rational module
\texttt{compute/lib/theorem\_w3\_holographic\_datum\_engine.py}
implements the scalar projection used below; it is not a proof of an
equivalence among these seven entries.

\medskip
\noindent\emph{Universal normalisations.}
The invariants $\kappa^{W_3}(\Walg_3) = 5c/6$ and
$K_3 = c + c' = 100$ are evaluations of two universal constructions.
The universal conductor functor
$K(\cA) = -c_{\mathrm{ghost}}(\BRST(\cA))$
yields $\kappa^{W_3}(\Walg_3) = c\,(H_3 - 1) = 5c/6$ via the
principal-DS BRST resolution, with $H_3 = 1 + 1/2 + 1/3 = 11/6$
(Corollary~\ref{cor:uc-K-WN} at $N = 3$,
Chapter~\ref{chap:universal-conductor}); the quadrichotomy classification
places $\Walg_3$ in class~M with shadow depth
$r_{\max}^{W_3} = \infty$, on the second lane (the $W$-line) of
which the closed-form coefficient sequence
$a_n = 2 \cdot 3^{n-2}\,(2n-4)!/[(n-2)!\,n!]$ produces the
doubled exponential growth rate
$C_{\Walg_3}^{W\text{-line}} = 12 = 2 \cdot 6$
(Theorem~\ref{thm:w3-wline-closed-form},
Theorem~\ref{thm:w3-wline-exponential-base},
Chapter~\ref{chap:shadow-quadrichotomy-platonic}); these two
universal results, plus the universal-class-M $T$-line value
$C_{\mathcal{A}} = 6$
(Theorem~\ref{thm:universal-class-M-C-is-6}), pin the conductor,
modular-characteristic, and single-line shadow entries used below.
The genus-$2$ cross-channel correction and the propagator variance
also require the finite $\Walg_3$ OPE data and the mixed-channel graph
sum. None of these scalar computations identifies the holographic
lift, the line/evaluation comparison category, or the derived-centre
bulk.
\medskip

\medskip\noindent\textbf{Rank-2 obstruction.}
Four structural features make $\Walg_3$ the natural rank-2 base case:
(i)~$\Walg_3$ is rank~$2$ with two generators of distinct conformal
weight ($T$ at weight~$2$, $W$ at weight~$3$), so the multi-weight
phenomena of Theorem~\ref{thm:multi-weight-genus-expansion} appear
nontrivially;
(ii)~the $W$--$W$ OPE is genuinely non-quadratic, involving the
composite $\Lambda = {:}TT{:} - \tfrac{3}{10}\partial^2 T$ with a
$c$-dependent coefficient $\alpha(c) = 16/(22+5c)$;
(iii)~the Verdier/Koszul branch is the nontrivial companion
$c \mapsto 100 - c$, producing the nonzero complementarity constant
$250/3$ rather than the affine-style vanishing sum;
(iv)~the genus-2 cross-channel correction
$\delta F_2(\Walg_3) = (c+204)/(16c)$ is the smallest manuscript
example of a nonvanishing multi-weight cross-channel free energy, and
the finite genus-$3$ window gives
$\delta F_3^{\mathrm{cross}}(\Walg_3)
= (5c^3+3792c^2+1{,}149{,}120c+217{,}071{,}360)/(138{,}240c^2)$,
which witnesses the negative resolution of
\hyperref[op:multi-generator-universality]{Open Problem
on multi-generator universality}.

\medskip\noindent\textbf{Conventions.}
Throughout this chapter, $c$ denotes the central charge of $\Walg_3$
at generic level $k \neq -3$. We use OPE-mode notation $a_{(n)}b$
for singular products, the $\lambda$-bracket conversion
$\{a_\lambda b\} = \sum_n \lambda^{(n)} a_{(n)} b$ with divided
powers $\lambda^{(n)} = \lambda^n/n!$,
and the cohomological grading
convention $|d| = +1$. The bar propagator is the weight-1 form
$d\log E(z,w)$ in \emph{both} variables, regardless of the conformal
weights of the fields it sews
(Remark~\ref{rem:propagator-weight-universality}).
The modular characteristic $\kappa$ is
the genus-1 obstruction coefficient $\mathrm{obs}_1 = \kappa
\cdot \lambda_1$, distinct from both the central charge $c$
and from $\kappa_{\mathrm{eff}}$. Every formula below containing
$(5c+22)^{-1}$ is taken on the non-singular Zamolodchikov surface
$c(5c+22)\neq0$.

\section{The seven entries of \texorpdfstring{$\mathcal H(\Walg_3)$}{H(W3)}}
\label{sec:w3hol-seven-entries}

The holographic modular Koszul datum
\[
\mathcal H(\cA)=
(\cA,\cA^i,\cA^!,\cC,r_\cA(z),\Theta_\cA,\nabla^{\mathrm{hol}})
\]
has a proved algebraic core and a conditional holographic lift
(Definition~\ref{def:holographic-modular-koszul-datum}). For
$\cA=\Walg_3^k$ the seven entries are the following.

\subsection{The algebra \texorpdfstring{$\cA = \Walg_3$}{A = W3}}
\label{subsec:w3hol-A}

The principal $\Walg$-algebra $\Walg_3^k =
\Walg^k(\mathfrak{sl}_3, f_{\mathrm{prin}})$ is the Drinfeld--Sokolov
reduction of $\widehat{\mathfrak{sl}}_3$ at the principal nilpotent.
It is freely strongly generated by the stress-energy tensor $T$
of conformal weight~$2$ and a primary field $W$ of conformal
weight~$3$; the central charge as a function of the level is
\begin{equation}
\label{eq:w3hol-central-charge}
c(\Walg_3^k) = 2 - \frac{24(k+2)^2}{k+3}
\end{equation}
(Drinfeld--Sokolov formula; see Chapter~\ref{chap:w-algebra-koszul},
Proposition~\ref{prop:w3-central-charge}).
The $T$--$T$ OPE reproduces the Virasoro algebra, the $T$--$W$ OPE
makes $W$ a weight-3 primary, and the $W$--$W$ OPE involves the
nonlinear composite
$\Lambda = {:}TT{:} - \tfrac{3}{10}\partial^2 T$ with $c$-dependent
coefficient $\alpha(c) = 16/(22+5c)$
(Theorem~\ref{thm:lambda-coefficients-derivation}).

\subsection{The bar-dual coalgebra \texorpdfstring{$\cA^i$}{A i}}
\label{subsec:w3hol-A-i}

The intrinsic dual object produced by the chiral bar construction is
\begin{equation}\label{eq:w3hol-ai}
\cA^i := H^\ast\bigl(B^{\mathrm{ch}}(\Walg_3^k)\bigr).
\end{equation}
It is a coalgebra. It is not the Verdier/Koszul algebra and not the
line category. The bar-cobar counit
$\Omega^{\mathrm{ch}}B^{\mathrm{ch}}(\cA)\to\cA$ is inversion on the
Koszul locus; it is not the construction of~$\cA^!$.

\subsection{The Verdier/Koszul branch \texorpdfstring{$\cA^!$}{A!}}
\label{subsec:w3hol-A-dual}

For $k'=-k-6$, the principal companion algebra $\Walg_3^{k'}$ has
central charge $100-c$
(Theorem~\ref{thm:central-charge-complementarity}). The proved
Koszul statement is characteristic-level:
$\kappa((\Walg_3^k)^!)=\kappa(\Walg_3^{k'})$
(Theorem~\ref{thm:w3-koszul-dual}). The stronger identification
$\cA^!\simeq \Walg_3^{k'}$ is the Verdier/Koszul rectification of
$\cA^i$ and is conditional on DS/bar transport; on that conditional
lane the OPE skeleton is the same as for~$\cA$ with $c$ replaced by
$100-c$.

\subsection{The derived centre and the line/evaluation model}
\label{subsec:w3hol-C}

The closed-sector slot is pinned to the chiral derived centre
\begin{equation}\label{eq:w3hol-derived-centre}
\cC :=
Z^{\mathrm{der}}_{\mathrm{ch}}(\Walg_3^k).
\end{equation}
It is represented by the chiral Hochschild cochain complex
$C^\bullet_{\mathrm{ch}}(\Walg_3^k,\Walg_3^k)$. This object is not
the bar-dual coalgebra, not the Verdier/Koszul branch, and not the
line category. Theorem~H is used here only in its generic Koszul-locus
form: it constrains this chiral Hochschild object to the
cohomological degrees \(\{0,1,2\}\). It is not topological
Hochschild theory, categorical Hochschild theory, or a physical
three-dimensional bulk without the typed open/closed comparison maps.
On the evaluation-generated core of the line side the
categorical Chari--Ginzburg construction gives
\begin{equation}\label{eq:w3hol-line-evaluation-category}
\cC_{\mathrm{line}}^{\mathrm{ev}}
\simeq \mathrm{Rep}_q(\mathfrak{sl}_3),
\qquad q=\exp(\pi i/(k+3)),
\end{equation}
(Theorem~\ref{thm:categorical-cg-all-types},
Corollary~\ref{cor:mc3-all-types}). Thick generation by
evaluation modules is unconditional for all simple types
(Corollary~\ref{cor:mc3-all-types}); the residual problem of
extending beyond evaluation modules is the downstream DK-4/5 question.
No equivalence
$\cC_{\mathrm{line}}^{\mathrm{ev}}\simeq \cC$ is asserted here.
The only use below is the boundary-linear comparison surface on
which the evaluation core supplies a model for line operators, while
$Z^{\mathrm{der}}_{\mathrm{ch}}(\Walg_3^k)$ remains the bulk entry.
The physical open/closed comparison, when present, is a pair of typed
maps
\begin{equation}\label{eq:w3hol-typed-comparison-maps}
A_{\mathrm{bulk}}^{\mathrm{HT}}
\xrightarrow{\;\beta_{\mathrm{der}}\;}
Z^{\mathrm{der}}_{\mathrm{ch}}(\Walg_3^k),
\qquad
\cC_{\mathrm{line}}^{\mathrm{ev}}
\xrightarrow{\;\iota_{\mathrm{ev}}\;}
\cC_{\mathrm{line}},
\end{equation}
plus compatibility of the induced bulk-to-boundary action with the
chiral Hochschild brace algebra. The map
\(\iota_{\mathrm{ev}}\) is the evaluation-core inclusion supplied by
Chari--Ginzburg. The map \(\beta_{\mathrm{der}}\) is a hypothesis of
the holographic lift; it is not produced by bar--cobar inversion and
not by the scalar identities in this chapter.

\subsection{The collision residue \texorpdfstring{$r(z)$}{r(z)}}
\label{subsec:w3hol-r}

The spectral $r$-matrix is the binary genus-0 collision residue of
the bar-intrinsic MC element, written in Verdier/Koszul coordinates:
\begin{equation}\label{eq:w3hol-r-collision}
r_{\Walg_3}(z) :=
\operatorname{Res}^{\mathrm{coll}}_{0,2}(\Theta_{\Walg_3})
\in \cA^! \widehat{\otimes} \cA^!(\!(z^{-1})\!).
\end{equation}
The pole structure is governed by the OPE poles minus one
because the bar propagator $d\log(z-w)$ absorbs one power of
$(z-w)$.
The four channel pieces are catalogued in
\S\ref{sec:w3hol-r-matrix-channels}.

\subsection{The MC element \texorpdfstring{$\Theta_{\cA}$}{Theta A}}
\label{subsec:w3hol-Theta}

By the bar-intrinsic construction
(Theorem~\ref{thm:mc2-bar-intrinsic}), the element
\begin{equation}
\Theta_{\Walg_3} := D_{\Walg_3} - d_0
\end{equation}
is automatically Maurer--Cartan in the convolution dg Lie algebra
$\Convstr(C^!_{\mathrm{ch}}, P^{\mathrm{ch}})$ because
$D_{\Walg_3}^2 = 0$ at the convolution level
(Theorem~\ref{thm:convolution-d-squared-zero}). No restriction to
simple Lie symmetry is needed. The genus expansion of $\Theta_{\cA}$
is computed in \S\ref{sec:w3hol-kappa} and
\S\ref{sec:w3hol-genus-2}.

\subsection{The shadow connection \texorpdfstring{$\nabla^{\mathrm{hol}}$}{nabla hol}}
\label{subsec:w3hol-nabla}

The flat shadow connection on the genus-0 $n$-point moduli is
\begin{equation}\label{eq:w3hol-shadow-connection}
\nabla^{\mathrm{hol}}_{0,n} = d - \mathrm{Sh}_{0,n}(\Theta_{\Walg_3}),
\end{equation}
the universal flat connection whose flat sections are the genus-0
correlators (Theorem~\ref{thm:shadow-connection}). Flatness is
the genus-0 projection of the MC equation
$D\Theta + \tfrac{1}{2}[\Theta,\Theta] = 0$. The corresponding
single-line shadow metrics on the $T$-line and the $W$-line are
computed in \S\ref{sec:w3hol-shadow-connection}.

\section{The modular characteristic \texorpdfstring{$\kappa(\Walg_3) = 5c/6$}{kappa(W3) = 5c/6}}
\label{sec:w3hol-kappa}

The modular characteristic of $\Walg_3$ is the genus-1 obstruction
coefficient
$\mathrm{obs}_1(\Walg_3) = \kappa(\Walg_3) \cdot \lambda_1$.

\begin{theorem}[Modular characteristic of \texorpdfstring{$\Walg_3$}{W3}; \ClaimStatusConditional]
\label{thm:w3hol-kappa-formula}
For generic $c \neq 0$,
\begin{equation}\label{eq:w3hol-kappa-w3}
\kappa(\Walg_3) \;=\; \frac{5c}{6}.
\end{equation}
This is the $N=3$ specialization of Theorem~\ref{thm:wn-obstruction}.
Equivalently $\kappa(\Walg_3) = c \cdot (H_3 - 1)$ where
$H_3 = 1 + \tfrac{1}{2} + \tfrac{1}{3}$ is the third harmonic number.
\end{theorem}

\begin{remark}[Channel decomposition]
\label{rem:w3hol-channel-decomp}
The modular characteristic decomposes additively over the
two generators:
\begin{equation}\label{eq:w3hol-kappa-channel}
\kappa(\Walg_3) \;=\; \kappa_T + \kappa_W,
\qquad \kappa_T = \frac{c}{2}, \qquad \kappa_W = \frac{c}{3}.
\end{equation}
The $T$-channel coincides with the Virasoro modular characteristic
because the Virasoro subalgebra of $\Walg_3$ is the full stress
tensor; the $W$-channel contributes $c/3$ from its two-point
normalization $\langle W(z)W(0)\rangle = (c/3)\,z^{-6}$
(equation~\eqref{eq:wn-2pt}), reflecting the formula
$\kappa_s = c/s$ for a weight-$s$ generator.
\end{remark}

\begin{remark}[Anomaly ratio \texorpdfstring{$\rho$}{rho}]
\label{rem:w3hol-anomaly-ratio}
The dimensionless anomaly ratio is
\begin{equation}\label{eq:w3hol-rho}
\rho(\Walg_3) \;=\; \frac{\kappa(\Walg_3)}{c} \;=\; H_3 - 1
\;=\; \frac{1}{2} + \frac{1}{3} \;=\; \frac{5}{6}.
\end{equation}
This admits an independent derivation from the $\mathfrak{sl}_3$
exponents $\{m_1, m_2\} = \{1, 2\}$:
$\rho = \sum_i 1/(m_i + 1) = 1/2 + 1/3 = 5/6$
(Corollary~\ref{cor:anomaly-ratio-ds}).
\end{remark}

\noindent\textbf{Independent normalisations.}
The same value is forced by the harmonic sum
$c(H_3-1)$, the channel sum $c/2+c/3$, the exponent formula
$c\sum_i(m_i+1)^{-1}$ for the $\mathfrak{sl}_3$ exponents
\(\{1,2\}\), the genus-$1$ \(d_2\)-coefficient in the bar spectral
sequence, and the vacuum two-point trace
$\sum_s \mathrm{tr}_{V_s}(L_0\,q^{L_0-c/24})|_{q\to1}$.
These five normalisations determine a single scalar:
\(\kappa(\Walg_3)=5c/6\).

\section{Koszul duality and complementarity}
\label{sec:w3hol-complementarity}

\begin{theorem}[Koszul conductor and self-dual point; \ClaimStatusProvedHere]
\label{thm:w3hol-conductor}
The Koszul conductor of $\Walg_3$ is
\begin{equation}\label{eq:w3hol-conductor}
K_3 \;=\; c + c' \;=\; 2(N-1)(2N^2 + 2N + 1)\big|_{N=3} \;=\; 100,
\end{equation}
equivalently $K_N = 4N^3 - 2N - 2$ at $N=3$
(Remark~\ref{rem:koszul-conductor-explicit}). The self-dual
central charge, fixed by $c \mapsto K_3 - c$, is
\begin{equation}\label{eq:w3hol-self-dual-point}
c^* \;=\; \frac{K_3}{2} \;=\; 50.
\end{equation}
\end{theorem}

\begin{proof}
Use the Fateev--Lukyanov Drinfeld--Sokolov central-charge formula in
the canonical parameterisation
\[
c(\Walg_N^k)
=(N-1)-\frac{N(N^2-1)(k+N-1)^2}{k+N}
\]
(Remark~\ref{rem:koszul-conductor-explicit}). For \(N=3\) this is
\[
c(k)=2-\frac{24(k+2)^2}{k+3}.
\]
The Feigin--Frenkel companion is \(k'=-k-6\), hence
\[
c(k')=2-\frac{24(k'+2)^2}{k'+3}
     =2+\frac{24(k+4)^2}{k+3}.
\]
Therefore
\[
c(k)+c(k')
=4+\frac{24((k+4)^2-(k+2)^2)}{k+3}
=4+96=100.
\]
Equivalently, the universal conductor formula
\(2(N-1)(2N^2+2N+1)=4N^3-2N-2\) gives \(K_3=100\).
The self-dual point is the unique fixed point of
\(c\mapsto K_3-c\), namely \(c^*=50\).
\end{proof}

\begin{theorem}[Complementarity sum; \ClaimStatusConditional]
\label{thm:w3hol-kappa-sum}
The modular characteristics of $\Walg_3$ and its Koszul dual sum to
\begin{equation}\label{eq:w3hol-kappa-sum}
\kappa(\Walg_3) + \kappa(\Walg_3^!) \;=\; \frac{5c}{6} +
\frac{5(100-c)}{6} \;=\; \frac{500}{6} \;=\; \frac{250}{3}.
\end{equation}
This sum is constant in $c$ and \emph{nonzero}; in particular, it
does not vanish at the self-dual point $c^* = 50$
(Theorem~\ref{thm:w3-genus1-complementarity}).
\end{theorem}

\begin{remark}[Theorem C is a scalar Verdier statement]
\label{rem:w3hol-theorem-c-scope}
Theorem~C is used in \eqref{eq:w3hol-kappa-sum} only after the two
objects have been typed: \(\cA^i\) is the bar-dual coalgebra,
\(\cA^!\) is the Verdier/Koszul branch on the DS/bar transport lane,
and \(\cC=Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)\) is the chiral
Hochschild closed-sector algebra. The equality
\(\kappa(\cA)+\kappa(\cA^!)=250/3\) is a scalar characteristic
identity. It is not a reconstruction theorem for \(\cA^!\) from
\(\cA^i\), and it is not an equivalence between the line/evaluation
category and the derived centre.
\end{remark}

\begin{remark}[Affine and principal-\texorpdfstring{$\Walg$}{W} complementarity]
\label{rem:w3hol-w-complementarity}
The vanishing of the complementarity sum is a feature of affine
Kac--Moody and free-field families only, where the Feigin--Frenkel
involution $k \mapsto -k - 2h^\vee$ is anti-symmetric around the
origin and $\kappa$ is linear in $k$. For $\Walg$-algebras the
involution acts on the central charge by $c \mapsto K_N - c$, an
affine involution with nonzero fixed-axis offset, so the
complementarity sum becomes $\rho_N \cdot K_N$. For $\Walg_3$
this is $(5/6) \cdot 100 = 250/3$, confirming the formula above.
The expression $\rho_N \cdot K_N$ is the universal closed form;
specializing it to a vanishing sum is permitted only when
$K_N = 0$, which excludes every $\Walg$-algebra.
\end{remark}

\begin{remark}[Self-dual point and critical string point]
\label{rem:w3hol-self-dual-vs-critical}
There are two distinguished central charges for $\Walg_3$.

\begin{itemize}
 \item \textbf{Self-dual point}\quad $c^* = 50$. This is the fixed
 point of the Feigin--Frenkel involution $c \mapsto 100 - c$. On the
 DS/bar transport lane it is the central charge at which the
 Verdier/Koszul companion has the same principal-\(\Walg_3\) central
 charge as~$\cA$; no off-lane isomorphism
 $\cA\simeq\cA^!$ is used. At this point the asymmetry
 $\delta_\kappa := \kappa - \kappa^!$
 vanishes (\emph{Koszul-pair self-duality}).

 \item \textbf{Critical string point}\quad $c_{\mathrm{crit}} = 100$.
 This is the unique central charge at which the matter-plus-ghost
 effective curvature
 $\kappa_{\mathrm{eff}} := \kappa(\text{matter}) +
 \kappa(\text{ghost})$
 vanishes, where the $\Walg_3$ ghost system contributes
 $\kappa(\text{ghost}) = -250/3$. This is the BRST anomaly
 cancellation point, not a Koszul self-duality.
\end{itemize}

The two are $50$ units apart on the $c$-axis and they vanish
\emph{different} quantities. See also
Remark~\ref{rem:w3-kappa-sums} for the parallel phenomenon in Virasoro
($c^*_{\mathrm{Vir}} = 13$, $c_{\mathrm{crit},\mathrm{Vir}} = 26$).
\end{remark}

\section{The collision residue \texorpdfstring{$r(z)$}{r(z)}: four channels}
\label{sec:w3hol-r-matrix-channels}

The spectral $r$-matrix decomposes by generator pair into four
channels: $T$--$T$, $T$--$W$, $W$--$T$, and $W$--$W$. Each is the
collision residue
$\mathrm{Res}^{\mathrm{coll}}_{0,2}(\Theta_\cA)$ extracted from the
corresponding OPE; by the pole orders
satisfy $\mathrm{ord}_r = \mathrm{ord}_{\mathrm{OPE}} - 1$ and
simple OPE poles drop entirely.

\begin{theorem}[Channel-by-channel \texorpdfstring{$r$}{r}-matrix;
\ClaimStatusProvedHere]
\label{thm:w3hol-r-channels}
The four channels of the $\Walg_3$ collision residue are:
\begin{align}
r_{TT}(z) &\;=\; \frac{c/2}{z^3} + \frac{2T}{z}, &
\text{poles } &\{3, 1\}; \label{eq:w3hol-r-TT}\\
r_{TW}(z) &\;=\; \frac{3W}{z}, &
\text{poles } &\{1\}; \label{eq:w3hol-r-TW}\\
r_{WT}(z) &\;=\; \frac{3W}{z}, &
\text{poles } &\{1\}; \label{eq:w3hol-r-WT}\\
r_{WW}(z) &\;=\; \frac{c/3}{z^5} + \frac{2T}{z^3} +
\frac{\partial T}{z^2} +
\frac{\tfrac{3}{10}\partial^2 T + \alpha(c)\,\Lambda}{z}, &
\text{poles } &\{5, 3, 2, 1\}, \label{eq:w3hol-r-WW}
\end{align}
where $\alpha(c) = 16/(22+5c)$ and
$\Lambda = {:}TT{:} - \tfrac{3}{10}\partial^2 T$. The maximum
pole order across all four channels is
\begin{equation}\label{eq:w3hol-kmax}
k_{\max} \;=\; 5,
\end{equation}
attained by the $W$--$W$ channel. More generally, in a diagonal
channel whose leading OPE pole has order $2h$, the leading collision
pole has order $2h-1$ after the $d\log$ absorption.
\end{theorem}

\begin{proof}
Each formula is the OPE
(Theorem~\ref{thm:lambda-coefficients-derivation}) with the bar
propagator factor $d\log(z-w) = dz/(z-w)$ absorbed. The OPE
\[
W(z)W(w) \;\sim\;
\frac{c/3}{(z-w)^6} + \frac{2T(w)}{(z-w)^4} +
\frac{\partial T(w)}{(z-w)^3} +
\frac{\tfrac{3}{10}\partial^2 T(w) + \alpha(c)\,\Lambda(w)}{(z-w)^2}
+ \frac{\tfrac{1}{15}\partial^3 T(w) + \tfrac{\alpha(c)}{2}
\partial\Lambda(w)}{z-w}
\]
has pole orders $\{6, 4, 3, 2, 1\}$. After $d\log$ absorption the
collision residue carries pole orders $\{5, 3, 2, 1\}$; the simple
OPE pole becomes the regular term and contributes nothing
to~$r$. The other channels are read off identically.
\end{proof}

\begin{remark}[An even pole order in a bosonic algebra]
\label{rem:w3hol-even-pole}
The $W$--$W$ channel exhibits an $r$-matrix pole at $z^{-2}$, which
is even-order. For a single-generator bosonic algebra the
$d\log$-absorbed parity sends $z^{-2n}$ to $z^{-(2n-1)}$, leaving
only odd-order poles; the $z^{-2}$ pole at $W$--$W$ arises from the
OPE pole at $z^{-3}$ (the $\partial T$ Taylor term), which exists in
$W$--$W$ but \emph{not} in $T$--$T$ because the $T$--$T$ OPE skips
odd-order poles between $z^{-4}$ and $z^{-2}$. The phenomenon is
intrinsic to multi-weight algebras; the landscape census
(Remark~\ref{rem:ww-even-poles-census}) records the OPE-by-OPE
breakdown.
\end{remark}

\section{The composite field \texorpdfstring{$\Lambda$}{Lambda} on primaries}
\label{sec:w3hol-lambda}

The weight-4 quasi-primary composite field
$\Lambda = {:}TT{:} - \tfrac{3}{10}\partial^2 T$ controls the
nonlinearity of the $W$--$W$ OPE. Its zero mode acts on highest
weight primaries by a quadratic polynomial in the conformal weight.

\begin{proposition}[Action of \texorpdfstring{$\Lambda_0$}{Lambda 0} on primaries; \ClaimStatusProvedElsewhere]
\label{prop:w3hol-lambda-on-primaries}
Let $|h, w\rangle$ be a primary state for $\Walg_3$ with conformal
weight $h$ (eigenvalue of $L_0$) and $W$-charge $w$ (eigenvalue of
$W_0$). Then
\begin{equation}\label{eq:w3hol-lambda-action}
\Lambda_0\,|h, w\rangle \;=\; \Bigl(h^2 - \tfrac{9h}{5}\Bigr)
|h, w\rangle.
\end{equation}
In particular the action of $\Lambda_0$ on a primary is a polynomial
in $h$ alone, independent of the $W$-charge $w$ and \emph{independent
of $c$}.
\end{proposition}

\begin{proof}
The mode expansion of the normal-ordered product is
\begin{equation}\label{eq:w3hol-tt-zero-mode}
({:}TT{:})_0 \;=\; 2\sum_{m \geq 1} L_{-m} L_m + L_0^2.
\end{equation}
On a primary state all $L_m|h,w\rangle = 0$ for $m \geq 1$, so the
sum collapses and $({:}TT{:})_0|h,w\rangle = h^2 |h,w\rangle$. The
zero mode of $\partial^2T$ uses the weight-four convention. Since
$T(z)=\sum_n L_n z^{-n-2}$, one has
$\partial^2T(z)=\sum_n(n+2)(n+3)L_n z^{-n-4}$; hence
$(\partial^2T)_0=6L_0$ as a weight-four field. Thus
$(\partial^2 T)_0 |h,w\rangle = 6h |h,w\rangle$. Combining,
\[
\Lambda_0|h,w\rangle
\;=\; \bigl(h^2 - \tfrac{3}{10}\cdot 6 h\bigr)|h,w\rangle
\;=\; \Bigl(h^2 - \tfrac{9h}{5}\Bigr)|h,w\rangle.\qedhere
\]
\end{proof}

\begin{corollary}[Roots of \texorpdfstring{$\Lambda_0$}{Lambda 0} on the conformal axis]
\label{cor:w3hol-lambda-roots}
The eigenvalue $h^2 - 9h/5$ vanishes on primaries with conformal

the nontrivial root $h = 9/5$ is independent of $c$. This is a
zero-mode statement on the conformal-weight axis; it does not assert
the existence of a unitary $\Walg_3$ module at that weight.
\end{corollary}

\begin{remark}[Independence from \texorpdfstring{$c$}{c}]
\label{rem:w3hol-lambda-c-independent}

\section{Propagator variance and mixing}
\label{sec:w3hol-propagator-variance}

The propagator variance is the multi-channel non-autonomy
invariant
$\delta_{\mathrm{mix}} = \sum_i f_i^2/\kappa_i -
(\sum_i f_i)^2/(\sum_i \kappa_i)$
of Theorem~\ref{prop:propagator-variance}. It vanishes precisely
when the gradient of the quartic contact contribution is parallel
to the curvature direction.

\begin{theorem}[Propagator variance for \texorpdfstring{$\Walg_3$}{W3}; \ClaimStatusConditional]
\label{thm:w3hol-propagator-variance}
The propagator variance of $\Walg_3$ at central charge $c$ is
\begin{equation}\label{eq:w3hol-prop-variance}
\delta_{\mathrm{mix}}(\Walg_3)
\;=\; \frac{1280\, P(c)^2}{c^3 (5c+22)^6},
\end{equation}
where the mixing polynomial is
\begin{equation}\label{eq:w3hol-mixing-poly}
P(\Walg_3) \;=\; 25c^2 + 100c - 428.
\end{equation}
On the positive-curvature chamber the variance is non-negative by
Cauchy--Schwarz and vanishes only at the two roots of $P$,
$c = -2 \pm \tfrac{4\sqrt{33}}{5}$. These are
curvature-proportionality loci for the rational shadow metric:
the diagonal flow on the joint $T$/$W$ line becomes autonomous
there. No representation-theoretic enhancement is asserted.
\end{theorem}

\begin{proof}
By Theorem~\ref{prop:propagator-variance}, the propagator variance
on a multi-channel algebra is
$\delta_{\mathrm{mix}} = (\sum f_i^2 / \kappa_i) -
(\sum f_i)^2 / (\sum \kappa_i)$
where $f_i$ are the per-channel quartic gradient coefficients and
$\kappa_i$ are the per-channel modular characteristics. For
$\Walg_3$ we have $\kappa_T = c/2$, $\kappa_W = c/3$, and the
finite-OPE quartic gradients
\[
f_T=\frac{200(c+14)}{c(5c+22)^2},
\qquad
f_W=\frac{640(15c+82)}{c(5c+22)^3}
\]
(Example~\ref{ex:w3-mixing-polynomial}). Substitution gives
\[
\frac{f_T^2}{c/2}+\frac{f_W^2}{c/3}
-\frac{(f_T+f_W)^2}{5c/6}
=\frac{1280(25c^2+100c-428)^2}{c^3(5c+22)^6}.
\]
This is \eqref{eq:w3hol-prop-variance}.
\end{proof}

\begin{corollary}[Proof of Cauchy--Schwarz non-negativity]
\label{cor:w3hol-prop-variance-positive}
The expression \eqref{eq:w3hol-prop-variance} is a square divided by
$c^3(5c+22)^6$. It is non-negative on the positive-curvature chamber
$c>0$. On real chambers with indefinite curvature this scalar
rational function is no longer a positivity statement; the invariant
meaning of $P(c)=0$ is the curvature-proportionality condition
$f_i/\kappa_i$ independent of~$i$.
\end{corollary}

\section{The shadow connection}

=\frac{4c^2}{9}
+\frac{40960}{3(5c+22)^3}t^2.

The vanishing of \(\alpha_W\) is the \(W\mapsto -W\) parity

\section{Commuting Hamiltonians and the holographic shadow}

\label{thm:w3hol-commuting-differentials}

\begin{equation}\label{eq:w3hol-commuting}
[H_i(\Walg_3), H_j(\Walg_3)] \;=\; 0
\qquad \forall\, i \neq j.
\end{equation}
The order bound $k_{\max} - 1 = 4$ is the $W$-line contribution

\begin{proof}
