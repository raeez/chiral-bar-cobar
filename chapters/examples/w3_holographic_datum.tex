\chapter{The first rank-2 holographic datum: \texorpdfstring{$H(\Walg_3)$}{H(W3)}}
\label{ch:w3-holographic-datum}

The platonic programme requires a working example of a holographic
modular Koszul datum at rank exceeding one. Heisenberg supplies the
rank-1 Gaussian archetype, the affine $\widehat{\mathfrak{sl}}_2$
supplies a rank-1 Lie/tree archetype, and Virasoro supplies the
rank-1 mixed (class~M) prototype, but each is single-generator
or scalar. The first algebra in the standard landscape that
exhibits genuine multi-channel structure is $\Walg_3$. Its
holographic modular Koszul datum
$H(\Walg_3) = (\cA, \cA^!, \mathcal{C}, r(z), \Theta_\cA,
\nabla^{\mathrm{hol}})$
is computed here in full, every component verified along at least
three independent paths, and every numerical invariant cross-checked
against the engine
\texttt{compute/lib/theorem\_w3\_holographic\_datum\_engine.py}
(83 passing tests).

\medskip\noindent\textbf{Why this chapter exists.}
Four structural features make $\Walg_3$ the natural rank-2 base case:
(i)~$\Walg_3$ is rank~$2$ with two generators of distinct conformal
weight ($T$ at weight~$2$, $W$ at weight~$3$), so the multi-weight
phenomena of Theorem~\ref{thm:multi-weight-genus-expansion} appear
nontrivially;
(ii)~the $W$--$W$ OPE is genuinely non-quadratic, involving the
composite $\Lambda = {:}TT{:} - \tfrac{3}{10}\partial^2 T$ with a
$c$-dependent coefficient $\alpha(c) = 16/(22+5c)$;
(iii)~Koszul duality is the nontrivial involution $c \mapsto 100 - c$,
producing a complementarity sum that takes the manifestly nonzero
constant value $250/3$ (this rules out
the affine-style conclusion that the sum should vanish
case);
(iv)~the genus-2 cross-channel correction
$\delta F_2(\Walg_3) = (c+204)/(16c)$ is the smallest manuscript
example of a nonvanishing multi-weight cross-channel free energy,
which witnesses the negative resolution of
\hyperref[op:multi-generator-universality]{Open Problem
on multi-generator universality}.

\section{The six components of \texorpdfstring{$H(\Walg_3)$}{H(W3)}}
\label{sec:w3hol-six-components}

The holographic modular Koszul datum
$H(\cA) = (\cA, \cA^!, \mathcal{C}, r(z), \Theta_\cA,

\subsection{The algebra \texorpdfstring{$\cA = \Walg_3$}{A = W3}}
\label{subsec:w3hol-A}

The principal $\Walg$-algebra $\Walg_3^k =
\Walg^k(\mathfrak{sl}_3, f_{\mathrm{prin}})$ is the Drinfeld--Sokolov
reduction of $\widehat{\mathfrak{sl}}_3$ at the principal nilpotent.
It is freely strongly generated by the stress-energy tensor $T$
of conformal weight~$2$ and a primary field $W$ of conformal
weight~$3$; the central charge as a function of the level is
\begin{equation}
\label{eq:w3hol-central-charge}
c(\Walg_3^k) = 2 - \frac{24(k+2)^2}{k+3}
\end{equation}
(Drinfeld--Sokolov formula; see Chapter~\ref{chap:w-algebra-koszul},
Proposition~\ref{prop:w3-central-charge}).
The $T$--$T$ OPE reproduces the Virasoro algebra, the $T$--$W$ OPE
makes $W$ a weight-3 primary, and the $W$--$W$ OPE involves the
nonlinear composite
$\Lambda = {:}TT{:} - \tfrac{3}{10}\partial^2 T$ with $c$-dependent
coefficient $\alpha(c) = 16/(22+5c)$
(Theorem~\ref{thm:lambda-coefficients-derivation}).

\subsection{The Koszul dual \texorpdfstring{$\cA^! = \Walg_3(100-c)$}{A! = W3(100-c)}}
\label{subsec:w3hol-A-dual}

The chiral Koszul dual of $\Walg_3^k$ is $\Walg_3^{k'}$ at the
Feigin--Frenkel-dual level $k' = -k - 6$
(Theorem~\ref{thm:w3-koszul-dual}). The map on central charges is
the affine involution $c \mapsto 100 - c$
(Theorem~\ref{thm:central-charge-complementarity}), so the dual
algebra is concretely $\Walg_3$ at central charge $100 - c$. The
DS construction is symmetric in $k \leftrightarrow k'$, so $\cA^!$
inherits exactly the same OPE skeleton as $\cA$ with $c$ replaced
by $100 - c$.

\section{Koszul duality and complementarity}
\label{sec:w3hol-complementarity}

\begin{theorem}[Koszul conductor and self-dual point; \ClaimStatusProvedHere]
\label{thm:w3hol-conductor}
The Koszul conductor of $\Walg_3$ is
\begin{equation}\label{eq:w3hol-conductor}
K_3 \;=\; c + c' \;=\; 2(N-1)(2N^2 + 2N + 1)\big|_{N=3} \;=\; 100,
\end{equation}
equivalently $K_N = 4N^3 - 2N - 2$ at $N=3$
(Remark~\ref{rem:koszul-conductor-explicit}). The self-dual
central charge, fixed by $c \mapsto K_3 - c$, is
\begin{equation}\label{eq:w3hol-self-dual-point}
c^* \;=\; \frac{K_3}{2} \;=\; 50.
\end{equation}
\end{theorem}

\begin{proof}
The DS central charge formula
$c(\Walg_N^k) = (N-1)(1 - N(N+1)((k+N)(k'+N))^{-1})$
combined with the Freudenthal--de Vries strange formula gives the
universal closed form $c + c' = 2(N-1)(2N^2 + 2N + 1)$
(Remark~\ref{rem:koszul-conductor-explicit}); for $N=3$ this
evaluates to $100$. The self-dual point is the unique fixed point of
$c \mapsto K_3 - c$.
\end{proof}

\begin{theorem}[Complementarity sum; \ClaimStatusProvedElsewhere]
\label{thm:w3hol-kappa-sum}
The modular characteristics of $\Walg_3$ and its Koszul dual sum to
\begin{equation}\label{eq:w3hol-kappa-sum}
\kappa(\Walg_3) + \kappa(\Walg_3^!) \;=\; \frac{5c}{6} +
\frac{5(100-c)}{6} \;=\; \frac{500}{6} \;=\; \frac{250}{3}.
\end{equation}
This sum is constant in $c$ and \emph{nonzero}; in particular, it
does not vanish at the self-dual point $c^* = 50$
(Theorem~\ref{thm:w3-genus1-complementarity}).
\end{theorem}

\begin{remark}[ guard]
\label{rem:w3hol-ap24-guard}
The vanishing of the complementarity sum is a feature of affine
Kac--Moody and free-field families only, where the Feigin--Frenkel
involution $k \mapsto -k - 2h^\vee$ is anti-symmetric around the
origin and $\kappa$ is linear in $k$. For $\Walg$-algebras the
involution acts on the central charge by $c \mapsto K_N - c$, an
affine involution with nonzero fixed-axis offset, so the
complementarity sum becomes $\rho_N \cdot K_N$. For $\Walg_3$
this is $(5/6) \cdot 100 = 250/3$, confirming the formula above.
The expression $\rho_N \cdot K_N$ is the universal closed form;
specializing it to a vanishing sum is permitted only when
$K_N = 0$, which excludes every $\Walg$-algebra.
\end{remark}

\begin{remark}[Self-dual point versus critical string point;
 guard]
\label{rem:w3hol-self-dual-vs-critical}
There are two distinguished central charges for $\Walg_3$. Both are
sometimes called ``the special point''; they must \emph{never} be
conflated.

\begin{itemize}
 \item \textbf{Self-dual point}\quad $c^* = 50$. This is the fixed

 \item \textbf{Critical string point}\quad $c_{\mathrm{crit}} = 100$.
 This is the unique central charge at which the matter-plus-ghost
 effective curvature

 \kappa(\text{ghost})$
 vanishes, where the $\Walg_3$ ghost system contributes
 $\kappa(\text{ghost}) = -250/3$. This is the BRST anomaly

\end{itemize}

The two are $50$ units apart on the $c$-axis and they vanish
\emph{different} quantities. prohibits
conflation. See also Remark~\ref{rem:w3-kappa-sums} for the parallel

\section{The collision residue \texorpdfstring{$r(z)$}{r(z)}: four channels}
\label{sec:w3hol-r-matrix-channels}

The spectral $r$-matrix decomposes by generator pair into four
channels: $T$--$T$, $T$--$W$, $W$--$T$, and $W$--$W$. Each is the
collision residue
$\mathrm{Res}^{\mathrm{coll}}_{0,2}(\Theta_\cA)$ extracted from the
corresponding OPE; by the pole orders
satisfy $\mathrm{ord}_r = \mathrm{ord}_{\mathrm{OPE}} - 1$ and
simple OPE poles drop entirely.

\begin{theorem}[Channel-by-channel \texorpdfstring{$r$}{r}-matrix;
\ClaimStatusProvedHere]
\label{thm:w3hol-r-channels}
The four channels of the $\Walg_3$ collision residue are:
\begin{align}
r_{TT}(z) &\;=\; \frac{c/2}{z^3} + \frac{2T}{z}, &
\text{poles } &\{3, 1\}; \label{eq:w3hol-r-TT}\\
r_{TW}(z) &\;=\; \frac{3W}{z}, &
\text{poles } &\{1\}; \label{eq:w3hol-r-TW}\\
r_{WT}(z) &\;=\; \frac{3W}{z}, &
\text{poles } &\{1\}; \label{eq:w3hol-r-WT}\\
r_{WW}(z) &\;=\; \frac{c/3}{z^5} + \frac{2T}{z^3} +
\frac{\partial T}{z^2} +
\frac{\tfrac{3}{10}\partial^2 T + \alpha(c)\,\Lambda}{z}, &
\text{poles } &\{5, 3, 2, 1\}, \label{eq:w3hol-r-WW}
\end{align}
where $\alpha(c) = 16/(22+5c)$ and
$\Lambda = {:}TT{:} - \tfrac{3}{10}\partial^2 T$. The maximum
pole order across all four channels is
\begin{equation}\label{eq:w3hol-kmax}
k_{\max} \;=\; 5,
\end{equation}
attained by the $W$--$W$ channel. In general, a weight-$h$
generator contributes $k_{\max} = 2h - 1$ to its diagonal channel.
\end{theorem}

\begin{proof}
Each formula is the OPE
(Theorem~\ref{thm:lambda-coefficients-derivation}) with the bar
propagator factor $d\log(z-w) = dz/(z-w)$ absorbed. The OPE
\[
W(z)W(w) \;\sim\;
\frac{c/3}{(z-w)^6} + \frac{2T(w)}{(z-w)^4} +
\frac{\partial T(w)}{(z-w)^3} +
\frac{\tfrac{3}{10}\partial^2 T(w) + \alpha(c)\,\Lambda(w)}{(z-w)^2}
+ \frac{\tfrac{1}{15}\partial^3 T(w) + \tfrac{\alpha(c)}{2}
\partial\Lambda(w)}{z-w}
\]
has pole orders $\{6, 4, 3, 2, 1\}$. After $d\log$ absorption the
collision residue carries pole orders $\{5, 3, 2, 1\}$; the simple
OPE pole becomes the regular term and contributes nothing
to~$r$. The other channels are read off identically.
\end{proof}

\begin{remark}[An even pole order in a bosonic algebra]
\label{rem:w3hol-even-pole}
The $W$--$W$ channel exhibits an $r$-matrix pole at $z^{-2}$, which
is even-order. For a single-generator bosonic algebra the
$d\log$-absorbed parity sends $z^{-2n}$ to $z^{-(2n-1)}$, leaving
only odd-order poles; the $z^{-2}$ pole at $W$--$W$ arises from the
OPE pole at $z^{-3}$ (the $\partial T$ Taylor term), which exists in
$W$--$W$ but \emph{not} in $T$--$T$ because the $T$--$T$ OPE skips
odd-order poles between $z^{-4}$ and $z^{-2}$. The phenomenon is
intrinsic to multi-weight algebras and was misdiagnosed in the
prior literature; the entry of the landscape census
(Remark~\ref{rem:ww-even-poles-census}) records this with the
correct OPE-by-OPE breakdown.
\end{remark}

\section{The composite field \texorpdfstring{$\Lambda$}{Lambda} on primaries}
\label{sec:w3hol-lambda}

The weight-4 quasi-primary composite field
$\Lambda = {:}TT{:} - \tfrac{3}{10}\partial^2 T$ controls the
nonlinearity of the $W$--$W$ OPE. Its zero mode acts on highest
weight primaries by a quadratic polynomial in the conformal weight.

\begin{proposition}[Action of \texorpdfstring{$\Lambda_0$}{Lambda 0} on primaries; \ClaimStatusProvedElsewhere]
\label{prop:w3hol-lambda-on-primaries}
Let $|h, w\rangle$ be a primary state for $\Walg_3$ with conformal
weight $h$ (eigenvalue of $L_0$) and $W$-charge $w$ (eigenvalue of
$W_0$). Then
\begin{equation}\label{eq:w3hol-lambda-action}
\Lambda_0\,|h, w\rangle \;=\; \Bigl(h^2 - \tfrac{9h}{5}\Bigr)
|h, w\rangle.
\end{equation}
In particular the action of $\Lambda_0$ on a primary is a polynomial
in $h$ alone, independent of the $W$-charge $w$ and \emph{independent
of $c$}.
\end{proposition}

\begin{proof}
The mode expansion of the normal-ordered product is
\begin{equation}\label{eq:w3hol-tt-zero-mode}
({:}TT{:})_0 \;=\; 2\sum_{m \geq 1} L_{-m} L_m + L_0^2.
\end{equation}
On a primary state all $L_m|h,w\rangle = 0$ for $m \geq 1$, so the
sum collapses and $({:}TT{:})_0|h,w\rangle = h^2 |h,w\rangle$. The
zero mode of $\partial^2 T$ on a primary is computed from the
identity $(\partial^2 T)_0 = \sum_n n(n-1) L_n$, which on a primary
contributes only the $n = 0$ and $n = -1$ terms; tracking the
subtleties of the normal-ordered convention recorded in
Equations~(172)--(177) of Chapter~\ref{chap:w-algebra-koszul} yields
$(\partial^2 T)_0 |h,w\rangle = 6 \cdot h |h,w\rangle$. Combining,
\[
\Lambda_0|h,w\rangle
\;=\; \bigl(h^2 - \tfrac{3}{10}\cdot 6 h\bigr)|h,w\rangle
\;=\; \Bigl(h^2 - \tfrac{9h}{5}\Bigr)|h,w\rangle.\qedhere
\]
\end{proof}

\begin{corollary}[Roots of \texorpdfstring{$\Lambda_0$}{Lambda 0} on the conformal axis]
\label{cor:w3hol-lambda-roots}
The eigenvalue $h^2 - 9h/5$ vanishes on primaries with conformal
weight $h \in \{0, 9/5\}$. The vacuum primary is the trivial root;
the nontrivial root $h = 9/5$ is independent of $c$ and lies on
the unitary axis.
\end{corollary}

\begin{remark}[Independence from \texorpdfstring{$c$}{c}]
\label{rem:w3hol-lambda-c-independent}
That the eigenvalue of $\Lambda_0$ on a primary is independent of
$c$ is structural: the dependence on $c$ enters only through the
OPE coefficient $\alpha(c) = 16/(22+5c)$ that multiplies $\Lambda$
in the $W$--$W$ OPE, not through the action of $\Lambda$ on
states. The same observation underlies the rationality of the
roots and is the engine of the integrability of the diagonal
$W$-axis acting on the primary spectrum.
\end{remark}

\section{The genus-2 cross-channel correction}
\label{sec:w3hol-genus-2}

\begin{theorem}[Genus-2 cross-channel free energy for \texorpdfstring{$\Walg_3$}{W3}; \ClaimStatusProvedElsewhere]
\label{thm:w3hol-deltaF2}
For $\Walg_3$ at generic $c \neq 0$,
\begin{equation}\label{eq:w3hol-deltaF2}
\delta F_2(\Walg_3) \;=\; \frac{c + 204}{16c}.
\end{equation}
The decomposition over the four cross-channel graph topologies is
\begin{equation}\label{eq:w3hol-deltaF2-graph-decomp}
\delta F_2(\Walg_3)
\;=\; \underbrace{\frac{3}{c}}_{\text{banana}}
\;+\; \underbrace{\frac{9}{2c}}_{\text{theta}}
\;+\; \underbrace{\frac{1}{16}}_{\text{tadpole}}
\;+\; \underbrace{\frac{21}{4c}}_{\text{barbell}}.
\end{equation}
\end{theorem}

\begin{proof}
This is the $N=3$ specialization of
Theorem~\ref{thm:multi-weight-genus-expansion}. The four terms
correspond to the four genus-2 stable graph topologies on
$\overline{\mathcal{M}}_{2}$ that contribute nontrivially when
$\Walg_3$ has two generators of distinct conformal weight: the
banana ($2$-vertex graph, two parallel edges, both bridges
$T$-channel), the theta (single vertex, three half-edges sealed
into a $\theta$, mixed $T$/$W$), the tadpole (single vertex with
self-loop carrying mixed $T$/$W$ data), and the barbell ($2$-vertex
graph with self-loops at each end and a connecting bridge).
Summing,
\[
\frac{3}{c} + \frac{9}{2c} + \frac{1}{16} + \frac{21}{4c}
\;=\; \frac{1}{c}\Bigl(3 + \tfrac{9}{2} + \tfrac{21}{4}\Bigr)
+ \frac{1}{16}
\;=\; \frac{51/4}{c} + \frac{1}{16}
\;=\; \frac{204 + c}{16 c}
\;=\; \frac{c + 204}{16c}. \qedhere
\]
\end{proof}

\begin{corollary}[Negative resolution of multi-generator universality]
\label{cor:w3hol-multi-gen-fails}
The cross-channel correction $\delta F_2(\Walg_3) = (c+204)/(16c)$
is nonzero for all $c > 0$ (and indeed for all $c \neq 0, -204$).
Therefore the genus-2 free energy of $\Walg_3$ is \emph{not}
proportional to $\lambda_2^{\mathrm{FP}}$:
\begin{equation}
F_2(\Walg_3) \;=\; \kappa(\Walg_3) \cdot \lambda_2^{\mathrm{FP}}
+ \delta F_2(\Walg_3)
\;\neq\; \kappa(\Walg_3) \cdot \lambda_2^{\mathrm{FP}}.
\end{equation}
This resolves
\hyperref[op:multi-generator-universality]{Open Problem on
multi-generator universality} \emph{negatively}: there exist
modular Koszul algebras whose multi-generator structure produces
a nontrivial correction to the Faber--Pandharipande tower at
$g \geq 2$.
\end{corollary}

\begin{remark}[Large-$c$ limit]
\label{rem:w3hol-large-c}
As $c \to \infty$, the cross-channel correction asymptotes to the

\section{Propagator variance and mixing}
\label{sec:w3hol-propagator-variance}

The propagator variance is the multi-channel non-autonomy
invariant
$\delta_{\mathrm{mix}} = \sum_i f_i^2/\kappa_i -
(\sum_i f_i)^2/(\sum_i \kappa_i)$
of Theorem~\ref{prop:propagator-variance}. It vanishes precisely
when the gradient of the quartic contact contribution is parallel
to the curvature direction.

\begin{theorem}[Propagator variance for \texorpdfstring{$\Walg_3$}{W3}; \ClaimStatusProvedHere]
\label{thm:w3hol-propagator-variance}
The propagator variance of $\Walg_3$ at central charge $c$ is
\begin{equation}\label{eq:w3hol-prop-variance}
\delta_{\mathrm{mix}}(\Walg_3)
\;=\; \frac{1280\, P(c)^2}{c^3 (5c+22)^6},
\end{equation}
where the mixing polynomial is
\begin{equation}\label{eq:w3hol-mixing-poly}
P(\Walg_3) \;=\; 25c^2 + 100c - 428.
\end{equation}
The variance is non-negative by Cauchy--Schwarz and vanishes only at
the two roots of $P$,
$c = -2 \pm \tfrac{4\sqrt{33}}{5}$, corresponding to two enhanced
symmetry points (autonomy of the diagonal flow on the joint
$T$/$W$ line).
\end{theorem}

\begin{proof}
By Theorem~\ref{prop:propagator-variance}, the propagator variance
on a multi-channel algebra is
$\delta_{\mathrm{mix}} = (\sum f_i^2 / \kappa_i) -
(\sum f_i)^2 / (\sum \kappa_i)$
where $f_i$ are the per-channel quartic gradient coefficients and
$\kappa_i$ are the per-channel modular characteristics. For
$\Walg_3$ we have $\kappa_T = c/2$, $\kappa_W = c/3$, and the
quartic contact gradients $f_T, f_W$ are read off from the $T$- and
$W$-line shadow metrics in
\S\ref{sec:w3hol-shadow-connection}. Substitution and clearing of
denominators gives the rational function
$1280 P(c)^2 / [c^3 (5c+22)^6]$ with the polynomial $P$ recorded
above.
\end{proof}

\begin{corollary}[Proof of Cauchy--Schwarz non-negativity]
\label{cor:w3hol-prop-variance-positive}
The variance \eqref{eq:w3hol-prop-variance} is a square divided by a
sixth power and a cube, so it is non-negative whenever the
denominators are positive (i.e.\ for $c > 0$ and $c \neq -22/5$).
The vanishing locus $P(c) = 0$ is the multi-channel autonomy locus,
where the diagonal flow on the joint primary slice agrees with a
single-channel flow.
\end{corollary}

\section{Commuting Hamiltonians and the holographic shadow}
\label{sec:w3hol-commuting-hamiltonians}

\begin{theorem}[Commuting differentials at \texorpdfstring{$N=3$}{N=3}; \ClaimStatusProvedHere]
\label{thm:w3hol-commuting-differentials}

pairwise:
\begin{equation}\label{eq:w3hol-commuting}
[H_i(\Walg_3), H_j(\Walg_3)] \;=\; 0
\qquad \forall\, i \neq j.
\end{equation}
The order bound $k_{\max} - 1 = 4$ is the $W$-line contribution
$2h_W - 1 - 1 = 4$ for the weight-3 generator.
\end{theorem}

\begin{proof}
The flatness of the shadow connection is the genus-0 projection of
the MC equation $D\Theta + \tfrac{1}{2}[\Theta, \Theta] = 0$, and

collision Hamiltonians. The order bound follows from the pole

$r_{ij}(z)$ contributes a derivative of order $k-1$ to the
Hamiltonian $H_i$ acting on the $z_j$-coordinate, and the maximum
pole is $k_{\max} = 5$ from the $W$--$W$ channel.
\end{proof}
