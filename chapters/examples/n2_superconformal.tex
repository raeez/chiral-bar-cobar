\chapter{\texorpdfstring{The $\mathcal{N}=2$ superconformal algebra}{The N=2 superconformal algebra}}\label{chap:n2-sca}

The $\mathcal{N}=2$ superconformal algebra is the definitive
witness in the standard landscape that chiral Koszulness and
Chevalley--Eilenberg Koszulness are genuinely distinct properties:
the CE complex has $H^2_{\mathrm{CE}} \neq 0$ at conformal
weight~$3$, yet the chiral bar complex has $H^2_{\mathrm{bar}} = 0$.
The sub-leading OPE modes in the chiral bar differential, invisible
to the CE complex, provide coboundary contributions that kill the
CE cocycles.

Wave-14 anchors. The W-type complementarity sum
$\kappa(c) + \kappa(6{-}c) = 1$ is an instance of the universal
Koszul-conductor identity
$\kappa + \kappa' = \varrho \cdot K$ analysed in
Chapter~\ref{chap:universal-conductor}; here $\varrho_{\mathcal{N}=2} = 1$
and $K_{\mathcal{N}=2} = 1$ \textup{(}distinct from the Virasoro
conductor $K = 13$ and the Bershadsky--Polyakov conductor $K = 196$;
cf.\ the per-family table in
Chapter~\ref{chap:universal-conductor}\textup{)}.
The multi-channel shadow profile (\S\ref{sec:n2-multi-channel-shadow})
inhabits all three of $\{\mathbf{G},\mathbf{L},\mathbf{M}\}$
simultaneously across $J$-, $G$-, $T$-channels respectively; the
overall class~$\mathbf{M}$ is governed by the deepest channel, in
accord with the channel maximum rule of
Chapter~\ref{chap:shadow-quadrichotomy-platonic}.

The Kazama--Suzuki coset decomposition $\mathrm{SCA}_c \simeq
\operatorname{Com}(\mathrm{U}(1),\, \widehat{\mathfrak{sl}}_{2,k}
\otimes \psi\bar\psi)$ determines the modular characteristic
$\kappa = (k{+}4)/4$; the Feigin--Frenkel involution
$k \mapsto -k-4$ identifies the Koszul dual at central charge
$c' = 6 - c$; the complementarity sum
$\kappa(c) + \kappa(6{-}c) = 1$ is constant. The shadow obstruction
tower has multi-channel structure: the $T$-channel is class~M
(infinite shadow depth, as for Virasoro), the $J$-channel is class~G
(Gaussian, terminating at degree~$2$), and the overall shadow class
is~M, governed by the deepest channel. All five main theorems are
verified.

\noindent
The modular Koszul triple
(Definition~\ref{def:modular-koszul-triple}) of the $\mathcal{N}=2$
SCA at generic level $k \neq -2$ is
\begin{equation}\label{eq:n2-triple}
\mathfrak{T}_{\mathrm{SCA}}
\;=\;
\bigl(\,\mathrm{SCA}_c,\;\;
\mathrm{SCA}_{6-c},\;\;
r_{\mathrm{SCA}}(z)\,\bigr),
\end{equation}
where the $r$-matrix has two bosonic channels: a triple pole
$(c/2)/z^3 + 2T/z$ from the Virasoro subalgebra and a simple
pole $(c/3)/z$ from the $\mathrm{U}(1)$ current~$J$.

The $\mathcal{N}=2$ SCA occupies a distinguished position in the
shadow depth classification. The Virasoro subalgebra places it in
class~M ($r_{\max} = \infty$), while the $\mathrm{U}(1)$ current
gives a class~G channel ($r_{\max} = 2$) and the fermionic
$G^+G^-$ pairing gives a class~L channel ($r_{\max} = 3$). The
overall class is~M, governed by the deepest channel.
Extended supersymmetry modifies the modular characteristic
relative to the Virasoro value: the coset decomposition
(Proposition~\ref{prop:n2-kappa}) yields
$\kappa = (k{+}4)/4 = (6{-}c)/(2(3{-}c))$, which diverges
at the free-field limit $c = 3$ and vanishes at $c = 6$
(corresponding to $k = -4$).


\section{OPE data and generators}

The $\mathcal{N}=2$ superconformal algebra has four generators:
\begin{equation}\label{eq:n2-generators}
T \;\;(\text{wt } 2,\; \text{bosonic}), \quad
J \;\;(\text{wt } 1,\; \text{bosonic}), \quad
G^+ \;\;(\text{wt } \tfrac{3}{2},\; \text{fermionic}), \quad
G^- \;\;(\text{wt } \tfrac{3}{2},\; \text{fermionic}).
\end{equation}
The stress tensor~$T$ generates a Virasoro subalgebra at central
charge~$c$; the $\mathrm{U}(1)$ current~$J$ has charge assignments
$\mathrm{ch}(G^+) = +1$, $\mathrm{ch}(G^-) = -1$,
$\mathrm{ch}(T) = \mathrm{ch}(J) = 0$; and the supercurrents
$G^\pm$ are fermionic primaries of conformal weight~$\frac{3}{2}$.

The OPE algebra is (OPE-mode convention: $a_{(n)}b$ denotes the
coefficient of $(z{-}w)^{-(n+1)}$):
\begin{align}
T_{(3)}T &= \tfrac{c}{2}, &
T_{(1)}T &= 2T, &
T_{(0)}T &= \partial T,
\label{eq:n2-ope-TT} \\
T_{(1)}J &= J, &
T_{(0)}J &= \partial J,
\label{eq:n2-ope-TJ} \\
T_{(1)}G^\pm &= \tfrac{3}{2}\,G^\pm, &
T_{(0)}G^\pm &= \partial G^\pm,
\label{eq:n2-ope-TG} \\
J_{(1)}J &= \tfrac{c}{3},
\label{eq:n2-ope-JJ} \\
J_{(0)}G^+ &= G^+, &
J_{(0)}G^- &= -G^-,
\label{eq:n2-ope-JG} \\
G^+_{(2)}G^- &= \tfrac{c}{3}, &
G^+_{(1)}G^- &= J, &
G^+_{(0)}G^- &= T + \tfrac{1}{2}\,\partial J,
\label{eq:n2-ope-GG} \\
G^\pm_{(n)}G^\pm &= 0 \quad\text{for all } n.
\label{eq:n2-ope-self}
\end{align}
The fermionic self-OPEs vanish identically. By vertex
superalgebra skew-symmetry, the reversed zero-mode product is
$G^-_{(0)}G^+ = T - \frac{1}{2}\,\partial J$
(the sign on $\partial J$ flips relative to
$G^+_{(0)}G^- = T + \frac{1}{2}\,\partial J$); the higher reversed
modes $G^-_{(n)}G^+$ for $n \geq 1$ receive derivative corrections
from the skew-symmetry formula.

\begin{remark}[Kazama--Suzuki coset realization]
\label{rem:n2-kazama-suzuki}
\index{Kazama--Suzuki coset}
\index{N=2 superconformal algebra!coset realization}
The $\mathcal{N}=2$ SCA at central charge
$c = 3k/(k{+}2)$ is realized as the Kazama--Suzuki coset
\begin{equation}\label{eq:n2-coset}
\mathrm{SCA}_c
\;\simeq\;
\operatorname{Com}\bigl(\mathrm{U}(1),\;
\widehat{\mathfrak{sl}}_{2,k} \otimes \psi\bar\psi\bigr),
\end{equation}
where $\psi\bar\psi$ denotes a complex free fermion pair and
$\mathrm{U}(1)$ is the diagonal current
$J_{\mathrm{diag}} = J^3_{\mathrm{KM}} + :\!\psi\bar\psi\!:$\,.
The level $k$ ranges over $\mathbb{C} \setminus \{-2\}$; at
$k = -2$ the Sugawara construction is undefined. The
free-field limit $k \to \infty$ gives $c \to 3$.
\end{remark}


\section{Modular characteristic}

\begin{proposition}[Modular characteristic of the $\mathcal{N}=2$
SCA;\; \ClaimStatusProvedHere]
\label{prop:n2-kappa}
\index{N=2 superconformal algebra!modular characteristic}
\index{kappa@$\kappa$!N=2 superconformal}
At central charge $c = 3k/(k{+}2)$ with $k \neq -2$:
\begin{equation}\label{eq:n2-sca-kappa}
\kappa(\mathrm{SCA}_c)
\;=\;
\frac{k{+}4}{4}
\;=\;
\frac{6{-}c}{2(3{-}c)}\,.
\end{equation}
\end{proposition}

\begin{proof}
The Kazama--Suzuki decomposition~\eqref{eq:n2-coset} gives
$\kappa$ as the coset difference of modular characteristics:
\begin{align*}
\kappa(\mathrm{SCA}_c)
&= \kappa(\widehat{\mathfrak{sl}}_{2,k})
 + \kappa(\psi\bar\psi)
 - \kappa(\mathrm{U}(1)_{\mathrm{diag}}) \\
&= \frac{3(k{+}2)}{4}
 + \frac{1}{2}
 - \Bigl(\frac{k}{2} + 1\Bigr)
\;=\;
\frac{3k + 6 + 2 - 2k - 4}{4}
\;=\;
\frac{k{+}4}{4}\,.
\end{align*}
The coset constituents:
$\kappa(\widehat{\mathfrak{sl}}_{2,k})
= \dim(\mathfrak{sl}_2)(k{+}h^\vee)/(2h^\vee)
= 3(k{+}2)/4$;
$\kappa(\psi\bar\psi) = 1/2$
(complex free fermion at $c = 1$);
$\kappa(\mathrm{U}(1)_{\mathrm{diag}})
= k/2 + 1$, since the diagonal current
$J_{\mathrm{diag}} = J^3_{\mathrm{KM}} + :\!\psi\bar\psi\!:$
has level $\ell = k/2 + 1$ (the Cartan generator $J^3$
contributes $k/2$ and the fermion bilinear contributes~$1$).
Substituting $k = 2c/(3{-}c)$ yields
$(6{-}c)/(2(3{-}c))$.
\end{proof}


\section{Koszul duality and complementarity}

\begin{proposition}[Complementarity for the $\mathcal{N}=2$
SCA;\; \ClaimStatusProvedHere]
\label{prop:n2-complementarity}
\index{N=2 superconformal algebra!Koszul duality}
\index{complementarity!N=2 superconformal}
The Feigin--Frenkel involution
$k \mapsto -k - 2h^\vee = -k - 4$
sends $c = 3k/(k{+}2)$ to
$c' = 6 - c$. The modular characteristics satisfy
\begin{equation}\label{eq:n2-sca-complementarity-sum}
\kappa(c) + \kappa(6{-}c) \;=\; 1\,.
\end{equation}
The self-dual point is $c = 3$ ($k \to \infty$).
\end{proposition}

\begin{proof}
Under $k \mapsto -k-4$:
$c' = 3(-k{-}4)/(-k{-}2) = 3(k{+}4)/(k{+}2)$.
Then $c + c' = 3k/(k{+}2) + 3(k{+}4)/(k{+}2) = 6$.
For the complementarity sum:
\[
\kappa(c) + \kappa(6{-}c)
\;=\;
\frac{6{-}c}{2(3{-}c)}
\;+\;
\frac{6{-}(6{-}c)}{2(3{-}(6{-}c))}
\;=\;
\frac{6{-}c}{2(3{-}c)}
\;+\;
\frac{c}{2(c{-}3)}
\;=\;
\frac{6{-}c}{2(3{-}c)}
\;-\;
\frac{c}{2(3{-}c)}
\;=\;
\frac{6 - 2c}{2(3{-}c)}
\;=\; 1\,.
\]
This is W-type complementarity: the sum is~$1$, not~$0$
(compare $\kappa + \kappa' = 0$ for affine KM and
$\kappa + \kappa' = 13$ for Virasoro).
\end{proof}


\section{Chiral Koszulness: CE versus chiral bar}

\begin{proposition}[PBW Koszulness of the $\mathcal{N}=2$ SCA;\;
\ClaimStatusProvedHere]
\label{prop:n2-koszulness}
\index{N=2 superconformal algebra!Koszulness}
\index{chiral Koszulness!N=2 superconformal}
The $\mathcal{N}=2$ superconformal algebra is chirally Koszul
at generic central charge. The PBW spectral sequence collapses
at~$E_2$, and $H^n(\barB(\mathrm{SCA}_c)) = 0$ for $n \geq 2$
in the bar-relevant range.
\end{proposition}

\begin{proof}
The $\mathcal{N}=2$ SCA is freely strongly generated by
$\{T, J, G^+, G^-\}$ at generic~$c$
(Adamovi\'c~\cite{Adamovic1999}: no singular vectors in the
vacuum Verma module at generic level).
By Proposition~\ref{prop:pbw-universality}, free strong generation
implies PBW collapse at~$E_2$, hence chiral Koszulness.

The CE complex of the negative-mode Lie superalgebra
$\mathfrak{g}_-$ has $H^2_{\mathrm{CE}} \neq 0$ at conformal
weight~$3$.
This does \emph{not} obstruct chiral Koszulness: the CE complex
uses only the mode-$0$ product $a_{(0)}b$ (the Lie bracket),
while the chiral bar differential incorporates all OPE modes via the
Orlik--Solomon form factor on configuration spaces
(Proposition~\ref{prop:pbw-universality}).
The sub-leading modes $G^+_{(1)}G^- = J$ and
$G^+_{(2)}G^- = c/3$
provide additional coboundary contributions in the PBW spectral
sequence that kill the CE cocycles at the $E_2$ page.
Verified computationally through conformal weight~$6$.
\end{proof}

\begin{remark}[The CE/chiral bar gap]
\label{rem:n2-ce-chiral-gap}
\index{N=2 superconformal algebra!CE vs chiral bar}
The $\mathcal{N}=2$ SCA provides a direct demonstration
in the standard landscape that CE Koszulness (vanishing of
$H^2_{\mathrm{CE}}(\mathfrak{g}_-)$) and chiral Koszulness
(vanishing of $H^2(\barB(A))$) are genuinely distinct properties.
For single-generator algebras (Heisenberg, Virasoro), the two
complexes agree at bar degree~$2$ because the mode-$0$ product
exhausts the generating OPE at that level. For multi-generator
algebras with mixed bosonic/fermionic statistics and sub-leading
OPE poles, the chiral bar complex is strictly finer.
The mechanism: at bar degree~$3$, the Orlik--Solomon differential on
$\operatorname{Conf}_3(\mathbb{C})$ activates the higher OPE modes
$G^+_{(1)}G^-$ and $G^+_{(2)}G^-$, which are invisible to
the CE complex but provide coboundaries in the PBW spectral sequence
that annihilate the CE $H^2$ classes.
\end{remark}


\section{Shadow obstruction tower: multi-channel structure}
\label{sec:n2-multi-channel-shadow}

The $\mathcal{N}=2$ SCA has four generators spanning three
independent primary lines in the cyclic deformation complex
$\Defcyc^{\mathrm{mod}}$
(Definition~\ref{def:modular-cyclic-deformation-complex}):
\begin{enumerate}[label=(\roman*),nosep]
\item The \emph{$T$-line} (stress tensor, conformal weight~$2$,
 bosonic): identical to the Virasoro line. The $TT$ OPE has
 maximal pole order~$4$; $d\log$ absorption
 (Remark~\ref{rem:bar-pole-absorption})
 gives collision depth $k_{\max} = 3$.
 Class~M: infinite shadow depth.
\item The \emph{$J$-line} ($\mathrm{U}(1)$ current, conformal
 weight~$1$, bosonic): identical to the Heisenberg line. The
 $JJ$ OPE has maximal pole order~$2$, giving $k_{\max} = 1$.
 Class~G: Gaussian, shadow depth $r_{\max} = 2$.
\item The \emph{$G$-lines} (supercurrents $G^\pm$, conformal
 weight~$\frac{3}{2}$, fermionic): the $G^\pm G^\pm$ OPEs
 vanish; the mixed $G^+ G^-$ OPE has maximal pole order~$3$,
 giving $k_{\max} = 2$. Class~L: Lie/tree, shadow depth
 $r_{\max} = 3$.
\end{enumerate}
The overall shadow class is~M (governed by the deepest channel),
and the shadow depth is $r_{\max} = \infty$. The modular
characteristic $\kappa = (k{+}4)/4$ receives contributions from all
channels via the Kazama--Suzuki coset decomposition
(Proposition~\ref{prop:n2-kappa}).

\begin{remark}[Shadow obstruction tower for the $\mathcal{N}=2$ SCA]
\textup{(}ALL-WEIGHT + $\delta F_g^{\mathrm{cross}}$\textup{})
\label{rem:n2-shadow-tower}
\index{shadow obstruction tower!N=2 superconformal}
For $\mathrm{SCA}_c$ at generic level, the shadow obstruction tower
in the modular convolution algebra $\gAmod$
(Definition~\ref{def:modular-convolution-dg-lie}) has modular
characteristic $\kappa = (k{+}4)/4$. Since the $\mathcal{N}=2$
SCA has generators at three distinct conformal weights
($1, \frac{3}{2}, 2$), it is a multi-weight algebra; the genus-$1$
identity $\mathrm{obs}_1 = \kappa \cdot \lambda_1$ holds
unconditionally \textup{($g = 1$; all-weight at $g = 1$ is unconditional)}, while the higher-genus scalar formula receives
cross-channel corrections $\delta F_g^{\mathrm{cross}}$ at $g \geq 2$
\textup{(all-weight, with cross-channel correction $\delta F_g^{\mathrm{cross}}$)}. The
tower does not terminate: the $T$-channel carries the
same infinite tower as the Virasoro algebra
(Chapter~\ref{chap:w-algebra-koszul}), including a nonzero
quintic obstruction $o_5^{(T)} \neq 0$. The $J$-channel terminates
at degree~$2$ (Gaussian) and the $G$-channel at degree~$3$
(Lie/tree, from the $G^+G^-$ bracket). Cross-channel interactions
between the $T$-, $J$-, and $G$-channels produce
additional multi-channel shadow coefficients at degrees $\geq 4$.
\end{remark}


\section{Spectral flow}
\label{sec:n2-spectral-flow}

The $\mathcal{N}=2$ SCA admits a one-parameter family of
automorphisms $\sigma_\theta$ ($\theta \in \mathbb{Z}$), the
\emph{spectral flow}, defined by
\begin{equation}\label{eq:n2-spectral-flow}
\sigma_\theta(T) = T + \theta\,J + \tfrac{c}{6}\,\theta^2,
\qquad
\sigma_\theta(J) = J + \tfrac{c}{3}\,\theta,
\qquad
\sigma_\theta(G^\pm) = G^\pm_{\pm\theta},
\end{equation}
where $G^\pm_{\pm\theta}$ denotes the shift of mode indices by
$\pm\theta$. At $\theta = 1/2$, the spectral flow intertwines
the Neveu--Schwarz and Ramond sectors.

Since $\sigma_\theta$ is an algebra automorphism, it preserves
all OPE structure constants and hence every shadow coefficient:
$S_r^{\sigma_\theta} = S_r$ for all~$r$. The modular
characteristic~$\kappa$, the cubic shadow~$C$, the quartic
resonance class~$Q$, and the full shadow obstruction tower are
spectral-flow invariants. This provides a non-trivial consistency
check: the NS and Ramond sectors have different field content
(integer vs.\ half-integer moded supercurrents) but identical
shadow data (Remark~\ref{rem:n2-spectral-flow-invariance}).


\begin{remark}[Comparison with the $\mathcal{N}=1$ and
$\mathcal{N}=4$ families]
\label{rem:n2-susy-hierarchy}
\index{superconformal hierarchy}
\index{N=1 super Virasoro}
\index{N=4 small superconformal}
The three fundamental superconformal families form a hierarchy
under extended supersymmetry:
\[
\renewcommand{\arraystretch}{1.3}
\begin{array}{cccccc}
\mathcal{N} & c & \kappa &
 c' & \kappa + \kappa' & \text{Self-dual} \\
\hline
0 \;(\text{Vir}) &
 c & c/2 & 26{-}c & 13 & c = 13 \\
1 \;(\text{SVir}) &
 c & (3c{-}2)/4 & 15{-}c & 41/4 & c = 15/2 \\
2 \;(\text{SCA}) &
 3k/(k{+}2) &
 (k{+}4)/4 & 6{-}c & 1 & c = 3 \\
4 \;(\text{small}) &
 6k &
 2k & -c{-}24 & -8 & c = -12 \\
\end{array}
\]
As $\mathcal{N}$ increases, the complementarity sum
$\kappa + \kappa'$ decreases: $13, \; 41/4, \; 1, \; {-}8$.
This reflects the increasing cancellation between bosonic and
fermionic contributions to the genus-$1$ obstruction class:
the modular characteristic is an invariant of the full algebra,
not of the Virasoro subalgebra alone.
\end{remark}
