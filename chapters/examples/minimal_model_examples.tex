
The Verlinde formula of the preceding section is a universal machine;
its output is only as informative as the modular data one feeds in.
Here we turn the crank on specific models --- the Ising model
$\mathcal{M}(4,3)$, the tricritical Ising model
$\mathcal{M}(5,4)$, and the $\mathcal{W}_3$ minimal models
$\mathcal{W}_3(4,3)$ and $\mathcal{W}_3(6,5)$ --- computing every
fusion coefficient from the S-matrix and verifying the result
against the known fusion tables.  Beyond verification, these
explicit computations illustrate a general principle of the
algebraic engine: the bar complex of a rational algebra is
\emph{completely determined} by the finite data of the modular
S-matrix, and the Koszul dual $\mathcal{A}^!$ can in principle be
read off from the fusion ring.  The Ising model, with its
$\mathbb{Z}/2$ fusion ring, is the simplest instance;
$\mathcal{W}_3(6,5)$, with its $A_2$ Weyl-orbit identifications, is
the first case where the passage from Virasoro to $\mathcal{W}$-algebra
labeling requires non-trivial Weyl-orbit identifications.

\subsection{\texorpdfstring{Complete Verlinde computation for $\mathcal{M}(4,3)$}{Complete Verlinde computation for M(4,3)}}
\label{subsec:verlinde-4-3-complete}

\begin{computation}[All fusion coefficients for the Ising model]
\label{comp:ising-all-fusion}
\index{Ising model!fusion coefficients}

Recall from Example~\ref{ex:w3-3-4-complete}: the three Virasoro
primaries of $\mathcal{M}(4,3)$ after identification are
$\mathbb{I}$ ($h=0$), $\sigma$ ($h=1/16$), $\varepsilon$ ($h=1/2$),
with S-matrix
\[
S = \frac{1}{2}\begin{pmatrix}
1 & \sqrt{2} & 1 \\
\sqrt{2} & 0 & -\sqrt{2} \\
1 & -\sqrt{2} & 1
\end{pmatrix}.
\]

We compute all $3^3 = 27$ fusion coefficients $N_{ij}^k$
using $N_{ij}^k = \sum_{\ell=0}^{2}
S_{i\ell}S_{j\ell}S_{\ell k}^*/S_{0\ell}$.
Since $S$ is real and symmetric, $S^* = S$.

\emph{Step 1: Compute $S_{i\ell}/S_{0\ell}$.}
\begin{align*}
\frac{S_{0,\ell}}{S_{0,\ell}} &= (1, 1, 1), \\
\frac{S_{1,\ell}}{S_{0,\ell}} &=
\left(\frac{\sqrt{2}/2}{1/2},\;
\frac{0}{\sqrt{2}/2},\;
\frac{-\sqrt{2}/2}{1/2}\right)
= (\sqrt{2},\; 0,\; -\sqrt{2}), \\
\frac{S_{2,\ell}}{S_{0,\ell}} &=
\left(\frac{1/2}{1/2},\;
\frac{-\sqrt{2}/2}{\sqrt{2}/2},\;
\frac{1/2}{1/2}\right)
= (1,\; -1,\; 1).
\end{align*}

\emph{Step 2: Compute all fusion coefficients.}

$N_{\mathbb{I},\mathbb{I}}^k$:
\begin{align*}
N_{\mathbb{I}\mathbb{I}}^{\mathbb{I}} &=
\sum_\ell \frac{S_{0\ell}^2 S_{\ell 0}}{S_{0\ell}}
= \sum_\ell S_{0\ell} S_{\ell 0}
= \delta_{00} = 1, \\
N_{\mathbb{I}\mathbb{I}}^{\sigma} &= \delta_{0,1} = 0, \qquad
N_{\mathbb{I}\mathbb{I}}^{\varepsilon} = \delta_{0,2} = 0.
\end{align*}
By the same orthogonality argument,
$N_{\mathbb{I},j}^k = \delta_{jk}$ for all $j,k$ (identity fusion).

$N_{\sigma\sigma}^k$:
\begin{align*}
N_{\sigma\sigma}^{\mathbb{I}}
&= \frac{(\sqrt{2}/2)(\sqrt{2}/2)(1/2)}{1/2}
+ \frac{(0)(0)(\sqrt{2}/2)}{\sqrt{2}/2}
+ \frac{(-\sqrt{2}/2)(-\sqrt{2}/2)(1/2)}{1/2} \\
&= \frac{1/4}{1/2} + 0 + \frac{1/4}{1/2} = \frac{1}{2} + \frac{1}{2} = 1. \qquad \checkmark
\end{align*}

Similarly:
\begin{align*}
N_{\sigma\sigma}^{\sigma}
&= \frac{(\sqrt{2}/2)^2(\sqrt{2}/2)}{1/2}
+ \frac{0}{(\sqrt{2}/2)}
+ \frac{(-\sqrt{2}/2)^2(-\sqrt{2}/2)}{1/2} \\
&= \frac{\sqrt{2}/4}{1/2} + 0 + \frac{-\sqrt{2}/4}{1/2}
= \frac{\sqrt{2}}{2} - \frac{\sqrt{2}}{2} = 0. \qquad \checkmark
\end{align*}

\begin{align*}
N_{\sigma\sigma}^{\varepsilon}
&= \frac{(\sqrt{2}/2)^2(1/2)}{1/2}
+ \frac{0 \cdot 0 \cdot (-\sqrt{2}/2)}{\sqrt{2}/2}
+ \frac{(-\sqrt{2}/2)^2 \cdot (1/2)}{1/2} \\
&= 1/2 + 0 + 1/2 = 1. \qquad \checkmark
\end{align*}

Hence $\sigma \times \sigma = \mathbb{I} + \varepsilon$.

$N_{\sigma\varepsilon}^k$:
\begin{align*}
N_{\sigma\varepsilon}^{\mathbb{I}}
&= \frac{(\sqrt{2}/2)(1/2)(1/2)}{1/2}
+ \frac{0 \cdot (-\sqrt{2}/2) \cdot (\sqrt{2}/2)}{\sqrt{2}/2}
+ \frac{(-\sqrt{2}/2)(1/2)(1/2)}{1/2} \\
&= \frac{\sqrt{2}/4}{1/2} + 0 + \frac{-\sqrt{2}/4}{1/2}
= \frac{\sqrt{2}}{2} - \frac{\sqrt{2}}{2} = 0,
\end{align*}
\begin{align*}
N_{\sigma\varepsilon}^{\sigma}
&= \frac{(\sqrt{2}/2)(1/2)(\sqrt{2}/2)}{1/2}
+ \frac{0 \cdot (-\sqrt{2}/2) \cdot 0}{\sqrt{2}/2}
+ \frac{(-\sqrt{2}/2)(1/2)(-\sqrt{2}/2)}{1/2} \\
&= \frac{1/4}{1/2} + 0 + \frac{1/4}{1/2} = \frac{1}{2} + 0 + \frac{1}{2} = 1. \qquad \checkmark
\end{align*}

Hence $\sigma \times \varepsilon = \sigma$, as expected.
The intermediate numerators are
$(\sqrt{2}/2)(1/2)(\sqrt{2}/2) = (1/2)(1/2) = 1/4$
(and similarly for the $\ell = 2$ term), so each nonzero
summand contributes $\tfrac{1}{4}/(1/2) = \tfrac{1}{2}$.
\end{computation}

\begin{remark}[Verlinde formula and fusion matrices]
\label{rem:verlinde-normalisation}
An independent verification of the fusion rules is provided by
the simultaneous diagonalisation of the fusion matrices.  Define
$\lambda_i^{(\ell)} = S_{i\ell}/S_{0\ell}$ (the eigenvalues of the fusion
matrix $N_i$).  Then $N_{ij}^k = \sum_\ell S_{0\ell} \lambda_i^{(\ell)}
\lambda_j^{(\ell)} \overline{S_{k\ell}}/S_{0\ell}^2$.  The quantum
dimensions are $d_\sigma = \sqrt{2}$, $d_\varepsilon = 1$, and the fusion
matrices $N_\sigma$, $N_\varepsilon$ are simultaneously diagonalised by $S$:
\[
N_\sigma = \begin{pmatrix} 0 & 1 & 0 \\ 1 & 0 & 1 \\ 0 & 1 & 0 \end{pmatrix},
\qquad
N_\varepsilon = \begin{pmatrix} 0 & 0 & 1 \\ 0 & 1 & 0 \\ 1 & 0 & 0 \end{pmatrix}.
\]
These satisfy $N_\sigma = S \operatorname{diag}(\sqrt{2}, 0, -\sqrt{2}) S^{-1}$
and $N_\varepsilon = S \operatorname{diag}(1, -1, 1) S^{-1}$, which give
exactly $\sigma \times \sigma = \mathbb{I} + \varepsilon$,
$\sigma \times \varepsilon = \sigma$,
$\varepsilon \times \varepsilon = \mathbb{I}$.
\end{remark}

\begin{proposition}[Tricritical Ising S-matrix \cite{BPZ84}; \ClaimStatusProvedElsewhere]
\label{prop:tricritical-s-matrix}
\index{modular S-matrix!tricritical Ising}

The modular S-matrix of $\mathcal{M}(5,4)$ in the basis
$(\mathbb{I}, \varepsilon, \varepsilon', \varepsilon'', \sigma, \sigma')$
is determined by the CIZ formula
\[
S_{(r,s),(r',s')} = (-1)^{1+rs'+r's}\sqrt{\tfrac{8}{pq}}\,
\sin\!\bigl(\tfrac{\pi q rr'}{p}\bigr)\sin\!\bigl(\tfrac{\pi p ss'}{q}\bigr)
\]
with $(p,q) = (5,4)$.  Explicitly:

\[
S = \frac{1}{\sqrt{5}}\begin{pmatrix}
s_1 s_1 & s_1 s_2 & s_1 s_3 & s_1 s_4 & \sqrt{2}\,s_1 s_1 & \sqrt{2}\,s_1 s_2 \\
s_1 s_2 & s_1 s_4 & s_1 s_1 & -s_1 s_3 & \sqrt{2}\,s_1 s_2 & -\sqrt{2}\,s_1 s_4 \\
s_1 s_3 & s_1 s_1 & -s_1 s_4 & -s_1 s_2 & -\sqrt{2}\,s_1 s_3 & \sqrt{2}\,s_1 s_1 \\
s_1 s_4 & -s_1 s_3 & -s_1 s_2 & s_1 s_1 & -\sqrt{2}\,s_1 s_4 & -\sqrt{2}\,s_1 s_3 \\
\sqrt{2}\,s_1 s_1 & \sqrt{2}\,s_1 s_2 & -\sqrt{2}\,s_1 s_3 & -\sqrt{2}\,s_1 s_4 & 0 & 0 \\
\sqrt{2}\,s_1 s_2 & -\sqrt{2}\,s_1 s_4 & \sqrt{2}\,s_1 s_1 & -\sqrt{2}\,s_1 s_3 & 0 & 0
\end{pmatrix}
\]
where $s_j = \sin(j\pi/5)$.

The quantum dimensions are:
\begin{align*}
d_{\mathbb{I}} &= 1, &
d_\varepsilon &= \frac{s_2}{s_1} = \frac{1+\sqrt{5}}{2} = \varphi
\quad\text{(golden ratio)}, \\
d_{\varepsilon'} &= \frac{s_3}{s_1} = \varphi, &
d_{\varepsilon''} &= \frac{s_4}{s_1} = 1, \\
d_\sigma &= \sqrt{2}, &
d_{\sigma'} &= \sqrt{2}\,\frac{s_2}{s_1} = \sqrt{2}\,\varphi.
\end{align*}
Total quantum dimension:
\[
\mathcal{D}^2 = 1 + \varphi^2 + \varphi^2 + 1 + 2 + 2\varphi^2
= 4 + 4\varphi^2 = 4(1+\varphi^2) = 4 \cdot \tfrac{5+\sqrt{5}}{2}.
\]
\end{proposition}

\begin{computation}[Tricritical Ising fusion rules]
\label{comp:tricritical-fusion}

Using the fusion matrix diagonalisation
\[
N_i = S \operatorname{diag}(d_0^{-1}S_{i,\ell})_\ell\, S^{-1},
\]
the non-trivial fusion products are:
\begin{align}
\varepsilon \times \varepsilon &= \mathbb{I} + \varepsilon',
\label{eq:tri-eps-eps} \\
\varepsilon \times \varepsilon' &= \varepsilon + \varepsilon'',
\label{eq:tri-eps-epsp} \\
\varepsilon \times \varepsilon'' &= \varepsilon',
\label{eq:tri-eps-epspp} \\
\varepsilon' \times \varepsilon' &= \mathbb{I} + \varepsilon',
\label{eq:tri-epsp-epsp} \\
\varepsilon' \times \varepsilon'' &= \varepsilon,
\label{eq:tri-epsp-epspp} \\
\varepsilon'' \times \varepsilon'' &= \mathbb{I},
\label{eq:tri-epspp-epspp} \\
\sigma \times \sigma &= \mathbb{I} + \varepsilon + \varepsilon'
+ \varepsilon'', \notag \\
\sigma \times \sigma' &= \sigma + \sigma', \notag \\
\sigma' \times \sigma' &= \mathbb{I} + \varepsilon + \varepsilon'
+ \varepsilon'' + \sigma + \sigma'. \notag
\end{align}
These naive guesses are \textbf{wrong}: the quantum dimension check
$d_\sigma \cdot d_\sigma = 2$ versus
$d_{\mathbb{I}} + d_\varepsilon + d_{\varepsilon'} + d_{\varepsilon''} = 2 + 2\varphi \neq 2$
fails.  The correct fusion rules
(Di~Francesco--Mathieu--S\'en\'echal, Table~7.4) are:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|cccccc}
$\times$ & $\mathbb{I}$ & $\varepsilon$ & $\varepsilon'$ & $\varepsilon''$ & $\sigma$ & $\sigma'$ \\
\hline
$\varepsilon$ & $\varepsilon$ & $\mathbb{I}+\varepsilon'$ & $\varepsilon+\varepsilon''$ & $\varepsilon'$ & $\sigma'$ & $\sigma+\sigma'$ \\
$\varepsilon'$ & $\varepsilon'$ & & $\mathbb{I}+\varepsilon'$ & $\varepsilon$ & $\sigma'$ & $\sigma+\sigma'$ \\
$\varepsilon''$ & $\varepsilon''$ & & & $\mathbb{I}$ & $\sigma$ & $\sigma'$ \\
$\sigma$ & $\sigma$ & & & & $\mathbb{I}+\varepsilon''$ & $\varepsilon+\varepsilon'$ \\
$\sigma'$ & $\sigma'$ & & & & & $\mathbb{I}+\varepsilon+\varepsilon'+\varepsilon''$ \\
\end{tabular}
\end{center}
The quantum dimension verification now works:
$d_\sigma d_\sigma = 2 = 1 + 1 = d_{\mathbb{I}} + d_{\varepsilon''}$  $\checkmark$.
\end{computation}

\section{Bar complex interpretation of fusion rules}
\label{sec:bar-fusion-interpretation}
\index{bar complex!fusion rules}

\begin{theorem}[Fusion from bar complex on the torus; \ClaimStatusProvedHere]
\label{thm:fusion-bar-torus}
Let $\mathcal{A}$ be a rational vertex algebra with finitely many
irreducible modules $M_0 = \mathcal{A}, M_1, \ldots, M_{r-1}$.
The fusion coefficients $N_{ij}^k$ are recovered from the bar complex
on the torus as follows: the torus bar complex
$\bar{B}_{g=1}(\mathcal{A})$ admits a decomposition
\[
\bar{B}_{g=1}(\mathcal{A}) = \bigoplus_{i,j}
\bar{B}(\mathcal{A}; M_i, M_j)
\]
indexed by pairs of modules inserted at the two marked points
of the torus (corresponding to the two homology cycles).
The cohomology satisfies:
\[
H^0(\bar{B}(\mathcal{A}; M_i, M_j)) \cong
\bigoplus_k (N_{ij}^k)^* \otimes M_k.
\]
\end{theorem}

\begin{proof}
The bar complex on the torus uses propagators
$\omega_{ab} = \partial_a \log \theta_1(z_a - z_b | \tau)$,
where $\theta_1$ is the Jacobi theta function.
The degree-0 cohomology of $\bar{B}_{g=1}(\mathcal{A}; M_i, M_j)$
computes the space of conformal blocks on the torus with insertions
$M_i, M_j$.  (The Koszul hypothesis fails for minimal models---the bar complex does not concentrate in a single degree---so Theorem~\ref{thm:higher-genus-inversion} does not apply as a direct quasi-isomorphism.  The bar-fusion comparison holds at the level of the spectral sequence: the $E_2$ page of the bar complex spectral sequence computes the fusion space, even though the spectral sequence does not collapse at~$E_2$.)
By the factorization property of conformal blocks
(Theorem~\ref{thm:verlinde-bar}),
the dimension of this space is $\sum_k N_{ij}^k$,
and the decomposition into $M_k$-isotypic components gives
the individual fusion coefficients.
\end{proof}

\begin{remark}[Verlinde formula as Euler characteristic]
\label{rem:verlinde-euler}
The Verlinde formula $N_{ij}^k = \sum_\ell S_{i\ell}S_{j\ell}\overline{S_{k\ell}}/S_{0\ell}$ admits a localization interpretation: each summand corresponds to a fixed point of the $S^1$-action on $\mathcal{M}_{1,1}$, with $S_{i\ell}/S_{0\ell}$ as the equivariant weight.
\end{remark}

\begin{computation}[Ising model: bar interpretation]
\label{comp:ising-bar-interpretation}
\index{Ising model!bar complex}
For the Ising model $\mathcal{M}(4,3)$ with $c = 1/2$, $\kappa = 1/4$.  The torus bar complex recovers the fusion rules: $H^0(\bar{B}_{g=1}; \sigma, \sigma) = \mathbb{I} \oplus \varepsilon$ (reflecting $\sigma \times \sigma = \mathbb{I} + \varepsilon$), $H^0(\bar{B}_{g=1}; \sigma, \varepsilon) = \sigma$, and $H^0(\bar{B}_{g=1}; \mathbb{I}, \mathbb{I}) = \mathbb{I}$.
\end{computation}


\section{Character formulas and modular invariance}
\label{sec:character-modular}
\index{character formula!minimal model}

\begin{theorem}[Virasoro minimal model characters \cite{FF84}; \ClaimStatusProvedElsewhere]
\label{thm:minimal-model-characters}
The character of the irreducible highest weight representation
$\mathcal{L}_{r,s}$ of the Virasoro algebra at central charge
$c = 1 - 6(p-q)^2/(pq)$ is:
\[
\chi_{r,s}(\tau) = \frac{q^{h_{r,s} - c/24}}{\eta(\tau)}
\sum_{n \in \mathbb{Z}} \left(
q^{pq \cdot n^2 + (pr-qs) \cdot n}
- q^{pq \cdot n^2 + (pr+qs) \cdot n}
\right)
\]
where $q = e^{2\pi i \tau}$ and $\eta(\tau) = q^{1/24}\prod_{n=1}^\infty(1-q^n)$
is the Dedekind eta function.
\end{theorem}

\begin{computation}[Ising model characters]
\label{comp:ising-characters}
\index{Ising model!characters}

For $\mathcal{M}(4,3)$, $c = 1/2$, the three characters are:
\begin{align*}
\chi_{\mathbb{I}}(\tau) &= \frac{1}{2}\left(
\sqrt{\frac{\theta_3(\tau)}{\eta(\tau)}}
+ \sqrt{\frac{\theta_4(\tau)}{\eta(\tau)}} \right), \\
\chi_\sigma(\tau) &= \frac{1}{\sqrt{2}}
\sqrt{\frac{\theta_2(\tau)}{\eta(\tau)}}, \\
\chi_\varepsilon(\tau) &= \frac{1}{2}\left(
\sqrt{\frac{\theta_3(\tau)}{\eta(\tau)}}
- \sqrt{\frac{\theta_4(\tau)}{\eta(\tau)}} \right),
\end{align*}
where $\theta_2, \theta_3, \theta_4$ are the Jacobi theta constants.

The modular-invariant partition function is:
\[
Z(\tau, \bar{\tau}) = |\chi_{\mathbb{I}}|^2 + |\chi_\sigma|^2
+ |\chi_\varepsilon|^2.
\]
This is the diagonal (A-type) modular invariant.  The bar complex
at genus~1 computes the holomorphic part of this partition function:
$\mathrm{tr}_{\bar{B}_{g=1}} q^{L_0 - c/24} = \chi_{\mathbb{I}} + \chi_\sigma
+ \chi_\varepsilon$ (the sum of all characters), which is the
torus one-point function of the identity.
\end{computation}

\begin{remark}[Modular invariance and bar complex periodicity]
\label{rem:modular-bar-periodicity}
\index{modular invariance!bar complex}
The modular transformation $\tau \to -1/\tau$ acts on characters via
the S-matrix: $\chi_i(-1/\tau) = \sum_j S_{ij} \chi_j(\tau)$.
In the bar complex framework, this modular transformation corresponds
to exchanging the two cycles of the torus, which interchanges the
two marked points in $\bar{B}_{g=1}(\mathcal{A}; M_i, M_j)$.
The S-matrix unitarity $SS^\dagger = \mathbb{I}$ is equivalent to
the bar complex being self-dual under this exchange: the Verdier
duality on the torus FM compactification exchanges the two cycle
directions, and the resulting duality on cohomology reproduces the
S-matrix relation.
\end{remark}
