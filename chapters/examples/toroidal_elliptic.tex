\section{Double affine setup}
\label{sec:double-affine}

\subsection{Definition}

\begin{definition}[Toroidal algebra]\label{def:toroidal}
\index{toroidal algebra|textbf}
\index{double affine algebra|textbf}
The \emph{toroidal algebra} (or double affine algebra) $U_{q,t}(\hat{\hat{\mathfrak{g}}})$ 
is generated by:
\begin{itemize}
\item Horizontal currents $x^\pm_{i,n}$, $\psi^\pm_{i,n}$ (affine in one direction)
\item Vertical currents $e_i(z)$, $f_i(z)$, $\psi_i(z)$ (affine in the other direction)
\end{itemize}
subject to:
\begin{enumerate}
\item \emph{Drinfeld--type relations} in each direction:
$[h_{i,m}, x^\pm_{j,n}] = \pm a_{ij} x^\pm_{j,m+n}$, \quad
$[x^+_{i,m}, x^-_{j,n}] = \delta_{ij} (\psi^+_{i,m+n} - \psi^-_{i,m+n})/(q_i - q_i^{-1})$
\item \emph{Cross relations} coupling horizontal and vertical:
$\psi_i(z) e_j(w) = g_{ij}(z/w) \, e_j(w) \psi_i(z)$
where $g_{ij}(u) = (u - q^{a_{ij}})/(u - q^{-a_{ij}})$
\item \emph{Serre relations} in both directions
\end{enumerate}
with two independent parameters $q$ and $t = q^k$. See Ginzburg--Kapranov--Vasserot
for the complete presentation.
\end{definition}

\begin{proposition}[Toroidal OPE; \ClaimStatusProvedElsewhere]\label{prop:toroidal-ope}
The OPE for toroidal currents involves both parameters:
\[
e_i(z) f_j(w) = \frac{\delta_{ij}}{(1 - q)(1 - t^{-1})} \cdot
\frac{\psi_i^+(w) - \psi_i^-(w)}{z - w} + \text{rational terms in } q, t
\]
\end{proposition}

\begin{proof}
We derive the OPE from the defining relations of Definition~\ref{def:toroidal}.

\emph{Step~1: Mode commutator.}
The Drinfeld-type relation (1) gives:
\[
[e_{i,m}, f_{j,n}] = \delta_{ij} \frac{\psi^+_{i,m+n} - \psi^-_{i,m+n}}{q_i - q_i^{-1}}.
\]
Form the generating functions $e_i(z) = \sum_m e_{i,m} z^{-m-1}$,
$f_j(w) = \sum_n f_{j,n} w^{-n-1}$, and
$\psi_i^\pm(w) = \sum_n \psi^\pm_{i,n} w^{-n}$.

\emph{Step~2: Extracting the OPE.}
Contracting the commutator against $z^{-m-1} w^{-n-1}$ and summing:
\[
e_i(z) f_j(w) - {:}e_i(z) f_j(w){:}
= \delta_{ij} \sum_{m} \frac{\psi^+_{i,m}(w) - \psi^-_{i,m}(w)}{q_i - q_i^{-1}} \cdot \frac{w^{-m}}{z - w}.
\]
For the toroidal algebra with parameters $q$ and $t = q^k$,
we have $q_i = q^{d_i}$ where $d_i = (\alpha_i, \alpha_i)/2$.
The singular part of the OPE is the residue at $z = w$:
\[
e_i(z) f_j(w) \sim \frac{\delta_{ij}}{q_i - q_i^{-1}} \cdot
\frac{\psi_i^+(w) - \psi_i^-(w)}{z - w}.
\]

\emph{Step~3: Two-parameter structure.}
The cross relation (2) introduces the second parameter.
From $\psi_i(z) e_j(w) = g_{ij}(z/w) \, e_j(w) \psi_i(z)$
with $g_{ij}(u) = (u - q^{a_{ij}})/(u - q^{-a_{ij}})$,
the normal ordering of $\psi$ against $e, f$ currents produces
corrections rational in both $q$ and $t$.  Specifically, expanding
$1/(q_i - q_i^{-1}) = 1/((q^{d_i} - q^{-d_i}))$ and using
$t = q^k$ to express the central element as
$c = (1 - q)(1-t^{-1})/(q_i - q_i^{-1})$ (the standard Ding--Iohara
normalization).  See~\cite{DI97} for the complete change-of-basis computation.
\end{proof}

\begin{theorem}[Toroidal as $\Eone$-chiral; \ClaimStatusConjectured]\label{thm:toroidal-e1}
The toroidal algebra $U_{q,t}(\hat{\hat{\mathfrak{g}}})$ is an $\Eone$-chiral algebra
in two ways:
\begin{enumerate}[label=(\roman*)]
\item As an $\Eone$-chiral algebra on $E_\tau$ (elliptic curve with modulus $\tau = \frac{\log t}{\log q}$).
\item As a doubly-graded $\Eone$-chiral algebra on $\bC^* \times \bC^*$.
\end{enumerate}
\end{theorem}

\begin{remark}[Scope]
The $\Eone$-chiral structure is expected from the RTT presentation of
$U_{q,t}(\hat{\hat{\mathfrak{g}}})$ and the Yang--Baxter equation satisfied by
the associated $R$-matrix.  We outline the construction below.  However,
a complete proof requires verifying associativity of the chiral product
directly from the OPE relations---the standard Beilinson--Drinfeld
approach via locality and the Cousin resolution applies only to
commutative (local) chiral algebras, not to the $\Eone$ setting.
A correct proof should proceed either by constructing a factorization
algebra on $E_\tau$ whose local operators recover the toroidal relations,
or by a direct OPE associativity verification using the RTT formalism.
\end{remark}

\begin{proof}[Proof outline]
We describe the expected $\Eone$-chiral structure.

\emph{Step~1: $R$-matrix braiding.}
The cross relation (2) of Definition~\ref{def:toroidal} gives the
structure function $g_{ij}(z/w) = (z/w - q^{a_{ij}})/(z/w - q^{-a_{ij}})$.
Define the $R$-matrix-valued braiding on generating currents by
\[
\sigma_{12} \circ \mu(e_i(z), e_j(w))
= g_{ij}(z/w) \cdot \mu(e_j(w), e_i(z)).
\]
Since $g_{ij}(u)$ is a rational function with $g_{ij}(u) \neq 1$
for generic $q$ and $a_{ij} \neq 0$, the braiding is non-trivial:
the algebra is associative but not commutative, hence strictly $\Eone$.

\emph{Step~2: Associativity (gap).}
The $R$-matrix $R_{ij}(u) = g_{ij}(u) \cdot \mathrm{Id}$ satisfies
the Yang--Baxter equation trivially (being scalar-valued).
The genuine content of associativity lies in verifying that the
full matrix-valued OPE structure---including the Serre-type relations
of Definition~\ref{def:toroidal}---defines an algebra over $\chirAss$
(Definition~\ref{def:chiral-ass-operad}).  This requires an explicit
verification that triple OPEs are consistent, which we do not
carry out here.

\emph{Step~3: Realization on $E_\tau$.}
Set $\tau = \log t / \log q$ and let $E_\tau = \bC^*/q^{\bZ}$.
The generating currents $e_i(z)$, $f_i(z)$, $\psi_i(z)$ are meromorphic
on $\bC^*$ with quasi-periodicity $e_i(qz) = q^{d_i} e_i(z)$
(from the grading), so they define sections of line bundles on $E_\tau$.
The OPE of Proposition~\ref{prop:toroidal-ope} has poles at
$z/w \in q^{\bZ}$, which descend to a single point on $E_\tau$.
This defines a candidate $\Eone$-chiral algebra structure on $E_\tau$,
pending verification of Step~2.

\emph{Step~4: Realization on $\bC^* \times \bC^*$.}
The horizontal modes $x^\pm_{i,n}$ depend on the spectral
parameter $u$ and the vertical modes $e_i(z)$ depend on $z$.
Together they give fields on $\bC^*_u \times \bC^*_z$ with the
two independent gradings (by horizontal and vertical mode number)
providing the doubly-graded structure.  The $\Eone$ property in
each direction follows from Step~1 applied to each factor.

Conditional on Step~2, the $\Eone$-chiral Koszul duality framework
(Theorem~\ref{thm:e1-chiral-koszul-duality}) applies
to $U_{q,t}(\hat{\hat{\mathfrak{g}}})$ in either realization.
\end{proof}


\subsection{Koszul dual of the toroidal algebra}

\begin{conjecture}[Toroidal Koszul dual; \ClaimStatusConjectured]
\label{conj:toroidal-koszul-dual}
\index{toroidal algebra!Koszul dual}
The $\Eone$-chiral Koszul dual of the toroidal algebra
$U_{q,t}(\hat{\hat{\mathfrak{g}}})$ is isomorphic to
$U_{q^{-1},t^{-1}}(\hat{\hat{\mathfrak{g}}})$,
the toroidal algebra with inverted parameters.
\end{conjecture}

\begin{remark}[Scope and evidence]\label{rem:toroidal-koszul-evidence}
This conjecture is conditional on the $\Eone$-chiral
structure (Theorem~\ref{thm:toroidal-e1}).  The evidence is
threefold:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{RTT presentation.}  The toroidal algebra has
an RTT presentation $R_{12}(u) L_1(u) L_2(v) = L_2(v) L_1(u)
R_{12}(u)$ with $R$-matrix $R(u) = R(u; q, t)$.  By the
$\Eone$-chiral Koszul duality
(Theorem~\ref{thm:e1-chiral-koszul-duality}),
the dual algebra has $R$-matrix $R(u; q, t)^{-1}$.
For the toroidal algebra, $R(u; q, t)^{-1} = R(u; q^{-1}, t^{-1})$
(the inversion formula for the Ding--Iohara $R$-matrix;
cf.\ the analogous formula for the Yangian in
Theorem~\ref{thm:yangian-koszul-dual}).
\item \emph{Affine degeneration.}  In the limit $t \to 0$
(or equivalently $k \to -\infty$, taking the ``first loop''
degeneration), $U_{q,t}(\hat{\hat{\mathfrak{g}}}) \to
U_q(\hat{\mathfrak{g}})$ and the conjectured dual reduces to
$U_{q^{-1}}(\hat{\mathfrak{g}})$, consistent with the
$q \leftrightarrow q^{-1}$ duality for affine quantum groups
(Theorem~\ref{thm:km-quantum-groups}).
\item \emph{Central charge complementarity.}  On the
elliptic curve $E_\tau$ with modulus $\tau = \log t/\log q$,
the inversion $q \to q^{-1}$ (with $t$ fixed)
sends $\tau \to -\tau$, reversing the complex structure
of the elliptic curve.
The bar-cobar duality exchanging $E_\tau$ with $E_{-\tau}$
is the elliptic analog of the Verdier duality on configuration
spaces (Theorem~\ref{thm:bar-cobar-verdier}).
\end{enumerate}
\end{remark}

\subsection{Connection to the three main theorems}

\begin{remark}[Three theorems for toroidal algebras]
\label{rem:toroidal-three-theorems}
\index{three theorems!toroidal}
Conditional on the $\Eone$-chiral structure
(Theorem~\ref{thm:toroidal-e1}), the three main
theorems have the following expected forms:
\begin{itemize}
\item \emph{Theorem~A} (Geometric bar-cobar duality):
the bar complex $\bar{B}^{\mathrm{ell}}
(U_{q,t}(\hat{\hat{\mathfrak{g}}}))$, constructed via
the elliptic propagator
(Construction~\ref{constr:elliptic-bar}), should
compute the Koszul dual coalgebra.  The Fay-based
$d^2 = 0$ (Proposition~\ref{prop:fay-implies-d-squared})
provides the required nilpotency.
\item \emph{Theorem~B} (Bar-cobar inversion):
$\Omega(\bar{B}(U_{q,t})) \xrightarrow{\sim} U_{q,t}$
should hold by the same spectral sequence argument as in
the rational case, with the elliptic corrections of
Theorem~\ref{thm:elliptic-vs-rational} contributing to
higher pages.
\item \emph{Theorem~C} (Complementarity):
the genus-$g$ obstruction $\mathrm{obs}_g(U_{q,t}) +
\mathrm{obs}_g(U_{q^{-1},t^{-1}})$ should be controlled
by $H^*(\overline{\mathcal{M}}_g, Z(U_{q,t}))$.
Since the toroidal algebra is $\Eone$ (not $\Einf$),
the moduli space $\overline{\mathcal{M}}_g$ is replaced by
a moduli space of \emph{framed} curves (or equivalently,
surfaces with oriented intervals).  Developing this
$\Eone$ genus theory is a significant open problem
(cf.\ Remark~\ref{rem:yangian-three-theorems}).
\end{itemize}
\end{remark}


\section{Elliptic quantum groups}
\label{sec:elliptic-quantum}

\subsection{Felder's elliptic quantum group}

\begin{definition}[Elliptic quantum group]\label{def:elliptic-quantum}
\index{elliptic quantum group|textbf}
The \emph{elliptic quantum group} $E_{q,p}(\mathfrak{g})$ (Felder) is defined by:
\begin{itemize}
\item The dynamical R-matrix $R(u, \lambda)$ depending on spectral parameter $u$ 
and dynamical parameter $\lambda$.
\item The RLL relations:
\[
R(u-v, \lambda) L_1(u, \lambda) L_2(v, \lambda - h^{(1)}) = L_2(v, \lambda) L_1(u, \lambda - h^{(2)}) R(u-v, \lambda)
\]
\end{itemize}
\end{definition}

\begin{definition}[Elliptic R-matrix]\label{def:elliptic-r}
The elliptic R-matrix for $\mathfrak{sl}_2$ is:
\[
R(u, \lambda) = \begin{pmatrix}
a(u) & 0 & 0 & 0 \\
0 & b(u, \lambda) & c(u, \lambda) & 0 \\
0 & c(u, -\lambda) & b(u, -\lambda) & 0 \\
0 & 0 & 0 & a(u)
\end{pmatrix}
\]
where $a(u), b(u, \lambda), c(u, \lambda)$ are expressed via theta functions:
\begin{align*}
a(u) &= \frac{\theta(u + \eta)}{\theta(\eta)} \\
b(u, \lambda) &= \frac{\theta(u)\theta(\lambda + \eta)}{\theta(\eta)\theta(\lambda)} \\
c(u, \lambda) &= \frac{\theta(u + \lambda)\theta(\eta)}{\theta(\lambda)\theta(u + \eta)}
\end{align*}
\end{definition}


\begin{remark}[Elliptic quantum groups and bar complex]
\label{rem:elliptic-qg-bar}
\index{elliptic quantum group!bar complex}
The elliptic quantum group $E_{q,p}(\mathfrak{g})$ is the natural
target of the Kazhdan--Lusztig-type equivalence for the toroidal algebra
on $E_\tau$.  At the level of bar complexes, we expect:
\begin{enumerate}[label=\textup{(\roman*)}]
\item The bar complex of $U_{q,t}(\hat{\hat{\mathfrak{g}}})$ on $E_\tau$
computes the chiral coalgebra $E_{q,p}(\mathfrak{g})$
(the elliptic analog of the Kazhdan--Lusztig coalgebra).
\item The dynamical parameter $\lambda$ in Definition~\ref{def:elliptic-quantum}
corresponds to the B-cycle monodromy of the elliptic propagator
$\omega^{\mathrm{ell}}_{ij}$ around $E_\tau$
(this is the elliptic analog of the level shift $k \to k + h^\vee$
in the affine case).
\item The dynamical Yang--Baxter equation for $R(u, \lambda)$ is the
elliptic deformation of the $d^2 = 0$ identity
(Proposition~\ref{prop:fay-implies-d-squared}), with the dynamical
parameter arising from the non-trivial $H^1(E_\tau)$.
\end{enumerate}
These connections are conjectural
(\ClaimStatusConjectured; the missing ingredient
is the full $\Eone$-chiral structure of
Theorem~\ref{thm:toroidal-e1}).
\end{remark}

\section{Elliptic R-matrices and theta functions}
\label{sec:elliptic-r-theta}

\subsection{Theta function identities}

\begin{definition}[Jacobi theta function]\label{def:theta}
The Jacobi theta function is:
\[
\theta(u | \tau) = \sum_{n \in \bZ} (-1)^n e^{\pi i \tau n(n-1) + 2\pi i n u} =
(e^{2\pi iu}; p)_\infty (p e^{-2\pi iu}; p)_\infty (p; p)_\infty
\]
where $p = e^{2\pi i \tau}$ and $(a; q)_\infty = \prod_{n \geq 0}(1 - aq^n)$.
\end{definition}

\begin{theorem}[Fay identity at genus 1; \ClaimStatusProvedElsewhere]\label{thm:fay}
\index{Fay trisecant identity|textbf}
For any four points $u_1, u_2, u_3, u_4$ on an elliptic curve $E_\tau$, the theta function
satisfies the Fay trisecant identity:
\[
\theta(u_1 - u_2)\,\theta(u_3 - u_4) - \theta(u_1 - u_3)\,\theta(u_2 - u_4) + \theta(u_1 - u_4)\,\theta(u_2 - u_3) = 0
\]
This is a \emph{bilinear} identity (each term is a product of two theta functions), which at genus~1 is the degenerate case of the general Fay trisecant identity on higher-genus curves.  The identity is equivalent to the Riemann addition formula and reduces to the Ptolemy relation $\sin(\alpha-\beta)\sin(\gamma-\delta) - \sin(\alpha-\gamma)\sin(\beta-\delta) + \sin(\alpha-\delta)\sin(\beta-\gamma) = 0$ in the rational limit $\tau \to i\infty$.

This identity underlies the Yang--Baxter equation for elliptic R-matrices:
the star-triangle relation is a consequence of the Fay identity applied to the
structure functions of the elliptic algebra.
\end{theorem}


\subsection{Fay identity and bar nilpotency}

\begin{proposition}[Fay identity implies elliptic $d^2 = 0$;
\ClaimStatusProvedHere]\label{prop:fay-implies-d-squared}
\index{Fay trisecant identity!and bar nilpotency}
On $\overline{C}_3(E_\tau)$, the elliptic bar differential satisfies
$d^2 = 0$.  The key algebraic input is the Fay trisecant identity
(Theorem~\ref{thm:fay}), which plays the role of the Arnold relation
in the rational case (Theorem~\ref{thm:bar-nilpotency-complete}).
\end{proposition}

\begin{proof}
The elliptic bar differential on degree-$2$ elements uses the
propagator $\omega_{ij}^{\mathrm{ell}} = d\log\theta_1(z_i - z_j
| \tau)$.  The computation of $d^2$ on a degree-$1$ element
$[a] \otimes 1$ produces terms on $\overline{C}_3(E_\tau)$ involving
the $2$-form $\omega_{12}^{\mathrm{ell}} \wedge \omega_{23}^{\mathrm{ell}}
+ \omega_{23}^{\mathrm{ell}} \wedge \omega_{31}^{\mathrm{ell}}
+ \omega_{31}^{\mathrm{ell}} \wedge \omega_{12}^{\mathrm{ell}}$,
exactly as in the rational case (cf.\
\S\ref{sec:bar-nilpotency-nine-terms-complete}).

The rational Arnold relation $\eta_{12} \wedge \eta_{23}
+ \eta_{23} \wedge \eta_{31} + \eta_{31} \wedge \eta_{12} = 0$
(where $\eta_{ij} = d\log(z_i - z_j)$) is a consequence of
the partial fraction identity
$(z_1 - z_2)^{-1}(z_2 - z_3)^{-1} + \text{cyclic} = 0$.
The elliptic analog replaces $1/(z_i - z_j)$ with
$\zeta_\tau(z_i - z_j) := \partial_z \log\theta_1(z_i - z_j|\tau)$,
the Weierstrass $\zeta$-function on $E_\tau$.  The required identity
is:
\begin{equation}\label{eq:elliptic-arnold}
d\zeta_\tau(z_1{-}z_2) \wedge d\zeta_\tau(z_2{-}z_3)
+ d\zeta_\tau(z_2{-}z_3) \wedge d\zeta_\tau(z_3{-}z_1)
+ d\zeta_\tau(z_3{-}z_1) \wedge d\zeta_\tau(z_1{-}z_2) = 0.
\end{equation}
This follows from the Fay identity
(Theorem~\ref{thm:fay}).  Differentiating the Fay identity
\[
\theta(u_1{-}u_2)\,\theta(u_3{-}u_4) -
\theta(u_1{-}u_3)\,\theta(u_2{-}u_4) +
\theta(u_1{-}u_4)\,\theta(u_2{-}u_3) = 0
\]
with respect to $u_1$ and $u_4$, then setting $u_4 = u_1$, yields:
\[
\zeta_\tau(u_1{-}u_2)\,\zeta_\tau(u_2{-}u_3)
+ \zeta_\tau(u_2{-}u_3)\,\zeta_\tau(u_3{-}u_1)
+ \zeta_\tau(u_3{-}u_1)\,\zeta_\tau(u_1{-}u_2)
= \wp_\tau(u_1{-}u_2) + \wp_\tau(u_2{-}u_3) + \wp_\tau(u_3{-}u_1)
\]
where $\wp_\tau = -\zeta_\tau'$ is the Weierstrass $\wp$-function
(the identity $\zeta(a)\zeta(b) + \zeta(b)\zeta(c) + \zeta(c)\zeta(a)
= \wp(a) + \wp(b) + \wp(c)$ for $a + b + c = 0$ is a classical
consequence of the Fay identity; see e.g.\ \cite{Fay73}).
Taking the exterior derivative $d_{z_1} \wedge d_{z_2}$ of both sides,
the right-hand side $\wp(z_1{-}z_2) + \wp(z_2{-}z_3) + \wp(z_3{-}z_1)$
contributes $d\wp(z_1{-}z_2) \wedge d\wp(z_2{-}z_3) + \cdots$, but
each $\wp(z_i{-}z_j)$ depends on a single difference variable $z_i - z_j$
(not on $z_1$ and $z_2$ independently), so
$d_{z_1}\wp(z_2{-}z_3) = 0$ and $d_{z_2}\wp(z_3{-}z_1) = 0$,
and the remaining terms cancel pairwise.  Therefore the left-hand side
also vanishes under $d_{z_1} \wedge d_{z_2}$, giving exactly~\eqref{eq:elliptic-arnold}.

The remaining terms in $d^2$ (involving OPE data of the chiral
algebra $\cA$) cancel by the same mechanism as in the rational case:
the bar differential decomposes as $d = d_{\mathrm{local}} +
d_{\mathrm{global}}$, where $d_{\mathrm{local}}^2 = 0$ by the OPE
associativity of $\cA$, and the cross-terms $d_{\mathrm{local}} \circ
d_{\mathrm{global}} + d_{\mathrm{global}} \circ d_{\mathrm{local}} = 0$
by the compatibility of the OPE with the propagator (which holds for
both the rational and elliptic propagators, since the residue at a
simple pole is the same: $\Res_{z=0} \zeta_\tau(z)\,dz = 1$).
\end{proof}

\begin{remark}[Rational limit]\label{rem:fay-to-arnold}
In the rational limit $\tau \to i\infty$, the theta function
$\theta_1(z|\tau) \to 2\sin(\pi z) \to 2\pi z$ (for $|z|$ small),
so $\zeta_\tau(z) \to 1/z$ and $\wp_\tau(z) \to 1/z^2$.
The identity~\eqref{eq:elliptic-arnold} reduces to the Arnold relation
$\eta_{12} \wedge \eta_{23} + \eta_{23} \wedge \eta_{31} +
\eta_{31} \wedge \eta_{12} = 0$
(Theorem~\ref{thm:arnold-relations}), and the
bar nilpotency reduces to
Theorem~\ref{thm:bar-nilpotency-complete}.
\end{remark}


\subsection{Bar complex with theta functions}

\begin{construction}[Elliptic bar complex]\label{constr:elliptic-bar}
The bar complex of an elliptic chiral algebra incorporates theta functions:

\emph{Forms.} Replace $d\log(z_1 - z_2)$ with:
\[
\omega_{12} = d\log\theta(z_1 - z_2 | \tau)
\]
the logarithmic derivative of theta.

\emph{Differential.} The residue is computed at the zeros of $\theta$:
\[
d_{\text{res}}([a|b] \otimes \omega_{12}) = \sum_{\theta(z_0) = 0} \Res_{z_1 = z_0 + z_2}[\cdots]
\]
The zeros of $\theta(u|\tau)$ are at $u = m + n\tau$ for $m, n \in \bZ$ (the lattice points).
\end{construction}

\begin{theorem}[Elliptic vs rational homology; \ClaimStatusProvedHere]\label{thm:elliptic-vs-rational}
Let $\cA$ be a chiral algebra of central charge $c$ on an elliptic curve $E_\tau$.
The homology of the elliptic bar complex admits a decomposition:
\[
H_n(\B^{\mathrm{ell}}(\cA)) \;\cong\; H_n(\B^{\mathrm{rat}}(\cA))
\;\oplus\; \bigoplus_{k \geq 1} \mathcal{E}^{(k)}_n(\cA, \tau)
\]
where the elliptic corrections $\mathcal{E}^{(k)}_n(\cA, \tau)$
are $\cA$-modules whose dependence on $\tau$ is through
(quasi-)modular forms. Specifically:
\begin{enumerate}[label=(\roman*)]
\item The leading correction $\mathcal{E}^{(1)}_n$ is
proportional to $E_2(\tau)$, the weight-2 quasi-modular
Eisenstein series, arising from the regularization of
the elliptic propagator $\partial_z \log\theta(z|\tau)$
relative to the rational propagator $1/z$
(cf.\ Construction~\ref{constr:elliptic-bar}).
\item Higher corrections $\mathcal{E}^{(k)}_n$ for $k \geq 2$
lie in the space of modular forms $M_{2k}(\mathrm{SL}_2(\bZ))$
of weight $2k$; the first truly modular correction is an
$E_4(\tau)$-coefficient.
\item The conformal dimension $h$ of the states entering the
bar complex constrains the weights: a state of dimension $h$
contributes to $\mathcal{E}^{(k)}_n$ only for $k \leq h + n$.
\item At the rational limit $\tau \to i\infty$ (i.e., $q = e^{2\pi i\tau} \to 0$),
the Eisenstein series $E_{2k}(\tau) \to 1$ and all corrections
degenerate to constants, recovering
$H_n(\B^{\mathrm{rat}}(\cA))$ up to scalar renormalization.
\end{enumerate}
For the Heisenberg algebra ($c = 1$), the corrections reduce to the
Eisenstein series hierarchy computed in
\S\ref{sec:heisenberg-genus-expansion-eisenstein}.
\end{theorem}

\begin{proof}
We construct the decomposition via a perturbative spectral sequence
comparing the elliptic and rational bar differentials.

\emph{Step~1 (Propagator decomposition).}
The elliptic bar complex (Construction~\ref{constr:elliptic-bar})
uses the propagator $\omega_{ij}^{\mathrm{ell}} = d\log\theta_1(z_i -
z_j|\tau)$, while the rational bar complex uses $\omega_{ij}^{\mathrm{rat}}
= d\log(z_i - z_j)$.  Near $z_i = z_j$, both propagators have the same
simple pole with residue~$1$, so their difference $\delta_{ij}(\tau) :=
\omega_{ij}^{\mathrm{ell}} - \omega_{ij}^{\mathrm{rat}}$ is a regular
$1$-form.  The classical expansion of the Weierstrass $\zeta$-function
$\zeta(z;\Lambda_\tau)$ in terms of the Eisenstein--Weierstrass series
$G_{2k}(\tau)$ gives the Laurent expansion:
\[
\partial_z\log\theta_1(z|\tau)
\;=\; \frac{1}{z} + \sum_{k=1}^{\infty} c_k(\tau)\,z^{2k-1},
\]
where the correction coefficients $c_k(\tau)$ are expressed in terms
of Eisenstein series via $G_{2k+2}(\tau) = 2\zeta_{\mathrm{R}}(2k+2)\,
E_{2k+2}(\tau)$:
\begin{itemize}
\item $c_1(\tau) = -\tfrac{\pi^2}{3}E_2(\tau) \in
\widetilde{M}_2(\mathrm{SL}_2(\bZ))$ (quasi-modular, weight~$2$),
since $c_1 = -2\eta_1$ where
$\eta_1 = \zeta(1/2;\Lambda_\tau) = \tfrac{\pi^2}{6}E_2(\tau)$;
\item $c_k(\tau) = (-1)^k G_{2k+2}(\tau) \in M_{2k+2}(\mathrm{SL}_2(\bZ))$
for $k \geq 2$ (holomorphic modular forms).
\end{itemize}
The expansion converges absolutely for $|z| < \min_{\omega \in
\Lambda_\tau \setminus\{0\}} |\omega|$.

\emph{Step~2 (Correction-order filtration).}
Define the \emph{correction-order filtration}
$\mathcal{F}^p \bar{B}^{\mathrm{ell}}_n(\cA)$ by
declaring that an element has filtration degree $\geq p$ if it arises
from at least $p$ insertions of the regular correction terms
$\{c_k(\tau)\,z^{2k-1}\}_{k \geq 1}$ in the propagator expansion.
Equivalently, replace each propagator by $1/z + t\cdot(c_1 z + c_2 z^3
+ \cdots)$ and expand in powers of the formal parameter~$t$; then
$\mathcal{F}^p$ consists of terms of order~$\geq t^p$.

Since the bar differential $d^{\mathrm{ell}}$ at degree~$n$
involves at most $n$ propagator factors, we have $\mathcal{F}^{n+1}
\bar{B}^{\mathrm{ell}}_n = 0$.  The filtration is compatible
with the differential: $d^{\mathrm{ell}}(\mathcal{F}^p) \subset
\mathcal{F}^p$ (inserting the bar differential does not decrease
the number of correction insertions).

\emph{Step~3 (Spectral sequence: $E_0$ and $E_1$ pages).}
The associated graded $\mathrm{gr}^0_{\mathcal{F}}
\bar{B}^{\mathrm{ell}}(\cA)$ is the bar complex computed using only the
singular part $1/z$ of the propagator.  Since the residue extraction
in the bar differential (which encodes the OPE data of~$\cA$) depends
only on the polar part, the differential on $\mathrm{gr}^0$ coincides
with the rational bar differential $d^{\mathrm{rat}}$.  Thus:
\[
E_1^{0,n} = H_n(\bar{B}^{\mathrm{rat}}(\cA)).
\]
More generally, $E_1^{p,n} = H_n(\bar{B}^{\mathrm{rat}}(\cA)) \otimes
\mathrm{Sym}^p_{\geq 1}\{c_k(\tau)\}$, where the symmetric product is
taken over all propagator correction coefficients contributing total
weight~$p$.

\emph{Step~4 (Modular structure via Zhu's theorem).}
The differential $d_r \colon E_r^{p,n} \to E_r^{p+r,n+1}$ on the
$r$-th page arises from the interaction of $r$ correction insertions
with the bar differential.  By Zhu's modular invariance
theorem~\cite{Zhu96}, the $n$-point genus-$1$ correlation functions
$\langle V_1(z_1) \cdots V_n(z_n) \rangle_{E_\tau}$ of a rational
vertex algebra $\cA$ transform as components of a vector-valued modular
form for $\mathrm{SL}_2(\bZ)$.  Since the bar complex at degree~$n$ is
built from such correlation functions composed with the propagator form,
and the correction coefficients $c_k(\tau)$ are themselves
(quasi-)modular, the differentials $d_r$ respect the modular weight
grading.  Consequently, the $E_\infty$ terms inherit a weight
decomposition from the ring $\widetilde{M}_*(\mathrm{SL}_2(\bZ))$
of quasi-modular forms.

\emph{Step~5 (Convergence and splitting).}
The spectral sequence converges because the filtration is bounded:
$\mathcal{F}^{n+1}\bar{B}^{\mathrm{ell}}_n = 0$ at each bar
degree~$n$.  At convergence, $H_n(\bar{B}^{\mathrm{ell}}(\cA))$
admits the filtration
\[
0 = \mathcal{F}^{n+1} H_n \subset \mathcal{F}^n H_n \subset \cdots
\subset \mathcal{F}^1 H_n \subset \mathcal{F}^0 H_n
= H_n(\bar{B}^{\mathrm{ell}}(\cA))
\]
with successive quotients $\mathcal{E}^{(k)}_n := E_\infty^{k,n}$.
The filtration \emph{splits}: the modular weight provides a grading
on the coefficient ring (quasi-modular forms of different weights are
linearly independent over~$\bC$), and the $E_\infty$ terms in
different weights cannot be related by extensions.  We therefore
obtain the direct sum decomposition
$H_n(\bar{B}^{\mathrm{ell}}(\cA)) \cong H_n(\bar{B}^{\mathrm{rat}}(\cA))
\oplus \bigoplus_{k \geq 1} \mathcal{E}^{(k)}_n(\cA, \tau)$
as claimed.

Properties~(i)--(iv) now follow from the structure of the correction
coefficients:
\begin{enumerate}[label=(\roman*)]
\item The leading correction involves $c_1(\tau) = -(\pi^2/3)\,E_2(\tau)$
(quasi-modular, weight~$2$).
\item Higher corrections involve products of
$c_k(\tau) \in M_{2k+2}(\mathrm{SL}_2(\bZ))$ for $k \geq 2$.
\item A bar element of degree~$n$ involves at most $n$ propagator
factors; a state of conformal dimension~$h$ constrains the Laurent
order of each propagator insertion to $\leq 2h - 1$, giving the bound
$\mathcal{E}^{(k)}_n = 0$ for $k > h + n$.
\item As $\tau \to i\infty$: $q \to 0$ and $E_{2k}(\tau) \to 1$, so all
corrections reduce to constants, recovering $H_n(\bar{B}^{\mathrm{rat}}(\cA))$
up to renormalization.
\end{enumerate}

\emph{Verification for the Heisenberg algebra.}
For $\cA = \mathcal{H}$ ($c = 1$), the genus-$1$ two-point function is
$\langle a(z_1)a(z_2)\rangle_{E_\tau} = \kappa\,G_\tau(z_1 - z_2)$
with $G_\tau(z) = \zeta_\tau(z) + (\pi^2 E_2(\tau)/3)\,z$
(Theorem~\ref{thm:heisenberg-genus-one-complete}).  The leading
correction $\mathcal{E}^{(1)}_n \propto E_2(\tau)$ arises from the
$c_1$ term.  The complete Eisenstein hierarchy of
\S\ref{sec:heisenberg-genus-expansion-eisenstein} gives
$\mathcal{E}^{(k)}_n \propto E_{2k}(\tau)$, confirming the general
pattern.
\end{proof}

\begin{remark}[Scope]
The proof assembles three ingredients:
(i)~the Laurent expansion of the Weierstrass $\zeta$-function in
terms of Eisenstein series (\ClaimStatusProvedElsewhere, classical);
(ii)~the convergence of the correction-order spectral sequence
(bounded filtration at each bar degree); and
(iii)~Zhu's modular invariance theorem for genus-$1$ $n$-point
functions of rational vertex algebras~\cite{Zhu96}
(\ClaimStatusProvedElsewhere).  The combination and the identification
of the splitting mechanism (Step~5) are new to this work.
The Heisenberg verification uses the explicit computations of
\S\ref{sec:heisenberg-genus-expansion-eisenstein}.
\end{remark}
