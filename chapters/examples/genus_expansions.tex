\subsection{Heisenberg algebra}\label{subsec:heisenberg-all-genera}

\begin{theorem}[Heisenberg free energy at all genera;
\ClaimStatusProvedHere]\label{thm:heisenberg-all-genera}
\index{Heisenberg algebra!genus expansion}
For the Heisenberg algebra $\mathcal{H}_\kappa$
of rank~$d$ at level~$\kappa \neq 0$, the genus-$g$
free energy is:
\begin{equation}\label{eq:heisenberg-Fg}
F_g(\mathcal{H}_\kappa) = d\kappa \cdot
\frac{2^{2g-1}-1}{2^{2g-1}} \cdot
\frac{|B_{2g}|}{(2g)!}
\end{equation}
for all $g \geq 1$.  The generating function is
\begin{equation}\label{eq:heisenberg-generating}
\sum_{g=1}^{\infty} F_g(\mathcal{H}_\kappa)\, x^{2g}
= d\kappa \left(\frac{x/2}{\sin(x/2)} - 1\right).
\end{equation}
\end{theorem}

\begin{proof}
By the genus universality theorem
(Theorem~\ref{thm:genus-universality}) with
$\kappa(\mathcal{H}_\kappa) = d\kappa$
(Theorem~\ref{thm:heisenberg-obs}) and
the Faber--Pandharipande formula
(Theorem~\ref{thm:mumford-formula}).
The generating function follows from
Proposition~\ref{prop:complementarity-genus-series}.
\end{proof}

\begin{remark}[Connection to partition functions]
\label{rem:heisenberg-partition}
The genus-$g$ free energy of the bar complex
should not be confused with the physical partition
function.  For a single free boson ($d = 1$,
$\kappa = 1$, $c = 1$):
\begin{itemize}
\item $g = 1$: the physical partition function
$Z_1(\tau) = (\mathrm{Im}\,\tau)^{-1/2}|\eta(\tau)|^{-2}$
(oscillator modes $+$ zero-mode integral), while
$F_1(\mathcal{H}_1) = 1/24$ captures only the
obstruction to extending the bar complex from genus~$0$
to genus~$1$.
\item $g = 2$: the physical partition function
involves the full period matrix
$\Omega \in \mathcal{H}_2$, while
$F_2(\mathcal{H}_1) = 7/5760$ is a single
tautological intersection number on
$\overline{\mathcal{M}}_2$.
\item General~$g$: the physical partition function
$Z_g = [\det'(\Delta_g)]^{-1/2}$ depends on all
of~$\mathcal{M}_g$, while $F_g$ extracts
the single class $\kappa \cdot \lambda_g \in
H^{2g}(\overline{\mathcal{M}}_g)$.
\end{itemize}
\end{remark}

\begin{remark}[$\hat{A}$-genus and the universal generating function]\label{rem:a-hat-genus}
\index{A-hat genus@$\hat{A}$-genus!generating function}
The generating function~\eqref{eq:heisenberg-generating} for a single boson ($d = 1$, $\kappa = 1$) is
\[
\sum_{g=1}^{\infty} F_g(\mathcal{H}_1)\, x^{2g} = \frac{x/2}{\sin(x/2)} - 1.
\]
Including the leading term, $1 + \sum_{g \geq 1} F_g x^{2g} = x/2 / \sin(x/2)$, which is precisely the $\hat{A}$-genus (the Todd class of the complexified tangent bundle, evaluated on $x$):
\begin{equation}\label{eq:a-hat-identification}
\hat{A}(x) = \frac{x/2}{\sinh(x/2)} \bigg|_{x \mapsto ix} = \frac{x/2}{\sin(x/2)}.
\end{equation}
This identification is not coincidental.  The Faber--Pandharipande formula evaluates tautological integrals $\int_{\overline{\mathcal{M}}_{g,1}} \psi^{2g-2} \lambda_g$ via the Bernoulli numbers, which are the Taylor coefficients of $x/(e^x - 1)$---the same function that generates the Todd class.  The bar-complex free energy is thus controlled by the universal multiplicative genus associated to the formal group law of the exponential function.

For general $\kappa$, $\sum_g F_g x^{2g} = \kappa(\hat{A}(x) - 1)$: the total genus expansion is $\kappa$ times the reduced $\hat{A}$-genus.
\end{remark}

\subsection{$\beta\gamma$-system}\label{subsec:betagamma-all-genera}

\begin{proposition}[$\beta\gamma$ genus expansion;
\ClaimStatusProvedHere]\label{prop:betagamma-all-genera}
\index{beta-gamma system@$\beta\gamma$-system!genus expansion}
For the $\beta\gamma$-system of conformal weights
$(\lambda, 1-\lambda)$:
\begin{enumerate}[label=\textup{(\roman*)}]
\item The genus-$g$ free energy is
$F_g(\beta\gamma) = \kappa(\beta\gamma) \cdot
\lambda_g^{\mathrm{FP}}$, where $\kappa(\beta\gamma)$
depends on the conformal weight~$\lambda$ via the
genus-$1$ curvature.
\item At genus $g \geq 2$, the physical partition function
\begin{equation}\label{eq:betagamma-partition}
Z_g^{\beta\gamma}(\lambda) =
\bigl[\det(\mathrm{Im}\,\Omega)\bigr]^{(1-2\lambda)(1-g)/2}
\cdot \bigl[\det'(\bar{\partial}_\lambda)\bigr]^{-1}
\end{equation}
depends on the full period matrix
$\Omega \in \mathcal{H}_g$, where
$\bar{\partial}_\lambda$ acts on sections of
$\omega_X^{\otimes \lambda}$ and the Riemann--Roch
index $\chi(\omega^{\otimes \lambda}) =
(2\lambda-1)(g-1)$ governs the zero-mode structure.
\item At genus~$1$, $Z_1^{\beta\gamma} =
|\eta(\tau)|^{-2}$ is independent of~$\lambda$
because $\omega_{E_\tau} \cong \mathcal{O}_{E_\tau}$.
\end{enumerate}
\end{proposition}

\begin{proof}
Part~(i) is a direct application of genus universality
(Theorem~\ref{thm:genus-universality}).
Part~(ii) is the standard result from the
Quillen determinant line bundle computation
(\ClaimStatusProvedElsewhere).
Part~(iii) follows from the triviality of the
canonical bundle on an elliptic curve.
\end{proof}

\section{Lattice VOAs at all genera}\label{sec:lattice-genera}
\index{lattice VOA!genus expansion}

\begin{theorem}[Lattice VOA free energy;
\ClaimStatusProvedHere]\label{thm:lattice-all-genera}
For a positive-definite even lattice $\Lambda$ of
rank~$d$, the lattice vertex algebra
$V_\Lambda = \mathcal{H}_1^{\otimes d} \otimes
\mathbb{C}[\Lambda]$
has genus-$g$ free energy:
\begin{equation}\label{eq:lattice-Fg}
F_g(V_\Lambda) = d \cdot
\frac{2^{2g-1}-1}{2^{2g-1}} \cdot
\frac{|B_{2g}|}{(2g)!}
\end{equation}
In particular, $F_g(V_\Lambda)$ depends only on
$\mathrm{rank}(\Lambda) = d$, not on the lattice
structure.
\end{theorem}

\begin{proof}
The lattice VOA is an extension of the rank-$d$
Heisenberg algebra $\mathcal{H}_1^{\otimes d}$ by
lattice vertex operators.  Since the lattice vertex
operators have conformal weight $\langle v, v
\rangle / 2 \geq 1$ for $v \neq 0$, they do not
contribute to the genus-$1$ curvature (which is
determined by the singular part of the OPE,
Theorem~\ref{thm:heisenberg-obs}).
Therefore $\kappa(V_\Lambda) =
\kappa(\mathcal{H}_1^{\otimes d}) = d$.
By genus universality
(Theorem~\ref{thm:genus-universality}),
$F_g(V_\Lambda) = d \cdot \lambda_g^{\mathrm{FP}}$.
\end{proof}

\begin{corollary}[Lattice-independence of genus expansion; \ClaimStatusProvedHere]
\label{cor:lattice-rank-only}
\index{lattice VOA!rank dependence}
For two positive-definite even lattices $\Lambda_1, \Lambda_2$ of the same rank $d$, the lattice VOAs $V_{\Lambda_1}$ and $V_{\Lambda_2}$ have identical genus-$g$ free energies:
\[
F_g(V_{\Lambda_1}) = F_g(V_{\Lambda_2}) = d \cdot \lambda_g^{FP}
\quad \text{for all } g \geq 1.
\]
In particular, the Leech lattice VOA $V_{\Lambda_{24}}$ and the Niemeier lattice VOA $V_{D_{24}^+}$ (both rank $24$) have the same bar-complex free energy at every genus, despite having different automorphism groups and different partition functions.
\end{corollary}

\begin{remark}[Partition functions vs.\ free energy]
\label{rem:lattice-partition-vs-free}
While the bar-complex free energy~\eqref{eq:lattice-Fg}
depends only on~$d$, the physical partition function
encodes the full lattice structure:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Genus~$1$:} $Z_1(V_\Lambda, \tau) =
\Theta_\Lambda(\tau)/\eta(\tau)^d$, where
$\Theta_\Lambda(\tau) = \sum_{v \in \Lambda}
q^{\langle v,v \rangle/2}$ is the lattice theta
function.
\item \emph{General~$g$:} $Z_g(V_\Lambda, \Omega) =
\Theta_\Lambda^{(g)}(\Omega)/F_g^{\mathrm{osc}}(\Omega)$,
where $\Theta_\Lambda^{(g)}(\Omega) =
\sum_{v \in \Lambda^g}
\exp(\pi i \, {}^tv \Omega v)$ is the degree-$g$
Siegel theta series and $F_g^{\mathrm{osc}}$
is the oscillator determinant.
\end{enumerate}
For unimodular even $\Lambda$
($\Lambda = \Lambda^*$,
$\langle v, v \rangle \in 2\mathbb{Z}$),
the Siegel theta series transforms as
$\Theta_\Lambda(\gamma \cdot \Omega) =
\det(C\Omega + D)^{d/2}\,\Theta_\Lambda(\Omega)$
for $\gamma = \bigl(\begin{smallmatrix}
A & B \\ C & D \end{smallmatrix}\bigr)
\in \mathrm{Sp}(2g, \mathbb{Z})$.
For non-unimodular lattices, the transformation
involves the Weil representation of
$\mathrm{Sp}(2g, \mathbb{Z})$ on
$\mathbb{C}[\Lambda^*/\Lambda]$.
\end{remark}

\section{$\mathcal{W}$-algebra free energy and Higgs bundles}
\label{sec:w-algebra-genera}
\index{W-algebra@$\mathcal{W}$-algebra!genus expansion}

\begin{theorem}[$\mathcal{W}$-algebra free energy at all genera;
\ClaimStatusProvedHere]\label{thm:w-algebra-all-genera}
For the principal $\mathcal{W}$-algebra
$\mathcal{W}^k(\mathfrak{g})$ at generic level
$k \neq -h^\vee$, the genus-$g$ free energy is
\begin{equation}\label{eq:w-Fg-general}
F_g\bigl(\mathcal{W}^k(\mathfrak{g})\bigr)
= \kappa\bigl(\mathcal{W}^k(\mathfrak{g})\bigr)
\cdot \frac{2^{2g-1}-1}{2^{2g-1}} \cdot
\frac{|B_{2g}|}{(2g)!}
\end{equation}
where $\kappa(\mathcal{W}^k(\mathfrak{g}))$ is
determined by the genus-$1$ curvature of the bar
complex and depends on the central charge via the
Drinfeld--Sokolov formula.  The complementarity sum
is
\[
\kappa\bigl(\mathcal{W}^k(\mathfrak{g})\bigr) +
\kappa\bigl(\mathcal{W}^{k'}(\mathfrak{g})\bigr)
\]
where $k' = -k - 2h^\vee$ is the Feigin--Frenkel
dual level.
\end{theorem}

\begin{proof}
Genus universality
(Theorem~\ref{thm:genus-universality}) applies
to $\mathcal{W}^k(\mathfrak{g})$ as a Koszul
chiral algebra
(Theorem~\ref{thm:w-algebra-koszul}).
The obstruction coefficient is read off from the
genus-$1$ bar complex curvature, and the
Faber--Pandharipande formula
(Theorem~\ref{thm:mumford-formula}) evaluates
the tautological integral.
\end{proof}

\begin{remark}[Connections to Higgs bundle geometry]
\label{rem:w-higgs-connections}
\index{Higgs bundle!W-algebra connection}
The $\mathcal{W}$-algebra genus expansion connects to
Hitchin geometry at three levels:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Classical limit.}
The Poisson algebra underlying
$\mathcal{W}^k(\mathfrak{g})$ is the algebra of
functions on the Hitchin base
$\mathcal{B}_g(\mathfrak{g}) =
\bigoplus_{i=1}^r H^0(\Sigma_g,
\omega_{\Sigma_g}^{\otimes d_i})$, where
$d_1, \ldots, d_r$ are the degrees of the
fundamental invariants of~$\mathfrak{g}$.
\item \emph{Symplectic volume.}
The Verlinde formula gives
$\dim \mathbb{V}_{g,k}(G) =
\int_{\mathcal{M}_{\mathrm{Higgs}}^g(\mathfrak{g})}
\exp(k\,\omega_{\mathrm{Hitchin}})$ at leading
order in~$k$, relating conformal block dimensions
to the symplectic volume of the Hitchin moduli
space.
\item \emph{AGT correspondence
\textup{(\ClaimStatusConjectured; physics)}:}
the genus-$g$ partition function of
$\mathcal{W}^k(\mathfrak{g})$ equals the
Nekrasov partition function of the corresponding
$4$d $\mathcal{N} = 2$ theory on~$\Sigma_g$.
This involves the full quantum structure, not
just the symplectic volume.
\end{enumerate}
At critical level $k = -h^\vee$, the free energy
vanishes ($\kappa = 0$) and the genus expansion
degenerates.  In this regime, the geometric
Langlands correspondence (\ClaimStatusProvedElsewhere;
Gaitsgory~\cite{Gai19}) identifies
$\mathcal{D}$-modules on $\mathrm{Bun}_G(\Sigma_g)$
with quasi-coherent sheaves on
$\mathrm{Loc}_{G^\vee}(\Sigma_g)$.
\end{remark}

\begin{conjecture}[$\mathcal{W}_3$ bar cohomology algebraicity; \ClaimStatusConjectured]
\label{conj:w3-algebraicity}
\index{W3 algebra@$\mathcal{W}_3$!bar generating function}
The bar cohomology generating function $P_{\mathcal{W}_3}(x) = \sum_{n \geq 1} \dim H^n(\barBgeom(\mathcal{W}_3^k))\, x^n$ is an algebraic function over $\mathbb{Q}(x)$.  The first four values $2, 5, 16, 52$ are consistent with a degree-$2$ algebraic function (quadratic over $\mathbb{Q}(x)$), but do not yet determine it uniquely.  If the DS discriminant principle (Remark~\ref{rem:ds-discriminant-invariant}) holds, the discriminant of $P_{\mathcal{W}_3}$ should match that of $P_{\widehat{\mathfrak{sl}}_3}$.

\emph{Scope.}  Verification requires computing $\dim H^5(\barBgeom(\mathcal{W}_3^k))$, which involves the $\mathcal{W}_3$ bar complex at degree $5$ (a computation not yet carried out).
\end{conjecture}

\begin{conjecture}[Non-simply-laced discriminant; \ClaimStatusConjectured]
\label{conj:non-simply-laced-discriminant}
\index{bar complex!generating function!non-simply-laced}
For non-simply-laced Lie algebras $\mathfrak{g}$ ($B_n$, $C_n$, $F_4$, $G_2$), the bar cohomology generating function $P_{\widehat{\mathfrak{g}}}(x)$ is algebraic with discriminant $\Delta_{\mathfrak{g}}(x)$ satisfying:
\begin{enumerate}[label=\textup{(\roman*)}]
\item $\Delta_{\mathfrak{g}}$ has a simple root at $x = 1/\dim\mathfrak{g}$ (determining the exponential growth rate);
\item the DS discriminant $\Delta_{\mathcal{W}(\mathfrak{g})}$ shares the same branch locus as $\Delta_{\mathfrak{g}}$;
\item for $G_2$: $\dim\mathfrak{g} = 14$, so the predicted growth rate is $14^n$.
\end{enumerate}

\emph{Scope.}  Even for $G_2$, only three data points are available (bar cohomology $14, 196, 2744 = 14^n$, which are \emph{chain-group} dimensions, not bar cohomology).  Computing $H^n(\barBgeom(\widehat{G}_{2,k}))$ requires the $G_2$ structure constants and is computationally demanding.
\end{conjecture}

\begin{conjecture}[Yangian bar generating function; \ClaimStatusConjectured]
\label{conj:yangian-non-algebraic}
\index{Yangian!bar generating function}
The bar cohomology generating function of $Y(\mathfrak{sl}_2)$ with values $4, 10, 28, \ldots$ (Table~\ref{tab:bar-dimensions}) is \emph{not} algebraic over $\mathbb{Q}(x)$, reflecting the $\Eone$-chiral (rather than $\Einf$-chiral) nature of the Yangian.  The absence of the locality axiom removes the constraint that forces algebraicity in the $\Einf$ case (Corollary~\ref{cor:algebraicity-koszul}).

\emph{Scope.}  Verification requires either computing further Yangian bar cohomology dimensions (degree $\geq 4$) or proving a general non-algebraicity criterion for $\Eone$-chiral algebras.
\end{conjecture}

\subsection{Genus-$g$ free energy}\label{subsec:sl2-free-energy}

\begin{theorem}[$\widehat{\mathfrak{sl}}_2$ free energy at all genera; \ClaimStatusProvedHere]\label{thm:sl2-all-genera}
For $\widehat{\mathfrak{sl}}_{2,k}$ at generic level $k \neq -2$, the genus-$g$ free energy is:
\begin{equation}\label{eq:sl2-Fg}
F_g(\widehat{\mathfrak{sl}}_{2,k}) = \frac{3(k+2)}{4} \cdot \frac{2^{2g-1}-1}{2^{2g-1}} \cdot \frac{|B_{2g}|}{(2g)!}
\end{equation}
where $B_{2g}$ are the Bernoulli numbers ($B_2 = \frac{1}{6}$, $B_4 = -\frac{1}{30}$, $B_6 = \frac{1}{42}$, $B_8 = -\frac{1}{30}$, $B_{10} = \frac{5}{66}$).
\end{theorem}

\begin{proof}
By the genus universality theorem (Theorem~\ref{thm:genus-universality}), the genus-$g$ obstruction class factors as $\mathrm{obs}_g(\cA) = \kappa(\cA) \cdot \lambda_g$, where $\lambda_g = c_g(\mathbb{E})$ is the top Chern class of the Hodge bundle on $\overline{\mathcal{M}}_g$.  For $\widehat{\mathfrak{sl}}_{2,k}$, the genus-1 curvature computation (Theorem~\ref{thm:sl2-genus1-curvature}) gives $\kappa = 3(k+2)/4$.

The free energy is the tautological integral:
\[
F_g(\cA) = \int_{\overline{\mathcal{M}}_{g,1}} \psi_1^{2g-2}\, \mathrm{obs}_g(\cA) = \kappa(\cA) \cdot \int_{\overline{\mathcal{M}}_{g,1}} \psi_1^{2g-2}\, \lambda_g
\]
The Faber--Pandharipande formula (Theorem~\ref{thm:mumford-formula}) evaluates the integral:
\[
\int_{\overline{\mathcal{M}}_{g,1}} \psi_1^{2g-2}\, \lambda_g = \frac{2^{2g-1}-1}{2^{2g-1}} \cdot \frac{|B_{2g}|}{(2g)!}
\]
Substituting $\kappa = 3(k+2)/4$ yields~\eqref{eq:sl2-Fg}.
\end{proof}

\subsection{Complementarity at every genus}\label{subsec:sl2-complementarity}

\begin{proposition}[$\widehat{\mathfrak{sl}}_2$ complementarity; \ClaimStatusProvedHere]\label{prop:sl2-complementarity-all-genera}
For all $g \geq 1$ and all $k \neq -2$:
\begin{equation}\label{eq:sl2-complementarity-allgenera}
F_g(\widehat{\mathfrak{sl}}_{2,k}) + F_g(\widehat{\mathfrak{sl}}_{2,-k-4}) = 0
\end{equation}
\end{proposition}

\begin{proof}
The obstruction coefficients satisfy:
\[
\kappa(\widehat{\mathfrak{sl}}_{2,k}) + \kappa(\widehat{\mathfrak{sl}}_{2,-k-4}) = \frac{3(k+2)}{4} + \frac{3(-k-4+2)}{4} = \frac{3(k+2)}{4} + \frac{3(-k-2)}{4} = 0
\]
Since $F_g$ is linear in $\kappa$, the result follows.
\end{proof}

\begin{remark}[Comparison with Virasoro complementarity]\label{rem:sl2-vs-vir-complementarity}
For $\widehat{\mathfrak{sl}}_2$, the obstruction coefficients cancel exactly ($\kappa + \kappa' = 0$), a consequence of the symmetric level-shift $k \mapsto -k-4$ fixing the midpoint $k = -2$.  For the Virasoro algebra $\mathrm{Vir}_c \simeq \mathcal{W}^k(\mathfrak{sl}_2)$, the complementarity takes a different form:
\[
\kappa(\mathrm{Vir}_c) + \kappa(\mathrm{Vir}_{26-c}) = \frac{c}{2} + \frac{26-c}{2} = 13
\]
so $F_g(\mathrm{Vir}_c) + F_g(\mathrm{Vir}_{26-c}) = 13 \cdot \lambda_g^{\mathrm{FP}}$, where $\lambda_g^{\mathrm{FP}} = \frac{2^{2g-1}-1}{2^{2g-1}} \cdot \frac{|B_{2g}|}{(2g)!}$.  The nonzero sum $\kappa + \kappa' = 13$ reflects the conformal anomaly of the $c = 26$ bosonic string: the combined obstruction of Virasoro and its dual equals half the bosonic string central charge times $\lambda_g^{\mathrm{FP}}$.

For the $\mathcal{W}_3$ algebra (where $c + c' = 100$), the analogous computation gives $\kappa + \kappa' = 250/3$.
\end{remark}

\begin{proposition}[Bivariate generating function; \ClaimStatusProvedHere]
\label{prop:bivariate-gf}
\index{generating function!bivariate}
The full genus-expanded free energy of $\widehat{\mathfrak{sl}}_{2,k}$ has a closed-form bivariate generating function in the genus variable $x$ and the level variable $t = k + 2$:
\begin{equation}\label{eq:bivariate-gf}
\sum_{g=1}^{\infty} F_g(\widehat{\mathfrak{sl}}_{2,k})\, x^{2g} = \frac{3t}{4}\left(\frac{x/2}{\sin(x/2)} - 1\right).
\end{equation}
This is linear in $t$ (hence in $k$) and is the product of the algebra-dependent factor $3t/4 = \kappa$ with the universal generating function $\hat{A}(x) - 1$.  For any Koszul chiral algebra, the analogous formula is
\begin{equation}\label{eq:bivariate-gf-general}
\sum_{g=1}^{\infty} F_g(\cA)\, x^{2g} = \kappa(\cA) \cdot \left(\frac{x/2}{\sin(x/2)} - 1\right).
\end{equation}
The bivariate structure reflects the factorization $F_g = \kappa \cdot \lambda_g^{FP}$ (Theorem~\ref{thm:genus-universality}).
\end{proposition}

\begin{proof}
Substituting $F_g = \kappa \cdot \lambda_g^{FP}$ into the series and using the classical identity $\sum_{g \geq 1} \lambda_g^{FP} x^{2g} = x/(2\sin(x/2)) - 1$ (which follows from the generating function for Bernoulli numbers: $\sum_{g \geq 1} \frac{(2^{2g-1}-1)|B_{2g}|}{(2g)!} x^{2g} = x/(2\sin(x/2)) - 1$).
\end{proof}

\subsection{Explicit genus-2 bar differential}\label{subsec:sl2-genus2-bar}
\index{bar complex!genus-2 differential}
\index{sl2@$\mathfrak{sl}_2$!genus-2 bar differential}

The genus-1 curvature of $\widehat{\mathfrak{sl}}_{2,k}$ was computed in
Theorem~\ref{thm:sl2-genus1-curvature} by extracting the B-cycle
monodromy of the elliptic propagator.  We now carry out the analogous
computation at genus~$2$: the first \emph{non-abelian} genus-$2$ bar
differential.  The Heisenberg case
(Theorem~\ref{thm:heisenberg-genus2-obstruction}) serves as template;
the new content is the interplay between the double-pole
coefficient~$k$ and the structure constants~$f^{ab}{}_c$, which
produces the dual Coxeter correction $h^\vee = 2$ in the curvature.

\subsubsection{Genus-2 propagator for affine Kac--Moody algebras}

\begin{proposition}[Non-abelian genus-2 propagator; \ClaimStatusProvedHere]
\label{prop:km-genus2-propagator}
\index{propagator!genus-2 non-abelian}
Let $\widehat{\mathfrak{g}}_k$ be an affine Kac--Moody algebra on a
genus-$2$ surface $\Sigma_2$ with period matrix
$\Omega \in \mathfrak{H}_2$ and normalized abelian differentials
$\omega_1, \omega_2$.  The genus-$2$ OPE propagator is:
\begin{equation}\label{eq:km-genus2-ope}
J^a(z) J^b(w) \sim
  k\,\kappa^{ab} \cdot \bigl(-\partial_w K^{(2)}(z,w|\Omega)\bigr)
  + f^{ab}{}_{c}\, J^c(w) \cdot K^{(2)}(z,w|\Omega)
\end{equation}
where $K^{(2)}(z,w|\Omega) = \partial_z \log E(z,w;\Omega)$
is the genus-$2$ chiral propagator
\textup{(}Definition~\textup{\ref{def:prime-form-g2})},
$\kappa^{ab} = (J^a, J^b)$ is the normalized Killing form,
and $f^{ab}{}_c$ are the structure constants.
The B-cycle monodromies are:
\begin{align}
K^{(2)}(z + \delta_{B_j}, w|\Omega)
  &= K^{(2)}(z, w|\Omega) - 2\pi i\, \omega_j(w)
  \label{eq:km-genus2-K-mono} \\
\partial_w K^{(2)}(z + \delta_{B_j}, w|\Omega)
  &= \partial_w K^{(2)}(z, w|\Omega) - 2\pi i\, \partial_w\omega_j(w)
  \label{eq:km-genus2-dK-mono}
\end{align}
for $j = 1, 2$.
\end{proposition}

\begin{proof}
The OPE~\eqref{eq:km-genus2-ope} is the standard replacement of the
flat-space propagator $1/(z-w)$ by the genus-$2$ propagator $K^{(2)}$:
the double pole $k\,\kappa^{ab}/(z-w)^2 = -k\,\kappa^{ab}\,\partial_w[1/(z-w)]$
becomes $-k\,\kappa^{ab}\,\partial_w K^{(2)}$, and the simple pole
$f^{ab}{}_c J^c(w)/(z-w)$ becomes $f^{ab}{}_c J^c(w) \cdot K^{(2)}$.
The B-cycle monodromy~\eqref{eq:km-genus2-K-mono} is
equation~\eqref{eq:genus2-B-quasi} from the Heisenberg computation.
Equation~\eqref{eq:km-genus2-dK-mono} follows by differentiation.
\end{proof}

\begin{remark}[Genus-1 vs genus-2 monodromies]\label{rem:g1-g2-mono-contrast}
At genus~$1$, the single B-cycle monodromy of $K^{(1)}$ is the
\emph{constant} $-2\pi i$, so $\partial_w K^{(1)}$ is doubly periodic
and only the simple-pole term contributes to the curvature.  At
genus~$2$, both B-cycle monodromies are \emph{function-valued}
(proportional to the abelian differentials $\omega_j$), so
$\partial_w\omega_j \neq 0$ in general and the \emph{double-pole term
also acquires a monodromy defect}.  This is the mechanism by which the
level~$k$ enters the genus-$2$ curvature directly, in addition to the
dual Coxeter correction from the structure constants.
\end{remark}

\subsubsection{The differential $d^{(2)}$ on $\bar{B}(\widehat{\mathfrak{sl}}_{2,k})$}

\begin{theorem}[Genus-2 bar differential for $\widehat{\mathfrak{sl}}_2$;
\ClaimStatusProvedHere]\label{thm:sl2-genus2-bar-differential}
\index{bar complex!genus-2 for sl2@genus-2 for $\mathfrak{sl}_2$}
Fix a basis $\{e, f, h\}$ of $\mathfrak{sl}_2$ with
$[e,f] = h$, $[h,e] = 2e$, $[h,f] = -2f$, and Killing form
$\kappa^{ef} = \kappa^{fe} = 1$, $\kappa^{hh} = 2$
\textup{(}all other entries zero\textup{)}.
The genus-$2$ component of the bar differential,
$d^{(2)}\colon \bar{B}^n(\widehat{\mathfrak{sl}}_{2,k}) \to
\bar{B}^{n-1}(\widehat{\mathfrak{sl}}_{2,k})$, is determined by its
action on degree-$2$ elements:
\begin{equation}\label{eq:sl2-d2-formula}
d^{(2)}\bigl(J^a \boxtimes J^b \cdot \eta_{12}\bigr)
= \sum_{j=1}^{2} \Bigl[
  -2\pi i \cdot k\,\kappa^{ab}\, \partial_w\omega_j
  \;+\; (-2\pi i) \cdot f^{ab}{}_{c}\, J^c \cdot \omega_j
\Bigr] \otimes [\beta_j]
\end{equation}
where $[\beta_j] \in H_1(\Sigma_2, \mathbb{Z})$ denotes the
cohomology class dual to the $j$-th B-cycle monodromy, and
$\omega_j$ is the $j$-th normalized abelian differential.
\end{theorem}

\begin{proof}
The bar differential at genus~$g$ decomposes as
$d = d^{(0)} + d^{(1)} + d^{(2)} + \cdots$
(Theorem~\ref{thm:genus-differential}).  The component $d^{(2)}$ is
the genus-$2$ correction arising from the B-cycle monodromies of the
propagator on~$\Sigma_2$.

\emph{Step~1: Double-pole contribution.}
The double-pole OPE $k\,\kappa^{ab}\, (-\partial_w K^{(2)})$ acquires
B-cycle monodromy $-2\pi i\, k\,\kappa^{ab}\, \partial_w\omega_j$
via~\eqref{eq:km-genus2-dK-mono}.  At genus~$1$, the analogous
monodromy $\partial_w\omega_\tau$ vanishes because $\omega_\tau = dz$
is constant on the elliptic curve, so the double pole does not
contribute to the genus-$1$ curvature.  At genus~$2$, the abelian
differentials $\omega_j$ are non-constant holomorphic sections of
$\omega_{\Sigma_2}$, so $\partial_w\omega_j \neq 0$ and the double
pole contributes.

\emph{Step~2: Simple-pole contribution.}
The simple-pole OPE $f^{ab}{}_c\, J^c \cdot K^{(2)}$ acquires
B-cycle monodromy $-2\pi i\, f^{ab}{}_c\, J^c \cdot \omega_j$
via~\eqref{eq:km-genus2-K-mono}.  This is identical in form to the
Heisenberg contribution (with $f^{ab}{}_c\, J^c$ replacing the abelian
level~$\kappa$) and is the mechanism by which the structure constants
of~$\mathfrak{sl}_2$ enter the genus-$2$ bar differential.

\emph{Step~3: Assembly.}
Summing over both B-cycles $j = 1, 2$
gives~\eqref{eq:sl2-d2-formula}.  The first term is proportional to the
level~$k$ and the Killing form~$\kappa^{ab}$ (the ``double-pole
channel''); the second is proportional to the structure
constants~$f^{ab}{}_c$ (the ``simple-pole channel'').
\end{proof}

\begin{computation}[Explicit action on basis elements]
\label{comp:sl2-d2-basis}
Evaluating~\eqref{eq:sl2-d2-formula} on each pair of
$\mathfrak{sl}_2$ generators.  For
the non-trivial Killing form entries $\kappa^{ef} = 1$, $\kappa^{hh} = 2$:

\medskip\noindent
\emph{Pair $(e, f)$:} $\kappa^{ef} = 1$, $f^{ef}{}_c J^c = h$.
\begin{equation}\label{eq:d2-ef}
d^{(2)}(e \boxtimes f \cdot \eta_{12})
= \sum_{j=1}^{2} \bigl[
  -2\pi i \cdot k\, \partial_w\omega_j
  + (-2\pi i) \cdot h \cdot \omega_j
\bigr] \otimes [\beta_j]
\end{equation}

\emph{Pair $(h, h)$:} $\kappa^{hh} = 2$, $f^{hh}{}_c = 0$.
\begin{equation}\label{eq:d2-hh}
d^{(2)}(h \boxtimes h \cdot \eta_{12})
= \sum_{j=1}^{2} \bigl[
  -2\pi i \cdot 2k\, \partial_w\omega_j
\bigr] \otimes [\beta_j]
\end{equation}
Only the double-pole channel contributes (the structure constants
$f^{hh}{}_c = 0$ vanish by the Cartan subalgebra property).

\medskip\noindent
\emph{Pair $(h, e)$:} $\kappa^{he} = 0$, $f^{he}{}_c J^c = 2e$.
\begin{equation}\label{eq:d2-he}
d^{(2)}(h \boxtimes e \cdot \eta_{12})
= \sum_{j=1}^{2} (-2\pi i) \cdot 2e \cdot \omega_j \otimes [\beta_j]
\end{equation}
Only the simple-pole channel contributes (the Killing form
$\kappa^{he} = 0$).

\medskip\noindent
\emph{Pair $(h, f)$:} $\kappa^{hf} = 0$, $f^{hf}{}_c J^c = -2f$.
\begin{equation}\label{eq:d2-hf}
d^{(2)}(h \boxtimes f \cdot \eta_{12})
= \sum_{j=1}^{2} (-2\pi i) \cdot (-2f) \cdot \omega_j \otimes [\beta_j]
\end{equation}

\medskip\noindent
\emph{Pair $(e, e)$:} $\kappa^{ee} = 0$, $f^{ee}{}_c = 0$.
\begin{equation}\label{eq:d2-ee}
d^{(2)}(e \boxtimes e \cdot \eta_{12}) = 0
\end{equation}
Both channels vanish, since $e$ is a root vector with vanishing
self-Killing form and $[e, e] = 0$.  Similarly
$d^{(2)}(f \boxtimes f \cdot \eta_{12}) = 0$.
\end{computation}

\subsubsection{Curvature at genus 2}

\begin{theorem}[Genus-2 curvature for $\widehat{\mathfrak{sl}}_2$;
\ClaimStatusProvedHere]\label{thm:sl2-genus2-curvature}
\index{curvature!genus-2 for sl2@genus-2 for $\mathfrak{sl}_2$}
The genus-$2$ obstruction class of $\widehat{\mathfrak{sl}}_{2,k}$ is:
\begin{equation}\label{eq:sl2-obs2}
\mathrm{obs}_2(\widehat{\mathfrak{sl}}_{2,k})
= \frac{3(k+2)}{4} \cdot \lambda_2
\in H^4(\overline{\mathcal{M}}_2,\, Z(\widehat{\mathfrak{sl}}_{2,k}))
\end{equation}
where $\lambda_2 = c_2(\mathbb{E})$ is the second Chern class of the
rank-$2$ Hodge bundle.  The coefficient $3(k+2)/4 = \kappa(\widehat{\mathfrak{sl}}_{2,k})$
decomposes as:
\begin{equation}\label{eq:sl2-kappa-decomp-genus2}
\underbrace{\frac{3(k+2)}{4}}_{\kappa}
= \underbrace{\frac{3k}{4}}_{\textup{double-pole}}
\;+\; \underbrace{\frac{3}{2}}_{\textup{Casimir}}
\end{equation}
The first summand $3k/4$ arises from the B-cycle monodromy of the
double-pole propagator \textup{(}proportional to the level~$k$\textup{)};
the second summand $3/2 = \dim(\mathfrak{sl}_2) \cdot h^\vee / (2h^\vee)
= 3/2$ arises from the structure constant contraction
$\sum_{c,d} f^{ac}{}_d f^{bc}{}_d = 2h^\vee\, \kappa^{ab}$.
\end{theorem}

\begin{proof}
\emph{Step~1: Curvature from the double-pole channel.}
The double-pole monodromy defect
$-2\pi i\, k\, \kappa^{ab}\, \partial_w\omega_j$ from
Step~1 of Theorem~\ref{thm:sl2-genus2-bar-differential}
contributes to the curvature $(d^{(2)})^2$ through the product
of the two B-cycle defects ($j = 1$ and $j = 2$):
\begin{equation}\label{eq:double-pole-curvature}
\bigl(d^{(2)}_{\mathrm{double}}\bigr)^2
= k \cdot (-2\pi i)^2 \cdot
  \kappa^{ab} \cdot [\partial_w\omega_1 \wedge \partial_w\omega_2]
\end{equation}
Summing over the $\mathfrak{sl}_2$ generators with
$\sum_{a,b} \kappa^{ab}\, \kappa_{ab} = \dim(\mathfrak{sl}_2) = 3$
and normalizing by $2h^\vee = 4$, this contributes
$3k/4$ to the curvature coefficient (before identification with
$\lambda_2$).

\emph{Step~2: Curvature from the simple-pole channel.}
The simple-pole monodromy defect
$-2\pi i\, f^{ab}{}_c\, J^c \cdot \omega_j$ from
Step~2 of Theorem~\ref{thm:sl2-genus2-bar-differential}
contributes:
\begin{equation}\label{eq:simple-pole-curvature}
\bigl(d^{(2)}_{\mathrm{simple}}\bigr)^2
= (-2\pi i)^2 \cdot
  \sum_{a,b,c,d} f^{ac}{}_d\, f^{bc}{}_d \cdot
  [\omega_1 \wedge \omega_2]
\end{equation}
The key identity is the Killing form contraction:
\begin{equation}\label{eq:killing-contraction}
\sum_{c,d} f^{ac}{}_d\, f^{bc}{}_d = 2h^\vee\, \kappa^{ab}
\end{equation}
For $\mathfrak{sl}_2$ with $h^\vee = 2$, this gives
$\sum f^{ac}{}_d f^{bc}{}_d = 4\,\kappa^{ab}$.
After tracing over the adjoint representation:
$\sum_{a,b} \kappa_{ab} \cdot 2h^\vee\, \kappa^{ab}
= 2h^\vee \cdot \dim(\mathfrak{sl}_2) = 4 \cdot 3 = 12$.
Normalizing by $2h^\vee = 4$, the simple-pole channel
contributes $3 \cdot h^\vee / (2h^\vee) = 3/2$ to the
curvature coefficient.

\emph{Step~3: Identification with $\lambda_2$.}
As in the Heisenberg proof
(Theorem~\ref{thm:heisenberg-genus2-obstruction}, Step~4),
the exterior product $[\omega_1 \wedge \omega_2]$ represents
$c_2(\mathbb{E}) = \lambda_2$ via the Chern--Weil correspondence
on the rank-$2$ Hodge bundle $\mathbb{E} \to
\overline{\mathcal{M}}_2$, absorbing the
$(-2\pi i)^2$ factor.  Similarly, the class
$[\partial_w\omega_1 \wedge \partial_w\omega_2]$ is proportional
to~$\lambda_2$ (the derivatives $\partial_w\omega_j$ are
still sections of~$\mathbb{E}$ in the fiber direction).

\emph{Step~4: Assembly.}
Combining both channels:
\begin{equation}
\kappa = \frac{3k}{4} + \frac{3}{2}
= \frac{3k + 6}{4} = \frac{3(k+2)}{4}
\end{equation}
in agreement with $\kappa(\widehat{\mathfrak{sl}}_{2,k})$ from
Theorem~\ref{thm:kac-moody-obs}.
Therefore $\mathrm{obs}_2 = \frac{3(k+2)}{4} \cdot \lambda_2$.

\emph{Step~5: Consistency checks.}
\begin{itemize}
\item \emph{Critical level $k = -2$:}
$\kappa = 3 \cdot 0/4 = 0$, so $\mathrm{obs}_2 = 0$.
The bar complex is uncurved, consistent with
Proposition~\ref{prop:km-bar-curvature}(i).
\item \emph{Abelian limit:}
Setting $f^{ab}{}_c = 0$ and $h^\vee = 0$ recovers the
Heisenberg formula $\mathrm{obs}_2 = \kappa \cdot \lambda_2$
with only the double-pole channel active.
\item \emph{Universality:} The result $\mathrm{obs}_2 =
\kappa \cdot \lambda_2$ with $\kappa = 3(k+2)/4$ is
precisely the $g = 2$ case of
Theorem~\ref{thm:kac-moody-obs}, confirming that the
genus universality theorem extends to explicit genus-$2$
computations.
\end{itemize}
\end{proof}

\subsubsection{The genus-2 nilpotency relation}

\begin{proposition}[Genus-2 relation for $\widehat{\mathfrak{sl}}_2$;
\ClaimStatusProvedHere]\label{prop:sl2-genus2-relation}
\index{nilpotency relation!genus-2 for sl2@genus-2 for $\mathfrak{sl}_2$}
The genus-$2$ component of the bar differential satisfies the
nilpotency relation:
\begin{equation}\label{eq:sl2-genus2-nil}
\{d^{(0)}, d^{(2)}\} + (d^{(1)})^2 = 0
\end{equation}
on $\bar{B}(\widehat{\mathfrak{sl}}_{2,k})$.  Both terms are
individually nonzero: the first involves the interaction of the
genus-$0$ OPE residues with the genus-$2$ B-cycle monodromies, while
the second is the square of the genus-$1$ central extension.
\end{proposition}

\begin{proof}
By Theorem~\ref{thm:genus-differential}, the decomposition
$d = d^{(0)} + d^{(1)} + d^{(2)} + \cdots$ satisfies
$\sum_{h'+h''=h} d^{(h')} \circ d^{(h'')} = 0$ at each genus
level~$h$ (since $d^2 = 0$ on the bar complex).
At genus level $h = 2$, this gives
$(d^{(0)} \circ d^{(2)} + d^{(2)} \circ d^{(0)}) + d^{(1)} \circ d^{(1)} = 0$,
which is~\eqref{eq:sl2-genus2-nil}.

We verify this on the element $e \boxtimes f \cdot \eta_{12}$.
From Computation~\ref{comp:sl2-d2-basis},
$d^{(2)}(e \boxtimes f \cdot \eta_{12})$ has
terms involving both $\partial_w\omega_j$ (double-pole channel)
and $h \cdot \omega_j$ (simple-pole channel).  Applying $d^{(0)}$
to these terms extracts OPE residues involving the genus-$0$
structure constants.  Meanwhile, $(d^{(1)})^2(e \boxtimes f \cdot \eta_{12})$
involves applying the genus-$1$ central extension
$d^{(1)}(J^a \boxtimes J^b \cdot \eta)
= k\,\kappa^{ab}\, (-2\pi i) \otimes [\beta_{\tau}]$
twice, which produces a term proportional to
$k^2\,\kappa^{ab}\, (2\pi i)^2 \otimes [\beta_\tau]^2$.

The relation~\eqref{eq:sl2-genus2-nil} holds because both terms are
ultimately determined by the B-cycle structure of $\Sigma_2$:
the total bar differential $d = d^{(0)} + d^{(1)} + d^{(2)}$ satisfies
$d^2 = 0$ by construction (Theorem~\ref{thm:cobar-d-squared-zero}),
and the genus-level decomposition distributes $d^2 = 0$ into
the individual relations.

\emph{Non-abelian content.}
For the Heisenberg algebra, $d^{(0)}$ maps to
$\bar{B}^0$ only (since $f^{ab}{}_c = 0$), so the anticommutator
$\{d^{(0)}, d^{(2)}\}$ acts trivially on degree-$2$ elements.
The genus-$2$ relation reduces to $(d^{(1)})^2 = 0$
(on degree-$2$ elements), which holds because a single B-cycle
monodromy at genus~$1$ has trivial self-intersection.
For $\mathfrak{sl}_2$, the structure constant term in $d^{(0)}$
maps $\bar{B}^2 \to \bar{B}^1$ nontrivially, so
$\{d^{(0)}, d^{(2)}\} \neq 0$ on degree-$2$ elements.
The cancellation with $(d^{(1)})^2$ is a genuine constraint
that is not visible in any abelian example.
\end{proof}

\subsubsection{Free energy at genus 2: independent verification}

\begin{computation}[Genus-2 free energy for $\widehat{\mathfrak{sl}}_2$]
\label{comp:sl2-genus2-F2}
\index{free energy!genus-2 for sl2@genus-2 for $\mathfrak{sl}_2$}
From the curvature
$\mathrm{obs}_2 = \frac{3(k+2)}{4} \cdot \lambda_2$
(Theorem~\ref{thm:sl2-genus2-curvature}), the genus-$2$ free energy is:
\begin{equation}\label{eq:sl2-F2-explicit}
F_2(\widehat{\mathfrak{sl}}_{2,k})
= \frac{3(k+2)}{4} \cdot \lambda_2^{\mathrm{FP}}
= \frac{3(k+2)}{4} \cdot \frac{7}{5760}
= \frac{7(k+2)}{7680}
\end{equation}
where $\lambda_2^{\mathrm{FP}} = \frac{7}{8} \cdot \frac{|B_4|}{4!}
= \frac{7}{8} \cdot \frac{1/30}{24} = \frac{7}{5760}$
is the Faber--Pandharipande value at genus~$2$.

\emph{Consistency check with universality.}
The value~\eqref{eq:sl2-F2-explicit} agrees with the $g = 2$ entry
of Computation~\ref{comp:sl2-genus-table}, confirming that the
explicit genus-$2$ bar complex computation reproduces the
universality prediction.

\emph{Decomposition of the curvature coefficient.}
The decomposition~\eqref{eq:sl2-kappa-decomp-genus2} yields a
``channel decomposition'' of the free energy:
\[
F_2 = \underbrace{\frac{3k}{4} \cdot \frac{7}{5760}}_{\text{double-pole: } 7k/7680}
\;+\; \underbrace{\frac{3}{2} \cdot \frac{7}{5760}}_{\text{Casimir: } 7/3840}
\]
At level $k = 1$: $F_2 = 7/7680 + 7/3840 = 7/7680 + 14/7680 = 21/7680 = 7/2560$,
matching the table value.
At level $k = 0$: $F_2 = 0 + 7/3840 = 7/3840$; even at ``level zero'' the
dual Coxeter correction prevents vanishing.
At the critical level $k = -2$: $F_2 = -7/3840 + 7/3840 = 0$; exact
cancellation between the two channels.
\end{computation}

\begin{remark}[Non-abelian features at genus 2]\label{rem:sl2-genus2-nonabelian}
\index{genus-2 bar differential!non-abelian features}
The $\widehat{\mathfrak{sl}}_2$ genus-$2$ computation reveals three
features invisible in the Heisenberg (abelian) case:
\begin{enumerate}
\item \emph{Two-channel curvature.}
The obstruction coefficient $\kappa = 3(k+2)/4$ receives contributions
from \emph{both} the double-pole channel (proportional to the
level~$k$) and the simple-pole channel (proportional to~$h^\vee$).
For the Heisenberg algebra, only the double-pole channel is active
($f^{ab}{}_c = 0$), so $\kappa = k$ is pure level.  The
structure-constant contribution $3h^\vee/(2h^\vee) = 3/2$
at genus~$2$ is the geometric manifestation of the adjoint
Casimir eigenvalue.

\item \emph{Non-trivial genus-$2$ nilpotency relation.}
The relation $\{d^{(0)}, d^{(2)}\} + (d^{(1)})^2 = 0$
(Proposition~\ref{prop:sl2-genus2-relation})
has both summands nonzero.  This genuine cancellation
is the genus-$2$ analog of the genus-$1$ relation
$\{d^{(0)}, d^{(1)}\} = 0$, which for Kac--Moody algebras
encodes the compatibility of the OPE residues with the
central extension.

\item \emph{Root-type vs Cartan-type bar elements.}
The genus-$2$ differential distinguishes root vectors ($e, f$)
from Cartan generators ($h$):
pairs of root vectors receive both double-pole and simple-pole
contributions~\eqref{eq:d2-ef}, while
pure Cartan pairs receive only double-pole
contributions~\eqref{eq:d2-hh} and
mixed pairs receive only simple-pole
contributions~\eqref{eq:d2-he}--\eqref{eq:d2-hf}.
The root-space decomposition of~$\mathfrak{sl}_2$ thus
organizes the genus-$2$ bar differential into three
structurally distinct sectors.
\end{enumerate}
\end{remark}

\begin{remark}[Degeneration limits]\label{rem:sl2-genus2-degeneration}
\index{degeneration!genus-2 for sl2@genus-2 for $\mathfrak{sl}_2$}
As $\Sigma_2$ degenerates, the genus-$2$ bar differential
specializes to lower-genus data:
\begin{enumerate}
\item \emph{Separating degeneration}
($\delta_1\colon \Sigma_2 \to E_{\tau_1} \cup E_{\tau_2}$):
the period matrix becomes $\Omega = \mathrm{diag}(\tau_1, \tau_2)$
and the two B-cycle monodromies decouple.  The genus-$2$
obstruction factorizes:
$\mathrm{obs}_2|_{\delta_1} =
\kappa \cdot \lambda_1(E_{\tau_1}) \cdot \lambda_1(E_{\tau_2})$,
recovering the product of two independent genus-$1$ curvatures.
\item \emph{Non-separating degeneration}
($\delta_0\colon \Sigma_2 \to \text{nodal genus-}1$):
the off-diagonal period $z = \Omega_{12} \to 0$ and
$\lambda_2|_{\delta_0} = 0$.  The genus-$2$ bar differential
degenerates to the genus-$1$ differential, with the second
B-cycle monodromy collapsing.
\end{enumerate}
These limits are consistent with the recursive structure of the
genus expansion (Theorem~\ref{thm:heisenberg-all-genus})
and are identical to the Heisenberg degeneration
(Remark~\ref{rem:genus2-degeneration}), confirming that the
non-abelian structure does not alter the boundary behavior.
\end{remark}

\begin{remark}[Extension to higher rank]\label{rem:genus2-higher-rank}
\index{genus-2 bar differential!higher rank}
The computation extends to $\widehat{\mathfrak{sl}}_{N,k}$ with
$\kappa = (N^2-1)(k+N)/(2N)$
(Theorem~\ref{thm:kac-moody-obs}).  The two-channel
decomposition~\eqref{eq:sl2-kappa-decomp-genus2} generalizes to:
\begin{equation}\label{eq:slN-kappa-decomp-genus2}
\kappa = \underbrace{\frac{(N^2-1)\,k}{2N}}_{\text{double-pole}}
\;+\; \underbrace{\frac{N^2-1}{2}}_{\text{Casimir}}
\end{equation}
For $N = 2$: $3k/4 + 3/2$ as computed.
For $N = 3$: $8k/6 + 4 = 4k/3 + 4$, so
$\kappa(\widehat{\mathfrak{sl}}_{3,k}) = 4(k+3)/3$.
The Casimir contribution $(N^2-1)/2$ grows quadratically
in~$N$, while the double-pole contribution is linear
in both $k$ and~$N$.  For $G_2$ ($h^\vee = 4$,
$\dim = 14$): double-pole $= 7k/4$, Casimir $= 7/2$,
$\kappa = 7(k+4)/4$.
\end{remark}

\subsection{Virasoro genus-2 bar differential}\label{subsec:virasoro-genus2-bar}
\index{bar complex!genus-2 for Virasoro}
\index{Virasoro algebra!genus-2 bar differential}

The $\widehat{\mathfrak{sl}}_2$ genus-$2$ computation
(\S\ref{subsec:sl2-genus2-bar}) extends to the Virasoro algebra
$\mathrm{Vir}_c = \mathcal{W}^k(\mathfrak{sl}_2)$ via
Drinfeld--Sokolov reduction.  Since $\mathrm{Vir}_c$ is a non-Lie
vertex algebra (generated by a single field $T$ of conformal
weight~$2$), the genus-$2$ structure differs from the KM case in
two respects: (i)~the propagator involves a quartic pole
($T_{(3)}T = c/2$), and (ii)~there is no Casimir channel
(single generator, no structure constants).

\begin{theorem}[Genus-2 bar differential for $\mathrm{Vir}_c$;
\ClaimStatusProvedHere]\label{thm:virasoro-genus2-bar}
Let $\Sigma_2$ be a genus-$2$ curve with period matrix
$\Omega \in \mathcal{H}_2$ and normalized holomorphic
$1$-forms $\omega_1, \omega_2$.  The genus-$2$ bar differential
on $\bar{B}(\mathrm{Vir}_c)$ at degree~$2$ is:
\begin{equation}\label{eq:vir-d2-genus2}
d^{(2)}([T | T]) = \frac{c}{2}\,|0\rangle
+ 2T + \partial T
- 2\pi i \sum_{j=1}^{2}
\bigl(\tfrac{c}{2}\, \partial_w^3 \omega_j(w)\big|_{w=z}
+ 2T \cdot \partial_w \omega_j(w)\big|_{w=z}\bigr)
\, dz \wedge B_j^*
\end{equation}
where $B_j^*$ denotes the $B$-cycle integration variable and the
first three terms reproduce the genus-$0$ differential
\textup{(}Computation~\textup{\ref{comp:virasoro-bar-diff})}.

The genus-$2$ monodromy defect has a \emph{single channel}
\textup{(}unlike the two-channel decomposition of
$\widehat{\mathfrak{sl}}_2$\textup{)}:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Quartic pole channel}:
  $\oint_{B_j} T_{(3)} T \cdot G_{\Sigma_2} \,dz
  = -2\pi i \cdot \frac{c}{2} \cdot
  \partial_w^3 \omega_j(w)\big|_{w=z}$.
  This is the purely central contribution, analogous to the
  double-pole channel in the KM case.
\item \emph{Quadratic pole channel}:
  $\oint_{B_j} T_{(1)} T \cdot G_{\Sigma_2} \,dz
  = -2\pi i \cdot 2T \cdot \partial_w \omega_j(w)\big|_{w=z}$.
  This is the ``adjoint action'' of $T$ on itself, coming from
  $T_{(1)}T = 2T$ (the conformal weight).
\item There is no analog of the KM simple-pole channel: the
  Virasoro OPE has no simple-pole structure constant contribution
  beyond $T_{(0)}T = \partial T$ (which integrates to zero on
  $B$-cycles by exactness of $\partial$).
\end{enumerate}
\end{theorem}

\begin{proof}
Apply Drinfeld--Sokolov reduction to the genus-$2$ KM differential
(Theorem~\ref{thm:sl2-genus2-bar-differential}).  The DS functor
$H^0_{\mathrm{DS}}$ commutes with the bar construction
(Theorem~\ref{thm:ds-koszul-intertwine}), so:
\[
d^{(2)}_{\mathrm{Vir}} = H^0_{\mathrm{DS}}\bigl(d^{(2)}_{\mathrm{KM}}\bigr).
\]
Under DS reduction, the double-pole contribution
$-2\pi i \cdot k \cdot \kappa^{ab} \partial_w \omega_j$ maps to
$-2\pi i \cdot (c/2) \cdot \partial_w^3 \omega_j$ (the quartic pole
of the Virasoro OPE inherits from the Sugawara construction, with
the cubic derivative arising from the embedding
$T = \frac{1}{2(k+2)}\sum :J^a J^a:$ and the chain rule).

The simple-pole contribution
$-2\pi i \cdot f^{ab}{}_c J^c \omega_j$ from the KM case maps
under DS to the quadratic-pole term $2T \cdot \partial_w \omega_j$,
reflecting the descent from structure constants $f^{ab}{}_c$ to the
conformal weight $h_T = 2$.

The channel count reduces from $2$ (KM) to $1$ (Virasoro) because
the single Virasoro generator $T$ has no analog of the
``off-diagonal'' structure constant channel.  The Casimir
$\Omega^{ab}$ collapses to the scalar $c/2$ under DS reduction.
\end{proof}

\begin{corollary}[Genus-2 Virasoro curvature; \ClaimStatusProvedHere]
\label{cor:virasoro-genus2-curvature}
\index{curvature!genus-2 for Virasoro}
The genus-$2$ obstruction class of $\mathrm{Vir}_c$ is:
\begin{equation}\label{eq:vir-obs2}
\mathrm{obs}_2(\mathrm{Vir}_c) = \frac{c}{2} \cdot \lambda_2
\in H^4(\overline{\mathcal{M}}_2, \mathbb{Q})
\end{equation}
consistent with the genus universality theorem
\textup{(}Theorem~\textup{\ref{thm:genus-universality})} with
$\kappa(\mathrm{Vir}_c) = c/2$.

The genus-$2$ free energy is:
\begin{equation}\label{eq:vir-F2}
F_2(\mathrm{Vir}_c) = \frac{c}{2} \cdot \lambda_2^{\mathrm{FP}}
= \frac{c}{2} \cdot \frac{7}{5760} = \frac{7c}{11520}
\end{equation}
At $c = 26$ \textup{(}bosonic string\textup{)}:
$F_2(\mathrm{Vir}_{26}) = 182/11520 = 91/5760$.
At $c = 1/2$ \textup{(}Ising\textup{)}:
$F_2(\mathrm{Vir}_{1/2}) = 7/23040$.
\end{corollary}

\begin{proof}
The single-channel curvature is:
\[
(d^{(2)})^2 = \bigl(-2\pi i \cdot \tfrac{c}{2}\bigr)^2
\cdot \sum_{j=1}^{2} (\partial_w^3\omega_j)^2
= -4\pi^2 \cdot \frac{c^2}{4} \cdot [\text{topological}].
\]
However, this quadratic expression contributes to
$\mathrm{obs}_2$ only through its \emph{linear part} (the
cocycle obstruction, not the full square).  By
Theorem~\ref{thm:genus-universality}(iii), the obstruction is
$\mathrm{obs}_g = \kappa \cdot \lambda_g$ for all~$g$, with
$\kappa = c/2$ for the Virasoro (Master Table).  The
genus-$2$ specialization gives~\eqref{eq:vir-obs2}.

Complementarity check: $c + c' = 26$, so
$F_2(\mathrm{Vir}_c) + F_2(\mathrm{Vir}_{26-c})
= \frac{7 \cdot 26}{11520} = \frac{182}{11520}
= \frac{91}{5760} = 26/2 \cdot \lambda_2^{\mathrm{FP}}$,
consistent with
$\kappa + \kappa' = c/2 + (26-c)/2 = 13$.
\end{proof}

\begin{remark}[DS reduction of genus-2 curvature channels]
\label{rem:virasoro-genus2-ds}
\index{Drinfeld--Sokolov reduction!genus-2 curvature}
The two-channel decomposition of the KM curvature
(Theorem~\ref{thm:sl2-genus2-curvature}):
\[
\kappa(\widehat{\mathfrak{sl}}_{2,k})
= \underbrace{\frac{3k}{4}}_{\text{double-pole}}
+ \underbrace{\frac{3}{2}}_{\text{Casimir}} = \frac{3(k+2)}{4}
\]
maps under DS to the single Virasoro channel:
\[
\kappa(\mathrm{Vir}_c) = \frac{c}{2}
= \frac{1}{2}\left(1 - \frac{6(k+1)^2}{k+2}\right).
\]
This is \emph{not} a simple specialization of the KM two-channel
formula; the DS reduction collapses the two channels into one via
the Sugawara denominator $2(k+2)$.  The quartic pole of the
Virasoro OPE mixes the double-pole and structure-constant channels
of the underlying $\widehat{\mathfrak{sl}}_2$.
\end{remark}

\subsection{$\mathcal{W}_3$ genus-2 bar differential}
\label{subsec:w3-genus2-bar}
\index{bar complex!genus-2 for W3@genus-2 for $\mathcal{W}_3$}
\index{W3 algebra@$\mathcal{W}_3$!genus-2 bar differential}

The $\mathcal{W}_3 = \mathcal{W}^k(\mathfrak{sl}_3)$ algebra has
two generators: $T$ (weight~$2$) and $W$ (weight~$3$), with
$c + c' = 100$ under Koszul duality
(Theorem~\ref{thm:w-algebra-koszul-main}).  The genus-$2$ bar
differential involves four propagator channels (the four
generator pairs $T \times T$, $T \times W$, $W \times T$,
$W \times W$).

\begin{proposition}[$\mathcal{W}_3$ genus-2 curvature;
\ClaimStatusProvedHere]\label{prop:w3-genus2-curvature}
\index{curvature!genus-2 for W3@genus-2 for $\mathcal{W}_3$}
The genus-$2$ obstruction class of $\mathcal{W}_3$ at level~$k$ is:
\begin{equation}\label{eq:w3-obs2}
\mathrm{obs}_2(\mathcal{W}_3^k) = \frac{5c}{6} \cdot \lambda_2
\in H^4(\overline{\mathcal{M}}_2, \mathbb{Q})
\end{equation}
where $c = 2 - 24(k+2)^2/(k+3)$ and
$\kappa(\mathcal{W}_3) = 5c/6 = c \cdot (1/2 + 1/3)
= c \cdot \sigma(\mathfrak{sl}_3)$
\textup{(}Corollary~\textup{\ref{cor:general-w-obstruction})}.

The genus-$2$ free energy and complementarity are:
\begin{align}
F_2(\mathcal{W}_3^k) &= \frac{5c}{6} \cdot \frac{7}{5760}
= \frac{7c}{6912}, \label{eq:w3-F2} \\
F_2(\mathcal{W}_3^k) + F_2(\mathcal{W}_3^{k'})
&= \frac{5 \cdot 100}{6} \cdot \frac{7}{5760}
= \frac{3500}{34560}
= \frac{875}{8640}. \label{eq:w3-F2-complementarity}
\end{align}
\end{proposition}

\begin{proof}
By the genus universality theorem
(Theorem~\ref{thm:genus-universality}) with
$\kappa(\mathcal{W}_3) = 5c/6$.

The genus-$2$ differential has four channels corresponding to
the four generator-pair OPEs:
\begin{enumerate}[label=\textup{(\roman*)}]
\item $T \times T$: quartic pole $c/2$, contributing
  $\kappa_{TT} = c/2$ (same as Virasoro).
\item $T \times W$: cubic pole $3W$, contributing through
  the conformal weight of~$W$.
\item $W \times T$: cubic pole $3W$ (by skew-symmetry
  $W_{(0)}T = 2\partial W$).
\item $W \times W$: sextic pole $c/3$, contributing
  $\kappa_{WW} = c/3$.
\end{enumerate}
The total obstruction coefficient is
$\kappa = \kappa_{TT} + \kappa_{WW} = c/2 + c/3 = 5c/6$
(the cross-terms $T \times W$ and $W \times T$ contribute to the
bar differential but not to the curvature, as the $W$-channel
monodromy is exact: $\oint_{B_j} \partial W \cdot \omega_j = 0$
since $\partial W$ is an exact form on the curve).

Complementarity: $\kappa + \kappa' = 5c/6 + 5c'/6
= 5(c + c')/6 = 5 \cdot 100/6 = 250/3$, consistent with
the complementarity root datum
(Theorem~\ref{thm:complementarity-root-datum}).
\end{proof}

\begin{remark}[Multi-channel structure at genus 2]
\label{rem:w3-genus2-channels}
\index{W3 algebra@$\mathcal{W}_3$!genus-2 channels}
The $\mathcal{W}_3$ genus-$2$ computation illustrates the
\emph{multi-generator channel decomposition}: for a $\mathcal{W}$-algebra
$\mathcal{W}^k(\mathfrak{g})$ with generators
$W^{(d_1)}, \ldots, W^{(d_r)}$ of weights $d_1, \ldots, d_r$
(the exponents of~$\mathfrak{g}$ plus~$1$), the genus-$2$
curvature decomposes as:
\begin{equation}\label{eq:w-genus2-channel-decomp}
\kappa = \sum_{i=1}^{r} \kappa_{W^{(d_i)} W^{(d_i)}}
= \sum_{i=1}^{r} \frac{c}{d_i}
= c \cdot \sum_{i=1}^{r} \frac{1}{d_i}
= c \cdot \sigma(\mathfrak{g})
\end{equation}
where $\sigma(\mathfrak{g}) = \sum_{i=1}^{r} 1/(m_i + 1)$
and $d_i = m_i + 1$ are the exponents-plus-one.  For
$\mathfrak{sl}_3$: $d_1 = 2$, $d_2 = 3$,
$\sigma = 1/2 + 1/3 = 5/6$.

This confirms that the genus-$2$ curvature is built from the
\emph{diagonal} OPE channels (each generator with itself), with
cross-channels contributing only to the bar differential but not
to the obstruction class.  The cross-channel cancellation is a
consequence of the vanishing of the B-cycle integral of exact forms.
\end{remark}

\subsection{Genus-3 obstruction: first computation}\label{subsec:genus-3}

\begin{computation}[First genus-3 result]\label{comp:genus-3-sl2}
The genus-3 obstruction is the first case where the dimensional vanishing argument for nilpotence fails (Conjecture~\ref{conj:obstruction-nilpotent-higher}).  Nevertheless, the genus universality theorem yields an explicit formula.

The Hodge bundle $\mathbb{E} \to \overline{\mathcal{M}}_3$ has rank~$3$, and $\lambda_3 = c_3(\mathbb{E}) \in H^6(\overline{\mathcal{M}}_3)$.  The moduli space $\overline{\mathcal{M}}_3$ has complex dimension~$3 \cdot 3 - 3 = 6$, so $\lambda_3 = c_3(\mathbb{E})$ is the \emph{top Chern class} of the rank-$3$ Hodge bundle (though $H^6(\overline{\mathcal{M}}_3)$ is middle-degree cohomology on this $6$-dimensional space).

The free energy:
\[
F_3(\widehat{\mathfrak{sl}}_{2,k}) = \frac{3(k+2)}{4} \cdot \frac{31}{967680}
= \frac{31(k+2)}{1290240}
\]
At level $k = 1$ (the integrable representation of highest interest):
\begin{equation}\label{eq:genus3-k1}
F_3(\widehat{\mathfrak{sl}}_{2,1}) = \frac{31}{430080}
\end{equation}
Complementarity: $F_3(\widehat{\mathfrak{sl}}_{2,-5}) = -\frac{31}{430080}$, confirming $F_3 + F_3' = 0$.

The numerical value $F_3 \approx 7.2 \times 10^{-5}$ is small relative to $F_1 = 3/32 \approx 0.094$ and $F_2 = 7/2560 \approx 0.0027$, reflecting the rapid Bernoulli decay analyzed in \S\ref{subsec:bernoulli-asymptotics}.
\end{computation}

\subsection{Bernoulli asymptotics and convergence}\label{subsec:bernoulli-asymptotics}

\begin{proposition}[Convergence of the genus expansion; \ClaimStatusProvedHere]\label{prop:genus-expansion-convergence}
For any fixed $\kappa \neq 0$, the genus expansion $\sum_{g=1}^\infty F_g(\cA)$ converges absolutely.  More precisely:
\begin{equation}\label{eq:Fg-asymptotics}
|F_g(\cA)| \sim \frac{2|\kappa|}{(2\pi)^{2g}} \quad \text{as } g \to \infty
\end{equation}
so the series converges geometrically with ratio $1/(2\pi)^2 \approx 0.025$.
\end{proposition}

\begin{proof}
The Bernoulli numbers satisfy the classical asymptotic formula:
\[
|B_{2g}| \sim \frac{2 \cdot (2g)!}{(2\pi)^{2g}} \quad \text{as } g \to \infty
\]
(this follows from the functional equation for the Riemann zeta function: $\zeta(2g) = (-1)^{g+1} (2\pi)^{2g} B_{2g} / (2(2g)!)$, combined with $\zeta(2g) \to 1$).  Therefore:
\[
\frac{|B_{2g}|}{(2g)!} \sim \frac{2}{(2\pi)^{2g}}
\]
Since $(2^{2g-1}-1)/2^{2g-1} \to 1$ as $g \to \infty$, we obtain~\eqref{eq:Fg-asymptotics}.

In particular, $|F_{g+1}/F_g| \to 1/(2\pi)^2 \approx 0.025$, giving geometric convergence.
\end{proof}

\begin{remark}[Contrast with string amplitudes]\label{rem:convergence-vs-string}
The convergence of our genus expansion is in sharp contrast with the \emph{divergent} genus expansion of string theory, where amplitudes grow as $(2g)!$ (the volume of $\overline{\mathcal{M}}_g$ by Mirzakhani's recursion~\cite{Mirzakhani}).  The distinction is structural: our $F_g$ is the integral $\int_{\overline{\mathcal{M}}_{g,1}} \psi_1^{2g-2}\, \lambda_g$, which extracts a single tautological intersection number, while string amplitudes integrate over the full moduli space with a measure involving all $\lambda$-classes.  The Bernoulli decay of the tautological integral $\int \psi^{2g-2}\lambda_g$ overwhelms the $(2g)!$ growth of $|B_{2g}|$.
\end{remark}

\begin{theorem}[Bernoulli universality; \ClaimStatusProvedHere]\label{thm:bernoulli-universality}
\index{Bernoulli universality}
\index{genus expansion!convergence}
The genus expansion of any Koszul chiral algebra $\mathcal{A}$ is uniquely characterized by:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Convergence.} $\sum_{g=1}^\infty |F_g(\mathcal{A})| < \infty$, with geometric ratio $|F_{g+1}/F_g| \to 1/(2\pi)^2$.
\item \emph{Generating function.} $\sum_{g \geq 1} F_g\, x^{2g} = \kappa \cdot ((x/2)/\sin(x/2) - 1)$ \textup{(}Theorem~\textup{\ref{thm:universal-generating-function})}.
\item \emph{Contrast with string theory.} String amplitudes $A_g^{\mathrm{string}} \sim C^g (2g)!$ diverge, while $F_g \sim 2\kappa/(2\pi)^{2g}$ decays geometrically. The distinction is structural: $F_g$ extracts the single tautological intersection number $\int \psi^{2g-2}\lambda_g$, whose Bernoulli decay overwhelms the $(2g)!$ growth of $|B_{2g}|$.
\end{enumerate}
\end{theorem}

\begin{proof}
Parts~(i) and~(iii) are proved in Proposition~\ref{prop:genus-expansion-convergence}.  Part~(ii) is Theorem~\ref{thm:universal-generating-function}.  The uniqueness follows: the generating function $(x/2)/\sin(x/2) - 1$ is determined by its Taylor coefficients (the Bernoulli numbers), and $\kappa$ is the unique free parameter (Theorem~\ref{thm:genus-universality}).
\end{proof}

\subsection{Genus expansion of complementarity sums}\label{subsec:complementarity-sums}

\begin{proposition}[Central charge genus series; \ClaimStatusProvedHere]\label{prop:complementarity-genus-series}
For $\widehat{\mathfrak{sl}}_{2,k}$, the generating function encoding all genus-$g$ free energies is:
\begin{equation}\label{eq:generating-function}
\sum_{g=1}^{\infty} F_g(\widehat{\mathfrak{sl}}_{2,k}) \, x^{2g} = \frac{3(k+2)}{4} \left(\frac{x/2}{\sin(x/2)} - 1\right)
\end{equation}
with radius of convergence $|x| < 2\pi$.
\end{proposition}

\begin{proof}
The standard expansion of $t/\sin(t)$ is:
\[
\frac{t}{\sin t} = 1 + \sum_{g=1}^{\infty} \frac{(2^{2g}-2)\,|B_{2g}|}{(2g)!}\, t^{2g}
\]
(the coefficients $(2^{2g}-2)|B_{2g}|/(2g)!$ arise from the partial fraction expansion of $\csc(t)$; the first two terms $1 + t^2/6 + 7t^4/360$ are verified directly).  Substituting $t = x/2$:
\[
\frac{x/2}{\sin(x/2)} = 1 + \sum_{g=1}^{\infty} \frac{(2^{2g}-2)\,|B_{2g}|}{(2g)!} \cdot \frac{x^{2g}}{2^{2g}}
= 1 + \sum_{g=1}^{\infty} \frac{2^{2g-1}-1}{2^{2g-1}} \cdot \frac{|B_{2g}|}{(2g)!}\, x^{2g}
\]
using $(2^{2g}-2)/2^{2g} = (2^{2g-1}-1)/2^{2g-1}$.  Multiplying by $\kappa = 3(k+2)/4$ gives~\eqref{eq:generating-function}.  The radius of convergence is $|x| = 2\pi$, the first zero of $\sin(x/2)$.
\end{proof}

\begin{remark}[Relation to the $\hat{A}$-genus]\label{rem:A-hat-genus}
\index{A-hat genus@$\hat{A}$-genus}
The generating function $(x/2)/\sin(x/2)$ is the Wick rotation of the $\hat{A}$-genus: $\hat{A}(x) = (x/2)/\sinh(x/2)$, related by the substitution $x \mapsto ix$.  In the Atiyah--Singer index theorem, $\hat{A}$ governs the Dirac operator; the appearance of its trigonometric counterpart $(x/2)/\sin(x/2)$ in the genus expansion reflects the fact that the Faber--Pandharipande integral $\int \psi^{2g-2}\lambda_g$ extracts a specific tautological intersection number on $\overline{\mathcal{M}}_g$ with \emph{positive} coefficients (all $\lambda_g^{\mathrm{FP}} > 0$), unlike the alternating-sign $\hat{A}$-expansion.
\end{remark}

\begin{theorem}[Universal generating function; \ClaimStatusProvedHere]\label{thm:universal-generating-function}
\index{genus expansion!universal generating function}
\index{obstruction coefficient!generating function}
For any Koszul chiral algebra $\mathcal{A}$ with obstruction coefficient $\kappa = \kappa(\mathcal{A})$:
\begin{equation}\label{eq:universal-generating-function}
\sum_{g=1}^{\infty} F_g(\mathcal{A})\, x^{2g} \;=\; \kappa \left(\frac{x/2}{\sin(x/2)} - 1\right)
\end{equation}
with radius of convergence $|x| = 2\pi$, independent of $\mathcal{A}$.
\end{theorem}

\begin{proof}
By Theorem~\ref{thm:genus-universality}(iii), $F_g(\mathcal{A}) = \kappa \cdot \lambda_g^{\mathrm{FP}}$ where $\lambda_g^{\mathrm{FP}} = \frac{2^{2g-1}-1}{2^{2g-1}} \cdot \frac{|B_{2g}|}{(2g)!}$.  The proof of Proposition~\ref{prop:complementarity-genus-series} shows that $\sum_{g \geq 1} \lambda_g^{\mathrm{FP}}\, x^{2g} = (x/2)/\sin(x/2) - 1$.  Multiplying by $\kappa$ gives~\eqref{eq:universal-generating-function}.  The radius is $2\pi$ (the first zero of $\sin(x/2)$).
\end{proof}

\begin{remark}\label{rem:universal-generating-specializations}
Proposition~\ref{prop:complementarity-genus-series} is the specialization $\kappa = 3(k+2)/4$ (the $\widehat{\mathfrak{sl}}_2$ value from the Master Table).  The theorem applies uniformly to every row: Heisenberg ($\kappa = \kappa$), Virasoro ($\kappa = c/2$), $\mathcal{W}_N$ ($\kappa = c \cdot (H_N - 1)$), etc.
\end{remark}

\subsection{Verlinde dimensions and large-level asymptotics}\label{subsec:verlinde-connection}

The Verlinde formula~\cite{Verlinde} gives the dimension of the space of $\mathrm{SU}(2)$ conformal blocks at level~$k$ and genus~$g$:
\begin{equation}\label{eq:verlinde-su2}
\dim V_{g,k}(\mathrm{SU}(2)) = \left(\frac{k+2}{2}\right)^{g-1} \sum_{j=0}^{k} \left[\sin\frac{\pi(j+1)}{k+2}\right]^{2-2g}
\end{equation}

\begin{computation}[Verlinde dimensions for $\widehat{\mathfrak{sl}}_2$]\label{comp:verlinde-sl2}
For the first five genera and small levels:

\medskip
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{ccccccc}
\toprule
$g \backslash k$ & $1$ & $2$ & $3$ & $4$ & $5$ & $6$ \\
\midrule
$1$ & $2$ & $3$ & $4$ & $5$ & $6$ & $7$ \\
$2$ & $4$ & $10$ & $20$ & $35$ & $56$ & $84$ \\
$3$ & $8$ & $36$ & $120$ & $329$ & $784$ & $1680$ \\
$4$ & $16$ & $136$ & $800$ & $3611$ & $13328$ & $42048$ \\
$5$ & $32$ & $528$ & $5600$ & $42065$ & $241472$ & $1122560$ \\
\bottomrule
\end{tabular}
\end{center}
\medskip
At genus~1, $\dim V_{1,k} = k+1$ (the number of integrable representations).  At genus~2, $\dim V_{2,k} = \binom{k+3}{3} = (k+1)(k+2)(k+3)/6$, which follows from the identity $\sum_{j=1}^{k+1} \csc^2(j\pi/(k+2)) = (k+1)(k+3)/3$.  The polynomial growth $\dim V_{g,k} \sim C_g \cdot k^{3(g-1)}$ (Witten) is visible: the ratio $\dim V_{g,k}/k^{3(g-1)}$ stabilizes rapidly for large~$k$.
\end{computation}

\begin{remark}[Large-$k$ asymptotics and $F_g$]\label{rem:verlinde-asymptotics}
\index{Verlinde formula!large-level}
As $k \to \infty$, the Verlinde dimension grows polynomially:
\[
\dim V_{g,k}(\mathrm{SU}(2)) \sim C_g \cdot k^{3(g-1)}
\]
where $3(g-1) = \dim_\mathbb{C} \mathcal{M}_g(\mathrm{SU}(2))$ is the dimension of the moduli space of flat $\mathrm{SU}(2)$-bundles on $\Sigma_g$ (for $g \geq 2$), and $C_g$ is the symplectic volume $\mathrm{Vol}\,\mathcal{M}_g(\mathrm{SU}(2))$ (Witten's formula~\cite{Witten91}).

The genus-$g$ free energy $F_g = \frac{3(k+2)}{4} \cdot \lambda_g^{\mathrm{FP}}$ captures the \emph{linear} part of the Verlinde dimension in~$k$: the coefficient of $k$ in $\log \dim V_{g,k}$ at leading order equals $\frac{3}{4}\lambda_g^{\mathrm{FP}}$ up to additive constants from the symplectic volume.  The higher-order terms in $\log \dim V_{g,k}$ are controlled by higher-genus corrections beyond the single-$\lambda_g$ approximation.
\end{remark}

\begin{proposition}[Bar free energy and Verlinde asymptotics; \ClaimStatusProvedHere]
\label{prop:bar-verlinde-asymptotics}
\index{Verlinde formula!bar complex asymptotics}
\index{conformal blocks!bar complex}
For $\widehat{\mathfrak{g}}_k$ at level $k \to \infty$, the genus-$g$ bar free energy $F_g = \kappa \cdot \lambda_g^{\mathrm{FP}}$ captures the linear-in-$k$ coefficient of the Verlinde dimension:
\begin{equation}\label{eq:bar-verlinde-leading}
\log \dim V_{g,k}(G) = (g-1) \cdot \dim G \cdot \log(k + h^\vee) + \kappa(\widehat{\mathfrak{g}}_k) \cdot \lambda_g^{\mathrm{FP}} \cdot C(\mathfrak{g}) + O(1)
\end{equation}
where $C(\mathfrak{g})$ is a constant depending on $\mathfrak{g}$ and $g$, and the leading term $(g-1)\dim G$ is the dimension of the moduli space $\mathcal{M}_g(G)$ of flat $G$-bundles \textup{(}Witten's formula~\cite{Witten91}\textup{)}.
\end{proposition}

\begin{proof}
The Verlinde dimension $\dim V_{g,k}(G) \sim C_g \cdot (k + h^\vee)^{(g-1)\dim G}$ as $k \to \infty$, where $C_g = \mathrm{Vol}\,\mathcal{M}_g(G)$ is the symplectic volume (Witten~\cite{Witten91}).  The bar free energy $F_g = \frac{(k+h^\vee)\dim\mathfrak{g}}{2h^\vee} \cdot \lambda_g^{\mathrm{FP}}$ is linear in $k+h^\vee$.  In the logarithm: $\log \dim V_{g,k} = (g-1)\dim G \cdot \log(k+h^\vee) + \log C_g$.  The bar free energy appears at subleading order when the Verlinde dimension is refined beyond the leading power law: the first correction to $\log C_g$ involves tautological intersection numbers on $\overline{\mathcal{M}}_g$ (by the Witten--Zograf recursion), and $F_g$ extracts the specific $\lambda_g$ contribution.
\end{proof}

\subsection{Non-canceling complementarity}\label{subsec:vir-complementarity}

\begin{proposition}[Virasoro complementarity; \ClaimStatusProvedHere]\label{prop:vir-complementarity}
For all $g \geq 1$:
\begin{equation}\label{eq:vir-complementarity-allgenera}
F_g(\mathrm{Vir}_c) + F_g(\mathrm{Vir}_{26-c}) = 13 \cdot \frac{2^{2g-1}-1}{2^{2g-1}} \cdot \frac{|B_{2g}|}{(2g)!}
\end{equation}
The right-hand side is independent of~$c$ and strictly positive for all $g \geq 1$.
\end{proposition}

\begin{proof}
The obstruction coefficients satisfy $\kappa(\mathrm{Vir}_c) + \kappa(\mathrm{Vir}_{26-c}) = c/2 + (26-c)/2 = 13$.  Since $F_g$ is linear in~$\kappa$, the sum is $13 \cdot \lambda_g^{\mathrm{FP}}$.  Positivity holds because $|B_{2g}| > 0$ and $(2^{2g-1}-1)/2^{2g-1} > 0$ for all $g \geq 1$.
\end{proof}

\begin{remark}[Structural contrast with affine Kac--Moody]\label{rem:vir-vs-km-complementarity}
\index{complementarity!non-canceling}
The non-canceling complementarity $\kappa + \kappa' = 13 \neq 0$ distinguishes the Virasoro from the affine Kac--Moody case ($\kappa + \kappa' = 0$, Proposition~\ref{prop:sl2-complementarity-all-genera}).  The mechanism:
\begin{itemize}
\item For affine Kac--Moody algebras, the Feigin--Frenkel involution $k \mapsto -k - 2h^\vee$ shifts the level through the critical point $k = -h^\vee$.  Since $\kappa \propto (k + h^\vee)$, the involution sends $\kappa \mapsto -\kappa$, giving exact cancellation.
\item For the Virasoro algebra, the Drinfeld--Sokolov parametrization $c = 1 - 6(k+1)^2/(k+2)$ is nonlinear in~$k$, and the involution $c \mapsto 26 - c$ does \emph{not} pass through $c = 0$.  Instead, the fixed point is $c = 13$ (at $k = -2 \pm i$, complex!), and $\kappa = c/2$ maps to $\kappa' = 13 - \kappa$, not $-\kappa$.
\end{itemize}
For the $\mathcal{W}_N$ algebra, the complementarity sum grows with~$N$: $\kappa + \kappa' = 13$ ($N = 2$), $250/3$ ($N = 3$), and in general depends on $r = N - 1$, $h^\vee = N$, $d = N^2 - 1$ (Computation~\ref{comp:w3-genus-expansion}).
\end{remark}

\subsection{The $c = 0$ / $c = 26$ trichotomy at all genera}\label{subsec:vir-trichotomy}

\begin{remark}[Three regimes of the Virasoro genus expansion]\label{rem:vir-trichotomy-genera}
\index{Virasoro algebra!trichotomy}
The genus-0 curvature trichotomy extends to all genera
(cf.\ Computation~\ref{comp:virasoro-curvature}):
\begin{enumerate}
\item \emph{$c = 0$ (uncurved).} $F_g = 0$ for all $g \geq 1$.  The bar complex $\bar{B}(\mathrm{Vir}_0)$ is a genuine cochain complex ($d^2 = 0$) at every genus, and the Koszul duality $\mathrm{Vir}_0^! \simeq \mathrm{Vir}_0$ is of classical type.  No quantum corrections arise.
\item \emph{$c$ generic.} $F_g = (c/2) \cdot \lambda_g^{\mathrm{FP}} \neq 0$ for all $g \geq 1$.  The bar complex is curved, and the curved $A_\infty$ structure is nontrivial at every genus.  The Koszul dual $\mathrm{Vir}_c^! \simeq \mathrm{Vir}_{26-c}$ is a \emph{different} Virasoro algebra, and the complementarity sum is $13 \cdot \lambda_g^{\mathrm{FP}}$.
\item \emph{$c = 26$ (maximal curvature for the uncurved dual).} $F_g(\mathrm{Vir}_{26}) = 13 \cdot \lambda_g^{\mathrm{FP}}$ for all~$g$.  The dual algebra $\mathrm{Vir}_0$ is uncurved, so $\mathrm{Vir}_{26}$ carries the entire complementarity budget.  This is the unique central charge at which the Koszul dual is uncurved.
\end{enumerate}
The generating function:
\begin{equation}\label{eq:vir-generating}
\sum_{g=1}^{\infty} F_g(\mathrm{Vir}_c)\, x^{2g} = \frac{c}{2}\left(\frac{x/2}{\sin(x/2)} - 1\right)
\end{equation}
has radius of convergence $|x| = 2\pi$ (independent of~$c$), reflecting the universal Bernoulli asymptotics of Proposition~\ref{prop:genus-expansion-convergence}.  At $c = 26$:
\[
\sum_{g=1}^\infty F_g(\mathrm{Vir}_{26})\, x^{2g} = 13\left(\frac{x/2}{\sin(x/2)} - 1\right)
\]
\end{remark}

\begin{remark}[Obstruction coefficient and the bosonic string]\label{rem:vir-bosonic-string-genus}
\index{bosonic string!obstruction coefficient}
The Virasoro obstruction coefficient $\kappa(\mathrm{Vir}_c) = c/2$ should be compared with the Heisenberg coefficient $\kappa(\mathcal{H}_\kappa) = \kappa$.  For a single free boson ($c = 1$): $\kappa(\mathcal{H}_1) = 1$ but $\kappa(\mathrm{Vir}_1) = 1/2$.  The factor~$2$ discrepancy reflects the difference between computing $F_g$ for the full vertex algebra (Heisenberg, with spin-$1$ generator $J$) versus its Virasoro subalgebra (with spin-$2$ generator $T = \frac{1}{2}\normord{JJ}$).  The genus expansion is an invariant of the \emph{algebra}, not merely of its central charge.

For the bosonic string matter sector ($26$ free bosons $\mathcal{H}_1^{\oplus 26}$, $c = 26$):
\[
F_g^{\text{matter}} = 26 \cdot \lambda_g^{\mathrm{FP}}, \qquad
F_g^{\text{ghost}} = -13 \cdot \lambda_g^{\mathrm{FP}}
\quad (\text{$bc$ ghost at $\lambda = 2$ has } \kappa = c_{bc}/2 = -13)
\]
so $F_g^{\text{total}} = 13 \cdot \lambda_g^{\mathrm{FP}} = F_g(\mathrm{Vir}_{26})$.
The equality $F_g^{\text{total}} = F_g(\mathrm{Vir}_{26})$ reflects the
bosonization equivalence at the level of genus expansions: the $26$-boson
system contributes $\kappa_{\text{matter}} = 26$ (one per free boson),
while the $bc$ ghost contributes
$\kappa_{\text{ghost}} = -13$, giving
$\kappa_{\text{total}} = 26 + (-13) = 13$.
Note that $\kappa_{\text{total}} = 13 = c_{\text{matter}}/2$
(i.e., $\kappa$ tracks the matter central charge, not $c_{\text{total}} = 0$).
\end{remark}

\frac{|B_{2g}|}{(2g)!}
\end{equation}
for all $g \geq 1$, where the obstruction coefficient is
\begin{equation}\label{eq:fermion-kappa}
\kappa(bc_\lambda) = \frac{c_{bc}}{2}
= -(6\lambda^2 - 6\lambda + 1).
\end{equation}
For the Koszul dual $\beta\gamma$-system,
$\kappa(\beta\gamma_\lambda) = 6\lambda^2 - 6\lambda + 1
= -\kappa(bc_\lambda)$.
\end{theorem}

\begin{proof}
By the genus universality theorem
(Theorem~\ref{thm:genus-universality}) applied to
the $bc$ system as a chiral algebra with stress-energy tensor
$T = \lambda\,(\partial b)\,c + (1-\lambda)\,b\,(\partial c)$.
The genus-$1$ curvature of the bar complex is determined by
the quartic pole $T_{(3)}T = c_{bc}/2$
(the same mechanism as Theorem~\ref{thm:vir-genus1-curvature}
applied to the Virasoro subalgebra), giving
$\kappa = c_{bc}/2 = -(6\lambda^2 - 6\lambda + 1)$.
The Faber--Pandharipande formula
(Theorem~\ref{thm:mumford-formula}) evaluates the
tautological integral.
\end{proof}

\begin{proposition}[$bc$--$\beta\gamma$ complementarity;
\ClaimStatusProvedHere]\label{prop:fermion-complementarity}
\index{complementarity!bc-betagamma@$bc$--$\beta\gamma$}
For all $g \geq 1$ and all conformal weights~$\lambda$:
\begin{equation}\label{eq:fermion-complementarity}
F_g(bc_\lambda) + F_g(\beta\gamma_\lambda) = 0
\end{equation}
\end{proposition}

\begin{proof}
The obstruction coefficients satisfy
$\kappa(bc_\lambda) + \kappa(\beta\gamma_\lambda)
= -(6\lambda^2 - 6\lambda + 1) + (6\lambda^2 - 6\lambda + 1) = 0$.
Since $F_g = \kappa \cdot \lambda_g^{\mathrm{FP}}$ is linear
in~$\kappa$, exact cancellation follows.
\end{proof}

\begin{remark}[Complementarity type classification]
\label{rem:fermion-complementarity-type}
The exact cancellation $\kappa + \kappa' = 0$ for
$bc \leftrightarrow \beta\gamma$ is of the same type
as affine Kac--Moody ($\kappa + \kappa' = 0$,
Proposition~\ref{prop:sl2-complementarity-all-genera}),
and contrasts with the Virasoro case
($\kappa + \kappa' = 13$,
Proposition~\ref{prop:vir-complementarity}).
The mechanism in both cases is the same: the operadic
duality $\mathrm{Lie} \leftrightarrow \mathrm{Com}$ (for
Kac--Moody) and $\mathrm{Ext} \leftrightarrow \mathrm{Sym}$
(for $bc$--$\beta\gamma$) produce level-shift involutions
that negate the obstruction coefficient.
The Virasoro and $\mathcal{W}_N$ algebras, by contrast,
arise from the self-dual operad $\mathrm{Ass}$ via
Drinfeld--Sokolov reduction, and the nonlinear
parametrization $c(k)$ prevents exact cancellation.
\end{remark}

\begin{proposition}[Classification of complementarity types;
\ClaimStatusProvedHere]\label{prop:complementarity-classification}
\index{complementarity!classification of types}
For Koszul chiral algebras, the complementarity sum
$\kappa(\cA) + \kappa(\cA^!)$ falls into one of three types:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Exact cancellation}
($\kappa + \kappa' = 0$):
affine Kac--Moody algebras $\widehat{\fg}_k$,
Heisenberg $\mathcal{H}_\kappa$,
$\mathrm{Ext}/\mathrm{Sym}$ pairs
($bc \leftrightarrow \beta\gamma$),
and lattice VOAs.  In each case, the level-shift involution
acts linearly on~$\kappa$, sending
$\kappa \mapsto -\kappa$.
\item \emph{Nonzero constant}
($\kappa + \kappa' = \text{const} \neq 0$):
Virasoro ($\kappa + \kappa' = 13$),
$\mathcal{W}_3$ ($\kappa + \kappa' = 250/3$),
and general $\mathcal{W}_N$
($\kappa + \kappa' = (c+c')(H_N - 1)$).
The constant depends on the Lie algebra and the
operadic structure, but not on the level.
\item \emph{Zero by triviality}:
algebras with $c = 0$ (topological), where
$\kappa = \kappa' = 0$ and $F_g = 0$ for all~$g$.
\end{enumerate}
\end{proposition}

\begin{proof}
Type~(i):  For affine Kac--Moody, $\kappa(\widehat{\fg}_k)
= (k + h^\vee) d / (2h^\vee)$
(Table~\ref{tab:master-invariants}).  The Feigin--Frenkel
involution $k \mapsto -k - 2h^\vee$ sends
$k + h^\vee \mapsto -(k + h^\vee)$, so
$\kappa \mapsto -\kappa$ and $\kappa + \kappa' = 0$.
The same argument applies to the Heisenberg ($\kappa \mapsto
-\kappa$ under the sign change of the pairing) and to
$bc \leftrightarrow \beta\gamma$ (where
$\kappa(bc) = -\kappa(\beta\gamma)$ by
Theorem~\ref{thm:fermion-all-genera}).

Type~(ii): For $\mathcal{W}_N$,
$\kappa = c \cdot (H_N - 1)$
(Theorem~\ref{thm:wn-obstruction}).
Since $c + c'$ is level-independent
(Theorem~\ref{thm:central-charge-complementarity}),
$\kappa + \kappa' = (c + c')(H_N - 1)$ is a constant
depending only on~$\fg$.

Type~(iii): If $c(\cA) = 0$, then $\kappa = 0$ by the
relation $\kappa = c/2$ (for the Virasoro subalgebra).
\end{proof}

\begin{theorem}[Complementarity sum as root datum invariant; \ClaimStatusProvedHere]
\label{thm:complementarity-root-datum}
\index{complementarity!root datum invariant}
\index{obstruction coefficient!complementarity sum}
For any Koszul pair arising from a simple Lie algebra $\mathfrak{g}$ via the constructions of this monograph, the complementarity sum $\kappa(\mathcal{A}) + \kappa(\mathcal{A}^!)$ is a level-independent invariant of the root datum:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Affine Kac--Moody:} $\kappa(\widehat{\mathfrak{g}}_k) + \kappa(\widehat{\mathfrak{g}}_{k'}) = 0$ \textup{(}exact cancellation\textup{)}.
\item \emph{Principal $\mathcal{W}$-algebra for $\mathfrak{sl}_N$:}
\begin{equation}\label{eq:wn-kappa-sum-general}
\kappa(\mathcal{W}_N^k) + \kappa(\mathcal{W}_N^{k'}) = 2(N{-}1)(2N^2{+}2N{+}1) \cdot (H_N - 1)
\end{equation}
where $H_N - 1 = \sum_{j=2}^N 1/j$.
\item \emph{General $\mathcal{W}(\mathfrak{g}, f_{\mathrm{prin}})$:}
\begin{equation}\label{eq:general-w-kappa-sum}
\kappa(\mathcal{W}^k(\mathfrak{g})) + \kappa(\mathcal{W}^{k'}(\mathfrak{g})) = (c + c') \cdot \sigma(\mathfrak{g})
\end{equation}
where $\sigma(\mathfrak{g}) = \sum_{i=1}^r 1/(m_i + 1)$ depends only on the exponents $m_1, \ldots, m_r$ of $\mathfrak{g}$, and $c + c'$ is level-independent \textup{(}Theorem~\textup{\ref{thm:central-charge-complementarity})}.
\end{enumerate}
\end{theorem}

\begin{proof}
Part~(i) is proved in Proposition~\ref{prop:sl2-complementarity-all-genera}: $\kappa(\widehat{\mathfrak{g}}_k) = (k+h^\vee) d / (2h^\vee)$ is linear in $k+h^\vee$, so the Feigin--Frenkel involution $k \mapsto k' = -k-2h^\vee$ sends $\kappa \mapsto -\kappa$.

Parts~(ii)--(iii): by Theorem~\ref{thm:wn-obstruction} and Corollary~\ref{cor:general-w-obstruction}, $\kappa = c \cdot \sigma(\mathfrak{g})$ where $\sigma(\mathfrak{g}) = \sum 1/(m_i+1)$ depends only on $\mathfrak{g}$.  Therefore $\kappa + \kappa' = (c+c') \cdot \sigma(\mathfrak{g})$.  The sum $c + c'$ is level-independent by Theorem~\ref{thm:central-charge-complementarity}: under $k \mapsto k' = -k-2h^\vee$, the $k$-dependent terms in the central charge formula cancel.  For $\mathfrak{sl}_N$: $\sigma = H_N - 1$ and $c+c' = 2(N-1)(2N^2+2N+1)$ (Table~\ref{tab:master-invariants}).
\end{proof}

\begin{remark}[Operadic origin of the trichotomy]
\label{rem:complementarity-operadic}
\index{operadic structure!complementarity type}
The complementarity type is determined by the operadic
structure underlying the chiral algebra:
\begin{itemize}
\item Type~(i) arises from $\mathrm{Lie}/\mathrm{Com}$
duality (Kac--Moody) or $\mathrm{Ext}/\mathrm{Sym}$ duality
($bc$--$\beta\gamma$), where the operadic Koszul involution
acts linearly on the obstruction parameter.
\item Type~(ii) arises when the chiral algebra is obtained
by Drinfeld--Sokolov reduction from a type~(i) algebra.
The DS functor introduces a nonlinear change of variables
($c$ as a rational function of~$k$), which prevents the
level-shift from negating~$\kappa$.
\item Type~(iii) is degenerate: the vanishing of both
$\kappa$ and~$\kappa'$ reflects the absence of
obstructions at all genera.
\end{itemize}
\end{remark}

\label{comp:sl2-critical-genera}

\index{critical level!vanishing genus expansion}

F_g(\widehat{\mathfrak{sl}}_{2,-2}) = 0

$d^2 = 0$ without corrections, and the cohomology recovers
$\mathfrak{z}(\widehat{\mathfrak{sl}}_{2,-2})$
(Proposition~\ref{prop:km-bar-curvature}).

\begin{remark}[Minimal model genus expansions]
\label{rem:minimal-model-genera}
\index{minimal model!genus expansion}
At a minimal model central charge
$c_{p,q} = 1 - 6(p-q)^2/(pq)$ with $\gcd(p,q) = 1$ and
$p, q \geq 2$, the genus expansion is
$F_g = (c_{p,q}/2) \cdot \lambda_g^{\mathrm{FP}}$.
The Virasoro Koszul dual has central charge
$c' = 26 - c_{p,q}$, which is \emph{not} in general a
minimal model central charge.  For example:
\begin{itemize}
\item Ising $(3,4)$: $c = 1/2$, dual $c' = 51/2$
  (not a minimal model).
\item Tricritical Ising $(4,5)$: $c = 7/10$,
  dual $c' = 253/10$ (not a minimal model).
\item Three-state Potts $(5,6)$: $c = 4/5$,
  dual $c' = 126/5$ (not a minimal model).
\end{itemize}
In each case, the complementarity sum is
$\kappa + \kappa' = c/2 + (26-c)/2 = 13$, reflecting the
universal Virasoro constant (Proposition~\ref{prop:vir-complementarity}).
The minimal model at $c_{p,q}$ and its complement at
$26 - c_{p,q}$ share this genus expansion budget, with
the minimal model carrying a small fraction
($c_{p,q}/26 \ll 1$ for all unitary minimal models) and
the complement carrying the remainder.
\end{remark}

\section{Growth rate analysis}\label{sec:growth-rate-genera}
\index{genus expansion!growth rate}
\index{convergence!genus expansion}

Since every genus expansion takes the factored form
$F_g = \kappa \cdot \lambda_g^{\mathrm{FP}}$
(Theorem~\ref{thm:universal-generating-function}), the
growth rate $|F_{g+1}/F_g|$ is independent of the algebra
and equals the ratio
$\lambda_{g+1}^{\mathrm{FP}}/\lambda_g^{\mathrm{FP}}$.

\begin{proposition}[Universal growth rate;
\ClaimStatusProvedHere]\label{prop:universal-growth-rate}
\index{growth rate!universal limit}
For any Koszul chiral algebra $\cA$ with $\kappa(\cA) \neq 0$:
\begin{equation}\label{eq:universal-growth-rate}
\frac{|F_{g+1}(\cA)|}{|F_g(\cA)|}
= \frac{\lambda_{g+1}^{\mathrm{FP}}}
{\lambda_g^{\mathrm{FP}}}
\;\longrightarrow\; \frac{1}{(2\pi)^2}
\quad \text{as } g \to \infty.
\end{equation}
The genus expansion $\sum_{g=1}^\infty F_g(\cA)$
converges absolutely with radius of convergence $2\pi$
in the variable~$x$ of the generating function~\eqref{eq:universal-generating-function}, independent
of~$\cA$.
\end{proposition}

\begin{proof}
The Faber--Pandharipande coefficient is
$\lambda_g^{\mathrm{FP}} = \frac{2^{2g-1}-1}{2^{2g-1}}
\cdot \frac{|B_{2g}|}{(2g)!}$.  The ratio:
\[
\frac{\lambda_{g+1}^{\mathrm{FP}}}
{\lambda_g^{\mathrm{FP}}}
= \frac{2^{2g+1}-1}{2^{2g+1}} \cdot
\frac{2^{2g-1}}{2^{2g-1}-1} \cdot
\frac{|B_{2g+2}|}{|B_{2g}|} \cdot
\frac{(2g)!}{(2g+2)!}
\]
The first two factors tend to~$1$.  By the asymptotic
formula $|B_{2g}| \sim 2(2g)!/(2\pi)^{2g}$
(from $\zeta(2g) \to 1$):
\[
\frac{|B_{2g+2}|}{|B_{2g}|} \cdot
\frac{(2g)!}{(2g+2)!}
\;\sim\;
\frac{2(2g+2)!/(2\pi)^{2g+2}}{2(2g)!/(2\pi)^{2g}}
\cdot \frac{(2g)!}{(2g+2)!}
= \frac{1}{(2\pi)^2}
\]
giving the limit~\eqref{eq:universal-growth-rate}.
The radius of convergence of the generating function
$\sum \lambda_g^{\mathrm{FP}} x^{2g} =
(x/2)/\sin(x/2) - 1$ is determined by the first
singularity of $(x/2)/\sin(x/2)$, namely the zero of
$\sin(x/2)$ at $x = 2\pi$.
\end{proof}

\begin{remark}[Contrast with string theory amplitudes]
\label{rem:growth-vs-string}
\index{string amplitudes!growth rate contrast}
The convergence of the genus expansion
($|F_g| \sim 2|\kappa|/(2\pi)^{2g}$,
geometric decay) contrasts sharply with string theory
scattering amplitudes, which grow as $(2g)!$ and produce
\emph{divergent} perturbation series requiring Borel
summation (see Remark~\ref{rem:convergence-vs-string}).
The distinction is structural: the genus expansion
$F_g = \kappa \cdot \lambda_g^{\mathrm{FP}}$ extracts a
\emph{single} tautological intersection number
$\int_{\overline{\mathcal{M}}_{g,1}} \psi_1^{2g-2}
\lambda_g$ at each genus, while string amplitudes
integrate over all of $\overline{\mathcal{M}}_g$ with a
measure involving all $\lambda$-classes.  The Bernoulli
decay of $\lambda_g^{\mathrm{FP}}$ overwhelms any
polynomial growth from the algebra.

The factored form $F_g = \kappa \cdot
\lambda_g^{\mathrm{FP}}$ reveals that the algebra
contributes only through the single number~$\kappa$:
the moduli space geometry (encoded in
$\lambda_g^{\mathrm{FP}}$) provides the universal
decay rate, while $\kappa$ sets the overall scale.
This separation of algebraic and geometric data is a
consequence of the genus universality theorem
(Theorem~\ref{thm:genus-universality}).
\end{remark}

\section{Multiplicative genus from Koszul duality}\label{sec:multiplicative-genus}
\index{multiplicative genus|textbf}
\index{genus expansion!multiplicative genus}

\begin{proposition}[Koszul duality determines a multiplicative genus; \ClaimStatusProvedHere]
\label{prop:multiplicative-genus}
Every Koszul chiral algebra $\mathcal{A}$ determines a multiplicative sequence via the generating function of Theorem~\textup{\ref{thm:universal-generating-function}}:
\begin{equation}\label{eq:multiplicative-genus}
\phi_{\mathcal{A}}(x) = \kappa(\mathcal{A}) \cdot \frac{x/2}{\sin(x/2)}
\end{equation}
This is the modified Todd class \textup{(}up to the additive constant $-\kappa$\textup{)}, related to the $\hat{A}$-genus by $x \mapsto ix$:
\begin{equation}
\hat{A}(x) = \frac{x/2}{\sinh(x/2)} \;=\; \phi_{\mathcal{A}}(ix)/\kappa.
\end{equation}
\end{proposition}

\begin{proof}
The generating function $\sum_{g \geq 1} F_g x^{2g} = \kappa \cdot ((x/2)/\sin(x/2) - 1)$ (Theorem~\ref{thm:universal-generating-function}) identifies $\phi_{\mathcal{A}}(x) = \kappa + \sum_{g \geq 1} F_g x^{2g} = \kappa \cdot (x/2)/\sin(x/2)$.  The function $(x/2)/\sin(x/2)$ is the generating function of the modified Todd class: its Taylor coefficients $\frac{2^{2g-1}-1}{2^{2g-1}} \cdot \frac{|B_{2g}|}{(2g)!}$ are the Faber--Pandharipande integrals $\lambda_g^{\mathrm{FP}}$.  The relation $\sin(ix) = i\sinh(x)$ gives $(x/2)/\sin(x/2)|_{x \mapsto ix} = (ix/2)/(i\sinh(x/2)) = (x/2)/\sinh(x/2) = \hat{A}(x)$.
\end{proof}

\begin{corollary}[Koszul complementarity of genera; \ClaimStatusProvedHere]
\label{cor:complementary-genera}
\index{multiplicative genus!complementarity}
For a Koszul pair $(\mathcal{A}, \mathcal{A}^!)$:
\begin{equation}\label{eq:genus-complementarity}
\phi_{\mathcal{A}}(x) + \phi_{\mathcal{A}^!}(x) = (\kappa + \kappa') \cdot \frac{x/2}{\sin(x/2)}
\end{equation}
The right-hand side depends only on the root datum \textup{(}Theorem~\textup{\ref{thm:complementarity-root-datum})}.  For affine Kac--Moody algebras \textup{(}$\kappa + \kappa' = 0$\textup{)}, the complementary genera cancel: $\phi_{\mathcal{A}} = -\phi_{\mathcal{A}^!}$.  For Virasoro \textup{(}$\kappa + \kappa' = 13$\textup{)}, the sum is $13 \cdot (x/2)/\sin(x/2)$.
\end{corollary}

\begin{proof}
Immediate from $\phi_{\mathcal{A}} = \kappa \cdot (x/2)/\sin(x/2)$ and linearity in~$\kappa$.
\end{proof}

\begin{remark}[Relation to index theory]\label{rem:genus-index-theory}
\index{index theory!genus expansion}
The appearance of the modified Todd class $(x/2)/\sin(x/2)$ (rather than the $\hat{A}$-genus $(x/2)/\sinh(x/2)$) reflects the \emph{positive} Faber--Pandharipande coefficients: all $\lambda_g^{\mathrm{FP}} > 0$ (Remark~\ref{rem:A-hat-genus}).  In the Atiyah--Singer framework, the $\hat{A}$-genus governs the Dirac operator and has alternating signs.  The Wick rotation $x \mapsto ix$ relates the two generating functions, but the underlying geometry is distinct: the genus expansion operates on the algebraic moduli space $\overline{\mathcal{M}}_g$ (with tautological intersection theory), while the $\hat{A}$-genus operates on smooth manifolds (with characteristic classes of the tangent bundle).
\end{remark}
