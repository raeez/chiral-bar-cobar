\section*{Master table of computed invariants}
\label{sec:master-table}
\index{master table of invariants|textbf}

Table~\ref{tab:master-invariants} records, for each Koszul chiral
algebra~$\cA$ computed in this monograph:
\begin{itemize}
\item the Koszul dual~$\cA^!$;
\item the central charge~$c(\cA)$;
\item the level-independent sum $c(\cA) + c(\cA^!)$ under Koszul duality

\begin{remark}[Genus-$1$ free energy and the conformal anomaly]\label{rem:genus-1-verification}
\index{free energy!genus one}
The Faber--Pandharipande $\lambda_g$ formula (Theorem~\ref{thm:genus-universality}(iii))
gives the genus-$g$ free energy
$F_g(\cA) = \kappa(\cA) \cdot \frac{2^{2g-1}-1}{2^{2g-1}} \cdot \frac{|B_{2g}|}{(2g)!}$.
At $g = 1$, $|B_2| = 1/6$, yielding
\[
F_1(\cA) = \frac{\kappa(\cA)}{24}.
\]
For the Heisenberg algebra at level $\kappa = 1$ (single free boson, $c = 1$),
$F_1(\mathcal{H}_1) = 1/24 = c/24$---precisely the conformal anomaly coefficient
$q^{-c/24}$ in the torus partition function $Z(\tau) = q^{-c/24}\sum_h d(h)\,q^h$.
For $\widehat{\mathfrak{sl}}_2$ at level~$k$:
$F_1 = 3(k{+}2)/96 = (k{+}2)/32$.
In general, $F_1 = \kappa/24$ differs from the conformal anomaly $c/24$ by the ratio $\kappa/c$, which is~$1$ for the Heisenberg algebra but varies for other algebras.
\end{remark}

\begin{remark}[Reading the table]\label{rem:reading-master-table}
Three structural features are visible from the table:
\begin{enumerate}
\item \emph{Complementarity.}
  Every $c + c'$ entry is independent of the level~$k$: it depends
  only on the root datum of~$\fg$.
  This is the content of Theorem~\ref{thm:central-charge-complementarity}.
\item \emph{Critical level.}
  Setting $\kappa(\cA) = 0$ (i.e., $t = k + h^\vee = 0$)
  simultaneously forces $\operatorname{obs}_g = 0$
  for all genera: the bar complex becomes \emph{uncurved}, and
  the Feigin--Frenkel center $\mathfrak{z}(\widehat{\fg}_{-h^\vee})$
  emerges as the degree-zero cohomology.
\item \emph{Free fields.}
  The exact Koszul dualities ($\psi \leftrightarrow \gamma$,
  $bc \leftrightarrow \beta\gamma$) satisfy
  $\kappa(\cA) + \kappa(\cA^!) = 0$
  (Theorem~\ref{thm:fermion-all-genera}): the individual
  obstruction coefficients $\kappa(\psi) = \tfrac{1}{4}$,
  $\kappa(bc) = c/2$ are nonzero (they measure the
  Virasoro subalgebra curvature), but the tensor product
  $bc \otimes \beta\gamma$ is uncurved.
  The Heisenberg is the abelian curved case, with
  $\kappa = \kappa$ recording the central extension.
\item \emph{Gaps.}
  The Heisenberg entry $c + c' = \text{---}$ reflects the fact that
  $\mathcal{H}_\kappa^! = \mathrm{Sym}^{\mathrm{ch}}(V^*)$ is
  curved (not a vertex algebra in the traditional sense), so
  its central charge is not defined by the Sugawara construction.
  The Yangian entries are blank because $Y(\fg)$ is an $\Eone$-chiral
  algebra, not conformally graded: it has no Virasoro subalgebra
  and hence no central charge or obstruction coefficient in the
  sense of the other rows.
\end{enumerate}
\end{remark}

\section*{Key results by category}

\emph{$\Einf$-Chiral Algebras (Vertex Algebras).}
\begin{enumerate}[label=(\roman*)]
\item \emph{Heisenberg}: Koszul dual is the commutative chiral algebra
$\mathrm{Sym}^{\mathrm{ch}}(V^*)$ with curvature $m_0 = -\kappa \cdot \mathbf{1}$;
the bar complex is the Chevalley--Eilenberg coalgebra
(Theorem~\ref{thm:heisenberg-koszul-dual-early}).
Complete genus-$1$ pipeline for $\widehat{\mathfrak{sl}}_2$ computed
in \S\ref{sec:sl2-genus-one-pipeline}.
\item \emph{$bc$--$\beta\gamma$}: Clifford/Sym exchange;
bar construction $\bar{B}(bc) = \Lambda^c(V^*)$, Koszul dual
$(bc)^! \cong \beta\gamma$
(\S\ref{sec:betagamma-koszul-dual}).
\item \emph{Affine Kac--Moody}: Koszul dual is
$\widehat{\fg}_{-k-2h^\vee}$; explicit computation through degree~$3$
for $\mathfrak{sl}_2$ (Theorem~\ref{thm:sl2-koszul-dual}),
general case via Serre duality on FM spaces
(Theorem~\ref{thm:universal-kac-moody-koszul}).
\item \emph{Virasoro and $\mathcal{W}$-algebras}:
$\mathcal{W}^k(\fg)^! \cong \mathcal{W}^{-k-2h^\vee}(\fg)$
(Theorem~\ref{thm:w-algebra-koszul-main}).
Complete genus-$1$ pipelines for $\mathrm{Vir}_c$
(\S\ref{sec:virasoro-genus-one-pipeline}, $c + c' = 26$, $\kappa = c/2$)
and $\mathcal{W}_3^k$
(\S\ref{sec:w3-genus-one-pipeline}, $c + c' = 100$, $\kappa = 5c/6$).
General $\mathcal{W}_N$: $\kappa = c \cdot (H_N - 1)$
(Theorem~\ref{thm:wn-obstruction}).
\end{enumerate}

\emph{$\Eone$-Chiral Algebras (Nonlocal Vertex Algebras).}
\begin{enumerate}[label=(\roman*)]
\item Lattice algebras with non-symmetric cocycles: first strictly $\Eone$ examples;
Koszul dual uses inverse cocycle.
\item R-twisted vertex algebras: Yang--Baxter equation ensures associativity;
Yangians are fundamental examples.
\item Yangians: $Y(\fg)^! \cong Y_{R^{-1}}(\fg)$, self-dual at the classical
level (Theorem~\ref{thm:yangian-koszul-dual}).
\item Toroidal and elliptic algebras: double affine structures with theta function OPEs.
\end{enumerate}

Deformation quantization.
\begin{enumerate}[label=(\roman*)]
\item Coisson algebras (PVA) quantize to $\Einf$-chiral algebras (vertex algebras)
via configuration space integrals (Theorem~\ref{thm:chiral-kontsevich}).
\item $\Pinf$-chiral algebras ($\Einf + L_\infty$) quantize to $\Eone$-chiral
algebras (nonlocal VAs; Construction~\ref{constr:quantization}).
\item Maurer--Cartan elements classify deformations at each level.
\end{enumerate}

\section*{Bar complex dimensions}
\label{sec:bar-dimensions}
\index{bar complex!dimension table|textbf}

Table~\ref{tab:bar-dimensions} records the dimension of the degree-$n$
bar \emph{cohomology} $\dim H^n(\barBgeom(\cA))$ for each algebra,
through degree~$5$.
(For Kac--Moody algebras, the \emph{chain-group} dimensions are much
larger: $\dim \bar{B}^n = (\dim\mathfrak{g})^n \cdot (n{-}1)!$,
the product of the generator factor and the Orlik--Solomon form
factor; see Remark~\ref{rem:bar-dims-level-independent}.)
These cohomology dimensions are computed from the explicit
generators-and-relations
presentations of the bar complex given in Part~XI;
see Section~\ref{sec:cross-ref-guide} below for precise references.
A dash~(---) indicates that the value is not currently known.
Six of the nine algebras in Table~\ref{tab:bar-dimensions} now have
closed-form generating functions (Heisenberg, free fermion, $bc$,
$\beta\gamma$, $\widehat{\mathfrak{sl}}_2$, Virasoro).
For $\widehat{\mathfrak{sl}}_3$, the degree-$4$ bar cohomology
requires computing the rank of a $24576 \times 786432$ matrix
(the chain-group dimensions grow as $8^n \cdot (n{-}1)!$),
which exceeds practical hand computation.  The $\mathcal{W}_3$
and Yangian $Y(\mathfrak{sl}_2)$ gaps similarly await either explicit machine
computation or the discovery of closed-form generating functions.

\begin{proposition}[Bar complex dimensions; \ClaimStatusProvedHere]
\label{prop:bar-dimensions}
The bar complex dimensions for the algebras of
Table~\textup{\ref{tab:master-invariants}} are as listed in
Table~\textup{\ref{tab:bar-dimensions}}.
\end{proposition}

\begin{proof}
Each entry is computed from the explicit generators-and-relations
presentation of the bar complex established in the referenced theorems.
The explicit formulas are:
\begin{enumerate}
\item \emph{Heisenberg} ($d = 1$, commutative):
  $\dim H^1 = 1$; for $n \geq 2$,
  $\dim H^n = p(n-2)$ where $p$ is the partition function
  (Theorem~\ref{thm:heisenberg-bar}).
\item \emph{Free fermion} ($d = 1$, fermionic):
  $\dim H^n = p(n-1)$
  (Theorem~\ref{thm:fermion-bar-coalg}).
\item \emph{$bc$ ghosts} ($d = 2$, fermionic):
  $\dim H^n = 2^n - n + 1$
  (Theorem~\ref{thm:betagamma-bar-complex},
  Proposition~\ref{prop:bc-betagamma-orthogonality}).
\item \emph{$\beta\gamma$ system} ($d = 2$, bosonic):
  at the lowest-weight truncation,
  $\dim H^n = [x^n]\sqrt{(1{+}x)/(1{-}3x)}$,
  with recurrence $n \cdot a(n) = 2n \cdot a(n{-}1) + 3(n{-}2) \cdot a(n{-}2)$
  (Theorem~\ref{thm:ds-bar-gf-discriminant}).
  See Remark~\ref{rem:betagamma-conventions}
  for the relation between different truncation conventions.
\item \emph{$\widehat{\mathfrak{sl}}_2$}:
  $\dim H^n = R(n+3)$ where $R$ denotes the Riordan numbers
  (OEIS~A005043).  This follows from the PBW basis structure
  (Theorem~\ref{thm:universal-kac-moody-koszul}) and the
  Koszul resolution of $\widehat{\mathfrak{sl}}_2$.
\item \emph{$\widehat{\mathfrak{sl}}_3$}: computed through
  degree~$3$ from the PBW basis.
\item \emph{Virasoro}:
  $\dim H^n = M(n{+}1) - M(n)$ where $M(n)$ is the $n$-th
  Motzkin number (OEIS~A001006).  This reflects the
  associahedron structure of $\overline{M}_{0,n+2}$
  (Theorem~\ref{thm:spectral-sequence-collapse}).
\item \emph{$\mathcal{W}_3$}: computed through degree~$4$
  in \S\ref{sec:w3-bar-degree3}.
\item \emph{Yangian $Y(\mathfrak{sl}_2)$}: computed through
  degree~$3$ from the RTT presentation
  (Theorem~\ref{thm:yangian-bar-rtt}).
\end{enumerate}
\end{proof}

\begin{remark}[Growth rates and Koszulness]\label{rem:bar-growth-koszul}
\index{Koszul algebra!bar complex growth}
The growth rate of $\dim H^n(\barBgeom(\cA))$ as $n \to \infty$ is
an invariant of the algebra with direct Koszul-theoretic meaning.
(Note: the bar \emph{chain-group} dimensions grow much faster;
for Kac--Moody algebras $\dim \bar{B}^n = (\dim\fg)^n \cdot (n{-}1)!$
is super-exponential.  The growth rates below refer to bar
\emph{cohomology}.)
\begin{enumerate}
\item \emph{Sub-exponential growth.}
  The Heisenberg and free fermion bar cohomologies grow
  sub-exponentially: $\dim H^n \sim
  e^{\pi\sqrt{2n/3}}/(4\sqrt{3}\,n)$ by the
  Hardy--Ramanujan asymptotics for the partition function.
\item \emph{Exponential growth.}
  All other algebras in the table have bar cohomology growing
  exponentially in~$n$, at rate~$\sim d^n$ where $d = \dim\fg$
  for Kac--Moody.
  The explicit formulas are:
  \begin{itemize}
  \item $bc$: $\dim H^n = 2^n - n + 1$
    (growth rate~$2^n$);
  \item $\beta\gamma$: $\dim H^n = [x^n]\sqrt{(1{+}x)/(1{-}3x)}$
    (growth rate~$3^n/n^{1/2}$);
  \item $\widehat{\mathfrak{sl}}_2$: $\dim H^n = R(n+3)$,
    the Riordan numbers (growth rate~$3^n/n^{3/2}$);
  \item Virasoro: $\dim H^n = M(n{+}1) - M(n)$,
    the first differences of Motzkin numbers
    (growth rate~$3^n/n^{3/2}$).
  \end{itemize}
\item In general, for a Koszul algebra, the exponential
  growth rate of $\dim H^n(\barBgeom)$ equals the
  exponential growth rate of the Koszul dual $\cA^!$.
  The Virasoro bar cohomology growth~$3^n$ is consistent
  with the infinite-dimensional nature of $\mathrm{Vir}_{26-c}$.
\end{enumerate}
\end{remark}

\begin{theorem}[DS reduction and bar cohomology generating functions; \ClaimStatusProvedHere]
\label{thm:ds-bar-gf-discriminant}
\index{Drinfeld--Sokolov reduction!bar cohomology generating function}
\index{bar complex!generating function!shared discriminant}
\index{Riordan numbers!relation to Motzkin numbers}
Let $\mathfrak{g}$ be a simple Lie algebra and
$\mathcal{W}^k(\mathfrak{g}) = H^0_{\mathrm{DS}}(\widehat{\mathfrak{g}}_k)$
the associated $\mathcal{W}$-algebra.
Define the bar cohomology generating functions
\[
P_{\widehat{\mathfrak{g}}}(x)
  = \sum_{n \geq 1} \dim H^n(\barBgeom(\widehat{\mathfrak{g}}_k))\, x^n,
\qquad
P_{\mathcal{W}}(x)
  = \sum_{n \geq 1} \dim H^n(\barBgeom(\mathcal{W}^k(\mathfrak{g})))\, x^n.
\]
Then $P_{\widehat{\mathfrak{g}}}$ and $P_{\mathcal{W}}$ are algebraic
functions sharing the same discriminant.

Explicitly, for $\mathfrak{g} = \mathfrak{sl}_2$:
\begin{align}
P_{\widehat{\mathfrak{sl}}_2}(x)
  &= \frac{1 + x - \sqrt{1 - 2x - 3x^2}}{2x(1+x)},
  \label{eq:gf-sl2-riordan} \\
P_{\mathrm{Vir}}(x)
  &= \frac{4x}{(1 - x + \sqrt{1 - 2x - 3x^2})^2},
  \label{eq:gf-vir-motzkin} \\
P_{\beta\gamma}(x)
  &= \sqrt{\frac{1+x}{1-3x}}
  = \frac{1+x}{\sqrt{1-2x-3x^2}}.
  \label{eq:gf-betagamma}
\end{align}
All three are algebraic functions of the discriminant
$\Delta(x) = 1 - 2x - 3x^2 = (1-3x)(1+x)$.
In particular, all three generating functions are ramified at
$x = 1/3$ and $x = -1$, and all three sequences share the
exponential growth rate~$3^n$
(with sub-exponential corrections $n^{-3/2}$ for
$\widehat{\mathfrak{sl}}_2$ and Virasoro,
$n^{-1/2}$ for~$\beta\gamma$).
\end{theorem}

\begin{proof}
The formulas~\eqref{eq:gf-sl2-riordan}
and~\eqref{eq:gf-vir-motzkin} are proved as follows.

\emph{$\widehat{\mathfrak{sl}}_2$ (Riordan).}
The bar cohomology $\dim H^n(\barBgeom(\widehat{\mathfrak{sl}}_{2,k}))
= R(n+3)$ where $R(n)$ are the Riordan numbers
(Theorem~\ref{thm:universal-kac-moody-koszul}).  The Riordan numbers
have generating function $\sum_{n \geq 0} R(n) x^n
= (1 + x - \sqrt{1-2x-3x^2})/(2x(1+x))$
(OEIS~A005043, \cite{BernhartRiordan}).
Shifting the index by $3$ gives~\eqref{eq:gf-sl2-riordan}.

\emph{Virasoro (Motzkin differences).}
The bar cohomology $\dim H^n(\barBgeom(\mathrm{Vir}_c))
= M(n{+}1) - M(n)$ where $M(n)$ are the Motzkin numbers
(Theorem~\ref{thm:spectral-sequence-collapse}).  The Motzkin
generating function is $M(x) = (1 - x - \sqrt{1-2x-3x^2})/(2x^2)$
(OEIS~A001006).  Then
$\sum_{n \geq 0}(M(n{+}1) - M(n))x^n = (M(x) - 1)/x - M(x) + 1$;
simplification gives~\eqref{eq:gf-vir-motzkin}.

\emph{$\beta\gamma$ system.}
At the lowest-weight truncation
(Remark~\ref{rem:betagamma-conventions}), the bar cohomology
satisfies the recurrence
$n \cdot a(n) = 2n \cdot a(n{-}1) + 3(n{-}2) \cdot a(n{-}2)$
with $a(0) = 1$, $a(1) = 2$.  This is the coefficient recurrence
for the ODE $(1{+}x)(1{-}3x)H'(x) = 2H(x)$, whose solution with
$H(0) = 1$ is $H(x) = \sqrt{(1{+}x)/(1{-}3x)}$,
giving~\eqref{eq:gf-betagamma}.
The first values $2, 4, 10, 26, 70, 192, 534, 1500$
are OEIS~A025565 (shifted by one).

\emph{Shared discriminant.}
All three generating functions are algebraic of degree~$2$ over
$\mathbb{Q}(x)$, with the same branch locus
$\Delta(x) = 1 - 2x - 3x^2 = 0$, i.e.\ $x = 1/3$ and $x = -1$.
The branch point at $x = 1/3$ (radius of convergence) determines the
exponential growth rate $3^n$ for all three sequences.
The sub-exponential correction $n^{-3/2}$ for
$\widehat{\mathfrak{sl}}_2$ and Virasoro follows from the
square-root singularity
(Theorem~VI.1 of~\cite{FlajoletSedgewick});
the $\beta\gamma$ correction is $n^{-1/2}$, reflecting the
$\Delta^{-1/2}$ (rather than $\Delta^{1/2}$) singularity type.

\emph{Explanation via DS reduction and Wakimoto.}
Since $\mathrm{Vir}_c = \mathcal{W}^k(\mathfrak{sl}_2)$ by
Drinfeld--Sokolov reduction, the functorial commutative
diagram~\eqref{eq:ds-kd-square}
(Theorem~\ref{thm:ds-koszul-intertwine}) gives
$\barBgeom(\mathrm{Vir}_c) \simeq
H^0_{\mathrm{DS}}(\barBgeom(\widehat{\mathfrak{sl}}_{2,k}))$.
The DS functor $H^0_{\mathrm{DS}}$ is a BRST cohomology that preserves
the algebraicity class of generating functions:
it acts on the bar complex as a \emph{rational} operation
(projection to BRST cohomology followed by a spectral-sequence
collapse), which transforms one algebraic function of
$\sqrt{\Delta(x)}$ into another.  The discriminant $\Delta(x)$
is therefore an invariant of the DS orbit.

The $\beta\gamma$ system enters the same discriminant family
through the Wakimoto representation:
$\widehat{\mathfrak{sl}}_{2,k}$ embeds into
$\beta\gamma \otimes \mathcal{H}$ (Theorem~\ref{thm:wakimoto-bar}),
and the embedding preserves the algebraic structure of bar
generating functions.  Since the Heisenberg factor~$\mathcal{H}$
contributes only sub-exponential growth
($\dim H^n = p(n{-}2)$, the partition function),
the exponential growth rate~$3^n$ and the discriminant~$\Delta(x)$
are inherited from $\widehat{\mathfrak{sl}}_2$ by both the
Virasoro quotient (via DS) and the $\beta\gamma$ factor (via Wakimoto).
\end{proof}

\begin{remark}[Discriminant as DS invariant]\label{rem:ds-discriminant-invariant}
\index{Drinfeld--Sokolov reduction!discriminant invariant}
Theorem~\ref{thm:ds-bar-gf-discriminant} suggests a general
principle: for any simple $\mathfrak{g}$, the discriminant
$\Delta_{\mathfrak{g}}(x)$ of the bar cohomology generating function
$P_{\widehat{\mathfrak{g}}}(x)$ should be preserved under DS reduction.
That is, $P_{\mathcal{W}^k(\mathfrak{g})}(x)$ should be algebraic
with the same branch locus as $P_{\widehat{\mathfrak{g}}}(x)$.
This would imply that $\widehat{\mathfrak{g}}_k$ and
$\mathcal{W}^k(\mathfrak{g})$ always have the same exponential
growth rate for bar cohomology dimensions.

For the remaining cases, the prediction is testable:
the $\mathcal{W}_3 = \mathcal{W}^k(\mathfrak{sl}_3)$ sequence
$2, 5, 16, 52, \ldots$ should be algebraic with discriminant
matching that of $P_{\widehat{\mathfrak{sl}}_3}(x)$, once the latter
is determined.
\end{remark}

\begin{remark}[$\beta\gamma$ bar dimensions: conventions]\label{rem:betagamma-conventions}
\index{betagamma@$\beta\gamma$ system!bar dimension conventions}
The $\beta\gamma$ bar complex is infinite-dimensional in each bar
degree $n \geq 1$ (Theorem~\ref{thm:bg-bar-coalg}(2)):
without a conformal weight bound,
$\dim(\bar{B}^n) = \infty$, since the augmentation ideal
$\overline{\beta\gamma}$ contains generators $\beta_{-m-\lambda}$
and $\gamma_{-m-1+\lambda}$ for all $m \geq 0$.
The generating function in the bigraded sense is
\[
\sum_{n,h} \dim(\bar{B}^n_h)\, q^h t^n
= \prod_{m \geq 0}
  \frac{1}{(1-q^{m+\lambda}t)(1-q^{m+1-\lambda}t)}.
\]
The values in Table~\ref{tab:bar-dimensions}
are \emph{bar cohomology} dimensions $\dim H^n(\bar{B}(\beta\gamma))$
at the lowest-weight truncation (only the modes
$\beta_{-\lambda}$ and $\gamma_{\lambda-1}$, no conformal descendants),
with closed-form generating function
$\sum_{n \geq 0} \dim H^n \cdot x^n = \sqrt{(1{+}x)/(1{-}3x)}$
(Theorem~\ref{thm:ds-bar-gf-discriminant}).

The conjecture $\dim(\bar{B}^n) = 2 \cdot 3^{n-1}$
(Conjecture~\ref{conj:betagamma-bar-dim}) gives the \emph{chain space}
dimension---the total number of bar-$n$ elements before passing to
cohomology.  This is the $\beta\gamma$ analogue of the Kac--Moody
formula $\dim(\bar{B}^n(\hat{\fg}_k)) = (\dim\fg)^n \cdot (n{-}1)!$
(Remark~\ref{rem:bar-dims-level-independent}).  In both cases,
the chain space is much larger than bar cohomology:
\begin{center}
\begin{tabular}{c|ccccc}
$n$ & 1 & 2 & 3 & 4 & 5 \\
\hline
chain $2 \cdot 3^{n-1}$ & 2 & 6 & 18 & 54 & 162 \\
cohom (Table) & 2 & 4 & 10 & 26 & 70
\end{tabular}
\end{center}
The inequality chain $\geq$ cohomology holds at every degree.
\end{remark}

\begin{remark}[Complementarity at each genus]\label{rem:genus-complementarity}
\index{complementarity!genus-g}
By Theorem~\ref{thm:quantum-complementarity-main},
$\operatorname{obs}_g(\cA) + \operatorname{obs}_g(\cA^!)
 = \lambda_g \cdot (\kappa + \kappa')$.
For affine Kac--Moody algebras,
$\kappa(\widehat{\fg}_k) + \kappa(\widehat{\fg}_{-k-2h^\vee}) = 0$
(the shifted level $t = k + h^\vee$ maps to $-t$, and
$\kappa = td/(2h^\vee)$ is linear in~$t$).
Hence $\operatorname{obs}_g(\widehat{\fg}_k)
 + \operatorname{obs}_g(\widehat{\fg}_{k'}) = 0$.
For $\widehat{\mathfrak{sl}}_2$:
$3(k{+}2)/96 + 3(-k{-}2)/96 = 0$.
At the critical level $k = -h^\vee$, both $\kappa = 0$ and $\kappa' = 0$,
so $\operatorname{obs}_g = \operatorname{obs}_g' = 0$.

For $\mathcal{W}$-algebras, $\kappa + \kappa' \neq 0$ in general:
$\kappa + \kappa' = 13$ for the Virasoro algebra and $250/3$ for
$\mathcal{W}_3$ (see Theorem~\ref{thm:genus-universality}(ii)
and Remark~\ref{rem:vir-vs-km-complementarity}).
\end{remark}

\begin{proposition}[Spectral sequence collapse; \ClaimStatusProvedHere]
\label{prop:spectral-collapse-summary}
\index{Koszul algebra!spectral sequence characterization}
For the algebras listed in Table~\textup{\ref{tab:spectral-data}}:
\begin{enumerate}
\item The free fermion, $bc$--$\beta\gamma$, and lattice VOA
  bar complexes are exact Koszul: the spectral sequence collapses
  at~$E_1$ and the bar differential is strict ($d^2 = 0$ on generators).
\item The Heisenberg, generic affine Kac--Moody, generic Virasoro,
  and generic $\mathcal{W}_3$ bar complexes are Koszul: the spectral
  sequence collapses at~$E_2$, and the dual algebra is identified
  with $E_2^{*,0}$.
\item At the critical level $k = -h^\vee$, the affine Kac--Moody
  spectral sequence has nontrivial differentials beyond~$E_2$,
  reflecting the appearance of the Feigin--Frenkel center
  $\mathfrak{z}(\widehat{\fg}_{-h^\vee})$ in the abutment.
\end{enumerate}
\end{proposition}

\begin{proof}
Claims (1) and (2) are proved in the individual example chapters:
Theorem~\ref{thm:fermion-bar-coalg} (fermion),
Theorem~\ref{thm:betagamma-bc-koszul} ($bc$--$\beta\gamma$),
Theorem~\ref{thm:heisenberg-bar} (Heisenberg),
Theorem~\ref{thm:universal-kac-moody-koszul} (Kac--Moody),
Theorem~\ref{thm:w-algebra-koszul-main} ($\mathcal{W}$-algebras),
and Theorem~\ref{thm:lattice:unimodular-self-dual} (lattice VOA).
Claim (3) at the critical level follows from
Theorem~\ref{thm:spectral-sequence-collapse}:
$E_2$-collapse is equivalent to Koszulness, and at $k = -h^\vee$
the bar complex acquires additional cohomology from the
Feigin--Frenkel center, so the spectral sequence does not degenerate
at~$E_2$.
\end{proof}

\begin{remark}[Yangian collapse conjecture]\label{rem:yangian-collapse-conj}
\ClaimStatusConjectured
\index{Yangian!spectral sequence collapse}
We expect the spectral sequence for $\barBgeom(Y(\fg))$ to collapse
at~$E_2$, which would establish the Koszulness of the Yangian as
an $\Eone$-chiral algebra.
The RTT presentation (Theorem~\ref{thm:yangian-bar-rtt}) gives a
quadratic-dual coalgebra whose Hilbert series is consistent with
$E_2$-collapse through degree~$3$.
A proof would require extending the FM compactification arguments
of Theorem~\ref{thm:spectral-sequence-collapse} to the $\Eone$
setting.
The scope of this extension is discussed in
Remark~\ref{rem:yangian-three-theorems}.
\end{remark}
