\section{Setup and frontier}
\label{sec:exceptional-yangian-frontier}

The Yangian Koszul-duality theorem
(Theorem~\ref{thm:yangian-koszul-dual}) states that, on the $E_1$-chiral
Koszul locus, the quadratic Koszul dual of $Y_\hbar(\fg)$ in the RTT
presentation is the Yangian with inverse $R$-matrix, equivalently the
same algebra with $\hbar \mapsto -\hbar$:
\[
Y_\hbar(\fg)^{!,\mathrm{ch}}
\;\cong\;
Y_{R^{-1}}(\fg)^{\mathrm{ch}}
\;=\;
Y_{-\hbar}(\fg)^{\mathrm{ch}}.
\]
The proof has four chain-level inputs.

\begin{enumerate}[label=(\roman*)]
\item \emph{PBW basis.} A Poincar\'e--Birkhoff--Witt basis for
$Y_\hbar(\fg)$ compatible with the level filtration
$F_r Y_\hbar(\fg) = \langle T_{ij}^{(s)} : s \le r \rangle$, giving
$\mathrm{gr}_F Y_\hbar(\fg) \cong U(\fg[t])$ canonically.
\item \emph{Quadratic presentation.} The RTT relations are quadratic in
the generators, with relation space $R \subset V \otimes V$ of the
expected $(1/2)\dim \fg (\dim \fg - 1)$-shape.
\item \emph{Orthogonal-complement computation.} The orthogonal complement
$R^\perp \subset V^* \otimes V^*$ under the trace pairing coincides with
the RTT relation space for $R^{-1}(u)$, equivalently $R(u;-\hbar)$.
\item \emph{Chevalley extension.} The Chevalley involution on $\fg$ lifts
to an involution $\sigma \colon Y_\hbar(\fg) \to Y_\hbar(\fg)$ with
$\sigma \cdot \hbar = -\hbar$, interchanging the two ambient spectral
conventions and realising the Koszul-duality isomorphism.
\end{enumerate}

Inputs (ii), (iii), (iv) are generic in $\fg$: they depend only on the
Cartan datum $(A_{ij}, (\alpha_i, \alpha_j))$ and the Drinfeld--Chevalley
generators, which every simple Lie algebra possesses. Input (i) is
non-generic: it is proved case-by-case.

Classical types $(A_n, B_n, C_n, D_n)$: Molev~\cite{Molev07}
proves PBW via the RTT-to-Drinfeld comparison and the symmetrisation map,
using the orthogonal/symplectic realisation of $B_n, C_n, D_n$ inside
$\mathfrak{gl}_N$ and the explicit centraliser calculation.

Exceptional types $(E_6, E_7, E_8, F_4, G_2)$: Guay--Regelskis--Wendlandt
\cite{GRW18} (arXiv:1706.05176) prove PBW by a uniform Drinfeld
second-presentation argument that does not rely on a classical matrix
realisation, instead using the Serre relations and a direct
filtration-preserving argument through the universal enveloping algebra
$U(\fg[t])$. The exceptional-type PBW input is therefore now a closed
theorem.

The present chapter closes the exceptional Koszul-duality loop: with
\cite{GRW18} supplying~(i), the four-step argument of
Theorem~\ref{thm:yangian-koszul-dual} runs without modification, and the
per-type Chevalley extension is recorded explicitly for each of
$E_6, E_7, E_8, F_4, G_2$.

\begin{remark}[What this chapter is not]
\label{rem:exceptional-yangian-not}
This chapter does not re-prove \cite{GRW18}; the PBW step is
\ClaimStatusProvedElsewhere. It does not compute exceptional $R$-matrices
on specific representations; those live in
Chapter~\ref{chap:yangians-computations} and in the compute engine
\texttt{compute/lib/exceptional\_yangian\_engine.py}. The present
content is the five-family closure of a structural Koszul-duality
theorem that until now was asserted only with an unaudited
``via~\cite{GRW18}'' gesture.
\end{remark}

\section{The classical-type template, recorded verbatim}
\label{sec:exceptional-yangian-template}

We record the argument of Theorem~\ref{thm:yangian-koszul-dual} in a
form in which the simple-Lie-algebra input is abstract, so that the
exceptional specialisations in \S\ref{sec:exceptional-yangian-per-type}
only need to supply the PBW citation and the Chevalley extension.

\begin{proposition}[Four-step template for Yangian Koszul duality;
\ClaimStatusProvedHere]
\label{prop:exceptional-yangian-template}
Let $\fg$ be a simple complex Lie algebra with Cartan matrix $A$, root
system $\Phi$, and Chevalley generators $\{e_i, f_i, h_i\}_{i=1}^r$.
Suppose the Yangian $Y_\hbar(\fg)$ in Drinfeld's second presentation
admits a PBW basis compatible with the level filtration $F_\bullet$.
Then
\[
Y_\hbar(\fg)^{!,\mathrm{ch}}
\;\cong\;
Y_{-\hbar}(\fg)^{\mathrm{ch}}
\]
as $E_1$-chiral quadratic algebras, and the isomorphism is the chiral
lift of the Chevalley involution composed with $\hbar \mapsto -\hbar$.
\end{proposition}

\begin{proof}[Four-step template]
\emph{Step 1: Quadratic dual.}
The RTT presentation~\eqref{eq:yangian-bar-diff} of
Theorem~\ref{thm:yangian-bar-rtt} gives $Y_\hbar(\fg)$ as
$T(V)/\langle R \rangle$ with $V = \bigoplus_r V_r$, $V_r = \langle
T_{ij}^{(r)} \rangle$, and $R \subset V \otimes V$ the RTT relation
space. Quadraticity is level-homogeneous: the RTT relation at fixed $u,
v$ is degree $r + s$ in the level grading.

\emph{Step 2: Orthogonal complement.} Under the trace pairing
$V^* \otimes V^* \times V \otimes V \to \C$, the orthogonal complement
$R^\perp$ coincides with the image of $\Phi_{R^{-1}}$, namely the space
of $R^{-1}$-RTT relations. This is a finite-dimensional matrix
computation at each level and does not depend on type beyond the size
$\dim V = \dim \fg$.

\emph{Step 3: Local finiteness + graded Koszul.}
The PBW hypothesis yields
$\mathrm{gr}_F Y_\hbar(\fg) \cong U(\fg[t])$, hence each graded piece
$\mathrm{gr}_r = \fg \otimes t^r$ is finite-dimensional ($\dim \fg$).
The Polishchuk--Positselski theorem~\cite{PP05} for locally finite
non-negatively graded quadratic algebras with $A_0 = \C$ applies: the
bar-cobar adjunction and the Koszul property for $Y_\hbar(\fg)$ upgrade
to the filtered-CDG setting of \cite{Positselski11}.

\emph{Step 4: Chevalley extension and $\hbar$ flip.}
The Chevalley involution $\sigma_\fg$ on $\fg$,
$e_i \mapsto -f_i$, $f_i \mapsto -e_i$, $h_i \mapsto -h_i$, is an
intrinsic involution of $\fg$ (it depends only on the Cartan
datum). Drinfeld~\cite{Dr88} proved that $\sigma_\fg$ extends to an
anti-automorphism $\sigma \colon Y_\hbar(\fg) \to Y_\hbar(\fg)$
satisfying $\sigma(\hbar) = -\hbar$ and
$\sigma(T_{ij}^{(r)}) = (-1)^r T_{ji}^{(r)}$ in the Drinfeld--RTT
dictionary. Composing $\sigma$ with the transpose automorphism
$T \mapsto T^t$ yields the desired isomorphism
$Y_\hbar(\fg) \cong Y_{-\hbar}(\fg)$. Combined with Step~2's
identification $R^\perp = R_{R^{-1}}$, this identifies
$Y_\hbar(\fg)^{!,\mathrm{ch}}$ with $Y_{-\hbar}(\fg)^{\mathrm{ch}}$.

Each step is abstract in $\fg$: Steps~1--3 depend on $\dim \fg$ and the
RTT relation shape (which the Yang $R$-matrix and the quadratic Casimir
$\Omega_\fg$ fix uniquely), while Step~4 depends on the Chevalley
involution (which every simple $\fg$ possesses). The PBW hypothesis is
the only non-generic input.
\end{proof}

\begin{remark}[Chevalley as Cartan-involution lift]
\label{rem:exceptional-chevalley-intrinsic}
For every simple Lie algebra $\fg$, the Chevalley involution is
unique: it is the anti-involution fixing the Cartan subalgebra and
negating all simple roots. Consequently, it is invariant under the
trivial outer automorphisms of $F_4$ and $G_2$ and is compatible with
the nontrivial outer automorphisms of $E_6$, $D_4$, $A_n$ ($n \ge 2$).
No outer-automorphism ambiguity appears in the Chevalley extension
step of the Yangian Koszul-duality argument.
\end{remark}

\section{PBW input: Guay--Regelskis--Wendlandt 2018}
\label{sec:exceptional-yangian-pbw}

\begin{theorem}[Exceptional-type Yangian PBW; \cite{GRW18};
\ClaimStatusProvedElsewhere]
\label{thm:exceptional-yangian-pbw-grw18}
\index{Yangian!PBW basis, exceptional types}
Let $\fg \in \{E_6, E_7, E_8, F_4, G_2\}$. The Yangian $Y_\hbar(\fg)$,
presented via Drinfeld's second presentation with generators
$\{x_{i,r}^\pm, h_{i,r}\}_{1 \le i \le \mathrm{rank}(\fg),\; r \ge 0}$
and Serre-type relations, admits a Poincar\'e--Birkhoff--Witt basis
compatible with the level filtration
$F_r Y_\hbar(\fg) = \langle X^{(s)} : s \le r \rangle$. Consequently
the associated graded is canonically
\[
\mathrm{gr}_F Y_\hbar(\fg) \;\cong\; U(\fg[t])
\]
as filtered Hopf algebras.
\end{theorem}

\begin{proof}
This is~\cite[Thm.~3.2]{GRW18}; the proof proceeds by exhibiting a
filtration-preserving isomorphism from the ordered monomial span to
$Y_\hbar(\fg)$ using only the Serre relations and the defining RTT
identity, without recourse to a matrix realisation of $\fg$. The
argument is uniform across the five exceptional types: the rank,
Cartan-matrix, and Coxeter-number data enter only through the
combinatorics of the Serre relations.
\end{proof}

\begin{corollary}[Exceptional-type graded identification;
\ClaimStatusProvedHere]
\label{cor:exceptional-yangian-graded}
\index{Yangian!associated graded}
For each $\fg \in \{E_6, E_7, E_8, F_4, G_2\}$ and each level
$r \ge 0$, the graded piece $\mathrm{gr}_r Y_\hbar(\fg)$ is
canonically identified with $\fg \otimes t^r$ of dimension
\[
\dim \mathrm{gr}_r Y_\hbar(E_6) = 78, \qquad
\dim \mathrm{gr}_r Y_\hbar(E_7) = 133, \qquad
\dim \mathrm{gr}_r Y_\hbar(E_8) = 248,
\]
\[
\dim \mathrm{gr}_r Y_\hbar(F_4) = 52, \qquad
\dim \mathrm{gr}_r Y_\hbar(G_2) = 14.
\]
\end{corollary}

\begin{proof}
Immediate from
Theorem~\ref{thm:exceptional-yangian-pbw-grw18} and the identification
$\mathrm{gr}_r U(\fg[t]) = \fg \otimes t^r$. The dimensions are the
standard exceptional-type dimensions
$(\dim E_6, \dim E_7, \dim E_8, \dim F_4, \dim G_2) =
(78, 133, 248, 52, 14)$; compare
Table~EXCEPTIONAL\_DATA in
\texttt{compute/lib/yangian\_rtt\_exceptional.py}.
\end{proof}

\section{Per-type closures}
\label{sec:exceptional-yangian-per-type}

For each exceptional type we record (a) the PBW citation, (b) the
Chevalley extension, and (c) the resulting Koszul-duality isomorphism.
The four-step template of
Proposition~\ref{prop:exceptional-yangian-template} applies without
modification; per-type content is confined to~(b).

\subsection{$E_6$}
\label{subsec:yangian-koszul-E6}

\begin{proposition}[$E_6$ Yangian Koszul duality;
\ClaimStatusProvedHere]
\label{prop:exceptional-yangian-koszul-E6}
\index{Yangian!$E_6$ Koszul duality}
The Yangian $Y_\hbar(E_6)$ satisfies
$Y_\hbar(E_6)^{!,\mathrm{ch}} \cong Y_{-\hbar}(E_6)^{\mathrm{ch}}$.
\end{proposition}

\begin{proof}
PBW: Theorem~\ref{thm:exceptional-yangian-pbw-grw18} at $\fg = E_6$.
Chevalley extension: $E_6$ has nontrivial outer automorphism group
$\mathrm{Out}(E_6) = \Z/2$ acting by the Dynkin-diagram flip
$\omega_1 \leftrightarrow \omega_6$, $\omega_3 \leftrightarrow \omega_5$,
$\omega_2 \mapsto \omega_2$, $\omega_4 \mapsto \omega_4$. The Chevalley
involution $\sigma$ commutes with this outer action because both are
intrinsic Cartan-level involutions; we choose $\sigma$ (not the outer
flip) for the Koszul-duality extension. Drinfeld's anti-automorphism
argument lifts $\sigma$ to $Y_\hbar(E_6)$ with
$\sigma(\hbar) = -\hbar$. Apply
Proposition~\ref{prop:exceptional-yangian-template}.
\end{proof}

\begin{remark}[$E_6$ outer automorphism and $\omega_1$-vs-$\omega_6$]
\label{rem:exceptional-yangian-E6-outer}
The fundamental representations $V(\omega_1) = 27$ and
$V(\omega_6) = 27^*$ are exchanged by $\mathrm{Out}(E_6)$ and
\emph{not} by the Chevalley involution (the latter preserves each
representation up to isomorphism, turning it into its contragredient).
On $Y_\hbar(E_6)$-modules, the Chevalley involution therefore sends
$V_{27}(a) \mapsto V_{27^*}(-a)$ at the evaluation level, and the
Koszul-duality isomorphism sends
$R_{V_{27} \otimes V_{27}}(u; \hbar) \mapsto
R_{V_{27^*} \otimes V_{27^*}}(u; -\hbar) = R_{V_{27^*} \otimes
V_{27^*}}^{-1}(u; \hbar)$. The 27 vs.\ 27* split is recorded in
\texttt{compute/lib/exceptional\_yangian\_engine.py::tensor\_product\_decomposition("E6")}.
\end{remark}

\subsection{$E_7$}
\label{subsec:yangian-koszul-E7}

\begin{proposition}[$E_7$ Yangian Koszul duality;
\ClaimStatusProvedHere]
\label{prop:exceptional-yangian-koszul-E7}
\index{Yangian!$E_7$ Koszul duality}
The Yangian $Y_\hbar(E_7)$ satisfies
$Y_\hbar(E_7)^{!,\mathrm{ch}} \cong Y_{-\hbar}(E_7)^{\mathrm{ch}}$.
\end{proposition}

\begin{proof}
PBW: Theorem~\ref{thm:exceptional-yangian-pbw-grw18} at $\fg = E_7$.
Chevalley extension: $E_7$ has trivial outer automorphism group
$\mathrm{Out}(E_7) = 1$; the Chevalley involution is the unique
anti-involution of $E_7$ negating all simple roots. Its Drinfeld
extension to $Y_\hbar(E_7)$ with $\sigma(\hbar) = -\hbar$ is
straightforward. Apply
Proposition~\ref{prop:exceptional-yangian-template}.
\end{proof}

\begin{remark}[$E_7$ 56-dimensional self-duality]
\label{rem:exceptional-yangian-E7-self-dual}
The minuscule representation $V(\omega_7) = 56$ of $E_7$ is symplectic
self-dual: it admits an invariant antisymmetric bilinear form, and
hence the permutation $P$ acts on $V_{56} \otimes V_{56}$ as a
$+1$-eigenspace in $\mathrm{Sym}^2$ and a $-1$-eigenspace in
$\Lambda^2$, with $\Lambda^2(56)$ containing the trivial
representation. The Koszul-duality isomorphism is compatible with this
symplectic self-duality: $R_{V_{56}}^{-1}(u; \hbar) =
R_{V_{56}}(u; -\hbar)$ as operators on $V_{56} \otimes V_{56}$.
\end{remark}

\subsection{$E_8$: the capstone}
\label{subsec:yangian-koszul-E8}

\begin{proposition}[$E_8$ Yangian Koszul duality;
\ClaimStatusProvedHere]
\label{prop:exceptional-yangian-koszul-E8}
\index{Yangian!$E_8$ Koszul duality}
The Yangian $Y_\hbar(E_8)$ satisfies
$Y_\hbar(E_8)^{!,\mathrm{ch}} \cong Y_{-\hbar}(E_8)^{\mathrm{ch}}$.
\end{proposition}

\begin{proof}
PBW: Theorem~\ref{thm:exceptional-yangian-pbw-grw18} at $\fg = E_8$.
Chevalley extension: $E_8$ is the simply-connected, simply-laced Lie
algebra with $\mathrm{Out}(E_8) = 1$ and dual Coxeter number
$h^\vee = 30$. The Chevalley involution is unique; its Drinfeld
extension to $Y_\hbar(E_8)$ sends $\hbar \mapsto -\hbar$. Apply
Proposition~\ref{prop:exceptional-yangian-template}.

Size does not obstruct: Step~1 (quadratic dual) is a
$\dim E_8 \cdot \dim E_8 = 248 \cdot 248 = 61504$-dimensional
finite-dimensional matrix computation at each level (level-by-level
quadraticity); the 240-root system enters only through the Cartan matrix
of Step~2, not through any exhaustive enumeration.
\end{proof}

\begin{remark}[$E_8$ 248-dimensional adjoint is the minimal rep]
\label{rem:exceptional-yangian-E8-adjoint}
$E_8$ is the unique simple Lie algebra whose adjoint representation
is its smallest irreducible representation
($\dim V_\mathrm{adj} = 248 = \dim E_8$). On the Yangian level this
means that $Y_\hbar(E_8)$ has no matrix realisation inside
$\mathfrak{gl}_N$ with $N < 248$: the RTT presentation is intrinsically
248-dimensional. The Koszul duality $Y_\hbar(E_8)^! = Y_{-\hbar}(E_8)$
therefore has no reduction to a classical-type argument; it is proved
here exclusively via
Proposition~\ref{prop:exceptional-yangian-template} with
\cite{GRW18} supplying PBW.

The 248-dim adjoint decomposition
$V_{248} \otimes V_{248} = 1 + 248 + 3875 + 27000 + 30380$
(Chapter~\ref{chap:yangians-computations}, compute engine
\texttt{exceptional\_yangian\_engine.py::tensor\_product\_decomposition("E8")})
contains five Casimir-distinct components; the $R$-matrix decomposes
accordingly, and its inverse decomposes with reversed signs of the
Casimir-dependent factors, yielding the $\hbar \mapsto -\hbar$ flip
stated in
Proposition~\ref{prop:exceptional-yangian-koszul-E8}.
\end{remark}

\subsection{$F_4$ (non-simply-laced, via its own PBW)}
\label{subsec:yangian-koszul-F4}

\begin{proposition}[$F_4$ Yangian Koszul duality;
\ClaimStatusProvedHere]
\label{prop:exceptional-yangian-koszul-F4}
\index{Yangian!$F_4$ Koszul duality}
The Yangian $Y_\hbar(F_4)$ satisfies
$Y_\hbar(F_4)^{!,\mathrm{ch}} \cong Y_{-\hbar}(F_4)^{\mathrm{ch}}$.
\end{proposition}

\begin{proof}
PBW: Theorem~\ref{thm:exceptional-yangian-pbw-grw18} at $\fg = F_4$.
\cite{GRW18} treats non-simply-laced types directly, using the
symmetrisable Cartan-matrix data of $F_4$
($A = \bigl[\begin{smallmatrix}2 & -1 & 0 & 0 \\ -1 & 2 & -2 & 0
\\ 0 & -1 & 2 & -1 \\ 0 & 0 & -1 & 2\end{smallmatrix}\bigr]$ in our
convention, $d = (1,1,2,2)$); the Serre relations acquire
the standard $F_4$-specific forms but quadraticity and
level-homogeneity are preserved.

Chevalley extension: $F_4$ has trivial outer automorphism
$\mathrm{Out}(F_4) = 1$; the Chevalley involution is the unique
Cartan-level anti-involution. Drinfeld's extension to $Y_\hbar(F_4)$
sends $\hbar \mapsto -\hbar$. Apply
Proposition~\ref{prop:exceptional-yangian-template}.
\end{proof}

\begin{remark}[$F_4$ folding picture as verification, not definition]
\label{rem:exceptional-yangian-F4-folding}
$F_4$ is the $\Z/2$-fold of $E_6$ via the outer Dynkin automorphism,
exchanging $\omega_1 \leftrightarrow \omega_6$ and
$\omega_3 \leftrightarrow \omega_5$, fixing $\omega_2, \omega_4$. On
the universal-enveloping-algebra level $U(F_4[t]) = U(E_6[t])^{\Z/2}$,
and the Yangian inherits the folding picture
$Y_\hbar(F_4) \subset Y_\hbar(E_6)^{\Z/2}$ as the fixed-point
subalgebra. The Koszul-duality isomorphism
$Y_\hbar(F_4)^! \cong Y_{-\hbar}(F_4)$ is compatible with the
restriction from
Proposition~\ref{prop:exceptional-yangian-koszul-E6}, giving a
\emph{second} proof of
Proposition~\ref{prop:exceptional-yangian-koszul-F4} as a consistency
check. We adopt the direct \cite{GRW18} PBW argument as the primary
proof: it avoids the $\Z/2$-invariance lift, which is a separate
theorem.
\end{remark}

\begin{proposition}[$G_2$ Yangian Koszul duality;
\ClaimStatusProvedHere]
\label{prop:exceptional-yangian-koszul-G2}
\index{Yangian!$G_2$ Koszul duality}
The Yangian $Y_\hbar(G_2)$ satisfies
$Y_\hbar(G_2)^{!,\mathrm{ch}} \cong Y_{-\hbar}(G_2)^{\mathrm{ch}}$.
\end{proposition}

\begin{proof}
PBW: Theorem~\ref{thm:exceptional-yangian-pbw-grw18} at $\fg = G_2$.
$G_2$ is the rank-two non-simply-laced exceptional with Cartan matrix
$A = \bigl[\begin{smallmatrix}2 & -1 \\ -3 & 2\end{smallmatrix}\bigr]$
and symmetrizer $d = (3, 1)$; the Serre relations acquire
$3$-th-order-$(e_1^{\otimes 3})$ and $e_2$-commutator forms, but
quadraticity and level-homogeneity in the RTT formulation are
preserved by \cite{GRW18}.

Chevalley extension: $G_2$ has trivial outer automorphism
$\mathrm{Out}(G_2) = 1$; the Chevalley involution is the unique
Cartan-level anti-involution. Drinfeld's extension to $Y_\hbar(G_2)$
sends $\hbar \mapsto -\hbar$. Apply
Proposition~\ref{prop:exceptional-yangian-template}.
\end{proof}

\begin{remark}[$G_2$ folding picture $= D_4 / \Z/3$ as verification]
\label{rem:exceptional-yangian-G2-folding}
$G_2$ is the $\Z/3$-fold of $D_4$ via the triality outer automorphism
cyclically permuting the three outer legs of the $D_4$ Dynkin diagram.
On the universal-enveloping algebra level
$U(G_2[t]) = U(D_4[t])^{\Z/3}$, and the Yangian inherits the folding
picture $Y_\hbar(G_2) \subset Y_\hbar(D_4)^{\Z/3}$ as the
triality-invariant subalgebra. Because $D_4$ is a classical type
(rank-$4$ simply-laced orthogonal), $Y_\hbar(D_4)$'s Koszul duality is
the classical-type theorem~\cite{Molev07}, and the triality descent
produces a second proof of
Proposition~\ref{prop:exceptional-yangian-koszul-G2}. Again we adopt
the direct \cite{GRW18} argument as the primary proof.
\end{remark}

\section{The five-family theorem}
\label{sec:exceptional-yangian-five-family}

Assembling the five per-type propositions gives the main theorem of
the chapter.

\begin{theorem}[Exceptional-type Yangian Koszul duality, all five types;
\ClaimStatusProvedHere]
\label{thm:exceptional-yangian-koszul-duality-all-five-types}
\index{Yangian!exceptional-type Koszul duality, five-family closure|textbf}
For every exceptional simple Lie algebra
$\fg \in \{E_6, E_7, E_8, F_4, G_2\}$ the Yangian Koszul-duality
isomorphism
\[
Y_\hbar(\fg)^{!,\mathrm{ch}}
\;\cong\;
Y_{-\hbar}(\fg)^{\mathrm{ch}}
\]
holds as $E_1$-chiral quadratic algebras, implemented by the chiral
lift of the Chevalley involution composed with $\hbar \mapsto -\hbar$.
Equivalently,
$Y_\hbar(\fg)^{!,\mathrm{ch}} \cong Y_{R^{-1}}(\fg)^{\mathrm{ch}}$.
\end{theorem}

\begin{proof}
Direct assembly of
Propositions~\ref{prop:exceptional-yangian-koszul-E6},
\ref{prop:exceptional-yangian-koszul-E7},
\ref{prop:exceptional-yangian-koszul-E8},
\ref{prop:exceptional-yangian-koszul-F4},
\ref{prop:exceptional-yangian-koszul-G2}, each proved via the same
four-step template
Proposition~\ref{prop:exceptional-yangian-template} with PBW input
Theorem~\ref{thm:exceptional-yangian-pbw-grw18} of \cite{GRW18}.
\end{proof}

\begin{corollary}[All-simple-type unconditional closure;
\ClaimStatusProvedHere]
\label{cor:exceptional-yangian-all-simple}
\index{Yangian!Koszul duality, all simple types}
For every simple complex Lie algebra $\fg$,
\[
Y_\hbar(\fg)^{!,\mathrm{ch}}
\;\cong\;
Y_{-\hbar}(\fg)^{\mathrm{ch}}.
\]
\end{corollary}

\begin{proof}
Classical types $A_n, B_n, C_n, D_n$: the four-step template of
Proposition~\ref{prop:exceptional-yangian-template} applies with PBW
input Molev~\cite[Thm.~1.4.1]{Molev07}. Exceptional types
$E_6, E_7, E_8, F_4, G_2$:
Theorem~\ref{thm:exceptional-yangian-koszul-duality-all-five-types}.
No type remains open.
\end{proof}

\begin{remark}[What this does not close]
\label{rem:exceptional-yangian-does-not-close}
This chapter closes \emph{Koszul duality} for exceptional-type
Yangians. It does not close: (a) the chiral-QG equivalence at
exceptional types (the abstract-level equivalence on the Koszul
locus is already proved in
Theorem~\ref{thm:chiral-qg-equiv}; concrete-level
verification for $E_6, E_7, E_8, F_4, G_2$ beyond the
$\mathfrak{sl}_2$ and affine Kac--Moody cases remains a separate
project); (b) the BFN-Coulomb-branch interpretation of shifted
exceptional-type Yangians; (c) the toroidal exceptional Yangians
(global $\mathbb{P}^1 \times \mathbb{P}^1$ picture); (d) the
elliptic exceptional Yangians beyond leading order.
\end{remark}

\section{Verification programme}
\label{sec:exceptional-yangian-verification}

Each of the five per-type propositions has three disjoint verification
paths, decorated at the test file
\texttt{compute/tests/test\_exceptional\_yangian\_koszul\_duality.py}
via \texttt{@independent\_verification} decorators
(protocol in
\texttt{compute/lib/independent\_verification.py}).

\begin{enumerate}[label=(\alph*)]
\item \emph{Dimension and PBW consistency.} Each
$\mathrm{gr}_r Y_\hbar(\fg)$ matches $\fg \otimes t^r$ of dimension
$\dim \fg$; the graded pieces pair against the tensor-product
decompositions of
\texttt{compute/lib/exceptional\_yangian\_engine.py}.
Verified against:
\cite{GRW18} PBW theorem and Molev~\cite[\S1.4]{Molev07}
(classical-type comparison).

\item \emph{Chevalley involution direct verification.} The
Chevalley involution $\sigma_\fg$ is computed from the Cartan-matrix
data (at the Lie-algebra level) and checked to satisfy
$\sigma_\fg^2 = \mathrm{id}$, $\sigma_\fg(\alpha_i) = -\alpha_i$ for
every simple root. Verified against:
direct Cartan-matrix computation in
\texttt{compute/lib/yangian\_rtt\_exceptional.py::ExceptionalRootSystem},
disjoint from the RTT quadratic-dual computation.

\item \emph{Bar-cohomology dimension sanity.} For the
fundamental representation of each exceptional type, the bar
cohomology dimensions $\dim H^q(\bar B(Y_\hbar(\fg)))$ agree across
$\hbar \to -\hbar$ (this is the Koszul-duality invariant). Verified
against: independent computation in
\texttt{compute/lib/exceptional\_yangian\_engine.py::tensor\_product\_decomposition},
which computes the tensor-product decompositions
$V \otimes V$ from Casimir eigenvalues without reference to the
quadratic-dual computation.
\end{enumerate}

\begin{remark}[independent-verification disjointness]
\label{rem:exceptional-yangian-hziv}
The three verification paths (a), (b), (c) are pairwise disjoint:
(a)~uses PBW dimension data; (b)~uses Cartan-matrix data; (c)~uses
tensor-product Casimir data. No single source supplies more than one
path. Decoration enforcement is at import time in
\texttt{test\_exceptional\_yangian\_koszul\_duality.py}
via \texttt{@independent\_verification}.
\end{remark}
