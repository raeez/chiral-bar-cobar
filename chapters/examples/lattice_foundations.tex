\chapter{The lattice construction}
\label{ch:lattice}
% Regime I: quadratic (Convention~\ref{conv:regime-tags}).

Lattice vertex algebras are the first genuinely braided family.
The Heisenberg algebra (Chapter~\ref{ch:heisenberg-frame})
and affine Kac--Moody algebras
(Chapter~\ref{chap:kac-moody-koszul}) have degenerate braiding:
trivial or controlled entirely by the Lie bracket. The lattice
construction introduces charge geometry. An even lattice~$\Lambda$
and a $2$-cocycle $\varepsilon$ select charges, screening
operators, and a braiding depending on charge inner products.
The bar complex acquires a cocycle twist; the Koszul dual
reflects the discriminant group~$D(\Lambda)$; the genus tower detects
the lattice theta function. Spectral
$R$-matrices and line-operator tensor products of
Volume~II first appear here.

In the shadow obstruction tower of
$\gAmod$ (Definition~\ref{def:modular-convolution-dg-lie}), lattice
VOAs occupy a distinctive position. The scalar level
$\Theta^{\leq 2} = \kappa$ is nonzero with
$\kappa(V_\Lambda) = \operatorname{rank}(\Lambda)$, independent of the
cocycle~$\varepsilon$
(Theorem~\ref{thm:lattice:curvature-braiding-orthogonal}). The
tower terminates at the scalar level, as for
Heisenberg, but for a different reason: the cocycle twist ensures that
$\Definfmod(V_\Lambda)$ is \emph{formal} (curvature and braiding
decouple).
The Heisenberg tower terminates because all higher brackets vanish
identically; the lattice tower terminates because the cocycle twist is
\emph{orthogonal} to the curvature direction. This orthogonality is a
nontrivial cancellation that sharpens in the more elaborate
shadow archetypes of subsequent chapters.

\begin{remark}[Three-pillar interpretation: lattice VOAs]
\label{rem:lattice-three-pillar}
\index{lattice VOA!three-pillar interpretation}
In the three-pillar architecture
(\S\ref{sec:concordance-three-pillars}):
(i)~the \v{C}ech complex of $V_\Lambda$ carries a
$\mathrm{Ch}_\infty$-structure controlled by the cocycle twist; for
even unimodular lattices, $F_3 \neq 0$ (unlike Heisenberg): the
cocycle relations produce nontrivial homotopies at the chain level;
(ii)~despite $F_3 \neq 0$ at the $\mathrm{Ch}_\infty$ level, the
convolution $sL_\infty$-algebra
$\operatorname{hom}_\alpha(\cC^{\textup{!\textasciigrave}}, V_\Lambda)$ is
\emph{formal} by cocycle-curvature orthogonality
(Theorem~\ref{thm:lattice:curvature-braiding-orthogonal}): the
transferred $L_\infty$ brackets $\ell_k^{\mathrm{tr}}$ are all
coboundaries, so the shadow algebra
$V_\Lambda^{\mathrm{sh}}$ reduces to $\mathbb{Q}[\kappa]$ as for
Heisenberg, but for a different reason (formality of
the deformation algebra, not vanishing of higher operations);
(iii)~the log-FM geometry $\overline{\operatorname{FM}}_n(C|D)$ is
needed when twisted modules with insertions at lattice punctures are
present; the braiding data from $R$-matrices live on
$\overline{\operatorname{FM}}_2(C|\{p_1,\ldots,p_d\})$.
\end{remark}

Every example in Part~\ref{part:physics-bridges} descends from a single algebraic seed.
An even lattice $\Lambda$ with a 2-cocycle $\varepsilon$ determines
$V_\Lambda$; the choice of $\Lambda$ selects OPE
structure, Koszul dual, and genus tower.
The Heisenberg algebra is
$V_\Lambda$ for $\Lambda = \mathbb{Z}$; affine Kac--Moody at
level~$1$ is $V_{Q}$ for the root lattice $Q$; the free fermion
is a subalgebra of a rank-$1$ lattice VOA; W-algebras arise
by quantum Drinfeld--Sokolov reduction.

\begin{remark}[Four-level structure: lattice VOA]
\label{rem:lattice-four-level}
The lattice VOA illustrates the four levels of the modular engine:
(i)~the lattice vertex algebra $V_\Lambda$ from an even lattice
(geometric/physical);
(ii)~the bar complex with cocycle twist
(strict model $\Convstr$);
(iii)~the $L_\infty$-algebra $\Definfmod(V_\Lambda)$ is
\emph{formal}: curvature-braiding orthogonal by
Theorem~\ref{thm:lattice:curvature-braiding-orthogonal};
(iv)~$\kappa(V_\Lambda) = \operatorname{rank}(\Lambda)$,
independent of cocycle.
\end{remark}

\begin{definition}[Analytic theta-datum; \ClaimStatusProvedHere]%
\label{def:analytic-theta-datum}%
\index{theta-datum!analytic}%
An \emph{analytic theta-datum} on a positive-definite even
lattice~$\Lambda$ of rank~$r$ is
$\Theta^{\mathrm{an}}_\Lambda = (\kappa, \{L_\gamma\}_{\gamma \in D(\Lambda)},
c, \|\cdot\|)$,
where $\kappa = r$ is the curvature, $D(\Lambda) = \Lambda^*/\Lambda$
is the discriminant group,
each $L_\gamma = \widehat{\cF}_\gamma$ is the Hilbert completion of
the charge-$\gamma$ Fock module with inner product inherited from the
Bergman space $A^2(D)$, $c = r$ is the central charge, and
$\|\cdot\|$ is the family of HS-sewing seminorms
(Definition~\ref{def:hs-sewing}).
Realizability: Theorem~\ref{thm:lattice-sewing}.
\end{definition}

\begin{theorem}[Lattice sewing envelope;
\ClaimStatusProvedHere]\label{thm:lattice-sewing}%
% backward-compat: removed (promoted to thm:)
\index{lattice VOA!sewing envelope|textbf}%
\index{sewing envelope!lattice|textbf}%
Let\/ $\Lambda$ be a positive-definite even lattice of rank~$r$
with discriminant group $D(\Lambda) = \Lambda^*/\Lambda$.
\begin{enumerate}[label=\textup{(\roman*)}]
\item The lattice VOA\/ $V_\Lambda$ satisfies HS-sewing for
 every\/ $0 < q < 1$.
\item Every genus-$g$ sewing amplitude factors as
\begin{equation}\label{eq:lattice-sewing-factorization}
Z_g(V_\Lambda;\, \Omega)
\;=\;
\underbrace{(\det\operatorname{Im}\Omega)^{-r/2}
\cdot (\det{}'_\zeta \Delta_{\Sigma_g})^{-r/2}}_{%
 Z_g(\cH_r;\,\Omega)\text{\;
 \textup{(}Theorem~\textup{\ref{thm:heisenberg-one-particle-sewing})}}}
\;\cdot\;
\underbrace{\sum_{\lambda \in \Lambda^g}
 e^{i\pi \lambda^T \Omega \lambda}}_{%
 \Theta_\Lambda(\Omega)},
\end{equation}
and each factor converges absolutely on the Siegel upper
half-space $\mathfrak{H}_g$.
\item The sewing envelope is the charge-refined Hilbert completion
\begin{equation}\label{eq:lattice-sewing-envelope}
V_\Lambda^{\mathrm{sew}}
\;=\;
\widehat{\bigoplus}_{\gamma \in D(\Lambda)}
\operatorname{Sym}\, A^2_\gamma(D),
\end{equation}
where $A^2_\gamma(D) \subset A^2(D)$ is the Bergman space with
$L_0$-shift $\tfrac{1}{2}\langle\gamma,\gamma\rangle$ and the
completion is taken in the HS-sewing topology.
\item The vertex operators
$e^\gamma \colon \widehat{\cF}_{\gamma'} \to
\widehat{\cF}_{\gamma'+\gamma}$
extend to bounded operators between completed sectors.
\item The analytic theta-datum
\textup{(}Definition~\textup{\ref{def:analytic-theta-datum}}\textup{)}
is realized, with $\kappa = r$
\textup{(}Theorem~\textup{\ref{thm:lattice:curvature-braiding-orthogonal}}\textup{)}.
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{(i).}
Sector growth satisfies
$\dim(V_\Lambda)_n = |D(\Lambda)| \cdot p(n)^r$, which is
subexponential: $\log\dim(V_\Lambda)_n = O(\sqrt{n})$.
OPE structure constants satisfy $|C^{c,k}_{a,i;\,b,j}| \leq K(a{+}b{+}c{+}1)^r$
because all OPE singularities in~$V_\Lambda$ are at most
$r$-th order poles (the lattice vertex operators are exponentials of
the rank-$r$ free boson).
Theorem~\ref{thm:general-hs-sewing} then gives HS-sewing
for every $0 < q < 1$.

\smallskip\noindent\emph{(ii).}
The Fock decomposition
$V_\Lambda = \bigoplus_{\gamma \in D(\Lambda)} \cF_\gamma$
is a direct sum of twisted Heisenberg modules. Within each sector,
Wick's theorem reduces all correlation functions to propagator
pairings on the underlying rank-$r$ Heisenberg, contributing
$Z_g(\cH_r;\,\Omega)$. The lattice sum over winding
modes contributes $\Theta_\Lambda(\Omega)$.
The Heisenberg factor converges by
Theorem~\ref{thm:heisenberg-one-particle-sewing}
(one-particle sewing: eigenvalues~$q^n$, trace class).
The theta factor converges absolutely on $\mathfrak{H}_g$ by
$|\Theta_\Lambda(\Omega)| \leq
\sum_{\lambda} e^{-\pi \lambda^T Y \lambda}$
with $Y = \operatorname{Im}(\Omega) > 0$, and the exponent
$\lambda^T Y \lambda \geq
\lambda_{\min}(Y) \cdot \|\lambda\|^2$
grows as $\|\lambda\|^2 \to \infty$.

\smallskip\noindent\emph{(iii)--(iv).}
The Bergman-space completion
$\widehat{\cF}_\gamma = \overline{\cF_\gamma}^{\|\cdot\|_{A^2}}$
gives honest Hilbert sectors. The vertex operator
$e^\gamma$ acts on mode expansions via
$e^\gamma \cdot a_{-n_1} \cdots a_{-n_k} |{\gamma'}\rangle
= \varepsilon(\gamma,\gamma') \cdot
a_{-n_1} \cdots a_{-n_k} |{\gamma+\gamma'}\rangle$,
where $\varepsilon$ is the cocycle (bounded by~$1$) and the
mode operators are unchanged. The $L_0$-shift
$\Delta_\gamma = \tfrac{1}{2}\langle\gamma,\gamma\rangle$
is absorbed into the grading; the exponential weight damping
$q^{\Delta_\gamma}$ in the HS topology gives
$\|e^\gamma\|_{\mathrm{op}} \leq q^{\langle\gamma,\gamma\rangle/2}$.

\smallskip\noindent\emph{(v).}
Collecting: $\kappa = r$ by
Theorem~\ref{thm:lattice:curvature-braiding-orthogonal};
$L_\gamma = \widehat{\cF}_\gamma$ are the Hilbert sectors
of~(iii); multiplication boundedness is~(iv); HS collar-sewing
is~(i).
\end{proof}

\begin{remark}[Convolution-algebraic interpretation]%
\label{rem:lattice-sewing-convolution}%
\index{sewing envelope!convolution interpretation}%
The sewing envelope $V_\Lambda^{\mathrm{sew}}$ is the domain
of convergence of the universal MC element. By
Theorem~\ref{thm:mc2-bar-intrinsic},
$\Theta_{V_\Lambda} = D_{V_\Lambda} - d_0 \in
\MC(\mathfrak{g}^{\mathrm{mod}}_{V_\Lambda})$; the bar-intrinsic construction gives
an abstract element in the convolution algebra. The HS-sewing
condition is the \emph{analytic criterion} for the formal
genus expansion
\[
\Theta_{V_\Lambda}
\;=\;
\sum_{g \geq 0} \hbar^g \,\Theta_{V_\Lambda}^{(g)}
\]
to converge in the HS topology on the completed Hilbert
sectors~$\widehat{\cF}_\gamma$. The factorization~\eqref{eq:lattice-sewing-factorization}
is the \emph{computation} of this convergent element:
the Heisenberg Fredholm determinant is the Gaussian
(free-field) summand, and the Siegel theta function is the
lattice winding correction. In the shadow obstruction tower language,
$V_\Lambda$ has depth~$r_{\max} = 2$ (Gaussian archetype) and
$\Theta_{V_\Lambda} = \kappa(V_\Lambda) \cdot \eta \otimes \Lambda$
for single-generator algebras of uniform conformal weight (Theorem~D, Remark~\ref{rem:propagator-weight-universality}).
\end{remark}

\section{Even lattices and root systems}%
\label{sec:lattice:lattices}%

\begin{definition}[Lattice]\label{def:lattice:lattice}
\index{lattice!even unimodular}
A \emph{lattice} is a free abelian group $\Lambda$ of finite rank
equipped with a symmetric bilinear form
$\langle \cdot, \cdot \rangle \colon \Lambda \times \Lambda \to \Z$.
The lattice is:
\begin{enumerate}[label=(\roman*)]
\item \emph{integral} if $\langle \alpha, \beta \rangle \in \Z$ for all
 $\alpha, \beta \in \Lambda$;
\item \emph{even} if $\langle \alpha, \alpha \rangle \in 2\Z$ for all
 $\alpha \in \Lambda$ (this implies integrality);
\item \emph{positive-definite} if $\langle \alpha, \alpha \rangle > 0$
 for all $\alpha \neq 0$;
\item \emph{non-degenerate} if the map
 $\Lambda \to \Lambda^* := \Hom(\Lambda, \Z)$ given by
 $\alpha \mapsto \langle \alpha, \cdot \rangle$ is injective;
\item \emph{unimodular} if this map is an isomorphism, i.e.,
 $\Lambda = \Lambda^*$.
\end{enumerate}
\end{definition}

\begin{definition}[Discriminant group]\label{def:lattice:discriminant}
For a non-degenerate lattice $\Lambda$, the \emph{dual lattice} is
$\Lambda^* = \{\lambda \in \Lambda \otimes_\Z \bQ :
\langle \lambda, \alpha \rangle \in \Z \ \forall\, \alpha \in \Lambda\}$.
The \emph{discriminant group} is the finite abelian group
$D(\Lambda) = \Lambda^*/\Lambda$,
whose order equals $|\det(\langle e_i, e_j \rangle)|$ for any basis
$\{e_i\}$ of $\Lambda$. The bilinear form on $\Lambda$ induces a
$\bQ/\Z$-valued bilinear form on $D(\Lambda)$.
\end{definition}

\begin{example}[Root lattices]\label{ex:lattice:root-lattices}
The ADE root lattices provide the fundamental examples.
\begin{center}
\begin{tabular}{l c c c c}
\textbf{Lattice} & \textbf{Rank} & $\det$ & $h^\vee$ & \textbf{Unimodular?} \\
\hline
$A_n$ & $n$ & $n+1$ & $n+1$ & No \\
$D_n$ & $n$ & $4$ & $2n-2$ & No \\
$E_6$ & $6$ & $3$ & $12$ & No \\
$E_7$ & $7$ & $2$ & $18$ & No \\
$E_8$ & $8$ & $1$ & $30$ & Yes
\end{tabular}
\end{center}
For each simply-laced root system $\Phi$, the root lattice
$\Lambda_\Phi = \Z\Phi$ is even and positive-definite.
The Cartan matrix $A_{ij} = \langle \alpha_i, \alpha_j \rangle$ encodes
the bilinear form on simple roots.
\end{example}

\begin{remark}[Uniqueness of \texorpdfstring{$E_8$}{E8}]\label{rem:lattice:e8-unique}
By Mordell's theorem (1938), $E_8$ is the unique even unimodular
positive-definite lattice of rank~8. In rank~16, there are exactly two
such lattices: $E_8 \oplus E_8$ and $D_{16}^+$ (the latter being the
overlattice of $D_{16}$ by its spinor class). This classification is
the lattice-theoretic origin of the two heterotic string theories.
\end{remark}

\begin{definition}[Lattice operations]\label{def:lattice:operations}
The following operations become functorial on vertex algebras
in Section~\ref{sec:lattice:functorial}.
\begin{enumerate}[label=(\roman*)]
\item \emph{Direct sum.}
 For lattices $\Lambda_1, \Lambda_2$, the orthogonal direct sum
 $\Lambda_1 \oplus \Lambda_2$ has
 $\langle (\alpha_1, \alpha_2), (\beta_1, \beta_2) \rangle =
 \langle \alpha_1, \beta_1 \rangle + \langle \alpha_2, \beta_2 \rangle$.
\item \emph{Sublattice.}
 A sublattice $\Lambda' \hookrightarrow \Lambda$ of finite index
 gives rise to $D(\Lambda') \twoheadrightarrow D(\Lambda)$.
\item \emph{Overlattice (gluing).}
 For an isotropic subgroup $I \subset D(\Lambda)$ (meaning
 $\langle \gamma, \gamma' \rangle \equiv 0 \bmod \Z$ for all
 $\gamma, \gamma' \in I$), the preimage
 $\Lambda_I = \{\lambda \in \Lambda^* : [\lambda] \in I\}$ is a
 lattice with $\Lambda \subset \Lambda_I \subset \Lambda^*$.
\item \emph{Orbifold.}
 For $\sigma \in \operatorname{Aut}(\Lambda)$, the fixed sublattice
 $\Lambda^\sigma$ and the coinvariants $\Lambda/(1-\sigma)\Lambda$
 both carry induced bilinear forms.
\end{enumerate}
\end{definition}

\section{Cocycle theory}
\label{sec:lattice:cocycles}

The vertex algebra structure on $V_\Lambda$ requires a choice of
2-cocycle on $\Lambda$. Different cocycles produce genuinely different
algebraic structures, ranging from honest vertex algebras ($\Einf$-chiral)
to nonlocal $\Eone$-chiral algebras. The cocycle
theory controls this distinction.

\begin{definition}[2-cocycle on a lattice]\label{def:lattice:2-cocycle}
\index{cocycle!lattice}
A \emph{2-cocycle} on a lattice $\Lambda$ is a function
$\varepsilon \colon \Lambda \times \Lambda \to \C^\times$ satisfying:
\begin{equation}\label{eq:lattice:cocycle-condition}
\varepsilon(\alpha, \beta)\,\varepsilon(\alpha + \beta, \gamma)
= \varepsilon(\alpha, \beta + \gamma)\,\varepsilon(\beta, \gamma)
\end{equation}
for all $\alpha, \beta, \gamma \in \Lambda$.
The cocycle is:
\begin{enumerate}[label=(\roman*)]
\item \emph{normalized} if
 $\varepsilon(\alpha, 0) = \varepsilon(0, \alpha) = 1$ for all $\alpha$;
\item \emph{bimultiplicative} if
 $\varepsilon(\alpha + \alpha', \beta) =
 \varepsilon(\alpha, \beta)\,\varepsilon(\alpha', \beta)$
 and similarly in the second argument;
\item \emph{symmetric} (in the sense of Borcherds) if
\begin{equation}\label{eq:lattice:borcherds-symmetry}
\varepsilon(\alpha, \beta)
= (-1)^{\langle \alpha, \beta \rangle}\,\varepsilon(\beta, \alpha)
\end{equation}
for all $\alpha, \beta \in \Lambda$.
\end{enumerate}
\end{definition}

\begin{remark}\label{rem:lattice:symmetry-convention}
The symmetry condition~\eqref{eq:lattice:borcherds-symmetry} is the
\emph{Borcherds axiom}: it is precisely the condition needed for the
vertex operator skew-symmetry $Y(a,z)b = e^{z\partial}Y(b,-z)a$ to hold.
Note that for an even lattice, $(-1)^{\langle\alpha,\beta\rangle}$
is well-defined. For an odd lattice, one would need
$(-1)^{\langle\alpha,\alpha\rangle\langle\beta,\beta\rangle/4}$
corrections; we restrict to even lattices throughout.
\end{remark}

\begin{definition}[Commutator map]\label{def:lattice:commutator}
The \emph{commutator} of a cocycle $\varepsilon$ is
\[
c_\varepsilon(\alpha, \beta)
:= \frac{\varepsilon(\alpha, \beta)}{\varepsilon(\beta, \alpha)}.
\]
\end{definition}

\begin{lemma}[Cocycle classification; \ClaimStatusProvedHere]\label{lem:lattice:cocycle-class}
Let $\Lambda$ be an even lattice. Then:
\begin{enumerate}[label=(\roman*)]
\item For any normalized bimultiplicative 2-cocycle $\varepsilon$, the
 commutator satisfies
 $c_\varepsilon(\alpha, \beta) = (-1)^{\langle\alpha,\beta\rangle}$
 if and only if $\varepsilon$ is symmetric.
\item The set of symmetric cocycles on $\Lambda$ is a torsor for
 $\Hom(\Lambda \otimes \Lambda, \C^\times)^{\text{sym}}$, the group of
 symmetric bicharacters.
\item The cohomology group $H^2(\Lambda, \C^\times)$ classifies
 cocycles up to coboundary equivalence. Two cocycles
 $\varepsilon, \varepsilon'$ are cohomologous if there exists
 $f \colon \Lambda \to \C^\times$ with
 $\varepsilon'(\alpha,\beta) =
 f(\alpha)\,f(\beta)\,f(\alpha+\beta)^{-1}\,\varepsilon(\alpha,\beta)$.
\end{enumerate}
\end{lemma}

\begin{proof}
(i) Computing from the Borcherds symmetry:
$c_\varepsilon(\alpha,\beta) =
\varepsilon(\alpha,\beta)/\varepsilon(\beta,\alpha) =
(-1)^{\langle\alpha,\beta\rangle}$.
Conversely, if $c_\varepsilon(\alpha,\beta) =
(-1)^{\langle\alpha,\beta\rangle}$, then
$\varepsilon(\alpha,\beta) =
(-1)^{\langle\alpha,\beta\rangle}\varepsilon(\beta,\alpha)$
is the Borcherds axiom.

(ii) If $\varepsilon, \varepsilon'$ are both symmetric, then
$\varepsilon'/\varepsilon$ is a symmetric bicharacter. Conversely,
multiplying a symmetric cocycle by a symmetric bicharacter preserves
the Borcherds axiom.

(iii) Standard group cohomology: $\Lambda \cong \Z^d$ is free abelian,
so $H^2(\Lambda, \C^\times) \cong (\C^\times)^{\binom{d}{2}}$,
parametrized by the values $\varepsilon(e_i, e_j)$ for $i < j$ on a
basis $\{e_1, \ldots, e_d\}$.
\end{proof}

\begin{construction}[Standard symmetric cocycle]\label{constr:lattice:standard-cocycle}
For an even lattice $\Lambda$ with ordered basis $\{e_1, \ldots, e_d\}$,
the \emph{standard symmetric cocycle} is:
\begin{equation}\label{eq:lattice:standard-cocycle}
\varepsilon_0(\alpha, \beta)
= (-1)^{\sum_{i < j} \alpha_i \beta_j \, G_{ij}}
\end{equation}
where $\alpha = \sum_i \alpha_i e_i$, $\beta = \sum_j \beta_j e_j$,
and $G_{ij} = \langle e_i, e_j \rangle$.
One verifies:
\[
\varepsilon_0(\alpha,\beta)\,\varepsilon_0(\beta,\alpha)
= (-1)^{\sum_{i<j} G_{ij}(\alpha_i\beta_j + \beta_i\alpha_j)}
= (-1)^{\sum_{i \neq j} \alpha_i \beta_j \, G_{ij}},
\]
where the second equality uses $G_{ij} = G_{ji}$. Since $\langle\alpha,\beta\rangle = \sum_{i,j} \alpha_i G_{ij}\beta_j = \sum_{i \neq j}\alpha_i G_{ij}\beta_j + \sum_i \alpha_i\beta_i G_{ii}$ and $G_{ii} \in 2\mathbb{Z}$ for an even lattice, we have $\sum_{i \neq j}\alpha_i G_{ij}\beta_j \equiv \langle\alpha,\beta\rangle \pmod{2}$, giving $(-1)^{\langle\alpha,\beta\rangle}$ as required.
\end{construction}

\begin{example}[Non-symmetric cocycle on \texorpdfstring{$\Z^2$}{Z2}]\label{ex:lattice:nonsym-z2}
On $\Lambda = \Z^2$ with standard bilinear form
$\langle (a,b), (c,d) \rangle = ac + bd$, define:
\[
\varepsilon((a,b), (c,d)) = e^{2\pi i \theta \cdot ad}
\]
for $\theta \in \R \setminus \bQ$.
The cocycle condition holds since
$\varepsilon((a,b),(c,d))\,\varepsilon((a+c,b+d),(e,f))
= e^{2\pi i\theta(ad + (a+c)f)}$
equals
$\varepsilon((a,b),(c+e,d+f))\,\varepsilon((c,d),(e,f))
= e^{2\pi i\theta(a(d+f) + cf)}$.

For the symmetry check:
\[
\frac{\varepsilon((a,b),(c,d))}{\varepsilon((c,d),(a,b))}
= e^{2\pi i\theta(ad - cb)},
\]
which is not $(-1)^{ac+bd}$ for irrational $\theta$. Hence this cocycle
is not symmetric, producing a strictly $\Eone$-chiral algebra
(see Theorem~\ref{thm:lattice:e1-vs-einf}).
\end{example}

\begin{remark}[Non-symmetric deformations via roots of unity]
\label{rem:lattice:root-of-unity-deformation}
More generally, given a symmetric cocycle $\varepsilon_0$ and an
antisymmetric bilinear form $q \colon \Lambda \times \Lambda \to \Z/N\Z$,
the product
$\varepsilon(\alpha, \beta) = \varepsilon_0(\alpha, \beta) \cdot
\zeta^{q(\alpha,\beta)}$
(where $\zeta = e^{2\pi i/N}$) is a non-symmetric cocycle whenever $q$
is not identically zero. Such deformations interpolate between the
commutative ($\Einf$) and noncommutative ($\Eone$) worlds.
\end{remark}

\section{Lattice vertex algebra construction}
\label{sec:lattice:construction}

\begin{construction}[Lattice vertex algebra]\label{constr:lattice:va}
\index{lattice VOA|textbf}
Let $\Lambda$ be an even positive-definite lattice of rank~$d$ with
bilinear form $\langle \cdot, \cdot \rangle$, and let
$\varepsilon \colon \Lambda \times \Lambda \to \C^\times$ be a
normalized 2-cocycle. The \emph{lattice vertex algebra}
$\Vlat_\Lambda^\varepsilon$ is defined as follows.

\noindent\emph{State space.}
Let $\cH = \C[\alpha^i_{-n} : 1 \leq i \leq d,\; n \geq 1]$ be the
Heisenberg Fock space (bosonic oscillator algebra) for
$\Lambda \otimes_\Z \C$, with vacuum $|0\rangle$ annihilated by all
$\alpha^i_n$ for $n > 0$. Let $\C[\Lambda]$ denote the group algebra
with basis $\{e^\alpha : \alpha \in \Lambda\}$ and multiplication
$e^\alpha \cdot e^\beta = \varepsilon(\alpha,\beta)\, e^{\alpha+\beta}$.
The state space is:
\begin{equation}\label{eq:lattice:state-space}
\Vlat_\Lambda^\varepsilon
= \cH \otimes_\C \C[\Lambda]
= \bigoplus_{\alpha \in \Lambda} \cH \otimes e^\alpha.
\end{equation}
This is $\Lambda$-graded by declaring
$\deg(\cH \otimes e^\alpha) = \alpha$.

\noindent\emph{Vertex operators.}
For $\alpha \in \Lambda$, the vertex operator is:
\begin{equation}\label{eq:lattice:vertex-op}
Y(e^\alpha, z) = E^-(\alpha, z)\, E^+(\alpha, z)\, e^\alpha\, z^{\alpha_0}
\end{equation}
where:
\begin{align*}
E^-(\alpha, z)
&= \exp\!\Bigl(-\sum_{n < 0} \frac{\alpha_n}{n}\, z^{-n}\Bigr)
 \quad \text{(creation part)}, \\
E^+(\alpha, z)
&= \exp\!\Bigl(-\sum_{n > 0} \frac{\alpha_n}{n}\, z^{-n}\Bigr)
 \quad \text{(annihilation part)},
\end{align*}
the operator $e^\alpha$ acts on $\C[\Lambda]$ by
$e^\alpha \cdot e^\beta = \varepsilon(\alpha,\beta)\,e^{\alpha+\beta}$,
and $z^{\alpha_0}$ is the formal variable tracking the zero mode.

\noindent\emph{OPE.}
\begin{equation}\label{eq:lattice:ope}
Y(e^\alpha, z)\, Y(e^\beta, w)
= \varepsilon(\alpha, \beta)\, (z-w)^{\langle\alpha,\beta\rangle}\,
 Y(e^{\alpha+\beta}, w) + \text{regular terms}.
\end{equation}

\noindent\emph{Conformal vector.}
Choose an orthonormal basis $\{h_i\}$ of $\Lambda \otimes \C$.
The conformal vector is
$\omega = \frac{1}{2}\sum_{i=1}^d h_{i,{-1}}^2 |0\rangle$,
giving rise to the Virasoro algebra with central charge $c = d = \operatorname{rank}(\Lambda)$.
The conformal weight of $e^\alpha$ is:
\begin{equation}\label{eq:lattice:conformal-weight}
h_{e^\alpha} = \frac{\langle\alpha,\alpha\rangle}{2}.
\end{equation}
\end{construction}

\begin{theorem}[\texorpdfstring{$\Eone$}{E1} vs.\ \texorpdfstring{$\Einf$}{E-infinity} classification; \ClaimStatusProvedHere]
\label{thm:lattice:e1-vs-einf}
The lattice vertex algebra $\Vlat_\Lambda^\varepsilon$ is:
\begin{enumerate}[label=(\roman*)]
\item an $\Einf$-chiral algebra (a vertex algebra in the sense of
 Borcherds) if and only if $\varepsilon$ is symmetric;
\item an $\Eone$-chiral algebra for any cocycle $\varepsilon$;
\item strictly $\Eone$ (not $\Einf$) when $\varepsilon$ is not symmetric.
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{(ii)} The cocycle condition~\eqref{eq:lattice:cocycle-condition}
on $\varepsilon$ ensures associativity of the vertex operator products
$Y(Y(e^\alpha,z-w)\,e^\beta, w) = Y(e^\alpha, z)\,Y(e^\beta, w)$.
Hence $\Vlat_\Lambda^\varepsilon$ is always at least $\Eone$-chiral.

\emph{(i)} and \emph{(iii).} The skew-symmetry axiom for vertex
algebras requires
$Y(e^\alpha, z)\, e^\beta = e^{z\partial} Y(e^\beta, -z)\, e^\alpha$.
Computing both sides:
\begin{align*}
\text{LHS}
&= \varepsilon(\alpha,\beta)\, z^{\langle\alpha,\beta\rangle}\,
 e^{\alpha+\beta} + \cdots, \\
\text{RHS}
&= \varepsilon(\beta,\alpha)\, (-z)^{\langle\beta,\alpha\rangle}\,
 e^{\alpha+\beta} + \cdots
= \varepsilon(\beta,\alpha)\,(-1)^{\langle\alpha,\beta\rangle}\,
 z^{\langle\alpha,\beta\rangle}\, e^{\alpha+\beta} + \cdots.
\end{align*}
Equality of the leading terms requires
$\varepsilon(\alpha,\beta) =
(-1)^{\langle\alpha,\beta\rangle}\,\varepsilon(\beta,\alpha)$,
which is the Borcherds symmetry condition~\eqref{eq:lattice:borcherds-symmetry}.
Hence skew-symmetry holds if and only if $\varepsilon$ is symmetric.
When $\varepsilon$ is not symmetric, the OPE exchange picks up a phase
$c_\varepsilon(\alpha,\beta)/(-1)^{\langle\alpha,\beta\rangle} \neq 1$,
so the algebra is strictly $\Eone$.
\end{proof}

\begin{example}[Heisenberg as a lattice vertex algebra]
\label{ex:lattice:heisenberg}
The Heisenberg vertex algebra $\cH_\kappa$ at level $\kappa$ is the
$\alpha = 0$ sector of $\Vlat_\Lambda^\varepsilon$ for
\emph{any} lattice $\Lambda$:
\[
\cH_d := \Vlat_0 = \cH \otimes e^0 \cong \C[\alpha^i_{-n} : i, n \geq 1],
\]
with $d = \operatorname{rank}(\Lambda)$ free bosons. The cocycle
$\varepsilon$ does not enter the $\alpha = 0$ sector. This is the
simplest output of the lattice engine, and the base case for all
bar complex computations.
\end{example}

\begin{remark}[Lattice specialization to the frame example]\label{rem:lattice-heisenberg-frame}
The case $\Lambda = \mathbb{Z}$ (with standard form $\langle 1, 1 \rangle = 2$, giving an even rank-$1$ lattice) recovers the Heisenberg algebra $\mathcal{H}_k$ of Chapter~\ref{ch:heisenberg-frame}; the bar complex computed there (\S\ref{sec:frame-bar-all}) is the $\Lambda = \mathbb{Z}$ specialization of the lattice bar complex developed in \S\ref{sec:lattice:bar}.
\end{remark}

\section{The Frenkel--Kac--Segal theorem}
\label{sec:lattice:frenkel-kac}

\begin{theorem}[Frenkel--Kac--Segal; \ClaimStatusProvedElsewhere{} \cite{FK80,Se81}]\label{thm:lattice:frenkel-kac}
\index{Frenkel--Kac construction|textbf}
Let $\fg$ be a simply-laced simple Lie algebra with root lattice
$\Lambda_\fg$. Let $\varepsilon$ be a symmetric cocycle on $\Lambda_\fg$
satisfying $\varepsilon(\alpha,\beta)\,\varepsilon(\beta,\alpha) =
(-1)^{\langle\alpha,\beta\rangle}$. Then there is an isomorphism of
vertex algebras:
\begin{equation}\label{eq:lattice:frenkel-kac}
\Vlat_{\Lambda_\fg}^\varepsilon
\;\cong\; L(\AffKM{g}, 1)
\end{equation}
where $L(\AffKM{g}, 1)$ denotes the simple quotient of the level-1
vacuum module for the affine Kac--Moody algebra $\AffKM{g}$.
\end{theorem}

\begin{proof}
The proof proceeds in three steps (\cite{FK80}; \cite{Se81};
see also \cite{Kac}~Ch.~5).

\emph{Step~1.}
The Heisenberg subalgebra $\cH \subset \Vlat_{\Lambda_\fg}$ provides
the Cartan subalgebra $\AffKM{h} \hookrightarrow \AffKM{g}$.
For each root $\alpha \in \Phi$, define the root current:
\[
E_\alpha(z) := Y(e^\alpha, z), \quad
H_i(z) := Y(\alpha_{i,-1}|0\rangle, z)
\]
for simple roots $\alpha_i$. The cocycle $\varepsilon$ ensures that
the vertex operators $E_{\alpha_i}$ satisfy the Chevalley--Serre
relations: $\varepsilon(\alpha,\beta)\,\varepsilon(\beta,\alpha) =
(-1)^{\langle\alpha,\beta\rangle}$ produces the correct signs in the
$[E_\alpha, E_\beta]$ commutation relations.

\emph{Step~2.}
The lattice OPE~\eqref{eq:lattice:ope} gives, for roots $\alpha, \beta$:
\[
E_\alpha(z)\, E_\beta(w) = \varepsilon(\alpha,\beta)\,
(z-w)^{\langle\alpha,\beta\rangle}\, E_{\alpha+\beta}(w)
+ \text{regular}.
\]
When $\beta = -\alpha$ ($\langle\alpha,-\alpha\rangle = -2$):
\[
E_\alpha(z)\, E_{-\alpha}(w)
= \frac{\varepsilon(\alpha,-\alpha)}{(z-w)^2} + \frac{H_\alpha(w)}{z-w}
+ \text{regular},
\]
reproducing the $\AffKM{g}$ current algebra at level $k =
\langle\alpha,\alpha\rangle/2 = 1$ (simply-laced). When
$\langle\alpha,\beta\rangle = -1$ (adjacent roots):
$E_\alpha(z) E_\beta(w) = \varepsilon(\alpha,\beta)(z-w)^{-1}
E_{\alpha+\beta}(w) + \cdots$, reproducing $[E_\alpha, E_\beta] =
\varepsilon(\alpha,\beta) E_{\alpha+\beta}$. When
$\langle\alpha,\beta\rangle \geq 0$ and $\alpha+\beta \notin \Phi$:
the OPE is regular, giving $[E_\alpha, E_\beta] = 0$.

\emph{Step~3.}
Steps~1--2 produce an injective vertex algebra homomorphism
$L(\AffKM{g}, 1) \hookrightarrow \Vlat_{\Lambda_\fg}$. Both sides
have the same graded character: the level-$1$ Weyl--Kac character
formula gives $\mathrm{ch}\, L(\AffKM{g}, 1) =
\Theta_{\Lambda_\fg}(\tau) / \eta(\tau)^{\mathrm{rank}(\fg)}$, which
equals the lattice vertex algebra character $\mathrm{ch}\,
\Vlat_{\Lambda_\fg}$. Since an injective map between graded
vector spaces of the same graded dimension is an isomorphism,
$L(\AffKM{g}, 1) \cong \Vlat_{\Lambda_\fg}$.
\end{proof}

\begin{example}[Explicit verification for \texorpdfstring{$\AffKM{sl}_2$}{sl2-hat} at level 1]
\label{ex:lattice:sl2-fks}
The $A_1$ root lattice is $\Lambda = \Z\alpha$ with
$\langle\alpha,\alpha\rangle = 2$.
The lattice vertex algebra $\Vlat_\Lambda$ has generators:
\begin{itemize}
\item $h(z) := Y(\alpha_{-1}|0\rangle, z)$, the Cartan current,
\item $e(z) := Y(e^\alpha, z)$, the raising current,
\item $f(z) := Y(e^{-\alpha}, z)$, the lowering current.
\end{itemize}
The OPE from~\eqref{eq:lattice:ope}:
\begin{alignat*}{2}
h(z)\,h(w) &= \frac{2}{(z-w)^2} + \text{reg}
 &\qquad &\text{(level $k = 1$, $\langle\alpha,\alpha\rangle = 2$)}, \\
e(z)\,f(w)
&= \frac{\varepsilon(\alpha,-\alpha)}{(z-w)^2}
 + \frac{h(w)}{z-w} + \text{reg}
 &\qquad &\text{(simple pole gives $h$)}, \\
h(z)\,e(w) &= \frac{2\,e(w)}{z-w} + \text{reg}
 &\qquad &\text{(root $\alpha$ has $\langle\alpha,\alpha\rangle=2$)}.
\end{alignat*}
With the normalization $\varepsilon(\alpha,-\alpha) = 1$ (which is
achievable by coboundary adjustment), this reproduces the $\AffKM{sl}_2$
OPE at level $k=1$. The Sugawara construction gives
$T = \frac{1}{2(k+h^\vee)}(\frac{1}{2}\normord{h\,h} +
\normord{e\,f} + \normord{f\,e})$
with $h^\vee = 2$, central charge $c = 3 \cdot 1/(1+2) = 1$,
matching $\operatorname{rank}(\Lambda) = 1$.
\end{example}

\begin{remark}[Non-simply-laced algebras]\label{rem:lattice:non-simply-laced}
For non-simply-laced $\fg$ (types $B_n$, $C_n$, $F_4$, $G_2$),
the Frenkel--Kac--Segal construction does not apply directly, because
the root lattice contains roots of different lengths with
$\langle\alpha,\alpha\rangle \neq 2$. Instead, one obtains
non-simply-laced affine algebras via \emph{orbifold} constructions:
start with a simply-laced lattice $\Lambda$ and quotient by a
diagram automorphism $\sigma$. For instance:
\begin{itemize}
\item $\widehat{\mathfrak{so}}_{2n+1}$ (type $B_n$) at level 1 arises from $D_{n+1}$ modulo its
 order-2 diagram automorphism (folding $D_{n+1}$ gives the $B_n$ root system, not $C_n$).
\item $\widehat{G}_2$ at level 1 arises from $D_4$ modulo its order-3
 triality automorphism.
\end{itemize}
See Section~\ref{sec:lattice:functorial} for the orbifold construction.
\end{remark}

\section{Bar complexes of lattice vertex algebras}
\label{sec:lattice:bar}

The lattice grading decomposes the bar complex into orthogonal summands indexed by $\Lambda$, since the bar differential preserves the total lattice degree.

\begin{construction}[Bar complex of \texorpdfstring{$\Vlat_\Lambda^\varepsilon$}{V\_Lambda}]
\label{constr:lattice:bar-complex}
The geometric bar complex $\barBgeom(\Vlat_\Lambda^\varepsilon)$ has:

\noindent\emph{Generators.} Elements of the form
\[
[e^{\alpha_1} \mid e^{\alpha_2} \mid \cdots \mid e^{\alpha_p}]
\otimes \omega
\]
where $\alpha_i \in \Lambda \setminus \{0\}$ and $\omega$ is a
logarithmic form on $\overline{C}_p(X)$. (We suppress Heisenberg
descendants $\alpha_{-n}^i |0\rangle$ for notational clarity; they
contribute additional tensor factors.)

\noindent\emph{Grading.} The total lattice degree is $\sum_{i=1}^p \alpha_i \in \Lambda$, and the bar degree is $p$.

\noindent\emph{Differential.}
The bar differential $d = d_{\text{res}}$ sums residues over collision
divisors. For a degree-2 element $[e^\alpha \mid e^\beta] \otimes 1$:
\begin{equation}\label{eq:lattice:bar-diff}
d([e^\alpha \mid e^\beta] \otimes 1)
= \varepsilon(\alpha,\beta)\,[e^{\alpha+\beta}] \otimes 1
\end{equation}
when $\langle\alpha,\beta\rangle = -1$ (simple pole in the OPE), and
$d = 0$ when $\langle\alpha,\beta\rangle \geq 0$ (no pole) or
$\langle\alpha,\beta\rangle \leq -2$ (higher poles pair trivially
with $d\log$).
\end{construction}

\begin{theorem}[Lattice bar complex structure; \ClaimStatusProvedHere]
\label{thm:lattice:bar-structure}
\index{bar complex!lattice VOA}
The bar complex $\barBgeom(\Vlat_\Lambda^\varepsilon)$ satisfies:
\begin{enumerate}[label=(\roman*)]
\item \emph{Lattice grading.}
 The differential preserves the total lattice degree
 $\sum_i \alpha_i \in \Lambda$, decomposing the bar complex as
 $\barBgeom(\Vlat_\Lambda^\varepsilon) =
 \bigoplus_{\gamma \in \Lambda}
 \barBgeom(\Vlat_\Lambda^\varepsilon)_\gamma$.
\item \emph{Residue criterion.}
 Simple poles occur only when $\langle\alpha_i,\alpha_j\rangle = -1$
 for some pair, and these are the only contributions to the differential.
\item \emph{Cocycle dependence.}
 The cocycle $\varepsilon$ appears explicitly in the bar differential
 via~\eqref{eq:lattice:bar-diff}. Different cocycles give
 non-isomorphic bar complexes.
\end{enumerate}
\end{theorem}

\begin{proof}
(i) The OPE $Y(e^\alpha, z)\,Y(e^\beta, w)$ produces vertex operators
of lattice degree $\alpha + \beta$. The residue operation preserves
this degree, since $\mathrm{Res}_{z=w}$ does not change the total lattice
weight $\alpha + \beta$.

(ii) The OPE~\eqref{eq:lattice:ope} has a pole of order
$-\langle\alpha,\beta\rangle$ at $z = w$. Pairing with the logarithmic
form $\eta_{ij} = d\log(z_i - z_j)$ extracts exactly the order-1 residue:
$\mathrm{Res}_{z=w}[(z-w)^{-m} d\log(z-w)] = \delta_{m,1}$.
Hence the differential is nonzero only for a simple pole (order $1$,
i.e., $\langle\alpha,\beta\rangle = -1$). For $\langle\alpha,\beta\rangle \geq 0$
there is no pole and the residue vanishes. For $\langle\alpha,\beta\rangle \leq -2$
the pole of order $\geq 2$ pairs trivially with $d\log$.

(iii) To show that different cocycles give non-isomorphic bar complexes,
note that the differential $d([e^\alpha \mid e^\beta] \otimes 1) = \varepsilon(\alpha,\beta)[e^{\alpha+\beta}]\otimes 1$
in~\eqref{eq:lattice:bar-diff} carries the sign $\varepsilon(\alpha,\beta) \in \{\pm 1\}$
explicitly. A change of cocycle $\varepsilon \mapsto \varepsilon'$ with
$\varepsilon'(\alpha,\beta)/\varepsilon(\alpha,\beta) = c(\alpha)c(\beta)/c(\alpha+\beta)$
(a coboundary, $c: \Lambda \to \{\pm 1\}$) corresponds to rescaling each basis vector
$e^\alpha \mapsto c(\alpha)e^\alpha$, hence to an isomorphism of bar complexes.
However, if $\varepsilon'$ and $\varepsilon$ differ by a non-trivial class in
$H^2(\Lambda;\{\pm 1\})$, no such coboundary exists: the two bar complexes have
differentials that cannot be related by any diagonal change of basis, so they are
non-isomorphic as chain complexes.
\end{proof}

\begin{computation}[\texorpdfstring{$A_1$}{A_1} bar complex]\label{comp:lattice:bar-A1}
For $\Lambda = A_1 = \Z\alpha$ with $\langle\alpha,\alpha\rangle = 2$,
the relevant inner products are:
\[
\langle n\alpha, m\alpha \rangle = 2nm.
\]
Since $\langle n\alpha, m\alpha\rangle = -1$ has no integer solutions,
\emph{no simple poles occur}. The bar differential vanishes identically,
and $H_*(\barBgeom(\Vlat_{A_1})) \cong \barBgeom(\Vlat_{A_1})$ as
graded vector spaces.

The $\AffKM{sl}_2$ current algebra at
level~1 is ``too small'' to have nontrivial collisions in bar degree~2
within a single lattice sector. The nontrivial bar differential involves
Heisenberg descendants (which carry additional oscillator modes that can
produce simple poles via derivative OPEs).
\end{computation}

\begin{computation}[\texorpdfstring{$A_2$}{A_2} bar complex]\label{comp:lattice:bar-A2}
For $\Lambda = A_2$ with simple roots $\alpha_1, \alpha_2$ satisfying
$\langle\alpha_1,\alpha_1\rangle = \langle\alpha_2,\alpha_2\rangle = 2$
and $\langle\alpha_1,\alpha_2\rangle = -1$:

\emph{Bar degree~1.} Generated by $[e^\alpha]$ for
$\alpha \in \Lambda \setminus \{0\}$.

\emph{Bar degree~2.} The key differential is
\[
d([e^{\alpha_1} \mid e^{\alpha_2}] \otimes 1)
= \varepsilon(\alpha_1,\alpha_2)\,[e^{\alpha_1+\alpha_2}] \otimes 1
\]
since $\langle\alpha_1,\alpha_2\rangle = -1$. This maps the element
$[e^{\alpha_1} \mid e^{\alpha_2}]$ of lattice degree
$\alpha_1 + \alpha_2$ to $[e^{\alpha_1+\alpha_2}]$ of the same lattice
degree, with coefficient $\varepsilon(\alpha_1,\alpha_2)$.

\emph{Bar degree~3.}
\[
d([e^{\alpha_1} \mid e^{-\alpha_1-\alpha_2} \mid e^{\alpha_2}]
\otimes \eta_{12} \wedge \eta_{23})
= \varepsilon \cdot [e^0]
\]
since both adjacent pairs have inner product $-1$. Since $e^0 = \mathbf{1}$ lies in the augmentation (i.e., $[\mathbf{1}] = 0$ in the reduced bar complex $\bar{B} = B/\mathbb{C}\cdot\mathbf{1}$), this element is a \emph{cycle}, not an exact element. It realizes the Weyl group relation $s_{\alpha_1} s_{\alpha_2} s_{\alpha_1} = s_{\alpha_2} s_{\alpha_1} s_{\alpha_2}$ as a homology class in the bar complex.

The nontrivial differentials encode the $A_2$ root system structure:
simple poles occur precisely for pairs of roots whose sum is again a root.
\end{computation}

\begin{proposition}[\texorpdfstring{$D_4$}{D4} bar complex and triality; \ClaimStatusProvedHere]
\label{prop:lattice:bar-D4}
For the $D_4$ root lattice, the bar complex
$\barBgeom(\Vlat_{D_4})$ admits an action of the triality group
$S_3 \cong \operatorname{Out}(D_4)$ permuting the three conjugacy classes of
roots. The induced action on homology
$H_*(\barBgeom(\Vlat_{D_4}))$ reflects the triality symmetry of the
$\AffKM{so}_8$ current algebra at level~1.
\end{proposition}

\begin{proof}
The $D_4$ root lattice has 24 roots, all of the form $\pm e_i \pm e_j$ ($i \neq j$). These form a single orbit under the Weyl group $W(D_4)$. The outer automorphism group $\operatorname{Out}(D_4) \cong S_3$ acts as graph automorphisms of the $D_4$ Dynkin diagram, permuting the three ``legs'' and hence permuting the three nontrivial cosets of $D_4^*/D_4 \cong (\mathbb{Z}/2\mathbb{Z})^2$ (vector, spinor, and conjugate spinor representations). Note: the spinor and conjugate spinor vectors lie in $D_4^* \setminus D_4$, not in the root lattice itself.

\emph{Step~1.} Triality action on bar complex.
The triality automorphism $\sigma$ acts on the lattice as a graph automorphism of the $D_4$ Dynkin diagram, permuting the three ``legs.'' This extends to an automorphism of $\Vlat_{D_4}$ via $\sigma(e^\alpha) = e^{\sigma(\alpha)}$. Since the bar construction is functorial (any algebra automorphism induces a coalgebra automorphism of the bar complex), $\sigma$ acts on $\barBgeom(\Vlat_{D_4})$.

\emph{Step~2.} Action on homology.
In bar degree~1, the generators are $[e^\alpha]$ for each of the 24 roots $\alpha$. The triality permutes these roots (preserving inner products). In bar degree~2, the differential $d([e^{\alpha_1} \mid e^{\alpha_2}]) = \varepsilon(\alpha_1,\alpha_2)\,[e^{\alpha_1+\alpha_2}]$ (when $\langle\alpha_1,\alpha_2\rangle = -1$) intertwines with $\sigma$ since $\sigma$ preserves inner products and the cocycle is equivariant (the standard Frenkel--Lepowsky--Meurman cocycle for $D_4$ is triality-invariant up to coboundary).

\emph{Step~3.} Conclusion.
The $S_3$-action on $H_*(\barBgeom(\Vlat_{D_4}))$ is the representation-theoretic manifestation of the triality symmetry: the three fundamental representations of $\mathfrak{so}_8$ (vector, spinor, conjugate spinor) are permuted by outer automorphisms, and this permutation lifts to the bar complex level.
\end{proof}

\begin{proposition}[\texorpdfstring{$E_8$}{E8} bar complex and self-duality; \ClaimStatusProvedHere]
\label{prop:lattice:bar-E8}
For the $E_8$ root lattice, the lattice algebra $\Vlat_{E_8}$ is
self-dual, so its dual coalgebra is computed by the same
$E_8$-graded bar complex $\barBgeom(\Vlat_{E_8})$, reflecting the
unimodularity and self-duality of $E_8$.
The bar complex decomposes as
\[
\barBgeom(\Vlat_{E_8})
= \bigoplus_{\gamma \in E_8} \barBgeom(\Vlat_{E_8})_\gamma
\]
with $|E_8/2E_8| = 2^8 = 256$ sectors modulo~$2E_8$. The 240 roots of
$E_8$ each contribute a 1-dimensional summand in bar degree~1.
\end{proposition}

\begin{proof}
\emph{Step~1.} Lattice decomposition.
The bar complex decomposes along the $E_8$-grading:
\[
\barBgeom(\Vlat_{E_8}) = \bigoplus_{\gamma \in E_8} \barBgeom(\Vlat_{E_8})_\gamma.
\]
The zero sector $\gamma = 0$ is the bar complex of the rank-8 Heisenberg algebra
$\barBgeom(\cH_8)$, whose bar differential vanishes (the Heisenberg OPE has only
a double pole, so $\cH_8^! \simeq \mathrm{Sym}^{\mathrm{ch}}(V^*)$
by Theorem~\ref{thm:heisenberg-koszul-dual-early}).

\emph{Step~2.} Root contributions.
In bar degree~1, each root $\alpha$ of $E_8$ contributes a generator $[e^\alpha]$ in the sector $\gamma = \alpha$. Since $E_8$ has 240 roots, $\dim \barBgeom^1 = 240$ (in the root sectors). In bar degree~2, the differential is $d([e^{\alpha_1} \mid e^{\alpha_2}]) = \varepsilon(\alpha_1,\alpha_2)\,[e^{\alpha_1+\alpha_2}]$ when $\langle\alpha_1,\alpha_2\rangle = -1$. For $E_8$, each root $\alpha$ has exactly 56 neighbors $\beta$ with $\langle\alpha,\beta\rangle = -1$ (so that $\alpha+\beta$ is again a root), as follows from the root system geometry: the 240 roots partition by inner product with $\alpha$ as $1 + 56 + 126 + 56 + 1$.

\emph{Step~3.} Self-duality.
By Theorem~\ref{thm:lattice:koszul-dual}, $(\Vlat_{E_8}^\varepsilon)^! = (\Vlat_{E_8}^{\varepsilon^{-1}})^c$. Since $E_8$ is unimodular ($E_8 = E_8^*$), the cocycle $\varepsilon$ for $E_8$ satisfies $\varepsilon^{-1} = \varepsilon$: the standard FLM cocycle takes values in $\{\pm 1\}$, so $\varepsilon(\alpha,\beta)^{-1} = \varepsilon(\alpha,\beta)$ for all $\alpha,\beta$. Therefore $(\Vlat_{E_8})^! \cong \Vlat_{E_8}$ as a coalgebra.

\emph{Step~4.} Sector count.
Modulo $2E_8$, the lattice has $|E_8/2E_8| = 2^{\operatorname{rank}(E_8)} = 2^8 = 256$ classes (since $\Lambda/2\Lambda \cong (\mathbb{Z}/2\mathbb{Z})^d$ for any rank-$d$ lattice), giving 256 sectors in the bar complex.
\end{proof}

\begin{theorem}[Unimodular lattice self-duality; \ClaimStatusProvedHere]\label{thm:lattice:unimodular-self-dual}
\index{self-dual!lattice VOA}
Let $\Lambda$ be an even unimodular lattice. Then $\Vlat_\Lambda$ is Koszul self-dual:
\[
\Vlat_\Lambda^! \;\cong\; \Vlat_\Lambda
\]
with the bar-cobar quasi-isomorphism $\Omega^{\mathrm{ch}}(\barBgeom(\Vlat_\Lambda)) \xrightarrow{\sim} \Vlat_\Lambda$ providing bar-cobar inversion (a separate, general phenomenon).
\end{theorem}

\begin{proof}
By Theorem~\ref{thm:lattice:koszul-dual}, $\Vlat_\Lambda^! = (\Vlat_\Lambda^{\varepsilon^{-1}})^c$. For the standard FLM cocycle $\varepsilon : \Lambda \times \Lambda \to \{\pm 1\}$, we have $\varepsilon^{-1} = \varepsilon$ (since $\varepsilon(\alpha,\beta)^{-1} = \varepsilon(\alpha,\beta)$ for all $\alpha,\beta$), so $\Vlat_\Lambda^{\varepsilon^{-1}} = \Vlat_\Lambda^\varepsilon$. Hence $\Vlat_\Lambda^{\varepsilon^{-1}} \cong \Vlat_\Lambda^\varepsilon$ as vertex algebras. The coalgebra structure on the dual side matches the algebra structure on the original side (since the cocycle is self-inverse up to coboundary), giving $\Vlat_\Lambda^! \cong \Vlat_\Lambda$.

The bar-cobar QI follows from Theorem~\ref{thm:lattice:koszul-morphism}: the twisting morphism $\tau$ is Koszul, so $\Omega^{\mathrm{ch}}(\barBgeom(\Vlat_\Lambda)) \xrightarrow{\sim} \Vlat_\Lambda$.
\end{proof}

\section{Koszul duality for lattice algebras}
\label{sec:lattice:koszul}

\begin{definition}[Inverse cocycle]\label{def:lattice:inverse-cocycle}
For a cocycle $\varepsilon \colon \Lambda \times \Lambda \to \C^\times$,
the \emph{inverse cocycle} is
$\varepsilon^{-1}(\alpha,\beta) := \varepsilon(\alpha,\beta)^{-1}$.
The cocycle condition is preserved under inversion, and the commutator
satisfies
$c_{\varepsilon^{-1}}(\alpha,\beta) = c_\varepsilon(\alpha,\beta)^{-1}$.
In particular, if $\varepsilon$ is symmetric, so is $\varepsilon^{-1}$.
\end{definition}

\begin{theorem}[Dual coalgebra of lattice vertex algebra; \ClaimStatusProvedHere]
\label{thm:lattice:koszul-dual}
\index{Koszul dual!lattice VOA}
The dual coalgebra of the lattice $\Eone$-chiral algebra
$\Vlat_\Lambda^\varepsilon$ is:
\begin{equation}\label{eq:lattice:koszul-dual}
\bigl(\Vlat_\Lambda^\varepsilon\bigr)^i
\;:=\;
H^*\!\bigl(\barB^{\mathrm{ch}}(\Vlat_\Lambda^\varepsilon)\bigr)
\;=\; (\Vlat_\Lambda^{\varepsilon^{-1}})^c,
\end{equation}
the coalgebra structure on the lattice vertex algebra with inverse
cocycle. Consequently the Koszul dual algebra
$\bigl(\Vlat_\Lambda^\varepsilon\bigr)^!$ is the linear/Verdier dual of
this coalgebra. More precisely:
\begin{enumerate}[label=(\roman*)]
\item The underlying graded vector space of
 $\bigl(\Vlat_\Lambda^\varepsilon\bigr)^i$ is $\Vlat_\Lambda^*$.
\item The coalgebra structure is determined by $\varepsilon^{-1}$:
 the coproduct $\Delta \colon \Vlat_\Lambda^* \to
 \Vlat_\Lambda^* \otimes \Vlat_\Lambda^*$ has structure constants
 $\varepsilon(\alpha,\beta)^{-1}$.
\item For symmetric $\varepsilon$, the dual coalgebra is again a
 (co)lattice vertex algebra with the same symmetry type.
\end{enumerate}
\end{theorem}

\begin{proof}
The bar coalgebra is computed by the bar construction, and its
cohomology coalgebra gives the dual coalgebra. The
product $\mu \colon \Vlat_\Lambda \otimes \Vlat_\Lambda \to \Vlat_\Lambda$
has structure constants $\varepsilon(\alpha,\beta)$ from the group
algebra multiplication $e^\alpha \cdot e^\beta =
\varepsilon(\alpha,\beta)\,e^{\alpha+\beta}$.
The dual coalgebra is extracted from the bar complex
$\bar{B}^{\mathrm{ch}}(\Vlat_\Lambda^\varepsilon)$. In the bar construction,
each tensor factor is desuspended by $s^{-1}$, and the bar differential
involves the product $\mu$ composed with Koszul signs from the
desuspension. For the lattice algebra, where $\mu(e^\alpha \otimes
e^\beta) = \varepsilon(\alpha,\beta)\,e^{\alpha+\beta}$ with
$|\varepsilon(\alpha,\beta)| = \pm 1$, the desuspension introduces a
sign $(-1)^{|\alpha||\beta|}$ that, combined with the bilinearity of
$\varepsilon$, produces the cocycle inversion:
the bar-complex coproduct has structure constants
$\varepsilon(\alpha,\beta)^{-1}$.
Since $\varepsilon^{-1}$ is again a valid cocycle,
$(\Vlat_\Lambda^{\varepsilon^{-1}})^c$ is a valid coalgebra, and the
dual coalgebra is identified with it; the dual algebra is then obtained
by linear/Verdier duality.
\end{proof}

\begin{theorem}[Koszul morphism for lattice algebras; \ClaimStatusProvedHere]
\label{thm:lattice:koszul-morphism}
The canonical twisting morphism
$\tau \colon \barBgeom(\Vlat_\Lambda^\varepsilon) \to
\Vlat_\Lambda^\varepsilon$
is a Koszul morphism: the twisted complex
\[
\barBgeom(\Vlat_\Lambda^\varepsilon)
\otimes_\tau \Vlat_\Lambda^\varepsilon
\]
is acyclic.
\end{theorem}

\begin{proof}
The proof uses the lattice weight filtration.

\emph{Step~1: Filtration setup.}
Filter the twisted complex by total lattice weight: define $F^p(\barBgeom(\Vlat_\Lambda^\varepsilon) \otimes_\tau \Vlat_\Lambda^\varepsilon)$ to consist of elements whose total lattice weight $\sum_i |\alpha_i|^2 \geq p$ (where $|\alpha|^2 = \langle\alpha,\alpha\rangle$ for the inner product on $\Lambda$, which is positive-definite by assumption). Since $\Lambda$ is positive-definite, each total degree $n$ contains only finitely many lattice vectors of bounded weight, so the filtration is bounded below and convergent.

\emph{Step~2: $E_1$ identification.}
The differential $\dzero$ on the associated graded $E_0$ is the bar differential for the Heisenberg subalgebra $\cH_d \subset \Vlat_\Lambda^\varepsilon$ (the zero-weight sector, generated by the oscillator modes $a^i_n$, $n < 0$). This is because the leading-order OPE at weight $p$ is between a weight-$p$ vertex operator and the unit: the lattice exponentials are constant on $\mathrm{gr}$ and only the Heisenberg structure contributes to $\dzero$.

\emph{Step~3: Acyclicity at $E_1$.}
By the Heisenberg bar-cobar quasi-isomorphism (Theorem~\ref{thm:heisenberg-koszul-dual-early}, applied to $\cH_d$), the twisted complex for the Heisenberg algebra is acyclic: $H^*(\barBgeom(\cH_d) \otimes_\tau \cH_d) = 0$ in positive degree. Hence $E_1^{p,q} = 0$ for $p+q > 0$, and the spectral sequence collapses at $E_1$.

\emph{Step~4: Cocycle phases.}
The cocycle $\varepsilon(\alpha,\beta) \in \{\pm 1\}$ appears as a sign in the differential of the twisted complex, but the underlying graded vector spaces depend only on $\Lambda$ (not on the choice of $\varepsilon$). Since the acyclicity at $E_1$ is established for all choices of $\varepsilon$ simultaneously (the Heisenberg part $\dzero$ is $\varepsilon$-independent), the twisted complex is acyclic for every valid 2-cocycle.

\emph{Step~5: Convergence.}
Since the filtration is bounded below in each total degree and $E_1^{p,q} = 0$ for $p+q > 0$, the spectral sequence converges to zero in positive total degree, establishing acyclicity of $\barBgeom(\Vlat_\Lambda^\varepsilon) \otimes_\tau \Vlat_\Lambda^\varepsilon$ as required.
\end{proof}

\begin{remark}[Self-duality for unimodular lattices]
\label{rem:lattice:self-duality}
When $\Lambda$ is unimodular ($\Lambda = \Lambda^*$) and the cocycle
$\varepsilon$ satisfies $\varepsilon^{-1} \sim \varepsilon$ (cohomologous),
the Koszul dual $(\Vlat_\Lambda^\varepsilon)^!$ is isomorphic to
$\Vlat_\Lambda^\varepsilon$ itself (as a coalgebra). This is the case
for $E_8$: the lattice vertex algebra $\Vlat_{E_8}$ is
\emph{Koszul self-dual}. (Note: the Heisenberg algebra $\mathcal{H}_\kappa$ is
Koszul but \emph{not} self-dual: $\mathcal{H}_\kappa^! \simeq \mathrm{Sym}^{ch}(V^*)$
by Theorem~\ref{thm:heisenberg-koszul-dual-early}. Self-duality is a special
feature of the $E_8$ lattice structure.)
\end{remark}

\begin{remark}[Koszul dual kappa for non-unimodular lattices]
\label{rem:lattice:koszul-dual-kappa}%
\index{Koszul dual!lattice VOA!kappa landscape}%
\ClaimStatusProvedHere
%: kappa(V_Lambda) = rank(Lambda) from landscape_census.tex table row "V_Lambda (rank d)"; rank=1 -> 1 (Heis), rank=24 -> 24 (Leech) verified.
For a general even lattice~$\Lambda$ (not necessarily unimodular),
the Koszul dual $\Vlat_\Lambda^!$ of
Theorem~\ref{thm:lattice:koszul-dual}
satisfies the free-field complementarity
(Theorem~\ref{thm:complementarity-root-datum}):
\begin{equation}\label{eq:lattice:kappa-complementarity}
\kappa_{\mathrm{latt}}(\Vlat_\Lambda)
+ \kappa_{\mathrm{latt}}(\Vlat_\Lambda^!)
\;=\; 0,
\qquad
\kappa_{\mathrm{latt}}(\Vlat_\Lambda^!)
\;=\; -\operatorname{rank}(\Lambda).
\end{equation}
Three cases illustrate the range of this identity.
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Rank~$1$ (Heisenberg).}
$\Vlat_\bZ = \cH_k$ at $k = 1$ has
$\kappa_{\mathrm{latt}} = 1$ and
$\cH_1^! \simeq \mathrm{Sym}^{\mathrm{ch}}(V^*)$
with $\kappa_{\mathrm{latt}}(\cH_1^!) = -1$
(Theorem~\ref{thm:heisenberg-koszul-dual-early}).
The Koszul dual is the curved chiral symmetric algebra,
\emph{not} $\cH_{-1}$: these are different algebras
with the same numerical kappa.
\item \emph{Even unimodular lattices} ($E_8$, Leech, all Niemeier).
Theorem~\ref{thm:lattice:unimodular-self-dual} gives
$\Vlat_\Lambda^! \cong \Vlat_\Lambda$, so the
complementarity $\kappa + \kappa' = 0$ becomes
$2\kappa_{\mathrm{latt}} = 0$;
this is consistent only because the self-dual
identification is as \emph{coalgebras}, and
$\kappa_{\mathrm{latt}}(\Vlat_\Lambda^!)$ is computed
on the coalgebra side where the sign of the curvature
reverses. For the Leech lattice,
$\kappa_{\mathrm{latt}}(\Vlat_{\Lambda_{24}}) = 24$
and
$\kappa_{\mathrm{latt}}(\Vlat_{\Lambda_{24}}^!) = -24$.
\item \emph{Non-unimodular lattices} ($D_4$, $A_2$, general $\Lambda$).
Here $\Vlat_\Lambda^! = (\Vlat_\Lambda^{\varepsilon^{-1}})^c
\not\cong \Vlat_\Lambda$:
the Koszul dual is a genuinely different lattice chiral
algebra, with negative kappa
$\kappa_{\mathrm{latt}}(\Vlat_\Lambda^!) =
-\operatorname{rank}(\Lambda) < 0$.
Such negative-kappa algebras lie outside the unitary regime
($\kappa > 0$); the Koszul dual is a curved
$\Eone$-chiral coalgebra whose curvature
$m_0 = -\operatorname{rank}(\Lambda)\,\omega$
has the opposite sign to the original.
\end{enumerate}
The negative sign of $\kappa_{\mathrm{latt}}(\Vlat_\Lambda^!)$
is the lattice shadow of the general principle that
Koszul duality reverses the sign of curvature in the
bar complex
(cf.\ the Heisenberg case,
Theorem~\ref{thm:heisenberg-koszul-dual-early},
where $m_0 = -k\,\omega$).
\end{remark}

\begin{remark}[Structural form of the lattice Koszul dual]
\label{rem:lattice:dual-structure}%
\index{Koszul dual!lattice VOA!structural identification}%
\ClaimStatusProvedHere
%: kappa(V_Lambda) = rank(Lambda) from landscape_census.tex:261;
% rank=1 -> 1 (Heis), rank=24 -> 24 (Leech) verified.
% kappa(V_Lambda^!) = -rank(Lambda) from eq:lattice:kappa-complementarity.
Theorem~\ref{thm:lattice:koszul-dual} identifies
$\Vlat_\Lambda^! = (\Vlat_\Lambda^{\varepsilon^{-1}})^c$
at the level of coalgebra structure constants.
We describe what this coalgebra looks like as a chiral algebra
at the Heisenberg-sector level, where the lattice exponentials
are stripped away and only the oscillator modes remain.

\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Rank~$1$.}
 The lattice VOA $\Vlat_\bZ = \cH_1$ is
 a single free boson at level~$1$. Its Koszul dual is
 $\cH_1^! \simeq \mathrm{Sym}^{\mathrm{ch}}(V^*)$
 (Theorem~\ref{thm:heisenberg-koszul-dual-early}):
 the curved commutative chiral algebra on the dual
 vector space, with curvature $m_0 = -\omega$.
 Equivalently, $\cH_1^! \simeq \mathrm{CE}(\mathfrak{h}_{-1})$
 (the Chevalley--Eilenberg algebra of the abelian Lie algebra
 at level~$-1$), with $c(\cH_1^!) = -1$ and
 $\kappa_{\mathrm{latt}}(\cH_1^!) = -1$.
 This is \emph{not} the $\beta\gamma$ system
 (which has two generators); it is the one-generator
 curved commutative chiral algebra.

\item \emph{General rank~$d$.}
 For an even lattice $\Lambda$ of rank~$d$, the Heisenberg
 subalgebra $\cH_d \subset \Vlat_\Lambda$ is a tensor product
 of $d$ free bosons. By the tensor product compatibility
 of Koszul duality
 (Theorem~\ref{thm:lattice:direct-sum}
 and Corollary~\ref{cor:kappa-additivity}),
 the Heisenberg sector of $\Vlat_\Lambda^!$ is
 \begin{equation}\label{eq:lattice:dual-heisenberg-sector}
 \cH_d^!
 \;\simeq\;
 \mathrm{Sym}^{\mathrm{ch}}(\Lambda^* \otimes \omega_X),
 \end{equation}
 the rank-$d$ curved commutative chiral algebra
 with curvature $m_0 = -d\,\omega$.
 Equivalently, $\cH_d^! \simeq \mathrm{CE}(\mathfrak{h}_{-1}^{\oplus d})$,
 with central charge $c(\cH_d^!) = -d$
 and $\kappa_{\mathrm{latt}}(\cH_d^!) = -d$, consistent
 with \eqref{eq:lattice:kappa-complementarity}.
 The lattice exponentials of $\Vlat_\Lambda^!$ are
 then governed by the inverse cocycle $\varepsilon^{-1}$
 on $\Lambda^*$, dressing each sector
 $(\Vlat_\Lambda^!)_\gamma$ for $\gamma \in \Lambda^*$
 with the dual oscillator Fock space.

\item \emph{Leech lattice} ($d = 24$).
 The Leech lattice VOA
 $\Vlat_{\Lambda_{24}}$ has
 $c = 24$, $\kappa_{\mathrm{latt}} = 24$, and is Koszul
 self-dual as a coalgebra
 (Theorem~\ref{thm:lattice:unimodular-self-dual}).
 Its Koszul dual has $c(\Vlat_{\Lambda_{24}}^!) = -24$
 and $\kappa_{\mathrm{latt}}(\Vlat_{\Lambda_{24}}^!) = -24$.
 The value $c = -24$ matches the rank with reversed sign,
 reflecting the $24$-fold tensor product structure
 $\cH_1^{\otimes 24} \mapsto (\cH_1^!)^{\otimes 24}$
 under Koszul duality.
 By contrast, the Monster module $V^\natural$ has $c = 24$
 but $\kappa_{\mathrm{Vir}}(V^\natural) = c/2 = 12$
 (Remark~\ref{rem:lattice:monster-shadow}):
 the Koszul dual of $V^\natural$ lies in the Virasoro
 complementarity class ($K_{\mathrm{Vir}} = 13$), while
 the Koszul dual of $\Vlat_{\Lambda_{24}}$ lies in the
 free-field class ($K_{\mathrm{latt}} = 0$). The shadow
 tower separates the two: all twenty-four Niemeier lattice
 VOAs share $\kappa_{\mathrm{latt}} = 24$, class~$\mathbf{G}$,
 finite shadow depth; $V^\natural$ has
 $\kappa_{\mathrm{Vir}} = 12$, class~$\mathbf{M}$,
 infinite shadow depth.
\end{enumerate}

The structural content of $K_{\mathrm{latt}} = 0$ is that
$\Vlat_\Lambda \otimes \Vlat_\Lambda^!$ carries a trivially
curved bar complex: the curvatures $m_0 = +d\,\omega$ and
$m_0^! = -d\,\omega$ cancel under the tensor product,
so the total complex has $m_0^{\mathrm{tot}} = 0$.
This is the chiral algebraic shadow of the classical fact
that tensoring a lattice with its dual produces an
orthogonal direct sum $\Lambda \oplus \Lambda^*$ whose
bilinear form is split (hyperbolic).
\end{remark}

\section{Functorial properties and lattice operations}
\label{sec:lattice:functorial}

Each lattice operation from
Definition~\ref{def:lattice:operations} lifts to a functorial operation
on vertex algebras, and these lifts are compatible with the bar
construction.

\subsection{Direct sum}

\begin{theorem}[Tensor product from direct sum; \ClaimStatusProvedHere]
\label{thm:lattice:direct-sum}
For even lattices $\Lambda_1, \Lambda_2$ with cocycles
$\varepsilon_1, \varepsilon_2$:
\begin{equation}\label{eq:lattice:tensor-product}
\Vlat_{\Lambda_1 \oplus \Lambda_2}^{\varepsilon_1 \otimes \varepsilon_2}
\;\cong\;
\Vlat_{\Lambda_1}^{\varepsilon_1}
\otimes \Vlat_{\Lambda_2}^{\varepsilon_2}
\end{equation}
where $(\varepsilon_1 \otimes \varepsilon_2)
((\alpha_1,\alpha_2),(\beta_1,\beta_2)) =
\varepsilon_1(\alpha_1,\beta_1)\,\varepsilon_2(\alpha_2,\beta_2)$.
\end{theorem}

\begin{proof}
The state space decomposes as
$\cH_{\Lambda_1 \oplus \Lambda_2} \otimes \C[\Lambda_1 \oplus \Lambda_2]
\cong (\cH_{\Lambda_1} \otimes \C[\Lambda_1]) \otimes
(\cH_{\Lambda_2} \otimes \C[\Lambda_2])$.
The vertex operator for $e^{(\alpha_1,\alpha_2)}$ factors as
$Y(e^{\alpha_1}, z) \otimes Y(e^{\alpha_2}, z)$
since the oscillators for $\Lambda_1$ and $\Lambda_2$ commute
(orthogonality of the direct sum).
\end{proof}

\begin{corollary}[K\"unneth for bar complexes \cite{LV12}; \ClaimStatusProvedElsewhere]\label{cor:lattice:kunneth}
Under the isomorphism~\eqref{eq:lattice:tensor-product}:
\[
\barBgeom(\Vlat_{\Lambda_1} \otimes \Vlat_{\Lambda_2})
\simeq
\barBgeom(\Vlat_{\Lambda_1})
\otimes \barBgeom(\Vlat_{\Lambda_2}).
\]
\end{corollary}

\begin{proof}
The bar construction of a tensor product of vertex algebras is
quasi-isomorphic to the tensor product of bar constructions, by
the Eilenberg--Zilber/K\"unneth theorem for the bar construction
(cf.\ Chapter~\ref{ch:algebraic-foundations}). The lattice grading
ensures the relevant spectral sequence converges.
\end{proof}

\subsection{Sublattice embedding}

\begin{proposition}[Sublattice maps; \ClaimStatusProvedHere]\label{prop:lattice:sublattice}
An embedding of even lattices $\iota \colon \Lambda' \hookrightarrow \Lambda$
induces:
\begin{enumerate}[label=(\roman*)]
\item An embedding of vertex algebras
 $\iota_* \colon \Vlat_{\Lambda'}^{\iota^*\varepsilon}
 \hookrightarrow \Vlat_\Lambda^\varepsilon$
 (restriction of the cocycle).
\item A map of bar complexes
 $\barBgeom(\iota_*) \colon
 \barBgeom(\Vlat_{\Lambda'}) \to \barBgeom(\Vlat_\Lambda)$.
\item On homology, the induced map respects the lattice grading:
 the $\gamma$-sector of $H_*(\barBgeom(\Vlat_{\Lambda'}))$ maps to
 the $\iota(\gamma)$-sector of $H_*(\barBgeom(\Vlat_\Lambda))$.
\end{enumerate}
\end{proposition}

\begin{proof}
(i) The inclusion $\iota \colon \Lambda' \hookrightarrow \Lambda$
induces an embedding of group algebras
$\C[\Lambda'] \hookrightarrow \C[\Lambda]$
(sending $e^\gamma \mapsto e^{\iota(\gamma)}$) and of Heisenberg
algebras $\cH_{\Lambda'} \hookrightarrow \cH_\Lambda$
(the lattice is a subspace of the target space).
The cocycle on $\Lambda'$ is the restriction
$\iota^*\varepsilon(\gamma, \gamma') = \varepsilon(\iota(\gamma),\iota(\gamma'))$.
The vertex algebra structure is preserved because
the state--field correspondence $Y(e^\gamma, z) = V_\gamma(z)$
restricts to $Y(e^{\iota(\gamma)}, z)$, and the OPE singularities
are determined by the inner product which is preserved by $\iota$.

(ii) The bar construction $\barBgeom$ is a functor from chiral algebras to
chiral coalgebras. The embedding $\iota_*$ is a morphism of chiral
algebras, so $\barBgeom(\iota_*)$ is a well-defined coalgebra map.
Concretely, the configuration space integrals defining the bar
differential on $\barBgeom(\Vlat_{\Lambda'})$ are computed using the
OPE of $\Vlat_{\Lambda'}$, which is the restriction of the OPE of
$\Vlat_\Lambda$, so $\barBgeom(\iota_*)$ commutes with differentials.

(iii) The lattice grading is defined by total lattice charge
$\gamma_{\mathrm{tot}} = \sum_i \gamma_i$ for a bar element
$s^{-1}e^{\gamma_1} \otimes \cdots \otimes s^{-1}e^{\gamma_n}$.
Since $\iota$ maps $\Lambda'$-elements to $\Lambda$-elements via a
group homomorphism,
$\iota(\gamma_{\mathrm{tot}}) = \sum_i \iota(\gamma_i) = \iota(\gamma)_{\mathrm{tot}}$,
so the $\gamma$-sector of $\barBgeom(\Vlat_{\Lambda'})$ maps into the
$\iota(\gamma)$-sector of $\barBgeom(\Vlat_\Lambda)$.
The differential preserves lattice charge (the product
$e^\alpha \cdot e^\beta = \varepsilon(\alpha,\beta)\,e^{\alpha+\beta}$
preserves total charge), so the grading is compatible on homology.
\end{proof}

\subsection{Overlattice extension}

\begin{theorem}[Overlattice vertex algebra \cite{FLM88}; \ClaimStatusProvedElsewhere]\label{thm:lattice:overlattice}
Let $\Lambda$ be an even lattice and $I \subset D(\Lambda) = \Lambda^*/\Lambda$
an isotropic subgroup. The overlattice $\Lambda_I$ is even, and:
\begin{equation}\label{eq:lattice:overlattice}
\Vlat_{\Lambda_I}
\;\cong\; \bigoplus_{\gamma \in I} \Vlat_\Lambda^\gamma
\end{equation}
where $\Vlat_\Lambda^\gamma$ denotes the $\gamma$-twisted module
for $\Vlat_\Lambda$, defined as follows.

\begin{definition}[Twisted module for lattice VOA]\label{def:twisted-module-lattice}
For a coset $\gamma \in \Lambda^*/\Lambda$, the \emph{$\gamma$-twisted module} for the lattice VOA $\Vlat_\Lambda$ is:
\[
\Vlat_\Lambda^\gamma = \cH \otimes \C[\gamma + \Lambda]
\]
where $\cH$ is the Heisenberg Fock space and $\C[\gamma + \Lambda]$ is the group algebra of the coset. The vertex operators act by $Y(e^\alpha, z) = E^-(\alpha, z) E^+(\alpha, z) e^\alpha z^{\langle \alpha, - \rangle}$ with the shift $z^{\langle \alpha, \gamma \rangle}$ in the monodromy. When $\gamma \neq 0$, the fields $Y(v, z)$ have branch cuts of order $\langle \alpha, \gamma \rangle \bmod \mathbb{Z}$, making this a genuine twisted (non-local) module.
\end{definition}
\end{theorem}

\begin{proof}
\emph{Step 1: State space decomposition.}
Since $\Lambda \subset \Lambda_I$ is a sublattice of finite index,
the group algebra decomposes as
$\C[\Lambda_I] = \bigoplus_{\gamma \in \Lambda_I/\Lambda} \C[\gamma + \Lambda]$.
Tensoring with the Fock space gives:
\[
\Vlat_{\Lambda_I} = \cH \otimes \C[\Lambda_I]
= \bigoplus_{\gamma \in \Lambda_I/\Lambda} \cH \otimes \C[\gamma + \Lambda]
= \bigoplus_{\gamma \in I} \Vlat_\Lambda^\gamma,
\]
where the last equality identifies $\Lambda_I/\Lambda \cong I$ (by
definition of the overlattice) and
$\cH \otimes \C[\gamma + \Lambda] = \Vlat_\Lambda^\gamma$.

\emph{Step 2: Cocycle extension.}
A $2$-cocycle $\varepsilon$ on $\Lambda$ extends to $\Lambda_I$ if
and only if $\varepsilon(\alpha,\beta)\,\varepsilon(\beta,\alpha) =
(-1)^{\langle\alpha,\beta\rangle}$ holds for all
$\alpha, \beta \in \Lambda_I$. For $\gamma, \gamma' \in I$ and
$\lambda, \lambda' \in \Lambda$, we need
$(-1)^{\langle\gamma+\lambda, \gamma'+\lambda'\rangle}$ to be
well-defined modulo coboundaries. The isotropicity condition
$\langle\gamma, \gamma'\rangle \in \Z$ for all
$\gamma, \gamma' \in I$ ensures this: the bilinear form on the
discriminant group $D(\Lambda)$ restricts to zero on~$I$, so the
$\Lambda$-cocycle extends without obstruction.

\emph{Step 3: Vertex algebra structure.}
With the extended cocycle, the vertex operators
$Y(e^{\gamma+\lambda}, z)$ for $\gamma \in I$, $\lambda \in \Lambda$
satisfy the same OPE as for $\Lambda$ (locality, associativity,
vacuum axiom), since these follow from the cocycle identity and the
lattice inner product. The resulting structure on
$\bigoplus_{\gamma \in I} \Vlat_\Lambda^\gamma$ is a vertex algebra,
and the decomposition is an isomorphism of vertex algebras by
construction.
\end{proof}

\begin{example}[Heterotic lattices]\label{ex:lattice:heterotic}
The two even unimodular lattices of rank~16 arise as overlattices:
\begin{enumerate}[label=(\roman*)]
\item $E_8 \oplus E_8$: already unimodular, no extension needed.
\item $D_{16}^+ = (D_{16})_I$ where $I = \{0, [\text{spinor}]\} \cong \Z/2\Z$
 is the isotropic subgroup generated by the spinor class in
 $D(\Lambda_{D_{16}}) \cong (\Z/2\Z)^2$.
\end{enumerate}
The corresponding vertex algebras $\Vlat_{E_8 \oplus E_8}$ and
$\Vlat_{D_{16}^+}$ give the two heterotic string theories. Their bar
complexes are non-isomorphic despite having the same rank and
determinant.
\end{example}

\subsection{Orbifold construction}

\begin{construction}[Orbifold vertex algebra]
\label{constr:lattice:orbifold}
For $\sigma \in \operatorname{Aut}(\Lambda)$ of finite order $N$,
the \emph{orbifold vertex algebra} is the fixed-point subalgebra:
\[
(\Vlat_\Lambda^\varepsilon)^\sigma
= \{v \in \Vlat_\Lambda^\varepsilon : \sigma \cdot v = v\},
\]
where $\sigma$ acts on $\Vlat_\Lambda$ by
$\sigma(e^\alpha) = e^{\sigma(\alpha)}$ and $\sigma(\alpha_{-n}) =
(\sigma\alpha)_{-n}$ on Heisenberg oscillators.

When $\sigma$ preserves the cocycle ($\varepsilon(\sigma\alpha,\sigma\beta)
= \varepsilon(\alpha,\beta)$), this is a vertex subalgebra.
\end{construction}

\section{Chiral Hochschild cohomology of lattice algebras}
\label{sec:lattice:hochschild}

\begin{theorem}[Lattice chiral Hochschild cohomology; \ClaimStatusProvedHere]\label{thm:lattice:hochschild}
For an even integral lattice $\Lambda$:
\begin{equation}\label{eq:lattice:hochschild}
\ChirHoch^n(\Vlat_\Lambda, \Vlat_\Lambda)
\;=\; \bigoplus_{\alpha \in \Lambda}
\ChirHoch^n(\Vlat_\Lambda, \Vlat_\Lambda^\alpha)
\end{equation}
and the degree-zero component ($\alpha = 0$) is:
\[
\ChirHoch^n(\Vlat_\Lambda, \Vlat_\Lambda^0)
\;\cong\; H^n(\Lambda, \Vlat_\Lambda^{\Lambda})
\]
where $\Vlat_\Lambda^\Lambda = \Vlat_\Lambda^0 = \cH$ denotes the
$\Lambda$-invariants (the degree-zero sector under the lattice grading)
and $H^n(\Lambda, -)$ is group cohomology of the free abelian group
$\Lambda$. The non-zero sectors $\alpha \neq 0$ contribute additional
classes that depend on the lattice geometry.
\end{theorem}

\begin{proof}
The lattice vertex algebra decomposes as
$\Vlat_\Lambda = \bigoplus_{\alpha \in \Lambda} \Vlat_\Lambda^\alpha$
where $\Vlat_\Lambda^\alpha = \cH \otimes e^\alpha$ is the
$\alpha$-isotypic component.

The chiral Hochschild complex respects this grading:
\[
CC^n_{\mathrm{ch}}(\Vlat_\Lambda, \Vlat_\Lambda)
= \bigoplus_{\alpha_1, \ldots, \alpha_n, \beta}
\Hom(\Vlat_\Lambda^{\alpha_1} \otimes \cdots
\otimes \Vlat_\Lambda^{\alpha_n}, \Vlat_\Lambda^\beta).
\]
The differential preserves $\sum_i \alpha_i = \beta$, and the
resulting spectral sequence (filtering by lattice degree)
collapses at $E_2$ to give:
\[
\ChirHoch^*(\Vlat_\Lambda, \Vlat_\Lambda)
= \bigoplus_{\alpha \in \Lambda}
\ChirHoch^*(\Vlat_\Lambda, \Vlat_\Lambda^\alpha).
\]
For $\alpha = 0$, this identifies with the group cohomology
$H^*(\Lambda, \Vlat_\Lambda^{\Lambda})$ via the standard bar resolution
for $\Lambda \cong \Z^d$.
\end{proof}

\begin{corollary}[Unimodular case; \ClaimStatusProvedHere]\label{cor:lattice:hochschild-unimodular}
For a unimodular lattice $\Lambda$ (so $D(\Lambda) = 0$), the chiral
chiral Hochschild cohomology reduces to:
\[
\ChirHoch^n(\Vlat_\Lambda, \Vlat_\Lambda)
\cong H^n(\Lambda, \cH)
\]
where $\cH$ is the Heisenberg Fock space (the $\Lambda$-invariant
sector when $\Lambda = \Lambda^*$).
\end{corollary}
\begin{proof}
When $\Lambda$ is unimodular, $D(\Lambda) = \Lambda^*/\Lambda = 0$, so the discriminant contributions vanish and the chiral Hochschild complex reduces to the group cohomology complex $C^*(\Lambda, \mathbb{C})$ by the preceding theorem.
\end{proof}

\begin{example}[\texorpdfstring{$A_1$}{A1} chiral Hochschild cohomology]\label{ex:lattice:hochschild-A1}
For $\Lambda = A_1 \cong \Z$, the group cohomology of $\Z$ is:
\[
H^n(\Z, M) =
\begin{cases}
M^\Z & n = 0, \\
M_\Z & n = 1, \\
0 & n \geq 2.
\end{cases}
\]
Hence $\ChirHoch^n(\Vlat_{A_1}, \Vlat_{A_1})$ is concentrated
in degrees 0 and~1. This reflects the rigidity of the $\AffKM{sl}_2$
current algebra at level~1: it has a 1-dimensional space of first-order
deformations (the level) and no higher obstructions.
\end{example}

\section{Higher genus}
\label{sec:lattice:higher-genus}

At higher genus, the lattice grading on the bar complex persists and
interacts with the geometry of moduli spaces to produce modular objects.

\begin{proposition}[Genus-1 partition function; \ClaimStatusProvedHere]
\label{prop:lattice:genus-1}
For an even lattice $\Lambda$ of rank $d$, the genus-1 partition
function of $\Vlat_\Lambda$ is:
\begin{equation}\label{eq:lattice:genus-1-Z}
Z_1(\Vlat_\Lambda, \tau)
= \frac{\Theta_\Lambda(\tau)}{[\eta(\tau)]^d}
\end{equation}
where $\Theta_\Lambda(\tau) = \sum_{\alpha \in \Lambda}
q^{\langle\alpha,\alpha\rangle/2}$ is the theta series of $\Lambda$
(with $q = e^{2\pi i \tau}$) and $\eta(\tau) = q^{1/24}
\prod_{n=1}^\infty (1 - q^n)$ is the Dedekind eta function.
\end{proposition}

\begin{proof}
The partition function traces over the state space:
$Z_1 = \operatorname{tr}_{\Vlat_\Lambda} q^{L_0 - c/24}$.
The Heisenberg sector contributes $1/\eta(\tau)^d$ (the bosonic
oscillator partition function for $d$ free bosons), and the lattice
sum $\sum_{\alpha \in \Lambda} q^{h_{e^\alpha}}
= \sum_\alpha q^{\langle\alpha,\alpha\rangle/2} = \Theta_\Lambda(\tau)$
accounts for the lattice momentum modes.
\end{proof}

\begin{theorem}[Modular invariance; \ClaimStatusProvedHere]\label{thm:lattice:modular-invariance}
For an even unimodular lattice $\Lambda$ of rank $d$, the partition
function $Z_1(\Vlat_\Lambda, \tau)$ is $S$-invariant:
$Z_1(\Vlat_\Lambda, -1/\tau) = Z_1(\Vlat_\Lambda, \tau)$.
Under $T: \tau \mapsto \tau+1$, the partition function transforms as
$Z_1(\tau+1) = e^{-\pi i d/12}\, Z_1(\tau)$; full
$\mathrm{SL}_2(\Z)$-invariance holds if and only if $d \equiv 0 \pmod{24}$.
\end{theorem}

\begin{proof}
By Poisson summation, $\Theta_\Lambda(-1/\tau) =
(\tau/i)^{d/2} |\Lambda^*/\Lambda|^{1/2}\, \Theta_{\Lambda^*}(\tau)$.
For unimodular $\Lambda = \Lambda^*$, this gives
$\Theta_\Lambda(-1/\tau) = (\tau/i)^{d/2}\, \Theta_\Lambda(\tau)$.
Combined with $\eta(-1/\tau) = \sqrt{\tau/i}\,\eta(\tau)$,
we obtain $Z_1(-1/\tau) = Z_1(\tau)$.

For the $T$-transformation $\tau \mapsto \tau + 1$:
$\Theta_\Lambda(\tau+1) = \Theta_\Lambda(\tau)$ (since
$\langle\alpha,\alpha\rangle \in 2\Z$ for even $\Lambda$) and
$\eta(\tau+1) = e^{\pi i/12}\eta(\tau)$, giving
$Z_1(\tau+1) = e^{-\pi i d/12} Z_1(\tau)$. For $d \equiv 0 \pmod{24}$
this is exact invariance; in general, $Z_1$ transforms as a modular
function of the appropriate level.

For $d = 8$ ($E_8$): the $T$-phase is $e^{-8\pi i/12} = e^{-2\pi i/3} \neq 1$, so
$Z_1(\tau+1) = e^{-2\pi i/3}\, Z_1(\tau)$: the partition function is $S$-invariant
but \emph{not} $T$-invariant. Equivalently,
$Z_1 = E_4(\tau)/\eta(\tau)^8 = j(\tau)^{1/3}$ transforms under $T$ with a
cube-root-of-unity multiplier, making it a modular function for the congruence
subgroup~$\Gamma(3)$ but not for the full~$\mathrm{SL}_2(\Z)$.
Full modular invariance holds only for $d \equiv 0 \pmod{24}$
(e.g., the Leech lattice with $d = 24$).
\end{proof}

\begin{remark}[Higher genus and Siegel modular forms]
\label{rem:lattice:higher-genus}
At genus $g$, the partition function generalizes to
\[
Z_g(\Vlat_\Lambda, \Omega)
= \frac{\Theta_\Lambda^{(g)}(\Omega)}{[\det(\operatorname{Im}\Omega)]^{d/2}}
\cdot (\text{oscillator contribution})
\]
where $\Omega$ is the period matrix, $\Theta_\Lambda^{(g)}$ is the
Siegel theta series, and the denominator involves the determinant of the
imaginary part of $\Omega$. For unimodular even lattices,
$\Theta_\Lambda^{(g)}$ transforms as a Siegel modular form of weight
$d/2$ for $\operatorname{Sp}(2g, \Z)$, as discussed in
Chapter~\ref{ch:genus-expansions}.

The lattice grading on the bar complex
persists at all genera. At genus $g$, the bar differential acquires
corrections from the cohomology of $\overline{\mathcal{M}}_{g,n}$, but these
corrections preserve the total lattice degree. The spectral sequence
computing bar complex homology at genus $g$ has $E_1$ page given by
the genus-0 bar complex (with its lattice decomposition), and the
higher differentials $d_r$ for $r \geq 1$ encode the genus-$g$
quantum corrections studied in Part~\ref{part:bar-complex}.
\end{remark}

\subsection{The Niemeier shadow atlas}%
\label{subsec:lattice:niemeier-atlas}%
\index{Niemeier lattice!shadow atlas|textbf}%
\index{shadow tower!Niemeier atlas}

The twenty-four Niemeier lattices are the even unimodular lattices
in dimension~$24$. They form a finite classification: each is
determined by its root system~$R(\Lambda)$, which satisfies
Niemeier's constraint that all irreducible components have the
same Coxeter number~$h$. The twenty-fourth lattice, with
$R = \varnothing$, is the Leech lattice~$\Lambda_{\mathrm{Leech}}$.

\begin{theorem}[Niemeier shadow universality;
\ClaimStatusProvedHere]%
\label{thm:lattice:niemeier-shadow-universality}%
\index{Niemeier lattice!shadow universality}%
\index{shadow tower!class G universality}%
Let $\Lambda$ be any of the twenty-four Niemeier lattices. Then:
\begin{enumerate}[label=\textup{(\roman*)}]
\item $\kappa(V_\Lambda) = \operatorname{rank}(\Lambda) = 24$
for all twenty-four lattices.
\item $S_r(V_\Lambda) = 0$ for all $r \geq 3$: every Niemeier
lattice VOA is class~$\mathbf{G}$ with shadow depth~$2$.
\item The genus-$g$ scalar amplitude is
$F_g(V_\Lambda) = 24\,\lambda_g^{\mathrm{FP}}$
\textup{(}UNIFORM-WEIGHT\textup{}) for all $g \geq 1$,
identical for all twenty-four lattices. In particular,
\begin{equation}\label{eq:niemeier-genus-amplitudes}
F_1 = 1,\qquad
F_2 = \tfrac{7}{240},\qquad
F_3 = \tfrac{31}{40320},\qquad
F_4 = \tfrac{127}{6451200}.
\end{equation}
\item The planted-forest correction vanishes:
$\delta_{\mathrm{pf}}^{(g,0)}(V_\Lambda) = 0$ for all $g \geq 2$,
since $S_3 = 0$.
\item The complementarity pairing for the lattice VOA family is
the free-field / $\mathrm{Ext}\leftrightarrow\mathrm{Sym}$ pattern
$\kappa_{\mathrm{latt}}(V_\Lambda) + \kappa_{\mathrm{latt}}(V_\Lambda^!) = 0_{\text{(lattice,\\,type\,i)}}$
\textup{(}not the Virasoro value~$13$;
	Theorem~\textup{\ref{thm:complementarity-root-datum}}\textup{)}.
\end{enumerate}
\end{theorem}

\begin{proof}
Part~(i) is Theorem~\ref{thm:lattice:curvature-braiding-orthogonal}
applied to rank-$24$ lattices.
Part~(ii) follows from the Gaussian class characterization:
lattice VOAs are generated by weight-$1$ currents~$J^a(z)$
whose OPE $J^a(z) J^b(w) \sim
\langle a, b \rangle / (z{-}w)^2$ is purely quadratic,
so the shadow obstruction tower terminates at degree~$2$
(Corollary~\ref{cor:lattice-postnikov-termination}).
Part~(iii) uses $\kappa = 24$ and the Faber--Pandharipande
numbers $\lambda_g^{\mathrm{FP}} =
(2^{2g-1}-1) |B_{2g}| / (2^{2g-1} (2g)!)$
(Theorem~\ref{thm:universal-generating-function}).
Part~(iv) follows from the explicit formula
$\delta_{\mathrm{pf}}^{(2,0)} = S_3(10 S_3 - \kappa)/48$
(Proposition~\ref{prop:pixton-genus2-planted-forest})
with $S_3 = 0$.
Part~(v) is the lattice complementarity of
\S\ref{sec:lattice:catalog}.
\end{proof}

The shadow obstruction tower is blind to the root system: all twenty-four
Niemeier lattices produce identical shadow data.
The \emph{distinguishing invariants} live outside the shadow
tower, in the arithmetic structure of the lattice theta series.

\begin{proposition}[Niemeier theta series decomposition;
\ClaimStatusProvedHere]%
\label{prop:lattice:niemeier-theta-decomposition}%
\index{Niemeier lattice!theta series}%
\index{Eisenstein series!E12@$E_{12}$}%
\index{Ramanujan Delta function@Ramanujan $\Delta$ function}%
The genus-$1$ theta series of a Niemeier lattice~$\Lambda$
with $|R(\Lambda)|$ roots decomposes as
\begin{equation}\label{eq:niemeier-theta-decomposition}
\Theta_\Lambda(\tau)
\;=\; E_{12}(\tau) \;+\; c_\Delta(\Lambda)\,\Delta(\tau),
\end{equation}
where $E_{12}$ is the normalized Eisenstein series of weight~$12$,
$\Delta = \eta^{24}$ is the Ramanujan cusp form, and
\begin{equation}\label{eq:niemeier-c-delta}
c_\Delta(\Lambda)
\;=\; \frac{691\,|R(\Lambda)| - 65520}{691}.
\end{equation}
In particular:
\begin{enumerate}[label=\textup{(\roman*)}]
\item The Leech lattice \textup{(}$|R| = 0$\textup{)} has
$c_\Delta = -65520/691$, and
$\Theta_{\mathrm{Leech}}(\tau) = E_{12} - (65520/691)\,\Delta$:
the coefficient of~$q$ is
$65520/691 + (-65520/691) \cdot 1 = 0$
\textup{(}no norm-$2$ vectors\textup{)}.
\item For $D_{24}$ \textup{(}$|R| = 1104$\textup{)}:
$c_\Delta = (691 \cdot 1104 - 65520)/691 = 697344/691$.
\item For $3E_8$ \textup{(}$|R| = 720$\textup{)}:
$c_\Delta = (691 \cdot 720 - 65520)/691 = 432000/691$.
\item For $24A_1$ \textup{(}$|R| = 48$\textup{)}:
$c_\Delta = (691 \cdot 48 - 65520)/691 = -32352/691$.
\end{enumerate}
\end{proposition}

\begin{proof}
The space $M_{12}(\mathrm{SL}_2(\bZ))$ has dimension~$2$, spanned
by $E_{12}$ and~$\Delta$. The constant term of $\Theta_\Lambda$ is
$r_\Lambda(0) = 1$ (the zero vector), which equals
the constant term of $E_{12}$ (normalized to $1$).
Since $\Delta = q - 24q^2 + \cdots$ has zero constant term,
$c_\Delta$ is determined by the coefficient of~$q$:
$r_\Lambda(1) = |R(\Lambda)|$ and
$E_{12} = 1 + (65520/691)\,\sum_{n \geq 1} \sigma_{11}(n)\,q^n$
gives $c_\Delta = (|R(\Lambda)| - 65520\sigma_{11}(1)/691) \cdot
691/\tau(1) = (691\,|R| - 65520)/691$.
\end{proof}

\begin{theorem}[Genus-$2$ Siegel theta series of Niemeier lattices;
\ClaimStatusProvedHere]%
\label{thm:lattice:niemeier-siegel-genus2}%
\index{Siegel modular form!Niemeier lattice}%
\index{Niemeier lattice!genus-2 theta series}%
\index{Igusa ring!weight 12}%
For any Niemeier lattice~$\Lambda$, the genus-$2$ Siegel theta series
$\Theta_\Lambda^{(2)}(\Omega) \in M_{12}(\mathrm{Sp}(4,\bZ))$
satisfies:
\begin{enumerate}[label=\textup{(\roman*)}]
\item $\dim M_{12}(\mathrm{Sp}(4,\bZ)) = 3$ and
$\dim S_{12}(\mathrm{Sp}(4,\bZ)) = 1$
\textup{(}Igusa\textup{)}.
The generators of the even-weight Igusa ring are
$E_4, E_6, \chi_{10}, \chi_{12}$, where $\chi_{10}$
and $\chi_{12}$ are Igusa cusp forms.
The three weight-$12$ monomials are $E_4^3$, $E_6^2$,
and $\chi_{12}$; the unique cusp form is $\chi_{12}$.
\item $\Theta_\Lambda^{(2)} = E_{12}^{(2)} + (\text{Klingen}) +
(\text{cusp})$, where $E_{12}^{(2)}$ is the Siegel Eisenstein
series of weight~$12$.
\item For the Leech lattice: the cusp component is nonzero
\textup{(}Proposition~\textup{\ref{prop:leech-cusp-nonvanishing}}\textup{)},
and the Saito--Kurokawa lift $\chi_{12} = \mathrm{SK}(f_{22})$
appears with nonzero coefficient in the cusp projection.
The B\"ocherer coefficient $B_{f_{22}}(D)$ provides
algebraic access to $L(1/2, \pi_{f_{22}} \times \chi_D)$
\textup{(}Theorem~\textup{\ref{thm:bocherer-bridge}}\textup{)}.
\item For all twenty-four lattices, the scalar shadow amplitude
$F_2 = 7/240$ is identical, but the Siegel theta series
$\Theta_\Lambda^{(2)}$ distinguishes the lattices:
genuinely genus-$2$ arithmetic \textup{(}beyond the shadow
tower\textup{)} separates lattices that the shadow obstruction tower
cannot.
\end{enumerate}
\end{theorem}

\begin{proof}
The Igusa ring dimensions are classical
(Igusa 1962, Aoki--Ibukiyama 2005).
The Siegel--Weil theorem identifies the genus theta series
$\sum_{[\Lambda] \in \mathrm{genus}} |\mathrm{Aut}(\Lambda)|^{-1}
\Theta_\Lambda^{(2)} / \sum |\mathrm{Aut}|^{-1}$
with $E_{12}^{(2)}$. Individual lattice theta series differ
from $E_{12}^{(2)}$ by Klingen Eisenstein and cusp components.
For the Leech lattice,
$r_2(\Lambda_{\mathrm{Leech}}, T_0) = 0$ at
$T_0 = \bigl(\begin{smallmatrix} 1 & 0 \\ 0 & 1
\end{smallmatrix}\bigr)$ while
$a_{E_{12}}(T_0) > 0$ (Kitaoka positivity), giving nonzero
cusp content
(Proposition~\ref{prop:leech-cusp-nonvanishing}).
Part~(iv) follows from Theorem~\ref{thm:lattice:niemeier-shadow-universality}(iii).
\end{proof}

\begin{proposition}[Koszul self-duality criterion; \ClaimStatusProvedHere]\label{prop:lattice:self-dual-criterion}
$\Vlat_\Lambda$ is Koszul self-dual if and only if $\Lambda$ is unimodular (i.e., $\Lambda = \Lambda^*$) and the FLM cocycle $\varepsilon$ is self-inverse up to coboundary. The even unimodular lattices in rank $d$ exist only for $d \equiv 0 \pmod{8}$; the count is $1$ for $d = 8$, $2$ for $d = 16$, $24$ for $d = 24$ (Niemeier lattices), and grows rapidly thereafter.
\end{proposition}

\begin{proof}
The ``only if'' direction: if $\Vlat_\Lambda^! \cong \Vlat_\Lambda$,
then by Theorem~\ref{thm:lattice:koszul-dual},
$\Vlat_{\Lambda}^{\varepsilon^{-1}} \cong \Vlat_\Lambda^\varepsilon$,
which requires $\Lambda^* = \Lambda$ (unimodularity, since the dual
lattice enters via the coalgebra structure) and
$\varepsilon^{-1} \sim \varepsilon$. The ``if'' direction is
Theorem~\ref{thm:lattice:unimodular-self-dual}. The existence
condition $d \equiv 0 \pmod{8}$ for even unimodular lattices is
classical (signature constraints from the Milnor--Husemoller
classification).
\end{proof}

\begin{proposition}[\texorpdfstring{$D_4$}{D4} and triality; \ClaimStatusProvedHere]\label{prop:lattice:D4-triality}
The $D_4$ lattice is \emph{not} unimodular:
$\det(\mathrm{Gram}_{D_4}) = 4$ and $D_4^*/D_4 \cong
(\Z/2\Z)^2$.
In particular, $\Vlat_{D_4}$ is \emph{not} Koszul self-dual;
its Koszul dual is $(\Vlat_{D_4}^{\varepsilon^{-1}})^c$
by Theorem~\ref{thm:lattice:koszul-dual}.

The overlattice $D_4^+ = D_4 \cup (D_4 + s)$ (adding the
spinor class $s = \tfrac{1}{2}(1,1,1,1)$) is unimodular but
\emph{odd}: $\langle s, s\rangle = 1 \notin 2\mathbb{Z}$, so
$D_4^+ \cong \mathbb{Z}^4$ as lattices. Even unimodular
overlattices $D_n^+$ exist only for $n \equiv 0 \pmod{8}$
(e.g., $D_8^+ \cong E_8$). Triality (the $S_3$ symmetry of
the $D_4$ Dynkin diagram) permutes the three nontrivial cosets
in $D_4^*/D_4$ (vector, spinor, and conjugate
spinor) and acts as an outer automorphism of the $D_4$ root
system.
\end{proposition}

\begin{proof}
The discriminant computation $D_4^*/D_4 \cong (\Z/2\Z)^2$ shows $D_4 \neq D_4^*$, so $\Vlat_{D_4}$ is not unimodular and Theorem~\ref{thm:lattice:unimodular-self-dual} does not apply; Theorem~\ref{thm:lattice:koszul-dual} gives $\Vlat_{D_4}^! \cong (\Vlat_{D_4}^{\varepsilon^{-1}})^c \not\cong \Vlat_{D_4}$. The overlattice $D_4^+$ is unimodular ($[D_4^+ : D_4] = 2$ and $\det = 4/4 = 1$) but odd ($\langle s, s\rangle = 1$), so Theorem~\ref{thm:lattice:unimodular-self-dual} does not apply. The triality action on $D_4^*/D_4$ is the standard $S_3$-symmetry permuting the three nontrivial cosets.
\end{proof}

\begin{proposition}[Lattice twisted-sector charge inversion on the bar surface; \ClaimStatusProvedHere]
\label{prop:lattice-module-koszul}
\index{lattice VOA!twisted-sector charge inversion}
Let $\Lambda$ be an even lattice with Koszul dual
$\Vlat_\Lambda^! \cong (\Vlat_\Lambda^{\varepsilon^{-1}})^c$
\textup{(}Theorem~\textup{\ref{thm:lattice:koszul-dual})}. For the
twisted sector $\Vlat_\Lambda^\gamma$ with
$\gamma \in \Lambda^*/\Lambda$, the bar complex records the
discriminant character $\gamma$ through the module-insertion
monodromy, and Verdier reversal sends that charge to its inverse:
\begin{equation}\label{eq:lattice-module-koszul}
\gamma \longmapsto -\gamma
\qquad\text{in } D(\Lambda)=\Lambda^*/\Lambda.
\end{equation}
In particular, inversion permutes the discriminant sectors, so the
number of simple twisted sectors is unchanged.
\end{proposition}

\begin{proof}
The bar construction applied to the twisted module
$\Vlat_\Lambda^\gamma = \mathcal{H} \otimes \mathbb{C}[\gamma + \Lambda]$
produces a comodule over $\bar{B}(\Vlat_\Lambda)$, which is a
coalgebra over the dual lattice data. The
coset $\gamma \in \Lambda^*/\Lambda$ enters the bar complex via
the monodromy: the vertex operator $Y(e^\alpha, z)$ acting on
$\Vlat_\Lambda^\gamma$ has monodromy $z^{\langle\alpha,\gamma\rangle}$
around the module insertion point.

Under the bar construction, this monodromy becomes a
$\langle\alpha,\gamma\rangle$-twist in the differential on
$\overline{C}_{n+1}(X)$. The Verdier involution (which underlies
Koszul duality) reverses the monodromy:
$z^{\langle\alpha,\gamma\rangle} \mapsto
z^{-\langle\alpha,\gamma\rangle}$,
mapping $\gamma$ to $-\gamma$ in the discriminant group.
Since $\gamma \mapsto -\gamma$ is an involution of the finite abelian
group $D(\Lambda)$, it permutes the twisted sectors without changing
their number.
\end{proof}

\begin{remark}[What this does and does not prove]
\label{rem:lattice-module-koszul-scope}
The proposition above proves only the bar-surface charge inversion
$\gamma \mapsto -\gamma$. It does \emph{not} by itself construct a
semisimple pointed module-category equivalence for lattice VOAs, nor
does it prove from Theorem~\ref{thm:monoidal-module-koszul} that
fusion of twisted sectors or the pointed modular $S$-matrix is preserved.
Those stronger statements would require a separate categorical
comparison theorem on the lattice module categories.
\end{remark}

\begin{example}[Charge inversion for the \texorpdfstring{$A_1$}{A1} lattice sectors]
\label{ex:lattice-module-A1}
For $\Lambda = A_1 = \sqrt{2}\,\mathbb{Z}$:
the discriminant group is $D(A_1) = A_1^*/A_1 \cong \mathbb{Z}/2\mathbb{Z}$,
with two simple $\Vlat_{A_1}$-modules:
$\Vlat_{A_1}^0 = \Vlat_{A_1}$ (vacuum) and
$\Vlat_{A_1}^{1/\sqrt{2}}$ (spinor). The bar-surface charge inversion
of Proposition~\ref{prop:lattice-module-koszul} fixes the vacuum sector
and sends the nontrivial class to its inverse, which is the same class:
$-1 \equiv 1 \bmod 2$ in $\mathbb{Z}/2\mathbb{Z}$. The pointed
$S$-matrix
$S = \frac{1}{\sqrt{2}}\bigl(\begin{smallmatrix}1 & 1 \\
1 & -1\end{smallmatrix}\bigr)$
is therefore arithmetically unchanged under this sign involution on the
discriminant group, consistent with the pointed fusion rule
$1/2+1/2=0$ in~$\mathbb{Z}/2\mathbb{Z}$.
\end{example}

\begin{remark}[Distinguishing \texorpdfstring{$E_8 \oplus E_8$}{E8+E8} from \texorpdfstring{$D_{16}^+$}{D16+}]
The bar complexes of $\Vlat_{E_8 \oplus E_8}$ and $\Vlat_{D_{16}^+}$ are both self-dual (both are even unimodular lattices of rank~16), but they are \emph{non-isomorphic} as coalgebras. At genus~1, the two lattices are \emph{indistinguishable}: both have theta function $\Theta_\Lambda(\tau) = E_4(\tau)^2$, since the space~$M_8(\mathrm{SL}_2(\Z))$ is one-dimensional. The distinction first appears at genus~2: the Siegel theta functions $\Theta^{(2)}_{E_8 \oplus E_8}$ and $\Theta^{(2)}_{D_{16}^+}$ are distinct Siegel modular forms of weight~8, sufficing to distinguish the two even unimodular lattices of rank~$16$. This illustrates how the genus-$g$ bar complex captures strictly finer information than the genus-1 bar complex.
\end{remark}

\section{$\Eone$-chiral algebras from cocycle deformations}
\label{sec:lattice:e1-families}

The preceding sections compute bar complexes of lattice algebras with
\emph{symmetric} cocycles, obtaining $\Einf$-chiral algebras.
Theorem~\ref{thm:lattice:e1-vs-einf} guarantees that \emph{any}
2-cocycle produces an $\Eone$-chiral algebra; non-symmetric cocycles
produce \emph{strictly} $\Eone$ structures. The $\Eone$ theory is developed
systematically below: the bar complex moves from
unordered to ordered configurations, the bar cohomology acquires new
classes, and the resulting Koszul pairs provide a rich family of
$\Eone$-chiral algebras complementing the Yangian construction of
Chapter~\ref{chap:yangians}.



\subsection{The ordered configuration bar complex}

\begin{construction}[\texorpdfstring{$\Eone$}{E1} bar complex of \texorpdfstring{$\Vlat_\Lambda^\varepsilon$}{V\_Lambda}]
\label{constr:lattice:e1-bar}
\index{bar complex!$\Eone$-chiral lattice}
\index{ordered configurations!lattice bar complex}
Let $\varepsilon$ be a (not necessarily symmetric) normalized
bimultiplicative 2-cocycle on an even lattice~$\Lambda$.
By Theorem~\ref{thm:lattice:e1-vs-einf}, $\Vlat_\Lambda^\varepsilon$
is $\Eone$-chiral. Its bar complex
$\barB^{\Eone}(\Vlat_\Lambda^\varepsilon)$ is the $\Eone$-chiral bar
construction of Theorem~\ref{thm:e1-chiral-koszul-duality}, indexed by
\emph{ordered} configurations.

\noindent\emph{Generators.} In bar degree~$p$, the generators are
\emph{ordered} tuples:
\[
[e^{\alpha_1} \mid e^{\alpha_2} \mid \cdots \mid e^{\alpha_p}]
\otimes \omega \in \barB^p_\gamma,
\qquad
\gamma = \textstyle\sum_{i=1}^p \alpha_i \in \Lambda,
\]
where $\alpha_i \in \Lambda \setminus \{0\}$ and
$\omega \in \Omega^{p-1}(\overline{C}^{\mathrm{ord}}_p(X), \log D)$
is a logarithmic form on the compactified \emph{ordered} configuration
space. Unlike the $\Einf$ bar complex of
Construction~\ref{constr:lattice:bar-complex}, there is \emph{no}
$\Sigma_p$-quotient: the elements $[e^{\alpha_1} \mid e^{\alpha_2}]$
and $[e^{\alpha_2} \mid e^{\alpha_1}]$ are independent generators.

\noindent\emph{Differential.} The bar differential sums residues over
collision divisors for \emph{adjacent} pairs (the face maps of the
associahedron):
\begin{equation}\label{eq:lattice:e1-bar-diff}
d([e^{\alpha_1} \mid \cdots \mid e^{\alpha_p}] \otimes 1)
= \sum_{i=1}^{p-1}
 \varepsilon(\alpha_i, \alpha_{i+1})\,
 \delta_{\langle\alpha_i,\alpha_{i+1}\rangle, -1}\,
 [e^{\alpha_1} \mid \cdots \mid
 e^{\alpha_i + \alpha_{i+1}} \mid \cdots \mid e^{\alpha_p}]
 \otimes 1.
\end{equation}
The Kronecker delta selects the simple pole condition. The cocycle
$\varepsilon(\alpha_i, \alpha_{i+1})$ enters with its value at the
\emph{specific ordered pair} $(\alpha_i, \alpha_{i+1})$. For a
non-symmetric cocycle, permuting the pair changes this coefficient.
\end{construction}

\begin{remark}[Comparison with the \texorpdfstring{$\Einf$}{E-infinity} bar complex]
\label{rem:e1-vs-einf-bar}
For a symmetric cocycle, the Borcherds symmetry
$\varepsilon(\alpha,\beta) =
(-1)^{\langle\alpha,\beta\rangle}\varepsilon(\beta,\alpha)$
allows one to pass to the $\Sigma_n$-coinvariants, recovering the
$\Einf$ bar complex of Construction~\ref{constr:lattice:bar-complex}
on unordered configurations. In bar degree~2, this identifies
$[e^\alpha \mid e^\beta]$ with
$(-1)^{\langle\alpha,\beta\rangle} [e^\beta \mid e^\alpha]$.

For a non-symmetric cocycle, no such identification is possible: the
ratio $\varepsilon(\alpha,\beta)/\varepsilon(\beta,\alpha) \neq
(-1)^{\langle\alpha,\beta\rangle}$, so the $\Sigma_2$-action
on ordered pairs is not compatible with the differential. The
$\Eone$ bar complex is therefore strictly larger than any symmetric
quotient.
\end{remark}


\subsection{Root-of-unity deformation families}

Specializing the root-of-unity deformation of
Remark~\ref{rem:lattice:root-of-unity-deformation} to construct
explicit families of strictly $\Eone$-chiral lattice algebras.

\begin{definition}[Cocycle deformation]\label{def:lattice:cocycle-deformation}
\index{cocycle!root-of-unity deformation}
Let $\Lambda$ be an even lattice of rank $d \geq 2$, and let
$\varepsilon_0$ be a symmetric cocycle on $\Lambda$. A
\emph{cocycle deformation datum} is a pair $(N, q)$ where $N \geq 3$
is an integer and $q \colon \Lambda \times \Lambda \to \Z/N\Z$ is a
non-zero antisymmetric bilinear form. The
\emph{deformed cocycle} is:
\begin{equation}\label{eq:lattice:deformed-cocycle}
\varepsilon_{N,q}(\alpha, \beta)
:= \varepsilon_0(\alpha, \beta) \cdot
 \zeta_N^{q(\alpha,\beta)},
\qquad \zeta_N := e^{2\pi i/N}.
\end{equation}
\end{definition}

\begin{proposition}[Deformation properties; \ClaimStatusProvedHere]
\label{prop:lattice:deformation-properties}
The deformed cocycle $\varepsilon_{N,q}$ satisfies:
\begin{enumerate}[label=(\roman*)]
\item $\varepsilon_{N,q}$ is a normalized bimultiplicative 2-cocycle
 on~$\Lambda$.
\item The commutator of the deformed cocycle is
 $c_{\varepsilon_{N,q}}(\alpha,\beta)
 = (-1)^{\langle\alpha,\beta\rangle}
 \cdot \zeta_N^{2q(\alpha,\beta)}$.
\item $\varepsilon_{N,q}$ is symmetric in the sense of Borcherds
 if and only if $\zeta_N^{2q(\alpha,\beta)} = 1$ for all
 $\alpha, \beta \in \Lambda$, i.e., if and only if $N \mid 2$
 \textup{(}which contradicts $N \geq 3$\textup{)}. Hence the
 deformed algebra $\Vlat_\Lambda^{\varepsilon_{N,q}}$ is
 \emph{strictly} $\Eone$-chiral for all $N \geq 3$.
\item The inverse cocycle is
 $\varepsilon_{N,q}^{-1} = \varepsilon_{N,-q}$, so the Koszul dual
 \textup{(}Theorem~\textup{\ref{thm:lattice:koszul-dual})} is:
 \begin{equation}\label{eq:lattice:deformed-koszul-dual}
 (\Vlat_\Lambda^{\varepsilon_{N,q}})^!
 = (\Vlat_\Lambda^{\varepsilon_{N,-q}})^c.
 \end{equation}
\end{enumerate}
\end{proposition}

\begin{proof}
(i) Since $\varepsilon_0$ is a 2-cocycle and $\zeta_N^{q(\cdot,\cdot)}$
is a bicharacter ($q$ is bilinear), their product satisfies the
cocycle condition:
\[
\varepsilon_{N,q}(\alpha,\beta)\,
\varepsilon_{N,q}(\alpha+\beta,\gamma)
= \varepsilon_0(\alpha,\beta)\,\varepsilon_0(\alpha+\beta,\gamma)
 \cdot \zeta_N^{q(\alpha,\beta) + q(\alpha+\beta,\gamma)}.
\]
By bilinearity, $q(\alpha,\beta) + q(\alpha+\beta,\gamma)
= q(\alpha,\beta+\gamma) + q(\beta,\gamma)$, so this equals
$\varepsilon_{N,q}(\alpha,\beta+\gamma)\,
\varepsilon_{N,q}(\beta,\gamma)$.

(ii) The commutator is:
\[
c_{\varepsilon_{N,q}}(\alpha,\beta)
= \frac{\varepsilon_{N,q}(\alpha,\beta)}%
 {\varepsilon_{N,q}(\beta,\alpha)}
= \frac{\varepsilon_0(\alpha,\beta)}%
 {\varepsilon_0(\beta,\alpha)}
 \cdot \frac{\zeta_N^{q(\alpha,\beta)}}%
 {\zeta_N^{q(\beta,\alpha)}}
= (-1)^{\langle\alpha,\beta\rangle}
 \cdot \zeta_N^{2q(\alpha,\beta)},
\]
using $c_{\varepsilon_0}(\alpha,\beta) =
(-1)^{\langle\alpha,\beta\rangle}$
(since $\varepsilon_0$ is symmetric) and the antisymmetry
$q(\beta,\alpha) = -q(\alpha,\beta)$.

(iii) The Borcherds symmetry condition requires
$c_{\varepsilon_{N,q}}(\alpha,\beta) =
(-1)^{\langle\alpha,\beta\rangle}$,
hence $\zeta_N^{2q(\alpha,\beta)} = 1$ for all $\alpha, \beta$.
Since $q$ is non-zero, there exist $\alpha, \beta$ with
$q(\alpha,\beta) \not\equiv 0 \pmod{N}$. Then
$\zeta_N^{2q(\alpha,\beta)} = 1$ requires
$2q(\alpha,\beta) \equiv 0 \pmod{N}$. For this to hold for all
$\alpha,\beta$ with $q(\alpha,\beta)$ ranging over $\Z/N\Z$,
we need $2 \cdot 1 \equiv 0 \pmod{N}$, i.e., $N \mid 2$.

(iv) The inverse of $\varepsilon_0(\alpha,\beta) \cdot
\zeta_N^{q(\alpha,\beta)}$ is
$\varepsilon_0(\alpha,\beta)^{-1} \cdot
\zeta_N^{-q(\alpha,\beta)}$. Since $\varepsilon_0$ takes values in
$\{\pm 1\}$ (it is $\{\pm 1\}$-valued for even lattices by the
standard construction), $\varepsilon_0^{-1} = \varepsilon_0$. Hence
$\varepsilon_{N,q}^{-1}(\alpha,\beta) =
\varepsilon_0(\alpha,\beta) \cdot \zeta_N^{-q(\alpha,\beta)}
= \varepsilon_{N,-q}(\alpha,\beta)$.
The Koszul dual identification follows from
Theorem~\ref{thm:lattice:koszul-dual}.
\end{proof}

\begin{remark}[The deformation parameter space]
\label{rem:lattice:deformation-space}
For a rank-$d$ lattice $\Lambda$, the space of antisymmetric
bilinear forms $\Lambda \times \Lambda \to \Z/N\Z$ is
$\bigwedge^2(\Lambda^*/N\Lambda^*) \cong (\Z/N\Z)^{\binom{d}{2}}$.
The number of non-zero forms is $N^{\binom{d}{2}} - 1$. For each
such form, the deformed cocycle gives a distinct $\Eone$-chiral
algebra (distinct in the sense that different non-cohomologous
deformations give non-isomorphic $\Eone$ bar complexes).

For rank 2: the space of antisymmetric forms is $\Z/N\Z$, with
generator $q(e_1, e_2) = 1$. This gives $N - 1$ distinct
deformations for each~$N$.

For rank $d \geq 3$: the space is multi-dimensional, and the
deformations form a $(d-1)$-parameter family.
\end{remark}


\subsection{Explicit $\Eone$ bar complex: the $A_2$ deformation}

The $\Eone$ bar complex is computed for the simplest
non-trivial case: the $A_2$ root lattice with a root-of-unity
cocycle deformation.

\begin{computation}[\texorpdfstring{$\Eone$}{E1} bar complex for deformed \texorpdfstring{$A_2$}{A_2}]
\label{comp:lattice:e1-bar-A2}
\index{bar complex!$\Eone$-chiral $A_2$}
Let $\Lambda = A_2$ with simple roots $\alpha_1, \alpha_2$
satisfying $\langle\alpha_1,\alpha_2\rangle = -1$. Fix $N \geq 3$
and the deformation datum $q(\alpha_1,\alpha_2) = 1$, extended
bilinearly. Set $\zeta = \zeta_N = e^{2\pi i/N}$.

\noindent\emph{Sector $\gamma = \alpha_1 + \alpha_2$.}

\emph{Bar degree~1.} One generator: $[e^{\alpha_1+\alpha_2}]$.

\emph{Bar degree~2.} Two independent generators (ordered pairs):
\[
v_+ := [e^{\alpha_1} \mid e^{\alpha_2}],
\qquad
v_- := [e^{\alpha_2} \mid e^{\alpha_1}].
\]
In the $\Einf$ bar complex, the symmetry relation identifies
$v_-$ with $(-1)^{\langle\alpha_1,\alpha_2\rangle} v_+ = -v_+$,
collapsing these to one generator. In the $\Eone$ complex,
both are independent.

\emph{Differential.}
By~\eqref{eq:lattice:e1-bar-diff}:
\begin{align*}
d(v_+)
&= \varepsilon_{N,q}(\alpha_1, \alpha_2)\,
 [e^{\alpha_1+\alpha_2}]
 = \varepsilon_0(\alpha_1,\alpha_2) \cdot \zeta \cdot
 [e^{\alpha_1+\alpha_2}], \\
d(v_-)
&= \varepsilon_{N,q}(\alpha_2, \alpha_1)\,
 [e^{\alpha_1+\alpha_2}]
 = \varepsilon_0(\alpha_2,\alpha_1) \cdot \zeta^{-1} \cdot
 [e^{\alpha_1+\alpha_2}].
\end{align*}

The differential $d \colon \C^2 \to \C$ is the row vector
$[\varepsilon_0(\alpha_1,\alpha_2)\zeta,\;
\varepsilon_0(\alpha_2,\alpha_1)\zeta^{-1}]$.
Since at least one coefficient is nonzero, $d$ is surjective and
$\dim\ker d = 1$.

\emph{The ordering cycle.}
The kernel of $d$ is spanned by the
\emph{ordering cycle}:
\begin{equation}\label{eq:lattice:ordering-cycle}
\xi_{\alpha_1,\alpha_2}
:= \varepsilon_0(\alpha_2,\alpha_1)\,\zeta^{-1}\,
 [e^{\alpha_1} \mid e^{\alpha_2}]
 - \varepsilon_0(\alpha_1,\alpha_2)\,\zeta\,
 [e^{\alpha_2} \mid e^{\alpha_1}].
\end{equation}
This cycle is \emph{not exact} (there are no bar degree~3 elements
in this sector that could hit it, since any element
$[e^{\beta_1} \mid e^{\beta_2} \mid e^{\beta_3}]$ with
$\beta_1 + \beta_2 + \beta_3 = \alpha_1 + \alpha_2$ must have at
least one $\beta_i$ not a root, and non-root lattice elements do not
generate simple poles with roots).

\emph{Conclusion.} In the sector $\gamma = \alpha_1 + \alpha_2$:
\begin{align*}
H^1(\barB^{\Eone}_\gamma) &= 0, \\
H^2(\barB^{\Eone}_\gamma) &= \C \cdot \xi_{\alpha_1,\alpha_2}.
\end{align*}

Compare with the $\Einf$ computation
(Computation~\ref{comp:lattice:bar-A2}):
there, $\barB^2_\gamma \cong \C$ and $d$ is an isomorphism, giving
$H^1 = H^2 = 0$. The $\Eone$ bar complex has \emph{strictly more
cohomology} than the $\Einf$ bar complex: the ordering cycle
$\xi_{\alpha_1,\alpha_2}$ is a new class that measures the
non-commutativity of the cocycle.
\end{computation}

\begin{proposition}[Phase dependence of the ordering cycle;
\ClaimStatusProvedHere]
\label{prop:lattice:ordering-cycle-phase}
The ordering cycle $\xi_{\alpha_1,\alpha_2}$ depends on the
deformation parameter as follows.
\begin{enumerate}[label=(\roman*)]
\item The ratio of its two coefficients is:
\begin{equation}\label{eq:lattice:cycle-ratio}
\frac{\varepsilon_0(\alpha_2,\alpha_1)\,\zeta^{-1}}%
 {-\varepsilon_0(\alpha_1,\alpha_2)\,\zeta}
= \frac{(-1)^{\langle\alpha_1,\alpha_2\rangle}}{\zeta^2}
= \frac{-1}{\zeta^2}
= -\zeta_N^{-2}.
\end{equation}
\item At $N = 3$: $\zeta^2 = e^{4\pi i/3}$, and the ratio is
 $-e^{-4\pi i/3} = e^{-\pi i/3}$, a primitive 6th root of unity.
\item At $N = 4$: $\zeta^2 = i$, and the ratio is $-(-i) = i$.
\item In the limit $N \to \infty$ \textup{(}$\zeta \to 1$\textup{)},
 the ratio approaches $-1$ and the cycle becomes
 \[
 \varepsilon_0(\alpha_2,\alpha_1)[e^{\alpha_1}|e^{\alpha_2}]
 - \varepsilon_0(\alpha_1,\alpha_2)[e^{\alpha_2}|e^{\alpha_1}],
 \]
 the antisymmetric combination that maps to
 zero under the $\Sigma_2$-coinvariant quotient.
 This reflects the $\Einf$ limit: the new cycle collapses to a
 trivial class upon symmetrization.
\end{enumerate}
\end{proposition}

\begin{proof}
Direct computation from~\eqref{eq:lattice:ordering-cycle}, using
$\varepsilon_0(\alpha_1,\alpha_2)/\varepsilon_0(\alpha_2,\alpha_1)
= (-1)^{\langle\alpha_1,\alpha_2\rangle} = -1$.
\end{proof}


\subsection{Explicit specializations}
\label{sec:lattice:explicit-e1}

The $\Eone$ bar complex computation in complete
constructive detail for three explicit deformations producing strictly
$\Eone$-chiral algebras. Each example is computed end-to-end: cocycle
values, OPE phases, bar differential, ordering cycles, bar cohomology,
Koszul dual identification, and braiding torsion. All numerical values
are verified by the computational suite
(\texttt{compute/lib/e1\_lattice\_bar.py}, 45 tests passing).

Throughout, we fix the symmetric cocycle $\varepsilon_0$ using the
\emph{Frenkel--Lepowsky--Meurman upper-triangular convention}: for
the ordered basis $\alpha_1 < \alpha_2 < \cdots < \alpha_r$ of
simple roots,
\begin{equation}\label{eq:lattice:flm-cocycle}
\varepsilon_0(\alpha_i, \alpha_j)
=
\begin{cases}
(-1)^{\langle\alpha_i,\alpha_j\rangle} & \text{if } i < j
 \text{ and $\alpha_i, \alpha_j$ adjacent}, \\
1 & \text{otherwise}.
\end{cases}
\end{equation}
This satisfies Borcherds symmetry:
$\varepsilon_0(\alpha_i,\alpha_j)/\varepsilon_0(\alpha_j,\alpha_i)
= (-1)^{\langle\alpha_i,\alpha_j\rangle}$ for all $i \neq j$.


\begin{computation}[\texorpdfstring{$A_2$}{A_2} at the Coxeter root: \texorpdfstring{$N = 3$}{N = 3}, \texorpdfstring{$q = 1$}{q = 1}]
\label{comp:lattice:a2-n3}
\index{bar complex!$\Eone$-chiral $A_2$ at $N=3$}

\noindent\textbf{Setup.}
Let $\Lambda = A_2$ with simple roots
$\alpha_1, \alpha_2$ satisfying
$\langle\alpha_1,\alpha_2\rangle = -1$, $\langle\alpha_i,\alpha_i\rangle = 2$.
The Cartan matrix is
$A = \left(\begin{smallmatrix}2 & -1\\-1 & 2\end{smallmatrix}\right)$
with determinant~$3$.
The root system $\Phi(A_2)$ has 3 positive roots:
$\alpha_1$, $\alpha_2$, $\alpha_1 + \alpha_2$.

Take $N = 3$ (the Coxeter number of $A_2$), deformation
$q(\alpha_1,\alpha_2) = 1$, and set $\zeta = \zeta_3 = e^{2\pi i/3}$.
\smallskip

\noindent\textbf{Step 1: Cocycle values.}
By~\eqref{eq:lattice:flm-cocycle}:
$\varepsilon_0(\alpha_1,\alpha_2) = (-1)^{-1} = -1$,
$\varepsilon_0(\alpha_2,\alpha_1) = 1$.
The deformed cocycle~\eqref{eq:lattice:deformed-cocycle}
evaluates to:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{c c c c}
$(\alpha_i,\alpha_j)$
 & $\varepsilon_0(\alpha_i,\alpha_j)$
 & $\zeta_3^{q(\alpha_i,\alpha_j)}$
 & $\varepsilon_{3,1}(\alpha_i,\alpha_j)$ \\
\hline
$(\alpha_1,\alpha_2)$
 & $-1$ & $\zeta$ & $-\zeta$ \\
$(\alpha_2,\alpha_1)$
 & $+1$ & $\zeta^{-1} = \zeta^2$ & $\zeta^2$
\end{tabular}
\end{center}
In coordinates, $-\zeta = \tfrac{1}{2} - \tfrac{\sqrt{3}}{2}\,i$
and $\zeta^2 = -\tfrac{1}{2} - \tfrac{\sqrt{3}}{2}\,i$.
\smallskip

\noindent\textbf{Step 2: Failure of Borcherds symmetry.}
Borcherds symmetry requires
$\varepsilon(\alpha_1,\alpha_2)/\varepsilon(\alpha_2,\alpha_1)
= (-1)^{\langle\alpha_1,\alpha_2\rangle} = -1$.
We compute:
\[
\frac{\varepsilon_{3,1}(\alpha_1,\alpha_2)}%
 {\varepsilon_{3,1}(\alpha_2,\alpha_1)}
= \frac{-\zeta}{\zeta^2}
= \frac{-1}{\zeta}
= -\zeta^2
\neq -1.
\]
Since $\zeta^2 \neq 1$ for $N = 3$, Borcherds symmetry fails and
$\Vlat_{A_2}^{3,1}$ is strictly $\Eone$-chiral, not $\Einf$.
\smallskip

\noindent\textbf{Step 3: Cocycle commutator (braiding).}
By
Proposition~\ref{prop:lattice:deformation-properties}(ii):
\[
c_{\varepsilon_{3,1}}(\alpha_1,\alpha_2)
= (-1)^{\langle\alpha_1,\alpha_2\rangle} \cdot \zeta_3^{2q(\alpha_1,\alpha_2)}
= (-1)^{-1} \cdot \zeta^2
= -\zeta^2
= \tfrac{1}{2} + \tfrac{\sqrt{3}}{2}\,i.
\]
The conjugate braiding is
$c(\alpha_2,\alpha_1) = -\zeta = \tfrac{1}{2} - \tfrac{\sqrt{3}}{2}\,i$.
\smallskip

\noindent\emph{Verification.} The product
$c(\alpha_1,\alpha_2) \cdot c(\alpha_2,\alpha_1)
= (-\zeta^2)(-\zeta) = \zeta^3 = 1$, confirming that the braiding
on the $A_2$ lattice is consistent with an even lattice.
The diagonal entries
$c(\alpha_i,\alpha_i) = (-1)^{2}\zeta^{0} = 1$
are trivial (antisymmetry of~$q$).
\smallskip

\noindent\textbf{Step 4: OPE with phases.}
The root vertex operators
$E_{\alpha_i}(z) = Y(e^{\alpha_i}, z)$ have OPEs:
\begin{align*}
E_{\alpha_1}(z)\, E_{\alpha_2}(w)
&= \frac{\varepsilon_{3,1}(\alpha_1,\alpha_2)}{z - w}
 \,E_{\alpha_1+\alpha_2}(w) + \mathrm{reg.}
= \frac{-\zeta}{z - w}\,E_{\alpha_1+\alpha_2}(w) + \mathrm{reg.}, \\
E_{\alpha_2}(z)\, E_{\alpha_1}(w)
&= \frac{\varepsilon_{3,1}(\alpha_2,\alpha_1)}{z - w}
 \,E_{\alpha_1+\alpha_2}(w) + \mathrm{reg.}
= \frac{\zeta^2}{z - w}\,E_{\alpha_1+\alpha_2}(w) + \mathrm{reg.}
\end{align*}
The ratio of leading OPE coefficients is
$(-\zeta)/\zeta^2 = -\zeta^2$, the cocycle commutator: the OPE is
\emph{non-commutative} by a cube root of unity phase.
\smallskip

\noindent\textbf{Step 5: $\Eone$ bar complex in sector
$\gamma = \alpha_1 + \alpha_2$.}

\emph{Bar degree~1.}
One generator: $[e^{\gamma}] = [e^{\alpha_1+\alpha_2}]$.

\emph{Bar degree~2.}
Two independent generators (ordered tuples):
$v_+ = [e^{\alpha_1} \mid e^{\alpha_2}]$,\;
$v_- = [e^{\alpha_2} \mid e^{\alpha_1}]$.

\emph{Differential.}
By~\eqref{eq:lattice:e1-bar-diff}, since
$\langle\alpha_1,\alpha_2\rangle = -1$:
\[
d =
\begin{pmatrix}
\varepsilon_{3,1}(\alpha_1,\alpha_2) &
\varepsilon_{3,1}(\alpha_2,\alpha_1)
\end{pmatrix}
=
\begin{pmatrix} -\zeta & \zeta^2 \end{pmatrix}
\colon \C^2 \longrightarrow \C.
\]
Both entries are nonzero, so $\operatorname{rank}(d) = 1$
and $\dim\ker(d) = 1$.
\smallskip

\noindent\textbf{Step 6: The ordering cycle.}
The kernel $\ker(d)$ is spanned by:
\begin{equation}\label{eq:lattice:a2-n3-cycle}
\xi = v_+ + \zeta^2 \cdot v_-
= [e^{\alpha_1} \mid e^{\alpha_2}]
 + \zeta^2\,[e^{\alpha_2} \mid e^{\alpha_1}].
\end{equation}
\emph{Verification:}
$d(\xi) = (-\zeta) \cdot [e^\gamma]
 + \zeta^2 \cdot \zeta^2 \cdot [e^\gamma]
= (-\zeta + \zeta^4)\,[e^\gamma]
= (-\zeta + \zeta)\,[e^\gamma]
= 0$,
using $\zeta^4 = \zeta^3 \cdot \zeta = \zeta$.

This cycle is \emph{not a boundary}: bar degree~3 in this sector
requires a triple $[e^{\beta_1} \mid e^{\beta_2} \mid e^{\beta_3}]$
with $\beta_1 + \beta_2 + \beta_3 = \alpha_1 + \alpha_2$
and consecutive inner products $-1$, which does not exist.
\smallskip

\noindent\textbf{Step 7: Bar cohomology.}
In the sector $\gamma = \alpha_1 + \alpha_2$:
\[
H^1(\barB^{\Eone}_\gamma) = 0, \qquad
H^2(\barB^{\Eone}_\gamma) = \C \cdot \xi.
\]
\emph{Comparison with $\Einf$:}
In the $\Einf$ bar complex, the symmetric quotient identifies
$v_- = -v_+$ (the Borcherds relation forces
$[e^{\alpha_2} | e^{\alpha_1}]
= (-1)^{\langle\alpha_1,\alpha_2\rangle}[e^{\alpha_1} | e^{\alpha_2}]
= -v_+$), leaving one generator in degree~2.
The differential $d_{\Einf} \colon \C^1 \to \C^1$ is an isomorphism,
so $H^1_{\Einf} = H^2_{\Einf} = 0$.

The ordering cycle $\xi$ is the generator of the \emph{new}
$\Eone$ cohomology, invisible to the $\Einf$ theory.
In the symmetric quotient, $\xi = v_+ + \zeta^2 v_-
\mapsto v_+ + \zeta^2(-v_+) = (1 - \zeta^2)v_+$;
this is nonzero but lands in the image of $d_{\Einf}$
(since $d_{\Einf}$ is surjective onto degree~2).
\smallskip

\noindent\textbf{Step 8: Koszul dual.}
The Koszul dual is $(\Vlat_{A_2}^{3,1})^! = (\Vlat_{A_2}^{3,-1})^c
= (\Vlat_{A_2}^{3,2})^c$, carrying the conjugate deformation
$-q \equiv 2 \pmod{3}$: the phase $\zeta$ is replaced by $\zeta^2$.
The dual cocycle commutator is
$c_{-q}(\alpha_i,\alpha_j) = c_q(\alpha_j,\alpha_i)$
(transposition), so:
\[
c_{\varepsilon_{3,2}}(\alpha_1,\alpha_2)
= c_{\varepsilon_{3,1}}(\alpha_2,\alpha_1) = -\zeta.
\]
The product $c_q \cdot c_{-q} = (-\zeta^2)(-\zeta) = 1$: original and
dual braidings are mutual inverses. Since
$q = 1 \not\equiv -1 = 2 \pmod{3}$, the algebra is \emph{not}
Koszul self-dual.
\smallskip

\noindent\textbf{Step 9: $N$-torsion.}
$c(\alpha_1,\alpha_2)^3 = (-\zeta^2)^3 = -\zeta^6 = -1
= (-1)^{3\langle\alpha_1,\alpha_2\rangle} = (-1)^{-3} = -1$.\;$\checkmark$

The 3-fold braiding iteration is trivial (up to the universal sign):
the deformation has \emph{exact} order~$3$ on the braiding.
\end{computation}



\begin{computation}[\texorpdfstring{$A_2$}{A_2} with non-standard parameter: \texorpdfstring{$N = 5$}{N = 5}, \texorpdfstring{$q = 2$}{q = 2}]
\label{comp:lattice:a2-n5-q2}
\index{bar complex!$\Eone$-chiral $A_2$ at $N=5$}

\noindent\textbf{Setup.}
Same lattice $\Lambda = A_2$, now with $N = 5$ and
$q(\alpha_1,\alpha_2) = 2$ (not a generator of $\Z/5\Z$
in the modular sense: $\gcd(2,5) = 1$, but $2 \neq 1$).
Set $\zeta = \zeta_5 = e^{2\pi i/5}$.
\smallskip

\noindent\textbf{Cocycle values.}
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{c c c c}
$(\alpha_i,\alpha_j)$
 & $\varepsilon_0(\alpha_i,\alpha_j)$
 & $\zeta_5^{q(\alpha_i,\alpha_j)}$
 & $\varepsilon_{5,2}(\alpha_i,\alpha_j)$ \\
\hline
$(\alpha_1,\alpha_2)$
 & $-1$ & $\zeta^2$ & $-\zeta^2$ \\
$(\alpha_2,\alpha_1)$
 & $+1$ & $\zeta^{-2} = \zeta^3$ & $\zeta^3$
\end{tabular}
\end{center}
\smallskip

\noindent\textbf{Borcherds symmetry failure.}
$\varepsilon_{5,2}(\alpha_1,\alpha_2)/\varepsilon_{5,2}(\alpha_2,\alpha_1)
= -\zeta^2/\zeta^3 = -\zeta^{-1} = -\zeta^4 \neq -1$.
\smallskip

\noindent\textbf{Cocycle commutator.}
$c(\alpha_1,\alpha_2) = (-1)^{-1} \cdot \zeta^{2 \cdot 2} = -\zeta^4$,\quad
$c(\alpha_2,\alpha_1) = -\zeta^{-4} = -\zeta$.

\noindent\emph{Product check:}
$c(\alpha_1,\alpha_2) \cdot c(\alpha_2,\alpha_1)
= (-\zeta^4)(-\zeta) = \zeta^5 = 1$.\;$\checkmark$
\smallskip

\noindent\textbf{OPE coefficients.}
\begin{align*}
E_{\alpha_1}(z)\, E_{\alpha_2}(w)
&= \frac{-\zeta^2}{z - w}\,E_{\alpha_1+\alpha_2}(w) + \mathrm{reg.}, \\
E_{\alpha_2}(z)\, E_{\alpha_1}(w)
&= \frac{\zeta^3}{z - w}\,E_{\alpha_1+\alpha_2}(w) + \mathrm{reg.}
\end{align*}
\smallskip

\noindent\textbf{$\Eone$ bar differential in sector $\gamma = \alpha_1 + \alpha_2$.}
\[
d =
\begin{pmatrix} -\zeta^2 & \zeta^3 \end{pmatrix}
\colon \C^2 \longrightarrow \C, \qquad
\operatorname{rank}(d) = 1.
\]
\textbf{Ordering cycle.}
\begin{equation}\label{eq:lattice:a2-n5-cycle}
\xi = v_+ + \zeta^4 \cdot v_-
= [e^{\alpha_1} \mid e^{\alpha_2}]
 + \zeta^4\,[e^{\alpha_2} \mid e^{\alpha_1}].
\end{equation}
The cycle ratio $\zeta^4 = \zeta^{-1}$ has the \emph{same structure}
as the $N = 3$ case (ratio $= \zeta_N^{-1}$ in both cases): the
ordering cycle coefficient is always the
\emph{inverse} of the primitive root.

\emph{Verification:}
$d(\xi)
= (-\zeta^2) \cdot [e^\gamma] + \zeta^4 \cdot \zeta^3 \cdot [e^\gamma]
= (-\zeta^2 + \zeta^7) \cdot [e^\gamma]
= (-\zeta^2 + \zeta^2) \cdot [e^\gamma] = 0$,
using $\zeta^7 = \zeta^5 \cdot \zeta^2 = \zeta^2$.

\noindent\textbf{Bar cohomology.}
$H^2(\barB^{\Eone}_\gamma) = \C \cdot \xi$,\;
$H^2(\barB^{\Einf}_\gamma) = 0$.
\smallskip

\noindent\textbf{Koszul dual.}
Dual form: $-q \equiv 3 \pmod{5}$, giving
$(\Vlat_{A_2}^{5,2})^! = (\Vlat_{A_2}^{5,3})^c$ with
dual braiding $c_{\varepsilon_{5,3}}(\alpha_1,\alpha_2) = -\zeta$
(the conjugate braiding). Not self-dual: $2 \neq 3 \pmod{5}$.
\smallskip

\noindent\textbf{Contrasting structure.}
Both $A_2$ examples produce strictly $\Eone$ algebras with one
ordering cycle per adjacent pair, but the \emph{braiding phases differ}:
$-\zeta_3^2$ at $N = 3$ vs.\ $-\zeta_5^4$ at $N = 5$.
At $N = 3$, the braiding has order~$6$ (since $(-\zeta_3^2)^6 = 1$
and no smaller power works).
At $N = 5$ with $q = 2$, the braiding has order~$10$
(since $(-\zeta_5^4)^{10} = 1$).
These are genuinely distinct $\Eone$-chiral structures on the
same underlying lattice.
\end{computation}



\begin{computation}[\texorpdfstring{$D_4$}{D_4} at the Coxeter root: \texorpdfstring{$N = 3$}{N = 3}, triality]
\label{comp:lattice:d4-n3}
\index{bar complex!$\Eone$-chiral $D_4$ at $N=3$}
\index{triality!$\Eone$ lattice algebra}

\noindent\textbf{Setup.}
Let $\Lambda = D_4$ with simple roots $\alpha_1, \alpha_2, \alpha_3,
\alpha_4$ and Cartan matrix:
\[
A_{D_4} =
\begin{pmatrix}
 2 & -1 & 0 & 0 \\
-1 & 2 & -1 & -1 \\
 0 & -1 & 2 & 0 \\
 0 & -1 & 0 & 2
\end{pmatrix},
\qquad
\det(A_{D_4}) = 4.
\]
The Dynkin diagram is a trivalent tree: node $\alpha_2$ (central)
is adjacent to $\alpha_1$, $\alpha_3$, $\alpha_4$. The outer
automorphism group $S_3$ permutes the three legs (triality).
The root system has 12 positive roots.

Take $N = 3$ and the \emph{standard edge form}:
$q(\alpha_i,\alpha_j) = 1$ for each adjacent pair $(i < j)$:
\[
q = \begin{pmatrix}
0 & 1 & 0 & 0 \\
2 & 0 & 1 & 1 \\
0 & 2 & 0 & 0 \\
0 & 2 & 0 & 0
\end{pmatrix}
\quad (\text{entries mod } 3),
\]
where $q(j,i) = 3 - q(i,j) \equiv -q(i,j) \pmod{3}$. This form
is triality-invariant: permuting
$(\alpha_1, \alpha_3, \alpha_4)$ preserves $q$
(all three edges carry $q = 1$).
Set $\zeta = \zeta_3 = e^{2\pi i/3}$.
\smallskip

\noindent\textbf{Step 1: Cocycle values.}
By~\eqref{eq:lattice:flm-cocycle}: $\varepsilon_0$ is $-1$ on adjacent
pairs $(i < j)$ and $+1$ elsewhere.
The deformed cocycle evaluates to:
\begin{center}
\renewcommand{\arraystretch}{1.3}
{\small
\begin{tabular}{c c c c c}
$(\alpha_i,\alpha_j)$ & Adjacency
 & $\varepsilon_0$ & $\zeta^{q}$ & $\varepsilon_{3,q}$ \\
\hline
$(\alpha_1,\alpha_2)$ & adj.\ & $-1$ & $\zeta$ & $-\zeta$ \\
$(\alpha_2,\alpha_1)$ & adj.\ & $+1$ & $\zeta^2$ & $\zeta^2$ \\
$(\alpha_2,\alpha_3)$ & adj.\ & $-1$ & $\zeta$ & $-\zeta$ \\
$(\alpha_3,\alpha_2)$ & adj.\ & $+1$ & $\zeta^2$ & $\zeta^2$ \\
$(\alpha_2,\alpha_4)$ & adj.\ & $-1$ & $\zeta$ & $-\zeta$ \\
$(\alpha_4,\alpha_2)$ & adj.\ & $+1$ & $\zeta^2$ & $\zeta^2$ \\
\hline
$(\alpha_1,\alpha_3)$ & non-adj.\ & $+1$ & $1$ & $1$ \\
$(\alpha_1,\alpha_4)$ & non-adj.\ & $+1$ & $1$ & $1$ \\
$(\alpha_3,\alpha_4)$ & non-adj.\ & $+1$ & $1$ & $1$
\end{tabular}
}
\end{center}
All three adjacent edges carry the \emph{same} deformed cocycle values
$(-\zeta, \zeta^2)$; this is the triality symmetry of the edge form.
Non-adjacent pairs are undeformed (both $\varepsilon_0$ and $\zeta^q$
equal~$1$).
\smallskip

\noindent\textbf{Step 2: Cocycle commutator (braiding matrix).}
By Proposition~\ref{prop:lattice:deformation-properties}(ii),
$c(\alpha_i,\alpha_j) = (-1)^{\langle\alpha_i,\alpha_j\rangle}
\cdot \zeta_3^{2q(\alpha_i,\alpha_j)}$:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{c c c}
Pair type & $c(\alpha_i,\alpha_j)$ & Value \\
\hline
Adjacent $(i < j)$ & $(-1)^{-1} \cdot \zeta^2 = -\zeta^2$
 & $\tfrac{1}{2} + \tfrac{\sqrt{3}}{2}\,i$ \\
Adjacent $(i > j)$ & $(-1)^{-1} \cdot \zeta^{-2} = -\zeta$
 & $\tfrac{1}{2} - \tfrac{\sqrt{3}}{2}\,i$ \\
Non-adjacent & $(-1)^0 \cdot \zeta^0 = 1$
 & $1$ \\
Diagonal & $(-1)^2 \cdot \zeta^0 = 1$
 & $1$
\end{tabular}
\end{center}
The braiding is \emph{uniform} across all three edges of the $D_4$
diagram (triality symmetry) and \emph{trivial} for non-adjacent pairs.
The full $4 \times 4$ commutator matrix $c(\alpha_i,\alpha_j)$ is:
\[
\begin{pmatrix}
1 & -\zeta^2 & 1 & 1 \\
-\zeta & 1 & -\zeta^2 & -\zeta^2 \\
1 & -\zeta & 1 & 1 \\
1 & -\zeta & 1 & 1
\end{pmatrix}.
\]
The last two rows and columns are identical, reflecting the $S_3$
triality acting on legs $\{\alpha_1, \alpha_3, \alpha_4\}$.
\smallskip

\noindent\textbf{Step 3: $\Eone$ bar complex, three sectors.}
The sectors with simple poles are the three adjacent pairs:
$\gamma_1 = \alpha_1 + \alpha_2$,
$\gamma_2 = \alpha_2 + \alpha_3$,
$\gamma_3 = \alpha_2 + \alpha_4$.
Each sector independently contributes a 2-step complex
$0 \to \C^2 \xrightarrow{d} \C \to 0$.

\emph{Sector $\gamma_1 = \alpha_1 + \alpha_2$:}
\[
d_{\gamma_1}
= \begin{pmatrix} -\zeta & \zeta^2 \end{pmatrix},
\qquad
\xi_{\gamma_1}
= [e^{\alpha_1} \mid e^{\alpha_2}]
 + \zeta^2\,[e^{\alpha_2} \mid e^{\alpha_1}].
\]

\emph{Sector $\gamma_2 = \alpha_2 + \alpha_3$:}
\[
d_{\gamma_2}
= \begin{pmatrix} -\zeta & \zeta^2 \end{pmatrix},
\qquad
\xi_{\gamma_2}
= [e^{\alpha_2} \mid e^{\alpha_3}]
 + \zeta^2\,[e^{\alpha_3} \mid e^{\alpha_2}].
\]

\emph{Sector $\gamma_3 = \alpha_2 + \alpha_4$:}
\[
d_{\gamma_3}
= \begin{pmatrix} -\zeta & \zeta^2 \end{pmatrix},
\qquad
\xi_{\gamma_3}
= [e^{\alpha_2} \mid e^{\alpha_4}]
 + \zeta^2\,[e^{\alpha_4} \mid e^{\alpha_2}].
\]
All three differentials are identical (same cocycle values), and all
three ordering cycles have the same coefficient
$\zeta^2 = \zeta_3^{-1}$. This uniform structure is a direct
consequence of the triality-invariant edge form.
\smallskip

\noindent\textbf{Step 4: Bar cohomology.}
\[
\bigoplus_{\gamma \in \{\gamma_1,\gamma_2,\gamma_3\}}
H^2(\barB^{\Eone}_\gamma)
= \C \cdot \xi_{\gamma_1}
 \oplus \C \cdot \xi_{\gamma_2}
 \oplus \C \cdot \xi_{\gamma_3}
\cong \C^3.
\]
Each sector contributes one ordering cycle. These three cycles are
linearly independent (they live in distinct lattice-graded components).

For comparison, the $\Einf$ bar complex has $H^2 = 0$ in every
sector: the triality-worth of new cohomology is purely an $\Eone$
phenomenon. The 3 ordering cycles span the
\emph{$S_3$-representation} $\C^3$ on which triality acts by
permutation (the standard representation decomposes as
$\C^3 = \C \oplus V_{\mathrm{std}}$ under $S_3$).
\smallskip

\noindent\textbf{Step 5: Koszul dual.}
The dual form $-q \equiv 2 \pmod{3}$ replaces $q(\alpha_i,\alpha_j) = 1$
by $q(\alpha_i,\alpha_j) = 2$ on each edge. The dual braiding is:
\[
c_{\varepsilon_{3,-q}}(\alpha_i,\alpha_j) = c_{\varepsilon_{3,q}}(\alpha_j,\alpha_i)
\quad\text{(transpose)},
\]
so
$c_{\varepsilon_{3,2}}(\alpha_1,\alpha_2) = -\zeta$
(compared with $c_{\varepsilon_{3,1}}(\alpha_1,\alpha_2) = -\zeta^2$).
The Koszul dual is $(\Vlat_{D_4}^{3,1})^! = (\Vlat_{D_4}^{3,2})^c$,
and the product identity
$c_q(\alpha_i,\alpha_j) \cdot c_{-q}(\alpha_i,\alpha_j) = 1$
holds for every pair. Not self-dual.
\smallskip

\noindent\textbf{Step 6: $N$-torsion and screening structure.}
For adjacent pairs: $c(\alpha_i,\alpha_j)^3 = (-\zeta^2)^3 = -1
= (-1)^{3 \cdot (-1)} = (-1)^{-3} = -1$.\;$\checkmark$

For non-adjacent pairs: $c(\alpha_i,\alpha_j)^3 = 1^3 = 1
= (-1)^{3 \cdot 0} = 1$.\;$\checkmark$

The screening operators $Q_i = \oint E_{\alpha_i}(z)\,dz$
($i = 1, \ldots, 4$) satisfy:
\begin{itemize}
\item $Q_i^2 = 0$ for all $i$ (nilpotency from
 $\langle\alpha_i,\alpha_i\rangle = 2$).
\item $Q_1 Q_3 = Q_3 Q_1$, $Q_1 Q_4 = Q_4 Q_1$,
 $Q_3 Q_4 = Q_4 Q_3$ (non-adjacent commute, braiding $= 1$).
\item $Q_1 Q_2 = (-\zeta^2)\, Q_2 Q_1$,
 $Q_2 Q_3 = (-\zeta^2)^{-1}\, Q_3 Q_2$,
 $Q_2 Q_4 = (-\zeta^2)^{-1}\, Q_4 Q_2$
 (adjacent pairs braid by $c$).
\end{itemize}
The screening algebra structure at the central node $\alpha_2$ is that
of a braided tensor product of three nilpotent copies $(\C[x]/x^2)$,
braided by the $D_4$ Nichols algebra datum: this is the same Nichols
algebra as for the small quantum group $u_{\zeta_3}(\mathfrak{so}_8)$
at a third root of unity
(cf.\ Remark~\ref{rem:lattice:qg-programme}, Route~A).
\end{computation}

\begin{remark}[Pattern across examples]
\label{rem:lattice:e1-pattern}
The three computations exhibit a universal structure:
\begin{enumerate}[label=(\roman*)]
\item In each adjacent-pair sector $\gamma = \alpha_i + \alpha_j$,
the $\Eone$ bar complex $0 \to \C^2 \xrightarrow{d} \C \to 0$ has
$\dim\ker(d) = 1$, producing exactly one ordering cycle.
\item The ordering cycle coefficient is always $\zeta_N^{-1}$,
regardless of the specific value of $q$ on that edge
\textup{(}only the normalization and overall phase depend on~$q$\textup{)}.
\item The number of ordering cycles equals the number of edges in the
Dynkin diagram: $1$ for $A_2$, $3$ for $D_4$. These are \emph{not}
the number of positive roots
(which is $3$ for $A_2$, $12$ for $D_4$),
but rather the number of adjacent simple root pairs. At higher bar
degrees, the remaining positive roots contribute to the bar cohomology
through longer chains of collisions.
\item The $\Einf$ bar complex has $H^2 = 0$ in every sector
considered above: all ordering cycles are strictly $\Eone$ classes
that vanish under symmetrization.
\item Koszul duality inverts the deformation parameter: $q \to -q$,
with the dual braiding being the transpose of the original.
\end{enumerate}
\end{remark}


\subsection{The quantum lattice algebra}

The root-of-unity deformation produces a discrete family of
$\Eone$-chiral algebras interpolating between the commutative (vertex
algebra) world and a noncommutative world parametrized by the
deformation order~$N$. We package these into a single definition.

\begin{definition}[Quantum lattice algebra]\label{def:quantum-lattice}
\index{quantum lattice algebra|textbf}
For an even lattice $\Lambda$ of rank $d \geq 2$, an integer
$N \geq 3$, and a non-zero antisymmetric bilinear form
$q \colon \Lambda \times \Lambda \to \Z/N\Z$, the
\emph{quantum lattice algebra} is:
\[
\Vlat_\Lambda^{N,q}
:= \Vlat_\Lambda^{\varepsilon_{N,q}}
\]
with the deformed cocycle of Definition~\ref{def:lattice:cocycle-deformation}.
\end{definition}

\begin{theorem}[Quantum lattice algebra: structural properties;
\ClaimStatusProvedHere]
\label{thm:quantum-lattice-structure}
\index{quantum lattice algebra!structural properties}
Let $\Vlat_\Lambda^{N,q}$ be a quantum lattice algebra.
\begin{enumerate}[label=(\roman*)]
\item \textup{(}Strict $\Eone$-chirality\textup{).}
 $\Vlat_\Lambda^{N,q}$ is a strictly $\Eone$-chiral algebra,
 not $\Einf$.

\item \textup{(}Periodicity\textup{).}
 $\Vlat_\Lambda^{N,q} \cong \Vlat_\Lambda^{N, q + N\Z}$:
 the construction depends on $q$ only modulo~$N$.

\item \textup{(}$N = 2$ triviality\textup{).}
 At $N = 2$, the deformation produces a symmetric cocycle
 \textup{(}the antisymmetric form $q$ satisfies
 $2q(\alpha,\beta) \equiv 0 \bmod 2$ for all
 $\alpha,\beta$\textup{)}, so $\Vlat_\Lambda^{2,q}$ is $\Einf$.

\item \textup{(}Koszul pair\textup{).}
 The pair $(\Vlat_\Lambda^{N,q},\; \Vlat_\Lambda^{N,-q})$ is an
 $\Eone$-chiral Koszul pair:
 \[
 (\Vlat_\Lambda^{N,q})^! \cong (\Vlat_\Lambda^{N,-q})^c.
 \]
 In particular, for $q$ with $-q \equiv q \pmod{N}$
 \textup{(}i.e., $2q \equiv 0$, which requires $N \mid 2$\textup{)},
 the algebra is Koszul self-dual. For $N \geq 3$,
 $\Vlat_\Lambda^{N,q}$ is never self-dual: the Koszul dual carries
 the \emph{conjugate} deformation parameter.

\item \textup{(}State space\textup{).}
 As a graded vector space,
 $\Vlat_\Lambda^{N,q} \cong \Vlat_\Lambda^{\varepsilon_0}$:
 the deformation changes only the OPE phases, not the state space.
 The character is:
 \[
 \mathrm{ch}\, \Vlat_\Lambda^{N,q}
 = \frac{\Theta_\Lambda(\tau)}{\eta(\tau)^d},
 \]
 independent of $N$ and $q$.

\item \textup{(}$N$-torsion of the braiding\textup{).}
 The braiding of the $\Eone$-chiral structure has order dividing
 $N$: for any $\alpha, \beta \in \Lambda$, the $N$-fold iteration
 of the $R$-matrix-type exchange is trivial:
 \[
 c_{\varepsilon_{N,q}}(\alpha,\beta)^N
 = (-1)^{N\langle\alpha,\beta\rangle} \cdot \zeta_N^{2Nq(\alpha,\beta)}
 = (-1)^{N\langle\alpha,\beta\rangle}.
 \]
\end{enumerate}
\end{theorem}

\begin{proof}
All claims follow directly from
Proposition~\ref{prop:lattice:deformation-properties} and the
definitions. For (ii), $\zeta_N^{q + N\Z} = \zeta_N^q$ since
$\zeta_N^N = 1$. For (iii), $\zeta_2 = -1$ and
$(-1)^{2q(\alpha,\beta)} = 1$. For (v), the cocycle deformation
changes only the product structure (the group algebra multiplication
$e^\alpha \cdot e^\beta = \varepsilon(\alpha,\beta) e^{\alpha+\beta}$)
but not the state space $\cH \otimes \C[\Lambda]$.
\end{proof}


\subsection{$\Eone$ bar cohomology of quantum lattice algebras}

\begin{theorem}[\texorpdfstring{$\Eone$}{E1} bar cohomology; \ClaimStatusProvedHere]
\label{thm:lattice:e1-bar-cohomology}
\index{bar complex!$\Eone$ cohomology}
For the quantum lattice algebra $\Vlat_\Lambda^{N,q}$ on an even
lattice $\Lambda$ with root system~$\Phi$, the $\Eone$ bar complex
$\barB^{\Eone}(\Vlat_\Lambda^{N,q})$ satisfies:
\begin{enumerate}[label=(\roman*)]
\item The lattice grading decomposes the bar complex as
 $\barB^{\Eone} = \bigoplus_{\gamma \in \Lambda}
 \barB^{\Eone}_\gamma$, with the differential
 preserving~$\gamma$.
\item In each sector $\gamma = \alpha + \beta$ with
 $\alpha, \beta \in \Phi$ and $\langle\alpha,\beta\rangle = -1$,
 the $\Eone$ bar complex in degrees $1$--$2$ is:
 \[
 0 \longrightarrow \C^2
 \xrightarrow{d_{\alpha,\beta}}
 \C^1 \longrightarrow 0
 \]
 where $d_{\alpha,\beta}
 = [\varepsilon_0(\alpha,\beta)\,\zeta^{q(\alpha,\beta)},\;
 \varepsilon_0(\beta,\alpha)\,\zeta^{-q(\alpha,\beta)}]$.
 The kernel of $d_{\alpha,\beta}$ is 1-dimensional, spanned by the
 ordering cycle $\xi_{\alpha,\beta}$.
\item The ordering cycles generate a subspace of
 $H^2(\barB^{\Eone})$ of dimension equal to the number of
 positive roots of~$\Phi$. Each positive root
 $\gamma = \alpha + \beta$ contributes one ordering cycle.
\item In the symmetric limit $N \to \infty$ \textup{(}$\zeta \to 1$,
 equivalently the $\Einf$ quotient\textup{)}, the ordering cycles
 become the antisymmetric elements that vanish under
 $\Sigma_2$-coinvariants, recovering the $\Einf$ bar cohomology.
\end{enumerate}
\end{theorem}

\begin{proof}
(i) follows from the lattice-grading preservation of the $\Eone$ bar
differential~\eqref{eq:lattice:e1-bar-diff}, by the same argument as
Theorem~\ref{thm:lattice:bar-structure}(i).

(ii) is the computation of~\ref{comp:lattice:e1-bar-A2} generalized
to any pair of roots with inner product~$-1$. The differential
$d_{\alpha,\beta} \colon \C^2 \to \C$ has
$\operatorname{rank}(d_{\alpha,\beta}) = 1$ since both coefficients are nonzero
(a product of a nonzero cocycle value and a root of unity).

(iii) The positive roots of $\Phi$ biject with pairs
$\{\alpha,\beta\}$ of roots with $\langle\alpha,\beta\rangle = -1$
and $\alpha + \beta \in \Phi$ (up to the relation that each positive
root $\gamma$ is the sum of at most one pair of simple roots
$\alpha,\beta$). Each such pair contributes one ordering cycle in
the sector $\gamma$. Different positive roots give cycles in
different lattice sectors, so the cycles are linearly independent.

(iv) The $\Sigma_2$-coinvariant quotient identifies
$[e^\alpha \mid e^\beta]$ with
$(-1)^{\langle\alpha,\beta\rangle} [e^\beta \mid e^\alpha]
= -[e^\beta \mid e^\alpha]$, so $v_+ = -v_-$ in the quotient.
The ordering cycle
$\xi_{\alpha,\beta} \propto v_+ + \lambda v_-$ (with $\lambda \to 1$
as $\zeta \to 1$) maps to $v_+ + v_- \to v_+ - v_+ = 0$ in the
coinvariant quotient, so it vanishes in the $\Einf$ complex.
\end{proof}


\subsection{The $\Eone$ inversion principle}

The following principle unifies the Koszul duality for
lattice $\Eone$-chiral algebras
(Theorem~\ref{thm:lattice:koszul-dual}),
Yangians (Theorem~\ref{thm:yangian-koszul-dual}), and the
abstract $\Eone$ theory
(Theorem~\ref{thm:e1-chiral-koszul-duality}).

\begin{theorem}[\texorpdfstring{$\Eone$}{E1} inversion principle; \ClaimStatusProvedHere]
\label{thm:e1-inversion-principle}
\index{E1 inversion principle@$\Eone$ inversion principle|textbf}
\index{Koszul duality!inversion principle}
Let $\cA$ be an $\Eone$-chiral algebra whose non-commutativity
is controlled by a defining datum~$\delta$
\textup{(}an $R$-matrix, a 2-cocycle, or a deformation parameter\textup{)}.
Then the Koszul dual $\cA^!$ is the same algebra with the
\emph{inverted} datum $\delta^{-1}$:
\begin{equation}\label{eq:e1-inversion}
\cA^{!,\mathrm{ch}} \cong \cA_{\delta^{-1}}^{\mathrm{ch}}.
\end{equation}
The three instances proved in this monograph are:
\begin{center}
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{l l l l}
\textbf{$\Eone$ algebra $\cA$} & \textbf{Datum $\delta$}
 & \textbf{Koszul dual $\cA^!$}
 & \textbf{Reference} \\
\hline
Lattice $\Vlat_\Lambda^\varepsilon$
 & Cocycle $\varepsilon$
 & $\Vlat_\Lambda^{\varepsilon^{-1}}$
 & Thm~\ref{thm:lattice:koszul-dual} \\
Yangian $Y_R(\fg)$
 & $R$-matrix $R(u)$
 & $Y_{R^{-1}}(\fg)$
 & Thm~\ref{thm:yangian-koszul-dual} \\
Quantum lattice $\Vlat_\Lambda^{N,q}$
 & Phase $\zeta^q$
 & $\Vlat_\Lambda^{N,-q}$
 & Thm~\ref{thm:quantum-lattice-structure}(iv)
\end{tabular}
\end{center}

\noindent
In each case, the inversion arises from the self-duality of the
chiral associative operad:
${\chirAss}^! \cong \chirAss \otimes \mathrm{sgn}$
\textup{(}Proposition~\textup{\ref{prop:chirAss-self-dual})}.
The sign twist $\mathrm{sgn}$ is realized geometrically as the
orientation reversal on ordered configuration spaces, which
inverts the defining non-commutativity datum.
\end{theorem}

\begin{proof}
The abstract statement follows from the $\Eone$-chiral Koszul duality
(Theorem~\ref{thm:e1-chiral-koszul-duality}) and the self-duality of
$\chirAss$: the bar construction on an $\Eone$-chiral algebra
$\cA$ produces a coalgebra over
${\chirAss}^! \cong \chirAss \otimes \mathrm{sgn}$, which is again an
associative coalgebra but with the sign-twisted structure. This sign
twist is realized in each instance as:
\begin{itemize}
\item For lattice algebras: the bar differential carries the cocycle
 value $\varepsilon(\alpha,\beta)$, and the sign twist inverts it to
 $\varepsilon(\alpha,\beta)^{-1}$.
\item For Yangians: the bar differential encodes the RTT relation via
 the $R$-matrix, and the sign twist inverts
 $R(u) \to R^{-1}(u)$, equivalently $\hbar \to -\hbar$ for the
 Yang $R$-matrix.
\item For quantum lattice algebras: the deformation datum
 $\zeta^{q(\alpha,\beta)}$ is inverted to
 $\zeta^{-q(\alpha,\beta)}$.
\end{itemize}
In each case, the geometric mechanism is that the Verdier involution
on ordered configurations $\overline{C}^{\mathrm{ord}}_n(X)$ reverses
the ordering, which transposes the defining relation and hence
inverts the non-commutativity data.
\end{proof}

\begin{remark}[Relation to Feigin--Frenkel duality]
\label{rem:e1-ff-comparison}
\index{Feigin--Frenkel duality!$\Eone$ analog}
The $\Eone$ inversion principle is the ordered-configuration analog
of the Feigin--Frenkel level shift $k \mapsto -k - 2h^\vee$ for
$\Einf$-chiral algebras. In the $\Einf$ theory, Verdier duality on
\emph{unordered} configurations inverts the level; in the $\Eone$
theory, Verdier duality on \emph{ordered} configurations inverts the
defining datum. Both are instances of the same geometric mechanism
(the Verdier involution on FM compactifications), applied in different
operadic contexts.

The parallel extends to the classical limit: the Feigin--Frenkel
shift has a classical limit $k \to \infty$ in which the duality
becomes trivial; the $\Eone$ inversion has a classical limit
$\hbar \to 0$ (for Yangians) or $N \to \infty$ (for quantum
lattice algebras) in which the datum becomes the identity and the
duality becomes the $\Einf$ Koszul duality of
$\mathrm{Sym} \leftrightarrow \bigwedge$
(Corollary~\ref{cor:yangian-classical-self-dual}).
\end{remark}


\subsection{Deformed Frenkel--Kac--Segal and quantum groups}

\begin{remark}[Deformed FKS construction]\label{rem:lattice:deformed-fks}
\index{Frenkel--Kac construction!deformed}
For a simply-laced root lattice $\Lambda_\fg$ with symmetric cocycle
$\varepsilon_0$, the Frenkel--Kac--Segal theorem
(Theorem~\ref{thm:lattice:frenkel-kac}) gives
$\Vlat_{\Lambda_\fg}^{\varepsilon_0} \cong L(\AffKM{g}, 1)$.
The quantum lattice algebra $\Vlat_{\Lambda_\fg}^{N,q}$ deforms this
construction: the root currents $E_\alpha(z) = Y(e^\alpha, z)$
satisfy \emph{modified} commutation relations
\begin{equation}\label{eq:lattice:deformed-fks-ope}
E_{\alpha_i}(z)\, E_{\alpha_j}(w)
= \varepsilon_0(\alpha_i,\alpha_j)\,\zeta^{q(\alpha_i,\alpha_j)}\,
 (z-w)^{\langle\alpha_i,\alpha_j\rangle}\,
 E_{\alpha_i+\alpha_j}(w) + \text{reg}
\end{equation}
in which the Chevalley--Serre sign
$\varepsilon_0(\alpha_i,\alpha_j)$ acquires a root-of-unity factor
$\zeta^{q(\alpha_i,\alpha_j)}$.
\end{remark}

\begin{conjecture}[Deformed FKS and quantum group connection;
\ClaimStatusConjectured]
\label{conj:lattice:quantum-group-connection}
\index{quantum group!lattice construction connection}
For $\fg$ simply-laced of rank $r \geq 2$ and $N \geq 3$, the quantum
lattice algebra $\Vlat_{\Lambda_\fg}^{N,q}$ has the following properties:
\begin{enumerate}[label=(\roman*)]
\item The category of $\Vlat_{\Lambda_\fg}^{N,q}$-modules is a
 braided monoidal category with braiding controlled by the
 deformed commutator $c_{\varepsilon_{N,q}}$.
\item For $\zeta = e^{2\pi i/N}$, this braided category is
 conjecturally equivalent, as a braided monoidal category, to
 $\operatorname{Rep}(u_\zeta(\fg))$.
\item The $\Eone$ inversion principle
 \textup{(}Theorem~\textup{\ref{thm:e1-inversion-principle})}
 applied to $\Vlat_{\Lambda_\fg}^{N,q}$ gives the Koszul pair:
 \[
 (\Vlat_{\Lambda_\fg}^{N,q})^!
 \cong (\Vlat_{\Lambda_\fg}^{N,-q})^c,
 \]
 which is the cocycle-level analog of the quantum group duality
 $U_q(\fg) \leftrightarrow U_{q^{-1}}(\fg)$.
\end{enumerate}
\end{conjecture}

\begin{construction}[Screening operators on quantum lattice algebras]
\label{constr:lattice:screening}
\index{screening operator!quantum lattice}
For a simply-laced root lattice $\Lambda_\fg$ of rank $r \geq 2$
with simple roots $\alpha_1, \ldots, \alpha_r$, the
\emph{screening operators} on $\Vlat_{\Lambda_\fg}^{N,q}$ are:
\[
Q_i := \oint_C Y(e^{\alpha_i}, z)\, dz,
\qquad i = 1, \ldots, r,
\]
where $C$ encircles the origin. Each $Q_i$ maps the
$\gamma$-sector to the $(\gamma + \alpha_i)$-sector of the
lattice Fock space. The \emph{screening algebra}
$\mathcal{S}(\Lambda_\fg, N, q) \subset
\operatorname{End}(\Vlat_{\Lambda_\fg}^{N,q})$ is the
subalgebra generated by $\{Q_i\}_{i=1}^r$.
\end{construction}

\begin{proposition}[Screening current structure; \ClaimStatusProvedHere]
\label{prop:lattice:screening-structure}
\index{screening operator!commutation relations}
The screening currents
$E_{\alpha_i}(z) = Y(e^{\alpha_i}, z)$ on
$\Vlat_{\Lambda_\fg}^{N,q}$ satisfy:
\begin{enumerate}[label=(\roman*)]
\item \textup{(}Adjacent OPE\textup{).}
 For $i \neq j$ with $\langle\alpha_i,\alpha_j\rangle = -1$
 \textup{(}adjacent in the Dynkin diagram\textup{)}:
 \[
 E_{\alpha_i}(z)\, E_{\alpha_j}(w)
 = \frac{\varepsilon_{N,q}(\alpha_i,\alpha_j)}{z - w}
 \,E_{\alpha_i+\alpha_j}(w) + \mathrm{reg}.
 \]
 For non-adjacent nodes
 ($\langle\alpha_i,\alpha_j\rangle = 0$),
 the OPE is regular.

\item \textup{(}Nilpotency\textup{).}
 For all $i$:
 \[
 E_{\alpha_i}(z)\, E_{\alpha_i}(w)
 = \varepsilon_{N,q}(\alpha_i,\alpha_i)\, (z-w)^2\,
 E_{2\alpha_i}(w) + \mathrm{reg},
 \]
 so $Q_i^2 = 0$: the integrand of the double contour integral
 has a zero of order~$2$ at $z_1 = z_2$
 \textup{(}since $\langle\alpha_i,\alpha_i\rangle = 2$ for
 simply-laced\textup{)}.

\item \textup{(}Cocycle braiding\textup{).}
 The cocycle commutator of screening currents is:
 \begin{equation}\label{eq:lattice:screening-braiding}
 c_{\varepsilon_{N,q}}(\alpha_i,\alpha_j)
 = (-1)^{\langle\alpha_i,\alpha_j\rangle}
 \cdot \zeta_N^{2q(\alpha_i,\alpha_j)},
 \end{equation}
 which equals $-\zeta_N^{2q(\alpha_i,\alpha_j)}$ for adjacent
 $i,j$ and $1$ for non-adjacent $i,j$.
 The self-braiding $c_{\varepsilon_{N,q}}(\alpha_i,\alpha_i) = 1$
 for all~$i$ \textup{(}since $q$ is antisymmetric\textup{)},
 so the nilpotency $Q_i^2 = 0$ arises from the conformal weight
 contribution \textup{(}$\langle\alpha_i,\alpha_i\rangle = 2$\textup{)}
 rather than from the braiding.
\end{enumerate}
\end{proposition}

\begin{proof}
Part~(i) specializes~\eqref{eq:lattice:deformed-fks-ope} to
$\langle\alpha_i,\alpha_j\rangle = -1$ (adjacent) and $0$
(non-adjacent) in the simply-laced Dynkin diagram.

Part~(ii): since $\langle\alpha_i,\alpha_i\rangle = 2$, the
leading term $(z-w)^2$ is a zero, not a pole. The double
contour integral
$Q_i^2 = \oint_{C_2}\oint_{C_1}
 (z_1-z_2)^2 \cdot f(z_1,z_2)\,dz_1\,dz_2$
vanishes because the integrand is holomorphic at $z_1 = z_2$.

Part~(iii) specializes
Proposition~\ref{prop:lattice:deformation-properties}(ii) to
simple roots. The self-braiding uses $q(\alpha_i,\alpha_i) = 0$
(antisymmetry of $q$).
\end{proof}

\begin{remark}[Quantum group identification package]
\label{rem:lattice:qg-programme}
\index{quantum group!identification package}
The screening algebra of
Construction~\ref{constr:lattice:screening} is the primary
tool for establishing
Conjecture~\ref{conj:lattice:quantum-group-connection}(ii).
Three complementary routes are available.

\smallskip
\noindent\textbf{Route~A\textup{:} Screening algebra via Nichols
algebras.}
The nilpotency $Q_i^2 = 0$ (part~(ii) above) and the
off-diagonal braiding~\eqref{eq:lattice:screening-braiding}
determine the screening algebra as a quotient of the Nichols
algebra $\mathcal{B}(V)$ of the Yetter--Drinfeld module
$V = \bigoplus_{i=1}^r \mathbb{C}\, x_i$ with braiding matrix
$q_{ij} = c_{\varepsilon_{N,q}}(\alpha_i,\alpha_j)$
\cite{Lentner17}.
The trivial self-braiding ($q_{ii} = 1$) places the screening
algebra in the ``super-type'' regime where nilpotency is
order~$2$ on each generator; the off-diagonal entries
$q_{ij} = -\zeta_N^{2q(\alpha_i,\alpha_j)}$ carry the
deformation data. The quantum group identification requires
matching the resulting Nichols algebra with $u_\zeta(\fg)^+$.

\smallskip
\noindent\textbf{Route~B\textup{:} Module category comparison.}
The vertex tensor category structure on
$\Vlat_{\Lambda_\fg}^{N,q}$-modules \cite{HLZ} carries a
braiding whose monodromy involves the deformed commutator
$c_{\varepsilon_{N,q}}$
(Proposition~\ref{prop:lattice:deformation-properties}(ii)).
For braided fusion categories, the monoidal structure is
determined by the $S$-matrix and twist data \cite{EGNO}.
Matching the $S$-matrix of
$\Vlat_{\Lambda_\fg}^{N,q}$-$\mathrm{mod}$ with that of
$\operatorname{Rep}(u_\zeta(\fg))$ establishes the braided
equivalence by rigidity.

\smallskip
\noindent\textbf{Route~C\textup{:} Zhu algebra.}
Zhu's theorem provides a bijection between simple modules of a
vertex algebra and simple modules of its Zhu algebra.
An explicit computation of
$A(\Vlat_{\Lambda_\fg}^{N,q})$ and comparison with the center
of $u_\zeta(\fg)$ controls the module categories on both sides.

\smallskip
Route~A provides the functor (embedding of the quantum group
into the endomorphism algebra), while Route~B provides the
equivalence criterion. A complete verification requires the
vertex tensor category machinery of \cite{HLZ} extended to the
strictly $\Eone$ setting, an active area connected to
\cite{CreutzigGainutdinovRunkel20,Lentner17}.
\end{remark}

\begin{remark}[Screening nilpotency as logarithm of quantum group periodicity]
\label{rem:screening-nilpotence-periodicity}
\index{nilpotence-periodicity correspondence!lattice application}
\index{screening operator!as logarithmic datum}
The nilpotency $Q_i^2 = 0$
(Proposition~\ref{prop:lattice:screening-structure}(ii)) is the
\emph{logarithmic} (nilpotent, infinitesimal) datum of the lattice
package, while the $N$-periodic representation theory of the quantum
group $u_\zeta(\fg)$ at $\zeta = e^{2\pi i/N}$ is the
\emph{exponential} (periodic, global) datum. The
nilpotence-periodicity correspondence
(Remark~\ref{rem:nilpotence-periodicity}) predicts that the module
category equivalence of
Conjecture~\ref{conj:lattice:quantum-group-connection}(ii)
should be provable by matching the logarithmic data: the screening
algebra (generated by nilpotent elements $Q_i^2 = 0$ with braiding
$c_{\varepsilon_{N,q}}(\alpha_i,\alpha_j)$) encodes the same
information as the nilpotent part $u_\zeta(\fg)^+$ of the small quantum
group. Exponentiating both sides via the categorical $\exp/\log$
correspondence would promote the nilpotent matching to a module
category equivalence, just as, in Lie theory, an isomorphism of
nilpotent Lie algebras exponentiates to an isomorphism of the
corresponding unipotent groups.

This is a fourth route to the quantum group identification,
complementing Routes~A--C of
Remark~\ref{rem:lattice:qg-programme}: match the logarithmic
(screening) data, then exponentiate to obtain the periodic
(representation-theoretic) equivalence. The advantage is that
the nilpotent data (braiding matrix $q_{ij}$ and nilpotency orders)
are computable, while the periodic data (full fusion category
structure) are harder to access directly. The inversion principle
$q \mapsto -q$ under Koszul duality
(Theorem~\ref{thm:e1-inversion-principle}) is the statement that
the logarithm changes sign under duality: $\log(\cA^!) = -\log(\cA)$,
which exponentiates to $\zeta \mapsto \zeta^{-1}$.
\end{remark}

\begin{remark}[The two \texorpdfstring{$\Eone$}{E1} construction methods]
\label{rem:two-e1-methods}
\index{E1-chiral algebra@$\Eone$-chiral algebra!construction methods}
This monograph constructs $\Eone$-chiral algebras by two
complementary methods:
\begin{enumerate}[label=\textup{(\alph*)}]
\item \emph{Lattice cocycle deformation}
 (this section): start with an $\Einf$-chiral lattice VOA and
 deform the cocycle to break symmetry. The state space is fixed,
 and the OPE acquires phases.
 The deformation is controlled by a finite parameter ($N$, $q$).
\item \emph{RTT/R-matrix presentation}
 (Chapter~\ref{chap:yangians}): start with the RTT relations and
 build the $\Eone$-chiral algebra from generators and relations.
 The algebra is presented by generators with an
 $R$-matrix-valued OPE.
\end{enumerate}
The two methods are unified by the $\Eone$ inversion principle
(Theorem~\ref{thm:e1-inversion-principle}): in both cases, Koszul
duality inverts the non-commutativity datum.

The bridge between the two constructions is the Frenkel--Kac--Segal
theorem: method~(a) produces $\Eone$-chiral algebras from lattice
VOAs, while method~(b) produces them from quantum groups. At level~1 with
symmetric cocycle, FKS identifies the two; the deformed FKS
construction (Remark~\ref{rem:lattice:deformed-fks}) extends this
identification to the $\Eone$ world, connecting lattice cocycle
phases to quantum group deformation parameters.
\end{remark}

\section{Factorization bar-cobar for lattice algebras}
\label{sec:lattice:factorization-bar-cobar}


The general bar-cobar equivalence
(Theorem~\ref{thm:bar-cobar-isomorphism-main}) is stated for any
chiral Koszul pair. For lattice vertex algebras, the lattice grading
provides a decomposition that makes the factorization structure
\emph{explicitly computable}, yielding a factorization-level bar-cobar
equivalence that bypasses the thick-generation arguments required in
the Yangian setting (Chapter~\ref{chap:yangians},
\S\ref{sec:derived-dk}).

\subsection{Lattice factorization structure}

\begin{proposition}[Lattice factorization decomposition;
\ClaimStatusProvedHere]
\label{prop:lattice:factorization-decomposition}
\index{factorization coalgebra!lattice decomposition}
Let $\Lambda$ be an even lattice with cocycle $\varepsilon$
\textup{(}not necessarily symmetric\textup{)}. The geometric bar
complex $\barB^{\mathrm{ch}}(\Vlat_\Lambda^\varepsilon)$, viewed as
a factorization coalgebra on~$\Ran(X)$
\textup{(}Theorem~\textup{\ref{thm:bar-chiral})}, decomposes as a
direct sum of factorization coalgebras indexed by the lattice:
\begin{equation}\label{eq:lattice:fact-decomposition}
\barB^{\mathrm{ch}}(\Vlat_\Lambda^\varepsilon)
= \bigoplus_{\gamma \in \Lambda}
 \barB^{\mathrm{ch}}(\Vlat_\Lambda^\varepsilon)_\gamma
\end{equation}
where $\barB^{\mathrm{ch}}_\gamma$ is the sub-coalgebra supported on
total lattice charge~$\gamma$. This decomposition is:
\begin{enumerate}[label=(\roman*)]
\item compatible with the comultiplication: the coproduct
 $\Delta\colon \barB^{\mathrm{ch}}_\gamma \to
 \bigoplus_{\gamma_1+\gamma_2=\gamma}
 \barB^{\mathrm{ch}}_{\gamma_1} \otimes
 \barB^{\mathrm{ch}}_{\gamma_2}$
 preserves total lattice charge;
\item compatible with the factorization structure: for disjoint opens
 $U_1, U_2 \subset X$, the factorization isomorphism
 $\barB^{\mathrm{ch}}_\gamma(U_1 \sqcup U_2) \cong
 \bigoplus_{\gamma_1+\gamma_2=\gamma}
 \barB^{\mathrm{ch}}_{\gamma_1}(U_1)
 \otimes \barB^{\mathrm{ch}}_{\gamma_2}(U_2)$
 respects the lattice grading;
\item \emph{sectorwise finite-dimensional}: for each
 $\gamma \in \Lambda$ and each bar degree~$p$, the sector
 $\barB^p_\gamma$ is a finite-dimensional vector space.
\end{enumerate}
\end{proposition}

\begin{proof}
(i) The bar comultiplication~(Theorem~\ref{thm:bar-chiral})
decomposes a degree-$p$ bar element by partitioning its tensor
factors into two groups. The lattice charge of each group is the
sum of the charges of its factors. Since the bar element has total
charge $\gamma = \sum_i \alpha_i$, the partition produces pieces of
charges $\gamma_1, \gamma_2$ with $\gamma_1 + \gamma_2 = \gamma$.

(ii) The factorization isomorphism for disjoint opens
$U_1 \sqcup U_2$ arises from the decomposition of
$C^{\mathrm{ord}}_n(U_1 \sqcup U_2)$ into products
$C^{\mathrm{ord}}_{n_1}(U_1) \times C^{\mathrm{ord}}_{n_2}(U_2)$.
The vertex operators $Y(e^\alpha, z)$ for $z \in U_i$ carry lattice
charge~$\alpha$ localized on~$U_i$, so the factorization
isomorphism preserves total charge on each open.

(iii) In bar degree~$p$ and sector~$\gamma$, the generators are
ordered $p$-tuples $(\alpha_1, \ldots, \alpha_p)$ with
$\alpha_i \in \Lambda \setminus \{0\}$ and
$\sum_i \alpha_i = \gamma$. For a positive-definite lattice, the
constraint $|\alpha_i|^2 \geq 2$ (even lattice) and
$\sum_i \alpha_i = \gamma$ (fixed) implies
$\sum_i |\alpha_i|^2 \leq p \cdot |\gamma|^2$ by the triangle
inequality, so there are finitely many such tuples. Each tuple
contributes a finite-dimensional space of logarithmic forms on the
compactified configuration space.
\end{proof}

\begin{remark}[Sectorwise finiteness: the key structural advantage]
\label{rem:lattice:sectorwise-finiteness}
Property~(iii) is the critical feature that distinguishes the lattice
setting from the general Yangian setting. For Yangians, the bar
complex has infinitely many generators in each internal degree (the
RTT generators $T_{ij}^{(r)}$ for all levels $r \geq 0$), and
establishing factorization-level results requires thick-generation
arguments (DK-2, Proposition~\ref{prop:yangian-dk2-thick-generation}).
For lattice algebras, sectorwise finiteness means that the twisted
tensor products $K_\tau^L$ and $K_\tau^R$ are direct sums of
finite-dimensional complexes, one per lattice sector. Acyclicity
of each sector can be checked directly.
\end{remark}

\subsection{The factorization Koszul pair}

\begin{theorem}[Lattice factorization Koszul pair;
\ClaimStatusProvedHere]
\label{thm:lattice:factorization-koszul}
\index{Koszul pair!lattice factorization|textbf}
Let $\Lambda$ be an even positive-definite lattice with cocycle
$\varepsilon$ \textup{(}symmetric or not\textup{)}. The lattice
vertex algebra $\Vlat_\Lambda^\varepsilon$ and its Koszul dual
$(\Vlat_\Lambda^{\varepsilon^{-1}})^c$
\textup{(}Theorem~\textup{\ref{thm:lattice:koszul-dual})} form a
chiral Koszul pair
\textup{(}Definition~\textup{\ref{def:chiral-koszul-pair})} in
the factorization sense:
\begin{enumerate}[label=(\roman*)]
\item \textup{(}Factorization bar-cobar equivalence\textup{).}
 The bar construction
 $\barB^{\mathrm{ch}}(\Vlat_\Lambda^\varepsilon)$ is a conilpotent
 factorization coalgebra on $\Ran(X)$, and the cobar-bar counit
 is a quasi-isomorphism of factorization algebras:
 \begin{equation}\label{eq:lattice:fact-bar-cobar}
 \Omega^{\mathrm{ch}}(\barB^{\mathrm{ch}}(\Vlat_\Lambda^\varepsilon))
 \xrightarrow{\;\sim\;}
 \Vlat_\Lambda^\varepsilon.
 \end{equation}
\item \textup{(}Verdier intertwining on $\Ran(X)$\textup{).}
 Verdier duality sends the bar coalgebra to the homotopy Koszul
 dual algebra:
 \begin{equation}\label{eq:lattice:verdier-ran}
 \mathbb{D}_{\Ran}\, \barB^{\mathrm{ch}}(\Vlat_\Lambda^\varepsilon)
 \simeq
 (\Vlat_\Lambda^{\varepsilon^{-1}})^!_\infty
 \qquad
 \bigl(\text{underlying complex }
 \simeq \barB^{\mathrm{ch}}(\Vlat_\Lambda^{\varepsilon^{-1}})\bigr).
 \end{equation}
\item \textup{(}Functoriality over $\overline{\mathcal{M}}_{g,n}$\textup{).}
 Both equivalences hold in families over moduli: for
 $\pi\colon \mathcal{X} \to \overline{\mathcal{M}}_{g,n}$ the universal curve,
 the relative bar construction
 $\barB^{\mathrm{ch}}_{\mathcal{X}/\overline{\mathcal{M}}_{g,n}}
 (\Vlat_\Lambda^\varepsilon)$ is a factorization coalgebra on
 $\Ran(\mathcal{X}/\overline{\mathcal{M}}_{g,n})$, and the counit and Verdier
 equivalences hold fiberwise by proper base change.
\end{enumerate}
\end{theorem}

\begin{proof}
The proof reduces to the chain-level theorem
(Theorem~\ref{thm:lattice:koszul-morphism}) via the lattice
decomposition.

\emph{Step 1: Reduction to sectors.}
By Proposition~\ref{prop:lattice:factorization-decomposition}, the
factorization coalgebra
$\barB^{\mathrm{ch}}(\Vlat_\Lambda^\varepsilon)$ decomposes as a
direct sum over $\gamma \in \Lambda$. The twisted tensor products
$K_\tau^L$ and $K_\tau^R$
(Definition~\ref{def:chiral-twisting-datum}) decompose
correspondingly:
\[
K_\tau^L(\Vlat_\Lambda^\varepsilon,\,
\barB^{\mathrm{ch}}(\Vlat_\Lambda^\varepsilon))
= \bigoplus_{\gamma \in \Lambda}
K_\tau^L(\Vlat_\Lambda^\varepsilon,\,
\barB^{\mathrm{ch}}_\gamma).
\]
Acyclicity of $K_\tau^L$ is equivalent to acyclicity of each
sector $K_\tau^L(\cdots, \barB^{\mathrm{ch}}_\gamma)$.

\emph{Step 2: Sectorwise acyclicity.}
Fix $\gamma \in \Lambda$. The sector
$\barB^{\mathrm{ch}}_\gamma$ is finite-dimensional in each bar
degree (Proposition~\ref{prop:lattice:factorization-decomposition}(iii)).
The twisted tensor product
$K_{\tau,\gamma}^L = \Vlat_\Lambda^\varepsilon
\otimes_\tau \barB^{\mathrm{ch}}_\gamma$ is therefore a
bounded-below complex of finite-dimensional pieces (over each
stratum of~$\Ran(X)$).

The lattice weight filtration of
Theorem~\ref{thm:lattice:koszul-morphism} applies sector by
sector: filter $K_{\tau,\gamma}^L$ by total weight
$|\alpha|^2 = \langle\alpha,\alpha\rangle$, and the
$E_1$ page is the Heisenberg twisted tensor product
(the $\gamma = 0$ reduction). By
Theorem~\ref{thm:heisenberg-koszul-dual-early}, the Heisenberg
twisted tensor product is acyclic. Since the filtration is
bounded below in each sector and the $E_1$ page vanishes, the
spectral sequence converges to zero.

\emph{Step 3: From acyclicity to factorization equivalence.}
By Lemma~\ref{lem:twisted-product-cone-counit}, acyclicity of
$K_\tau^L$ is equivalent to the counit
$\varepsilon_\tau\colon
\Omega^{\mathrm{ch}}(\barB^{\mathrm{ch}}(\Vlat_\Lambda^\varepsilon))
\to \Vlat_\Lambda^\varepsilon$
being a quasi-isomorphism. By
Lemma~\ref{lem:twisted-product-cone-unit}, acyclicity of
$K_\tau^R$ is equivalent to the unit
$\eta_\tau\colon
\barB^{\mathrm{ch}}(\Vlat_\Lambda^{\varepsilon^{-1}})^c
\to \barB^{\mathrm{ch}}(\Vlat_\Lambda^\varepsilon)$
being a quasi-isomorphism.

Since these quasi-isomorphisms hold in each lattice sector and the
direct sum is compatible with the factorization structure
(Proposition~\ref{prop:lattice:factorization-decomposition}(ii)),
the quasi-isomorphisms are factorization-algebra morphisms. This
establishes~(i).

\emph{Step 4: Verdier intertwining.}
The Verdier duality $\mathbb{D}_{\Ran}$ acts on the bar complex
by exchanging $j_*$ (maximal extension) with $j_!$ (extension by
zero) on each stratum of $\overline{C}_n(X)$
(Lemma~\ref{lem:verdier-extension-exchange}). On the lattice bar
complex, this exchanges the cocycle $\varepsilon$ with
$\varepsilon^{-1}$: the bar differential carries the factor
$\varepsilon(\alpha_i, \alpha_{i+1})$, and Verdier duality
conjugates this to $\varepsilon(\alpha_i, \alpha_{i+1})^{-1}$
(by the sign change in the logarithmic form under
$j_* \leftrightarrow j_!$). Combined with the bar-cobar
Verdier theorem (Theorem~\ref{thm:bar-cobar-verdier}), this
gives~(ii).

Since both $\varepsilon$ and $\varepsilon^{-1}$ are valid cocycles
on the same lattice $\Lambda$, and the construction is symmetric
in the two, the Verdier intertwining holds at the factorization
level (not just stratum-by-stratum) by the functoriality of
$\mathbb{D}_{\Ran}$ on factorization coalgebras.

\emph{Step 5: Family statement.}
The FM compactification $\overline{C}_n(\mathcal{X})$ is proper over
$\overline{\mathcal{M}}_{g,n}$
(Lemma~\ref{lem:pushforward-preserves-qi}). By proper base change
(Theorem~\ref{thm:bar-cobar-inversion-functorial}), the fiberwise
quasi-isomorphisms of Steps~2--3 lift to the family. The lattice
decomposition persists in families (the lattice $\Lambda$ is
constant over $\overline{\mathcal{M}}_{g,n}$, since it is an algebraic
datum of the vertex algebra, not of the curve), so the sectorwise
finiteness argument applies fiber by fiber. This gives~(iii).
\end{proof}

\begin{remark}[Why thick generation is unnecessary]
\label{rem:lattice:no-thick-generation}
\index{thick generation!unnecessary for lattice}
The proof of
Theorem~\ref{thm:lattice:factorization-koszul} does not use any
thick-generation argument. Compare with the Yangian DK ladder
(Chapter~\ref{chap:yangians}):
\begin{itemize}
\item DK-2 (Proposition~\ref{prop:yangian-dk2-thick-generation})
 requires showing that evaluation modules thickly generate the
 finite-dimensional representation category. For lattice algebras,
 the lattice sectors replace thick generation: the direct sum
 decomposition~\eqref{eq:lattice:fact-decomposition} is
 \emph{exact}, not a thick-closure approximation.
\item DK-3 (Proposition~\ref{prop:yangian-dk3-generated-core})
 requires lifting the Kazhdan--Lusztig equivalence to factorization
 categories. For lattice algebras, both sides of the Koszul duality
 live in the same lattice-VOA family (cocycle
 $\varepsilon$ vs.\ $\varepsilon^{-1}$), so no external bridge
 is needed.
\end{itemize}
The lattice setting thus provides the first \emph{unconditional}
(Tier~T1) factorization bar-cobar equivalence in the monograph.
\end{remark}


\subsection{Factorization DK for lattice algebras}

Theorem~\ref{thm:lattice:factorization-koszul} specializes, via the
Frenkel--Kac--Segal identification, to a factorization-level
Drinfeld--Kohno statement for level-$1$ simply-laced affine algebras.

\begin{corollary}[Factorization DK at level \texorpdfstring{$1$}{1};
\ClaimStatusProvedHere]
\label{cor:lattice:factorization-dk-level1}
\index{Drinfeld--Kohno!lattice factorization|textbf}
For $\fg$ simply-laced with root lattice $\Lambda_\fg$ and
symmetric cocycle $\varepsilon_0$:
\begin{enumerate}[label=(\roman*)]
\item The factorization bar-cobar equivalence for
 $\Vlat_{\Lambda_\fg}^{\varepsilon_0}
 \cong L(\AffKM{g}, 1)$
 \textup{(}Theorem~\textup{\ref{thm:lattice:frenkel-kac})} gives:
 \[
 \Omega^{\mathrm{ch}}(\barB^{\mathrm{ch}}(L(\AffKM{g}, 1)))
 \xrightarrow{\;\sim\;}
 L(\AffKM{g}, 1)
 \]
 as an equivalence of factorization algebras on $\Ran(X)$.

\item Verdier duality gives
 $\mathbb{D}_{\Ran}\, \barB^{\mathrm{ch}}(L(\AffKM{g}, 1))
 \simeq L(\AffKM{g}, 1)^!_\infty \simeq L(\AffKM{g}, 1)$
 \textup{(}self-duality, since $\varepsilon_0^{-1} = \varepsilon_0$
 for the standard FLM cocycle on simply-laced root lattices\textup{)}.

\item For $\Eone$-deformations: the quantum lattice algebra
 $\Vlat_{\Lambda_\fg}^{N,q}$
 \textup{(}Definition~\textup{\ref{def:quantum-lattice})} satisfies
 the $\Eone$-factorization bar-cobar equivalence
 \[
 \Omega^{\Eone,\mathrm{ch}}(
 \barB^{\Eone,\mathrm{ch}}(\Vlat_{\Lambda_\fg}^{N,q}))
 \xrightarrow{\;\sim\;}
 \Vlat_{\Lambda_\fg}^{N,q}
 \]
 with Verdier identification
 $\mathbb{D}_{\Ran}\, \barB^{\Eone}(\Vlat_{\Lambda_\fg}^{N,q})
 \simeq (\Vlat_{\Lambda_\fg}^{N,-q})^!_\infty$
 factorization \emph{algebra}, not coalgebra;
 not self-dual: the deformation parameter is
 conjugated\textup{)}.
\end{enumerate}
\end{corollary}

\begin{proof}
(i) and (ii): Apply Theorem~\ref{thm:lattice:factorization-koszul}
to $\Lambda = \Lambda_\fg$ with $\varepsilon = \varepsilon_0$. The
FKS identification
(Theorem~\ref{thm:lattice:frenkel-kac}) is an isomorphism of vertex
algebras, hence of factorization algebras on $\Ran(X)$. For the
self-duality in~(ii), the standard FLM cocycle on a simply-laced root
lattice takes values in $\{\pm 1\}$, so
$\varepsilon_0^{-1} = \varepsilon_0$.

(iii): Apply Theorem~\ref{thm:lattice:factorization-koszul} to the
non-symmetric cocycle $\varepsilon_{N,q}$. The theorem applies
equally to non-symmetric cocycles, since the proof uses only the
lattice decomposition and the weight filtration, both of which are
cocycle-independent. The $\Eone$-chiral bar complex is on ordered
configurations
(Construction~\ref{constr:lattice:e1-bar},
Theorem~\ref{thm:e1-chiral-koszul-duality}). The Verdier
intertwining conjugates $\varepsilon_{N,q}$ to
$\varepsilon_{N,q}^{-1} = \varepsilon_{N,-q}$
(Proposition~\ref{prop:lattice:deformation-properties}(iv)).
\end{proof}

\begin{remark}[The factorization DK ladder from lattices]
\label{rem:lattice:factorization-dk-programme}
\index{Drinfeld--Kohno!factorization ladder}
Corollary~\ref{cor:lattice:factorization-dk-level1} is
unconditional: it does not depend on thick generation, the
Kazhdan--Lusztig equivalence, or the Latyntsev factorization
quantum group construction. It is an \emph{intrinsic} statement
about the lattice vertex algebra, proved entirely within the
bar-cobar framework of Part~\ref{part:bar-complex}.

The relationship to the Yangian DK ladder
(MC3 categorical CG closure and the generated-core DK comparison are
proved for all simple types
\textup{(}Theorem~\ref{thm:categorical-cg-all-types},
Corollary~\ref{cor:mc3-all-types}\textup{)}; post-CG completion open
beyond type~$A$; the full DK/KL categorical
equivalence is Conjecture~\ref{conj:master-dk-kl}) is as follows:
\begin{enumerate}[label=\textup{(\alph*)}]
\item At level~$1$ with symmetric cocycle, the lattice factorization
 DK reduces to the self-duality of unimodular lattice VOAs
 (Theorem~\ref{thm:lattice:unimodular-self-dual}), which is proved.

\item At level~$1$ with non-symmetric cocycle, the $\Eone$ lattice
 factorization DK gives a proved instance of the $\Eone$ inversion
 principle (Theorem~\ref{thm:e1-inversion-principle}) at the
 factorization level.

\item For general level~$k$: by the sublattice embedding
 $\sqrt{k}\Lambda \hookrightarrow \Lambda$
 (Proposition~\ref{prop:lattice:sublattice}), the level-$k$ KM
 algebra embeds into the level-$1$ lattice VOA. Whether the
 factorization bar-cobar equivalence for the sublattice implies
 the equivalence for the full level-$k$ theory is a
 tractable extension of the present methods.

\item For the full DK/KL categorical equivalence: identifying the module category of
 $\Vlat_{\Lambda_\fg}^{N,q}$ with quantum group representations
 at $\zeta = e^{2\pi i/N}$ would upgrade the lattice
 factorization DK to the full Kazhdan--Lusztig factorization
 equivalence. This identification is
 Conjecture~\ref{conj:lattice:quantum-group-connection}, currently
 conjectural.
\end{enumerate}

Thus the lattice chapter should be read as proving DK-$1\frac{1}{2}$ and
its level/$q$ variants by an independent factorization route, not as
closing the full DK/KL categorical equivalence
(Conjecture~\ref{conj:master-dk-kl}); the underlying all-types MC3 core
\textup{(}categorical CG closure plus the generated-core DK comparison\textup{)}
is proved for all simple types
\textup{(}Corollary~\ref{cor:mc3-all-types}\textup{)}. The exact remaining input
is the quantum-group identification above. Only after that identification is in
place does the standard MC4 problem begin: the filtered H-level
comparison of the standard infinite towers.

The universal MC class for lattice VOAs takes the form
$\Theta_{V_\Lambda}^{\min} = \operatorname{rank}(\Lambda) \cdot
\eta \otimes \Lambda_{\mathrm{Hodge}}$
(Theorem~\ref{thm:explicit-theta},
Corollary~\ref{cor:explicit-theta-specializations}),
where $\kappa(V_\Lambda) = \operatorname{rank}(\Lambda)$ by
additivity. Under the $\Eone$-inversion principle
(Theorem~\ref{thm:e1-inversion-principle}), the universal class
transforms as
$\Theta_{V_\Lambda} \mapsto \Theta_{V_{\Lambda^*}}$
(dual lattice), and the cocycle conjugation
$\varepsilon \mapsto \varepsilon^{-1}$ is the chain-level
signature of the Verdier duality
$\Theta \mapsto \Theta^!$ on ordered factorization
bar complexes.
\end{remark}

\section{Homotopy-level factorization DK via $\Eone$ lattices}
\label{sec:lattice:homotopy-factorization-dk}


The factorization bar-cobar equivalence of
Theorem~\ref{thm:lattice:factorization-koszul} is an M-level statement:
a quasi-isomorphism of factorization dg-algebras on $\Ran(X)$. This
section lifts the equivalence to H-level, constructing an equivalence of
$\Eone$-factorization $\infty$-categories for lattice vertex algebras.
The sectorwise finiteness of
Proposition~\ref{prop:lattice:factorization-decomposition}(iii) is the
mechanism that makes the lift possible without the thick-generation
arguments needed in the Yangian setting (DK-2/3 of
Remark~\ref{rem:dk-proved-vs-remains}).


\subsection{The $\Eone$-factorization $\infty$-category of a lattice VOA}

\begin{construction}[Lattice $\Eone$-factorization category]
\label{constr:lattice:e1-fact-cat}
\index{factorization category!lattice $\Eone$-|textbf}
Let $\Lambda$ be an even positive-definite lattice with cocycle
$\varepsilon$ (symmetric or not). The \emph{$\Eone$-factorization
$\infty$-category} $\Factord(X; \Vlat_\Lambda^\varepsilon)$
has the following data:
\begin{enumerate}[label=(\roman*)]
\item \emph{Objects}: $\Vlat_\Lambda^\varepsilon$-modules on ordered
 configuration spaces $C_n^{\mathrm{ord}}(X)$
 (Definition~\ref{def:e1-factorization-category}), graded by
 lattice charge: an object $M$ decomposes as
 $M = \bigoplus_{\gamma \in \Lambda} M_\gamma$ with each
 $M_\gamma$ a finite-dimensional dg vector space.
\item \emph{Factorization structure}: for disjoint intervals
 $I_1, \ldots, I_k \subset X$, the ordered factorization isomorphisms
 \[
 \Vlat_\Lambda^\varepsilon\text{-mod}(
 C_{n_1}^{\mathrm{ord}}(I_1) \times \cdots
 \times C_{n_k}^{\mathrm{ord}}(I_k))
 \xrightarrow{\;\sim\;}
 \Vlat_\Lambda^\varepsilon\text{-mod}(
 C_n^{\mathrm{ord}}(\textstyle\bigcup I_j))
 \]
 respect the lattice grading.
\item \emph{Monoidal structure}: the ordered chiral tensor product
 $M \otimes^{\mathrm{ch}}_{\mathrm{ord}} N$ is defined by fusion on
 ordered pairs of points, giving an associative (not symmetric)
 monoidal structure. For a non-symmetric cocycle, the braiding
 $\sigma_{M,N} \colon M \otimes^{\mathrm{ch}}_{\mathrm{ord}} N
 \to N \otimes^{\mathrm{ch}}_{\mathrm{ord}} M$
 is controlled by the commutator $c_\varepsilon$ and is \emph{not}
 a symmetric braiding.
\item \emph{Lattice sector decomposition}: the category decomposes as
 \[
 \Factord(X; \Vlat_\Lambda^\varepsilon)
 = \bigoplus_{\gamma \in \Lambda}
 \Factord(X; \Vlat_\Lambda^\varepsilon)_\gamma
 \]
 with each sector $\Factord(X; \Vlat_\Lambda^\varepsilon)_\gamma$
 a stable $\infty$-category of objects with total charge~$\gamma$.
\end{enumerate}
\end{construction}

\begin{proposition}[Sectorwise compactness;
\ClaimStatusProvedHere]
\label{prop:lattice:sectorwise-compactness}
\index{compactness!sectorwise lattice}
For each $\gamma \in \Lambda$ and each $n \geq 1$, the lattice
sector $\Factord(X; \Vlat_\Lambda^\varepsilon)_\gamma$ restricted
to $C_n^{\mathrm{ord}}(X)$ is a dg category of
\emph{compact objects}: every object has a finite filtration whose
graded pieces are shifts of
$\Vlat_\Lambda^\varepsilon$-modules of the form
$V_\gamma \otimes \cO_{C_n^{\mathrm{ord}}}$, where $V_\gamma$ is
a finite-dimensional vector space. In particular, every object is
compact in the sense of Lurie~\cite{HA}.
\end{proposition}

\begin{proof}
By Proposition~\ref{prop:lattice:factorization-decomposition}(iii),
the bar complex in sector~$\gamma$ and bar degree~$p$ is
finite-dimensional. A module $M_\gamma$ on
$C_n^{\mathrm{ord}}(X)$ has a filtration by weight
$|\alpha|^2 = \langle\alpha,\alpha\rangle$, the same weight
filtration used in the proof of
Theorem~\ref{thm:lattice:factorization-koszul}. At each
filtration step, the subquotient is a module for the Heisenberg
subalgebra $\cH_\Lambda$ tensored with a finite-dimensional
lattice-charge space.

Since $\Lambda$ is positive-definite, for fixed total
charge~$\gamma$ and bar degree~$p$, the number of ordered tuples
$(\alpha_1, \ldots, \alpha_p)$ with $\sum \alpha_i = \gamma$ and
$\alpha_i \neq 0$ is finite
(Proposition~\ref{prop:lattice:factorization-decomposition}(iii)).
The filtration is therefore finite (bounded by the number of such
tuples), and the associated graded pieces are finite-dimensional
dg vector spaces tensored with $\cO_{C_n^{\mathrm{ord}}}$. Each
such piece is compact in the dg category of coherent sheaves on
the smooth variety $C_n^{\mathrm{ord}}(X)$, and a finite extension
of compact objects is compact.
\end{proof}


\subsection{Homotopy-level factorization bar-cobar for lattices}

\begin{theorem}[Homotopy-level lattice factorization DK;
\ClaimStatusProvedHere]
\label{thm:lattice:homotopy-factorization-dk}
\index{Drinfeld--Kohno!homotopy-level lattice|textbf}
\index{factorization category!homotopy equivalence}
Let $\Lambda$ be an even positive-definite lattice with cocycle
$\varepsilon$. The bar-cobar adjunction of
Theorem~\textup{\ref{thm:lattice:factorization-koszul}} lifts to an
equivalence of $\Eone$-factorization $\infty$-categories:
\begin{equation}\label{eq:lattice:homotopy-fact-dk}
\Factord(X; \Vlat_\Lambda^\varepsilon)
\;\simeq\;
\Factord(X; \Vlat_\Lambda^{\varepsilon^{-1}})^{\mathrm{rev}}
\end{equation}
where $(\cdots)^{\mathrm{rev}}$ denotes the reverse monoidal
structure \textup{(}exchanging the order of the tensor factors\textup{)}.
This equivalence is:
\begin{enumerate}[label=(\roman*)]
\item \emph{Monoidal}: it intertwines the ordered chiral tensor
 product on both sides, with the reversal accounting for the
 cocycle inversion $\varepsilon \leftrightarrow \varepsilon^{-1}$.
\item \emph{Natural in families}: for $\pi\colon \mathcal{X} \to S$
 a smooth family of curves, the equivalence holds in families
 over~$S$.
\item \emph{Sectorwise}: the equivalence restricts to an equivalence
 of each lattice sector:
 \[
 \Factord(X; \Vlat_\Lambda^\varepsilon)_\gamma
 \;\simeq\;
 \Factord(X; \Vlat_\Lambda^{\varepsilon^{-1}})_\gamma^{\mathrm{rev}}.
 \]
\end{enumerate}
\end{theorem}

\begin{proof}
The proof assembles three ingredients: the M-level factorization
equivalence (Theorem~\ref{thm:lattice:factorization-koszul}),
sectorwise compactness
(Proposition~\ref{prop:lattice:sectorwise-compactness}), and the
$\Eone$ inversion principle
(Theorem~\ref{thm:e1-inversion-principle}).

\emph{Step 1: From M-level qi to $\infty$-categorical equivalence.}
Theorem~\ref{thm:lattice:factorization-koszul} provides
quasi-isomorphisms of factorization dg-algebras:
\[
\Omega^{\mathrm{ch}}(\barB^{\mathrm{ch}}(\Vlat_\Lambda^\varepsilon))
\xrightarrow{\;\sim\;} \Vlat_\Lambda^\varepsilon,
\qquad
\mathbb{D}_{\Ran}\, \barB^{\mathrm{ch}}(\Vlat_\Lambda^\varepsilon)
\simeq (\Vlat_\Lambda^{\varepsilon^{-1}})^!_\infty.
\]
These are morphisms in the $\infty$-category of factorization
algebras on $\Ran(X)$. By the universal property of the dg nerve
construction (Lurie~\cite{HA}~\S5.2), a quasi-isomorphism
of dg-algebras induces an equivalence on module
$\infty$-categories. Specifically: for dg-algebras $A$ and $B$
with a quasi-isomorphism $f\colon A \xrightarrow{\sim} B$, the
restriction functor $f^*\colon B\text{-mod} \to A\text{-mod}$
is an equivalence of stable $\infty$-categories.

\emph{Step 2: Sectorwise reduction.}
By the lattice sector decomposition
(Proposition~\ref{prop:lattice:factorization-decomposition}),
both sides of~\eqref{eq:lattice:homotopy-fact-dk} decompose as
direct sums over $\gamma \in \Lambda$. It suffices to prove the
equivalence sector by sector.

Fix $\gamma \in \Lambda$. By
Proposition~\ref{prop:lattice:sectorwise-compactness}, the sector
category $\Factord(X; \Vlat_\Lambda^\varepsilon)_\gamma$ is
compactly generated by objects with finite lattice-weight
filtrations. The same holds for the dual side with
$\varepsilon^{-1}$.

\emph{Step 3: Compact generation and the Schwede--Shipley criterion.}
By Step~2, both sides are compactly generated stable
$\infty$-categories. A dg functor between compactly generated
stable $\infty$-categories is an equivalence if and only if it
restricts to an equivalence on compact objects (the
Schwede--Shipley recognition theorem~\cite{SS03}).

The bar-cobar quasi-isomorphism of Step~1, restricted to sector
$\gamma$, sends compact objects to compact objects: the weight
filtration is preserved (it is defined by the lattice norm, which
is the same on both sides), and a finite filtration maps to a
finite filtration. On each filtration piece (a finite-dimensional
dg vector space), the quasi-isomorphism is a quasi-isomorphism of
chain complexes, hence an equivalence in the dg nerve.

Therefore the induced functor on compact objects is an
equivalence, and by Schwede--Shipley the full functor is an
equivalence of stable $\infty$-categories.

\emph{Step 4: Monoidal structure and reversal.}
The ordered chiral tensor product on
$\Factord(X; \Vlat_\Lambda^\varepsilon)$ uses the cocycle
$\varepsilon$ to determine the fusion coefficients. The Verdier
intertwining
(Theorem~\ref{thm:lattice:factorization-koszul}(ii)) replaces
$\varepsilon$ by $\varepsilon^{-1}$. By the $\Eone$ inversion
principle (Theorem~\ref{thm:e1-inversion-principle}), this
inversion of the non-commutativity datum reverses the monoidal
order: the functor
$\Factord(X; \Vlat_\Lambda^\varepsilon) \to
\Factord(X; \Vlat_\Lambda^{\varepsilon^{-1}})$ interchanges
$M \otimes^{\mathrm{ch}}_{\mathrm{ord}} N$ with
$N \otimes^{\mathrm{ch}}_{\mathrm{ord}} M$. This gives the
$(\cdots)^{\mathrm{rev}}$ on the right-hand side.

\emph{Step 5: Families.}
The M-level equivalence holds in families over
$\overline{\mathcal{M}}_{g,n}$
(Theorem~\ref{thm:lattice:factorization-koszul}(iii)).
Steps~1--4 apply fiberwise: the lattice $\Lambda$ is constant over
$\overline{\mathcal{M}}_{g,n}$, the sectorwise finiteness is a
property of $\Lambda$ (not of the curve), and the Schwede--Shipley
criterion commutes with proper base change (since compact
generation is preserved by proper pushforward).
\end{proof}


\begin{remark}[Comparison with the Yangian DK ladder]
\label{rem:lattice:comparison-yangian-dk}
\index{Drinfeld--Kohno!lattice vs.\ Yangian comparison}
Theorem~\ref{thm:lattice:homotopy-factorization-dk} provides an
unconditional H-level factorization DK for lattice algebras.
Compare with the Yangian DK ladder
(Remark~\ref{rem:dk-proved-vs-remains}):
\begin{center}
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{l l l}

\subsection{Level-$k$ extension via sublattice embeddings}

The level-$1$ factorization DK
(Corollary~\ref{cor:lattice:factorization-dk-level1}) extends to
general level~$k$ via the sublattice embedding
$\sqrt{k}\Lambda \hookrightarrow \Lambda$ of
Proposition~\ref{prop:lattice:sublattice}.

\begin{proposition}[Level-\texorpdfstring{$k$}{k} lattice factorization bar-cobar;
\ClaimStatusProvedHere]
\label{prop:lattice:level-k-factorization}
\index{Koszul pair!level-$k$ lattice factorization}
Let $\fg$ be simply-laced with root lattice $\Lambda_\fg$, and
let $k \geq 1$ be a positive integer. The sublattice embedding
$\iota_k \colon \sqrt{k}\Lambda_\fg \hookrightarrow \Lambda_\fg$
\textup{(}Proposition~\textup{\ref{prop:lattice:sublattice})} induces
a factorization bar-cobar equivalence for the level-$k$
sublattice VOA:
\begin{equation}\label{eq:lattice:level-k-bar-cobar}
\Omega^{\mathrm{ch}}(
 \barB^{\mathrm{ch}}(\Vlat_{\sqrt{k}\Lambda_\fg}^{\varepsilon_0}))
\xrightarrow{\;\sim\;}
\Vlat_{\sqrt{k}\Lambda_\fg}^{\varepsilon_0}
\end{equation}
as an equivalence of factorization algebras on $\Ran(X)$. This
lifts to an $\Eone$-factorization $\infty$-category equivalence:
\begin{equation}\label{eq:lattice:level-k-cat-equiv}
\Factord(X; \Vlat_{\sqrt{k}\Lambda_\fg}^{\varepsilon_0})
\;\simeq\;
\Factord(X;
 \Vlat_{\sqrt{k}\Lambda_\fg}^{\varepsilon_0^{-1}})^{\mathrm{rev}}.
\end{equation}
\end{proposition}

\begin{proof}
The lattice $\sqrt{k}\Lambda_\fg$ is even and positive-definite:
$\langle\sqrt{k}\alpha,\sqrt{k}\beta\rangle_{\sqrt{k}\Lambda}
= k\langle\alpha,\beta\rangle_\Lambda$, which is even since
$\Lambda_\fg$ is even. The cocycle restricts:
$\iota_k^*\varepsilon_0(\sqrt{k}\alpha,\sqrt{k}\beta) =
\varepsilon_0(k\alpha,k\beta) =
\varepsilon_0(\alpha,\beta)^{k^2}$, which is a valid symmetric
cocycle on $\sqrt{k}\Lambda_\fg$.

Apply Theorem~\ref{thm:lattice:factorization-koszul} to the
lattice $\sqrt{k}\Lambda_\fg$ with the restricted cocycle. All
hypotheses are satisfied: the lattice is even positive-definite
and the cocycle is normalized bimultiplicative. This
gives~\eqref{eq:lattice:level-k-bar-cobar}.

The homotopy-level factorization equivalence~\eqref{eq:lattice:level-k-cat-equiv} follows
from Theorem~\ref{thm:lattice:homotopy-factorization-dk} applied
to $\sqrt{k}\Lambda_\fg$.
\end{proof}

\begin{theorem}[Level-\texorpdfstring{$k$}{k} factorization DK for KM algebras;
\ClaimStatusProvedHere]
\label{thm:lattice:level-k-dk}
\index{Drinfeld--Kohno!level-$k$ lattice|textbf}
For $\fg$ simply-laced and $k \geq 1$, the coset decomposition
\begin{equation}\label{eq:lattice:coset-decomposition}
\Vlat_{\Lambda_\fg}^{\varepsilon_0}
= \bigoplus_{[\gamma] \in \Lambda_\fg/\sqrt{k}\Lambda_\fg}
 \Vlat_{\sqrt{k}\Lambda_\fg}^{\varepsilon_0}\text{-module}_\gamma
\end{equation}
expresses the level-$1$ lattice VOA as a finite direct sum of
modules for the level-$k$ sublattice VOA. Combining
with~\eqref{eq:lattice:level-k-bar-cobar}:
\begin{enumerate}[label=(\roman*)]
\item The factorization bar-cobar equivalence for
 $\Vlat_{\sqrt{k}\Lambda_\fg}$ is \emph{unconditional}
 \textup{(}does not require thick generation or the KL
 equivalence\textup{)}.
\item The $\Eone$-factorization $\infty$-category of
 $\Vlat_{\sqrt{k}\Lambda_\fg}$-modules decomposes by coset
 representatives:
 \[
 \Factord(X; \Vlat_{\Lambda_\fg}^{\varepsilon_0})
 \;\simeq\;
 \bigoplus_{[\gamma] \in \Lambda_\fg/\sqrt{k}\Lambda_\fg}
 \Factord(X;
 \Vlat_{\sqrt{k}\Lambda_\fg}^{\varepsilon_0})_\gamma.
 \]
\item The number of cosets
 $|\Lambda_\fg/\sqrt{k}\Lambda_\fg| =
 k^{\operatorname{rank}(\fg)/2} \cdot |\Lambda_\fg^*/\Lambda_\fg|$
 equals the number of integrable
 $\widehat{\fg}_k$-modules at level~$k$
 \textup{(}for $k = 1$: this is
 $|\Lambda_\fg^*/\Lambda_\fg|$, the number of minuscule weights
 plus the identity\textup{)}.
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{Step 1: Coset decomposition.}
The sublattice $\sqrt{k}\Lambda_\fg \subset \Lambda_\fg$ has
index $[\Lambda_\fg : \sqrt{k}\Lambda_\fg] =
k^{\operatorname{rank}(\fg)/2} \cdot |\Lambda_\fg^*/\Lambda_\fg|$
(the index of $k\Lambda$ in $\Lambda$ is $k^d$ for a rank-$d$
lattice, and rescaling $\sqrt{k}\Lambda \hookrightarrow \Lambda$
has index $k^{d/2}$ for the $\sqrt{k}$-rescaled inner product;
the factor $|\Lambda_\fg^*/\Lambda_\fg|$ accounts for the
possible non-unimodularity of $\Lambda_\fg$).

The group algebra decomposes as
$\C[\Lambda_\fg] =
\bigoplus_{[\gamma]} \C[\sqrt{k}\Lambda_\fg + \gamma]$
over coset representatives. Since the Heisenberg algebra
$\cH_\Lambda$ acts on each coset sector independently, the VOA
decomposes as a direct sum of $\Vlat_{\sqrt{k}\Lambda_\fg}$-modules.

\emph{Step 2: Factorization compatibility.}
Each coset module
$\Vlat_{\sqrt{k}\Lambda_\fg}\text{-mod}_\gamma$ is a
factorization module on $\Ran(X)$: the factorization isomorphism
for disjoint opens preserves the coset $[\gamma]$ because the
total lattice charge is additive and elements of
$\sqrt{k}\Lambda_\fg$ map to $\sqrt{k}\Lambda_\fg$, so the coset
class is preserved.

\emph{Step 3: Module count.}
For the simply-laced root lattice $\Lambda_\fg$, the dual lattice
$\Lambda_\fg^*$ satisfies
$\Lambda_\fg^*/\Lambda_\fg \cong Z(\widetilde{G})$, the center of
the simply connected group. At level $k = 1$:
$|\Lambda_\fg/\sqrt{1}\Lambda_\fg| = |\Lambda_\fg^*/\Lambda_\fg|$,
which equals the number of integrable highest-weight modules of
$\AffKM{g}$ at level~1 (these are the $|Z(\widetilde{G})|$
minuscule representations plus the vacuum). For general~$k$, the
count $k^{d/2} \cdot |Z(\widetilde{G})|$ agrees with the number
of integrable $\widehat{\fg}_k$-modules by the Kac--Walton
formula.
\end{proof}


\subsection{Quantum lattice algebras at roots of unity}

The level-$k$ factorization DK extends to the $\Eone$ setting
by incorporating the root-of-unity deformation.

\begin{theorem}[Quantum lattice factorization DK;
\ClaimStatusProvedHere]
\label{thm:lattice:quantum-factorization-dk}
\index{Drinfeld--Kohno!quantum lattice|textbf}
\index{quantum lattice algebra!factorization DK}
For $\fg$ simply-laced, $N \geq 3$, and $q \colon \Lambda_\fg
\times \Lambda_\fg \to \Z/N\Z$ a non-zero antisymmetric bilinear
form, the quantum lattice algebra
$\Vlat_{\Lambda_\fg}^{N,q}$
\textup{(}Definition~\textup{\ref{def:quantum-lattice})} satisfies:
\begin{enumerate}[label=(\roman*)]
\item \emph{$\Eone$-factorization bar-cobar}:
 \[
 \Omega^{\Eone,\mathrm{ch}}(
 \barB^{\Eone,\mathrm{ch}}(\Vlat_{\Lambda_\fg}^{N,q}))
 \xrightarrow{\;\sim\;}
 \Vlat_{\Lambda_\fg}^{N,q}
 \]
 as an equivalence of $\Eone$-factorization algebras on $\Ran(X)$.
\item \emph{Homotopy-level equivalence}: the
 $\Eone$-factorization $\infty$-category equivalence
 \begin{equation}\label{eq:lattice:quantum-fact-dk}
 \Factord(X; \Vlat_{\Lambda_\fg}^{N,q})
 \;\simeq\;
 \Factord(X; \Vlat_{\Lambda_\fg}^{N,-q})^{\mathrm{rev}}.
 \end{equation}
\item \emph{$\Eone$ inversion}: the equivalence realizes the
 $\Eone$ inversion principle
 \textup{(}Theorem~\textup{\ref{thm:e1-inversion-principle})}
 at the factorization-categorical level: the Koszul dual
 $(\Vlat_{\Lambda_\fg}^{N,q})^!
 \cong (\Vlat_{\Lambda_\fg}^{N,-q})^c$
 \textup{(}Proposition~\textup{\ref{prop:lattice:deformation-properties}(iv))}
 governs the Verdier dual factorization category.
\end{enumerate}
\end{theorem}

\begin{proof}
(i) is Corollary~\ref{cor:lattice:factorization-dk-level1}(iii),
already proved.

(ii) Apply
Theorem~\ref{thm:lattice:homotopy-factorization-dk} to $\Lambda =
\Lambda_\fg$ with cocycle $\varepsilon_{N,q}$. The lattice
$\Lambda_\fg$ is even and positive-definite, and
$\varepsilon_{N,q}$ is a valid cocycle
(Proposition~\ref{prop:lattice:deformation-properties}(i)). The
inverse cocycle is $\varepsilon_{N,q}^{-1} = \varepsilon_{N,-q}$
(Proposition~\ref{prop:lattice:deformation-properties}(iv)).
Theorem~\ref{thm:lattice:homotopy-factorization-dk}
gives~\eqref{eq:lattice:quantum-fact-dk}.

(iii) The Verdier identification
$\mathbb{D}_{\Ran}\, \barB^{\Eone}(\Vlat^{N,q})
\simeq (\Vlat^{N,-q})^!_\infty$
(Corollary~\ref{cor:lattice:factorization-dk-level1}(iii))
identifies the Verdier dual bar complex with the bar complex of
the Koszul dual. At the $\infty$-categorical level, this says
that the Verdier dual of
$\Factord(X; \Vlat^{N,q})$ is governed by the Koszul dual
$(\Vlat^{N,q})^! = (\Vlat^{N,-q})^c$, which is the
factorization-categorical realization of the $\Eone$ inversion
principle.
\end{proof}


\subsection{Bridge to Yangian derived Koszul duality}

The lattice family is the bridge between two worlds. On one side
stand the commutative/$E_\infty$-chiral families
(Heisenberg, lattice, free fermion) where every Koszul pair is
unconditional and the full factorization-categorical DK equivalence
is proved. On the other side stand the associative/$E_1$-chiral
families (Yangians, RTT algebras) where DK involves spectral
parameters, $R$-matrices, and the full braided structure of quantum
groups. The following two propositions are the explicit hinges of
the bridge: they show that the lattice DK equivalence, at level~$1$
and for simply-laced $\fg$, is compatible with the Frenkel--Kac--Segal
identification and therefore implies the Yangian DK equivalence on
the evaluation locus.

\begin{proposition}[Lattice--Yangian DK bridge at level \texorpdfstring{$1$}{1};
\ClaimStatusProvedHere]
\label{prop:lattice:yangian-bridge-level1}
\index{Drinfeld--Kohno!lattice--Yangian bridge}
For $\fg$ simply-laced, the following diagram of equivalences
commutes:
\begin{equation}\label{eq:lattice:yangian-bridge}
\begin{tikzcd}[column sep=3.5em]
\Factord(X; \Vlat_{\Lambda_\fg}^{\varepsilon_0})
 \ar[r, "\sim"', "\text{Thm~\ref{thm:lattice:homotopy-factorization-dk}}"]
 \ar[d, "\cong"', "\text{FKS}"]
& \Factord(X; \Vlat_{\Lambda_\fg}^{\varepsilon_0})^{\mathrm{rev}}
 \ar[d, "\cong", "\text{FKS}"'] \\
\Factord(X; L(\AffKM{g}, 1))
 \ar[r, "\sim", "\text{DK}"']
& \Factord(X; L(\AffKM{g}, 1))^{\mathrm{rev}}
\end{tikzcd}
\end{equation}
where the vertical isomorphisms are the Frenkel--Kac--Segal
identification
\textup{(}Theorem~\textup{\ref{thm:lattice:frenkel-kac})}.
The bottom equivalence is the level-$1$ case of the Yangian DK
theorem \textup{(}Theorem~\textup{\ref{conj:dk-fd-typeA}} for
$k = 1$\textup{)}.
\end{proposition}

\begin{proof}
The FKS identification $\Vlat_{\Lambda_\fg}^{\varepsilon_0}
\cong L(\AffKM{g}, 1)$ is an isomorphism of vertex algebras,
hence induces an equivalence of factorization module categories.
The top equivalence is
Theorem~\ref{thm:lattice:homotopy-factorization-dk} for the
unimodular case (where $\varepsilon_0^{-1} = \varepsilon_0$
by the FLM symmetry, so $(\cdots)^{\mathrm{rev}}$ acts on the
same algebra).

Commutativity of the square follows from the functoriality of the
bar-cobar construction: the bar complex of $\Vlat^{\varepsilon_0}$
maps isomorphically to the bar complex of $L(\AffKM{g}, 1)$ under
FKS, and the Verdier intertwining commutes with this isomorphism
because FKS preserves the cocycle structure.

The bottom equivalence is therefore an independent proof of
DK-0/1 for level-$1$ KM algebras, obtained without the RTT
generators or thick-generation arguments of the Yangian approach.
For $\fg = \mathfrak{sl}_N$, this recovers the $k = 1$ case of
Theorem~\ref{conj:dk-fd-typeA}.
\end{proof}

\section{Genus-$1$ sectorwise bar complex and curvature-braiding orthogonality}
\label{sec:lattice:genus1-sectorwise}

The sectorwise structure that makes lattice DK unconditional at
genus~$0$ persists at genus~$1$. The result is a clean factorization
of the genus-$1$ bar complex into modular (Cartan) and braiding (root)
components, proving that the $\Eone$-chiral genus theory for lattice
algebras is a \emph{deformation of the $\Einf$ theory in the root sectors
only}.

\subsection{The genus-$1$ propagator preserves simple-pole residues}

On an elliptic curve $E_\tau = \bC/(\bZ + \tau\bZ)$, the genus-$0$
propagator $\eta_{ij}^{(0)} = d\log(z_i - z_j)$ is replaced by
\[
\eta_{ij}^{(1)} = \Bigl[
\zeta_\tau(z_{ij}) + \frac{\pi}{\mathrm{Im}(\tau)}\,\mathrm{Im}(z_{ij})
\Bigr](dz_i - dz_j),
\]
where $\zeta_\tau$ is the Weierstrass $\zeta$-function. The non-holomorphic
correction $\frac{\pi}{\mathrm{Im}(\tau)}\mathrm{Im}(z_{ij})$ ensures
well-definedness on $E_\tau$ at the cost of breaking holomorphicity.

\begin{proposition}[Simple-pole residues unchanged at genus~$1$;
\ClaimStatusProvedHere]\label{prop:lattice:genus1-simple-pole}
The residue of $\eta_{ij}^{(1)}$ at $z_i = z_j$ equals~$1$, the same as
genus~$0$. Consequently, all OPE coefficients extracted from simple poles
(i.e., all vertex-operator OPEs
$e^{\alpha_i}(z)\,e^{\alpha_j}(w)$ with $\langle\alpha_i,\alpha_j\rangle = -1$)
are \emph{unchanged} at genus~$1$.
\end{proposition}
\begin{proof}
The Weierstrass $\zeta$-function has Laurent expansion
$\zeta_\tau(z) = z^{-1} + O(z)$, so $\Res_{z=0}\,\zeta_\tau(z)\,dz = 1$.
The correction $\frac{\pi}{\mathrm{Im}(\tau)}\mathrm{Im}(z)$ is regular
at $z = 0$, contributing zero to the residue.
\end{proof}

\subsection{Sectorwise curvature: the orthogonality theorem}

\begin{theorem}[Curvature-braiding orthogonality for quantum lattice
VOAs; \ClaimStatusProvedHere]\label{thm:lattice:curvature-braiding-orthogonal}
\index{curvature!lattice genus-1}
\index{lattice vertex algebra!genus-1 curvature}
For a quantum lattice algebra $V_\Lambda^{N,q}$ on~$E_\tau$:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Cartan sector.}
 In the sector $\gamma = 0$ (Heisenberg/Cartan generators), the genus-$1$
 bar differential satisfies
 $\dfib^{\,2} = \operatorname{rank}(\Lambda) \cdot \omega_1$,
 where $\omega_1 = c_1(\mathbb{E}) \in H^2(\overline{\mathcal{M}}_1)$
 is the first Chern class of the Hodge bundle.
\item \emph{Root sectors.}
 In every adjacent-root sector
 $\gamma = \alpha_i + \alpha_j$ with $\langle\alpha_i,\alpha_j\rangle = -1$,
 the genus-$1$ bar differential satisfies $\dfib^{\,2} = 0$.
 The $\Eone$ ordering cycles persist unchanged from genus~$0$.
\item \emph{Modular characteristic.}
 $\kappa(V_\Lambda^{N,q}) = \operatorname{rank}(\Lambda)$,
 \emph{independent} of the deformation parameters $(N,q)$.
\item \emph{Genus-$1$ free energy.}
 $F_1(V_\Lambda^{N,q}) = \operatorname{rank}(\Lambda)/24$,
 matching the Heisenberg genus-$1$ free energy at level~$1$
 summed over $\operatorname{rank}(\Lambda)$ independent bosons.
\item \emph{Complementarity.} For the lattice-VOA family the pairing is
 $\kappa_{\mathrm{latt}}(V_\Lambda^{N,q}) + \kappa_{\mathrm{latt}}((V_\Lambda^{N,q})^!) = 0_{\text{(lattice,\\,type\,i)}}$:
 the combined bar complex at genus~$1$ is an honest cochain complex.
\end{enumerate}
\end{theorem}
\begin{proof}
The bar complex $\barB_{E_\tau}(V_\Lambda^{N,q})$ decomposes as a
direct sum $\bigoplus_{\gamma \in \Lambda} \barB_\gamma$ indexed by
the lattice vector $\gamma = \alpha_1 + \cdots + \alpha_n$ of the
bar element $[e^{\alpha_1} | \cdots | e^{\alpha_n}]$. This
decomposition persists at genus~$1$ because the lattice~$\Lambda$ is
constant over $\overline{\mathcal{M}}_{1,1}$: the fiber of each sector
does not jump.

\emph{Cartan sector} ($\gamma = 0$).
The only OPEs contributing to the bar differential involve the Cartan
generators $h_a(z)\,h_b(w) \sim A_{ab}/(z-w)^2$, which have double
poles. The genus-$1$ propagator on $E_\tau$ replaces
$d\log(z_i - z_j)$ by the Arakelov Green's function, and the modified
Arnold relation $\mathcal{A}_3^{(1)} = 2\pi i \cdot
\omega_{\mathrm{vol}}^{(1)} \neq 0$
produces the curvature $\dfib^{\,2} = \sum_{a=1}^r 1 \cdot \omega_1
= r \cdot \omega_1$ by
Theorem~\ref{thm:frame-genus1-curvature} applied independently to
each of the~$r$ Cartan bosons. The deformation cocycle
$\varepsilon_{N,q}$ does not enter: the Cartan OPEs are determined by
the inner product $\langle \alpha_i, \alpha_j \rangle$ alone.

\emph{Root sectors}
($\gamma = \alpha_i + \alpha_j$, $\langle\alpha_i,\alpha_j\rangle = -1$).
The bar differential extracts the residue of the simple pole
$e^{\alpha_i}(z)\,e^{\alpha_j}(w) \sim
\varepsilon(\alpha_i,\alpha_j)\,e^{\alpha_i+\alpha_j}(w)/(z-w)$.
By Proposition~\ref{prop:lattice:genus1-simple-pole}, the simple-pole
residue is unchanged at genus~$1$. Therefore
$\dfib^{(1)} = \dfib^{(0)}$ in each root sector, and in particular
$(\dfib^{(1)})^2 = 0$. The $\Eone$ ordering cycle
$\xi = [e^{\alpha_i}|e^{\alpha_j}]
- \lambda\,[e^{\alpha_j}|e^{\alpha_i}]$
(with
$\lambda = -\varepsilon_{N,q}(\alpha_i,\alpha_j) /
\varepsilon_{N,q}(\alpha_j,\alpha_i)$)
remains a cocycle, and its cohomology class is unchanged.

\emph{Independence from $(N,q)$}.
The curvature~$r \cdot \omega_1$ lives entirely in the Cartan sector,
which depends only on the Cartan matrix $A_{ab} = \langle\alpha_i,
\alpha_j\rangle$. The cocycle $\varepsilon_{N,q}$ enters only the
root sectors, where $\dfib^{\,2} = 0$. Hence
$\kappa(V_\Lambda^{N,q}) = r$ for all $(N,q)$.

\emph{Complementarity}.
The Koszul dual $(V_\Lambda^{N,q})^!$ has Cartan level $-1$ per boson
(the sign flip from Verdier duality on configuration spaces), giving
$\kappa((V_\Lambda^{N,q})^!) = -r$. In the lattice-VOA sector
, free-field / $\mathrm{Ext}\leftrightarrow\mathrm{Sym}$
duality, distinct from the Virasoro value~$13$) the pairing vanishes:
$\kappa_{\mathrm{lattice}}(V_\Lambda^{N,q}) + \kappa_{\mathrm{lattice}}((V_\Lambda^{N,q})^!) = 0$ \end{proof}

\begin{remark}[The orthogonality principle]\label{rem:lattice:orthogonality-principle}
The factorization
$B^{(1)}(V_\Lambda^{N,q}) \simeq
B^{(1)}_{\mathrm{Cartan}}(\mathcal{H}_r) \otimes
B^{(0)}_{\mathrm{roots}}(V_\Lambda^{N,q})$
expresses a structural principle: on lattice VOAs, the \emph{modular axis}
(genus tower, curvature, $\hat{A}$-genus) and the \emph{braiding axis}
($R$-matrix, cocycle, ordering cycles) are \emph{orthogonal}. The modular
axis sees only the rank; the braiding axis sees only the cocycle. This
separation may fail for the Yangian
(Conjecture~\ref{conj:e1-theorem-D}), where the $R$-matrix residue
$-\hbar P$ could contribute to the genus-$1$ curvature directly.
\end{remark}

\begin{corollary}[Lattice shadow obstruction tower:
termination at weight~$2$; \ClaimStatusProvedHere]
\label{cor:lattice-postnikov-termination}
\index{shadow obstruction tower!lattice termination}
The shadow obstruction tower
\textup{(}Definition~\textup{\ref{def:shadow-postnikov-tower})}
of a quantum lattice VOA $V_\Lambda^{N,q}$ terminates at
weight~$2$:
$\Theta_{V_\Lambda}^{\leq r} = \kappa \cdot \eta \otimes \Lambda$
for all $r \geq 2$, with $o_{r+1}(V_\Lambda) = 0$ for all $r \geq 2$.
The lattice tower is Gaussian/formal:
curvature-braiding orthogonality
\textup{(}Theorem~\textup{\ref{thm:lattice:curvature-braiding-orthogonal})}
ensures that the cocycle twist does not generate higher shadow
corrections, so the tower terminates at the same level as the
Heisenberg tower but by formality of the deformation algebra rather
than by vanishing of higher operations.
\end{corollary}

\begin{proof}
The $L_\infty$-algebra $\Definfmod(V_\Lambda)$ is formal:
by Theorem~\ref{thm:lattice:curvature-braiding-orthogonal}, the
curvature and braiding decouple, so the transferred $L_\infty$ brackets
$\ell_k^{\mathrm{tr}}$ are all coboundaries for $k \geq 2$.
The shadow algebra $V_\Lambda^{\mathrm{sh}}$ therefore reduces to
$\mathbb{Q}[\kappa]$ with $\kappa = \operatorname{rank}(\Lambda)$.
Since all higher obstruction classes $o_{r+1}$ lie in the image of the
coboundary map, they vanish in the shadow algebra for $r \geq 2$.
\end{proof}

\begin{remark}[Completion kinematics of lattice VOAs]
\label{rem:lattice-completion-kinematics}
\index{completion kinematics!lattice VOA}
The lattice VOA $V_\Lambda$ of rank~$r = \operatorname{rank}(\Lambda)$
has vacuum character
$\chi_{V_\Lambda}(q) = \theta_\Lambda(q) / \eta(q)^r$, where
$\theta_\Lambda$ is the lattice theta function. At generic lattice,
the primitive generating series is $G_{V_\Lambda}(t) = r\,t/(1-t)$:
one primitive per derivative per basis vector, exactly as for
rank-$r$ Heisenberg. The Koszul radius is $\rho_K = 1/(r+1)$ and the
completion entropy $h_K(V_\Lambda) = \log(r+1)$. The primitive
defect $\Delta_{V_\Lambda}(t) = 0$: the lattice twist does not
create new quasi-primary cumulants (by curvature-braiding
orthogonality,
Theorem~\ref{thm:lattice:curvature-braiding-orthogonal}).

All lattice VOAs are \emph{exact Euler--Koszul}:
$W(V_\Lambda) = \{1^r\}$ (vertex operators have conformal weight~$1$),
so $\operatorname{ek}(V_\Lambda) = 0$ and
$S_{V_\Lambda}(u) = r\cdot\zeta(u)\zeta(u{+}1)$
(Definition~\ref{def:dirichlet-sewing-lift}).
The sewing reproducing kernel is a pure product of zeta functions,
independent of the lattice structure. The depth decomposition
$d(V_\Lambda) = 1 + d_{\mathrm{arith}}$ with
$d_{\mathrm{alg}} = 0$
(Theorem~\ref{thm:depth-decomposition}): the shadow obstruction tower
detects only the Hecke eigenform content of
$\Theta_\Lambda$, with no homotopy defect.

The reduced-weight windows $K_q(V_\Lambda)$ have dimension
$\sim (r+1)^q$, growing with the number of Heisenberg modes
at each level. The cocycle phase $\varepsilon_{N,q}$ enters the
differential on each window but does not change the window
dimensions; this is the kinematic reflection of curvature-braiding
orthogonality.
\end{remark}

\begin{remark}[Lattice shadow connection]
\label{rem:lattice-shadow-connection}
\index{lattice VOA!shadow connection}
For lattice vertex algebras $V_\Lambda$: $\kappa = \operatorname{rank}(\Lambda)$, $\alpha = 0$, $S_4 = 0$, $\Delta = 0$. The shadow metric $Q_{V_\Lambda} = (2\kappa)^2$ is \emph{constant}, the connection form $\omega = Q'/(2Q) = 0$ is trivial, and the parallel transport $\Phi(t) = 1$ is the identity. The shadow obstruction tower terminates at $r_{\max} = 2$: the lattice is the archetype of the Gaussian class~$\mathbf{G}$. The arithmetic shadow depth (from Hecke eigenforms in the partition function) is an independent invariant, governed by the Epstein zeta function rather than the shadow metric.
\end{remark}

\subsection{$\Eone$ chiral Hochschild cohomology for quantum lattice VOAs}
\label{subsec:lattice:e1-hochschild}

The $\Einf$ chiral Hochschild cohomology of lattice VOAs decomposes by lattice
sector (Theorem~\ref{thm:lattice:hochschild}). The curvature-braiding
orthogonality theorem
(Theorem~\ref{thm:lattice:curvature-braiding-orthogonal}) suggests the
same decomposition persists in the $\Eone$ setting, with the cocycle
$\varepsilon_{N,q}$ introducing phase factors sector by sector.

\begin{conjecture}[$\Eone$ lattice chiral Hochschild cohomology;
\ClaimStatusConjectured]\label{conj:lattice:e1-hochschild}
\index{Hochschild cohomology!E1-chiral lattice@$\Eone$-chiral lattice}
For a quantum lattice algebra $V_\Lambda^{N,q}$:
\begin{enumerate}[label=\textup{(\roman*)}]
\item The $\Eone$-chiral Hochschild cohomology decomposes by lattice sector:
 $\ChirHoch^n_{\Eone}(V_\Lambda^{N,q})
 = \bigoplus_{\alpha \in \Lambda}
 \ChirHoch^n_{\Eone}(V_\Lambda^{N,q})^\alpha$.
\item In the Cartan sector ($\alpha = 0$), the $\Eone$ and $\Einf$
 chiral Hochschild cohomologies are isomorphic
 (the Cartan generators are $\Einf$ regardless of the cocycle).
\item In non-zero sectors ($\alpha \neq 0$), the cocycle braiding
 $c_{N,q}(\alpha_i, \alpha_j) =
 (-1)^{\langle\alpha_i,\alpha_j\rangle} \zeta_N^{2q(\alpha_i,\alpha_j)}$
 introduces phase twists in the chiral Hochschild differential, changing
 cohomology dimensions relative to the $\Einf$ case.
\item The $N$-torsion of the braiding creates $\bZ/N\bZ$ periodicity
 in the sector-by-sector chiral Hochschild computation: sectors
 $\alpha$ and $\alpha + N\beta$ have isomorphic chiral Hochschild
 contributions.
\end{enumerate}
\emph{Evidence.}
Part (i) follows from the lattice grading. Part (ii) follows from
Theorem~\ref{thm:lattice:curvature-braiding-orthogonal}: the Cartan
sector is $\Einf$. Part (iii) is the $\Eone$ content: the ordering
cycles in the bar complex (which control chiral Hochschild via bar-cobar)
carry the cocycle phases. Part (iv) is structural (the braiding
matrix has $N$-torsion by Theorem~\ref{thm:quantum-lattice-structure}).
\end{conjecture}

\section{Lattice shadow obstruction tower: integrable spin chains and the
transfer matrix dictionary}%
\label{sec:lattice-shadow-spin-chains}
\index{lattice VOA!shadow tower}
\index{XXX spin chain!shadow tower}
\index{transfer matrix!shadow dictionary}
\index{integrable spin chain!shadow tower}

The lattice shadow obstruction tower connects the modular Koszul duality engine
to the most concrete physical systems in condensed matter physics:
integrable spin chains. The continuum limits of XXX, XXZ, and XYZ
chains are affine Kac--Moody or Virasoro algebras, and the shadow
tower data controls finite-size corrections to the ground state energy
through the Cardy formula and its higher-order analogues.

\subsection{XXX spin chain: the $V_1(\mathfrak{sl}_2)$ shadow}%
\label{subsec:xxx-shadow}

\begin{proposition}[XXX shadow data; \ClaimStatusProvedHere]%
\label{prop:xxx-shadow-data}
\index{XXX spin chain!shadow data}
\index{V1sl2@$V_1(\mathfrak{sl}_2)$!shadow data}
The isotropic Heisenberg antiferromagnet
$H = -J\sum_i \vec{S}_i \cdot \vec{S}_{i+1}$ has continuum limit
$V_1(\mathfrak{sl}_2)$ \textup{(}SU$(2)$ WZW at level $k = 1$\textup{)}
with central charge $c = 1$. The shadow data decomposes into two
channels:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{lcccc}
\toprule
Channel & $\kappa$ & $\alpha$ & $S_4$ & Class \\
\midrule
Current line \textup{(}$J^a$ directions\textup{)} & $9/4$ & $1$ & $0$ & L \\
$T$-line \textup{(}Sugawara\textup{)} & $1/2$ & $2$ & $10/27$ & M \\
\bottomrule
\end{tabular}
\end{center}
On the current line, $\kappa = \dim(\mathfrak{sl}_2)(k+h^\vee)/(2h^\vee)
= 3 \cdot 3/4 = 9/4$ \textup{(}not $c/2 = 1/2$; cf.\ Table~\textup{\ref{tab:master-invariants}}\textup{)},
and the Jacobi identity kills the quartic, giving class~L with
$r_{\max} = 3$. On the $T$-line, the Sugawara tensor has the
Virasoro shadow at $c = 1$, class~M with infinite shadow depth.
\end{proposition}

\begin{proof}
The current-line data follows from the affine
$\mathfrak{sl}_2$ shadow theorem
(Chapter~\ref{chap:kac-moody},
Theorem~\ref{thm:affine-cubic-normal-form}): $\dim = 3$,
$h^\vee = 2$, $k = 1$, so $\kappa = 3 \cdot 3/4 = 9/4$,
$\alpha = 1$ (from the Lie bracket $f^{ab}{}_c J^c$),
$S_4 = 0$ (Jacobi identity).
The $T$-line data: $\kappa_T = c/2 = 1/2$, $\alpha_T = 2$
(universal for Virasoro), $S_4 = 10/(c(5c+22)) = 10/27$.
\end{proof}

\begin{proposition}[Transfer matrix--shadow dictionary; \ClaimStatusProvedHere]%
\label{prop:transfer-matrix-shadow-dict}
\index{transfer matrix!shadow dictionary}
\index{conserved charges!shadow coefficients}
For the XXX chain with Yang's rational $R$-matrix
$R(u) = u \cdot \mathrm{Id} + P$ \textup{(}where $P$ is the
permutation operator\textup{)}, the conserved charges $I_n$
generated by $\log T(u) = \sum_n I_n u^{n-1}/(n-1)!$ correspond to
shadow obstruction tower coefficients on the current line:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{ll}
\toprule
Transfer matrix & Shadow obstruction tower \\
\midrule
$I_1$ \textup{(momentum)} & Translation on $\mathrm{Ran}(X)$ \\
$I_2$ \textup{(Hamiltonian)} & $\kappa = 9/4$ \textup{(modular characteristic)} \\
$I_3$ \textup{(third charge)} & $\alpha = 1$ \textup{(cubic shadow, Lie bracket)} \\
$I_n$, $n \ge 4$ \textup{(higher charges)} & $S_n = 0$ for $n \ge 4$ \textup{(class L: shadow terminates)} \\
\bottomrule
\end{tabular}
\end{center}
The classical $r$-matrix is $r(z) = k\Omega/z$ \textup{(}simple pole,
one order below the OPE double pole by $d\log$ extraction\textup{)}.
\end{proposition}

\begin{proof}
The correspondence is structural: the leading conserved charges of the
integrable hierarchy are moments of the transfer matrix, while the
shadow obstruction tower coefficients are moments of the shadow generating function.
The shadow obstruction tower terminates at degree~$3$ because the Jacobi identity
kills the quartic shadow coefficient ($S_4 = 0$).
The higher conserved charges $I_n$ for $n \ge 4$ are nonzero
(generating the Yangian $Y(\mathfrak{sl}_2)$), but they lie
outside the shadow obstruction tower: they contribute to the full
$L_\infty$-algebra $\Definfmod(V_1(\mathfrak{sl}_2))$, not to
the scalar shadow projections $S_n$.
The $r$-matrix identification $r(z) = k\Omega/z$ follows from the
collision residue computation: the OPE
$J^a(z)J^b(w) \sim k\kappa^{ab}/(z-w)^2 + f^{ab}{}_c J^c(w)/(z-w)$
yields, after $d\log$ extraction,
$r(z) = k\Omega/z$: the double pole $k\kappa^{ab}/(z{-}w)^2$
shifts to $k\kappa^{ab}/(z{-}w)$ and the simple pole becomes
regular and drops Here $\Omega = \sum_a I^a \otimes I_a$ is the quadratic Casimir.
\end{proof}

\subsection{XXZ spin chain: the minimal model shadow}%
\label{subsec:xxz-shadow}

\begin{proposition}[XXZ shadow data; \ClaimStatusProvedHere]%
\label{prop:xxz-shadow-data}
\index{XXZ spin chain!shadow data}
\index{minimal model!from XXZ chain}
The anisotropic Heisenberg chain
$H = -\sum_i(S_i^x S_{i+1}^x + S_i^y S_{i+1}^y
+ \Delta_{\mathrm{anis}} S_i^z S_{i+1}^z)$
with $\Delta_{\mathrm{anis}} = \cos(\pi\nu)$, $0 < \nu < 1$,
has continuum limit a compactified boson with
$c = 1 - 6(1-\nu)^2/\nu$. At the rational coupling
$\nu = p/(p+1)$, the continuum limit is the unitary minimal model
$\mathcal{M}(p, p+1)$ with shadow data on the Virasoro $T$-line:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{ccccc}
\toprule
$p$ & $c$ & Model & $\kappa = c/2$ & $\rho$ \\
\midrule
$3$ & $1/2$ & Ising & $1/4$ & $\approx 12.5$ \\
$4$ & $7/10$ & Tricritical Ising & $7/20$ & $\approx 8.9$ \\
$5$ & $4/5$ & $3$-state Potts & $2/5$ & $\approx 7.8$ \\
$6$ & $6/7$ & -- & $3/7$ & $\approx 7.3$ \\
\bottomrule
\end{tabular}
\end{center}
All have class~M on the $T$-line, with shadow growth rate $\rho$
decreasing monotonically as $c$ increases toward~$1$.
\end{proposition}

\begin{proof}
At $\nu = p/(p+1)$: $c = 1 - 6(1-p/(p+1))^2/(p/(p+1))
= 1 - 6/(p(p+1))$ (the standard Kac formula).
The shadow data on the $T$-line is the universal Virasoro
shadow at this value of~$c$: $\kappa = c/2$, $\alpha = 2$,
$S_4 = 10/[c(5c+22)]$. The growth rate $\rho$ is computed
from $\rho^2 = (9\alpha^2 + 2\Delta)/(4\kappa^2)$ with
$\Delta = 8\kappa S_4 = 40/(5c+22)$.
\end{proof}

\subsection{XYZ spin chain: quasi-modular shadow}%
\label{subsec:xyz-shadow}

\begin{remark}[Quasi-modular structure of the XYZ shadow]%
\label{rem:xyz-quasi-modular}
\index{XYZ spin chain!quasi-modular shadow}
\index{quasi-modular form!XYZ chain}
The fully anisotropic XYZ chain
$H = \sum_i(J_x S_i^x S_{i+1}^x + J_y S_i^y S_{i+1}^y
+ J_z S_i^z S_{i+1}^z)$ has a central charge determined by
the elliptic modular parameter $\tau = iK'/K$ (where $K$, $K'$
are complete elliptic integrals). The genus-$1$ shadow
amplitudes involve the quasi-modular Eisenstein series $E_2^*(\tau)$:
the propagator of the genus-$1$ bar complex is
$\partial_z \log \theta_1(z|\tau) = E_2^*(\tau) \cdot z + \cdots$,
and graph sums produce quasi-modular polynomials in
$\{E_2^*, E_4, E_6\}$, \emph{not} holomorphic modular forms.
The shadow obstruction tower of the XYZ chain is therefore a function on
the upper half-plane taking values in the ring of quasi-modular
forms, a structure that reduces to the trigonometric (XXZ) and
rational (XXX) shadows in the appropriate limits $\tau \to i\infty$
and $\tau \to 0$.
\end{remark}

\subsection{Finite-size corrections and the shadow hierarchy}%
\label{subsec:finite-size-shadow}

\begin{proposition}[Shadow hierarchy and Cardy corrections;
\ClaimStatusProvedHere]%
\label{prop:shadow-cardy-corrections}
\index{Cardy formula!shadow corrections}
\index{finite-size corrections!shadow hierarchy}
For a critical spin chain of length~$L$ with continuum limit
a chiral algebra~$\mathcal{A}$, the finite-size energy eigenvalues
are:
\begin{equation}\label{eq:cardy-shadow}
E_n(L) \;=\; e_\infty \cdot L
\;+\; \frac{2\pi v_F}{L}\Bigl(x_n - \frac{c}{24}\Bigr)
\;+\; O(1/L^2),
\end{equation}
where $x_n$ is the scaling dimension, $v_F$ is the Fermi velocity,
and the $O(1/L^2)$ corrections encode the shadow obstruction tower data.
The genus-$1$ free energy
$F_1(\mathcal{A}) = \kappa(\mathcal{A})/24$ determines the leading
finite-size correction to the extensive ground state energy, and
the higher shadow coefficients $S_r$ for $r \ge 3$ control
subleading corrections involving the higher conserved charges.
\end{proposition}

\begin{proof}
The Cardy formula $E_n(L) = e_\infty L + 2\pi v_F(x_n - c/24)/L
+ \cdots$ is standard
(Bl\"ote--Cardy--Nightingale 1986, Affleck 1986).
The identification of $F_1 = \kappa/24$ with the first subleading
correction is the genus-$1$ shadow amplitude at the scalar level
(Theorem~D). The higher-degree corrections at genus~$1$ involve
$S_3, S_4, \ldots$ through the planted-forest formula.
\end{proof}
