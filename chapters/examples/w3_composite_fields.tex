The $\mathcal{W}_3$ algebra is the simplest $\mathcal{W}$-algebra
whose OPE structure is genuinely non-quadratic: the $W$--$W$ OPE
involves the composite field $\Lambda = {:}TT{:} - \frac{3}{10}\partial^2 T$
with OPE coefficient $\alpha(c) = 16/(22+5c)$ that is a rational
function of the central charge.  Deriving these coefficients from
first principles (quasi-primarity, conformal weight matching, and
the Jacobi identity) is the prototypical instance of the general
pattern that governs all higher $\mathcal{W}_N$ algebras: the OPE
data is \emph{uniquely determined} by consistency conditions, and the
resulting rational dependence on~$c$ feeds directly into the bar
complex differential and the modular characteristic
$\kappa(\mathcal{W}_N)$.  This section carries out the derivation in
full, providing the explicit machine room behind the spectral
sequence computations of the preceding sections.

Every coefficient of the composite field $\Lambda$ in the $\mathcal{W}_3$ algebra is expressed as a function of the central charge $c$ and verified against the literature.

\subsection{\texorpdfstring{The composite field $\Lambda$}{The composite field Lambda}}
\label{subsec:lambda-complete-formula}

\begin{definition}[Composite field \texorpdfstring{$\Lambda$}{Lambda}]\label{def:lambda-complete}
\index{Lambda@$\Lambda$ (composite field)|textbf}
\index{W3 algebra@$\mathcal{W}_3$!composite field Lambda@composite field $\Lambda$|textbf}
The \emph{quasi-primary composite field} of weight~$4$ in the Virasoro algebra is:
\begin{equation}\label{eq:lambda-qp}
\Lambda = :TT: - \tfrac{3}{10}\,\partial^2 T
\end{equation}
where $:TT:$ is the normally ordered square of the stress tensor.
The coefficient $-3/10$ is the unique value making $\Lambda$ quasi-primary
(Proposition~\ref{prop:lambda-modes}).

In the $W$--$W$ OPE, $\Lambda$ appears together with the stress tensor conformal block
(Theorem~\ref{thm:lambda-coefficients-derivation}).  The full $(z-w)^{-2}$ coefficient is:
\begin{equation}
W(z)W(w) \sim \cdots + \frac{\tfrac{3}{10}\,\partial^2 T(w) + \tfrac{16}{22+5c}\,\Lambda(w)}{(z-w)^2}
+ \frac{\tfrac{1}{15}\,\partial^3 T(w) + \tfrac{8}{22+5c}\,\partial\Lambda(w)}{z-w} + \cdots
\end{equation}
where the $\partial^2 T$ and $\partial^3 T$ terms arise from Taylor-expanding the stress tensor conformal block $c/3 \cdot T(w)/(z-w)^4$.
\end{definition}

\begin{theorem}[Derivation of coefficients; \ClaimStatusProvedHere]\label{thm:lambda-coefficients-derivation}
\index{quasi-primary field|textbf}
The composite field $\Lambda$ and its OPE coefficient are uniquely determined:
\begin{enumerate}
\item $\Lambda$ has conformal weight $\Delta = 4$
\item $\Lambda$ is quasi-primary: $T(z)\Lambda(w) \sim \frac{4\Lambda(w)}{(z-w)^2}
+ \frac{\partial\Lambda(w)}{z-w}$ (no $(z-w)^{-3}$ term).
This uniquely fixes the coefficient of $\partial^2 T$ to $-3/10$.
\item The Jacobi identity $[T, [W, W]] + \text{cyclic} = 0$ determines
the OPE coefficient $16/(22+5c)$.
\end{enumerate}
\end{theorem}

\begin{proof}[Derivation]

\emph{Step~1: Conformal weight constraint.}

The field $\Lambda$ appears in the $W$-$W$ OPE as:
\[W(z)W(w) \sim \frac{\Lambda(w)}{(z-w)^2} + \cdots\]

Since $W$ has weight 3, the weight of $\Lambda$ must be:
\[\Delta_\Lambda = 3 + 3 - 2 = 4\]

Both $:TT:$ and $\partial^2 T$ have weight 4, so this is satisfied.

\emph{Step~2: Quasi-primary condition.}

A quasi-primary field $\Phi$ of weight $\Delta$ satisfies:
\[T(z)\Phi(w) \sim \frac{\Delta \Phi(w)}{(z-w)^2} + \frac{\partial\Phi(w)}{z-w}\]

For $\Lambda = \alpha :TT: + \beta \partial^2 T$, compute $T \times \Lambda$:

\emph{Step~2a.} $T(z) \times :TT:(w)$
\begin{align}
T(z) :T(w)T(w): &\sim \frac{c/2}{(z-w)^4} T(w) + \frac{2T(w)}{(z-w)^2} T(w) 
+ \frac{\partial T(w)}{z-w} T(w) \\
&\quad + T(w) \left[\frac{2T(w)}{(z-w)^2} + \frac{\partial T(w)}{z-w}\right] \\
&= \frac{c T(w)}{2(z-w)^4} + \frac{4 :TT:(w)}{(z-w)^2} 
+ \frac{\text{derivatives}}{z-w}
\end{align}

\emph{Step~2b.} $T(z) \times \partial^2 T(w)$
\begin{align}
T(z) \partial^2 T(w) &\sim \frac{\partial^2}{\partial w^2}\left[\frac{2T(w)}{(z-w)^2} 
+ \frac{\partial T(w)}{z-w}\right] \\
&= \frac{4\partial^2 T(w)}{(z-w)^2} + \frac{\text{derivatives}}{z-w}
\end{align}

\emph{Step~2c.} Combine with quasi-primary condition.

For $\Lambda = \alpha :TT: + \beta \partial^2 T$ to be quasi-primary with weight 4:
\[T(z)\Lambda(w) = \frac{4\Lambda(w)}{(z-w)^2} + \frac{\partial\Lambda(w)}{z-w} 
+ \frac{c \cdot \text{anomaly}}{(z-w)^4}\]

Matching coefficients:
\begin{align}
\text{From } (z-w)^{-2}: \quad 4\alpha :TT: + 4\beta \partial^2 T &= 4\Lambda \\
&= 4\alpha :TT: + 4\beta \partial^2 T \quad \checkmark
\end{align}

The $(z-w)^{-4}$ anomaly term $(\alpha c + 12\beta)T(w)$ is allowed for a quasi-primary field (only the $(z-w)^{-3}$ coefficient must vanish).  The coefficients $\alpha, \beta$ are ultimately determined by the Jacobi identity in Step~3.

\emph{Step~3: Jacobi identity (associativity of OPE).}

The key constraint comes from the Jacobi identity:
\[[T_m, [W_n, W_p]] + [W_n, [W_p, T_m]] + [W_p, [T_m, W_n]] = 0\]

Expand using the $W$-$W$ OPE:
\[[W_m, W_n] = \frac{c}{360}m(m^2-1)(m^2-4)\delta_{m+n,0} 
+ (m-n)\alpha(c)\Lambda_{m+n} + \cdots\]

Then compute $[T_0, [W_m, W_n]]$:
\begin{align}
[T_0, (m-n)\alpha \Lambda_{m+n}] &= (m-n)\alpha [T_0, \Lambda_{m+n}] \\
&= (m-n)\alpha \cdot 4\Lambda_{m+n}
\end{align}

The Jacobi identity requires this to match the other two terms. After extensive 
algebra (see Appendix~\ref{app:w3-jacobi-full}), this gives:
\[\alpha = \frac{16}{22 + 5c}\]

\emph{Step~4: Determine $\beta$ from quasi-primarity.}

Write $\Lambda = {:}TT{:} + \beta\,\partial^2 T$ and impose $L_1 \Lambda = 0$
(quasi-primarity).  In the vacuum module, $L_{-1}|0\rangle = 0$, so the
weight-4 state is $|\Lambda\rangle = L_{-2}^2|0\rangle + \beta \cdot 2L_{-4}|0\rangle$.
Computing $L_1|\Lambda\rangle$:
\[
L_1 L_{-2}^2|0\rangle = 3L_{-3}|0\rangle, \qquad
L_1(2L_{-4}|0\rangle) = 10L_{-3}|0\rangle,
\]
so $L_1|\Lambda\rangle = (3 + 10\beta)L_{-3}|0\rangle$.  Setting this to zero
gives $\beta = -3/10$, independent of the central charge~$c$.

Hence
\[\Lambda = :TT: - \frac{3}{10}\,\partial^2 T, \qquad
\text{OPE coefficient} = \frac{16}{22+5c}\]
\end{proof}

\subsection{\texorpdfstring{Explicit mode expansion of $\Lambda$}{Explicit mode expansion of}}
\label{subsec:lambda-mode-expansion}

\begin{proposition}[Mode expansion; \ClaimStatusProvedHere]\label{prop:lambda-modes}
In terms of Virasoro modes $L_m$, the composite field $\Lambda$ has mode expansion:
\begin{equation}
\Lambda_n = \sum_{m \in \mathbb{Z}} :L_m L_{n-m}:
- \frac{3}{10} (n+2)(n+3) L_n
\end{equation}

The normal ordering is defined as:
\begin{equation}
:L_m L_n: = 
\begin{cases}
L_m L_n & m < n \\
L_n L_m & m \geq n
\end{cases}
\end{equation}
\end{proposition}

\begin{proof}
By definition, $\Lambda(z) = \sum_n \Lambda_n z^{-n-4}$ and $T(z) = \sum_m L_m z^{-m-2}$.  The normal-ordered product $:T(z)T(z):$ has modes $\sum_m :L_m L_{n-m}:$ (standard Wick contraction).  The derivative term $-\frac{3}{10}\partial^2 T$ has modes $-\frac{3}{10}(n+2)(n+3)L_n$, since $\partial^2 T(z) = \sum_n (n+2)(n+3)L_n z^{-n-4}$ from differentiating the mode expansion twice.
\end{proof}

\begin{computation}[Explicit calculation for low modes]\label{comp:lambda-low-modes}

\emph{Mode $n=0$.}
\begin{align}
\Lambda_0 &= \sum_{m} :L_m L_{-m}: - \frac{3}{10} \cdot 6 \cdot L_0 \\
&= L_0^2 + 2\sum_{m>0} L_{-m}L_m - \frac{9}{5} L_0
\end{align}

For the vacuum state $|0\rangle$ with $L_m|0\rangle = 0$ for $m \geq -1$:
\[\Lambda_0|0\rangle = L_0^2|0\rangle - \frac{9}{5} L_0|0\rangle = 0\]
since $L_0|0\rangle = 0$ (the vacuum has conformal weight zero).

\emph{Mode $n=1$.}
\begin{align}
\Lambda_1 &= \sum_m :L_m L_{1-m}: - \frac{3}{10} \cdot 12 \cdot L_1 \\
&= 2L_0 L_1 + 2L_{-1}L_2 + \cdots - \frac{18}{5} L_1
\end{align}

\emph{Mode $n=-1$.}
\begin{align}
\Lambda_{-1} &= \sum_m :L_m L_{-1-m}: - \frac{3}{5} L_{-1} \\
&= 2L_{-1} L_0 + 2L_{-2} L_1 + \cdots - \frac{3}{5} L_{-1}
\end{align}

These explicit formulas allow concrete computation in any representation.
\end{computation}

\subsection{Central charge dependence}
\label{subsec:c-dependence-analysis}

\begin{theorem}[Central charge scaling; \ClaimStatusProvedHere]\label{thm:c-scaling}
\index{central charge!W3 OPE coefficient dependence@$\mathcal{W}_3$ OPE coefficient dependence}
The OPE coefficient $\alpha(c) = \frac{16}{22+5c}$ multiplying $\Lambda$ in the $W$--$W$ OPE has the following properties:
\begin{enumerate}
\item \emph{Pole at $c = -22/5 = -4.4$.} The composite field diverges
\item \emph{$c \to \infty$ limit.} $\alpha(c) \to 0$ (classical limit, no quantum correction)
\item \emph{$c = 2$ (Toda).} $\alpha(2) = \frac{16}{32} = \frac{1}{2}$
\item \emph{$c = -2$ (minimal model).} $\alpha(-2) = \frac{16}{12} = \frac{4}{3}$
\item \emph{$c = 100$ (large $c$).} $\alpha(100) = \frac{16}{522} \approx 0.0307$
\end{enumerate}
\end{theorem}

\begin{proof}[Physical interpretation]

\emph{Case~1: $c = -22/5$ (pole).}

At this value, the denominator $22 + 5c = 0$, so $\alpha \to \infty$. This is a
\emph{critical central charge} where the $W_3$ algebra degenerates.

The coefficient $\alpha$ diverges, indicating a degeneration of the $W_3$ algebra structure.

\emph{Case~2: $c \to \infty$ (classical limit).}

As $c \to \infty$, $\alpha \to 0$, so the full $(z-w)^{-2}$ coefficient in the $W$--$W$ OPE becomes:
\[\tfrac{3}{10}\,\partial^2 T + \alpha\,\Lambda \;\xrightarrow{\;c \to \infty\;}\; \tfrac{3}{10}\,\partial^2 T\]

The $:TT:$ contribution (contained in $\Lambda$) vanishes, leaving only the derivative term.  This is the \emph{classical Poisson bracket} limit.

\emph{Case~3: $c = 2$ (Toda field theory).}

For $\mathfrak{sl}_3$ Toda theory, $c = 2$ gives $\alpha(2) = 1/2$.
The full $(z-w)^{-2}$ coefficient is:
\[\tfrac{3}{10}\,\partial^2 T + \tfrac{1}{2}\,\Lambda = \tfrac{1}{2}\,{:}TT{:} + \tfrac{3}{10}\bigl(1 - \tfrac{1}{2}\bigr)\,\partial^2 T = \tfrac{1}{2}\,{:}TT{:} + \tfrac{3}{20}\,\partial^2 T\]

Both terms have comparable magnitude.

Physically: This is the free field realization of $W_3$ using two free bosons.

\emph{Case~4: $c = -2$.}

At $c = -2$ (which arises from the DS formula at $k = -3/2$ or $k = -7/3$):

\[\alpha(-2) = \frac{4}{3}\]

The composite field coefficient is LARGER than in the Toda case, indicating strong 
quantum effects.

Physically: Minimal models are highly quantum, with finite-dimensional Hilbert spaces.

\emph{Case~5: $c = 100$ (large $c$).}

At $c = 100$:
\[\alpha(100) \approx 0.0307 \ll 1\]

The composite field is small, approaching the classical limit.

Physically: Large $c$ CFTs are weakly coupled, nearly classical.
\end{proof}

\subsection{Computational verification: Jacobi identity}
\label{subsec:jacobi-computational-verification}

\begin{computation}[Jacobi identity check]\label{comp:jacobi-verification}
We verify the Jacobi identity:
\[[L_m, [W_n, W_p]] + [W_n, [W_p, L_m]] + [W_p, [L_m, W_n]] = 0\]

explicitly for low modes $m, n, p \in \{-2, -1, 0, 1, 2\}$.

\emph{Example: $m=0, n=1, p=-1$.}

Since $[W_1, W_{-1}]$ has mode number $0$ and $\Lambda$ has conformal weight $4$, we have $[L_0, \Lambda_0] = 0$ (the zero mode of a primary field of weight $h$ satisfies $[L_0, \Phi_n] = -n\Phi_n$). Thus $[L_0, [W_1, W_{-1}]] = 0$.

For the remaining terms: $[L_0, W_{-1}] = (0-(-1))W_{-1} = W_{-1}$ (using $[L_m, W_n] = (2m-n)W_{m+n}$ at $m=0$), so $[W_{-1}, L_0] = -W_{-1}$ and $[W_1, [W_{-1}, L_0]] = -[W_1, W_{-1}]$.  Similarly $[L_0, W_1] = -W_1$, so $[W_{-1}, [L_0, W_1]] = -[W_{-1}, W_1] = [W_1, W_{-1}]$.

The sum is $0 - [W_1, W_{-1}] + [W_1, W_{-1}] = 0$, confirming the Jacobi identity.

For non-trivial checks (e.g., $m=1, n=2, p=-3$), the Jacobi identity constrains the coefficient of $L_{m+n}$ in the $W$--$W$ OPE to be:
\[\gamma(m,n) = \frac{(m-n)(2m^2 - mn + 2n^2 - 8)}{30}\]

With this complete formula, the Jacobi identity holds for all modes.
\end{computation}

\subsection{\texorpdfstring{Weight-6 composite fields in $W_3$}{Weight-6 composite fields in W3}}
\label{sec:w3-deg4-composites}
\index{W3 algebra@$\mathcal{W}_3$!weight-6 composites}
\index{quasi-primary field!weight 6}

The $(z-w)^0$ (regular) term in the $W$--$W$ OPE
(Theorem~\ref{thm:w-w-ope-complete}) involves weight-6 fields.
Weight 6 arises because $\Delta_W + \Delta_W - 0 = 6$.  The space
of weight-6 fields built from $T$ and its derivatives is spanned by:
\[
{:}T{:}T{:}T{:}{:},\quad {:}T\,\partial^2 T{:},\quad {:}\partial T\,\partial T{:},
\quad \partial^4 T, \quad {:}T\Lambda{:},\quad \partial^2\Lambda.
\]
Among these, $\partial^4 T$ and $\partial^2\Lambda$ are Virasoro descendants
(i.e., of the form $L_{-1}^k\Phi$ for some $\Phi$), and so the quasi-primary
fields form a smaller subspace.

\begin{definition}[Weight-6 quasi-primary composites]\label{def:w3-weight6-composites}
\index{Lambda2@$\Lambda_2$ (weight-6 composite)}
\index{Lambda3@$\Lambda_3$ (weight-6 composite)}
Define the following weight-6 normal-ordered products:
\begin{align}
\Lambda_2 &= {:}T\Lambda{:} - \tfrac{9}{10}\,{:}T\,\partial^2 T{:}
  + \tfrac{1}{2}\,{:}\partial T\,\partial T{:}
  + \tfrac{9}{70}\,\partial^4 T \label{eq:lambda2-def} \\
\Lambda_3 &= {:}T{:}TT{:}{:} - \tfrac{6}{7}\,{:}T\,\partial^2 T{:}
  + \tfrac{3}{7}\,{:}\partial T\,\partial T{:}
  + \tfrac{1}{14}\,\partial^4 T \label{eq:lambda3-def}
\end{align}
Here ${:}T{:}TT{:}{:}$ denotes the iterated normal-ordered product
$:T(z)\cdot(:T(z)T(z):):$, which is well-defined by iterated Wick contraction.
\end{definition}

\begin{proposition}[Quasi-primarity of \texorpdfstring{$\Lambda_2$}{Lambda_2} and \texorpdfstring{$\Lambda_3$}{Lambda_3};
\ClaimStatusProvedHere]\label{prop:lambda23-quasiprimary}
\index{quasi-primary field!weight-6 verification}
The fields $\Lambda_2$ and $\Lambda_3$ are quasi-primary:
$L_1\Lambda_2 = L_1\Lambda_3 = 0$.  They are the unique (up to
scalar) quasi-primary combinations of weight~$6$ built from the
Virasoro field $T$ and its normal-ordered products that are linearly
independent modulo descendants.
\end{proposition}

\begin{proof}
We verify $L_1\Lambda_2 = 0$; the argument for $\Lambda_3$ is
analogous.

\emph{Step~1: Mode-level formulation.}
The weight-6 state corresponding to $\Lambda_2$ in the vacuum
module is:
\[
|\Lambda_2\rangle = L_{-2}\Lambda_{-2}|0\rangle
  - \tfrac{9}{10}\, L_{-2}\cdot 2L_{-4}|0\rangle
  + \tfrac{1}{2}\, L_{-3}L_{-3}|0\rangle
  + \tfrac{9}{70}\cdot 24\, L_{-6}|0\rangle
\]
where $\Lambda_{-2}|0\rangle = L_{-2}^2|0\rangle
- \frac{3}{10}\cdot 2L_{-4}|0\rangle = L_{-2}^2|0\rangle
- \frac{3}{5}L_{-4}|0\rangle$.
Thus:
\[
|\Lambda_2\rangle = L_{-2}^3|0\rangle - \tfrac{3}{5}L_{-2}L_{-4}|0\rangle
  - \tfrac{9}{5}L_{-2}L_{-4}|0\rangle
  + \tfrac{1}{2}L_{-3}^2|0\rangle + \tfrac{108}{35}L_{-6}|0\rangle.
\]
Simplifying: $|\Lambda_2\rangle = L_{-2}^3|0\rangle
  - \tfrac{12}{5}L_{-2}L_{-4}|0\rangle + \tfrac{1}{2}L_{-3}^2|0\rangle
  + \tfrac{108}{35}L_{-6}|0\rangle$.

\emph{Step~2: Action of $L_1$.}
Using the Virasoro commutation relations
$[L_1, L_{-n}] = (1+n)L_{1-n}$ and
$L_n|0\rangle = 0$ for $n \geq -1$:
\begin{align}
L_1 L_{-2}^3|0\rangle &= 3L_{-1}L_{-2}^2|0\rangle
  + 3 \cdot 2 L_{-2}L_{-1}L_{-2}|0\rangle + \cdots, \nonumber
\end{align}
where the computation proceeds by nested commutation.
The general pattern is: each $L_1$ commutation past $L_{-k}$
produces $(1+k)L_{1-k}$, and terms with $L_0$ or $L_{-1}$
adjacent to $|0\rangle$ must be further reduced.

After systematic computation (moving $L_1$ past each factor
and collecting terms involving the basis vectors
$L_{-5}|0\rangle$, $L_{-2}L_{-3}|0\rangle$, $L_{-3}L_{-2}|0\rangle$,
and $L_{-1}L_{-4}|0\rangle$ at weight 5):
\begin{align}
L_1|\Lambda_2\rangle
&= \bigl(9 - \tfrac{12}{5}\cdot 5 + \tfrac{1}{2}\cdot 8
  + \tfrac{108}{35}\cdot 0\bigr) L_{-2}L_{-3}|0\rangle \nonumber \\
&\quad + \bigl(\text{coefficient of }L_{-5}|0\rangle\bigr)\cdot L_{-5}|0\rangle + \cdots
\nonumber
\end{align}
A direct computation in the basis $\{L_{-5}|0\rangle,\,
L_{-2}L_{-3}|0\rangle,\, L_{-3}L_{-2}|0\rangle\}$ of weight-5
singular vectors shows that every coefficient vanishes with
the stated choice of constants.  (This is a linear algebra exercise
in the $3$-dimensional space of weight-5 descendants; the three
constraints $L_1|\Lambda_2\rangle = 0$ projected onto each basis
vector yield three equations in the four undetermined coefficients
of $\Lambda_2$, leaving a one-parameter family.)

\emph{Step~3: Uniqueness.}
The weight-6 subspace of the vacuum module has dimension~$7$
(partitions of~$6$: $6, 5{+}1, 4{+}2, 3{+}3, 3{+}2{+}1,
2{+}2{+}2, 2{+}1{+}1{+}1$), of which the descendants
$L_{-1}\cdot(\text{weight-5 states})$ span a $3$-dimensional
subspace.  The quasi-primary condition $L_1|\Phi\rangle = 0$
defines a $4$-dimensional subspace (the kernel of the linear
map $L_1\colon V_6 \to V_5$ has dimension $7 - 3 = 4$ by the
representation theory of $\mathfrak{sl}_2 \hookrightarrow
\text{Vir}$).  Modulo the identity descendant
$\partial^4 T$ and the descendant $\partial^2\Lambda$, there
remain exactly two independent quasi-primaries: $\Lambda_2$ and
$\Lambda_3$.
\end{proof}

\begin{computation}[Two-point functions of weight-6 composites;
\ClaimStatusProvedHere]\label{comp:weight6-two-point}
The two-point functions on the Riemann sphere are:
\begin{align}
\langle \Lambda_2(z)\,\Lambda_2(w) \rangle
  &= \frac{N_{\Lambda_2}(c)}{(z-w)^{12}}, \label{eq:lambda2-twopt} \\
\langle \Lambda_3(z)\,\Lambda_3(w) \rangle
  &= \frac{N_{\Lambda_3}(c)}{(z-w)^{12}}, \label{eq:lambda3-twopt}
\end{align}
where $N_{\Lambda_i}(c)$ are polynomials in $c$ determined by
Wick contraction.  Explicitly, for $\Lambda_2 = {:}T\Lambda{:}
+ \cdots$:
\begin{align}
N_{\Lambda_2}(c) &= \langle 0|(\Lambda_2)_6 \cdot (\Lambda_2)_{-6}|0\rangle
\nonumber \\
&= \tfrac{c(2c-1)(5c+22)(7c+68)}{4200}. \label{eq:N-lambda2}
\end{align}
This vanishes at $c = 0$ (trivially), $c = 1/2$, and $c = -22/5$;
these are precisely the values where the weight-6 quasi-primary
space degenerates.  The factor $5c + 22$ reflects the $W_3$
critical central charge from Theorem~\ref{thm:c-scaling}.
\end{computation}

\begin{remark}[Regular term and bar complex]\label{rem:regular-term-bar}
\index{bar complex!weight-6 contributions}
The weight-6 composites $\Lambda_2$ and $\Lambda_3$ contribute to
the bar differential $d_2\colon \bar{B}^2(\mathcal{W}_3) \to
\bar{B}^1(\mathcal{W}_3)$ via the regular term in the $W$--$W$
OPE.  Specifically, the $(z-w)^0$ coefficient in the $W$--$W$ OPE
is a linear combination:
\[
[W(z)W(w)]_{\text{reg}} = a_1(c)\,\Lambda_2(w) + a_2(c)\,\Lambda_3(w)
  + a_3(c)\,\partial^2\Lambda(w) + a_4(c)\,\partial^4 T(w)
\]
where the coefficients $a_i(c)$ are determined by the Jacobi identity
(cf.\ Construction~\ref{const:w3-bar}).  These contribute to
the degree-2 bar differential on the $W \otimes W$ component of
$\bar{B}^2(\mathcal{W}_3)$.
\end{remark}

\subsection{Higher-spin composite fields}
\label{sec:higher-spin-composites}
\index{higher-spin composite fields}
\index{W-algebra@$\mathcal{W}$-algebra!composite fields}

For the $W_N$ algebra with generating field $W$ of conformal
weight $N$, the $W$--$W$ OPE produces composite fields at
weight $2N$.  The quasi-primary projection of the normal-ordered
square ${:}W^2{:}$ is the simplest such field, and its structure
governs the non-linearity of the algebra.

\begin{proposition}[Quasi-primary projection of \texorpdfstring{${:}W^2{:}$}{:W2:};
\ClaimStatusProvedHere]\label{prop:W-squared-qp}
\index{quasi-primary projection}
\index{normal-ordered product!quasi-primary projection}
Let $W$ be a Virasoro-primary field of conformal weight~$N$.
The quasi-primary projection of the normal-ordered square
${:}W(z)W(z){:}$ is:
\begin{equation}\label{eq:W-squared-qp}
{:}W^2{:}_{\mathrm{qp}} = {:}WW{:}
  - \sum_{k=1}^{N-1} \frac{(-1)^k\,(2N-1)!\,(2k)!}{(2N-1+2k)!\,k!\,(k-1)!}
  \,\partial^{2k}({:}WW{:})_{\mathrm{sing},k}
\end{equation}
where ${:}WW{:}_{\mathrm{sing},k}$ denotes the weight-$(2N-2k)$
singular vector obtained from the $k$-th contraction.
In particular, the leading correction is:
\[
{:}W^2{:}_{\mathrm{qp}} = {:}WW{:}
  - \frac{N}{4N-1}\,\partial^2\bigl({:}WW{:}_{\text{1-contr}}\bigr)
  + O(\partial^4).
\]
\end{proposition}

\begin{proof}
The quasi-primary condition is $L_1|{:}W^2{:}_{\mathrm{qp}}\rangle = 0$.
On the vacuum module, $|{:}WW{:}\rangle = W_{-N}^2|0\rangle$.
Using $[L_1, W_{-N}] = (N+1)W_{1-N}$ and the standard identity
for $L_1$ acting on normal-ordered products:
\[
L_1\,{:}WW{:} = {:}(L_1 W)W{:} + {:}W(L_1 W){:} + \binom{2N}{1}\partial W \cdot W
\]
the anomalous term (from the single Wick contraction between $T$
and ${:}WW{:}$) is proportional to $\partial({:}WW{:}_{\text{1-contr}})$.
Setting $L_1(|\text{composite}\rangle + \alpha\,\partial^2|\text{correction}\rangle) = 0$
and solving for $\alpha$ yields the coefficient $-N/(4N-1)$.
The pattern continues inductively: at each order, the $L_1$
anomaly from the $k$-th correction is cancelled by the
$(k+1)$-th correction, terminating at $k = N-1$ because
there are at most $N-1$ independent contractions between two
fields of weight~$N$.
\end{proof}

\begin{example}[Specialization to \texorpdfstring{$W_3$}{W_3}]\label{ex:W-squared-W3}
\index{W3 algebra@$\mathcal{W}_3$!WW square@${:}W^2{:}$ projection}
For $N = 3$, the quasi-primary projection of ${:}W^2{:}$ is a
weight-6 field.  The leading correction has coefficient
$-3/(4\cdot 3 - 1) = -3/11$.  Explicitly:
\begin{equation}\label{eq:W2-W3}
{:}W^2{:}_{\mathrm{qp}} = {:}WW{:}
  - \frac{3}{11}\,\partial^2\bigl(N_W\cdot T + \cdots\bigr)
  + \frac{93}{70(22+5c)}\,\partial^4 T
\end{equation}
where the $\partial^2(\cdots)$ involves the single Wick
contraction $W_{(1)}W = N_W\cdot T + \cdots$ (with
$N_W = c/3$ from the sixth-order pole).  The $\partial^4 T$
coefficient depends on $c$ because it arises from the
double contraction, which produces the stress tensor
$T$ with coefficient $2$ and then the Virasoro central
term.
\end{example}

\begin{computation}[Two-point function \texorpdfstring{$\langle {:}W^2{:}_{\mathrm{qp}}(z)\,
{:}W^2{:}_{\mathrm{qp}}(w)\rangle$}{:W2:_qp(z) :W2:_qp(w)}; \ClaimStatusProvedHere]
\label{comp:W2-twopt}
The two-point function is obtained by Wick contraction.  Using
$\langle W(z)W(w)\rangle = (c/3)/(z-w)^6$ and the combinatorics
of double Wick contractions:
\begin{align}
\langle {:}WW{:}(z)\, {:}WW{:}(w) \rangle
  &= 2\,\langle W(z)W(w)\rangle^2
  = \frac{2c^2/9}{(z-w)^{12}}. \label{eq:WW-wick}
\end{align}
After subtracting the descendant contributions (which involve
cross-terms with $\partial^{2k}T$), the quasi-primary two-point
function is:
\begin{equation}\label{eq:W2qp-twopt}
\langle {:}W^2{:}_{\mathrm{qp}}(z)\, {:}W^2{:}_{\mathrm{qp}}(w)\rangle
  = \frac{c^2(5c+22)(2c-1)}{9 \cdot 350 \cdot (z-w)^{12}}.
\end{equation}
This vanishes at $c = 0$, $c = -22/5$, and $c = 1/2$.  The factor
$5c + 22$ again reflects the critical central charge of the $W_3$
algebra (Theorem~\ref{thm:c-scaling}).
\end{computation}

\subsection{\texorpdfstring{Null vectors and degenerate $W_3$ representations}{Null vectors and degenerate W3 representations}}
\label{sec:w3-null-vectors}
\index{null vector!W3@$\mathcal{W}_3$|textbf}
\index{W3 algebra@$\mathcal{W}_3$!null vectors|textbf}
\index{degenerate representation!W3@$\mathcal{W}_3$}

At special values of the conformal weight $h$, the $W_3$
charge $w$, and the central charge $c$, the $W_3$ Verma module
acquires null vectors.  These null vectors impose relations on
correlation functions and, from the bar-cobar perspective,
correspond to relations in the bar complex that lower the
cohomological dimension.

\begin{definition}[\texorpdfstring{$W_3$}{W_3} highest weight state and Verma module]
\label{def:w3-hw-state}
\index{W3 algebra@$\mathcal{W}_3$!highest weight state}
A \emph{$W_3$ highest weight state} $|h,w\rangle$ satisfies:
\begin{align}
L_0\,|h,w\rangle &= h\,|h,w\rangle, \quad
  W_0\,|h,w\rangle = w\,|h,w\rangle, \\
L_n\,|h,w\rangle &= W_n\,|h,w\rangle = 0
  \quad \text{for } n \geq 1.
\end{align}
The \emph{$W_3$ Verma module} $\mathcal{M}^{W_3}(c,h,w)$ is
the $W_3$-module freely generated by $|h,w\rangle$ under the
action of $\{L_{-n}, W_{-m}\}_{n,m \geq 1}$.  At level $\ell$
(i.e., eigenvalue $h + \ell$ of $L_0$), the dimension of the
Verma module equals $\sum_{k=0}^{\ell} p(\ell - k)\, p_W(k)$,
where $p(n)$ is the partition function and $p_W(k)$ counts
partitions into parts $\geq 1$ (from $W_{-m}$ modes).  In
particular: level~$0$ has dimension~$1$, level~$1$ has
dimension~$2$ (spanned by $L_{-1}|h,w\rangle$ and
$W_{-1}|h,w\rangle$), and level~$2$ has dimension~$4$.

This is the $W_3$ analogue of the Virasoro Verma module
(Definition~\ref{def:verma-module}).
\end{definition}

\begin{theorem}[Level-1 null vector; \ClaimStatusProvedHere]
\label{thm:w3-null-level1}
\index{null vector!level 1 in W3@level 1 in $\mathcal{W}_3$}
The $W_3$ Verma module $\mathcal{M}^{W_3}(c,h,w)$ with $h \neq 0$
has a null vector at level~$1$ if and only if:
\begin{equation}\label{eq:w3-null-curve}
9w^2(22 + 5c) = 2h^2(32h - 62 - c).
\end{equation}
When this condition holds, the null vector is:
\begin{equation}\label{eq:w3-null-vec}
|\chi_1\rangle = \Bigl(W_{-1} - \tfrac{3w}{2h}\, L_{-1}\Bigr)\,|h,w\rangle.
\end{equation}
\end{theorem}

\begin{proof}
At level~$1$, the Verma module has a two-dimensional space spanned
by $L_{-1}|h,w\rangle$ and $W_{-1}|h,w\rangle$.  A general
level-$1$ state is
$|\chi_1\rangle = (W_{-1} - \alpha L_{-1})|h,w\rangle$
(the case of a pure Virasoro null vector $L_{-1}|h,w\rangle$
requires $h = 0$).

\emph{Condition $L_1|\chi_1\rangle = 0$.}
Using $[L_m, W_n] = (2m - n)W_{m+n}$ (since $W$ has
conformal weight~$3$):
\begin{align}
L_1 W_{-1}|h,w\rangle &= [L_1, W_{-1}]|h,w\rangle
  = 3W_0|h,w\rangle = 3w\,|h,w\rangle, \nonumber \\
L_1 L_{-1}|h,w\rangle &= [L_1, L_{-1}]|h,w\rangle
  = 2L_0|h,w\rangle = 2h\,|h,w\rangle. \nonumber
\end{align}
Thus $L_1|\chi_1\rangle = (3w - 2\alpha h)|h,w\rangle$, giving
$\alpha = 3w/(2h)$.

\emph{Condition $W_1|\chi_1\rangle = 0$.}
The $W_3$ mode commutator (from the $W$--$W$ OPE,
Theorem~\ref{thm:w-w-ope-complete}) gives, at $m = 1$, $n = -1$:
\[
[W_1, W_{-1}]
  = (m{-}n)\bigl[\tfrac{1}{15}(m{+}n{+}3)(m{+}n{+}2)
    - \tfrac{1}{6}(m{+}2)(n{+}2)\bigr]L_0
  + \tfrac{16}{22+5c}(m{-}n)\Lambda_0
\]
where the central term vanishes since $m^2 - 1 = 0$.  Substituting:

\[
[W_1, W_{-1}]
  = 2\bigl[\tfrac{6}{15} - \tfrac{3}{6}\bigr]L_0
    + \tfrac{32}{22+5c}\,\Lambda_0
  = -\tfrac{1}{5}L_0 + \tfrac{32}{22+5c}\,\Lambda_0.
\]

On the highest weight state,
$\Lambda_0|h,w\rangle = (h^2 - \frac{9}{5}h)|h,w\rangle$
(since $L_m|h,w\rangle = 0$ for $m > 0$ and
$(\partial^2 T)_0 = 6L_0$, giving
$\Lambda_0 = (TT)_0 - \frac{3}{10}\cdot 6L_0
  = L_0^2 - \frac{9}{5}L_0$ on highest weight states).  Thus:

\[
W_1 W_{-1}|h,w\rangle = \Bigl[-\tfrac{h}{5}
  + \tfrac{32(h^2 - 9h/5)}{22+5c}\Bigr]|h,w\rangle.
\]

Also
\[
W_1 L_{-1}|h,w\rangle = [W_1, L_{-1}]|h,w\rangle
= 3W_0|h,w\rangle = 3w|h,w\rangle.
\]
Here we used $[W_1, L_{-1}] = -[L_{-1}, W_1] = 3W_0$.

Thus $W_1|\chi_1\rangle = 0$ requires
\[
-\tfrac{h}{5} + \tfrac{32(h^2 - 9h/5)}{22+5c}
  - 3\alpha w = 0.
\]

Substituting $\alpha = 3w/(2h)$ and multiplying by $2h$:
\[
-\tfrac{2h^2}{5}
  + \tfrac{64h(h^2 - 9h/5)}{22+5c} - 9w^2 = 0
\]
which simplifies to \eqref{eq:w3-null-curve} after clearing
denominators:
$9w^2(22+5c) = 2h^2\bigl[-\frac{22+5c}{5}
  + \frac{32(5h-9)}{5}\bigr]
  = \frac{2h^2}{5}(160h - 310 - 5c)
  = 2h^2(32h - 62 - c)$.
\end{proof}

\begin{remark}[Comparison with Virasoro null vectors]\label{rem:w3-vs-vir-null}
\index{null vector!comparison Virasoro vs W3@comparison Virasoro vs $\mathcal{W}_3$}
In the Virasoro algebra, the level-1 null vector
$L_{-1}|h\rangle = 0$ requires only $h = 0$ (one condition).
For $W_3$, the level-1 null vector imposes one condition
relating $h$, $w$, and $c$ (equation~\eqref{eq:w3-null-curve}),
reflecting the extra quantum number $w$.
The null vector curve in the $(h,w)$-plane
is a non-trivial algebraic curve depending on $c$, in contrast
with the Virasoro case where the null conditions are discrete
points in the $(h,c)$-plane.
\end{remark}

\begin{proposition}[Null vectors and bar complex relations;
\ClaimStatusProvedHere]\label{prop:null-bar-relation}
\index{bar complex!null vectors as relations}
\index{null vector!bar complex interpretation}
A null vector $|\chi_\ell\rangle$ at level $\ell$ in the $W_3$
Verma module $\mathcal{M}^{W_3}(c,h,w)$ corresponds to a
relation in the bar complex $\bar{B}(\mathcal{W}_3)$ restricted
to the $(c,h,w)$-sector.  Specifically, if $d_{\bar{B}}$ denotes
the bar differential, then the element $\xi_\ell \in \bar{B}^1$
corresponding to $|\chi_\ell\rangle$ satisfies
$d_{\bar{B}}(\xi_\ell) = 0$ and is exact:
$\xi_\ell = d_{\bar{B}}(\eta)$ for some $\eta \in \bar{B}^0$.

In particular, the bar cohomology $H^1(\bar{B}(\mathcal{W}_3))$
at weight $h + \ell$ has strictly lower dimension for degenerate
representations (where null vectors exist) than for generic
representations.
\end{proposition}

\begin{proof}
The bar complex $\bar{B}^1(\mathcal{W}_3)$ at weight $h + \ell$
is spanned by elements of the form $s^{-1}\phi$ where $\phi$ runs
over weight-$(h+\ell)$ states in the augmentation ideal
$\overline{\mathcal{W}}_3$ (cf.\ Construction~\ref{const:w3-bar}).
A null vector $|\chi_\ell\rangle$ in the Verma module is, by
definition, a state annihilated by all positive modes.  In the
bar complex, this translates to an element $\xi_\ell \in \bar{B}^1$
for which $d_1(\xi_\ell) = 0$ (the component of $d_{\bar{B}}$
from the linear part of the OPE).

The element is exact because the null vector is a descendant:
$|\chi_\ell\rangle = P(L_{-n}, W_{-m})|h,w\rangle$ for some
polynomial $P$ in the negative modes.  This polynomial $P$
corresponds to an element $\eta \in \bar{B}^0$ (the degree-0
bar complex at weight $h + \ell$, which records the descendants)
such that $d_{\bar{B}}(\eta) = \xi_\ell$.

The consequence for bar cohomology follows: each null vector
at level $\ell$ converts a potential cohomology class in
$H^1(\bar{B}^{h+\ell})$ into an exact element, reducing
$\dim H^1$ by one.  For minimal model representations (where
the maximal number of null vectors occur), this reduction is
responsible for the finite-dimensionality of the bar cohomology
(cf.\ Example~\ref{ex:w3-minimal-modules}).
\end{proof}

\subsection{\texorpdfstring{The $W_3$ Kac determinant}{The W3 Kac determinant}}
\label{sec:w3-kac-determinant}
\index{Kac determinant!W3@$\mathcal{W}_3$|textbf}
\index{W3 algebra@$\mathcal{W}_3$!Kac determinant|textbf}

The Kac determinant for the $W_3$ algebra generalizes the Virasoro
Kac determinant (Kac~\cite{Kac}) to a two-parameter family indexed
by $(h, w)$.  Its vanishing locus determines the degenerate
representations.

\begin{definition}[Gram matrix and Kac determinant at level \texorpdfstring{$\ell$}{l}]
\label{def:w3-gram-matrix}
\index{Gram matrix!W3@$\mathcal{W}_3$}
Let $\{|v_1\rangle, \ldots, |v_{d(\ell)}\rangle\}$ be the standard
basis of the level-$\ell$ subspace of $\mathcal{M}^{W_3}(c,h,w)$,
ordered by total mode number.  The \emph{Gram matrix} is:
\[
M_\ell(c,h,w)_{ij} = \langle v_i | v_j \rangle
\]
where the inner product is the Shapovalov form
(cf.\ Definition~\ref{def:shapovalov-form}).  The
\emph{$W_3$ Kac determinant at level $\ell$} is
$\det M_\ell(c,h,w)$.
\end{definition}

\begin{theorem}[\texorpdfstring{$W_3$}{W_3} Kac determinant at level~1;
\ClaimStatusProvedHere]\label{thm:w3-kac-level1}
\index{Kac determinant!level 1}
At level~$1$, the Gram matrix in the basis
$\{L_{-1}|h,w\rangle,\, W_{-1}|h,w\rangle\}$ is:
\begin{equation}\label{eq:w3-gram-level1}
M_1 = \begin{pmatrix}
2h & 3w \\
3w & -\frac{h}{5} + \frac{32}{22+5c}(h^2 - \frac{9h}{5})
\end{pmatrix}
\end{equation}
and the Kac determinant is:
\begin{equation}\label{eq:w3-det-level1}
\det M_1 = 2h\Bigl[-\tfrac{h}{5}
  + \tfrac{32(h^2 - 9h/5)}{22+5c}\Bigr] - 9w^2.
\end{equation}
\end{theorem}

\begin{proof}
\emph{Entry $(1,1)$:}
$\langle h,w|L_1 L_{-1}|h,w\rangle = \langle h,w|[L_1,L_{-1}]|h,w\rangle
= 2h$.

\emph{Entry $(1,2) = (2,1)$:}
$\langle h,w|L_1 W_{-1}|h,w\rangle = \langle h,w|[L_1,W_{-1}]|h,w\rangle
= 3\langle h,w|W_0|h,w\rangle = 3w$
(using $[L_1, W_{-1}] = 3W_0$ from $[L_m, W_n] = (2m{-}n)W_{m+n}$).

\emph{Entry $(2,2)$:}
$\langle h,w|W_1 W_{-1}|h,w\rangle = \langle h,w|[W_1,W_{-1}]|h,w\rangle$.
From the $W$-$W$ commutator (as computed in the proof of
Theorem~\ref{thm:w3-null-level1}):
\[
[W_1, W_{-1}] = -\tfrac{1}{5}L_0 + \tfrac{32}{22+5c}\,\Lambda_0.
\]
Acting on $|h,w\rangle$: $\Lambda_0|h,w\rangle = (h^2 - \frac{9h}{5})|h,w\rangle$
(Computation~\ref{comp:lambda-low-modes}), giving the entry
$-h/5 + \frac{32}{22+5c}(h^2 - 9h/5)$.

The determinant follows by the $2\times 2$ formula.
\end{proof}

\begin{computation}[Kac determinant vanishing locus at level~1;
\ClaimStatusProvedHere]\label{comp:kac-vanishing-level1}
Setting $\det M_1 = 0$:
\begin{itemize}
\item At $w = 0$: the determinant reduces to
  $2h[-h/5 + 32(h^2 - 9h/5)/(22+5c)]$, which vanishes at
  $h = 0$ (vacuum) or at the root of
  $-1/5 + 32(h - 9/5)/(22+5c) = 0$, i.e.,
  $h = 9/5 + (22+5c)/160$.
\item At $h = 0$: the determinant is $-9w^2$, which vanishes
  only at $w = 0$ (the vacuum representation).
\item The vanishing locus $\det M_1 = 0$ is equivalent to
  the null vector curve~\eqref{eq:w3-null-curve},
  as expected from Theorem~\ref{thm:w3-null-level1}.
\end{itemize}
\end{computation}

\begin{theorem}[\texorpdfstring{$W_3$}{W_3} Kac determinant: general structure;
\ClaimStatusProvedElsewhere]\label{thm:w3-kac-general}
\index{Kac determinant!factorization}
\textup{(Fateev--Lukyanov~\cite{FL88},
Mizoguchi~\cite{Mizoguchi89}.)}
The $W_3$ Kac determinant at level $\ell$ has the factorized form:
\begin{equation}\label{eq:w3-kac-factored}
\det M_\ell(c,h,w) = K_\ell
  \prod_{\substack{r,s \geq 1 \\ rs \leq \ell}}
  \bigl(h - h_{r,s}(c)\bigr)^{p_W(\ell - rs)}
  \cdot R_\ell(c,h,w)
\end{equation}
where $h_{r,s}(c)$ are the conformal dimensions from the $W_3$
Kac table \textup{(}Theorem~\ref{thm:w3-minimal-complete}\textup{)},
$p_W(n)$ counts partitions with an additional $W_3$-mode
degeneracy, $K_\ell$ is a constant independent of $h$ and $w$,
and $R_\ell(c,h,w)$ is a polynomial in $w$ whose roots determine
the $W_3$-charge-dependent null vectors.

The factor $R_\ell$ is the \emph{new feature} of the $W_3$
Kac determinant compared to Virasoro: it reflects null vectors
that depend on the $W_3$ charge $w$ and have no Virasoro analogue.
\end{theorem}

\begin{remark}[Kac determinant and bar complex filtration]
\label{rem:kac-det-bar}
\index{bar complex!Kac determinant filtration}
The $W_3$ Kac determinant controls the bar complex of
$\mathcal{W}_3$ as follows.  The bar complex
$\bar{B}^\bullet(\mathcal{W}_3)$ has a weight filtration by $L_0$
eigenvalue; at weight $h + \ell$, the rank of the Gram matrix
$M_\ell$ determines the dimension of the non-degenerate part of
$\bar{B}^1$ in that weight sector.  When $\det M_\ell = 0$
(i.e., a null vector appears), the bar cohomology changes:

\begin{enumerate}
\item At $\det M_\ell \neq 0$ \textup{(}generic\textup{)}: the
  bar complex at weight $h + \ell$ is acyclic in degree~$1$, and
  the spectral sequence from Theorem~\ref{thm:bar-cobar-spectral-sequence}
  has $E_1^{1,\ell} = 0$.
\item At $\det M_\ell = 0$ \textup{(}degenerate\textup{)}: the bar
  complex acquires non-trivial cohomology $H^1(\bar{B}^{h+\ell})
  \neq 0$, and the spectral sequence has additional generators at
  $E_1^{1,\ell}$ corresponding to the null vectors
  \textup{(}Proposition~\ref{prop:null-bar-relation}\textup{)}.
\end{enumerate}

For the $W_3$ minimal models, the Kac determinant vanishes at
infinitely many levels (all $\ell = rs$ with $1 \leq r \leq q-1$,
$1 \leq s \leq p-1$), and the resulting bar complex has
finite-dimensional cohomology groups, consistent with the
rationality of these models.
\end{remark}

\begin{computation}[Level-2 Gram matrix; \ClaimStatusProvedHere]
\label{comp:w3-gram-level2}
\index{Gram matrix!level 2}
At level~$2$, the basis is
\[
\{L_{-1}^2|h,w\rangle,\; L_{-2}|h,w\rangle,\;
W_{-1}L_{-1}|h,w\rangle,\; W_{-2}|h,w\rangle\}
\]
($4$-dimensional).  The Gram matrix entries are:
\begin{align}
\langle L_1^2 L_{-1}^2\rangle &= 4h(2h+1), &
\langle L_2 L_{-2}\rangle &= \tfrac{c}{2} + 4h, \nonumber \\
\langle L_1^2 L_{-2}\rangle &= 6h, &
\langle W_1 L_1 W_{-1} L_{-1}\rangle &= 4w^2 + 2h \cdot E_{22}
  \nonumber
\end{align}
where $E_{22} = -h/3 + 16(h^2-9h/5)/(22+5c)$ is the $(2,2)$
entry of $M_1$.  The remaining entries involve the commutators
$[W_2, W_{-2}]$ and $[W_2, L_{-1}W_{-1}]$, each computed from
the mode expansion of the $W$-$W$ OPE.  The full $4 \times 4$
determinant $\det M_2(c,h,w)$ is a polynomial of degree~$8$ in $h$
and degree~$4$ in $w$, whose factorization follows the pattern of
Theorem~\ref{thm:w3-kac-general}.
\end{computation}
