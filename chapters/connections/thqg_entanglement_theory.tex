\begin{proposition}[Scalar, holographic, and open/closed lanes;
\ClaimStatusConditional]
\label{prop:thqg-entanglement-lane-separation}
\index{entanglement dictionary!lane separation}
The dictionary above has three logically distinct lanes.
\begin{enumerate}[label=\textup{(\roman*)}]
\item The \emph{scalar shadow lane} consists of identities in the
shadow obstruction tower, for example
\[
\Theta^{\le 2,(g)}_\cA
=\kappa(\cA)\lambda_g^{\mathrm{FP}}
\]
and, for the Virasoro dual pair,
\[
\kappa(\mathrm{Vir}_c)+\kappa(\mathrm{Vir}_{26-c})
=c/2+(26-c)/2=13 .
\]
These are algebraic identities in the genus-$g$ obstruction class.

\item The \emph{physical holographic lane} starts only after a
holomorphic-topological theory~$T$ and the comparison maps in
Definition~\ref{def:holographic-modular-koszul-datum} are supplied.
The algebraic class
$\kappa(\cA)\lambda_g^{\mathrm{FP}}$ occupies the area slot; it is
not the number $\mathrm{Area}/(4G_N)$ until the holographic
normalization identifies the two.

\item The \emph{open/closed lane} supplies typed maps
\[
C^\bullet_{\mathrm{ch}}(\cA,\cA)
\longrightarrow
\operatorname{End}(\cM)
\]
from the chiral derived centre to boundary endomorphisms.  It does
not supply a Hilbert tensor product, a positive density matrix, or a
partial trace until the finite-type or trace-class sewing completion
has been chosen.
\end{enumerate}
\end{proposition}

\begin{proof}
Part~\textup{(i)} is the scalar trace clause of
Theorem~\ref{thm:mc2-bar-intrinsic} together with the Virasoro
normalization $\kappa(\mathrm{Vir}_c)=c/2$.  Part~\textup{(ii)} is
the typed content of Definition~\ref{def:holographic-modular-koszul-datum}:
the datum contains comparison maps from the HT bulk side to the
chiral derived-centre side; it does not identify physical
functionals by itself.  Part~\textup{(iii)} is the open/closed realization
(Theorem~\ref{thm:thqg-swiss-cheese} and
Construction~\ref{constr:thqg-oc-mc-element}); tensor products and
traces enter only in the sewing envelope.
\end{proof}

\begin{remark}[Two sources of entanglement; direct sum versus tensor product]
\label{rem:thqg-two-entanglements}
\index{entanglement!direct sum versus tensor product}
The Lagrangian decomposition controls dimensions, not entanglement
states.
The Lagrangian decomposition
$\cH_g(\cA) = Q_g(\cA) \oplus Q_g(\cA^!)$
is a \emph{direct sum}.  It carries classical correlations:
every vector $v = \alpha+\beta \in V\oplus W$ has a unique
decomposition into its $V$- and $W$-components.  Quantum
entanglement enters only through an actual tensor product.

Potential quantum entanglement in the algebraic framework
comes from a different source: the \emph{sewing construction}
and the \emph{open/closed mixed sector}.
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Sewing entanglement.}
 Cutting a genus-$g$ surface along a separating curve
 produces two bordered surfaces. The factorization homology
 on each piece gives boundary state data.  When the HS-sewing
 hypotheses of Theorem~\ref{thm:general-hs-sewing} supply the
 Hilbert completion $H_\cA=FH(\cA;S^1)$ and trace-class sewing
 operator, the separating cut lands in
 $H_\cA\widehat{\otimes}H_\cA$.  Only at this completed level
 does the genus-$g$ surface define a candidate entangled state.

\item \emph{Open/closed entanglement.}
 The Swiss-cheese theorem
 (Theorem~\ref{thm:thqg-swiss-cheese}) decomposes the
 holographic system into an open boundary sector and a closed
 chiral derived-centre sector.  For a boundary module $\cM$ at
 a puncture, the mixed projection is a closed-to-boundary map
 \[
 C^\bullet_{\mathrm{ch}}(\cA,\cA)\longrightarrow
 \operatorname{End}(\cM),
 \]
 not a canonical tensor-factor decomposition.  On a finite-type
 or trace-class completion it may be dualized to an element of a
 completed tensor product.  A density matrix and its entropy exist
 only after that positivity and trace-class structure is supplied.
\end{enumerate}
The Lagrangian decomposition $Q_g \oplus Q_g^!$ provides
\emph{dimensional constraints} (the Page bound,
Theorem~\ref{thm:thqg-page-constraint}) that constrain
any entanglement entropy from sources~(i)--(ii),
but it does not itself carry entanglement.
\end{remark}

\subsection{Dimensional constraints from the Lagrangian structure}
\label{subsec:thqg-entanglement-dimensional}
\index{entanglement!dimensional constraints|textbf}

\begin{definition}[Lagrangian capacity]
\label{def:thqg-lagrangian-capacity}
\index{Lagrangian capacity|textbf}
For a gravitational input $(\cA, X, \langle \cdot, \cdot \rangle, k)$
and genus $g \ge 1$, the \emph{Lagrangian capacity} is
\begin{equation}\label{eq:thqg-lagrangian-capacity}
C_g(\cA)
\;:=\;
\dim Q_g(\cA).
\end{equation}
If $Q_g(\cA)$ is not finite-dimensional, set
$C_g(\cA)=\infty$; the logarithmic Page inequalities below are
asserted only in the finite-capacity case.
This is the dimension of the $\cA$-eigenspace in the
Lagrangian decomposition
$\cH_g(\cA) = Q_g(\cA) \oplus Q_g(\cA^!)$
(Theorem~\ref{V2-thm:thqg-III-eigenspace-decomposition}).
\end{definition}

\begin{theorem}[Lagrangian constraint theorem;
\ClaimStatusConditional]
\label{thm:thqg-lagrangian-constraint}
\index{Lagrangian constraint theorem|textbf}
\textit{Type signature.}
Quadrant: Open. Presentation: shifted-symplectic dg.
Level: $3$ (ambient genus-$g$ centre complex).
Hypothesis package: gravitational input
\textup{(}Definition~\ref{def:thqg-standing-hypotheses}\textup{)};
finite Lagrangian capacities $C_g(\cA), C_g(\cA^!) < \infty$;
nondegenerate Verdier pairing on the genus-$g$ ambient centre
\textup{(}Theorem~\ref{thm:quantum-complementarity-main}\textup{)}.

\medskip
Fix $g\ge1$.  For a gravitational input satisfying the hypotheses of
Theorem~\ref{thm:quantum-complementarity-main} in genus~$g$, and with
finite Lagrangian capacities in that genus:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \textup{(Complementarity sum.)}
 $C_g(\cA) + C_g(\cA^!) = \dim \cH_g(\cA)
 = \dim H^*(\overline{\cM}_g, \cZ(\cA))$.

\item \textup{(Genus-$1$ structure.)}
 In the case $g=1$,
 $C_1(\cA) = C_1(\cA^!) = 1$ and $\dim \cH_1(\cA) = 2$.

\item \textup{(Self-dual halving.)}
 For $\cA \cong \cA^!$ in this genus:
 $C_g(\cA) = C_g(\cA^!)
 = \tfrac{1}{2}\dim \cH_g(\cA)$.

\item \textup{(Scalar contribution.)}
 If $\kappa(\cA)\lambda_g^{\mathrm{FP}}\neq0$, the scalar shadow
 ($r = 2$) contributes one line to $Q_g(\cA)$: the class
 $\kappa(\cA) \cdot \lambda_g^{\mathrm{FP}} \in Q_g(\cA)$
 spans that line.  If the class vanishes, the scalar line is zero.

\item \textup{(Virasoro scalar complementarity.)}
 $F_g^{\mathrm{sc}}(\mathrm{Vir}_c) + F_g^{\mathrm{sc}}(\mathrm{Vir}_{26-c})
 = 13 \cdot \lambda_g^{\mathrm{FP}}$.
\end{enumerate}
\end{theorem}

\begin{proof}
Part (i): Theorem~\ref{thm:quantum-complementarity-main}.
Part (ii): Theorem~\ref{thm:thqg-III-genus-1-holographic}.
Part (iii): Corollary~\ref{cor:thqg-III-dimension-parity}.
Part (iv): the scalar MC element
$\Theta^{\le 2,(g)} = \kappa \cdot \lambda_g^{\mathrm{FP}}$
is a multiple of a fixed cohomology class; a nonzero multiple spans
one line in $Q_g(\cA)$, while the zero multiple contributes no line.
Part (v): linearity of $F_g^{\mathrm{sc}}$ in $\kappa$,
combined with $\kappa(\mathrm{Vir}_c) + \kappa(\mathrm{Vir}_{26-c})
= c/2 + (26-c)/2 = 13$.
\end{proof}

\begin{proposition}[Genus-one capacity and higher-genus scope;
\ClaimStatusConditional]
\label{prop:thqg-genus-one-higher-genus-scope}
\index{genus one!entanglement capacity}
\index{higher genus!entanglement scope}
At genus~$1$ the Lagrangian capacity lane is rigid:
\[
C_1(\cA)=C_1(\cA^!)=1,\qquad \dim\cH_1(\cA)=2 .
\]
Hence any Hilbert-space entropy whose two factors are bounded by
these capacities is zero.  For $g\ge2$, the same conclusion is not
available from the genus-one theorem.  One must use
Theorem~\ref{thm:quantum-complementarity-main} at that genus, the
finite-capacity hypothesis, and a sewing completion that realizes the
algebraic summands as bounds on Hilbert factors.
\end{proposition}

\begin{proof}
The genus-one equalities are part~\textup{(ii)} of
Theorem~\ref{thm:thqg-lagrangian-constraint}.  A bipartite state with
factor dimensions at most $1$ has reduced density matrix of rank at
most~$1$, so its von~Neumann entropy is~$0$.  For $g\ge2$ the
dimensions are those of the genus-$g$ homotopy eigenspaces in
Theorem~\ref{thm:quantum-complementarity-main}; no statement about
Hilbert tensor factors follows until the sewing completion supplies
them.
\end{proof}

\begin{remark}[Why the Lagrangian capacity bounds entanglement]
\label{rem:thqg-capacity-bounds-entanglement}
\index{entanglement!bounded by Lagrangian capacity}
Any von~Neumann entropy $S$ of a state in a Hilbert space
$H_A \otimes H_B$ satisfies $S \le \log\min(\dim H_A, \dim H_B)$.
If the spatial entanglement of a boundary subregion at genus $g$
is controlled by the ambient complex $\cH_g$, the Lagrangian
decomposition constrains:
\[
S_{\mathrm{EE}}^{(g)}(\cA)
\;\le\;
\log\min(C_g(\cA),\, C_g(\cA^!)).
\]
The complementarity sum $C_g + C_g' = \dim\cH_g$ then implies
the \emph{Page-type capacity bound}: for self-dual algebras,
\[
S_{\mathrm{EE}}^{(g)}(\cA)
\;\le\;
\log C_g(\cA)
\;=\;
\log\!\bigl(\tfrac{1}{2}\dim\cH_g(\cA)\bigr).
\]
A bound of the form $\tfrac12\log(\dim H_{\mathrm{tot}})$ belongs
to a tensor product $H_A\otimes H_B$ with
$\dim H_{\mathrm{tot}}=\dim H_A\dim H_B$; it is not a consequence
of the direct-sum identity $\dim\cH_g=C_g+C_g'$.
The capacity inequality is a formal consequence of the standard
von~Neumann bound and Theorem~C once the Hilbert realization has
factors bounded by the Lagrangian capacities; Theorem~D supplies the
scalar area class.  Saturation by a physical entanglement entropy is
conjectural (\S\ref{subsec:thqg-entanglement-qes}).
\end{remark}

S(\cM) \;:=\;
-\Tr\bigl(\rho_{\cM} \log \rho_{\cM}\bigr),

where $\rho_{\cM}$ is the reduced density matrix on the open sector
obtained by tracing out the closed sector.

\begin{remark}[The annulus trace as partial trace]
\label{rem:thqg-annulus-as-partial-trace}

hypothesis on that realization; it is not a formal consequence of
the annulus trace theorem alone.
\end{remark}

\subsection{The algebraic quantum extremal surface}
\label{subsec:thqg-entanglement-qes}
\index{quantum extremal surface!algebraic|textbf}
\index{QES|see{quantum extremal surface}}

The quantum extremal surface (QES) formula
\cite{Engelhardt-Wall14}
\begin{equation}\label{eq:thqg-qes-physical}
S(R) \;=\; \min_{\gamma}\,\operatorname{ext}\Bigl[
\frac{\mathrm{Area}(\gamma)}{4G_N}
\;+\; S_{\mathrm{bulk}}(\Sigma_\gamma)
\Bigr]
\end{equation}
generalizes the Ryu--Takayanagi formula \cite{Ryu-Takayanagi06}
by including bulk quantum corrections.  The shadow obstruction tower
provides an algebraic approximation scheme; identifying that scheme
with the physical QES functional is a separate holographic input.

\begin{definition}[QES approximant]
\label{def:thqg-qes-approximant}
\index{QES approximant|textbf}
For a gravitational input and integer $r \ge 2$, the
\emph{$r$-th order QES approximant} at genus $g$ is:
\begin{equation}\label{eq:thqg-qes-approximant}
S_{\mathrm{QES}}^{(g, \le r)}(\cA)
\;:=\;
\underbrace{
\kappa(\cA) \cdot \lambda_g^{\mathrm{FP}}
}_{\text{scalar obstruction / area slot (degree $2$)}}
\;+\;
\underbrace{
\sum_{j=3}^{r} \delta F^{(g)}_j(\cA)
}_{\text{closed-sector corrections (degrees $3, \ldots, r$)}}
\end{equation}
where $\delta F^{(g)}_j(\cA)$ is the genus-$g$ free-energy
increment contributed by the degree-$j$ shadow jet
$\Sh_j(\cA)$. At $r = 2$, this recovers the scalar
free energy $F_g^{\mathrm{sc}}(\cA) = \kappa \cdot \lambda_g^{\mathrm{FP}}$
on the uniform-weight scalar lane.
\end{definition}

\begin{theorem}[QES convergence; \ClaimStatusConditional]
\label{thm:thqg-qes-convergence}
\index{QES!convergence}
\textit{Type signature.}
Quadrant: Open. Presentation: chain-level. Level: $5$ (modular scalars).
Hypothesis package: finite shadow depth $r_{\max}(\cA) < \infty$, or
bounded class-$\mathsf M$ shadow radius $\rho(\cA) < 1$ with uniform
genus-$g$ bound $|\delta F^{(g)}_j(\cA)| \le K_g\rho^j j^{-5/2}$.

\medskip
The algebraic QES approximants converge under the finite-depth or
bounded class-$\mathsf M$ hypotheses:
\[
F_g(\cA)
\;:=\;
\lim_{r \to \infty}
S_{\mathrm{QES}}^{(g, \le r)}(\cA)
\]
exists at every fixed genus $g\ge1$ in the following cases:
\begin{enumerate}[label=\textup{(\roman*)}]
\item $r_{\max}(\cA)<\infty$, where the series terminates.
\item $r_{\max}(\cA)=\infty$, $\rho(\cA)<1$, and the genus-$g$
 functional on shadow degree $j$ is uniformly bounded, so that
 $|\delta F^{(g)}_j(\cA)| \le K_g\rho(\cA)^j j^{-5/2}$.
\end{enumerate}
Outside these hypotheses the expression is a formal shadow expansion,
not a convergence theorem.
\end{theorem}

\begin{proof}
If $r_{\max}<\infty$, no tail remains after degree $r_{\max}$.  In
class~$\mathsf M$, the shadow-radius theorem
(Theorem~\ref{thm:shadow-radius}) gives the
$\rho(\cA)^j j^{-5/2}$ decay on the shadow-metric coefficient lane.
The assumed uniform bound for the genus-$g$ functional is precisely
the hypothesis that transfers this estimate to
$\delta F^{(g)}_j$.  Since $\rho(\cA)<1$, comparison with
$\sum_j\rho^j j^{-5/2}$ gives absolute convergence.
\end{proof}

\begin{remark}[The MC equation as extremization]
\label{rem:thqg-mc-as-extremization}
\index{Maurer--Cartan equation!as extremization}
The physical QES formula \eqref{eq:thqg-qes-physical} requires
extremization over surfaces.  The algebraic stationarity condition
carried by the shadow tower is the Maurer--Cartan equation.
The bar-intrinsic MC element $\Theta_\cA = D_\cA - d_0$
\emph{satisfies} this equation
(Theorem~\ref{thm:mc2-bar-intrinsic}):
the construction gives a flat algebraic deformation.  It becomes the
surface extremality equation only after the holographic dictionary
identifies shadow variations with geometric variations of~$\gamma$.
\end{remark}

\begin{conjecture}[QES identification]
\ClaimStatusConjectured
\label{conj:thqg-qes-identification}
\index{QES!identification with physical formula}
Under the holographic dictionary of
Definition~\textup{\ref{def:holographic-modular-koszul-datum}},
assume that the HT bulk comparison identifies the scalar obstruction
line with the geometric area functional and identifies the
degree-$j$ closed-sector shadow correction with the $j$-th
bulk-entropy correction.  Then the QES approximant
$S_{\mathrm{QES}}^{(g,\le r)}(\cA)$ at degree~$r$ coincides with the
$r$-th order expansion of the physical QES formula
\eqref{eq:thqg-qes-physical}.  Without these comparison maps it is
only the algebraic closed-sector approximant.
\end{conjecture}

\subsection{Replica symmetry and R\'enyi entropy}
\label{subsec:thqg-entanglement-replica}
\index{replica symmetry|textbf}
\index{R\'enyi entropy|textbf}

\begin{construction}[Replica manifold]
\label{constr:thqg-replica-manifold}
\index{replica manifold|textbf}
For a genus-$g$ surface $\Sigma_g$ and integer $n \ge 2$,
the \emph{$n$-fold replica manifold} is the cyclic $n$-fold
branched cover
$\Sigma_{g,n} \to \Sigma_g$
branched over two points $\{p, q\} \subset \Sigma_g$,
with each branch point fully ramified
\textup{(}a single preimage of ramification index $n$\textup{)}.
By the Riemann--Hurwitz formula:
\begin{equation}\label{eq:thqg-replica-genus}
2g(\Sigma_{g,n}) - 2
\;=\; n(2g - 2) + 2(n-1),
\qquad\text{so}\qquad
g(\Sigma_{g,n}) \;=\; ng.
\end{equation}
\begin{itemize}
\item $g = 0$: $g(\Sigma_{0,n}) = 0$
 \textup{(}the map $z \mapsto z^n$ on $\mathbb{P}^1$\textup{)}.
 The replica manifold is again a sphere; the R\'enyi entropy
 is computed by twist-operator two-point functions
 \cite{Calabrese-Cardy04}, not by higher-genus partition functions.

\item $g = 1$: $g(\Sigma_{1,n}) = n$
 \textup{(}the $n$-fold cover of the torus is genus $n$\textup{)}.
 The R\'enyi entropy at genus $1$ involves genuinely higher-genus
 partition functions.

\item $g \ge 2$: $g(\Sigma_{g,n}) = ng$, growing linearly in $n$.
\end{itemize}
\end{construction}

\begin{conjecture}[Replica partition function; \ClaimStatusConjectured]
\label{conj:thqg-replica}
\index{replica!partition function}
The $n$-fold replica partition function of a gravitational input
$(\cA, X, \langle \cdot, \cdot \rangle, k)$ at genus $g \ge 1$
is:
\begin{equation}\label{eq:thqg-replica-pf}
Z_n^{(g)}(\cA)
\;:=\;
Z(\cA;\, \Sigma_{g,n}),
\end{equation}
the partition function of $\cA$ evaluated on the replica manifold
of genus $ng$. The R\'enyi entropy is:
\begin{equation}\label{eq:thqg-renyi}
S_n^{(g)}(\cA)
\;:=\;
\frac{1}{1-n}
\log\frac{Z_n^{(g)}(\cA)}{Z^{(g)}(\cA)^n}.
\end{equation}
The von~Neumann entropy is recovered in the $n \to 1$ limit:
$S^{(g)}(\cA) = \lim_{n \to 1} S_n^{(g)}(\cA)$.
\end{conjecture}

 $\langle Q_g(\cA^!), Q_g(\cA^!) \rangle_g = 0$
 \textup{(}isotropy of each summand, the symplectic shadow of

\end{enumerate}
For \emph{unitary} $\cA$ with positive-definite Shapovalov form,
$\sigma$ also acts as an anti-linear involution

\end{conjecture}

Part~(iii) is the isotropy of each Lagrangian summand under
the Verdier pairing

The anti-linearity statement for unitary algebras requires
the Shapovalov form, which provides a Hermitian structure
compatible with the $\mathbb{C}$-anti-linear action of $*$ on the
center.  The analytic Tomita $J$ axioms, including domain and
closability, require passage to the sewing envelope
$\cA^{\mathrm{sew}}$ (Theorem~\ref{thm:general-hs-sewing}) together
with a faithful normal state.
\end{remark}

\begin{remark}[KMS condition from complementarity]

\subsection{Error correction from Koszul duality}
\label{subsec:thqg-entanglement-error-correction}
\index{error correction!from Koszul duality|textbf}
\index{holographic error correction|textbf}

The bar-cobar adjunction provides an algebraic encoding/decoding
package that can be compared, after completion, with the quantum
error-correcting codes of holographic entanglement
\cite{Almheiri-Dong-Harlow15, Pastawski-Yoshida-Harlow-Preskill15}.

\begin{proposition}[Bar-cobar code structure;
\ClaimStatusConditional]
\label{prop:thqg-barcobar-error-correction}
\index{bar-cobar!as code structure}
The bar-cobar adjunction
$\Omega^{\mathrm{ch}}\dashv B^{\mathrm{ch}}$ (cobar left, bar right)
endows each Koszul chiral algebra $\cA$ with:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Bar package.}
 The right adjoint assigns to $\cA$ the chiral bar coalgebra
 $B^{\mathrm{ch}}(\cA)$; its cohomology is the bar-dual coalgebra
 $\cA^i$.

\item \emph{Cobar counit.}
 The left adjoint assigns to a chiral coalgebra $C$ the chiral cobar
 algebra $\Omega^{\mathrm{ch}}(C)$; for $C=B^{\mathrm{ch}}(\cA)$ the
 counit is
 \[
 \Omega^{\mathrm{ch}}B^{\mathrm{ch}}(\cA)\longrightarrow \cA .
 \]

\item \emph{Exact inversion on the Koszul locus.}
 Theorem~B gives
 $\Omega^{\mathrm{ch}}B^{\mathrm{ch}}(\cA) \xrightarrow{\sim} \cA$:
 the counit is a quasi-isomorphism.  This is inversion, not Verdier
 Koszul duality.

\item \emph{Verdier dual sector.}
 The line algebra $\cA^!$ is obtained from $\cA^i$ only after the
 finite-type or completed continuous Verdier duality hypotheses hold;
 in that case the line-operator category is modelled by
 $\mathcal{C} \simeq \cA^!\text{-}\mathsf{mod}$.
\end{enumerate}
\end{proposition}

\begin{proof}
Parts (i)--(ii) are the two adjoints.  Part (iii) is Theorem~B.  Part
(iv) is the Verdier boundary in Theorem~\ref{thm:thqg-g5-yangian}.
\end{proof}

\begin{remark}[Bar-cobar and holographic error correction]
\label{rem:thqg-hqec-analogy}
\index{holographic error correction!bar-cobar scope}
The bar-cobar package is algebraic; holographic quantum error
correction (HQEC) is Hilbert-theoretic.  The former becomes a literal
Hilbert-space code only after the physical inner product, projection,
error algebra, and recovery maps have been constructed.  In HQEC
\cite{Almheiri-Dong-Harlow15}:
the logical Hilbert space (bulk) embeds isometrically into the
physical Hilbert space (boundary), and subregion duality gives
recovery maps. In the algebraic framework:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\small
\begin{tabular}{ll}
\textbf{HQEC} & \textbf{Bar-cobar} \\
\midrule
Boundary algebra & $\cA$ \\
Bar coalgebra (coupling data) & $B(\cA)$ \\
Encoding & Bar construction $B$ \\
Decoding (recovers $\cA$) & Cobar construction $\Omega$ \\
Closed logical algebra & chiral derived centre
$C^\bullet_{\mathrm{ch}}(\cA,\cA)$ \\
Code perfection & Koszulness ($\Omega \circ B \simeq \id$) \\
Subregion recovery & Open/closed map $\cO$
\end{tabular}
\end{center}
The key difference: HQEC uses Hilbert spaces and isometries;
the bar-cobar uses dg algebras/coalgebras and quasi-isomorphisms.
The passage between the two requires the sewing envelope
and the analytic completion.
\end{remark}

\begin{remark}[Knill--Laflamme boundary]
\label{rem:thqg-knill-laflamme-boundary}
\index{Knill--Laflamme condition!THQG scope}
The Knill--Laflamme equation belongs to the subspace-code lane, not
to the bar-cobar functor lane.  Given a Hilbert code subspace
$\cC\subset H_{\mathrm{phys}}$, projection $P_\cC$, and error
operators $E_a$, the condition is
\[
P_\cC E_a^\dagger E_b P_\cC
\;=\;
\lambda_{ab}P_\cC .
\]
In this manuscript the corresponding algebraic prerequisite is the
Verdier-isotropic subspace code
$\mathbf Q_g(\cA)\subset \mathbf C_g(\cA)$
of Proposition~\ref{prop:hc-knill-laflamme}.  The functor code
$\Omega^{\mathrm{ch}}B^{\mathrm{ch}}(\cA)\to\cA$ supplies exact
bar-cobar recovery on the Koszul locus; it does not by itself supply
$P_\cC$, the adjoint $E_a^\dagger$, or the scalar matrix
$(\lambda_{ab})$.  At genus~$1$ the one-dimensional subspace code
makes the Hilbert-space Knill--Laflamme condition tautological after a
unitary inner product is supplied.  At genus $g\ge2$ the physical
condition is an additional unitarity and sewing-envelope statement.
\end{remark}

\begin{theorem}[Subregion structure from the open/closed realization;
\ClaimStatusConditional]
\label{thm:thqg-entanglement-wedge}
\index{entanglement wedge|textbf}
\index{subregion duality!algebraic|textbf}
The open/closed realization provides a subregion decomposition:
\begin{enumerate}[label=\textup{(\roman*)}]
\item A boundary module $\cM$ at a puncture defines the
 ``subregion'' (Theorem~\ref{thm:thqg-swiss-cheese}).

\item The chiral derived-centre operators reconstructible from $\cM$
 are the image of the open/closed map
 $\cO \colon C^\bullet_{\mathrm{ch}}(\cA, \cA) \to
 \operatorname{End}(\cM)$.

\item The closed projection
 $\pi_{\mathrm{cl}}(\Theta^{\mathrm{oc}}_\cA) = \Theta_\cA$
 recovers the closed-sector MC element
 (Theorem~\ref{thm:thqg-oc-projection}).

\item The annulus trace
 $\int_{S^1_p} \cM \simeq HH_*(\cM)$
 (Theorem~\ref{thm:thqg-annulus-trace})
 computes the categorical trace; after trace-class Hilbert completion
 it models the partial trace.
\end{enumerate}
\end{theorem}

\begin{proof}
Items \textup{(i)--(iii)} are the proved open/closed realization.
Item~\textup{(iv)} is Theorem~\ref{thm:thqg-annulus-trace}; the
identification with a Hilbert-space partial trace is precisely the
trace-class completion hypothesis, not an additional algebraic
consequence.
\end{proof}

\subsection{The algebraic Page constraint}
\label{subsec:thqg-entanglement-page}
\index{Page constraint!algebraic|textbf}
\index{Page curve|textbf}

\begin{theorem}[Algebraic Page constraint; \ClaimStatusConditional]
\label{thm:thqg-page-constraint}
\textit{Type signature.}
Quadrant: Open. Presentation: shifted-symplectic dg + Hilbert-bound.
Level: $3$ (ambient centre complex) + $5$ (Virasoro scalar shadow).
Hypothesis package: gravitational input; finite Lagrangian capacities;
nondegenerate Verdier pairing
\textup{(}Theorem~\ref{thm:quantum-complementarity-main}\textup{)};
for the gravity-line interpretation of part~\textup{(iv)}, the
BTZ / Cardy hypothesis package of
Remark~\ref{rem:thqg-ent-algebraic-sector}.

\medskip
For a gravitational input $(\cA, X, \langle \cdot, \cdot \rangle, k)$
satisfying the hypotheses of
Theorem~\ref{thm:quantum-complementarity-main}, and any genus
$g \ge 1$ with finite Lagrangian capacities:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \textup{(Dimensional Page bound.)}
 Any von~Neumann entropy $S$ of a state in a bipartite system
 whose factors have dimensions at most $C_g(\cA)$ and $C_g(\cA^!)$
 satisfies:
 \begin{equation}\label{eq:thqg-page-bound}
 S
 \;\le\;
 \log\min\bigl(C_g(\cA),\; C_g(\cA^!)\bigr).
 \end{equation}

\item \textup{(Complementarity sum.)}
 \begin{equation}\label{eq:thqg-page-complementarity}
 C_g(\cA) + C_g(\cA^!)
 \;=\;
 \dim \cH_g(\cA).
 \end{equation}

\item \textup{(Self-dual capacity.)}
 For $\cA \cong \cA^!$:
 $S \le \log\!\bigl(\tfrac{1}{2}\dim \cH_g(\cA)\bigr)$.

\item \textup{(Scalar complementarity.)}
 At the scalar level:
 \begin{equation}\label{eq:thqg-page-virasoro-scalar}
 F_g^{\mathrm{sc}}(\mathrm{Vir}_c)
 + F_g^{\mathrm{sc}}(\mathrm{Vir}_{26-c})
 \;=\;
 13 \cdot \lambda_g^{\mathrm{FP}}
 \quad\textup{on the uniform-weight scalar lane}.
 \end{equation}
\end{enumerate}
\end{theorem}

\begin{proof}
Part (i): standard von Neumann bound.
Part (ii): Theorem~\ref{thm:quantum-complementarity-main}.
Part (iii): Corollary~\ref{cor:thqg-III-dimension-parity}.
Part (iv): $F_g^{\mathrm{sc}}=\kappa\lambda_g^{\mathrm{FP}}$ on the
uniform-weight scalar lane, and
$\kappa(\mathrm{Vir}_c) + \kappa(\mathrm{Vir}_{26-c}) = 13$.
\end{proof}

\begin{remark}[The Page transition at the self-dual point]
\label{rem:thqg-page-transition}
\index{Page transition|textbf}
For the Virasoro family, scalar complementarity is the involution
$c\mapsto 26-c$.  The self-dual value is $c=13$, where the two scalar
coefficients agree.  For $c>13$ the smaller scalar coefficient lies on
the Verdier-dual side; for $c<13$ it lies on the original side.  Thus
the dimensional Page bound changes its controlling summand at
$c=13$.  Calling this change the Hawking/Page transition is an
additional gravitational interpretation, not a theorem of the
Lagrangian decomposition alone. The named gravitational interpretation
is supplied by the BTZ / Cardy hypothesis package of
Remark~\ref{rem:thqg-ent-algebraic-sector}: with modular invariance,
vacuum (identity-block) dominance in the high-temperature
channel~\cite{Cardy86}, BTZ saddle stability~\cite{BTZ92}, and
Maloney--Witten / Manschot--Moore Farey-tail
completion~\cite{Maloney-Witten07}, the algebraic Page transition at
$c=13$ identifies
with the Hawking--Page phase transition in the bulk gravity dual.
\end{remark}

\begin{remark}[Constructive approximation of the Page constraint]
\label{rem:thqg-page-constructive}
\index{Page constraint!constructive approximation}
The shadow obstruction tower provides a finite-order algebraic
approximation to the Page constraint at each degree:
\begin{enumerate}[label=\textup{(\arabic*)}]
\item Degree $2$ (scalar): the constraint is
 \eqref{eq:thqg-page-virasoro-scalar}.
\item Degree $3$ (cubic): the Virasoro dual scalar sum is
 $S_3(c)+S_3(26-c)=4$.  Vanishing at degree~$3$ occurs only for
 families whose cubic shadow vanishes.
\item Degree $4$ (quartic): the discriminant complementarity
 for the Virasoro dual pair is the explicit identity
 \[
 \Delta(c)+\Delta(26-c)
 =
 \frac{6960}{(5c+22)(152-5c)}
 \]
 from Theorem~\ref{thm:shadow-connection}; it is not a
 family-independent constant.
\item All degrees: $\Theta_\cA = \varprojlim \Theta_\cA^{\le r}$
 satisfies the full algebraic constraint whenever the finite-depth or
 bounded class-$\mathsf M$ convergence hypotheses hold.
\end{enumerate}
\end{remark}
