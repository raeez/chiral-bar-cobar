\section{Symplectic code structure from Lagrangian isotropy}
\label{sec:hc-knill-laflamme}
\index{symplectic code!from Lagrangian isotropy|textbf}

\begin{remark}[Two code structures]
\label{rem:hc-two-codes}
\index{holographic code!two structures}
The algebraic framework produces two distinct but related
code structures:
\begin{enumerate}[label=\textup{(\alph*)}]
\item The \emph{subspace code}: $\cC = \mathbf{Q}_g(\cA) \subset
  \mathbf{C}_g(\cA) = \cH$
  (Lagrangian summand inside the ambient complex).
  This is the code relevant for the Knill--Laflamme condition.
\item The \emph{functor code}: encoding $B \colon \cA \to B(\cA)$,
  decoding $\Omega \colon B(\cA) \to \cA$, with exact recovery
  $\Omega \circ B \simeq \id$ on the Koszul locus.
  This is the code relevant for holographic bulk reconstruction.
\end{enumerate}
The two are related: the subspace code governs the
genus-by-genus phase-space decomposition, while the functor
code governs the full algebraic reconstruction.  The Koszulness
condition~(K4) controls both: it ensures Lagrangian splitting~(a)
and exact recovery~(b).
\end{remark}

\begin{proposition}[Knill--Laflamme from Lagrangian isotropy;
\ClaimStatusProvedHere]
\label{prop:hc-knill-laflamme}
\index{Knill--Laflamme condition|textbf}
\index{Lagrangian isotropy!Knill--Laflamme}
Let $(\cA, \cA^!)$ be a \emph{unitary} chiral Koszul pair
\textup{(}positive-definite Shapovalov form\textup{)}
satisfying the
hypotheses of Theorem~\textup{\ref{thm:quantum-complementarity-main}(C2)}.
Define:
\begin{itemize}
\item $\cH := \mathbf{C}_g(\cA)$ \quad
  \textup{(}ambient complex, the ``physical Hilbert space''\textup{)},
\item $\cC := \mathbf{Q}_g(\cA)$ \quad
  \textup{(}Lagrangian summand, the ``code subspace''\textup{)},
\item $\cC^\perp := \mathbf{Q}_g(\cA^!)$ \quad
  \textup{(}complementary Lagrangian, the ``error space''\textup{)}.
\end{itemize}
Then the following three-layer error-correction structure holds:

\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Verdier isotropy \textup{(}code/error decoupling\textup{)}.}
  The Verdier pairing $\langle \cdot, \cdot \rangle_{\mathbb{D}}$
  vanishes on $\cC \times \cC$ and on
  $\cC^\perp \times \cC^\perp$:
  \begin{equation}\label{eq:hc-verdier-isotropy}
  \langle v, w \rangle_{\mathbb{D}} = 0
  \quad\text{for } v, w \in \cC
  \;\;\text{(resp.\ both in } \cC^\perp\text{)}.
  \end{equation}
  This is the algebraic necessary condition for quantum error
  correction: code states are invisible to error states under
  the natural pairing.

\item \emph{Shapovalov cross-pairing \textup{(}non-degenerate, not orthogonal\textup{)}.}
  Define $\langle v, w \rangle_S := \langle v, \sigma(w)
  \rangle_{\mathbb{D}}$.
  Since $\cC = V^+$ and $\cC^\perp = V^-$ are the $\pm1$-eigenspaces
  of the Verdier involution~$\sigma$
  \textup{(}Proposition~\textup{\ref{prop:lagrangian-eigenspaces}(iii)}\textup{)},
  the Shapovalov form restricts to:
  \begin{equation}\label{eq:hc-shapovalov-cross-pairing}
  \langle v, w \rangle_S = -\langle v, w \rangle_{\mathbb{D}}
  \quad\text{for } v \in \cC,\; w \in \cC^\perp.
  \end{equation}
  In particular, $\cC$ and $\cC^\perp$ are \emph{not}
  Shapovalov-orthogonal in general: the Verdier cross-pairing
  $\cC \times \cC^\perp \to \bC$ is non-degenerate
  \textup{(}it is the Lagrangian polarization\textup{)},
  so the Shapovalov cross-pairing is equally non-degenerate
  with a sign flip.
  The bipartite decomposition
  $\cH = \cC \oplus \cC^\perp$ is a \emph{Lagrangian}
  splitting \textup{(}Verdier-isotropic summands\textup{)},
  not an orthogonal one.
\end{enumerate}

\smallskip\noindent
\emph{Scope.}\;
Part~(i) establishes the \emph{Lagrangian prerequisite}
for the symplectic code: each summand is
self-orthogonal under the Verdier pairing,
so code states decouple from each other and error
states decouple from each other.
Part~(ii) shows the cross-pairing is non-degenerate
with a sign flip, giving a \emph{symplectic}
code/error structure rather than an orthogonal one.
The full Knill--Laflamme condition
$P_\cC E^\dagger E P_\cC = c(E) P_\cC$
requires the physical Hilbert space inner product
\textup{(}from the state--operator correspondence
in unitary theories\textup{)}, which is a separate
structure from the Verdier pairing.
At genus~$1$
\textup{(}$\dim \cC = 1$\textup{)}, the full KL condition
is automatic regardless.
\end{proposition}

\begin{proof}
\emph{Part (i).}
The Verdier involution $\sigma$ satisfies
$\langle \sigma(v), \sigma(w) \rangle_{\mathbb{D}} =
-\langle v, w \rangle_{\mathbb{D}}$
(anti-commutativity,
Proposition~\ref{prop:lagrangian-eigenspaces}(ii)).
For $v, w \in \cC = V^+$ (eigenvalue $+1$ of $\sigma$):
$\langle v, w \rangle_{\mathbb{D}} =
\langle \sigma(v), \sigma(w) \rangle_{\mathbb{D}} =
-\langle v, w \rangle_{\mathbb{D}}$,
so $\langle v, w \rangle_{\mathbb{D}} = 0$.
The same argument with eigenvalue $-1$ gives
isotropy of $\cC^\perp$.

\emph{Part (ii).}
For $v \in \cC = V^+$ and $w \in \cC^\perp = V^-$:
$\sigma(v) = v$ and $\sigma(w) = -w$, so
$\langle v, w \rangle_S = \langle v, \sigma(w)
\rangle_{\mathbb{D}} = \langle v, -w \rangle_{\mathbb{D}}
= -\langle v, w \rangle_{\mathbb{D}}$.
Since $V^+$ and $V^-$ are complementary Lagrangians,
the Verdier cross-pairing
$\langle -, - \rangle_{\mathbb{D}}\colon V^+ \times V^- \to \bC$
is a perfect pairing
\textup{(}non-degeneracy of the total form plus isotropy
of each factor\textup{)}.
Hence $\langle v, w \rangle_S = -\langle v, w \rangle_{\mathbb{D}}$
is equally non-degenerate.
\end{proof}

\begin{remark}[Relation to Almheiri--Dong--Harlow]
\label{rem:hc-adh-comparison}
\index{Almheiri--Dong--Harlow}
The Lagrangian decomposition
$\cH = \cC \oplus \cC^\perp$
is the algebraic incarnation of the
Almheiri--Dong--Harlow \cite{Almheiri-Dong-Harlow15}
entanglement wedge / complementary wedge decomposition.
The splitting is \emph{symplectic} (Verdier-isotropic
summands with non-degenerate cross-pairing,
\textup{(\ref{eq:hc-shapovalov-cross-pairing})}),
not orthogonal.
The bulk operator~$\Phi$ is in the entanglement wedge
of $\cC$ if and only if its restriction to
$\cC^\perp$ vanishes under the Verdier pairing,
precisely the condition
for $\Phi$ to be a ``code operator'' acting within the
code subspace.
\end{remark}

\begin{theorem}[Symplectic code structure; \ClaimStatusProvedHere]
\label{thm:hc-symplectic-code}
\index{symplectic code|textbf}
\index{holographic code!symplectic structure|textbf}
Let $(\cA, \cA^!)$ be a chirally Koszul pair satisfying the
hypotheses of Theorem~\textup{\ref{thm:quantum-complementarity-main}(C2)}.
The Lagrangian decomposition
$\mathbf{C}_g(\cA) = \mathbf{Q}_g(\cA) \oplus \mathbf{Q}_g(\cA^!)$
defines a \emph{symplectic code} in the following precise sense:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Lagrangian code subspace.}
  The code subspace $\cC := \mathbf{Q}_g(\cA)$ is Lagrangian:
  it is Verdier-isotropic
  \textup{(}Proposition~\textup{\ref{prop:hc-knill-laflamme}(i)}\textup{)}
  and half-dimensional
  \textup{(}$\dim \cC = \tfrac{1}{2}\dim \mathbf{C}_g(\cA)$\textup{)}.

\item \emph{Symplectic error correction.}
  Errors preserving the Lagrangian structure are correctable
  by projection along $\mathbf{Q}_g(\cA^!)$: the recovery map is
  the symplectic projection
  \[
    \pi_\cC \colon \mathbf{C}_g(\cA) \longrightarrow
    \mathbf{Q}_g(\cA)
  \]
  along $\mathbf{Q}_g(\cA^!)$, determined by the Lagrangian
  splitting.  For any operator $E$ preserving the Lagrangian
  decomposition \textup{(}i.e., $E(\cC) \subset \cC$\textup{)},
  $\pi_\cC \circ E$ acts as the identity on~$\cC$.

\item \emph{Non-degenerate cross-pairing.}
  The Verdier pairing restricted to
  $\cC \times \mathbf{Q}_g(\cA^!)$ is a perfect pairing
  with sign flip
  \textup{(}Proposition~\textup{\ref{prop:hc-knill-laflamme}(ii)}\textup{)},
  providing the non-degenerate duality between
  code and error spaces that replaces orthogonal
  complementarity in the symplectic setting.
\end{enumerate}
\end{theorem}

\begin{proof}
Part~(i) follows from Theorem~C:
$\mathbf{Q}_g(\cA) + \mathbf{Q}_g(\cA^!) = \mathbf{C}_g(\cA)$
with the summands complementary Lagrangians under the Verdier
pairing.  Isotropy is
Proposition~\ref{prop:hc-knill-laflamme}(i);
half-dimensionality is the Lagrangian condition.

Part~(ii): given the direct sum decomposition
$\mathbf{C}_g(\cA) = \cC \oplus \cC^\perp$ with
$\cC^\perp = \mathbf{Q}_g(\cA^!)$, the projection
$\pi_\cC$ is well-defined.  If $E(\cC) \subset \cC$,
then for $v \in \cC$: $E(v) = v' + w$ with
$v' \in \cC$, $w \in \cC^\perp$; but $E(v) \in \cC$
forces $w = 0$, so $\pi_\cC(E(v)) = E(v)$ and
recovery is exact.

Part~(iii) is Proposition~\ref{prop:hc-knill-laflamme}(ii):
the Verdier cross-pairing is non-degenerate by the
Lagrangian polarization, with the Shapovalov form
giving $\langle v, w \rangle_S = -\langle v, w
\rangle_{\mathbb{D}}$.
\end{proof}

\begin{remark}[Unitary Koszul algebras and physical QEC]
\label{rem:hc-unitary-physical-qec}
\index{Knill--Laflamme condition!unitary case}
\index{Tomita--Takesaki!modular conjugation}
For \emph{unitary} Koszul algebras
\textup{(}positive-definite Shapovalov form, physical
state--operator correspondence\textup{)},
the physical inner product provides a Hermitian
structure $\langle -, - \rangle_{\mathrm{phys}}$
on $\mathbf{C}_g(\cA)$ in addition to the Verdier
pairing.  The question of whether
$\mathbf{Q}_g(\cA)$ and $\mathbf{Q}_g(\cA^!)$ are
\emph{orthogonal} in this physical inner product
is the content of the Tomita--Takesaki analogy:
if the Verdier involution $\sigma$ acts
anti-unitarily with respect to
$\langle -, - \rangle_{\mathrm{phys}}$
\textup{(}as the modular conjugation $J$ does in
standard modular theory\textup{)}, then Lagrangian
isotropy upgrades to Hilbert space orthogonality,
recovering the full Knill--Laflamme condition
$P_\cC E^\dagger E P_\cC = c(E) P_\cC$.
At genus~$1$, where $\dim \mathbf{Q}_g(\cA) = 1$,
this is automatic \textup{(}the KL condition for a
$1$-dimensional code subspace is trivially satisfied\textup{)}.
\end{remark}

\begin{conjecture}[Physical QEC at higher genus;
\ClaimStatusConjectured]
\label{conj:hc-physical-qec}
\index{Knill--Laflamme condition!higher genus conjecture}
For a unitary chirally Koszul pair $(\cA, \cA^!)$,
the Verdier involution $\sigma$ acts as an
anti-unitary operator on $\mathbf{C}_g(\cA)$
equipped with the physical inner product
for all $g \ge 1$.
Equivalently, $\mathbf{Q}_g(\cA) \perp_{\mathrm{phys}}
\mathbf{Q}_g(\cA^!)$,
and the symplectic code of
Theorem~\textup{\ref{thm:hc-symplectic-code}} is
simultaneously an orthogonal quantum error-correcting
code satisfying the Knill--Laflamme condition.
\end{conjecture}

\section{Koszulness is exact holographic reconstruction}
\label{sec:hc-koszulness-exact-qec}
\index{exact holographic reconstruction|textbf}

\begin{theorem}[\textbf{G12}: Koszulness $\Leftrightarrow$
exact holographic reconstruction; \ClaimStatusProvedHere]
\label{thm:hc-koszulness-exact-qec}
\index{G12|textbf}
For a chiral algebra $\cA$ on a smooth projective curve~$X$,
the following are equivalent:
\begin{enumerate}[label=\textup{(\roman*)}]
\item $\cA$ is chirally Koszul
  \textup{(}any of the $10$ unconditional conditions K1--K10 of
  Theorem~\textup{\ref{thm:koszul-equivalences-meta}}\textup{)}.

\item The bar-cobar code admits exact recovery:
  $\Omega(B(\cA)) \xrightarrow{\sim} \cA$
  \textup{(}this is condition K4; the equivalence
  \textup{(i)}$\Leftrightarrow$\textup{(ii)} is tautological\textup{)}.

\item The holographic dual admits exact bulk reconstruction:
  every bulk operator in $C^\bullet_{\mathrm{ch}}(\cA, \cA)$ is
  recoverable from the boundary data $B(\cA)$.
\end{enumerate}
When $\cA$ is Koszul:
\begin{enumerate}[label=\textup{(\alph*)}]
\item The Lagrangian decomposition
  $\cH_g = \cC \oplus \cC^\perp$
  \textup{(}Theorem~C\textup{)}
  provides the code/error decomposition.

\item The Lagrangian code/error decomposition holds with
  symplectic \textup{(}not orthogonal\textup{)} cross-pairing
  \textup{(}Proposition~\textup{\ref{prop:hc-knill-laflamme}}\textup{)}.
  The full Knill--Laflamme condition requires the
  physical inner product from the state--operator
  correspondence, which at genus~$1$ is automatic.

\item The shadow depth $r_{\max}(\cA)$ classifies the
  redundancy structure
  \textup{(}Theorem~\textup{\ref{thm:hc-shadow-redundancy}}\textup{)}.

\item The entanglement entropy
  \textup{(}Theorem~\textup{\ref{thm:ent-scalar-entropy}},
  Chapter~\textup{\ref{chap:entanglement-modular-koszul}}\textup{)}
  is the code's information entropy.
\end{enumerate}
\end{theorem}

\begin{proof}
(i) $\Leftrightarrow$ (ii):
Characterization~K4 of
Theorem~\ref{thm:koszul-equivalences-meta} states that
$\Omega(B(\cA)) \to \cA$ is a quasi-isomorphism if and only
if $\cA$ is Koszul.
The quasi-isomorphism is the \emph{exact recovery} of the
code: the decoding functor $\Omega$ perfectly inverts the
encoding $B$ on the Koszul locus.  This is Theorem~B.

(ii) $\Rightarrow$ (iii):
Exact recovery gives $\cA$ from $B(\cA)$.
Since the chiral Hochschild complex
$C^\bullet_{\mathrm{ch}}(\cA, \cA)$ is a functor of $\cA$
(functorial in quasi-isomorphisms of chiral algebras),
the composition
$B(\cA) \xrightarrow{\Omega} \cA
\xrightarrow{C^\bullet_{\mathrm{ch}}}
C^\bullet_{\mathrm{ch}}(\cA, \cA)$
recovers the full bulk data from the encoded data.
By the open/closed realization
(Theorem~\ref{thm:thqg-swiss-cheese}),
this reconstruction extends to the holographic datum.

(iii) $\Rightarrow$ (i):
If $\cA$ is not Koszul, the counit
$\Omega(B(\cA)) \to \cA$ is not a quasi-isomorphism:
$H^*(\Omega(B(\cA))) \not\cong H^*(\cA)$.
Then the bulk operators depending on the ``lost'' cohomology
classes cannot be recovered from $B(\cA)$,
contradicting~(iii).

Consequences (a)--(d) follow from the respective references:
(a)~Theorem~C,
(b)~Proposition~\ref{prop:hc-knill-laflamme},
(c)~Theorem~\ref{thm:hc-shadow-redundancy},
(d)~Theorem~\ref{thm:ent-scalar-entropy}.
\end{proof}

\begin{remark}[Converse: non-Koszul $\Rightarrow$ code failure]
\label{rem:hc-non-koszul-failure}
\index{non-Koszul!code failure}
The theorem predicts that non-Koszul chiral algebras correspond
to gravitational theories with \emph{information loss}.
Concrete candidates: simple quotients $L_k(\mathfrak{g})$ at
admissible levels where vacuum null vectors break PBW
degeneration~(K1).  At these points, the bar spectral sequence
fails to collapse at $E_2$, the cobar recovery map has
non-trivial kernel, and exact bulk reconstruction fails.
Physically, this is a \emph{gravitational phase transition}:
the holographic error-correcting code degenerates.
\end{remark}

\section{Shadow depth as redundancy structure}
\label{sec:hc-redundancy}
\index{shadow depth!redundancy structure|textbf}
\index{redundancy channels|textbf}

\begin{conjecture}[Shadow depth classifies code distance]
\label{conj:thqg-shadow-depth-code-distance}
\index{shadow depth!code distance conjecture}
The shadow depth $r_{\max}(\cA)$ of a chirally Koszul algebra
classifies the redundancy structure of its holographic code:
the number of independent recovery channels equals
$r_{\max} - 2$.
\end{conjecture}

\begin{theorem}[Shadow depth controls redundancy;
\ClaimStatusProvedHere]
\label{thm:hc-shadow-redundancy}
\index{shadow depth!error correction}
For a chirally Koszul algebra $\cA$, the shadow depth
$r_{\max}(\cA)$ controls the \emph{redundancy structure}
of the holographic code as follows.

\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Fundamental datum.}
  The scalar-level shadow $\kappa(\cA)$
  \textup{(}arity~$2$\textup{)} is the fundamental logical
  datum: the minimum information required for code
  reconstruction.

\item \emph{Recursive recovery.}
  For each arity $r \ge 3$, the shadow
  $\Sh_r(\Theta_\cA) \in \cA^{\mathrm{sh}}_{r,\bullet}$
  \textup{(}as a cohomology class in the shadow algebra\textup{)}
  is uniquely determined by the lower-arity data
  $\{\Sh_s\}_{s < r}$ via the Maurer--Cartan equation
  \textup{(}Theorem~\textup{\ref{thm:recursive-existence}}\textup{)}.
  The representative $\Theta^{(r)}_\cA$ is determined up to
  gauge equivalence \textup{(}controlled by $H^1$ of the
  deformation complex at arity~$r$\textup{)}, but the shadow
  class itself is canonical.
  Each recovery step provides one \emph{redundancy channel}:
  erasure of the arity-$r$ data is correctable.

\item \emph{Redundancy count.}
  The number of independent redundancy channels is:
  \begin{center}
  \renewcommand{\arraystretch}{1.3}
  \begin{tabular}{ccll}
  \textbf{Class} & $r_{\max}$ & \textbf{Channels}
    & \textbf{Prototype} \\
  \midrule
  G & $2$ & $0$ & Heisenberg, lattice \\
  L & $3$ & $1$ & Affine Kac--Moody \\
  C & $4$ & $2$ & $\beta\gamma$ \\
  M & $\infty$ & $\aleph_0$ \textup{(}countably infinite\textup{)}
    & Virasoro, $W_N$
  \end{tabular}
  \end{center}
  For class~M with shadow radius $\rho(\cA) < 1$, the
  infinite recovery procedure converges absolutely.
\end{enumerate}
This resolves
Conjecture~\textup{\ref{conj:thqg-shadow-depth-code-distance}}:
shadow depth classifies the \emph{redundancy structure}
\textup{(}number of recovery channels\textup{)}, with
the code distance in the arity filtration equal to $2$
for all families \textup{(}the scalar datum is always
essential\textup{)}.
\end{theorem}

\begin{proof}
\emph{Part (i).}  The scalar shadow $\kappa(\cA)$ is the
genus-$1$ obstruction class (Theorem~D), determining the
modular curvature.  It cannot be recovered from lower-arity
data (there is no arity $< 2$).

\emph{Part (ii).}  The recursive existence theorem
(Theorem~\ref{thm:recursive-existence}) constructs the
shadow obstruction tower $\Theta_\cA = \varprojlim \Theta_\cA^{\le r}$
by solving the MC equation at each arity.  At arity $r \ge 3$,
the equation
$d(\Theta^{(r)}) + \tfrac{1}{2}\sum_{r_1+r_2=r}
[\Theta^{(r_1)}, \Theta^{(r_2)}] = 0$
determines the \emph{shadow class}
$[\Theta^{(r)}] \in H^0(\operatorname{gr}^r \fg^{\mathrm{mod}})$
from $\{\Theta^{(s)}\}_{s < r}$.
The representative $\Theta^{(r)}$ is determined up to
the gauge freedom controlled by~$H^1$, but the
cohomology class is canonical
($H^2$ controls the obstruction, both finite-dimensional at each arity).
This provides the recovery channel: given $\kappa$ and
lower-arity data, reconstruct the arity-$r$ shadow.

\emph{Part (iii).}
Class~G: $\Sh_r = 0$ for $r \ge 3$; zero channels.
Class~L: $\Sh_3 \ne 0$, $\Sh_r = 0$ for $r \ge 4$; one channel.
Class~C: $\Sh_4 \ne 0$ (and possibly $\Sh_3$), $\Sh_r = 0$
for $r \ge 5$; two channels.
Class~M: $\Sh_r \ne 0$ for all $r$; countably many channels.
Convergence for class~M follows from the shadow growth rate
bound $|\Sh_r| = O(\rho(\cA)^r \cdot r^{-5/2})$
(Theorem~\ref{thm:shadow-radius}), with absolute convergence
for $\rho < 1$.
\end{proof}
