\chapter[Master Reconstruction Theorem]%
{Master Reconstruction Theorem: the open Beilinson tower on a tangential
log curve}
\label{ch:master-reconstruction}

\begin{abstract}
On a tangential log curve $(X,D,\tau)$ the open quadrant of
$(\mathrm{Open}\cup\mathrm{CY})\times(\mathrm{Cat}\cup\mathrm{Chain})$
is organized into a five-level tower of mathematical objects connected
by named adjunctions. From the factorization dg-category $\cC^\op$ on
$(X,D,\tau)$, a vacuum chart $b$ produces the chart algebra
$A_b = \End_\cC(b)$; the chiral bar coalgebra
$B(A_b)=T^c(s^{-1}\bar A_b)$ is the twisting object dual to $A_b$;
the chiral derived centre $\DerZch(A_b)\simeq \ChirHoch^\bullet(A_b)$
is the closed-sector cochain object; line and brane operators are the
modules and bimodules over the centre; the modular trace and the
clutching system on $\Mbar_{g,n}$ project the entire stack onto its
scalar shadow. Each forgetful step has a reconstruction theorem with a
named hypothesis package: Morita, the chiral Koszul reflection
$K^2\simeq\mathrm{id}$, Hochschild concentration in $\{0,1,2\}$, the
Drinfeld-double recovery, modular reconstruction. The
$\mathbb Z/2$-fixed sublocus $\KSDual$ under the Verdier involution
$A\mapsto A^!$ is the locus where every reconstruction step is an
equivalence; on $\KSDual$ the Master Reconstruction is the $E_1$ open
shadow of a single Bourbaki-structural datum, and Theorems~A, B, C, D,
H are corollaries indexed by tower level. Vertical holographic
equivalences identify each level with its companion in Vol~II
($\SCchtop$-brace bulk; gravity-line completion) and Vol~III (6d hCS
quartic anomaly; $\PhiFA_d$; $\SpCh_{\Sigma_{d-1},C}$; Drinfeld
double via $Y^+(X)$; $\kappa_\BKM(\Phi_N)=c_N(0)/2$).
\end{abstract}

\section{The seven primitive data}
\label{sec:mr-seven-data}
%===========================================================

The open quadrant carries one structure with seven slots, all defined
before any choice of presentation.

\begin{definition}[Open holographic datum on a tangential log curve;
\ClaimStatusProvedHere]
\label{def:mr-open-datum}
\index{open holographic datum!seven primitive data}
An \emph{open holographic datum on a tangential log curve} is a tuple
\[
 (X,\,D,\,\tau;\;\cC^\op,\,b,\,A_b,\,\DerZch(\cC),\,\Theta_\cC,\,\Tr_\cC),
\]
where:
\begin{enumerate}[label=\textup{(\arabic*)}]
\item $X$ is a smooth projective curve over a field $k$ of
characteristic zero;
\item $D=\sum_i p_i$ is a reduced effective divisor on $X$;
\item $\tau$ is a tangential structure along $D$ (a choice of formal
nonzero tangent vector at each $p_i\in D$);
\item $\cC^\op$ is a factorization dg-category on $(X,D,\tau)$ in the
Francis--Gaitsgory sense, equipped with its Verdier-symmetric
monoidal structure on $\Ran(X,D,\tau)$;
\item $b\in\cC^\op$ is a vacuum chart, a choice of compact generator
sufficient for Morita reconstruction;
\item $A_b=\End_\cC(b)$ is the chart algebra at $b$, an augmented
factorization $E_1$-chiral algebra on $(X,D,\tau)$;
\item $\DerZch(\cC)$ is the chiral derived centre of $\cC^\op$,
identified through the Hochschild-cochain object on a chosen chart by
\[
\DerZch(\cC) \simeq C^\bullet_{\ch}(A_b,A_b),
\qquad b\in\cC^\op \text{ Morita-generating;}
\]
\item $\Theta_\cC$ is the universal Maurer--Cartan element of the
ordered convolution dg Lie algebra
$\mathfrak g^{E_1}_{\cC}=\mathrm{Conv}(B(A_b),A_b)$, satisfying
\[
 D\Theta_\cC + \tfrac12[\Theta_\cC,\Theta_\cC] = 0;
\]
\item $\Tr_\cC$ is the modular trace, a section of the determinant
gerbe over $\Mbar_{g,n}$ together with a clutching system compatible
with the gluing strata.
\end{enumerate}
The category of open holographic data on $(X,D,\tau)$ is denoted
$\mathsf{OpenDatum}(X,D,\tau)$.
\end{definition}

The seven slots are the primitive data; everything else in the open
quadrant is constructed from them. Slots~(4) and~(5) carry the
geometry and the chart; slot~(6) carries the chart-level algebra;
slot~(7) is the closed-sector cochain object; slot~(8) is the
universal MC element witnessing the chiral structure on the curve;
slot~(9) is the modular trace witnessing the family-level theorems on
$\Mbar_{g,n}$.

\begin{remark}[Type signature of the datum]
\label{rem:mr-type-signature-datum}
The datum lives in the type
$(\mathrm{Open},\,\mathrm{Cat}\&\mathrm{Chain},\,\text{level }0\text{ to }5,\,
H_0\text{--}H_5)$, where $H_0,\ldots,H_5$ are the hypothesis packages
of the six reconstruction theorems below. Quadrant Open: the curve
$X$ is the geometry. Presentation Cat$\&$Chain: $\cC^\op$ is a
$(\infty,1)$-category with a chain-level Hochschild presentation.
\end{remark}

\section{The five-level open Beilinson tower}
\label{sec:mr-tower}
%===========================================================

The seven primitive data organize into a tower of forgetful functors
between five mathematical objects, with reconstruction theorems
returning each step.

\begin{definition}[Open Beilinson tower on $(X,D,\tau)$;
\ClaimStatusProvedHere]
\label{def:mr-tower}
\index{Beilinson tower!open quadrant}
The \emph{open Beilinson tower} on $(X,D,\tau)$ is the diagram
\[
\begin{aligned}
 \cC^\op
 &\quad\text{[level 0: factorization dg-category]}
\\[-2pt]
 \quad\downarrow\;F_0\colon b\mapsto \End_\cC(b)
\\[-2pt]
 A_b
 &\quad\text{[level 1: chart algebra]}
\\[-2pt]
 \quad\downarrow\;F_1\colon A_b\mapsto T^c(s^{-1}\bar A_b)
\\[-2pt]
 B(A_b)=T^c(s^{-1}\bar A_b)
 &\quad\text{[level 2: bar / twisting coalgebra]}
\\[-2pt]
 \quad\downarrow\;F_2\colon \text{Hochschild cochains}
\\[-2pt]
 \DerZch(A_b)\simeq \ChirHoch^\bullet
 &\quad\text{[level 3: universal closed sector;}
\\[-2pt]
 &\quad\text{\phantom{[level 3: } physical bulk via OCA, Vol~II]}
\\[-2pt]
 \quad\downarrow\;F_3\colon \text{module / bimodule formation}
\\[-2pt]
 \mathsf{Line}(A_b),\;\mathsf{Brane}(A_b)
 &\quad\text{[level 4: line / brane operators]}
\\[-2pt]
 \quad\downarrow\;F_4\colon \text{modular trace}+\text{clutching}
\\[-2pt]
 (\kappa(A_b),\,Z(A_b),\,F_g(A_b))
 &\quad\text{[level 5: scalar shadow]}
\end{aligned}
\]
The forgetful functor $F_0$ assigns the endomorphism algebra at the
chosen vacuum; $F_1$ is the bar functor with desuspension and
deconcatenation; $F_2$ is the formation of chiral Hochschild
cochains; $F_3$ produces the categories of modules over $A_b$ and of
bimodules over $\DerZch(A_b)$; $F_4$ takes the modular trace on
$\Mbar_{g,n}$ together with the clutching system at the boundary
divisors.
\end{definition}

\begin{remark}[Five objects, never conflated]
\label{rem:mr-five-objects}
\index{five-object firewall}
The tower distinguishes five distinct objects which the
master-concordance type discipline forbids ever to identify
(Convention~\ref{conv:master-five-object-firewall}):
\[
 A_b,\quad B(A_b),\quad A_b^{i}=H^\bullet B(A_b),
 \quad A_b^{!},\quad \DerZch(A_b).
\]
The arrow $\Omegach B(A_b)\xrightarrow{\sim} A_b$ is bar--cobar
inversion. The arrow $\mathbb{D}_{\Ran} B(A_b)\rightsquigarrow
A_b^{!,\co}_\infty$ is Verdier--Koszul duality. The arrow
$A_b\mapsto C^\bullet_{\ch}(A_b,A_b)$ is the Hochschild-centre
construction. None of these arrows is replaceable by another.
\end{remark}

\section{Reconstruction theorems with named hypothesis packages}
\label{sec:mr-reconstruction}
%===========================================================

The open Beilinson tower is anchored by five reconstruction theorems.
Each names the hypothesis package under which the corresponding
forgetful step has a left or two-sided inverse.

\begin{theorem}[$F_0$-reconstruction: Morita;
\ClaimStatusProvedHere]
\label{thm:mr-morita}
\index{reconstruction!Morita; level 0 to 1}
\textup{[Type: Open, Cat, $0\leftrightarrow 1$, hypothesis package
$H_0$.]}
Let $\cC^\op$ be a factorization dg-category on $(X,D,\tau)$ and
$b\in\cC^\op$ a compact generator. The functor
\[
 \cC^\op \longrightarrow A_b\text{-}\mathsf{mod}^\fact,
 \qquad
 c \longmapsto \mathrm{Hom}_\cC(b,c),
\]
is an equivalence of factorization $(\infty,1)$-categories. Conversely,
$A_b$-modules recover $\cC^\op$ once $b$ is fixed.

\textbf{Hypothesis package $H_0$.} $\cC^\op$ is small, idempotent
complete, and dualizable in the Lurie symmetric-monoidal $\infty$-cat
of factorization presentable stable categories on $(X,D,\tau)$;
$b$ is compact and Morita-generating in the sense of
Francis--Gaitsgory.
\end{theorem}

\begin{proof}
Lurie's Morita theorem in the factorization context
(Francis--Gaitsgory, \cite[\S4.2]{Francis2012}; promoted to
$(\infty,2)$ by \cite[Chapter~IV.5]{GR17}). Compact generation, idempotent completion, and
dualizability are exactly the hypotheses needed for reconstruction.
\end{proof}

\begin{theorem}[$F_1$-reconstruction: chiral Koszul reflection
$K^2\simeq\mathrm{id}$, Theorem~A;
\ClaimStatusConditional]
\label{thm:mr-A}
\index{reconstruction!Theorem A; level 1 to 2}
\textup{[Type: Open, Chain, $1\leftrightarrow 2$, hypothesis package
$H_1=$ (H1)+(H2)+(H3) of \S\ref{ch:theorem-A-infinity-2}.]}
The bar--cobar adjunction
\[
 \Omegach_X\;:\;\mathsf{CoAlg}^{\fact,\conil,\co}_X
 \rightleftarrows
 \mathsf{Alg}^{\fact,\aug,\comp}_X\;:\;\Bbarch_X
\]
restricts on the Koszul locus $\Kosz(X)$ to a chain-level
quasi-isomorphism
$\Omegach_X\Bbarch_X(A)\xrightarrow{\sim} A$. This is
Theorem~\ref{thm:koszul-reflection} in the form $K^2\simeq\mathrm{id}$
on $\Kosz(X)$.

\textbf{Hypothesis package $H_1$.}
\textup{(H1)} $A$ is augmented;
\textup{(H2)} $A$ is augmentation-ideal complete (weakly curved in the
Positselski sense);
\textup{(H3)} the conformal-weight bar pieces
$(\Bbarch_X(A))_w$ are finite-dimensional for each $w$.
On the standard landscape $\mathsf G\cup\mathsf L\cup\mathsf C\cup
\mathsf M\cup\mathsf B$, hypothesis (H3) holds with the class-$\mathsf
M$ row taken in its weight-completed pro-conilpotent ambient
$\widehat A=\varprojlim_N A/A_{\geq N}$.
\end{theorem}

\begin{proof}
Theorem~\ref{thm:koszul-reflection}, with proof body
\S\ref{ch:theorem-A-infinity-2}; Vallette--Hinich--Positselski
specialised to the chiral ambient by
\cite{BD04,Val16,Hinich2003,Positselski2018}.
\end{proof}

\begin{theorem}[$F_2$-reconstruction via Theorem~H: Hochschild
concentration in $\{0,1,2\}$;
\ClaimStatusConditional]
\label{thm:mr-H}
\index{reconstruction!Theorem H; level 1 to 3}
\textup{[Type: Open, Chain, $1\rightarrow 3$, hypothesis package
$H_3=H_1\cup\{\text{Koszul, PBW, finite-dim conformal pieces}\}$.]}
On the Koszul / PBW locus, the chiral Hochschild cochain object of
the chart algebra is concentrated in degrees $\{0,1,2\}$:
\[
 \ChirHoch^n(A_b) = 0
 \quad\text{for every } n\notin\{0,1,2\},
\]
and on the affine Kac--Moody row $\ChirHoch^1(V_k(\mathfrak g))\cong
\mathfrak g$ at non-critical level. Hence the centre lane reads at
most three nontrivial groups.

\textbf{Hypothesis package $H_3$.} Hypothesis package $H_1$ together
with Koszulness in the Shelton--Yuzvinsky sense and a PBW basis on
each conformal-weight piece. Class $\mathsf M$ requires the
weight-completed / pro-conilpotent ambient or the Felder BRST chain
model.
\end{theorem}

\begin{proof}
Theorem~\ref{thm:hochschild-concentration-E1} on the ordered $E_1$ side
before averaging, established on the first page of the
collision-depth spectral sequence on $\overline{C}^{\mathrm{ord}}_{p+2}(X)$
via pure-braid Orlik--Solomon Koszulness. Symmetric concentration
follows from the averaging step
$\Av\colon \mathfrak g^{E_1}_A \to \mathfrak g^{\mathrm{mod}}_A$
which preserves the $\{0,1,2\}$ envelope.
\end{proof}

\begin{theorem}[$F_3$-reconstruction: Drinfeld-double recovery;
\ClaimStatusConditional]
\label{thm:mr-drinfeld-double}
\index{reconstruction!Drinfeld double; level 4 to 3}
\textup{[Type: Open, Cat, $4\rightarrow 3$, hypothesis package $H_4$.]}
On the Koszul locus, the closed-sector centre is recovered from line
and brane operators by Drinfeld-double formation: the bimonoidal
category of bimodules over $\DerZch(A_b)$ is equivalent to a Drinfeld
double of the line category over $A_b$,
\[
 \mathsf{Bimod}(\DerZch(A_b))
 \;\simeq\;
 \mathcal Z_{\mathrm{Dr}}(\mathsf{Line}(A_b))
\]
where $\mathcal Z_{\mathrm{Dr}}$ is the Drinfeld centre of a braided
monoidal category.

\textbf{Hypothesis package $H_4$.} $A_b$ admits a finite-type chart
presentation; the line category $\mathsf{Line}(A_b)$ is rigid braided
and admits compact generation; the elliptic global obstruction
$\mathrm{obs}^{(1)}_{\mathrm{double}}\in
H^2(\mathfrak{grt}^{\mathrm{ell}},\mathfrak{sp}(A_b)\otimes
\mathfrak{sp}(A_b^!))$ vanishes, equivalently $r_{\max}(A_b)=2$ at
genus~$1$. At genus zero the package reduces to rigid braiding only.
\end{theorem}

\begin{proof}
The genus-zero statement is the Joyal--Street Drinfeld-centre theorem
applied to the line category, with the chiral promotion of
\cite[Chapter~7]{EGNO}. The genus-$1$ obstruction is recorded in
Frontier~F5 and the elliptic associator package of Enriquez
$\mathrm{GRT}^{\mathrm{ell}}_1(\mathbb Q)$. The conditional status
reflects the open frontier at $g\geq 2$.
\end{proof}

\begin{theorem}[$F_4$-reconstruction: modular reconstruction;
\ClaimStatusConditional]
\label{thm:mr-modular}
\index{reconstruction!modular; level 5 to 4}
\textup{[Type: Open, Cat, $5\rightarrow 4$, hypothesis package
$H_5$.]}
The modular trace and clutching data on $\Mbar_{g,n}$ recover the
line category from the scalar shadow:
\begin{enumerate}[label=\textup{(\roman*)}]
\item the boundary clutching maps at separating and non-separating
divisors, together with $\Tr_\cC$ on each stratum, determine the
line category up to Morita equivalence;
\item the partition function
$Z_g(A_b) = \int_{\Mbar_{g,n}} \kappa(A_b)\cdot \lambda_g$
($\lambda_g$ the Hodge class) is the universal scalar shadow at
uniform weight on the genus stratum.
\end{enumerate}

\textbf{Hypothesis package $H_5$.} Modular tensor data: rigidity,
non-degenerate $S$-matrix, factorization at all boundary divisors of
$\Mbar_{g,n}$, and consistency of the clutching system with the
chiral fusion. Multi-weight refinement carries the cross-channel
correction $\delta F_g^{\mathrm{cross}}$ at $g\geq 2$.
\end{theorem}

\begin{proof}
Modular reconstruction in the chiral category is the
Beilinson--Drinfeld factorization of \cite[\S 4.5]{BD04} together with
Wakimoto--Tsuchiya--Ueno clutching at the boundary divisors. The
uniform-weight scalar lane is Theorem~\ref{thm:genus-universality}
$\mathrm{obs}_g=\kappa\cdot\lambda_g$. Multi-weight extension with
$\delta F_g^{\mathrm{cross}}$ is recorded in
Theorem~\ref{thm:multi-weight-genus-expansion}.
\end{proof}

\section{The Master Reconstruction Theorem}
\label{sec:mr-master}
%===========================================================

The five reconstruction theorems compose into a single Bourbaki-level
structural statement: every level of the open Beilinson tower is
accessible from any other through a finite chain of named adjunctions
with explicit hypothesis packages.

\begin{theorem}[Master Reconstruction Theorem on the open quadrant]
\label{thm:mr-master}
\ClaimStatusConditional
\index{Master Reconstruction Theorem!open quadrant}
\index{open Beilinson tower!five-level reconstruction}
\textup{[Type: Open, Cat$\,\&$Chain, $0\rightarrow 5$, hypothesis
package $H_0\cup H_1\cup H_3\cup H_4\cup H_5$.]}
Let $(X,D,\tau)$ be a tangential log curve over $k$,
$\cC^\op$ a factorization dg-category with vacuum chart $b$, and
$A_b=\End_\cC(b)$. Under the hypothesis package
$H_0\cup H_1\cup H_3\cup H_4\cup H_5$ the open Beilinson tower
\[
 \cC^\op \rightleftarrows A_b
 \rightleftarrows B(A_b)
 \xrightarrow{F_2} \DerZch(A_b)
 \xrightarrow{F_3}\mathsf{Line}(A_b)
 \xrightarrow{F_4}\bigl(\kappa,Z,F_g\bigr)
\]
satisfies the following four properties:
At genus zero on $\KSDual$ the Drinfeld-double and multi-weight
residuals vanish from the statement. At higher genus the theorem has
the conditional strength of the packages $H_4$ and $H_5$.
\begin{enumerate}[label=\textup{(M\arabic*)}]
\item \label{M1}
\emph{Step-wise reconstruction.}
$F_0$ has left inverse Morita reconstruction
(Theorem~\ref{thm:mr-morita}); $F_1$ has two-sided inverse on the
Koszul locus given by chiral Koszul reflection $K^2\simeq\mathrm{id}$
(Theorem~\ref{thm:mr-A}); $F_2$ on the Koszul locus has image
concentrated in $\{0,1,2\}$ (Theorem~\ref{thm:mr-H}); $F_3$ has left
inverse Drinfeld-double recovery
(Theorem~\ref{thm:mr-drinfeld-double}); $F_4$ has left inverse
modular reconstruction (Theorem~\ref{thm:mr-modular}).
\item \label{M2}
\emph{Universal Maurer--Cartan witness.}
The element $\Theta_\cC$ of the ordered convolution dGLA
$\mathfrak g^{E_1}_\cC = \mathrm{Conv}(B(A_b),A_b)$ controls every
forgetful step: its binary collision residue $r_{A_b}(z)$ is the
universal twisting morphism, its full bar projection is the
$F_1$-image, its averaged image
$\mathrm{Sh}_{g,n}(\Theta_\cC)$ is the connection determining the
$F_4$-image.
\item \label{M3}
\emph{Universal chain-homotopy normalisation.}
The chain homotopy controlling the $F_1$-counit is
$h_{A_b} = h_{\mathrm{LV}} / \mathcal N(A_b)$ where
$\mathcal N(A_b)$ is the chain-homotopy norm of
Convention~\ref{conv:A-two-kappa-shriek}
\textup{(}equivalently $\kappa(A_b)+\kappa^!_{\mathrm{LV}}(A_b)$,
distinct from the algebra-level scalar Verdier sum
$K^\kappa(A_b)$\textup{)},
with archetype witnesses
$\mathcal N(\mathsf G)=1$ giving $h_\mathsf G$ trivial,
$\mathcal N(\mathsf L)=2(k+h^\vee)$ giving Sugawara
$h_\mathsf L = 1/(2(k+h^\vee))$,
$\mathcal N(\mathsf C)=1$ giving $h_\mathsf C$ free-field OPE,
$\mathcal N(\mathsf M)=c(5c+22)/10$ giving weight-completed
Felder BRST,
$\mathcal N(\mathsf B)=8$ giving Mukai-K3 Heisenberg.
\item \label{M4}
\emph{Five-archetype dichotomy.}
Every chart algebra $A_b$ in the standard landscape lies in one of
the five archetypes
$\mathsf G\cup\mathsf L\cup\mathsf C\cup\mathsf M\cup\mathsf B$
with shadow depth $r_{\max}\in\{2,3,4,\infty,5\}$, and admits the
$5\times 5$ $\kappa$-stratification matrix
\[
 (\kappa_{\mathrm{cat}},\,\kappa^{\mathrm{Hodge}}_{\mathrm{ch}},\,
 \kappa^{\mathrm{Heis}}_{\mathrm{ch}},\,\kappa_\BKM,\,
 \kappa_{\mathrm{fiber}}).
\]
The K3$\times E$ row anchors $\mathsf B$ at $(0,0,3,5,24)$.
\end{enumerate}
\end{theorem}

\begin{proof}
\ref{M1} is the conjunction of
Theorems~\ref{thm:mr-morita}, \ref{thm:mr-A}, \ref{thm:mr-H},
\ref{thm:mr-drinfeld-double}, and \ref{thm:mr-modular}, each on its
declared hypothesis package. \ref{M2} is the universal
Maurer--Cartan element of
Theorem~\ref{thm:mc2-bar-intrinsic} together with the collision
residue identification of
Theorem~\ref{thm:collision-residue-twisting}. \ref{M3} is recorded in
Chapter~\ref{ch:theorem-A-infinity-2} with archetype witnesses
inscribed there; the normalisation
$h_{A_b} = h_{\mathrm{LV}} / \mathcal N(A_b)$, with the level-$2$
Verdier-Ran-on-bar norm
$\mathcal N(A_b) = \kappa(A_b)+\kappa^!_{\mathrm{LV}}(A_b)$
(distinct from the algebra-level $K^\kappa$, coinciding only on
class~$\mathsf B$), is the universal
denominator unifying Theorems~A and~C. \ref{M4} is the master
classification of Chapter~\ref{ch:landscape-census}, with the K3$\times
E$ anchor proved in Vol~III $\mathsf B$-row.
\end{proof}

\section{The KSDual fixed sublocus}
\label{sec:mr-ksdual}
%===========================================================

The Verdier--Koszul involution $A\mapsto A^!$ on the Koszul locus is
an involutive equivalence: $A^{!!}\simeq A$ on the standard landscape.
The fixed sublocus is the place where every step of the Master
Reconstruction degenerates from an adjunction with comparison map to
a strict equivalence.

\begin{definition}[KSDual sublocus;
\ClaimStatusProvedHere]
\label{def:mr-ksdual}
\index{KSDual!definition}
\index{Verdier involution!fixed locus}
The \emph{KSDual sublocus} is the $\mathbb Z/2$-fixed sublocus of the
Koszul locus under the Verdier--Koszul involution
$\sigma\colon A_b\mapsto A_b^!$:
\[
 \KSDual(X,D,\tau) := \bigl\{\,A_b\in\Kosz(X,D,\tau)
 \;:\;\sigma(A_b)\simeq A_b\,\bigr\}.
\]
On a chart algebra $A_b\in\KSDual$, the scalar shadow constants
$\kappa(A_b)+\kappa(A_b^!)$ collapse to twice the self-dual ceiling.
\end{definition}

\begin{theorem}[Master Reconstruction degenerates to equivalence on
$\KSDual$;
\ClaimStatusConditional]
\label{thm:mr-ksdual}
\index{KSDual!Master Reconstruction equivalence}
\textup{[Type: Open, Cat$\,\&$Chain, $0\rightarrow 5$, hypothesis
package $H_0\cup H_1\cup H_3\cup H_4\cup H_5\cup\{\sigma\text{-fixed}\}$.]}
Let $A_b\in\KSDual(X,D,\tau)$. Then the Master Reconstruction Theorem
\textup{(}Theorem~\ref{thm:mr-master}\textup{)} is an equivalence in
the strong sense \textup{(}not adjunction with comparison map\textup{)}
at every level of the tower:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Hochschild concentration is exact in $\{0,1,2\}$:}
$\ChirHoch^n(A_b)=0$ for $n\notin\{0,1,2\}$, $\ChirHoch^0(A_b)=k$,
$\ChirHoch^2(A_b)$ pairs nondegenerately with $\ChirHoch^0(A_b)$
under the cyclic Verdier pairing.
\item \emph{The five-archetype dichotomy stabilises:} on $\KSDual$,
the archetype label is invariant under deformation in the open chart
moduli, hence chart-independent.
\item \emph{Universal Trace Identity is the equivariant signature:}
the modular trace satisfies the involutive identity
\[
 \Tr_\cC(A_b) = \Tr_\cC(A_b^!),
\]
hence the partition function $Z_g(A_b)$ admits a $\mathbb Z/2$-graded
decomposition into Verdier-symmetric and Verdier-antisymmetric pieces,
and the antisymmetric piece is the equivariant signature of the
involution.
\end{enumerate}
\end{theorem}

\begin{proof}
On $\KSDual$, the chiral Koszul reflection $K^2\simeq\mathrm{id}$ of
Theorem~\ref{thm:mr-A} is the identity on the chart $A_b\simeq A_b^!$;
hence the comparison map of Theorem~\ref{thm:mr-master}\ref{M1}
collapses to the identity at every level. The Hochschild
concentration sharpens to exactness because on the self-dual locus
the Verdier symmetric ambient pairs $\ChirHoch^0$ with $\ChirHoch^2$
nondegenerately, and pairs $\ChirHoch^1$ with itself
antisymmetrically; the chiral analogue is the cyclic Verdier pairing

\section{The structural meta-theorem}

\begin{theorem}[Master Reconstruction as Bourbaki structure]
\label{thm:mr-bourbaki}
\ClaimStatusConditional
\index{Master Reconstruction!Bourbaki structure}
\textup{[Type: Open, Cat$\,\&$Chain, $0\text{--}5$, hypothesis package

The species of the structure is the type signature of
Definition~\ref{def:mr-open-datum}. The morphisms are defined

Theorem~\ref{thm:mr-master}: an endomorphism preserving $\Theta_\cC$
and acting as the identity at any level acts as the identity on the
universal MC element, hence on every projection of it.
\end{proof}
