Entanglement enters the modular-Koszul tower through a scalar
projection and a higher shadow projection.  The scalar projection is
the Calabrese--Cardy single-interval law
\[
 S_{\EE}^{\mathrm{sc}}(\Sigma)
 = \frac{c(\cA)}{3}\log\frac{L}{\varepsilon},
\]
which depends on the Virasoro central charge, not on the programme
characteristic~$\kappa$ except in families where a separate
normalisation identifies the two.  For Virasoro,
$\kappa(\mathrm{Vir}_c)=c/2$ and
$\mathrm{Vir}_c^!\simeq\mathrm{Vir}_{26-c}$, hence the scalar
duality sum is
\[
 S_{\EE}^{\mathrm{sc}}(\mathrm{Vir}_c)
 +S_{\EE}^{\mathrm{sc}}(\mathrm{Vir}_{26-c})
 = \frac{26}{3}\log\frac{L}{\varepsilon}.
\]
This is the entanglement shadow of the conductor identity
$c+c^!=26$, equivalently
$\kappa(\mathrm{Vir}_c)+\kappa(\mathrm{Vir}_{26-c})=13$.

The higher projection is algebraic.  The shadow tower gives
classes~$\Sh_r(\Theta_\cA)$ and bounds their growth; a chosen
replica functional can turn those classes into numbers
$\delta S_r$.  The vanishing and convergence statements below are
therefore theorems about the shadow tower.  Their interpretation as
Faulkner--Lewkowycz--Maldacena bulk entropy corrections, or as a
quantum-extremal-surface variational principle, requires a
gravitational dual and is recorded only as a physical heuristic.

\section{Entanglement entropy from the replica trick

\subsection{Twist operators and the modular characteristic}

\begin{setup}[Entanglement framework]
\label{setup:ent-framework}
Let $(\cA, \cA^!)$ be a chiral Koszul pair
(Definition~\ref{def:thqg-standing-hypotheses})
on a smooth projective curve~$X$ of genus~$0$,
with modular characteristic~$\kappa(\cA)$
and central charge~$c(\cA)$.
Fix an interval $\Sigma = [0, L] \subset X$
with UV cutoff~$\varepsilon$.
The R\'enyi and von~Neumann entropies of~$\Sigma$
in the vacuum state are defined via the replica trick:
\begin{equation}\label{eq:ent-renyi-def}
S_n(\Sigma)
\;:=\;
\frac{1}{1-n}\,\log \Tr(\rho_\Sigma^n),
\qquad
S_{\EE}(\Sigma)
\;:=\;
\lim_{n \to 1} S_n(\Sigma)
\;=\;
-\Tr(\rho_\Sigma \log \rho_\Sigma),
\end{equation}
where $\rho_\Sigma$ is the reduced density matrix obtained by
tracing over the complement $\Sigma^c$.
\end{setup}

\begin{lemma}[Twist operator dimension; \ClaimStatusProvedElsewhere]
\label{lem:ent-twist-dimension}
\index{twist operator!conformal dimension}
The $n$-fold twist operator $\sigma_n$ creating the $n$-sheeted
branched cover has conformal dimension
\begin{equation}\label{eq:ent-twist-dim}
h_n
\;=\;
\frac{c(\cA)}{24}\Bigl(n - \frac{1}{n}\Bigr).
\end{equation}
\end{lemma}

\begin{proof}
This is the standard result of Calabrese--Cardy
\cite[Eq.~(2.5)]{Calabrese-Cardy04},
following from the Schwarzian derivative of
$z \mapsto z^{1/n}$ applied to the stress tensor.
For the Virasoro family, $c = 2\kappa$; the
general relationship between $\kappa$ and~$c$ is family-dependent
(see Setup~\ref{setup:ent-framework}).
\end{proof}

\subsection{Calabrese--Cardy from the modular characteristic}

\begin{theorem}[Entanglement entropy at the scalar level;
\ClaimStatusProvedHere]
\label{thm:ent-scalar-entropy}
\index{Calabrese--Cardy formula!from modular characteristic}
\index{entanglement entropy!scalar level|textbf}
In the framework of Setup~\textup{\ref{setup:ent-framework}},
the scalar-level R\'enyi and von~Neumann entropies of a
single interval are:
\begin{align}
S_n^{\mathrm{sc}}(\Sigma)
&\;=\;
\frac{c(\cA)}{6}\Bigl(1 + \frac{1}{n}\Bigr)
\log\frac{L}{\varepsilon},
\label{eq:ent-renyi-scalar}
\\[4pt]
S_{\EE}^{\mathrm{sc}}(\Sigma)
&\;=\;
\frac{c(\cA)}{3}\,\log\frac{L}{\varepsilon}.
\label{eq:ent-vn-scalar}
\end{align}
\end{theorem}

\begin{proof}
The replica partition function on the $n$-sheeted branched cover is
(non-chiral convention, left$+$right movers):
\begin{equation}\label{eq:ent-replica-partition}
\log Z_n
\;=\;
-\frac{c(\cA)}{6}\Bigl(n - \frac{1}{n}\Bigr)
\log\frac{L}{\varepsilon},
\end{equation}
which follows from the two-point function
$\langle \sigma_n(0)\,\bar\sigma_n(L) \rangle
= (L/\varepsilon)^{-2(h_n + \bar{h}_n)}
= (L/\varepsilon)^{-4h_n}$
with~$h_n = \bar{h}_n$ from~\eqref{eq:ent-twist-dim}
\textup{(}non-chiral: both chiralities contribute equally\textup{)}.
Since $Z_1 = 1$ by normalization:
\[
S_n
= \frac{\log Z_n}{1-n}
= \frac{-(c/6)(n - 1/n)}{1-n}\,\log(L/\varepsilon)
= \frac{c}{6}\cdot\frac{n+1}{n}\,\log(L/\varepsilon),
\]
giving~\eqref{eq:ent-renyi-scalar}. The limit
$\lim_{n \to 1} S_n = (c/3)\log(L/\varepsilon)$
gives~\eqref{eq:ent-vn-scalar}.
\end{proof}

\begin{remark}[Connection to the $\hat{A}$-genus]
\label{rem:ent-ahat-connection}
\index{A-hat genus@$\hat{A}$-genus!entanglement}
The scalar free energy
$F_g^{\mathrm{sc}}(\cA) = \kappa(\cA) \cdot \lambda_g^{\mathrm{FP}}$
\textup{(}UNIFORM-WEIGHT\textup{})
(Corollary~\ref{cor:free-energy-ahat-genus})
resums as $F(\cA;\hbar) = (\kappa/\hbar^2)[\hat{A}(i\hbar)-1]$.
The replica partition function~\eqref{eq:ent-replica-partition}
is determined by~$c$ alone; at the scalar level, the
$\hat{A}$-genus controls the \emph{genus} expansion of the
partition function, while the \emph{replica} index~$n$
determines the sewing parameters of the branched cover.
The two expansions (genus and replica) are orthogonal
projections of the same universal MC element~$\Theta_\cA$.
\end{remark}

\section{The entanglement complementarity constraint}
\label{sec:ent-complementarity}
\index{complementarity!entanglement|textbf}

\begin{theorem}[Entanglement complementarity; \ClaimStatusConditional]
\label{thm:entanglement-complementarity}
\index{entanglement complementarity|textbf}
Let $(\cA, \cA^!)$ be a chiral Koszul pair with modular
characteristics $\kappa(\cA)$, $\kappa(\cA^!)$.
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Koszul duality sum rule.}
At the scalar level, the entanglement entropies of~$\cA$
and its Koszul dual satisfy:
\begin{equation}\label{eq:ent-complementarity-sum}
S_{\EE}^{\mathrm{sc}}(\cA)
+ S_{\EE}^{\mathrm{sc}}(\cA^!)
\;=\;
\frac{c(\cA) + c(\cA^!)}{3}\,
\log\frac{L}{\varepsilon}.
\end{equation}
For the Virasoro family,
$c(\mathrm{Vir}_c) + c(\mathrm{Vir}_{26-c}) = c + (26 - c) = 26$
\textup{(}equivalently,
$\kappa(\mathrm{Vir}_c) + \kappa(\mathrm{Vir}_{26-c}) = 13$
via Theorem~\textup{\ref{thm:anomaly-koszul}}\textup{)},
giving the universal sum:
\begin{equation}\label{eq:ent-virasoro-sum}
S_{\EE}^{\mathrm{sc}}(\mathrm{Vir}_c)
+ S_{\EE}^{\mathrm{sc}}(\mathrm{Vir}_{26-c})
\;=\;
\frac{26}{3}\,\log\frac{L}{\varepsilon}.
\end{equation}

\item \emph{Self-dual pinning.}
At the self-dual point $c = 13$
\textup{(}$\mathrm{Vir}_{13}^! = \mathrm{Vir}_{13}$\textup{)},
$S_{\EE}(\cA) = S_{\EE}(\cA^!)$, and the complementarity
constraint pins the entropy:
\begin{equation}\label{eq:ent-self-dual}
S_{\EE}^{\mathrm{sc}}(\mathrm{Vir}_{13})
\;=\;
\frac{13}{3}\,\log\frac{L}{\varepsilon}.
\end{equation}

\item \emph{Shadow-level complementarity.}
Beyond the scalar level, the algebraic Virasoro shadow sums
\[
D_r^{\mathrm{sh}}(c)
:= S_r(\mathrm{Vir}_c)+S_r(\mathrm{Vir}_{26-c})
\]
are controlled by
Remark~\textup{\ref{rem:higher-degree-complementarity}}:
\begin{itemize}
\item Degree $2$: $D_2^{\mathrm{sh}} = \kappa + \kappa' = 13$
 \textup{(}universal constant\textup{)}.
\item Degree $3$: $D_3^{\mathrm{sh}} = S_3(c) + S_3(26{-}c) = 4$
 \textup{(}universal constant\textup{)}.
\item Degree $\ge 4$: $D_r^{\mathrm{sh}}(c)$ acquires non-trivial
 $c$-dependence through the Lee--Yang denominators
 $(5c{+}22)$ and $(152{-}5c)$.
\end{itemize}
The discriminant complementarity
$\Delta(\cA) + \Delta(\cA^!) =
6960/[(5c{+}22)(152{-}5c)]$
\textup{(}Theorem~\textup{\ref{thm:shadow-connection}(vi)}\textup{)}
governs the degree-$4$ structure.
These are algebraic shadow identities.  They do not identify
$D_r^{\mathrm{sh}}(c)$ with physical entropy corrections until a
replica functional and, for bulk entropy, a gravitational dual have
been specified.
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{Part (i).}
By Theorem~\ref{thm:ent-scalar-entropy},
$S_{\EE}^{\mathrm{sc}}(\cA)
= (c(\cA)/3)\log(L/\varepsilon)$.
The sum follows by linearity.
The Virasoro specialization uses
$c(\mathrm{Vir}_c) = c$ and
the duality law $c + c' = 26$
(Theorem~\ref{thm:anomaly-koszul}).

\emph{Part (ii).}
At $c = 13$: $c = c' = 13$, so
$S_{\EE}^{\mathrm{sc}}
= (13/3)\log(L/\varepsilon)$.

\emph{Part (iii).}
The degree-$2$ and degree-$3$ universality of the duality sums
$D_2^{\mathrm{sh}} = 13$, $D_3^{\mathrm{sh}} = 4$ is proved in
Remark~\ref{rem:higher-degree-complementarity}.
The degree-$\ge 4$ $c$-dependence follows from the
discriminant complementarity formula
(Theorem~\ref{thm:shadow-connection}(vi)), whose
numerator $6960$ is a universal constant but whose
denominators $(5c{+}22)(152{-}5c)$ depend on~$c$.
\end{proof}

\begin{remark}[Entanglement as complementarity projection]
\label{rem:ent-as-projection}
\index{entanglement entropy!as complementarity projection}
The complementarity theorem decomposes the gravitational
phase space into Lagrangian halves:
$\mathcal{H}_g(\cA) = Q_g(\cA) \oplus Q_g(\cA^!)$.
The entanglement entropy~\eqref{eq:ent-vn-scalar}
is the scalar projection of this decomposition,
measuring the size of each Lagrangian summand through
$c(\cA)$, which determines the twist operator
dimension.
The modular characteristic~$\kappa(\cA)$ controls the genus-$1$
obstruction; the ratio
$c/\kappa$ is family-dependent (equal to~$2$ for Virasoro but not in general).
The Verdier involution $\sigma$
(Theorem~\ref{thm:thqg-g4-s-duality})
exchanges the two summands, enforcing~\eqref{eq:ent-complementarity-sum}.
\end{remark}

\section{Shadow corrections and entanglement complexity}
\label{sec:ent-shadow-corrections}
\index{entanglement complexity!shadow depth|textbf}
\index{shadow corrections!entanglement entropy}

\begin{proposition}[Shadow entropy functionals by depth;
\ClaimStatusConditional]
\label{prop:ent-complexity-classification}
\index{entanglement complexity!four classes|textbf}
Let $\mathcal R_{n,r}$ be a family of linear replica
functionals on the degree-$r$ shadow projection of the
$n$-sheeted replica geometry, regular at $n=1$.  Define the
degree-$r$ shadow entropy coefficient by
\[
\delta S_r^{\mathrm{sh}}(\cA;\Sigma)
\;:=\;
-\,\left.\frac{\partial}{\partial n}\right|_{n=1}
\mathcal R_{n,r}\!\left(
\Sh_r(\Theta_\cA|_{\Sigma_n})\right).
\]
The algebraic vanishing and convergence of
$\delta S_r^{\mathrm{sh}}$ are determined by the shadow tower
$\Theta_\cA^{\le r}$.
The four shadow depth classes
\textup{(}Theorem~\textup{\ref{thm:shadow-archetype-classification}}\textup{)}
yield four shadow-complexity classes:

\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{l@{\quad}c@{\quad}c@{\quad}l}
\textbf{Class} & $r_{\max}$ & \textbf{Corrections}
& \textbf{Prototype} \\
\midrule
\textbf{G} \textup{(Gaussian)} & $2$ &
 none beyond the scalar term &
 Heisenberg, lattice \\
\textbf{L} \textup{(Lie/tree)} & $3$ &
 one cubic shadow coefficient &
 Affine Kac--Moody \\
\textbf{C} \textup{(Contact)} & $4$ &
 quartic coefficient \textup{(}the cubic may vanish\textup{)} &
 $\beta\gamma$ \\
\textbf{M} \textup{(Mixed)} & $\infty$ &
 infinite tower $\{\delta S_r^{\mathrm{sh}}\}_{r \ge 3}$ &
 Virasoro, $W_N$ \\
\end{tabular}
\end{center}
In particular:
\begin{enumerate}[label=\textup{(\alph*)}]
\item For class~G algebras, every algebraic shadow contribution
 $\delta S_r^{\mathrm{sh}}$ with $r\ge3$ vanishes.  The
 Calabrese--Cardy term is therefore the complete contribution
 of the shadow model.
\item For class~M algebras, if the replica functionals are
 uniformly bounded by
 $|\mathcal R_{n,r}(x)|\le C_0\|x\|\log(L/\varepsilon)$ near
 $n=1$, then
 $|\delta S_r^{\mathrm{sh}}| \le C \cdot \rho(\cA)^r \cdot r^{-5/2}
 \cdot \log(L/\varepsilon)$ where $\rho(\cA)$ is the shadow
 radius \textup{(Theorem~\textup{\ref{thm:shadow-radius}}\textup{)}.}
 The correction series converges absolutely when
 $\rho(\cA) < 1$, i.e., for $c > c^* \approx 6.125$
 in the Virasoro family.
\end{enumerate}
\end{proposition}

\begin{proof}
The shadow obstruction tower $\Theta_\cA^{\le r}$ consists of
finite-order projections of the bar-intrinsic MC element
(Theorem~\ref{thm:mc2-bar-intrinsic}). On the replica
geometry~$\Sigma_n$ (the $n$-sheeted branched cover),
$\Theta_\cA^{\le r}$ contributes to
$\log Z_n$ through the genus expansion of the partition
function restricted to graphs of degree at most~$r$; applying
$\mathcal R_{n,r}$ and differentiating at $n=1$ gives the
coefficient $\delta S_r^{\mathrm{sh}}$.
The shadow depth $r_{\max}(\cA)$ is the
smallest~$r$ at which the tower stabilizes
(Theorem~\ref{thm:shadow-archetype-classification});
for~$r > r_{\max}$, all higher shadows vanish and
$\delta S_r^{\mathrm{sh}} = 0$.

Part~(a): For class~G, $r_{\max} = 2$, so all shadows of
degree $\ge 3$ vanish identically. The scalar
level is the complete shadow content.

Part~(b): The bound on $|\delta S_r^{\mathrm{sh}}|$ follows from the
shadow growth rate
(Theorem~\ref{thm:shadow-radius}):
$|\Sh_r(\Theta_\cA)| \le C_1 \rho(\cA)^r r^{-5/2}$.
The assumed uniform operator bound on $\mathcal R_{n,r}$ and its
$n$-derivative propagates this estimate to the replica coefficient:
$|\delta S_r^{\mathrm{sh}}| \le C \rho(\cA)^r r^{-5/2}
\log(L/\varepsilon)$.
Absolute convergence of $\sum_{r \ge 3} |\delta S_r^{\mathrm{sh}}|$
follows from comparison with the geometric series
$\sum \rho^r < \infty$ when $\rho < 1$.
\end{proof}

\begin{remark}[The shadow radius as entanglement convergence radius]
\label{rem:ent-convergence-radius}
\index{shadow radius!entanglement convergence}
For Virasoro at central charge $c$:
$\rho(c) = \sqrt{(180c+872)/(c^2(5c+22))}$.
At the self-dual point $c = 13$: $\rho \approx 0.467$
(convergent algebraic shadow expansion).
At the critical string $c = 26$: $\rho \approx 0.233$
(strongly convergent algebraic shadow expansion).
At the Ising point $c = 1/2$: $\rho > 1$
(the normalized shadow power series has radius below the
unit sewing disk).

The transition at $c = c^* \approx 6.125$, where
$\rho = 1$, separates the \emph{perturbative regime}
(shadow coefficients give an absolutely convergent correction
series for bounded replica functionals) from the
\emph{asymptotic regime}.  Borel--\'Ecalle summability is a
separate genus-expansion assertion; it does not follow from the
shadow-radius estimate alone.
\end{remark}

\begin{corollary}[Shadow expansion of the entropy functional;
\ClaimStatusConditional]
\label{cor:ent-full-entropy}
\index{entanglement entropy!full formula}
For a chirally Koszul algebra $\cA$ in the convergent
regime $\rho(\cA) < 1$, and for a replica functional satisfying
the boundedness hypothesis of
Proposition~\textup{\ref{prop:ent-complexity-classification}},
the shadow entropy functional is:
\begin{equation}\label{eq:ent-full}
S_{\EE}^{\mathrm{sh}}(\cA)
\;=\;
\frac{c(\cA)}{3}\,\log\frac{L}{\varepsilon}
\;+\;
\sum_{r=3}^{r_{\max}(\cA)}
\delta S_r^{\mathrm{sh}}(\cA),
\end{equation}
where the sum terminates at $r = r_{\max}$ for classes
G, L, C and converges absolutely for class~M
with $\rho < 1$. The underlying Virasoro shadow sums are universal
constants for $r \le 3$ and acquire $c$-dependence for $r \ge 4$
\textup{(}Theorem~\textup{\ref{thm:entanglement-complementarity}(iii)}\textup{)}.
\end{corollary}

\section{The quantum extremal surface as shadow

\label{sec:ent-qes}

\begin{conjecture}[QES from shadow connection; \ClaimStatusConjectured]
\label{conj:ent-qes}
\index{quantum extremal surface!from shadow connection|textbf}
The quantum extremal surface formula
\begin{equation}\label{eq:ent-qes-formula}
S_{\QES}
\;=\;
\min_\gamma\Bigl[
\frac{\mathrm{Area}(\gamma)}{4G_N}
+ S_{\mathrm{bulk}}(\Sigma_\gamma)
\Bigr]
\end{equation}
admits the following heuristic decomposition in the modular Koszul framework
\textup{(}the proved algebraic content is the shadow obstruction tower structure; the
identification with the physical QES formula requires a gravitational
dual\textup{)}:
\begin{enumerate}[label=\textup{(\roman*)}]
\item In a holographic normalization satisfying the RT dictionary,
 the area term has boundary value
 $\mathrm{Area}(\gamma)/(4G_N) = S_{\EE}^{\mathrm{sc}}(\cA)
 = (c/3)\log(L/\varepsilon)$.

\item The bulk entropy $S_{\mathrm{bulk}}$ is represented, on the
 shadow side, by the correction series
 $\sum_{r \ge 3} \delta S_r^{\mathrm{sh}}(\cA)$.

\item The stationarity condition
 $\nabla_\gamma S_{\QES} = 0$
 is the physical image of a Ward identity of the modular shadow connection
 $\nabla^{\mathrm{hol}}_{g,n} = d - \Sh_{g,n}(\Theta_\cA)$
 restricted to the replica geometry.
\end{enumerate}
\end{conjecture}

\begin{remark}[Algebraic evidence]
Part~(i) is Theorem~\ref{thm:ent-scalar-entropy}.
Part~(ii) is
Proposition~\ref{prop:ent-complexity-classification}.
For part~(iii), the shadow connection
$\nabla^{\mathrm{hol}} = d - \Sh(\Theta_\cA)$ satisfies
$(\nabla^{\mathrm{hol}})^2 = 0$ (proved as Theorem~\ref{thm:thqg-g6-soft-graviton}
in~\S\ref{sec:thqg-soft-graviton-theorems}).
Its Ward identities give
$\nabla^{\mathrm{hol}}(\Sh_r) + o^{(r)} = 0$
at each degree $r \ge 3$. On the replica geometry,
$S_{\QES}$ is modelled by a functional of the shadow projections
$\{\Sh_r|_{\Sigma_n}\}_{r \ge 2}$; the stationarity
$\delta S_{\QES}/\delta\gamma = 0$ is the content of
the Ward identity at the appropriate degree.
\end{remark}

\begin{remark}[RT as scalar approximation]
\label{rem:ent-rt-scalar}
\index{Ryu--Takayanagi formula!as scalar approximation}
The Ryu--Takayanagi formula
$S_{\mathrm{RT}} = \mathrm{Area}(\gamma_{\min})/(4G_N)$
\cite{Ryu-Takayanagi06}
is the scalar-level entanglement entropy
$S_{\EE}^{\mathrm{sc}}(\cA) = (c/3)\log(L/\varepsilon)$.
The quantum corrections
(Faulkner--Lewkowycz--Maldacena~\cite{Faulkner-Lewkowycz-Maldacena13},
Engelhardt--Wall~\cite{Engelhardt-Wall14})
upgrading RT to the quantum extremal surface formula are
heuristically identified with the shadow coefficients
$\delta S_r^{\mathrm{sh}}$; this identification requires a gravitational dual
and is not a theorem of the present framework.

The shadow depth classification measures the
\emph{quantum complexity of the RT formula}:
class~G (no corrections), class~L (one correction),
class~C (two corrections), class~M (infinite but convergent
tower for $\rho < 1$).
\end{remark}

For each standard family, the scalar entanglement entropy

the entropy coefficient is the central-charge coefficient

\begin{proof}

all class~G and~L algebras have finite shadow obstruction towers

\begin{remark}[Additivity under independent sums]
\label{rem:ent-additivity}
\index{entanglement entropy!additivity}
For independent chiral algebras $\cA = \cA_1 \oplus \cA_2$
with vanishing mixed OPE

$\kappa(\cA) = \kappa(\cA_1) + \kappa(\cA_2)$
and therefore
$S_{\EE}(\cA) = S_{\EE}(\cA_1) + S_{\EE}(\cA_2)$
at the scalar level. The shadow corrections also
separate by independent sum factorization.
\end{remark}

\begin{remark}[Monotonicity of the R\'enyi spectrum]
\label{rem:ent-renyi-monotone}
\index{Renyi entropy@R\'enyi entropy!monotonicity}
The R\'enyi entropy
$S_n^{\mathrm{sc}} = (c/6)(1+1/n)\log(L/\varepsilon)$
is non-increasing in $n \ge 1$ at the scalar level,
with limits:

\begin{theorem}[BTZ entropy from the modular characteristic;
\ClaimStatusConditional]
\label{thm:ent-btz-entropy}
\index{BTZ black hole!entropy|textbf}

charge $c = 3\ell/(2G)$ \textup{(}Brown--Henneaux\textup{)}.
For a BTZ black hole of mass~$M$ and angular momentum~$J$

$\bar{h} = (M\ell - J)/2$:

\item \emph{Leading entropy.}
The Bekenstein--Hawking entropy is given by the Cardy
formula with effective conformal weights
$h_{\mathrm{eff}} = h - c/24$,
$\bar{h}_{\mathrm{eff}} = \bar{h} - c/24$:
\begin{equation}\label{eq:ent-btz-leading}
S_{\mathrm{BH}}
\;=\;
2\pi\sqrt{\frac{c\,h_{\mathrm{eff}}}{6}}
+ 2\pi\sqrt{\frac{c\,\bar{h}_{\mathrm{eff}}}{6}}
\;=\;
2\pi\sqrt{\frac{\kappa\,h_{\mathrm{eff}}}{3}}
+ 2\pi\sqrt{\frac{\kappa\,\bar{h}_{\mathrm{eff}}}{3}},
\end{equation}
where $\kappa = \kappa(\mathrm{Vir}_c) = c/2$
\textup{(}so $c/6 = \kappa/3$\textup{)}.
In the semiclassical regime $h \gg c/24$,
$h_{\mathrm{eff}} \approx h$ and the formula reduces to
$S_{\mathrm{BH}} \approx 2\pi\sqrt{c\,h/6}$.
For the non-rotating case $J = 0$, the exact formula gives
$S_{\mathrm{BH}} = 2\pi\sqrt{c(M\ell - c/24)/6}$.

\item \emph{Logarithmic correction.}

\begin{equation}\label{eq:ent-btz-log}

\;=\;
S_{\mathrm{BH}}
\;-\;\frac{3}{2}\log\frac{S_{\mathrm{BH}}}{2\pi}
\;+\; O(1).
\end{equation}
The coefficient $-3/2$ is universal: it arises from the
second derivative of the free energy at the saddle point
and is independent of the shadow depth class.

$F_g^{\mathrm{sc}}(\mathrm{Vir}_c) = \kappa\cdot\lambda_g^{\mathrm{FP}}$
\textup{(}UNIFORM-WEIGHT\textup{})
\textup{(}Corollary~\textup{\ref{cor:free-energy-ahat-genus}}\textup{)}

\begin{equation}\label{eq:ent-btz-genus-expansion}

\;=\;

where $C_g$ depends on the shadow free energies
$F_1, \ldots, F_g$ through the saddle-point expansion.

$\delta_{\mathrm{pf}}^{(g,0)}$ contribute starting at $g = 2$.
\end{enumerate}

and vacuum dominance in the high-temperature limit give
$\log Z(\tau) \sim \pi c/(12\,\mathrm{Im}\,\tau)$ as

$h_{\mathrm{eff}} = h - c/24$ yields
the leading entropy $2\pi\sqrt{c\,h_{\mathrm{eff}}/6}
= 2\pi\sqrt{\kappa\,h_{\mathrm{eff}}/3}$.

\emph{Part~(iii).}

For Virasoro (class~M), the planted-forest corrections

\begin{conjecture}[Page curve from complementarity; \ClaimStatusConjectured]
\label{conj:ent-page-curve}
\index{Page curve|textbf}

is the elementary crossing-time computation given the two linear
growth rates below. The physical identification of $c/6$ and
$(26-c)/6$ with Hawking radiation emission and island absorption
respectively, and the identification of the crossing with the
Page transition, are heuristic:
they require a gravitational dual in which the bar complex
$B(\cA^!)$ controls the island saddle, an identification
that is not a theorem of the present framework.

\smallskip
Let $(\mathrm{Vir}_c, \mathrm{Vir}_{26-c})$ be the Virasoro
Koszul pair with entanglement growth rates
$v = c/6$ \textup{(}matter\textup{)} and
$v' = (26-c)/6$ \textup{(}Koszul dual\textup{)}.
The Page curve of the radiation entropy is:
\begin{equation}\label{eq:ent-page-curve}
S_{\mathrm{rad}}(t)
\;=\;
\min\!\bigl(
 S_{\mathrm{Hawking}}(t),\;
 S_{\mathrm{island}}(t)
\bigr),
\end{equation}
where:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Hawking phase} ($t < t_P$):
 $S_{\mathrm{Hawking}}(t) = (c/6)\,t$.

\item \emph{Island phase} ($t > t_P$):
 $S_{\mathrm{island}}(t)
 = S_{\mathrm{BH}} - ((26-c)/6)\,t$.

\item \emph{Page time}:
 \begin{equation}\label{eq:ent-page-time}
 t_P \;=\; \frac{3\,S_{\mathrm{BH}}}{13}
 \;=\;
 \frac{6\,S_{\mathrm{BH}}}{c + (26-c)}.
 \end{equation}
 The denominator is the complementarity constant
 $c + c' = 26$, equivalently $\kappa + \kappa' = 13$
 \textup{(}Theorem~\textup{\ref{thm:entanglement-complementarity}}\textup{)}.
\end{enumerate}
\end{conjecture}

\begin{remark}[Evidence]
The Hawking phase growth rate $c/6$ is the
Calabrese--Cardy entanglement production rate
after a local quench, identified with Hawking radiation
emission. The island phase uses the Koszul dual rate
$(26-c)/6$: once the bar complex $B(\cA^!)$ dominates
$B(\cA)$, the entanglement is controlled by the dual
modular characteristic $\kappa' = (26-c)/2$.

The Page time is determined by the crossing condition:
the two branches
$S_{\mathrm{Hawking}}(t) = (c/6)t$ (increasing)
and
$S_{\mathrm{island}}(t) = S_{\mathrm{BH}} - ((26{-}c)/6)t$
(decreasing) intersect when
\[
\frac{c}{6}\,t_P = S_{\mathrm{BH}} - \frac{26{-}c}{6}\,t_P,
\qquad
t_P\,\frac{c + (26{-}c)}{6} = S_{\mathrm{BH}},
\qquad
t_P = \frac{6\,S_{\mathrm{BH}}}{26}
= \frac{3\,S_{\mathrm{BH}}}{13}.
\]
\end{remark}

\begin{corollary}[Self-dual Page curve; \ClaimStatusConjectured]
\label{cor:ent-page-self-dual}
\index{Page curve!self-dual point}
At the self-dual point $c = 13$:
\begin{enumerate}[label=\textup{(\alph*)}]
\item The Hawking and island growth rates are equal:
 $v = v' = 13/6$.
\item The Page time is $t_P = 3S_{\mathrm{BH}}/13$
 and the Page curve is symmetric about~$t_P$.
\item The Page entropy is
 $S_{\mathrm{Page}} = S_{\mathrm{BH}}/2$,
 reflecting the perfect $\cA \leftrightarrow \cA^!$ symmetry.
\end{enumerate}
\end{corollary}

\begin{proof}
At $c = 13$: $\kappa = \kappa' = 13/2$, so $v = v' = 13/6$.
The crossing condition gives
$t_P = 6S_{\mathrm{BH}}/26 = 3S_{\mathrm{BH}}/13$.
The Page entropy $S_{\mathrm{Page}}
= (13/6)\cdot 3S_{\mathrm{BH}}/13 = S_{\mathrm{BH}}/2$.
\end{proof}

\begin{remark}[Island formula as Koszul dual dominance]
\label{rem:ent-island-koszul}
\index{island formula!Koszul interpretation}
In the language of
Penington~\cite{Penington19} and
Almheiri--Engelhardt--Marolf--Maxfield~\cite{AEMM19},
the \emph{island} is the region of the bulk where the
Koszul dual bar complex $B(\cA^!)$ dominates the matter

in the complementarity sum
$\mathbf{C}_g(\cA) = \mathbf{Q}_g(\cA) \oplus

(Theorem~\ref{thm:quantum-complementarity-main}).
The physical content of the Page curve is that
entanglement growth is controlled by $\kappa(\cA)$ at
early times and by $\kappa(\cA^!)$ at late times;
the complementarity constant $\kappa + \kappa' = 13$
determines the crossover.
\end{remark}

\begin{remark}[Page curve from Koszul complementarity; \ClaimStatusHeuristic]
\label{rem:page-curve-complementarity}
\index{Page curve!from complementarity constant}
The Page curve emerges from the Virasoro complementarity sum
$\kappa(\mathrm{Vir}_c) + \kappa(\mathrm{Vir}_{26-c}) = 13$

\end{remark}

\begin{proposition}[Real-root unitary submodule entanglement;

\;=\;

\;=\;

\]

S_{\mathrm{EE}}(V^{\mathrm{unit}})
\;=\;
\frac{2}{3}\,\log(L/\varepsilon)
\;>\;0,

$2$-plane. The remaining $c_{\mathrm{eff}}/24 - 2/24 = -168/24 = -7$

$c = \mathrm{rk}(P) = 2$. The Calabrese--Cardy formula

Remark~\ref{rem:ent-negative} is the partition-function

entanglement comes only from the unitary submodule with central

\section{Topological entanglement and ground-state degeneracy from the 3d TQFT}
\label{sec:ent-3d-tqft}

The scalar entanglement formula
$S_{\EE} = (c/3)\log(L/\varepsilon)$ of
Theorem~\ref{thm:ent-scalar-entropy} tracks the $1+1$d boundary CFT
Calabrese--Cardy coefficient. When the boundary chiral algebra
$\mathbf H_{\Delta_5}$ sits on the 3d bulk holographic dual -- the
non-semisimple MTC 3d TQFT
$Z^{\mathrm{DGGPR}}$ of
Vol~III Chapter~\ref{V3-ch:modular-trace}
\S\ref{V3-sec:modtr-3d-tqft-state-spaces} -- a second
entanglement contribution appears: the
\emph{topological entanglement entropy} $\gamma = \log\mathcal D$ of
Kitaev--Preskill 2006 \emph{PRL}~96 \S 2 and Levin--Wen 2006 \emph{PRL}~96
\S 2, extended to the non-semisimple setting by Gainutdinov--Runkel 2017
\emph{JMP}~58 \S 4. The topological correction is a sub-leading constant
in $S_{\EE}$ that detects the non-trivial topological order of the
3d TQFT.

\begin{theorem}[Topological entanglement from non-semisimple total quantum dimension]
\label{thm:ent-topological-entanglement}
\ClaimStatusProvedHere
Let $\mathcal C =
\mathrm{Rep}^{\mathrm{fd}}(\mathfrak u_{\zeta_8}
(\widehat{\mathfrak m}_{\Delta_5}))$ be the Kerler--Lyubashenko MTC of
Vol~III Chapter~\ref{V3-ch:modular-trace}
Theorem~\ref{V3-thm:modtr-Plancherel-KerlerLyubashenko}, and let
$Z^{\mathrm{DGGPR}}$ be the corresponding De~Renzi--Gainutdinov--Geer--Patureau-Mirand--Runkel
3d TQFT (arXiv:2003.09814). The total quantum dimension of $\mathcal C$ is
\[
 \mathcal D(\mathbf H_{\Delta_5})
 \;=\;
 \sqrt{\sum_{\lambda\in\Lambda_\infty}d_\lambda^2},
 \qquad
 d_\lambda
 \;:=\;
 \dim_{\mathrm{qu}}(P_\lambda),
\]
where the sum runs over indecomposable projective covers (not over simples)
and $d_\lambda$ is the Kerler--Lyubashenko quantum dimension. The
topological entanglement entropy of a disc-like spatial region in the
corresponding 3d TQFT is
\[
 \boxed{\;
 \gamma(\mathbf H_{\Delta_5})
 \;=\;
 \log\mathcal D(\mathbf H_{\Delta_5}).\;}
\]
The total entanglement entropy of an interval $\Sigma = [0, L]$ of
length $L\gg\varepsilon$ in the boundary CFT of the 3d TQFT is
\[
 S_{\EE}(\Sigma)
 \;=\;
 \frac{c(\mathbf H_{\Delta_5})}{3}\log\frac{L}{\varepsilon}
 \;-\;
 2\,\gamma(\mathbf H_{\Delta_5})
 \;+\;
 \mathcal O(1/L),
\]
with the coefficient $-2\gamma$ reflecting the two disconnected
boundary components of the entangling interval (Kitaev--Preskill 2006
eq.~(5)).
\end{theorem}

\begin{proof}
The Kitaev--Preskill 2006 \emph{PRL}~96 eq.~(5) topological-entanglement
formula was derived for semisimple MTCs in the string-net /
Turaev--Viro setting. Its non-semisimple extension is
Gainutdinov--Runkel 2017 Thm.~4.12: the formula
$\gamma = \log\mathcal D$ holds in any non-semisimple finite tensor
category after replacing the sum over simples by a sum over
indecomposable projective covers and the quantum dimension by the
Kerler--Lyubashenko 2001 LMS~LNS~262 \S5 quantum trace.

The Calabrese--Cardy $(c/3)\log(L/\varepsilon)$ leading term is
Theorem~\ref{thm:ent-scalar-entropy}; the topological subtraction
$-2\gamma$ is the constant correction from the two endpoints of the
entangling interval (each endpoint contributes $-\gamma$ by the
Kitaev--Preskill disc subtraction). The non-semisimple $\mathcal D$
involves the projective-cover dimensions $d_\lambda$; for
$\mathbf H_{\Delta_5}$, these are controlled by the Radford $S^4$
formula (Radford 1994 \emph{Amer.~J.~Math.}~116 \S 4 Thm.~4):
$\mathcal D^2 = |\mathrm{Hopf\ dim}(\mathfrak u_{\zeta_8})|\cdot
v_{\mathrm{ribbon}}^{-1}$, with $v_{\mathrm{ribbon}}\in\mu_8$ the
ribbon-twist anomaly of
Kerler--Lyubashenko 2001 \S 5.4.

\emph{Multi-path verification}:
(i) Topological: Kitaev--Preskill 2006 eq.~(5) $\gamma = \log\mathcal D$;
(ii) Non-semisimple extension: Gainutdinov--Runkel 2017 Thm.~4.12;
(iii) Hopf-algebraic: Radford 1994 Thm.~4 + Kerler--Lyubashenko 2001
\S 5.4;
(iv) Semisimple-limit consistency: at the semisimple quotient
$\mathcal C^{\mathrm{ss}}$, $\mathcal D^2 = \sum_{\lambda\ \mathrm{simple}}
d_\lambda^2$ recovers Kirillov--Ostrik 2002 \emph{Adv.~Math.}~171
Prop.~4.1.6.
\end{proof}

\begin{remark}[Topological entanglement as a derived-centre invariant]

for holographic codes (Pastawski--Yoshida--Harlow--Preskill 2015

\end{remark}
