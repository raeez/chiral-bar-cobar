\section{The holographic modular Koszul datum}
\label{sec:thqg-holographic-datum}

\begin{definition}[Standing hypotheses]
\label{def:thqg-standing-hypotheses}
A \emph{gravitational input} is a quadruple $(\cA, X, \langle\cdot,\cdot\rangle, k)$ where:
\begin{enumerate}[label=\textup{(\roman*)}]
\item $\cA$ is a chiral algebra on a smooth projective curve $X$, chirally Koszul
  in the sense of Theorem~\ref{thm:koszul-equivalences-meta} (any of the ten
  unconditionally equivalent characterizations);
\item $\cA$ is cyclically admissible
  (Definition~\ref{def:cyclically-admissible}): conformal grading, exhaustive
  filtration, bounded poles, invariant pairing $\langle\cdot,\cdot\rangle$;
\item the HS-sewing criterion holds: polynomial OPE growth and
  subexponential sector growth (Theorem~\ref{thm:general-hs-sewing});
\item the level $k$ avoids the critical value (for $\widehat{\mathfrak{g}}$:
  $k \ne -h^\vee$; for universal $\mathcal{W}$-algebras: generic level).
\end{enumerate}
All results below assume these hypotheses unless explicitly relaxed.
\end{definition}

\begin{definition}[Holographic modular Koszul datum]
\label{def:thqg-holographic-datum}
Given a gravitational input $(\cA, X, \langle\cdot,\cdot\rangle, k)$, the
\emph{holographic modular Koszul datum} is the sextuple
\begin{equation}\label{eq:thqg-holographic-datum}
\mathcal{H}(\cA) \;=\; \bigl(\cA,\;\cA^!,\;\mathcal{C},\;r(z),\;
\Theta_\cA,\;\nabla^{\mathrm{hol}}\bigr)
\end{equation}
where:
\begin{enumerate}[label=\textup{(\alph*)}]
\item $\cA^! = (H^*(B(\cA)))^\vee$ is the Koszul dual chiral algebra;
\item $\mathcal{C} \simeq \cA^!\text{-}\mathsf{mod}$ is the line-operator category;
\item $r(z) = \Res^{\mathrm{coll}}_{0,2}(\Theta_\cA) \in
  \cA^! \widehat{\otimes} \cA^![[z^{-1},z]]$ is the collision residue;
\item $\Theta_\cA := D_\cA - d_0 \in \MC(\gAmod)$ is the bar-intrinsic universal
  MC element (Theorem~\ref{thm:mc2-bar-intrinsic});
\item $\nabla^{\mathrm{hol}}_{g,n} = d - \operatorname{Sh}_{g,n}(\Theta_\cA)$
  is the modular shadow connection.
\end{enumerate}
Every component is a projection of $\Theta_\cA$.
\end{definition}

\begin{definition}[Shadow depth and archetype classification]
\label{def:thqg-shadow-archetype}
The \emph{shadow depth} $r_{\max}(\cA) \in \{2,3,4,\ldots\} \cup \{\infty\}$ is
the smallest $r$ such that the obstruction class
$o^{(r+1)}(\cA) \in H^2(F^{r+1}\gAmod / F^{r+2}\gAmod, d_2)$ vanishes
(or $\infty$ if no such $r$ exists).  The four archetype classes
(Definition~\ref{def:shadow-depth-classification}) are:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\small
\begin{tabular}{clll}
\textbf{Class} & \textbf{Name} & $r_{\max}$ & \textbf{Standard examples} \\
\hline
G & Gaussian & $2$ & Heisenberg, free fermion, lattice \\
L & Lie/tree & $3$ & Affine $\widehat{\mathfrak{g}}$, symplectic fermion \\
C & Contact & $4$ & $\beta\gamma$, $bc$ \\
M & Mixed & $\infty$ & Virasoro, $\mathcal{W}_N$ ($N \ge 3$)
\end{tabular}
\end{center}
\end{definition}

\begin{definition}[Gravitational phase space]
\label{def:thqg-gravitational-phase-space}
The \emph{gravitational phase space} at genus $g$ is the ambient complex
$\mathbf{C}_g(\cA)$ equipped with the shifted-symplectic Lagrangian splitting
\begin{equation}\label{eq:thqg-lagrangian-splitting}
\mathbf{C}_g(\cA) \;\simeq\; \mathbf{Q}_g(\cA) \oplus \mathbf{Q}_g(\cA^!)
\end{equation}
from the complementarity theorem (Theorem~\ref{thm:quantum-complementarity-main})
and its proved Verdier/ambient shifted-symplectic upgrades
\textup{(}Proposition~\ref{prop:ptvv-lagrangian},
Theorem~\ref{thm:ambient-complementarity-fmp}\textup{)}.
The summands $\mathbf{Q}_g(\cA)$ and $\mathbf{Q}_g(\cA^!)$ are the bulk and defect
sectors.  The Verdier involution $\sigma$ interchanges them.
\end{definition}

\section{The sixteen results}
\label{sec:thqg-ten-theorems}

\begin{theorem}[\textbf{G1}: Perturbative finiteness]\ClaimStatusProvedHere
\label{thm:thqg-g1-finiteness}
Let $(\cA, X, \langle\cdot,\cdot\rangle, k)$ be a gravitational input.
For every $g \ge 0$ and $n \ge 1$, the genus-$g$ amplitude
$\langle \Theta_\cA^{\otimes n} \rangle_g$ converges absolutely.
No UV renormalization is needed: the Fulton--MacPherson compactification
$\overline{C}_n(X)$ resolves all collision singularities, and
the arity-cutoff lemma (Lemma~\ref{lem:arity-cutoff}) bounds the contributing
graphs at each genus.
\end{theorem}

\begin{theorem}[\textbf{G2}: Shadow-depth classification]\ClaimStatusProvedHere
\label{thm:thqg-g2-complexity}
The shadow depth $r_{\max}(\cA)$ equals the minimal
transferred homotopy-algebraic complexity of the genus-$0$
modular deformation package:
\[
\ell_j^{(0),\mathrm{tr}} = 0 \qquad \text{for } j > r_{\max}(\cA).
\]
In particular:
\begin{enumerate}[label=\textup{(\roman*)}]
\item $r_{\max} = 2$: only the quadratic shadow survives (Gaussian);
\item $r_{\max} = 3$: the cubic shadow is nonzero and no higher arity survives (Lie);
\item $r_{\max} = 4$: the cubic term vanishes, the quartic term survives, and no higher arity survives (Contact);
\item $r_{\max} = \infty$: infinitely many transferred arities survive (Mixed).
\end{enumerate}
The shadow-formality identification
(Proposition~\ref{prop:shadow-formality-low-arity}) is proved at arities $2,3,4$.
\end{theorem}

\begin{theorem}[\textbf{G3}: Symplectic polarization]\ClaimStatusProvedHere
\label{thm:thqg-g3-polarization}
Let $(\cA, X, \langle\cdot,\cdot\rangle, k)$ be a gravitational input with
nondegenerate pairing at the chain level.  The complementarity splitting
\eqref{eq:thqg-lagrangian-splitting} is a shifted-symplectic Lagrangian
decomposition: both summands are Lagrangian with respect to the
$(-1)$-shifted symplectic form on $\mathbf{C}_g(\cA)$.  The Verdier
involution acts as gravitational CPT conjugation.
\end{theorem}

\begin{theorem}[\textbf{G4}: Gravitational $S$-duality]\ClaimStatusProvedHere
\label{thm:thqg-g4-s-duality}
The Koszul involution $\cA \mapsto \cA^!$ acts as an algebraic
analogue of gravitational $S$-duality
at the level of genus-$g$ partition functions
\textup{(}the identification with physical strong/weak $S$-duality
requires a gravitational dual\textup{)}.  The modular characteristic
satisfies a duality constraint: $\kappa(\cA^!) = -\kappa(\cA)$ for affine
Kac--Moody and free-field algebras, recovering the Feigin--Frenkel involution
$k \mapsto -k - 2h^\vee$; for $\mathcal{W}$-algebras the complementarity sum
$\kappa + \kappa'$ is a nonzero constant (Theorem~D).  The MC elements are
related by Verdier conjugation:
$\Theta_{\cA^!} = \sigma(\Theta_\cA)$ in $\gAmod$.
\end{theorem}

\begin{theorem}[\textbf{G5}: Gravitational Yangian]\ClaimStatusProvedHere
\label{thm:thqg-g5-yangian}
The collision residue $r(z) = \Res^{\mathrm{coll}}_{0,2}(\Theta_\cA)$
satisfies the classical Yang--Baxter equation.  On the affine lineage,
this residue recovers the proved dg-shifted Yangian realization of
Vol~II, Theorem~thm:Koszul\_dual\_Yangian; more generally it is the
classical input for the conjectural perturbative gravitational-Yangian
package.  Independently, the line-operator category is the open-colour
module category $\mathcal{C} \simeq \cA^!\text{-}\mathsf{mod}$
\textup{(}Theorem~\textup{\ref{thm:lines_as_modules})}.  The CYBE follows
from the MC equation projected to genus~$0$, arity~$3$, via the Arnold
relations on $\overline{C}_3(X)$.
\end{theorem}

\begin{theorem}[\textbf{G6}: Soft graviton theorems]\ClaimStatusProvedHere
\label{thm:thqg-g6-soft-graviton}
The Ward identities of the modular shadow connection
$\nabla^{\mathrm{hol}}_{g,n} = d - \operatorname{Sh}_{g,n}(\Theta_\cA)$
at genus~$0$, arity~$2$ produce identities that algebraically reproduce
the structure of the Weinberg--BMS leading and subleading
soft graviton theorems \textup{(}the identification with physical soft
graviton emission requires a gravitational dual\textup{)}.  The all-arity master equation
$\nabla_{\mathrm{H}}(\operatorname{Sh}_r) + o^{(r)} = 0$
gives nonlinear soft theorems at each arity $r \ge 3$.  The shadow depth
$r_{\max}(\cA)$ controls the number of independent soft theorems:
\begin{center}
\renewcommand{\arraystretch}{1.1}
\small
\begin{tabular}{cl}
$r_{\max}$ & Soft theorems \\
\hline
$2$ & Leading only \\
$3$ & Leading + subleading \\
$4$ & Leading + subleading + sub-subleading \\
$\infty$ & Full infinite tower
\end{tabular}
\end{center}
\end{theorem}

\begin{theorem}[\textbf{G7}: Modular bootstrap]\ClaimStatusProvedHere
\label{thm:thqg-g7-bootstrap}
The MC equation decomposes by genus:
\begin{equation}\label{eq:thqg-genus-bootstrap}
D_\cA\bigl(\Theta_\cA^{(g)}\bigr) +
\sum_{\substack{g_1 + g_2 = g \\ g_1, g_2 \ge 0}}
\tfrac{1}{2}\bigl[\Theta_\cA^{(g_1)}, \Theta_\cA^{(g_2)}\bigr] = 0.
\end{equation}
The equation determines $\Theta_\cA^{(g)}$ inductively from
$\Theta_\cA^{(0)}, \ldots, \Theta_\cA^{(g-1)}$ and the obstruction class
$o^{(g)} \in H^2(\Defcyc^{\mathrm{mod}})$.  The genus spectral sequence
(Construction~\ref{const:vol1-genus-spectral-sequence}) has $E_1$~page
with $p = 0$ (tree), $p = 1$ (one-loop), $p \ge 2$ (higher genus).
Convergence of the $2$-dimensional tower (genus $\times$ arity) is proved
by MC5.
\end{theorem}

\begin{theorem}[\textbf{G8}: Holographic reconstruction]\ClaimStatusProvedHere
\label{thm:thqg-g8-reconstruction}
The holographic modular Koszul datum $\mathcal{H}(\cA)$ is algorithmically
reconstructed from the boundary OPE data of $\cA$:
\begin{enumerate}[label=\textup{(\roman*)}]
\item $\cA \;\xmapsto{\text{bar-intrinsic}}\; \Theta_\cA$
  (Theorem~\ref{thm:mc2-bar-intrinsic});
\item $\cA \;\xmapsto{\text{bar + Verdier}}\; \cA^!$;
\item $\Theta_\cA \;\xmapsto{\Res^{\mathrm{coll}}_{0,2}}\; r(z)$;
\item $r(z) \;\xmapsto{\text{RTT}}\; \Ydg(\cA^!)$, hence
  $\mathcal{C} \simeq \cA^!\text{-}\mathsf{mod}$;
\item $\Theta_\cA \;\xmapsto{\operatorname{Sh}_{g,n}}\;
  \nabla^{\mathrm{hol}}_{g,n}$.
\end{enumerate}
Each step uses only the data produced by preceding steps.
\end{theorem}

\begin{theorem}[\textbf{G9}: Critical string dichotomy]\ClaimStatusProvedHere
\label{thm:thqg-g9-critical-string}
For the Virasoro family, the Koszul involution reads
$\mathrm{Vir}_c^! \simeq \mathrm{Vir}_{26-c}$.  At $c = 26$:
\begin{enumerate}[label=\textup{(\roman*)}]
\item The Koszul dual is
  $\mathrm{Vir}_{26}^! \simeq \mathrm{Vir}_0$; self-conjugacy
  $\cA \simeq \cA^!$ occurs at $c = 13$, not at $c = 26$.
\item The dual partner $\mathrm{Vir}_0$ lies on the uncurved scalar
  locus: $\kappa(\mathrm{Vir}_0) = 0$, hence its scalar class vanishes
  and its scalar genus package satisfies
  $F_g(\mathrm{Vir}_0) = 0$ for all $g \geq 1$.
\item The depth-zero resonance shadow
  (Remark~\ref{rem:virasoro-resonance-model}) therefore saturates only
  at the scalar level; the resonance-filtered bar-cobar is not proved
  to collapse at higher arity from this scalar cancellation alone.
\item The complementarity splitting is maximally asymmetric rather than
  self-conjugate: the nonzero scalar budget is carried by
  $\mathrm{Vir}_{26}$, while the dual $\mathrm{Vir}_0$ side is
  scalar-uncurved.
\end{enumerate}
\end{theorem}

\begin{theorem}[\textbf{G10}: Partition function convergence]\ClaimStatusProvedHere
\label{thm:thqg-g10-fredholm}
For the Heisenberg algebra, the genus-$g$ partition function admits a
Fredholm determinant representation
\begin{equation}\label{eq:thqg-fredholm-det}
Z_g(\cH_k) \;=\; \det\bigl(1 - K_g(\cH_k)\bigr),
\end{equation}
where $K_g(\cH_k)$ is a trace-class sewing kernel built from the Bergman
kernel on the genus-$g$ surface
(Theorem~\textup{\ref{thm:heisenberg-one-particle-sewing}}).
For the entire standard landscape of universal algebras, the
HS-sewing criterion
(Theorem~\textup{\ref{thm:general-hs-sewing}}) gives absolute convergence
of the genus-$g$ partition function; the Fredholm determinant form
\textup{(}\ref{eq:thqg-fredholm-det}\textup{)} is established only for
the Heisenberg algebra.
\end{theorem}

\begin{conjecture}[\textbf{G11}: Algebraic entanglement entropy; \ClaimStatusConjectured]
\label{conj:thqg-g11-entanglement}
For a gravitational input, the genus-$g$ Lagrangian capacity
$C_g(\cA)$
(Definition~\ref{def:thqg-lagrangian-capacity},
Conjecture~\ref{conj:thqg-spatial-entanglement})
is well-defined for unitary $\cA$, satisfies the Page bound
$S_{\mathrm{EE}}^{(g)} \le \min(\log\dim Q_g(\cA), \log\dim Q_g(\cA^!))$
(Theorem~\ref{thm:thqg-page-constraint}),
and admits a convergent shadow obstruction tower expansion.
At genus $1$: $S_{\mathrm{EE}}^{(1)} = 0$.
\end{conjecture}

\begin{conjecture}[\textbf{G12$'$}: Algebraic quantum extremal surface; \ClaimStatusConjectured]
\label{conj:thqg-g12-qes}
The spatial entanglement entropy of a boundary subregion $\cM$ is
determined by the open/closed MC element $\Theta^{\mathrm{oc}}_\cA$
(Conjecture~\ref{conj:thqg-spatial-entanglement}),
with the leading term $\kappa(\cA) \cdot \lambda_g^{\mathrm{FP}}$
(algebraic Ryu--Takayanagi) and convergent quantum corrections from
the shadow obstruction tower (Theorem~\ref{thm:thqg-qes-convergence}).
The MC equation serves as the extremization condition.
\end{conjecture}

\begin{conjecture}[\textbf{G13}: Modular flow from shadow connection; \ClaimStatusConjectured]
\label{conj:thqg-g13-modular-flow}
The shadow connection $\nabla^{\mathrm{sh}}$ generates the modular
automorphism group (Conjecture~\ref{conj:thqg-modular-flow}).
The Verdier involution $\sigma$ has the structural properties of the Tomita conjugation $J$
(Proposition~\ref{conj:thqg-tomita-from-verdier}),
the shadow metric $Q_L$ determines the modular Hamiltonian,
and the KMS condition at $\beta = 2\pi$ follows from the
$(-1)$-monodromy of $\nabla^{\mathrm{sh}}$.
\end{conjecture}

\begin{theorem}[\textbf{G14}: Holographic code structure]\ClaimStatusProvedHere
\label{thm:thqg-g14-error-correction}
The bar-cobar adjunction provides a code structure
(Proposition~\ref{prop:thqg-barcobar-error-correction}):
encoding via $B$, decoding via $\Omega$, exact inversion from
Koszulness (Theorem~B).  The open/closed realization provides
subregion structure (Theorem~\ref{thm:thqg-entanglement-wedge}),
with the annulus trace as algebraic partial trace
(Theorem~\ref{thm:thqg-annulus-trace}).
The identification as a literal quantum error-correcting code
requires the analytic completion.
\end{theorem}

\begin{theorem}[\textbf{G15}: Algebraic Page constraint]\ClaimStatusProvedHere
\label{thm:thqg-g15-page}
The complementarity theorem (Theorem~C) imposes dimensional constraints
at all genera (Theorem~\ref{thm:thqg-page-constraint}):
$C_g(\cA) + C_g(\cA^!) = \dim\cH_g(\cA)$, where $C_g(\cA) = \dim Q_g(\cA)$
is the Lagrangian capacity.  Any entanglement entropy in a bipartite system
with factors of these dimensions satisfies
$S \le \log\min(C_g(\cA), C_g(\cA^!))$.
The self-dual point $\cA \cong \cA^!$ is the algebraic Page transition.
\end{theorem}

\begin{conjecture}[\textbf{G16}: Replica structure; \ClaimStatusConjectured]
\label{conj:thqg-g16-replica}
The R\'enyi entropy at genus $g$ is computed by the shadow obstruction tower
on the replica manifold $\Sigma_{g,n}$ of genus $ng$
(Proposition~\ref{conj:thqg-replica}).
The genus expansion of the shadow obstruction tower controls each term of the
R\'enyi series.  The analytic continuation $n \to 1$ recovers the
von~Neumann entropy.
\end{conjecture}

\section{The dependency theorem}
\label{sec:thqg-dependency-theorem}

\begin{theorem}[Dependency theorem]\ClaimStatusProvedHere
\label{thm:thqg-dependency}
Let $(\cA, X, \langle\cdot,\cdot\rangle, k)$ be a gravitational input.
All ten proved theorems \textup{(G1)--(G10), (G14), (G15)} follow from:
\begin{enumerate}[label=\textup{(\roman*)}]
\item the bar-intrinsic MC element $\Theta_\cA := D_\cA - d_0 \in \MC(\gAmod)$
  (Theorem~\ref{thm:mc2-bar-intrinsic}); and
\item the identity $D_\cA^2 = 0$, at the convolution level
  (from $\partial^2 = 0$ on $\overline{\mathcal{M}}_{g,n}$)
  and at the ambient level with planted-forest corrections
  (Theorem~\ref{thm:ambient-d-squared-zero}).
\end{enumerate}
Each result is a projection of $\Theta_\cA$ to specific genus, arity,
weight filtration level, or representation-theoretic component:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\small
\begin{tabular}{cll}
\textbf{Projection} & \textbf{Object} & \textbf{Result} \\
\hline
$\Theta_\cA^{\le 2}$ & $\kappa(\cA)$ (modular characteristic)
  & G1, G7, G10 \\
$\Theta_\cA^{\le 3}$ & $\mathfrak{C}(\cA)$ (cubic shadow)
  & G2 (L-type), G6 \\
$\Theta_\cA^{\le 4}$ & $\mathfrak{Q}(\cA)$ (quartic resonance class)
  & G2 (C-type), G6 \\
$\Theta_\cA|_{g=0,n=2}$ & $r(z)$ (binary shadow)
  & G5 \\
$\Theta_\cA|_{g=0,n=3}$ & CYBE
  & G5 \\
$\mathbf{C}_g(\cA), \sigma$ & Lagrangian splitting
  & G3, G4 \\
$\Theta_\cA|_g$ & $\Theta_\cA^{(g)}$ (genus component)
  & G7, G10 \\
$\cA \mapsto \cA^!$ & Koszul involution
  & G4, G9 \\
$\varprojlim_r \Theta_\cA^{\le r}$ & Full MC element
  & G8 \\
$\cH_g, \sigma, \Theta_\cA^{(g)}$ & Lagrangian capacity $C_g$
  & G11, G15 \\

  & G12$'$, G14 \\

  & G13 \\
$B \dashv \Omega, \cA^!$ & bar-cobar code
  & G14 \\
$\Sigma_{g,n}$ & replica manifold
  & G16
\end{tabular}
\end{center}
\end{theorem}

The conjectures \textup{(G11, G12$'$, G13, G16)} use the same projections of~$\Theta_\cA$ but require analytic or physical hypotheses beyond the proved algebraic core.

\section{Notation and grading}
\label{sec:thqg-notation}

\begin{enumerate}[label=\textup{(\alph*)}]
\item Grading is \textbf{cohomological} ($|d| = +1$) throughout.
  The bar construction uses desuspension.

\item $\gAmod = \Convstr(\mathcal{C}^!_{\mathrm{ch}},
  \mathcal{P}^{\mathrm{ch}})$ is the modular convolution dg Lie
  algebra (Definition~\ref{def:modular-convolution-dg-lie}), the
  strict model of $\Convinf(\mathcal{C}^!_{\mathrm{ch}},
  \mathcal{P}^{\mathrm{ch}})$.

\item $B(\cA)$ is the bar coalgebra; $\cA^! = (H^*(B(\cA)))^\vee$
  is the Koszul dual; $\Omega(B(\cA)) \simeq \cA$ is bar-cobar inversion.

\item \emph{HT} abbreviates ``holomorphic-topological.''
  An HT theory is a 3d $\mathcal{N} = 2$ theory in the holomorphic twist.

\item $\mathrm{Vir}_c^! \simeq \mathrm{Vir}_{26-c}$.  Self-duality at
  $c = 13$, \textbf{not} $c = 26$.  The critical string value $c = 26$ is
  the point where $\cA^! \simeq \mathrm{Vir}_0$ on the uncurved scalar
  locus, not a trivial Virasoro algebra.
\end{enumerate}
