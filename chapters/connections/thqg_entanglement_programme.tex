\begin{remark}[Two sources of entanglement; direct sum versus tensor product]
\label{rem:thqg-two-entanglements}
\index{entanglement!direct sum versus tensor product}
A critical structural distinction governs the entire programme.
The Lagrangian decomposition
$\cH_g(\cA) = Q_g(\cA) \oplus Q_g(\cA^!)$
is a \emph{direct sum}, not a tensor product.
A direct sum carries classical correlations but not quantum
entanglement: every vector $v = \alpha + \beta \in V \oplus W$
has a unique decomposition into its $V$- and $W$-components,
so there is no ``entanglement'' to speak of.

Genuine quantum entanglement in the algebraic framework
arises from a different source: the \emph{sewing construction}
and the \emph{open/closed tensor product}.
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Sewing entanglement.}
  Cutting a genus-$g$ surface along a separating curve
  produces two bordered surfaces.  The factorization homology
  on each piece gives a vector in the boundary Hilbert space
  $H_\cA = FH(\cA; S^1)$.  The partition function on the
  closed surface is the inner product of these two vectors.
  The \emph{sewing} (Theorem~\ref{thm:general-hs-sewing},
  MC5) provides a genuine tensor product structure
  $H_\cA \otimes H_\cA$, and the state created by the
  genus-$g$ surface can be genuinely entangled.

\item \emph{Open/closed entanglement.}
  The Swiss-cheese theorem
  (Theorem~\ref{thm:thqg-swiss-cheese}) decomposes the
  holographic system into open (boundary) and closed (bulk)
  sectors.  A boundary module $\cM$ at a puncture gives
  a tensor product decomposition via
  $\operatorname{End}(\cM) \otimes Z(\cA)$.
  The open/closed MC element $\Theta^{\mathrm{oc}}_\cA$
  is a state in this tensor product, and its entanglement
  entropy is well-defined.
\end{enumerate}
The Lagrangian decomposition $Q_g \oplus Q_g^!$ provides
\emph{dimensional constraints} (the Page bound,
Theorem~\ref{thm:thqg-page-constraint}) that constrain
any entanglement entropy from sources~(i)--(ii),
but it does not itself carry entanglement.
\end{remark}

\subsection{Dimensional constraints from the Lagrangian structure}
\label{subsec:thqg-entanglement-dimensional}
\index{entanglement!dimensional constraints|textbf}

\begin{definition}[Lagrangian capacity]
\label{def:thqg-lagrangian-capacity}
\index{Lagrangian capacity|textbf}
For a gravitational input $(\cA, X, \langle \cdot, \cdot \rangle, k)$
and genus $g \ge 1$, the \emph{Lagrangian capacity} is
\begin{equation}\label{eq:thqg-lagrangian-capacity}
C_g(\cA)
\;:=\;
\dim Q_g(\cA).
\end{equation}
This is the dimension of the $\cA$-eigenspace in the
Lagrangian decomposition
$\cH_g(\cA) = Q_g(\cA) \oplus Q_g(\cA^!)$
(Theorem~\ref{thm:thqg-III-eigenspace-decomposition}).
\end{definition}

\begin{theorem}[Lagrangian constraint theorem;
\ClaimStatusProvedHere]
\label{thm:thqg-lagrangian-constraint}
\index{Lagrangian constraint theorem|textbf}
For a gravitational input:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \textup{(Complementarity sum.)}
  $C_g(\cA) + C_g(\cA^!) = \dim \cH_g(\cA)
  = \dim H^*(\overline{\cM}_g, \cZ(\cA))$.

\item \textup{(Genus-$1$ structure.)}
  $C_1(\cA) = C_1(\cA^!) = 1$ and $\dim \cH_1(\cA) = 2$.

\item \textup{(Self-dual halving.)}
  For $\cA \cong \cA^!$: $C_g(\cA) = C_g(\cA^!)
  = \tfrac{1}{2}\dim \cH_g(\cA)$ for all $g \ge 1$.

\item \textup{(Scalar contribution.)}
  The scalar shadow ($r = 2$) contributes exactly one
  dimension to $C_g$: the class
  $\kappa(\cA) \cdot \lambda_g^{\mathrm{FP}} \in Q_g(\cA)$
  spans a one-dimensional subspace.

\item \textup{(Virasoro scalar complementarity.)}
  $F_g^{\mathrm{sc}}(\mathrm{Vir}_c) + F_g^{\mathrm{sc}}(\mathrm{Vir}_{26-c})
  = 13 \cdot \lambda_g^{\mathrm{FP}}$.
\end{enumerate}
\end{theorem}

\begin{proof}
Part (i): Theorem~\ref{thm:quantum-complementarity-main}.
Part (ii): Theorem~\ref{thm:thqg-III-genus-1-holographic}.
Part (iii): Corollary~\ref{cor:thqg-III-dimension-parity}.
Part (iv): the scalar MC element
$\Theta^{\le 2,(g)} = \kappa \cdot \lambda_g^{\mathrm{FP}}$
is a nonzero multiple of a fixed cohomology class, spanning
a one-dimensional subspace of $Q_g(\cA)$.
Part (v): linearity of $F_g^{\mathrm{sc}}$ in $\kappa$,
combined with $\kappa(\mathrm{Vir}_c) + \kappa(\mathrm{Vir}_{26-c})
= c/2 + (26-c)/2 = 13$.
\end{proof}

\begin{remark}[Why the Lagrangian capacity bounds entanglement]
\label{rem:thqg-capacity-bounds-entanglement}
\index{entanglement!bounded by Lagrangian capacity}
Any von~Neumann entropy $S$ of a state in a Hilbert space
$H_A \otimes H_B$ satisfies $S \le \log\min(\dim H_A, \dim H_B)$.
If the spatial entanglement of a boundary subregion at genus $g$
is controlled by the ambient complex $\cH_g$, the Lagrangian
decomposition constrains:
\[
S_{\mathrm{EE}}^{(g)}(\cA)
\;\le\;
\log\min(C_g(\cA),\, C_g(\cA^!)).
\]
The complementarity sum $C_g + C_g' = \dim\cH_g$ then implies
the \emph{Page-type bound}: the entanglement cannot exceed
$\tfrac{1}{2}\log\dim\cH_g$ for self-dual algebras.
This bound is a proved consequence of Theorems~C and~D;
its saturation by a physical entanglement entropy is
conjectural (\S\ref{subsec:thqg-entanglement-qes}).
\end{remark}

\subsection{Spatial entanglement from the open/closed structure}
\label{subsec:thqg-entanglement-spatial}
\index{entanglement!spatial|textbf}
\index{Ryu--Takayanagi formula!algebraic|textbf}

The genuine tensor product structure for spatial entanglement
comes from the open/closed realization
(\S\ref{sec:thqg-open-closed-realization}).

\begin{conjecture}[Spatial entanglement from modules]
\label{conj:thqg-spatial-entanglement}
\index{spatial entanglement!from modules}
Let $\cM$ be an $\cA$-module category at a puncture $p$ of the
tangential log curve $(X, D, \tau)$.  The open/closed MC element
$\Theta^{\mathrm{oc}}_\cA$
(Construction~\ref{constr:thqg-oc-mc-element})
lives in the open/closed convolution algebra
$\mathfrak{g}^{\mathrm{oc}}_{\cA, \cM}$,
whose mixed-sector components provide a tensor product coupling
between the open sector $\operatorname{End}(\cM)$ and the
closed sector $C^\bullet_{\mathrm{ch}}(\cA, \cA)$.
The \emph{spatial entanglement entropy of the subregion $\cM$} is:
\begin{equation}\label{eq:thqg-spatial-entanglement}
S(\cM) \;:=\;
-\Tr\bigl(\rho_{\cM} \log \rho_{\cM}\bigr),
\end{equation}
where $\rho_{\cM}$ is the reduced density matrix on the open sector
obtained by tracing out the closed sector.
\end{conjecture}

\begin{remark}[The annulus trace as partial trace]
\label{rem:thqg-annulus-as-partial-trace}
\index{annulus trace!as partial trace}
The annulus trace theorem (Theorem~\ref{thm:thqg-annulus-trace})
gives $\int_{S^1_p} \cM \simeq HH_*(\cM)$.
This is the algebraic incarnation of the partial trace: it
integrates out the boundary degrees of freedom at the puncture,
leaving the Hochschild invariants.  In the entanglement computation,
$HH_0(\cM)$ controls the leading contribution to $S(\cM)$.
\end{remark}

\subsection{The algebraic quantum extremal surface}
\label{subsec:thqg-entanglement-qes}
\index{quantum extremal surface!algebraic|textbf}
\index{QES|see{quantum extremal surface}}

The quantum extremal surface (QES) formula
\cite{Engelhardt-Wall14}
\begin{equation}\label{eq:thqg-qes-physical}
S(R) \;=\; \min_{\gamma}\,\operatorname{ext}\Bigl[
\frac{\mathrm{Area}(\gamma)}{4G_N}
\;+\; S_{\mathrm{bulk}}(\Sigma_\gamma)
\Bigr]
\end{equation}
generalizes the Ryu--Takayanagi formula \cite{Ryu-Takayanagi06}
by including bulk quantum corrections.  The shadow obstruction tower provides
the algebraic realization.

\begin{definition}[QES approximant]
\label{def:thqg-qes-approximant}
\index{QES approximant|textbf}
For a gravitational input and integer $r \ge 2$, the
\emph{$r$-th order QES approximant} at genus $g$ is:
\begin{equation}\label{eq:thqg-qes-approximant}
S_{\mathrm{QES}}^{(g, \le r)}(\cA)
\;:=\;
\underbrace{
\kappa(\cA) \cdot \lambda_g^{\mathrm{FP}}
}_{\text{area term (arity $2$)}}
\;+\;
\underbrace{
\sum_{j=3}^{r} \delta F^{(g)}_j(\cA)
}_{\text{bulk corrections (arities $3, \ldots, r$)}}
\end{equation}
where $\delta F^{(g)}_j(\cA)$ is the genus-$g$ free-energy
increment contributed by the arity-$j$ shadow jet
$\Sh_j(\cA)$.  At $r = 2$, this recovers the scalar
free energy $F_g^{\mathrm{sc}}(\cA) = \kappa \cdot \lambda_g^{\mathrm{FP}}$.
\end{definition}

\begin{theorem}[QES convergence; \ClaimStatusProvedHere]
\label{thm:thqg-qes-convergence}
\index{QES!convergence}
The QES approximants converge:
\[
F_g(\cA)
\;:=\;
\lim_{r \to \infty}
S_{\mathrm{QES}}^{(g, \le r)}(\cA)
\]
exists for every genus $g \ge 1$ and every gravitational input.
For shadow depth $r_{\max} < \infty$, the series terminates.
For $r_{\max} = \infty$, the convergence rate is controlled
by the shadow growth rate:
$|\delta F^{(g)}_j(\cA)| = O(\rho(\cA)^j \cdot j^{-5/2})$.
\end{theorem}

\begin{proof}
Convergence of the shadow obstruction tower
(Theorem~\ref{thm:recursive-existence}) gives convergence of
$\Theta_\cA^{\le r} \to \Theta_\cA$.  The genus-$g$ free energy
is a continuous functional of $\Theta_\cA^{\le r}$
(it is the genus-$g$ component of the MC element projected
to the Lagrangian summand), so convergence of the approximants
follows.  The decay estimate is the shadow growth theorem
(Theorem~\ref{thm:shadow-radius}).
\end{proof}

\begin{remark}[The MC equation as extremization]
\label{rem:thqg-mc-as-extremization}
\index{Maurer--Cartan equation!as extremization}
The physical QES formula \eqref{eq:thqg-qes-physical} requires
extremization over surfaces.
In the algebraic framework, the extremization is the
Maurer--Cartan equation.
The bar-intrinsic MC element $\Theta_\cA = D_\cA - d_0$
\emph{automatically} satisfies this equation
(Theorem~\ref{thm:mc2-bar-intrinsic}):
the construction produces the extremal configuration
directly, without a variational principle.
\end{remark}

\begin{conjecture}[QES identification]
\label{conj:thqg-qes-identification}
\index{QES!identification with physical formula}
Under the holographic dictionary of
Definition~\textup{\ref{def:thqg-holographic-datum}},
the QES approximant $S_{\mathrm{QES}}^{(g,\le r)}(\cA)$
at arity $r$ coincides with the $r$-th order expansion of the
physical QES formula \eqref{eq:thqg-qes-physical}: the area term
$\kappa \cdot \lambda_g^{\mathrm{FP}}$ is $\mathrm{Area}/(4G_N)$,
and the higher-arity corrections $\delta F_j^{(g)}$ are the bulk
entropy contributions.
\end{conjecture}

\subsection{Replica symmetry and R\'enyi entropy}
\label{subsec:thqg-entanglement-replica}
\index{replica symmetry|textbf}
\index{R\'enyi entropy|textbf}

\begin{construction}[Replica manifold]
\label{constr:thqg-replica-manifold}
\index{replica manifold|textbf}
For a genus-$g$ surface $\Sigma_g$ and integer $n \ge 2$,
the \emph{$n$-fold replica manifold} is the cyclic $n$-fold
branched cover
$\Sigma_{g,n} \to \Sigma_g$
branched over two points $\{p, q\} \subset \Sigma_g$,
with each branch point fully ramified
\textup{(}a single preimage of ramification index $n$\textup{)}.
By the Riemann--Hurwitz formula:
\begin{equation}\label{eq:thqg-replica-genus}
2g(\Sigma_{g,n}) - 2
\;=\; n(2g - 2) + 2(n-1),
\qquad\text{so}\qquad
g(\Sigma_{g,n}) \;=\; ng.
\end{equation}
\begin{itemize}
\item $g = 0$: $g(\Sigma_{0,n}) = 0$
  \textup{(}the map $z \mapsto z^n$ on $\mathbb{P}^1$\textup{)}.
  The replica manifold is again a sphere; the R\'enyi entropy
  is computed by twist-operator two-point functions
  \cite{Calabrese-Cardy04}, not by higher-genus partition functions.

\item $g = 1$: $g(\Sigma_{1,n}) = n$
  \textup{(}the $n$-fold cover of the torus is genus $n$\textup{)}.
  The R\'enyi entropy at genus $1$ involves genuinely higher-genus
  partition functions.

\item $g \ge 2$: $g(\Sigma_{g,n}) = ng$, growing linearly in $n$.
\end{itemize}
\end{construction}

\begin{proposition}[Replica partition function; \ClaimStatusHeuristic]
\label{prop:thqg-replica}
\index{replica!partition function}
The $n$-fold replica partition function of a gravitational input
$(\cA, X, \langle \cdot, \cdot \rangle, k)$ at genus $g \ge 1$
is:
\begin{equation}\label{eq:thqg-replica-pf}
Z_n^{(g)}(\cA)
\;:=\;
Z(\cA;\, \Sigma_{g,n}),
\end{equation}
the partition function of $\cA$ evaluated on the replica manifold
of genus $ng$.  The R\'enyi entropy is:
\begin{equation}\label{eq:thqg-renyi}
S_n^{(g)}(\cA)
\;:=\;
\frac{1}{1-n}
\log\frac{Z_n^{(g)}(\cA)}{Z^{(g)}(\cA)^n}.
\end{equation}
The von~Neumann entropy is recovered in the $n \to 1$ limit:
$S^{(g)}(\cA) = \lim_{n \to 1} S_n^{(g)}(\cA)$.
\end{proposition}

\begin{proof}[Argument \textup{(}\ClaimStatusHeuristic{}\textup{)}]
Standard replica argument \cite{Calabrese-Cardy04, Lewkowycz-Maldacena13}
adapted to the algebraic framework.  The
heuristic status reflects two issues: (1)~the analytic continuation
$n \to 1$ requires the Carlson theorem, which demands growth bounds
on $Z_n^{(g)}$ in $n$ that have not been established algebraically;
(2)~the evaluation on the replica manifold of genus $ng$ requires
the sewing construction (MC5) at genus $ng$, which is proved but
computationally nontrivial for large~$n$.
\end{proof}

\begin{remark}[Shadow obstruction tower on the replica manifold]
\label{rem:thqg-shadow-on-replica}
\index{shadow tower!on replica manifold}
At the scalar level, the replica partition function is:
\[
\log Z_n^{(g),\mathrm{sc}}(\cA)
\;=\;
\kappa(\cA) \cdot
\sum_{h=0}^{\infty} \hbar^{2h-2}\, \lambda_h^{\mathrm{FP}},
\]
evaluated at genus $ng$.
The shadow obstruction tower controls the higher-arity corrections:
each arity-$r$ shadow jet contributes to $Z_n^{(g)}$
through its projection to genus~$ng$.  The convergence of
the shadow obstruction tower ensures that these corrections are summable
for all $n$ and~$g$.
\end{remark}

\subsection{Modular flow from the shadow connection}
\label{subsec:thqg-entanglement-modular-flow}
\index{modular flow|textbf}
\index{Tomita--Takesaki theory!algebraic incarnation|textbf}

\begin{proposition}[Verdier involution as Tomita conjugation;
\ClaimStatusHeuristic]
\label{prop:thqg-tomita-from-verdier}
\index{Verdier involution!as Tomita $J$}
The Verdier involution $\sigma$ on $\cH_g(\cA)$
shares the following algebraic properties with the Tomita
conjugation operator $J$:
\begin{enumerate}[label=\textup{(\roman*)}]
\item $\sigma^2 = \id$ \textup{(}involutive\textup{)};
\item $\sigma$ exchanges the two Lagrangian summands:
  $\sigma(Q_g(\cA)) = Q_g(\cA^!)$ \textup{(}swaps ``left''
  and ``right''\textup{)};
\item the Verdier pairing satisfies
  $\langle Q_g(\cA), Q_g(\cA) \rangle_g = 0$ and
  $\langle Q_g(\cA^!), Q_g(\cA^!) \rangle_g = 0$
  \textup{(}isotropy of each summand, the algebraic analogue
  of ``$J$ maps the algebra to its commutant''\textup{)}.
\end{enumerate}
For \emph{unitary} $\cA$ with positive-definite Shapovalov form,
$\sigma$ additionally acts as an anti-linear involution
with respect to the unitarity inner product, completing the
analogy with $J$.
\end{proposition}

\begin{proof}
Parts (i)--(ii) are the proved properties of the Verdier
involution
(Theorem~\ref{thm:thqg-III-eigenspace-decomposition}).
Part~(iii) is the isotropy of each Lagrangian summand under
the Verdier pairing
(\S\ref{subsec:thqg-III-shifted-symplectic}).
The anti-linearity statement for unitary algebras requires
the Shapovalov form, which provides a Hermitian structure
compatible with the $\mathbb{C}$-anti-linear action of $*$
on the center.  The full verification that $\sigma$ satisfies
all axioms of the Tomita $J$ operator (including the correct
domain and closability conditions) is heuristic: it requires
the passage from the algebraic to the analytic setting via
the sewing envelope $\cA^{\mathrm{sew}}$
(Theorem~\ref{thm:general-hs-sewing}).
\end{proof}

\begin{conjecture}[Shadow connection generates modular flow]
\label{conj:thqg-modular-flow}
\index{modular flow!from shadow connection}
\index{shadow connection!as modular flow generator}
The shadow connection
$\nabla^{\mathrm{sh}} = d - Q_L'/(2Q_L)\, dt$
(Theorem~\ref{thm:shadow-connection})
generates an algebraic analogue of the modular automorphism group:
\begin{enumerate}[label=\textup{(\roman*)}]
\item The logarithmic singularities of $\nabla^{\mathrm{sh}}$
  at the zeros of $Q_L$ correspond to the singular points of
  the modular flow.
\item The monodromy $-1$ of $\nabla^{\mathrm{sh}}$
  (the Koszul sign) corresponds to the
  anti-periodicity of the modular flow at $\beta = 2\pi$.
\item The shadow depth $r_{\max}(\cA)$ controls the complexity
  of the modular Hamiltonian: class~G gives a quadratic
  Hamiltonian, class~L cubic, and so on.
\end{enumerate}
\end{conjecture}

\begin{remark}[KMS condition from complementarity]
\label{rem:thqg-kms-from-complementarity}
\index{KMS condition!from complementarity}
The KMS condition
$\omega(a\, \sigma_{i\beta}(b)) = \omega(b\, a)$
expresses a symmetry between two halves of a thermal system.
The complementarity exchange principle
(Corollary~\ref{cor:thqg-III-complementarity-exchange})
expresses an analogous symmetry:
$\Phi(\cA) + \Phi(\cA^!) = \Phi_{\mathrm{total}}$.
Both relate two Lagrangian halves of a decomposed system.
The shadow connection provides the bridge: its monodromy
$-1$ is the algebraic KMS periodicity, and
$\sigma$ interchanges the two halves.
The precise relationship requires the analytic completion
(sewing envelope) to make the von~Neumann algebra structure
available.
\end{remark}

\subsection{Error correction from Koszul duality}
\label{subsec:thqg-entanglement-error-correction}
\index{error correction!from Koszul duality|textbf}
\index{holographic error correction|textbf}

The bar-cobar adjunction provides a natural encoding/decoding
structure that parallels, at the structural level, the
quantum error-correcting codes of holographic entanglement
\cite{Almheiri-Dong-Harlow15, Pastawski-Yoshida-Harlow-Preskill15}.

\begin{proposition}[Bar-cobar code structure;
\ClaimStatusProvedHere]
\label{prop:thqg-barcobar-error-correction}
\index{bar-cobar!as code structure}
The bar-cobar adjunction $B \dashv \Omega$ endows each
Koszul chiral algebra $\cA$ with:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Encoding.}
  $B \colon \cA \to B(\cA)$:
  the ``logical'' algebra $\cA$ is mapped into the
  bar coalgebra $B(\cA)$.

\item \emph{Decoding.}
  $\Omega \colon B(\cA) \to \Omega(B(\cA))$:
  the cobar construction provides recovery.

\item \emph{Exact inversion on the Koszul locus.}
  Theorem~B gives $\Omega(B(\cA)) \xrightarrow{\sim} \cA$:
  the recovery map is a quasi-isomorphism, recovering $\cA$
  exactly from $B(\cA)$.

\item \emph{Dual algebra.}
  $\cA^! = (H^*(B(\cA)))^\vee$: the Koszul dual is
  the cohomology of the bar complex, dualized.
  The line-operator category
  $\mathcal{C} \simeq \cA^!\text{-}\mathsf{mod}$.
\end{enumerate}
\end{proposition}

\begin{proof}
Parts (i)--(ii) are the bar and cobar functors.
Part (iii) is Theorem~B.
Part (iv) is the definition of $\cA^!$ and
Theorem~\ref{thm:thqg-g5-yangian}.
\end{proof}

\begin{remark}[Structural analogy with holographic error correction]
\label{rem:thqg-hqec-analogy}
\index{holographic error correction!structural analogy}
The bar-cobar structure is a \emph{structural analogy} with
holographic quantum error correction (HQEC), not a literal
identification.  In HQEC \cite{Almheiri-Dong-Harlow15}:
the logical Hilbert space (bulk) embeds isometrically into the
physical Hilbert space (boundary), and subregion duality gives

\subsection{The algebraic Page constraint}
\label{subsec:thqg-entanglement-page}
\index{Page constraint!algebraic|textbf}
\index{Page curve|textbf}

\begin{theorem}[Algebraic Page constraint; \ClaimStatusProvedHere]
\label{thm:thqg-page-constraint}
For a gravitational input $(\cA, X, \langle \cdot, \cdot \rangle, k)$
and any genus $g \ge 1$:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \textup{(Dimensional Page bound.)}
  Any von~Neumann entropy $S$ of a state in a bipartite system
  whose factors have dimensions at most $C_g(\cA)$ and $C_g(\cA^!)$
  satisfies:
  \begin{equation}\label{eq:thqg-page-bound}
  S
  \;\le\;
  \log\min\bigl(C_g(\cA),\; C_g(\cA^!)\bigr).
  \end{equation}

\item \textup{(Complementarity sum.)}
  \begin{equation}\label{eq:thqg-page-complementarity}
  C_g(\cA) + C_g(\cA^!)
  \;=\;
  \dim \cH_g(\cA).
  \end{equation}

\item \textup{(Self-dual saturation.)}
  For $\cA \cong \cA^!$:
  $S \le \tfrac{1}{2}\log\dim \cH_g(\cA)$.

\item \textup{(Scalar complementarity.)}
  At the scalar level:
  \begin{equation}\label{eq:thqg-page-virasoro-scalar}
  F_g^{\mathrm{sc}}(\mathrm{Vir}_c)
  + F_g^{\mathrm{sc}}(\mathrm{Vir}_{26-c})
  \;=\;
  13 \cdot \lambda_g^{\mathrm{FP}}.
  \end{equation}
\end{enumerate}
\end{theorem}

\begin{proof}
Part (i): standard von Neumann bound.
Part (ii): Theorem~\ref{thm:quantum-complementarity-main}.
Part (iii): Corollary~\ref{cor:thqg-III-dimension-parity}.
Part (iv): $\kappa(\mathrm{Vir}_c) + \kappa(\mathrm{Vir}_{26-c}) = 13$.
\end{proof}

\begin{remark}[The Page transition at the self-dual point]
\label{rem:thqg-page-transition}
\index{Page transition|textbf}
For the Virasoro family:
\begin{itemize}
\item $c > 13$: $C_g(\mathrm{Vir}_c) > C_g(\mathrm{Vir}_{26-c})$
  at the scalar level.  The Page bound is controlled by the
  \emph{dual} (bulk) capacity: this is the Hawking phase.
\item $c = 13$: $C_g = C_g'$, the self-dual point.
  Boundary and bulk capacities are equal.
\item $c < 13$: $C_g(\mathrm{Vir}_c) < C_g(\mathrm{Vir}_{26-c})$.
  The Page bound is controlled by the boundary capacity:
  information-recovery phase.
\end{itemize}
The Koszul involution $c \mapsto 26 - c$ interchanges the two
phases.  The fixed point $c = 13$ is the algebraic Page transition
(Proposition~\ref{prop:thqg-V-c13-self-duality}).
\end{remark}

\begin{remark}[Constructive approximation of the Page constraint]
\label{rem:thqg-page-constructive}
\index{Page constraint!constructive approximation}
The shadow obstruction tower provides a constructive, convergent
approximation to the Page constraint at each arity:
\begin{enumerate}[label=\textup{(\arabic*)}]
\item Arity $2$ (scalar): the constraint is
  \eqref{eq:thqg-page-virasoro-scalar}.
\item Arity $3$ (cubic): $C(\cA) + C(\cA^!) = 0$
  (independent sum factorization,
  Proposition~\ref{prop:independent-sum-factorization}).
\item Arity $4$ (quartic): the discriminant complementarity
  $\Delta(\cA) + \Delta(\cA^!) = \text{constant}$
  (Theorem~\ref{thm:shadow-connection}).
\item All arities: $\Theta_\cA = \varprojlim \Theta_\cA^{\le r}$
  satisfies the full constraint at all genera simultaneously.
\end{enumerate}
No other approach to quantum gravity provides a convergent,
finite-order approximation scheme for entanglement constraints.
\end{remark}

\begin{conjecture}[\textbf{G11}: Spatial entanglement from modules]
