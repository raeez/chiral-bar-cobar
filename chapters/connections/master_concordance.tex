\chapter{Master concordance: the seven-face dictionary across three volumes}
\label{ch:master-concordance}

\providecommand{\Tr}{\operatorname{Tr}}

\begin{abstract}
The common datum in the seven-face dictionary is the genus-zero
binary collision residue
\[
 r_\cA(z)
 = \operatorname{Res}^{\mathrm{coll}}_{0,2}(\Theta_\cA),
 \qquad
 \Theta_\cA = D_\cA-d_0.
\]
It is neither the chiral algebra $\cA$, nor the bar coalgebra
$B(\cA)$, nor the Verdier--Koszul dual algebra $\cA^!$, nor the
derived chiral centre.  Bar--cobar inversion recovers $\cA$;
Verdier duality produces $\cA^!$; chiral Hochschild cochains produce
$Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)$.  The tables below record the
normalizations under which the same collision residue appears as
twisting morphism, line-operator twist, Poisson-vertex $r$-matrix,
commuting Hamiltonian seed, Yangian classical limit, Sklyanin tensor,
and Gaudin generator.
\end{abstract}

\section{The five objects and three lanes}
\label{sec:master-concordance-five-objects}

\begin{convention}[Five-object firewall]
\label{conv:master-five-object-firewall}
For an augmented chiral algebra $\cA$ on a curve, the dictionary uses
five different objects:
\[
\begin{array}{rcl}
\cA && \text{the chiral algebra;}\\[2pt]
B(\cA)=T^c(s^{-1}\bar\cA) && \text{the ordered bar coalgebra,
with }\bar\cA=\ker\epsilon;\\[2pt]
\cA^{\mathrm i}=H^\bullet B(\cA) && \text{the Koszul-dual
coalgebra;}\\[2pt]
\cA^!=((\cA^{\mathrm i})^\vee) && \text{the post-Verdier
dual algebra;}\\[2pt]
Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)
 = C^\bullet_{\mathrm{ch}}(\cA,\cA) && \text{the closed-sector
cochain object, i.e. the derived chiral centre.}
\end{array}
\]
The three arrows are different:
\[
\Omega B(\cA)\xrightarrow{\sim}\cA
\quad\text{is inversion,}\qquad
\mathbb{D}_{\Ran}B(\cA)\rightsquigarrow \cA^!
\quad\text{is Verdier--Koszul duality,}
\]
and
\[
\cA\longmapsto C^\bullet_{\mathrm{ch}}(\cA,\cA)
\quad\text{is the Hochschild-centre construction.}
\]
The inversion counit never enters the centre lane, and no Hochschild
complex is a substitute for the Verdier dual algebra.
\end{convention}

\begin{convention}[Primitive structural datum]
\label{conv:master-primitive-datum}
The Open quadrant fixes a primitive datum of seven structural
elements,
\[
\bigl(X,\;D,\;\tau;\;\;\cC^{\mathrm{op}},\;b,\;A_b,\;
\Theta_\cC,\;Z^{\mathrm{der}}_{\mathrm{ch}}(\cC),\;\Tr_\cC\bigr),
\]
in the order of construction along the Beilinson tower:
\begin{enumerate}[label=\textup{(\arabic*)}]
\item the tangential log curve $(X, D, \tau)$, with $X$ a smooth
projective curve, $D\subset X$ a divisor of marked points, and
$\tau$ a tangential structure at $D$;
\item the factorisation dg-category $\cC^{\mathrm{op}}$ on
$(X,D,\tau)$, level $0$ in the Beilinson tower;
\item the chosen vacuum $b\in\cC^{\mathrm{op}}$;
\item the chart algebra $A_b=\End_\cC(b)$, level $1$;
\item the Maurer--Cartan twisting datum $\Theta_\cC=D_{A_b}-d_0$,
producing the bar coalgebra $B(A_b)=T^c(s^{-1}\bar A_b)$ at
level $2$;
\item the derived chiral centre
$Z^{\mathrm{der}}_{\mathrm{ch}}(\cC) \simeq
C^\bullet_{\mathrm{ch}}(A_b,A_b)$, level $3$, the
\emph{universal closed sector} \textup{(}physical bulk only after the
open--closed comparison
$\mathrm{OCA}\colon\mathcal{O}^{\mathrm{phys}}_{\mathrm{bulk}}(\cA)
\xrightarrow{\sim} Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)$ is supplied;
see Convention~\ref{conv:master-oca}\textup{)};
\item the modular trace $\Tr_\cC$, with clutching on the open
category, producing the level-$5$ scalar shadow.
\end{enumerate}
The chart algebra $A_b$ is not the primitive open object;
$\cC^{\mathrm{op}}$ on $(X,D,\tau)$ is.  The bar coalgebra $B(A_b)$
is not the bulk; $Z^{\mathrm{der}}_{\mathrm{ch}}(\cC)$ is.  The
modular shadow is not a closed-algebra property; it is the image of
$\Tr_\cC$ on the open category.
\end{convention}

\begin{convention}[Open--closed comparison; OCA]
\label{conv:master-oca}
The derived chiral centre $Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)
= C^\bullet_{\mathrm{ch}}(\cA,\cA)$ is the \emph{universal closed
sector} of the chart algebra $\cA = A_b$: it is constructed from
$\cA$ alone, by chiral Hochschild cochains, internal to the boundary
theory. It is \emph{not automatically} the physical bulk of a realised
holomorphic--topological QFT.

A physical-bulk statement for a QFT $T$ whose boundary chart algebra
is $\cA$ requires a separately constructed \emph{open--closed algebra
comparison} \textup{(}OCA\textup{)}
\[
\mathrm{OCA}_T\colon
\mathcal{O}^{\mathrm{phys}}_{\mathrm{bulk}}(T)
\xrightarrow{\;\sim\;}
Z^{\mathrm{der}}_{\mathrm{ch}}(\cA),
\]
a quasi-isomorphism compatible with the
$\mathsf{SC}^{\mathrm{ch,top}}$-action of the closed colour on $\cA$
\textup{(}Vol~II \texttt{thm:sc-chtop-deligne}\textup{)}. Without
$\mathrm{OCA}_T$, $Z^{\mathrm{der}}_{\mathrm{ch}}(\cA)$ remains the
universal closed sector internal to the boundary theory; the equality
``derived chiral centre $=$ physical bulk'' is not a theorem but a
named comparison hypothesis. The pair
$(\mathcal{O}^{\mathrm{phys}}_{\mathrm{bulk}}(T), \cA)$ becomes the
physical $\mathsf{SC}^{\mathrm{ch,top}}$-bulk--boundary system only
after $\mathrm{OCA}_T$ is exhibited.

The map $\mathrm{OCA}_T$ is constructed on the holomorphic--topological
locus where Vol~II's Costello--Gwilliam factorisation calculus applies
and the local formal-Darboux theorem
\textup{(}\texttt{thqg\_open\_closed\_realization.tex}\textup{)} provides
the bulk model. Off that locus, the residual is named and the bulk
claim is $\ClaimStatusConditional$.
\end{convention}

\begin{proposition}[OCA discharge for the 3d B-model on
$T^*\mathbb{A}^n$ via HKR; $\ClaimStatusProvedHere$]
\label{prop:master-oca-bmodel}
For the 3d topological B-model on the cotangent target
$T^*\mathbb{A}^n$ with boundary algebra
$A = \mathcal{O}(\mathbb{A}^n) = k[x_1, \ldots, x_n]$
\textup{(}Dirichlet boundary condition\textup{),} the
$\mathrm{OCA}$ comparison of Convention~\ref{conv:master-oca} exists
and is the Hochschild--Kostant--Rosenberg isomorphism. Explicitly,
\[
\mathrm{OCA}\colon
\mathcal{O}^{\mathrm{phys}}_{\mathrm{bulk}}(T^*\mathbb{A}^n)
 \;=\; \Lambda^\bullet T_{\mathbb{A}^n}
 \;=\; k[x_1, \ldots, x_n] \otimes \Lambda^\bullet[\xi_1, \ldots, \xi_n]
\;\xrightarrow{\;\sim\;}\;
Z^{\mathrm{der}}_{\mathrm{ch}}(A)
 \;=\; \mathrm{HH}^\bullet(A, A),
\]
with $\xi_i$ corresponding to the inner derivation
$\partial / \partial x_i$ in cohomological degree~$+1$.
\end{proposition}

\begin{proof}
The bulk physical operator algebra in the 3d topological B-model on
a cotangent target $T^*X$ is identified with the polyvector fields
$\Lambda^\bullet T_X$ via the cotangent BV
quantisation~\cite[Ch.~11]{CG-vol2}: the Costello--Gwilliam
factorisation algebra reduces, on the unique critical locus
$X \hookrightarrow T^*X$, to the polynomial-coefficient exterior
algebra in the cotangent direction, with cohomological grading from
ghost number $|\xi_i| = +1$.

For $X = \mathbb{A}^n$ with global trivialisation
$T_{\mathbb{A}^n} \cong \mathbb{A}^n \times k^n$:
\[
\mathcal{O}^{\mathrm{phys}}_{\mathrm{bulk}}(T^*\mathbb{A}^n)
 \;=\; \mathcal{O}(\mathbb{A}^n) \otimes \Lambda^\bullet(k^n)
 \;=\; k[x_1, \ldots, x_n] \otimes \Lambda^\bullet[\xi_1, \ldots, \xi_n].
\]

The Hochschild--Kostant--Rosenberg theorem~\cite{HKR62}
\textup{(}see also \cite[\S 9.4.4]{Wei94}\textup{)} asserts that for
a smooth commutative $k$-algebra $A$,
\[
\mathrm{HH}^\bullet(A, A)
 \;\cong\; \Lambda^\bullet_A \, \mathrm{Der}_k(A).
\]
Applied to the polynomial ring $A = k[x_1, \ldots, x_n]$
\textup{(}smooth\textup{):}
$\mathrm{Der}_k(A)$ is the free $A$-module on
$\partial / \partial x_i$, and
\[
\mathrm{HH}^\bullet(A, A)
 \;=\; A \otimes \Lambda^\bullet \langle
 \partial / \partial x_1, \ldots, \partial / \partial x_n
 \rangle.
\]
Identifying $\xi_i \leftrightarrow \partial / \partial x_i$ \textup{(}both
in cohomological degree $+1$\textup{)} yields the stated
quasi-isomorphism. This identification is the
$\mathrm{OCA}$ quasi-isomorphism, exhibiting the 3d B-model on
$T^*\mathbb{A}^n$ as a concrete physical realisation in which the
universal closed sector coincides with the physical bulk.
\end{proof}

\begin{remark}[Scope and extension]
\label{rem:master-oca-bmodel-scope}
The HKR identification extends to $X$ smooth affine over $k$ by the
same argument, with the trivialisation $T_X$ replaced by the
$\mathcal O_X$-module of derivations. For $X$ smooth projective, the
sheafy HKR \textup{(}polyvector fields = $\bigoplus_p
H^q(X, \Lambda^p T_X)$ in bigraded $(p, q)$\textup{)} together with
the Hodge--de Rham spectral sequence supplies the closed-sector
identification at the level of sheaf cohomology; the OCA at this
level is the Calaque--Van~den~Bergh sheafy HKR. Beyond the
cotangent-target setting the construction of physical bulk requires
the full Costello--Gwilliam BV factorisation calculus, and the OCA
becomes a non-trivial theorem rather than a direct
HKR application.
\end{remark}

\begin{lemma}[OCA extension: smooth affine and smooth projective via
sheafy HKR; $\ClaimStatusProvedHere$ on smooth affine,
$\ClaimStatusConditional$ on smooth projective on full Costello--Gwilliam
BV identification]
\label{lem:master-oca-sheafy-hkr-extension}
The OCA of Proposition~\ref{prop:master-oca-bmodel} extends:
\begin{enumerate}[label=\textup{(\arabic*)}]
\item \emph{Smooth affine $X$ over $k$.}
For $A = \mathcal O(X)$ the regular function ring of a smooth affine
variety $X$,
\[
\mathrm{OCA}_X\colon
\Gamma(X, \Lambda^\bullet T_X)
\;\xrightarrow{\;\sim\;}\;
\mathrm{HH}^\bullet(A, A),
\]
the HKR isomorphism for smooth commutative algebras.
\item \emph{Smooth projective $X$ over $k$ \textup{(}conditional on
the Costello--Gwilliam BV bulk identification\textup{).}}
The 3d B-model on $T^*X$ with derived boundary category
$D^b(\mathrm{Coh}(X))$ has OCA the sheafy HKR
\[
\mathrm{OCA}_X\colon
\mathrm R\Gamma\bigl(X, \Lambda^\bullet T_X\bigr)
\;\xrightarrow{\;\sim\;}\;
\mathrm{HH}^\bullet\bigl(D^b(\mathrm{Coh}(X))\bigr)
\;=\;
\bigoplus_{p+q = \bullet} H^q\bigl(X, \Lambda^p T_X\bigr).
\]
\end{enumerate}
\end{lemma}

\begin{proof}
\textup{(1)} For $X$ smooth affine, $A = \mathcal O(X)$ is a smooth
commutative $k$-algebra; the HKR theorem~\cite{HKR62}
\textup{(}\cite[\S 9.4.4]{Wei94}\textup{)} gives
$\mathrm{HH}^\bullet(A, A) \cong \Lambda^\bullet_A \mathrm{Der}_k(A)
\cong \Gamma(X, \Lambda^\bullet T_X)$. The argument of
Proposition~\ref{prop:master-oca-bmodel} for $X = \mathbb A^n$
applies verbatim with the polynomial-ring trivialisation
replaced by the locally free $\mathcal O_X$-module structure on
$T_X$.

\textup{(2)} For $X$ smooth projective, $D^b(\mathrm{Coh}(X))$ is
proper smooth and admits a Serre functor; its Hochschild cohomology
is computed by the sheafy HKR
\textup{(}see, e.g., the classical statement in
\cite[\S 9.4]{Wei94} extended to schemes via Swan-type sheaf
cohomology; modern treatment in
Calaque--Willwacher~\cite{CalaqueWillwacher21} for the higher-genus
operadic refinement\textup{).} The Hodge decomposition of polyvector
field sheaf cohomology gives the bigraded RHS.

The conditional status reflects that the 3d B-model physical-bulk
operator algebra on $T^*X$ for projective $X$ requires the full
Costello--Gwilliam BV factorisation calculus
\textup{(}\cite[Ch.~11--14]{CG-vol2}\textup{);} that this physical
bulk equals derived polyvector fields on $X$ is the Rozansky--Witten
identification, established for hyperkähler $X$ in the appropriate
holomorphic-topological twist and conjectural in full generality.
\end{proof}

\begin{corollary}[OCA for Rozansky--Witten on $T^*K3$;
$\ClaimStatusProvedHere$ on the HKR / Hodge identification,
$\ClaimStatusConditional$ on the Rozansky--Witten bulk identification]
\label{cor:master-oca-rozansky-witten-k3}
For $Y = K3$ smooth projective, the 3d B-model on $T^*K3$
\textup{(}the Rozansky--Witten model on $K3$\textup{)} has bulk
Hochschild cohomology
\[
\mathrm{HH}^p\bigl(D^b(\mathrm{Coh}(K3))\bigr) =
\begin{cases}
1 & p = 0,\,4,\\
22 & p = 2,\\
0 & \text{otherwise,}
\end{cases}
\qquad\sum_p \dim \mathrm{HH}^p = 24.
\]
\end{corollary}

\begin{proof}
By the sheafy HKR
\textup{(}Lemma~\ref{lem:master-oca-sheafy-hkr-extension}~\textup{(2)}\textup{)}
applied to $X = K3$:
\[
\mathrm{HH}^p(K3) \;=\; \bigoplus_{q + r = p} H^q(K3, \Lambda^r T_{K3}).
\]
$K3$ has trivial canonical bundle, so $\Lambda^2 T_{K3} \cong
K_{K3}^{\vee} \cong \mathcal O_{K3}$. The holomorphic symplectic form
$\sigma \in H^0(K3, \Omega^2)$ gives $T_{K3} \cong \Omega^1_{K3}$,
hence $H^q(K3, T_{K3}) = h^{1, q}(K3)$. The K3 Hodge diamond
$h^{0,0} = h^{2,0} = h^{0,2} = h^{2,2} = 1$, $h^{1,1} = 20$, with
all $h^{p, q}$ for $p + q$ odd equal to zero, yields:
\begin{align*}
\mathrm{HH}^0 &= H^0(K3, \mathcal O_{K3}) = 1,\\
\mathrm{HH}^1 &= H^0(T_{K3}) \oplus H^1(\mathcal O_{K3}) = 0 + 0 = 0,\\
\mathrm{HH}^2 &= H^0(\Lambda^2 T_{K3}) \oplus H^1(T_{K3}) \oplus
                H^2(\mathcal O_{K3}) = 1 + 20 + 1 = 22,\\
\mathrm{HH}^3 &= H^1(\Lambda^2 T_{K3}) \oplus H^2(T_{K3}) = 0 + 0 = 0,\\
\mathrm{HH}^4 &= H^2(\Lambda^2 T_{K3}) = 1.
\end{align*}
Total: $1 + 0 + 22 + 0 + 1 = 24 = \chi_{\mathrm{top}}(K3)$, agreeing
with the rank of the K3 Mukai cohomology lattice
$\Lambda_{\mathrm{Muk}}(K3) = H^*(K3, \mathbb Z)$ as expected from the
Mukai--C{\u a}ld{\u a}raru identity
$\dim \mathrm{HH}^*(\mathrm{Coh}(X)) = \dim H^*(X, \mathbb C)$ for
smooth projective $X$.

The conditional status on Rozansky--Witten reflects that the
identification of this Hochschild cohomology with the physical bulk
of the 3d B-model on $T^*K3$ uses the Rozansky--Witten interpretation
of the cotangent BV theory; the identification is established for
hyperk{\"a}hler targets including $K3$ \textup{(}Rozansky--Witten 1997;
Roberts--Willerton 2010\textup{).}
\end{proof}

\begin{corollary}[OCA hierarchy across K3 Hilbert schemes via
Mukai--C{\u a}ld{\u a}raru + Göttsche; $\ClaimStatusProvedHere$ on the
total-dim count, $\ClaimStatusConditional$ on the
hyperk{\"a}hler B-model bulk identification]
\label{cor:master-oca-k3-hilbert-hierarchy}
For each $n \geq 1$, the Hilbert scheme $K3^{[n]}$ is a smooth

\sum_{n \geq 0} q^n\, \dim H^*(K3^{[n]}, \mathbb C)
\;=\;
\prod_{m \geq 1}
 \frac{(1 + q^m)^{2 \cdot 1}\,(1 + q^m)^{2 \cdot 1}}
      {(1 - q^m)^{20}\,(1 - q^m)^{2 \cdot 2}}.
\]
The first several values:
$\dim H^*(K3^{[1]}) = 24$,
$\dim H^*(K3^{[2]}) = 324$,
$\dim H^*(K3^{[3]}) = 3200$, ...
extending the K3 OCA discharge of

\textup{(}the $n = 1$ case\textup{)} to an infinite hyperk{\"a}hler

\end{corollary}

\begin{proof}
Smoothness and hyperk{\"a}hler structure of $K3^{[n]}$ are standard

\textup{(}Lemma~\ref{lem:master-oca-sheafy-hkr-extension}~\textup{(2)}\textup{)}

\begin{corollary}[OCA for 3d B-model on $T^*E$ via sheafy HKR;
$\ClaimStatusProvedHere$ on the HKR / Hodge identification,
$\ClaimStatusConditional$ on the Rozansky--Witten bulk identification]
\label{cor:master-oca-bmodel-elliptic}
For $Y = E$ a smooth elliptic curve, the 3d B-model on $T^*E$ has
bulk Hochschild cohomology
\[
\mathrm{HH}^p\bigl(D^b(\mathrm{Coh}(E))\bigr) =
\begin{cases}
1 & p = 0,\,2,\\
2 & p = 1,\\
0 & \text{otherwise,}
\end{cases}
\qquad \sum_p \dim \mathrm{HH}^p = 4.
\]
\end{corollary}

\begin{proof}
$E$ has trivial canonical bundle and $T_E \cong \mathcal O_E$
\textup{(}translation-invariant frame on an abelian variety\textup{).}
Since $E$ is a complex $1$-fold, $\Lambda^p T_E = 0$ for $p \geq 2$.
The Hodge diamond
$h^{0,0} = h^{1,0} = h^{0,1} = h^{1,1} = 1$ \textup{(}all others
zero\textup{)} gives
$H^q(E, T_E) = H^q(E, \mathcal O_E) =
\begin{cases} 1 & q = 0, 1\\ 0 & q \geq 2\end{cases}$.
The sheafy HKR
\textup{(}Lemma~\ref{lem:master-oca-sheafy-hkr-extension}~\textup{(2)}\textup{)}
yields:
\begin{align*}
\mathrm{HH}^0 &= H^0(\mathcal O_E) = 1,\\
\mathrm{HH}^1 &= H^0(T_E) \oplus H^1(\mathcal O_E) = 1 + 1 = 2,\\
\mathrm{HH}^2 &= H^1(T_E) = 1,
\end{align*}
with $\mathrm{HH}^p = 0$ for $p \geq 3$. Total
$\sum_p \dim \mathrm{HH}^p = 4 = b_0(E) + b_1(E) + b_2(E) =
\dim H^*(E, \mathbb C)$, in accord with the Mukai--C{\u a}ld{\u a}raru
identity.
\end{proof}

\begin{corollary}[OCA for the quintic CY 3-fold;
$\ClaimStatusProvedHere$ on HKR/Hodge,
$\ClaimStatusConditional$ on B-model bulk identification]
\label{cor:master-oca-quintic}

3d B-model on $T^*Y_5$ has bulk Hochschild cohomology
\[
\bigl(\dim \mathrm{HH}^p(D^b(\mathrm{Coh}(Y_5)))\bigr)_{p = 0,\ldots,6}
\;=\;
(1,\, 0,\, 101,\, 4,\, 101,\, 0,\, 1),
\qquad
\sum_p \dim \mathrm{HH}^p = 208.
\]
\end{corollary}

\begin{proof}
The quintic is a smooth Calabi--Yau $3$-fold with Hodge numbers
$h^{0, 0} = h^{3, 3} = 1$, $h^{1, 1} = 1$, $h^{2, 1} = h^{1, 2} = 101$,
$h^{3, 0} = h^{0, 3} = 1$, $h^{2, 2} = 1$, other Hodge numbers zero.
For a CY $3$-fold, the holomorphic volume form provides
$\Lambda^p T_Y \cong \Omega^{3-p}_Y$ \textup{(}via
$K_Y^\vee = \det T_Y \cong \mathcal O_Y$\textup{).} Hence
$H^q(\Lambda^p T_Y) = h^{3 - p,\, q}$.

By sheafy HKR
\textup{(}Lemma~\ref{lem:master-oca-sheafy-hkr-extension}~\textup{(2)}\textup{)}:
\begin{align*}
\mathrm{HH}^0 &= h^{3, 0} = 1,\\
\mathrm{HH}^1 &= h^{2, 0} + h^{3, 1} = 0 + 0 = 0,\\
\mathrm{HH}^2 &= h^{1, 0} + h^{2, 1} + h^{3, 2} = 0 + 101 + 0 = 101,\\
\mathrm{HH}^3 &= h^{0, 0} + h^{1, 1} + h^{2, 2} + h^{3, 3}
               = 1 + 1 + 1 + 1 = 4,\\
\mathrm{HH}^4 &= h^{0, 1} + h^{1, 2} + h^{2, 3} = 0 + 101 + 0 = 101,\\
\mathrm{HH}^5 &= h^{0, 2} + h^{1, 3} = 0 + 0 = 0,\\
\mathrm{HH}^6 &= h^{0, 3} = 1.
\end{align*}
Total $1 + 0 + 101 + 4 + 101 + 0 + 1 = 208 = \dim H^*(Y_5, \mathbb C)$,
consistent with the Mukai--C{\u a}ld{\u a}raru identity.

The $\mathrm{HH}^3 = 4$ contribution carries the four protected
generators \textup{(}volume class, Kähler class, anti-canonical class,
top class\textup{);} the $\mathrm{HH}^2 = 101$ and $\mathrm{HH}^4 =
101$ contributions are the complex-structure deformations of the
quintic, which under mirror symmetry exchange with the Kähler-moduli
side of the mirror quintic $\tilde Y_5$.
\end{proof}

\begin{corollary}[Schouten--Nijenhuis bracket as closed-colour

[-, -]_{\mathrm{brace}}\bigl|_{\mathrm{HH}^\bullet}
 \;\xleftrightarrow{\;\mathrm{HKR}\;}\;
[-, -]_{SN}\bigl|_{\Lambda^\bullet T_{\mathbb A^n}}.
\]
Consequently the bulk operator algebra of the 3d B-model on

\end{corollary}

\begin{proof}

associative algebra $A$ was introduced
in~\cite{Ger63}; it gives a degree-$-1$ graded Lie bracket compatible
with the cup product, exhibiting $\mathrm{HH}^\bullet(A, A)$ as a

with the $\mathsf{SC}^{\mathrm{ch,top}}$ closed colour.

\textup{(}Proposition~\ref{prop:master-oca-bmodel}\textup{)} sends a

 \cdot f(x) \;\in\; \mathrm{HH}^p(A, A)
\]
to the corresponding polyvector field
$f(x) \, \xi_{i_1} \wedge \cdots \wedge \xi_{i_p}
\in \Lambda^p T_{\mathbb A^n}$. Under this identification the
Gerstenhaber bracket on $\mathrm{HH}^\bullet$ becomes the
Schouten--Nijenhuis bracket
\[
[X, Y]_{SN}
 \;=\; \mathcal L_X Y - (-1)^{(|X|-1)|Y|}\, \iota_{dY} X
\]
on $\Lambda^\bullet T_{\mathbb A^n}$ \textup{(}see, e.g.,
\cite[\S 9.4]{Wei94}, or, in the chiral / curved generalisation,
\cite{CL20}\textup{).}
\end{proof}

\begin{lemma}[Scalar non-faithfulness; $\ClaimStatusProvedHere$]
\label{lem:master-scalar-non-faithfulness}
The graded Euler character
\[
\chi\colon \mathrm{FiltCh}_{\mathsf{SC}^{\mathrm{ch,top}}}
\longrightarrow \mathbb Z(\!(q)\!)
\]
is not faithful. An equality $\chi(C) = \chi(C')$ does not determine
\textup{(}i\textup{)} a quasi-isomorphism $C \simeq C'$;
\textup{(}ii\textup{)} an
$\mathsf{SC}^{\mathrm{ch,top}}$-morphism;
\textup{(}iii\textup{)} a Hall product; \textup{(}iv\textup{)} a
Drinfeld pairing; \textup{(}v\textup{)} a current-envelope locality
structure; or \textup{(}vi\textup{)} a derived-centre trace map.
\end{lemma}

\begin{proof}
Choose a filtered acyclic complex $K$ with $\chi(K) = 0$. Then
$\chi(C) = \chi(C \oplus K)$, while $C$ and $C \oplus K$ may carry
inequivalent filtered products, coproducts, trace maps, or
open--closed mixed operations. Equivalently, replace a filtered
differential $d$ by $d + \delta$ with $\delta$ of higher filtration
degree: the associated-graded Euler characteristic is unchanged, so
$\chi$ does not see $\delta$; the chain-level product structure does.
\end{proof}

\begin{theorem}[Heterotic non-faithfulness witness at $c = 16$;
$\ClaimStatusProvedHere$]
\label{thm:master-scalar-nonfaithful-witness-c16}
The lattice vertex operator algebras
\[
\cA_1 \;=\; \Vlat_{E_8 \oplus E_8},
\qquad
\cA_2 \;=\; \Vlat_{D_{16}^+}
\;=\; \Vlat_{\mathrm{Spin}(32)/\mathbb Z_2},
\]
attached to the two even unimodular positive-definite lattices of
rank~$16$ \textup{(}\cite{Bor86, FLM88}; see
Example~\ref{ex:lattice:heterotic}\textup{)} satisfy
\begin{enumerate}[label=\textup{(\roman*)}]
\item the same central charge $c(\cA_1) = c(\cA_2) = 16$;
\item the same graded character
\[
\chi(\cA_1)(q) \;=\; \chi(\cA_2)(q)
\;=\; \frac{E_8(q)}{\eta(q)^{16}},
\qquad
E_8(q) = 1 + 480\sum_{n\geq 1}\sigma_7(n)\,q^n;
\]
\item non-isomorphic chiral algebra structure, hence inequivalent as
$\mathsf{SC}^{\mathrm{ch,top}}$-objects on the open colour.
\end{enumerate}
This is the concrete realisation of
Lemma~\ref{lem:master-scalar-non-faithfulness} at the smallest
central charge where the standard landscape exhibits two distinct
chiral algebras with equal graded character.
\end{theorem}

\begin{proof}
\textup{(}i\textup{)} For a lattice VOA $\Vlat_L$ with $L$ even
integral of rank $r$, the central charge is $c(\Vlat_L) = r$. Both
lattices have rank $16$.

\textup{(}ii\textup{)} The graded character is
$\chi(\Vlat_L)(q) = \theta_L(q) / \eta(q)^r$. For $L$ even unimodular
positive-definite of rank $r$ divisible by $8$, the theta function
$\theta_L$ is a holomorphic modular form of weight $r/2$ on
$\mathrm{SL}_2(\mathbb Z)$. At $r = 16$ the space
$M_8(\mathrm{SL}_2(\mathbb Z))$ is one-dimensional, spanned by the
Eisenstein series $E_8(q)$ \textup{(}cusp forms start at weight
$12$\textup{)}; both $\theta_{E_8 \oplus E_8}$ and
$\theta_{D_{16}^+}$ have constant term~$1$, hence
$\theta_{E_8 \oplus E_8} = \theta_{D_{16}^+} = E_8(q)$.

\textup{(}iii\textup{)} The weight-$1$ subspaces are the affine
Lie algebras attached to the norm-$2$ vectors of $L$:
\[
(\cA_1)_1 \;=\; \widehat{\mathfrak e_8 \oplus \mathfrak e_8}_1,
\qquad
(\cA_2)_1 \;=\; \widehat{\mathfrak{so}(32)}_1
                                 = \widehat{\mathfrak d_{16}}_1,
\]
both of dimension $496$, both of rank $16$, both with level-$1$
Sugawara central charge $16$ \textup{(}consistent with
\textup{(}ii\textup{)}\textup{)}. The horizontal Lie algebras are
nonetheless non-isomorphic: $\mathfrak e_8 \oplus \mathfrak e_8$ has
two simple summands of rank $8$ each, while $\mathfrak d_{16}$ is
simple of rank $16$. Equivalently, their Dynkin diagrams differ
\textup{(}two disconnected copies of $E_8$ versus the connected
$D_{16}$ fork\textup{),} and so do the multisets of fundamental
degrees: $\{2,8,12,14,18,20,24,30\}^{\sqcup 2}$ for
$\mathfrak e_8 \oplus \mathfrak e_8$ versus
$\{2,4,6,\ldots,28,30,16\}$ for $\mathfrak d_{16}$ \textup{(}where
the extra $16$ is the Pfaffian degree\textup{).} The OPEs on
$(\cA)_1 \otimes (\cA)_1$ therefore have inequivalent affine
Lie-algebra structure, and $\cA_1 \not\simeq \cA_2$ as chiral
algebras, hence as $\mathsf{SC}^{\mathrm{ch,top}}$-objects on the
open colour.
\end{proof}

\begin{lemma}[$\delta$ at the abelian Cartan window: trivial Drinfeld
doubling with Mukai pairing; $\ClaimStatusProvedHere$ on the
classical Drinfeld-double formalism]
\label{lem:master-delta-cartan-window}
Restrict the K3-BKM positive half to its Cartan part:
$Y^+_{\mathrm{Hall}}(K3)\bigl|_{\mathfrak h} =
U(\mathfrak h_{\Delta_5})$, where
$\mathfrak h_{\Delta_5} \cong \Lambda^{2,1}_{\mathrm{Muk}} \otimes
\mathbb C$ is the rank-$3$ abelian Cartan subalgebra. The Drinfeld
doubling of Convention~\ref{conv:master-hall-borcherds-datum}
restricted to this window is the trivial classical doubling
\[
D\bigl(\mathfrak h_{\Delta_5}\bigr)
\;=\;
\mathfrak h_{\Delta_5} \;\oplus\; \mathfrak h^*_{\Delta_5},
\]
a Manin triple $(\mathfrak h \oplus \mathfrak h^*, \mathfrak h,
\mathfrak h^*)$ with bracket
\[
[X, Y]_D = 0,
\qquad
[\xi, \eta]_D = 0,
\qquad
[X, \xi]_D = \langle X, \xi\rangle_{\mathrm{Muk}}
\quad
\text{for } X, Y \in \mathfrak h,\;
\xi, \eta \in \mathfrak h^*,
\]
where $\langle \cdot, \cdot\rangle_{\mathrm{Muk}}$ is the Mukai pairing
on $\Lambda^{2,1}_{\mathrm{Muk}}$ identifying $\mathfrak h \cong
\mathfrak h^*$. The cobracket on the Cartan-only sector vanishes
identically; non-trivial cobracket contributions $\delta(x_\alpha)$
for $\alpha \in \Phi^{\mathrm{imag}}_{\Delta_5}$ arise only when
imaginary roots are included
\textup{(}per Gritsenko--Nikulin~\cite[\S 5,
Thm.~5.3]{GritsenkoNikulin1998}; cf.~the cobracket formula above\textup{).}
\end{lemma}

\begin{proof}
For an abelian Lie algebra $\mathfrak h$ with non-degenerate
pairing $\langle\cdot,\cdot\rangle$, the Drinfeld double $D(\mathfrak
h)$ is constructed as the Manin triple $(\mathfrak h \oplus
\mathfrak h^*, \mathfrak h, \mathfrak h^*)$ with the canonical pairing
\textup{(}Etingof--Schiffmann~\cite[\S 4]{EtingofSchiffmann98}\textup{).}
The bracket on $D(\mathfrak h)$ is determined by:
\begin{itemize}[noitemsep]
\item $\mathfrak h$ has trivial bracket \textup{(}abelian by
hypothesis\textup{);}
\item $\mathfrak h^*$ inherits a bracket from the cobracket $\delta$
on $\mathfrak h$: $\delta(X) = \sum a_i \otimes b_i \in \mathfrak h
\otimes \mathfrak h$ gives $[X^*, Y^*]_{\mathfrak h^*} = \sum
\langle a_i, X^*\rangle\langle b_i, Y^*\rangle \cdot \xi$ for
$\xi \in \mathfrak h^*$;
\item mixed bracket $[X, \xi]_D = \langle X, \xi\rangle$ by Manin-triple
construction.
\end{itemize}
For the K3-BKM Cartan, the cobracket on Cartan-only generators is
zero \textup{(}the cobracket structure on $\mathfrak g_{\Delta_5}$
takes Cartan-only inputs to imaginary-root-supported outputs, which
vanish in the Cartan-only restriction\textup{).} Hence
$[\xi, \eta]_D = 0$ on the Cartan-only window. The Mukai pairing
gives the canonical mixed bracket.

This is the simplest finite-window discharge of $\delta$ in
Convention~\ref{conv:master-hall-borcherds-datum}: at the Cartan-only
sector, $\delta$ is the trivial Manin triple of an abelian Lie algebra
with Mukai pairing; the non-trivial $\delta$ on imaginary-root windows
remains the open residual.
\end{proof}

\begin{theorem}[Raw chain-level T1--T5 discharge for $\mathrm{Vir}_{26}$
via bosonic-string BRST; $\ClaimStatusProvedHere$]
\label{thm:master-vir-26-bc-brst-topologisation}
At matter central charge $c_m = 26$, the Virasoro algebra
$\mathrm{Vir}_{26}$ tensored with the conformal-weight-$(2, -1)$
fermionic ghost system $(b, c)$ of central charge $c_{bc} = -26$
admits a raw chain-level discharge of all five topologisation
obstructions T1--T5
\textup{(}Convention~\ref{conv:master-topologisation}\textup{)}.
Explicitly, let
\[
T_{bc} = -2{:}b\partial c{:} - {:}\partial b\, c{:},
\qquad
T_{\mathrm{tot}} = T_m + T_{bc},
\]
so that $c_{\mathrm{tot}} = c_m + c_{bc} = 0$. Define
\[
Q_{\mathrm{BRST}}
 = \oint \frac{dz}{2\pi i}\,
   {:}c(z)\bigl(T_m + \tfrac12 T_{bc}\bigr)(z){:},
\qquad
G = b.
\]
Then $Q_{\mathrm{BRST}}^2 = 0$ at $c_m = 26$ and
\[
[Q_{\mathrm{BRST}}, b(z)] = T_{\mathrm{tot}}(z).
\]
\end{theorem}

\begin{proof}
The central charge of a fermionic $(b, c)$ system of weight $(h, 1-h)$
is $c_{bc}(h) = -2(6h^2 - 6h + 1)$
\textup{(}Polchinski~\cite[eq.~(2.5.12)]{Pol98}\textup{);} at $h = 2$
this gives $c_{bc} = -26$.

BRST nilpotence $Q_{\mathrm{BRST}}^2 = 0$ at $c_m = 26$ is the
critical-dimension cancellation~\cite[\S 4.3]{Pol98}: the OPE
$j_{\mathrm{BRST}}(z) j_{\mathrm{BRST}}(w)$ has a residue proportional
to $c_{\mathrm{tot}} = c_m + c_{bc}$, which vanishes exactly when
$c_m = 26$.

The identity $[Q_{\mathrm{BRST}}, b(z)] = T_{\mathrm{tot}}(z)$ follows
from
\[
j_{\mathrm{BRST}}(z) b(w)
 \;\sim\; \frac{T_{\mathrm{tot}}(w)}{z - w} + \cdots,
\]
which is computed using $b(z) c(w) \sim 1/(z - w)$ and the standard
$cT_m$, $cT_{bc}$ OPEs, then contour deformation
$\oint_w dz/(2\pi i)$ extracts the first-order pole.

The five T-axioms now follow.

\textbf{T1} \textup{(}construct $G$ with $[Q, G] = T$\textup{):}
$G = b$, $Q = Q_{\mathrm{BRST}}$, $[Q, b] = T_{\mathrm{tot}}$ as above.

\textbf{T2} \textup{(}filtration / completion compatibility\textup{):}
the ghost-number filtration $\mathrm{gh}(b) = -1$,
$\mathrm{gh}(c) = +1$, $\mathrm{gh}(Q_{\mathrm{BRST}}) = +1$ is bounded
below on each conformal-weight piece and preserved by both
$Q_{\mathrm{BRST}}$ and $G$.

\textbf{T3} \textup{(}finite propagation\textup{):}
each mode $b_n$, $c_n$, $L^{\mathrm{tot}}_n$ shifts $L_0$ by $-n$
\textup{(}so weight-bounded subspaces map to weight-bounded
subspaces\textup{);} $Q_{\mathrm{BRST}}$ and $G$ act on each
weight-bounded subspace with bounded mode-shift, so propagation is
finite.

\textbf{T4} \textup{(}SDR-independence of transferred
$E_3$-operations\textup{):}
BRST cohomology
$H^\bullet(\mathrm{Vir}_{26} \otimes V_{bc}, Q_{\mathrm{BRST}})$
is invariant under canonical transformations preserving the
$(L_n, b_n, c_n)$ algebra; the transferred $E_3$-structure is the
topological-string operator product algebra of~\cite{FGZ86}, which
is SDR-independent by the standard homotopy-transfer argument
applied to the Koszul resolution of the ghost vacuum.

\textbf{T5} \textup{(}anomaly cancellation in $H^1/H^2$\textup{):}
$Q_{\mathrm{BRST}}^2 = 0$ \emph{is} the anomaly cancellation
condition. Concretely,
$Q_{\mathrm{BRST}}^2 = (1/2)\{Q_{\mathrm{BRST}}, Q_{\mathrm{BRST}}\}$
is proportional to the Virasoro central extension class
$c_{\mathrm{tot}} \in H^2(\mathrm{Witt}, \mathbb C) \cong \mathbb C$,
which vanishes at $c_{\mathrm{tot}} = c_m + c_{bc} = 0$.
\end{proof}

\begin{theorem}[Generalised matter-ghost BRST T1--T5 discharge at
arbitrary matter central charge $c \geq -1$;
$\ClaimStatusProvedHere$]
\label{thm:master-generalized-matter-ghost-brst}

generator $T_{\mathfrak A}$ and central charge $c_{\mathfrak A} \geq
-1$. Choose a fermionic ghost system $bc(h, 1-h)$ with weights given

\[

\[
T_{\mathrm{tot}} = T_{\mathfrak A} + T_{bc},
\quad
Q_{\mathrm{BRST}} = \oint \frac{dz}{2\pi i}\, {:}c(z)
\bigl(T_{\mathfrak A} + \tfrac{1}{2} T_{bc}\bigr)(z){:},
\quad
G = b,
\]
with $Q_{\mathrm{BRST}}^2 = 0$ \textup{(}from
$c_{\mathfrak A} + c_{bc} = 0$\textup{)} and
$[Q_{\mathrm{BRST}}, b(z)] = T_{\mathrm{tot}}(z)$.

In particular:
\begin{itemize}[noitemsep]
\item $\mathfrak A = \mathrm{Vir}_{26}$, ghosts $bc(2, -1)$: recovers
the bosonic-string BRST of
Theorem~\ref{thm:master-vir-26-bc-brst-topologisation}.
\item $\mathfrak A = \beta\gamma$ at conformal weight $\lambda \in
\mathbb R$ \textup{(}with $c = 12\lambda^2 - 12\lambda + 2 \geq -1$

\item $\mathfrak A = \mathrm{Vir}_c$ minimal model at $c = 1 - 6/m(m+1) \in
(0, 1)$ for $m \geq 2$: ghosts $bc(h_\pm, h_\mp)$ with real weights
$h_\pm \in (0, 1)$ depending on $m$.
\end{itemize}
\end{theorem}

\begin{proof}
The fermionic $bc(h, 1-h)$ system has stress tensor
$T_{bc} = -h\, {:}b\partial c{:} - (1-h)\, {:}\partial b\, c{:}$ and
central charge $c_{bc}(h) = -2(6h^2 - 6h + 1)$
\textup{(}\cite[eq.~(2.5.12)]{Pol98}\textup{).} Setting
$c_{bc}(h) = -c_{\mathfrak A}$ gives the quadratic
$12h^2 - 12h + 2 + c_{\mathfrak A} = 0$, equivalently
$h(h-1) = (c_{\mathfrak A} - 2)/12$, with discriminant $1 + (c_{\mathfrak A} - 2)/3 =
(c_{\mathfrak A} + 1)/3$; real roots exist iff
$c_{\mathfrak A} \geq -1$.

The BRST construction proceeds as in
Theorem~\ref{thm:master-vir-26-bc-brst-topologisation}: $T_{bc}$ has
the OPE structure of a fermionic stress tensor, $T_{\mathfrak A}$
commutes with $T_{bc}$ \textup{(}matter and ghost sectors are
independent\textup{),} and the only OPE singularity of
$j_{\mathrm{BRST}}(z) j_{\mathrm{BRST}}(w)$ at leading order is the
conformal-anomaly term proportional to $c_{\mathfrak A} + c_{bc} = 0$,
so $Q_{\mathrm{BRST}}^2 = 0$ on the operator product algebra.

The bracket $[Q_{\mathrm{BRST}}, b(z)] = T_{\mathrm{tot}}(z)$ follows
from the OPE
$j_{\mathrm{BRST}}(z) b(w) \sim T_{\mathrm{tot}}(w) / (z - w) + \cdots$
by the same contour-deformation argument as in
Theorem~\ref{thm:master-vir-26-bc-brst-topologisation}, using
$b(z) c(w) \sim 1/(z - w)$ \textup{(}fermionic\textup{)} and the
universal structure of the BRST current independent of matter type.

T2--T4 follow as in
Theorem~\ref{thm:master-vir-26-bc-brst-topologisation}: ghost-number
filtration with $\mathrm{gh}(b) = -1$, $\mathrm{gh}(c) = +1$; bounded
mode shifts of $T, b, c$; SDR-independence of BRST cohomology under
canonical transformations.

T5 is the anomaly cancellation: $Q_{\mathrm{BRST}}^2 = 0$ on the full
operator product algebra is equivalent to $c_{\mathrm{tot}} = 0$,
which holds by construction.
\end{proof}

\begin{corollary}[Per-spin antighost commutativity for linear

\begin{itemize}[noitemsep]

\end{theorem}

\begin{proof}
\textbf{T1.} The N=2 SCA OPE is
\[
G^+(z) G^-(w) \;\sim\; \frac{2c/3}{(z-w)^3} + \frac{2 J(w)}{(z-w)^2}
 + \frac{2T(w) + \partial J(w)}{z - w} + \mathrm{reg}.
\]
Hence
\[
\{Q, G(z)\}
 = \tfrac{1}{2} \oint_z \frac{dw}{2\pi i}\, G^+(w) G^-(z)
 = \tfrac{1}{2}\bigl(2T(z) + \partial J(z)\bigr)
 = T_{\mathrm{top}}(z).
\]

\textbf{T5} \textup{(}anomaly cancellation, raw chain at every
$c$\textup{).}
$U(1)$ $R$-charge conservation forces $G^+(z) G^+(w) \sim 0$
\textup{(}both fields carry $R$-charge $+1$; the OPE residue
would carry $R$-charge $+2$, but no operator of $R$-charge
$\pm 2$ exists in the N=2 SCA generator set\textup{).} Therefore
\[
Q^2 = \tfrac{1}{2}\{Q, Q\}
 = \tfrac{1}{2}\oint dz\, dw\, G^+(z) G^+(w) = 0
 \quad \text{at every } c.
\]
Crucially, no central-charge constraint is required: unlike the
bosonic-string BRST of
Theorem~\ref{thm:master-vir-26-bc-brst-topologisation} which forces
$c_{\mathrm{tot}} = 0$, the N=2 topological twist is anomaly-free
intrinsically.

\textbf{T2.} $R$-charge grading $\mathrm{gh}(J) = 0$,
$\mathrm{gh}(G^\pm) = \pm 1$, $\mathrm{gh}(T) = 0$ gives a
$\mathbb Z$-graded ghost-number filtration; $Q$ raises ghost number
by $+1$, $G$ lowers by $-1$.

\textbf{T3.} Each generator's modes $T_n, J_n, G^\pm_n$ shift $L_0$
by $-n$; each weight-bounded subspace is preserved by both
$Q$ and $G$ with bounded mode-shift.

\textbf{T4.} BRST cohomology $H^\bullet(\mathfrak A, Q)$ is invariant
under canonical transformations preserving the N=2 SCA generator
structure; the transferred $E_3$-operations on
$H^\bullet(\mathfrak A, Q)$ are SDR-independent by standard
homotopy-transfer applied to the $Q$-acyclic complement.
\end{proof}

\section{Cross-volume specialization}
\label{sec:master-concordance-cross-volume}

Vol~II and Vol~III do not add inverse arrows to the Vol~I
bar--cobar lane.  They specialize it.

\begin{enumerate}[label=\textup{(\roman*)}]
\item Vol~II uses the two-coloured dioperad
$\mathsf{SC}^{\mathrm{ch,top}}$.  Its mixed operations are
directional: the closed colour acts on the open colour, and the open
colour does not act back on the closed colour.  The open-colour
bar--cobar comparison supplies line modules over the Koszul dual; the
closed-sector cochain object is still the chiral Hochschild complex.
\item Vol~III uses the two-stage factorisation
\[
\Phi_d^{(\Sigma_{d-1},C)}
=\mathrm{Sp}^{\mathrm{ch}}_{\Sigma_{d-1},C}
\circ\Phi_d^{\mathrm{FA}}.
\]
Stage~$1$ produces an $E_d$ holomorphic factorisation algebra.
Stage~$2$ integrates along $\Sigma_{d-1}$ and restricts to a reference
curve $C$, producing an $E_1$-chiral shadow.  This specialization is
not inversion.
\item The shared Theorem~A surface is the curve-side $E_1$ chiral
shadow after Stage~$2$.  A single CY$_d$ category may therefore have
many $E_1$ shadows, indexed by $(\Sigma_{d-1},C)$.
\end{enumerate}

The five-level Beilinson tower of
Convention~\ref{conv:master-primitive-datum} carries a vertical
equivalence at every level to its Vol~II or Vol~III companion
\textup{(}\texttt{thm:mr-vertical-dictionary} in
\texttt{master\_reconstruction.tex}\textup{);} the dedicated chapters
exist for levels $0$ and $4$
\textup{(}\texttt{vertical\_equivalence\_level\_0.tex},
\texttt{vertical\_equivalence\_level\_4.tex}\textup{).}  The
intermediate levels $1$ and $2$ admit the following lemma-level
inscriptions, stating the chart-algebra and bar-coalgebra vertical
equivalences as direct consequences of the dictionary theorem.

\begin{lemma}[Level-$1$ vertical equivalence: chart algebra
$A_b$ $\leftrightarrow$ $\Phi_d^{\mathrm{FA}}$ at chosen vacuum;
$\ClaimStatusConditional$ on hypothesis package $H_{V1}$]
\label{lem:master-vertical-level-1}
Suppose the Open vacuum $b \in \cC^{\mathrm{op}}$ on the factorisation
dg-category over $(X, D, \tau)$ corresponds via the vacuum-target
correspondence to a CY-$d$ target $c$ in the CY-side category
$\mathrm{CY}_d\text{-Cat}$. Then the chart algebra
\[
A_b \;=\; \End_{\cC}(b)
\quad\text{(Open, level~$1$)}
\]
is quasi-isomorphic as an $E_1$-chiral algebra to the evaluation of
the Vol~III Stage-$1$ factorisation-algebra functor at $c$:
\[
A_b \;\simeq\; \Phi_d^{\mathrm{FA}}(\cC)\bigl|_c
\quad\text{(Vol~III, level~$1$).}
\]
The equivalence is conditional on hypothesis package $H_{V1}$ of
\texttt{thm:mr-vertical-dictionary}: Kontsevich--Tamarkin
$E_d$-formality of $\mathrm{CY}_d\text{-Cat}$ for $d \leq 3$, via
three-vanishing of the obstruction class.
\end{lemma}

\begin{proof}
The vacuum-target correspondence supplies a chosen identification
$b \leftrightarrow c$ at chart level. By the
$E_d$-formality~\cite{Kon99,CalaqueWillwacher21} of the CY-side
category, $\Phi_d^{\mathrm{FA}}(\cC)\bigl|_c$ is the $E_d$-formal
factorisation-algebra evaluation at $c$, equivalent to the
endomorphism algebra
$\End_{\Phi_d^{\mathrm{FA}}(\cC)}(c)$ on the CY side. This equals
$\End_{\cC^{\mathrm{op}}}(b)$ on the Open side by the chosen
identification, hence equals $A_b$.
\end{proof}

\begin{lemma}[Level-$2$ vertical equivalence: bar coalgebra
$B(A_b)$ $\leftrightarrow$ $\mathrm{Sp}^{\mathrm{ch}}_{\Sigma_{d-1},
C}$ specialisation; $\ClaimStatusConditional$ on
$H_{V1} \cup H_{V2}$]
\label{lem:master-vertical-level-2}
Under the vacuum-target correspondence of
Lemma~\ref{lem:master-vertical-level-1}, the Open bar coalgebra at
level~$2$ and the Vol~III two-stage specialisation at level~$2$
agree:
\[
B(A_b)
\;\simeq\;
B\bigl(\mathrm{Sp}^{\mathrm{ch}}_{\Sigma_{d-1}, C}
       \circ \Phi_d^{\mathrm{FA}}(\cC)\bigr)
\quad\text{as }E_1\text{-chiral coalgebras over }(\mathrm{Ass}^{\mathrm{ch}})^!.
\]
The equivalence is conditional on $H_{V1}$
\textup{(}Lemma~\ref{lem:master-vertical-level-1}\textup{)} and
$H_{V2}$ \textup{(}well-definedness of the Stage-$2$ specialisation
functor $\mathrm{Sp}^{\mathrm{ch}}_{\Sigma_{d-1}, C}$ for the chosen
$(\Sigma_{d-1}, C)$\textup{)}, both of
\texttt{thm:mr-vertical-dictionary}.
\end{lemma}

\begin{proof}
By Lemma~\ref{lem:master-vertical-level-1},
$A_b \simeq \Phi_d^{\mathrm{FA}}(\cC)\bigl|_c$. The Stage-$2$
specialisation $\mathrm{Sp}^{\mathrm{ch}}_{\Sigma_{d-1}, C}$ at the
chosen reference curve $C$ produces an $E_1$-chiral algebra on $C$
whose evaluation at the chosen target $c$ agrees with the Stage-$1$
evaluation, i.e.,
$\mathrm{Sp}^{\mathrm{ch}}_{\Sigma_{d-1}, C}(\Phi_d^{\mathrm{FA}}(\cC))\bigl|_c
\simeq A_b$ as $E_1$-chiral algebras
\textup{(}hypothesis $H_{V2}$\textup{).} Apply the bar functor
$B = T^c(s^{-1} \overline{(\cdot)})$ to both sides: by functoriality,
$B(A_b) \simeq B(\mathrm{Sp}^{\mathrm{ch}}_{\Sigma_{d-1}, C}
\circ \Phi_d^{\mathrm{FA}}(\cC))$ as $E_1$-chiral coalgebras over
$(\mathrm{Ass}^{\mathrm{ch}})^!$.
\end{proof}

\begin{lemma}[Level-$5$ vertical equivalence: modular trace
$\leftrightarrow$ universal Borcherds weight;
$\ClaimStatusConditional$ on $H_{V5}$]
\label{lem:master-vertical-level-5}
At level $5$ of the Beilinson tower, the Open-side modular trace
$\mathrm{Tr}_{\cC^{\mathrm{op}}}(A_b)$ on the modular shadow

$\Delta_5$:

\item the Open modular trace produces the BPS partition function
$Z_{\mathrm{BPS}}^{K3 \times E} =

$M_{-10}^{\mathrm{mero}}(\mathrm{Sp}_4(\mathbb Z))$
\textup{(}meromorphic modular form of weight $-10$ on

\item the Vol~III universal Borcherds weight reads
$\kappa_{\mathrm{BKM}}(\Delta_5) = c_5(0)/2 = 5$ as the weight of
$\Delta_5$ in $S_5(K(1))$
\textup{(}weight-$5$ cusp form\textup{);}
\end{enumerate}
both invariants are jointly determined by $\Delta_5$, with~\textup{(1)}
the inverse partition function and~\textup{(2)} the modular weight.
\end{lemma}

\begin{proof}
The Borcherds product $\Delta_5$ for the K3-BKM algebra

\bigl(1 - e^{2\pi i \langle \alpha, Z\rangle}\bigr)^{\mathrm{mult}(\alpha)}
\;=\;\Delta_5(Z),
\]
exhibiting $\Delta_5$ as the Weyl--Kac denominator of
$\mathfrak g_{\Delta_5}$.

\textup{(}1\textup{)} The Open-side modular trace on the K3$\times$E

\textup{(}2\textup{)} The Vol~III universal Borcherds weight identity

Both \textup{(1)} and \textup{(2)} are scalar invariants determined

\textup{(}\texttt{thm:mr-vertical-dictionary}\textup{).}

\section{Boundary of the dictionary}
\label{sec:master-concordance-boundary}

The seven-face atlas is a theorem on the comparison locus of
Theorem~\ref{thm:master-seven-face}.  Five proof obligations remain
outside that locus.
\begin{enumerate}
\item The term-by-term comparison between the $\mathcal W_N$
Hamiltonians produced by $\Theta_{\mathcal W_N}$ and the operators
of Gaiotto--Zeng is a normalization problem beyond the algebraic
commutativity theorem.
\item Genus $g\geq2$ multi-weight algebras carry the cross-channel
correction
$\delta F_g^{\mathrm{cross}}(\cA)$; the seven-face scalar statement
uses the uniform-weight projection.
\item Arbitrary-nilpotent non-principal $\mathcal W$-algebra
BV duality is known only on the stated hook-type corridor.
\item The Drinfeld-centre presentation of the closed sector does not
follow from bar--cobar inversion or Verdier duality alone; it requires
the separate chiral Hochschild comparison.
\item The Vol~III K3 row compares with F5--F6 only through the
Hall--Drinfeld/CoHA carrier.  It is not a Drinfeld Yangian
$Y_\hbar(\mathfrak g_{\Delta_5})$; the constants
$K^{\kappa_{\mathrm{ch}}}(\mathrm{Mukai}(K3))=8$,
$\kappa_{\mathrm{BKM}}(\Delta_5)=5$,
$\kappa_{\mathrm{ch}}^{\mathrm{Heis}}(K3\times E)=3$, and
$\kappa_{\mathrm{cat}}(K3\times E)=0$ live on distinct scalar lanes.
\end{enumerate}
