\section{Principal contributions}\label{sec:principal-contributions}

\begin{enumerate}[label=\emph{(\arabic*)}]

\item \emph{Configuration space bar-cobar adjunction
  (Theorems~\ref{thm:bar-cobar-isomorphism-main}
  and~\ref{thm:higher-genus-inversion}).}
  The bar and cobar constructions for chiral algebras are realized
  via explicit integrals over Fulton--MacPherson compactifications
  $\overline{C}_n(X)$, with the bar differential given by residues
  at collision divisors and the cobar differential by inclusions
  of strata.  The adjunction is proved for all Koszul chiral algebras
  and all genera.  Theorem~A decomposes into three layers
  (Remark~\ref{rem:theorem-A-decomposition}):
  the fundamental theorem of chiral twisting morphisms
  ($\mathrm{A}_0$, Theorem~\ref{thm:fundamental-twisting-morphisms}),
  bar concentration
  ($\mathrm{A}_1$, Theorem~\ref{thm:bar-concentration}), and
  the Verdier-geometric duality
  ($\mathrm{A}_2$, Theorem~\ref{thm:bar-cobar-isomorphism-main}).
  This is the first complete proof of chiral Koszul
  duality via configuration space geometry.

\item \emph{Genus universality
  (Theorem~\ref{thm:genus-universality}).}
  For any Koszul chiral algebra~$\cA$, the genus-$g$ obstruction
  factors as $\mathrm{obs}_g(\cA) = \kappa(\cA) \cdot \lambda_g$,
  where $\kappa$ is an algebra-dependent constant and $\lambda_g$
  is the universal Faber--Pandharipande tautological class
  on~$\overline{\mathcal{M}}_g$.
  The genus-independent coefficient $\kappa(\cA)$ is determined
  by the genus-$1$ curvature.  This result is new.

\item \emph{Deformation-obstruction complementarity
  (Theorem~\ref{thm:quantum-complementarity-main}).}
  For a Koszul pair $(\cA, \cA^!)$, the genus-$g$ obstructions
  satisfy $\mathrm{obs}_g(\cA) + \mathrm{obs}_g(\cA^!) =
  [\text{characteristic class of } Z(\cA)]$ in $H^*(\overline{\mathcal{M}}_g)$.
  The Kodaira--Spencer map is constructed for all Koszul pairs
  and all genera (Theorem~\ref{thm:kodaira-spencer-chiral-complete}).
  This is the first formulation and proof of complementarity
  for chiral algebras.

\item \emph{$\Eone$-chiral Koszul duality
  (Theorem~\ref{thm:e1-chiral-koszul-duality}).}
  The bar-cobar adjunction extends to $\Eone$-chiral algebras
  (nonlocal vertex algebras with $R$-matrix braiding), covering
  Yangians and toroidal algebras.  This goes beyond the
  $\Einf$ (commutative/local) setting of BD and FG.

\item \emph{Explicit computations.}
  Complete Koszul duality computations for all standard families:
  Heisenberg, free fermion, $bc$-$\beta\gamma$, affine Kac--Moody
  ($\widehat{\fg}_k \leftrightarrow \widehat{\fg}_{-k-2h^\vee}$),
  Virasoro
  \textup{(}same-family complementarity partner
  $\mathrm{Vir}_{26-c}$ at M/S-level, with the
  $\mathcal{W}_\infty$ realization retained as programme\textup{)},
  $\mathcal{W}_N$ algebras, Yangians.
  Genus-$g$ free energies computed through $g = 10$ with
  Bernoulli asymptotics.  Bar cohomology dimensions through
  degree~$10$.  These explicit results are new.

\item \emph{Derived oper space from the bar complex
  (Theorem~\ref{thm:oper-bar}).}
  At the critical level $k = -h^\vee$, the bar complex is uncurved
  and recovers the derived space of
  $\fg^\vee$-opers on the formal disc:
  $H^n(\barB(\widehat{\fg}_{-h^\vee}))
  \cong \Omega^n(\mathrm{Op}_{\fg^\vee}(D))$.
  The $H^0$ identification gives the Feigin--Frenkel
  center~\cite{Feigin-Frenkel}; the $H^1$ identification
  gives K\"ahler differentials of the oper space; and the
  full derived identification follows from the bar-Ext
  identification combined with the
  Frenkel--Teleman theorem~\cite{FT06}.
  This provides a vertex-algebraic construction of a
  central ingredient in the geometric Langlands programme,
  independent of the categorical approach of
  Gaitsgory et al.~\cite{GLC24} and
  Raskin~\cite{Raskin24}.

\item \emph{Chiral Hochschild cohomology
  (Theorems~\ref{thm:w-algebra-hochschild},
  \ref{thm:virasoro-hochschild},
  and~\ref{thm:critical-level-cohomology}).}
  For the principal $\Walg$-algebra $\Walg^k(\fg)$ at generic level:
  $\ChirHoch^*(\Walg^k(\fg)) \cong \mathbb{C}[\Theta_1, \ldots, \Theta_r]$
  with $\Theta_i$ in degree~$m_i + 1$ (exponents of~$\fg$).
  Rank~$1$ (Virasoro): strict period~$2$.
  Rank $r \geq 2$: quasi-period $\operatorname{lcm}(m_1+1, \ldots, m_r+1)$.
  At critical level $k = \critLevel$:
  exterior$\otimes$polynomial structure
  $\Lambda^*(P_1, \ldots, P_r) \otimes \mathbb{C}[\Theta_1, \ldots, \Theta_r]$
  (Fuks--Feigin--Tsygan).
  Functorial under Koszul duality via the Connes
  operator~$S$ (Theorem~\ref{thm:main-koszul-hoch},
  Corollary~\ref{cor:hochschild-cup-exchange}).
  This promotes the Hochschild results to a
  principal contribution~(Theorem~H).

\end{enumerate}

\begin{proposition}[GLZ as special case; \ClaimStatusProvedHere]\label{prop:glz-special-case}
The Gui--Li--Zeng quadratic duality is a special case of the $\chirAss$-duality of this monograph:
\[
\text{GLZ duality} = \chirAss\text{-duality} |_{\Einf\text{-chiral}} |_{\text{quadratic}}
\]
Restricting to $\Einf$-chiral algebras with quadratic presentations recovers their theory.
\end{proposition}

\begin{proof}
An $\Einf$-chiral algebra $\cA$ with quadratic presentation has a quadratic relation ideal $R \subset j_*j^*(\cA^{\boxtimes 2})$.  Its $\chirAss$ Koszul dual is determined by bar degrees $\leq 2$: the generators $V^*$ in $\bar{B}^1$ and the orthogonal complement of relations $R^\perp$ in $\bar{B}^2$ (the higher bar degrees, while nonzero, contribute no independent data for quadratic algebras).  The Koszul dual coalgebra $\bar{B}(\cA)$ is then determined by $R^\perp$, and the Koszul dual algebra $\cA^! = \bar{B}(\cA)^\vee$ is recovered by linear duality under the chiral pairing $j_*j^*(\cA^{\boxtimes 2}) \otimes j_*j^*((\cA^!)^{\boxtimes 2}) \to \omega_X$.  This is precisely the quadratic dual construction of GLZ.

For the $\Einf$ restriction: our $\chirAss$ bar complex uses the full associative chiral product $\mu\colon j_*j^*(\cA \boxtimes \cA) \to \Delta_!\cA$.  When $\cA$ is $\Einf$-chiral (i.e., the chiral bracket is skew-symmetric), the bar complex acquires a commutative coalgebra structure, and the Koszul dual is a Lie chiral algebra.  This recovers the FG framework as a special case (see Theorem~\ref{thm:fg-from-assch}).
\end{proof}

\begin{theorem}[FG duality from \texorpdfstring{$\chirAss$}{Ass-ch} self-duality; \ClaimStatusProvedHere]\label{thm:fg-from-assch}
The Francis--Gaitsgory $\chirCom$-$\chirLie$ duality is the associated graded of the $\chirAss$ self-duality under the PBW filtration.  Precisely: for a quadratic $\Einf$-chiral algebra $\cA$, the $\chirAss$ bar complex $\bar{B}_{\chirAss}(\cA)$ carries a filtration $F_\bullet$ (induced by the symmetrization degree of the $\Sigma$-action on $C_n(X)$) such that:
\begin{equation}\label{eq:fg-from-assch}
\mathrm{gr}^F \bar{B}_{\chirAss}(\cA) \;\cong\; \bar{B}_{\chirCom}(\mathrm{Sym}^{\mathrm{ch}}(\cA)) \;\cong\; \mathrm{CE}^{\chirLie}(\cA)
\end{equation}
and the associated spectral sequence degenerates at $E_1$ for Koszul algebras.
\end{theorem}

\begin{proof}
\emph{Step~1: PBW filtration.}
The chiral bar complex $\bar{B}^n_{\chirAss}(\cA)$ consists of sections on $\overline{C}_{n+1}(X)$ with coefficients in $\cA^{\boxtimes(n+1)} \otimes \Omega^n_{\log}$.  Define $F_p \bar{B}^n$ as the span of elements symmetric under at least $(n-p)$ transpositions of configuration space points.  This is the chiral analogue of the arity filtration on the classical associative bar complex (Loday--Vallette \cite{LV12}, \S9.3.2).

\emph{Step~2: Associated graded identification.}
On $\overline{C}_n(X)$, the $\Sigma_n$-representation decomposes as $j_*j^*(\cA^{\boxtimes n}) = \mathrm{Sym}^n_{\mathrm{ch}}(\cA) \oplus \bigwedge^2_{\mathrm{ch}}(\cA) \otimes \cdots$.  The PBW filtration separates these: $F_0/F_1 = \mathrm{Sym}^n_{\mathrm{ch}}(\cA)$ (the fully symmetric part) is the commutative chiral bar complex, and $F_1/F_2$ involves the Lie bracket (the antisymmetric part).  The associated graded $\mathrm{gr}^F \bar{B}_{\chirAss}$ is thus the Chevalley--Eilenberg complex for the Lie chiral algebra structure, which is $\bar{B}_{\chirCom}(\mathrm{Sym}^{\mathrm{ch}}(\cA))$ by the classical identification of the commutative bar complex with Chevalley--Eilenberg chains.

\emph{Step~3: Spectral sequence degeneration.}
For Koszul $\cA$, the spectral sequence $E_1^{p,q} = H^{p+q}(\mathrm{gr}^F_p \bar{B})$ degenerates at $E_1$.  To see this, note that the associated graded $\mathrm{gr}^F \bar{B}_{\chirAss}(\cA)$ is the Chevalley--Eilenberg complex (Step~2), and the Koszul property of $\cA$ as a chiral algebra implies that this commutative bar complex has cohomology concentrated on the diagonal $p + q = 0$ (i.e., $E_1^{p,q} = 0$ for $p + q \neq 0$).  Since $d_1\colon E_1^{p,q} \to E_1^{p+1,q}$ shifts the total degree $p+q$ by~$1$, it maps from the surviving diagonal to the zero region, hence $d_1 = 0$ and $E_2 = E_1$.  All higher differentials vanish for the same degree reason.  Therefore $H^*(\bar{B}_{\chirAss}(\cA)) \cong H^*(\mathrm{gr}^F \bar{B}_{\chirAss}(\cA))$, and the FG duality is recovered as the associated graded of our $\chirAss$ duality.

\emph{Step~4: Comparison with FG.}
The FG theorem states: for an $\Einf$-chiral algebra $\cA$, the chiral Koszul dual is a Lie chiral algebra $\cA^{!,\chirLie}$.  In the present framework: the Koszul dual coalgebra $\bar{B}_{\chirAss}(\cA)$ is an $\Eone$-chiral coalgebra, and the Koszul dual algebra $\cA^{!,\chirAss} = \bar{B}_{\chirAss}(\cA)^\vee$ is an $\Eone$-chiral algebra; its associated graded under PBW is $\cA^{!,\chirLie} = \bar{B}_{\chirCom}(\cA)^\vee$.  (The cobar $\Omega(\bar{B}_{\chirAss}(\cA))$ recovers~$\cA$ itself by bar-cobar inversion, not~$\cA^!$.)  The FG duality is recovered.
\end{proof}

\begin{remark}[Key comparison]\label{rem:ayala-francis-comparison}
The Ayala--Francis framework~\cite{AF15, AF20} establishes non-abelian
Poincar\'{e} duality as an equivalence between factorization homology and
factorization cohomology for framed $\mathsf{E}_n$-algebras on $n$-manifolds.
Our Theorem~A (Theorem~\ref{thm:bar-cobar-isomorphism-main}) is the
specialization to dimension~$n = 2$ (complex curves viewed as real
oriented surfaces) with the additional
structure of:
\begin{enumerate}[label=(\roman*)]
\item \emph{Holomorphic enrichment.}
  The AF framework works with topological $\mathsf{E}_n$-algebras and
  locally constant cosheaves.  We work with holomorphic chiral algebras
  ($\mathcal{D}$-modules on curves), where the Fulton--MacPherson
  compactification provides the correct holomorphic compactification
  (not just the topological one).  The bar differential involves
  residues of meromorphic forms, not just topological data.
\item \emph{Explicit formulas.}
  The AF duality is an $\infty$-categorical equivalence; our realization
  via configuration space integrals gives explicit chain-level formulas
  for the bar differential (residues at collision divisors) and the
  cobar differential (inclusions of FM strata).
\item \emph{Higher genus.}
  For genus $g \geq 1$ curves, the configuration space
  $C_n(\Sigma_g)$ acquires additional topology ($H^1(\Sigma_g)$),
  and the bar complex inherits new generators from the handle
  cycles.  The AF framework predicts this (factorization homology
  on non-simply-connected manifolds), but the explicit identification
  with moduli space integrals and the Hodge bundle (Theorem~C)
  is new.
\end{enumerate}
\end{remark}

\begin{remark}[Key comparison]\label{rem:feigin-frenkel-comparison}
The Feigin--Frenkel involution $k \mapsto -k - 2h^\vee$ on affine
Kac--Moody levels~\cite{Feigin-Frenkel} is, in this monograph's framework,
a \emph{consequence} of chiral Koszul duality: the bar-cobar adjunction on
$\overline{C}_n(X)$ produces $\widehat{\fg}_{-k-2h^\vee}$ as the Koszul
dual of $\widehat{\fg}_k$ (Theorem~\ref{thm:universal-kac-moody-koszul}).
The geometric mechanism is Verdier duality on the Fulton--MacPherson
compactification, which exchanges the chiral bracket
$\mu\colon j_*j^*(\cA \boxtimes \cA) \to \Delta_!\cA$
with the co-bracket on the Koszul dual.

At the critical level $k = -h^\vee$, the Feigin--Frenkel center
$\mathfrak{z}(\widehat{\fg}_{-h^\vee})$ is recovered as the degree-zero
homology of the (uncurved) bar complex: $\kappa = 0$ at $k = -h^\vee$,
so the bar complex is a genuine cochain complex, and
$H^0(\bar{B}(\widehat{\fg}_{-h^\vee})) \cong \mathfrak{z}(\widehat{\fg}_{-h^\vee})$
(Proposition~\ref{prop:km-generic-acyclicity}).

The extensions beyond FF are:
\begin{enumerate}[label=(\roman*)]
\item The complementarity formula $c(\widehat{\fg}_k) + c(\widehat{\fg}_{-k-2h^\vee}) = 2d$
  (Theorem~\ref{thm:central-charge-complementarity}), which shows the
  central charge sum is a root datum invariant, independent of the level.
\item The genus-$g$ obstruction formula $\mathrm{obs}_g = \kappa \cdot \lambda_g$
  (Theorem~\ref{thm:genus-universality}), which extends the FF involution
  to all genera via the Hodge bundle on $\overline{\mathcal{M}}_g$.
\item The Virasoro and $\mathcal{W}_N$ generalizations, where the
  Drinfeld--Sokolov parametrization introduces nonlinear level shifts
  (e.g., $c \mapsto 26 - c$ for Virasoro) that do not arise from a linear
  involution on levels.
\end{enumerate}
\end{remark}

\begin{remark}[Key comparison]\label{rem:cg-comparison}
The Costello--Gwilliam factorization algebra framework~\cite{CG17}
relates to ours through the Beilinson--Drinfeld equivalence between
chiral algebras and factorization algebras on curves (BD, Chapter~3).
The precise comparison:
\begin{enumerate}[label=(\roman*)]
\item \emph{Factorization homology.}
  The CG factorization homology $\int_X \cA$ computes the same object
  as our chiral homology $H_*(\bar{B}^{\mathrm{ch}}(\cA))$.  Our explicit
  realization via configuration space integrals provides chain-level
  formulas that the CG framework treats abstractly.
\item \emph{Koszul duality.}
  The CG framework includes Koszul duality for factorization algebras
  (CG, Vol.~2, Chapter~5), specialized to holomorphic factorization
  algebras on curves.  Their Koszul duality, restricted to this setting,
  recovers our chiral Koszul duality.  Our contribution is the explicit
  identification of the Koszul dual for all standard families
  (Kac--Moody, Virasoro, $\mathcal{W}_N$, etc.) via the bar-cobar
  adjunction on Fulton--MacPherson spaces.
\item \emph{Quantum corrections.}
  The CG perturbative quantization framework produces $A_\infty$
  structures from Feynman diagram expansions (CG, Vol.~2, Chapter~3).
  Our curved $A_\infty$ structure on the bar complex is the
  chiral-algebraic counterpart: the curvature $m_0 = \kappa \cdot
  \mathbf{1}$ arises from the highest-order OPE pole (quartic for
  Virasoro, double for Heisenberg), not from a Feynman diagram sum.
  The genus universality theorem shows that these two perspectives
  produce the same genus-$g$ obstruction class.
\item \emph{BV-BRST.}
  The CG BV formalism produces the same BRST complex as our bar
  construction when applied to the chiral algebra of a 2d holomorphic
  field theory (Chapter~\ref{ch:bv-brst}).  The identification
  $\bar{B}(\cA) \simeq C^{\mathrm{BRST}}(\cA)$ connects the
  algebraic (bar complex) and physical (BRST cohomology) perspectives.
\item \emph{Monoidal bar-cobar.}
  Booth--Lazarev~\cite{BL24} construct monoidal model structures on
  categories of coalgebras, providing the homotopical infrastructure
  for the bar-cobar adjunction to be a Quillen equivalence in the
  monoidal setting.  This is realized by the fusion product preservation theorem
  (Theorem~\ref{thm:fusion-bar-cobar}): the bar construction
  is a monoidal functor from the tensor category of chiral
  algebra modules to the comodule category over the bar coalgebra.
\end{enumerate}
\end{remark}

\begin{remark}[Convention for Koszul dual isomorphisms]
In Loday--Vallette \cite{LV12}, \S7.1--7.2, the Koszul dual includes the sign representation: $\Ass^! = \Ass \otimes \mathrm{sgn}$ and $\Com^! = \Lie\{1\}$ (with operadic suspension). In this manuscript, we absorb the sign twist into the bar desuspension: our bar construction uses the shifted generators $s^{-1}\mathcal{A}$ whose sign already accounts for the $\mathrm{sgn}$-twist. Concretely, our convention $\Ass^! \cong \Ass$ means that the dual cooperad is isomorphic to $\Ass$ as a \emph{graded} cooperad after this shift. All bar-cobar computations in the text use the desuspended convention consistently, so no sign errors propagate. See Appendix~\ref{app:signs} for the explicit translation between conventions.
\end{remark}

\begin{remark}[Chiral enhancement]
We lift the entire LV framework to the chiral setting, replacing vector spaces with D-modules and tensor products with chiral tensor products.
\end{remark}

\begin{remark}[Synthesis]\label{rem:concordance-synthesis}
The common thread running through the comparisons above is that each
framework---Beilinson--Drinfeld's chiral algebras, Francis--Gaitsgory's
chiral Koszul duality, Costello--Gwilliam's factorization algebras,
Loday--Vallette's operadic Koszul theory---captures one projection of a
single underlying phenomenon: \emph{the duality between commutative and
Lie structures, mediated by configuration spaces}.  The contribution of
the present monograph is the explicit chain-level realization of this
duality via configuration space integrals, which makes all projections
simultaneously computable.  The bar complex $\bar{B}^{\mathrm{ch}}(\cA)$
is the meeting point: it is BD's chiral homology, FG's Koszul dual
coalgebra, CG's factorization homology on the Ran space, and LV's bar
construction---all incarnated as a single cochain complex with an
explicitly computable differential.
\end{remark}

\section{Conjecture stratification}
\label{sec:conjecture-stratification}

The ${\sim}150$ conjectured claims in this monograph are not peers.
We stratify them into three tiers by the nature and difficulty of
the missing input.

\smallskip\noindent\textbf{Tier~1 --- Foundational.}
Conjectures whose resolution would extend the proved theorems
to new structural levels.  These require substantial new
mathematics:
\begin{itemize}
\item Coderived Ran formalism for non-Koszul chiral algebras:
  the categorical extension of Theorem~B off the Koszul locus.
\item Virasoro and $\mathcal{W}_{\infty}$ Koszul duals: explicit
  identification of the dual chiral algebras.
\end{itemize}

\smallskip
\noindent
The former higher-genus PBW concentration conjecture is now resolved
for the standard finite-type interacting families by
Theorems~\ref{thm:pbw-allgenera-km},
\ref{thm:pbw-allgenera-virasoro},
and~\ref{thm:pbw-allgenera-principal-w}.

\smallskip\noindent\textbf{Tier~2 --- Consequence.}
Conjectures that follow from known structures by explicit
computation or extension of existing proofs.  Each is a multi-week
project but requires no fundamentally new ideas:
\begin{itemize}
\item Derived Drinfeld--Kohno ladder beyond the theorematic
  evaluation-generated core:
  ordered $\Eone$-factorization with $R$-matrix inversion,
  factorization-level Kazhdan transport, and dg-shifted comparison.
\item Kazhdan--Lusztig from bar-cobar: the remaining stages
  beyond the proved chain-level framework.
\item Shifted-symplectic enhancement: upgrading the Lagrangian
  complementarity to a shifted symplectic structure on the
  moduli of Koszul pairs.
\item Fusion preservation under Koszul duality.
\end{itemize}

\smallskip\noindent\textbf{Tier~3 --- Physics horizon.}
Physics-facing conjectures that require input from quantum field
theory or string theory.  These are correctly conjectural within
a mathematics monograph:
\begin{itemize}
\item BV = bar identification for general field theories
  (beyond the holomorphic setting).
\item Path integral $=$ bar-cobar: the full Feynman--Koszul
  dictionary at all genera.
\item Holographic Koszul duality: AdS/CFT via the bar-cobar template.
\item $\mathsf{E}_n$ extension to higher-dimensional field theories.
\item AGT comparison via bar-cobar/Koszul duality.
\end{itemize}

\begin{proposition}[Standard-tower MC5 reduction after MC3 and realized

\label{prop:standard-tower-mc5-reduction}
Consider the standard principal

itself: the remaining MC5 content is exactly the genuswise comparison
between the realized H-level targets and the corresponding BV/BRST
complexes, compatibly with truncation, Verdier duality, and clutching.

\emph{This is a dependency theorem, not a closure theorem for the
packet identities themselves.  Its hypotheses are exactly the realized

stabilized chain
\[
\text{MC3 enlargement}
\Longrightarrow
\text{MC4 realized H-level package}
\Longrightarrow
\text{MC5 BV/BRST comparison}.
\]
Its scope is only the standard principal and RTT towers; it does not
assert the same closure principle for non-principal orbit families or

\begin{proof}
Assumption~\textup{(ii)} says that on the standard principal
\texorpdfstring{$W_\infty$}{W_infty} side the MC4 comparison has
already been reduced to exact stagewise packet identities on the finite
detection system~$\mathcal{I}_N$.  The additional

Corollary~\ref{cor:winfty-stage4-residue-four-channel}; it does not add
a further tower-wide algebraic obstruction once the packet identities
themselves are assumed.

side the H-level dg-shifted target is already realized by
Proposition~\ref{prop:yangian-typea-realization-criterion}, and that
the remaining completed categorical packet closes by either
Corollary~\ref{cor:yangian-typea-realization-plus-compacts} or
Corollary~\ref{cor:yangian-typea-realization-plus-fundamental-packet}
or, on the spectral vector-line locus,
Corollary~\ref{cor:yangian-typea-realization-plus-dg-packet}.
Thus, after the MC3 enlargement of assumption~\textup{(i)}, no new
infinite-tower algebraic input remains on the standard Yangian tower.

Therefore the standard principal \texorpdfstring{$W_\infty$}{W_infty}
and standard type-$A$ Yangian towers contribute no remaining
infinite-generator algebraic obstruction beyond the realized MC3/MC4
packages.  The only further content is the genuswise BV/BRST/bar
comparison supplied by assumption~\textup{(iv)}, namely the passage
from the realized H-level targets to the corresponding BV/BRST
complexes in the sense of
Theorem~\ref{thm:algebraic-string-dictionary} and
Theorem~\ref{thm:master-theta}, compatibly with truncation, Verdier
duality, and clutching.  This is exactly the claimed MC5 reduction.
\end{proof}

co-contra correspondence for conilpotent coalgebras
(Theorem~\ref{thm:fact-co-contra-general}).
\emph{Remaining gap}: off-Koszul bar-cobar inversion
(Conjecture~\ref{conj:off-koszul-ran-inversion}).

\smallskip\noindent
\textbf{H2\,: $\mathcal{W}_\infty$ H-level realization (MC4).}
\textbf{Resolved} by Theorem~\ref{thm:winfty-factorization-kd}: the
factorization target~$\mathcal{W}_\infty^{\mathrm{fact}}$ is
constructed via $\Einf$-sectorwise convergence at each finite
stage, with DS--KD intertwining
(Theorem~\ref{thm:ds-koszul-intertwine}) identifying
$(W_N^k)^! \simeq \widehat{\mathfrak{sl}}_{N,-k-2N}$ and
Mittag--Leffler
(Corollary~\ref{cor:winfty-standard-mc4-package}) controlling the
inverse limit.  The residue identities
$\mathsf{C}^{\mathrm{res}} = \mathsf{C}^{\mathrm{DS}}$ follow
from the chain-level content of the intertwining theorem.
The remaining $\mathcal{W}_\infty$ frontier is the large-$N$
coupling identification $\lambda \mapsto 1 - \lambda$
(Conjecture~\ref{conj:w-infty-bar}).

\smallskip\noindent
\textbf{H3\,: Thick generation beyond $\operatorname{Rep}_{\mathrm{fd}}$ (MC3).}
Thick generation by evaluation modules is proved for
$\operatorname{Rep}_{\mathrm{fd}}$
at all~$N$ in type~A
(Theorem~\ref{thm:eval-core-identification}) and for
$\mathcal{O}_{\mathrm{poly}}$ at all~$N$
(Remark~\ref{rem:catO-thick-generation-verification}).
The gap is extension to non-polynomial infinite-dimensional modules
(prefundamental, asymptotic) in the completed/coderived setting.
\emph{Strategy}: combine the BGG criterion
(Proposition~\ref{prop:bgg-criterion}) with the KL reduction
(Proposition~\ref{prop:thick-gen-projective}).

\smallskip\noindent
\textbf{H4\,: Baxter derived lift (MC3 path~\textup{(i)}).}
$K_0$-level TQ relations are verified at $498$~tests for
$\mathfrak{sl}_2$ and~$\mathfrak{sl}_3$
(Conjecture~\ref{conj:baxter-exact-triangles}), including
derived-level lifts to distinguished triangles and the octahedral
axiom for iterated TQ ($226$~tests).
\emph{Gap}: the categorical lift from short exact sequences in the
heart of the $t$-structure on
$D^b(\mathcal{O}^{\mathrm{sh}}_{\leq 0})$ to natural
transformations of exact functors.

\smallskip\noindent
\textbf{H5\,: Prefundamental generation (MC3 path~\textup{(ii)}).}
Hernandez--Jimbo--Zhang negative prefundamental modules~$L^-_a$
for $Y(\mathfrak{sl}_2)$ are now implemented, with character
$\operatorname{ch}(L^-_a)(q) = \prod_{n \geq 1} (1 - q^{-2n})^{-1}$
and weight multiplicities $\dim(L^-_a)_{-2k} = p(k)$
($186$~tests, including QQ-system Wronskian checks).
\emph{Gap}: thick generation of
$\mathcal{O}^{\mathrm{sh}}_{\leq 0}$ by
$\{V_n(a)\} \cup \{L^-_a\}$ requires verifying that every
standard module admits a finite resolution by
evaluation$\oplus$prefundamental summands.

\smallskip\noindent
\textbf{H6\,: Pro-Weyl M-level lift (MC3 path~\textup{(iii)}).}
Shadow-level convergence of the pro-Weyl system is proved:
Mittag-Leffler condition holds, $R^1\!\varprojlim = 0$, and
chain-level Ext computations verify the derived inverse limit
($344$~tests).  \emph{Gap}: construct a dg-algebra or
$A_\infty$ model for the pro-Weyl limit
$M(\Psi) \simeq R\!\varprojlim W_m$ and prove quasi-isomorphism
with the shifted Yangian Verma.

\smallskip\noindent
\textbf{H7\,: Holomorphic-to-real propagator extension (MC5).}
The genus-$0$ BRST/bar identification
(Theorem~\ref{thm:brst-bar-genus0}) uses holomorphic propagators
on~$\mathbb{P}^1$.  Extension to $g \geq 1$ requires passing from
holomorphic propagators on $\Sigma_g$ to real propagators
(\`a~la Kontsevich) with Costello's renormalization data.
The first missing step is the disk-local packet of
Conjecture~\ref{conj:disk-local-perturbative-fm}: identify the local
perturbative BRST brackets with the bar residue operations on
$C_2$ and the three codimension-$1$ strata of~$\overline{C}_3$.
By Proposition~\ref{prop:disk-local-binary-ternary-reduction}, this is
already the full independent local comparison problem.
On the anomaly-free genus-$0$ BRST/bar locus of
Corollary~\ref{cor:disk-local-ternary-on-brstbar-locus}, the binary
part is already theorematic, so the remaining local check there is the
ternary comparison on~$\overline{C}_3$.
Once that ternary comparison is compactified to
$\overline{M}_{0,4} \cong \bP^1$ with logarithmic boundary behavior,
Proposition~\ref{prop:compactified-ternary-two-channel} reduces it
further to any two of the three boundary channels; on the standard
$0,1,\infty$ chart,
Corollary~\ref{cor:genus0-standard-chart-two-residues} makes this the
explicit pair of residues at $0$ and~$1$, equivalently the two scalar
coefficients of $d\log t$ and $d\log(1-t)$ by
Corollary~\ref{cor:genus0-two-coefficient-packet}, i.e. the named
coefficient identities
\(\mathsf{a}_{\mathrm{pert}}=\mathsf{a}_{\mathrm{bar}}\) and
\(\mathsf{b}_{\mathrm{pert}}=\mathsf{b}_{\mathrm{bar}}\).
Once that packet is compactified and matched with the FM Stokes
relations, the higher-genus propagator problem becomes an analytic
genus-lifting problem rather than a local dictionary problem.
This is downstream of MC1--MC4.

\smallskip\noindent
\textbf{H8\,: DS transport of $T_{\mathrm{br}}$ (spectral branch).}
Drinfeld--Sokolov reduction preserves the discriminant
$\Delta_{\cA}(x) = \det(1 - x \, T_{\mathrm{rec}})$
(Theorem~\ref{thm:ds-bar-gf-discriminant}).  For rank~$1$
($\mathfrak{sl}_2$), preservation of the full transfer operator
$T_{\mathrm{br}}$ is automatic.  \emph{Gap}: for rank $\geq 2$
($\mathfrak{sl}_3$ onward), is $T_{\mathrm{br}}$ itself
DS-invariant (up to similarity), or only its determinant?
Computational verification at $\mathfrak{sl}_3$
($154$~tests) confirms that the DS-invariant sub-discriminant
$(1 - 3x - x^2)$ is preserved, while the dominant eigenvalue
changes ($8 \to 1$), so full $T_{\mathrm{br}}$~similarity fails;
see Conjecture~\ref{conj:ds-spectral-branch-preservation}
for the refined formulation.
\end{remark}

\subsection{Homotopy templates for conjectures}
\label{subsec:homotopy-templates}
\index{conjectures!homotopy templates|textbf}

The stratification above classifies conjectures by \emph{difficulty}.
The following classification is orthogonal: it sorts conjectures by
\emph{homotopy type}---what kind of mathematical object each conjecture
asserts the existence or properties of.  Every conjecture in this
monograph fits one of seven templates
(Convention~\ref{conv:hms-levels}).

\begin{convention}[Homotopy templates]\label{conv:homotopy-templates}
\index{homotopy templates|textbf}
\begin{enumerate}[label=\textbf{Type \Roman*.}, leftmargin=3em]
\item \emph{Existence of a homotopy object.}
  A filtered cyclic $L_\infty$-algebra, MC element,
  or factorization (co)algebra is asserted to exist with specified
  compatibility.  \emph{Example:}
  $\Theta_{\cA}$ (Theorem~\ref{thm:master-theta}).

\item \emph{Equivalence of homotopy categories.}
  Two $\infty$-categories (or stable/E$_1$-monoidal variants)
  are equivalent; current chain models give a presentation on a
  dense sublocus.
  \emph{Examples:} full coderived bar-cobar
  (Conjecture~\ref{conj:master-dk-kl}),
  full DK, KL from bar-cobar.

\item \emph{Shifted-symplectic / Lagrangian enhancement.}
  A pairing already constructed at the model level upgrades to an
  $n$-shifted symplectic structure on a formal moduli problem.
  \emph{Examples:} complementarity upgrades, moduli of Koszul pairs.

\item \emph{Trace / index statement.}
  A scalar invariant is the image of a universal deformation object
  under a trace or pushforward functor.
  \emph{Examples:} genus generating series ($\hat{A}$-genus),
  GRR for modular deformations.

\item \emph{Periodicity / autoequivalence.}
  A finite homotopy action, graded autoequivalence, or periodic
  filtration operator exists on a homotopy object; numerical periods
  are its shadow.
  \emph{Examples:} reflected modular periodicity,
  quantum group periodicity at roots of unity.

\item \emph{Filtered / pro-object comparison.}
  A theorematic completed model exists stagewise, and the remaining
  problem is to identify the intended H-level target by compatible
  finite quotients and finite-detection criteria.
  \emph{Examples:} $\mathcal{W}_\infty$ duals
  (Conjecture~\ref{conj:master-infinite-generator}),
  Yangian towers.

\item \emph{Physics-dictionary conjecture.}
  Two deformation problems---one algebraic, one
  field-theoretic---are controlled by quasi-isomorphic dg~Lie or
  $L_\infty$-algebras.
  \emph{Examples:} BRST $=$ bar
  (Conjecture~\ref{conj:master-bv-brst}),
  BV $=$ bar-cobar, path integral $=$ factorization.
\end{enumerate}
\end{convention}

\begin{remark}[Using the templates]\label{rem:template-usage}
Each individual conjecture in the monograph carries a scope remark
identifying its tier (Foundational / Consequence / Physics).
The homotopy template provides the complementary structural
information: \emph{what kind of homotopy-theoretic statement}
the conjecture makes.  The resolved PBW theorem and the four remaining
master conjectures above are respectively of Types~I, II, II, VI,
and~VII.
\end{remark}

\subsection{Higher-dimensional Koszul duality}
\label{subsec:higher-dim-kd}
\index{Koszul duality!higher-dimensional}

\begin{theorem}[\texorpdfstring{$\mathsf{E}_n$}{En} Koszul duality via configuration
space integrals; \ClaimStatusProvedElsewhere]\label{thm:en-koszul-duality-conc}
\index{En-algebra@$\mathsf{E}_n$-algebra!Koszul duality}
For an $\mathsf{E}_n$-algebra $A$ on an oriented $n$-manifold~$M$,
the bar-cobar adjunction admits an explicit realization via
configuration space integrals on the Fulton--MacPherson
compactification $\overline{C}_k(M)$.  All four components
are proved:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \textup{[\ClaimStatusProvedElsewhere; Idrissi~\cite{Idrissi22}]}
  The bar differential is given by integration along collision
  divisors $D_I \subset \overline{C}_k(M)$, with the propagator
  determined by a Green's form $G \in \Omega^{n-1}(M \times M
  \setminus \Delta)$.  Poincar\'e duality on configuration spaces
  ensures $d^2 = 0$ for all~$n$.
\item \textup{[\ClaimStatusProvedHere]}
  For $n = 2$ with complex structure \textup{(}complex curves viewed as
  real oriented surfaces\textup{)}, this recovers the topological
  shadow of the chiral bar-cobar duality of
  Theorem~\textup{\ref{thm:bar-cobar-isomorphism-main}}.
\item \textup{[\ClaimStatusProvedElsewhere;
  Totaro~\cite{Totaro96}, Lambrechts--Voli\'c~\cite{LambrechtsVolic14},
  Knudsen~\cite{Knudsen18}]}
  For $n \geq 3$, the Totaro/graph-complex relations govern the
  higher-dimensional chain models:
  the Arnold relations are replaced by Totaro's presentation of
  $H^*(\overline{C}_k(\mathbb{R}^n))$, and the bar cohomology is
  governed by $\mathsf{E}_n$-operadic Koszul duality.
\item \textup{[\ClaimStatusProvedHere]}
  The Poincar\'e--Koszul duality of Ayala--Francis
  \textup{(}\cite{AF15}, Theorem~7.8\textup{)} is recovered as the
  $\infty$-categorical shadow of this explicit chain-level
  construction.
\end{enumerate}
\end{theorem}

\begin{proposition}[\texorpdfstring{$n = 2$}{n = 2} recovery and AF comparison;
\ClaimStatusProvedHere]\label{prop:en-n2-recovery}\label{prop:en-n1-recovery}
Parts \textup{(ii)} and \textup{(iv)} of
Theorem~\textup{\ref{thm:en-koszul-duality}} hold:
\begin{enumerate}[label=\textup{(\alph*)}]
\item At $n = 2$, on a complex curve~$X$ viewed as a real oriented
  surface, the FM propagator class admits the holomorphic
  representative $\omega_{ij} = d\log(z_i - z_j)$
  recovers the bar differential of
  Definition~\textup{\ref{def:bar-differential-complete}},
  and the Arnold relations
  \textup{(}Appendix~\textup{\ref{app:arnold-relations})} give $d^2 = 0$.
\item The resulting bar-cobar adjunction is
  Theorem~\textup{\ref{thm:bar-cobar-isomorphism-main}},
  which is the chain-level refinement of the $\infty$-categorical
  Poincar\'e--Koszul duality of Ayala--Francis~\cite{AF15}
  restricted to $n = 2$
  \textup{(}Remark~\textup{\ref{rem:ayala-francis-comparison})}.
\end{enumerate}
\end{proposition}

\begin{proof}
Part (a) is the content of \S\ref{sec:bar-cobar}: the
bar differential is $d = \sum_{i < j} \mathrm{Res}_{D_{ij}}(\omega_{ij} \wedge -)$
with $\omega_{ij}$ the FM propagator on the curve~$X$.  Arnold relations
ensure $d^2 = 0$ (Corollary~\ref{cor:bar-d-squared-zero-arnold}).  Part~(b):
by Remark~\ref{rem:ayala-francis-comparison}, the AF equivalence
$\int_M A \simeq \int_M A^!$ specializes at $n = 2$ to our
bar-cobar adjunction, with the factorization homology integral
computed via FM configuration space integrals.
\end{proof}

\begin{remark}[Proof status]\label{rem:en-scope}
All four parts are proved.  Part~(i) by
Idrissi~\cite{Idrissi22}, who establishes that Poincar\'e duality on
configuration spaces ensures $d^2 = 0$ for the propagator-based
differential on $\mathsf{E}_n$-algebras for all~$n$.  Parts~(ii)
and~(iv) by Proposition~\ref{prop:en-n2-recovery} below.  Part~(iii)
follows from a chain of three results: (a)~Totaro's
presentation~\cite{Totaro96} of
$H^*(\overline{C}_k(\mathbb{R}^n))$; (b)~$\mathsf{E}_n$ formality
(Lambrechts--Voli\'c~\cite{LambrechtsVolic14}: the operadic
quasi-isomorphism $C_*(\mathsf{E}_n) \xrightarrow{\sim}
H_*(\mathsf{E}_n)$ transfers the bar-cobar adjunction to the
homological level); (c)~Knudsen's $\mathsf{E}_n$ Koszul
duality~\cite{Knudsen18}, which provides PBW and Koszul duality
theorems for higher enveloping algebras $U_n(\mathfrak{g})$.
Together, these establish that the bar cohomology of any
$\mathsf{E}_n$-algebra is governed by $\mathsf{E}_n$-operadic Koszul
duality, with the Totaro presentation providing the explicit
generators and relations.

\emph{What remains as a computational refinement} is writing down
explicit closed-form expressions for the bar differential in terms of
Totaro generators at each~$n \geq 3$---the propagator and FM
compactification exist, and $d^2 = 0$ is proved, but the individual
formulas (analogous to the Arnold-based computations at $n = 2$) have
not been tabulated.  This is a concrete calculation, not a
theoretical gap.
(Contributing to Conjecture~\ref{conj:master-bv-brst}.)
\end{remark}

\begin{remark}[Proof assembly for part~(iii)]\label{rem:en-proof-strategy}
\index{En-algebra@$\mathsf{E}_n$-algebra!Koszul duality!proof strategy}
Part~(iii) of Theorem~\ref{thm:en-koszul-duality} is proved by
assembling the following known ingredients into the abstract
chain-level construction.

\emph{Step~1: Formality.}
The $\mathsf{E}_n$ operad is formal over $\mathbb{Q}$ for all
$n \geq 2$: Kontsevich~\cite{Kon99} and Tamarkin~\cite{Tamarkin00}
proved this for $n = 2$;
Lambrechts--Voli\'c~\cite{LambrechtsVolic14} established the
general case via configuration space integrals on semi-algebraic
models for $\overline{C}_k(\mathbb{R}^n)$.  Formality provides a
quasi-isomorphism $C_*(\mathsf{E}_n) \xrightarrow{\sim}
H_*(\mathsf{E}_n)$ of operads, so the bar-cobar adjunction can be
studied at the level of homology.

\emph{Step~2: Totaro's presentation.}
For $n \geq 3$, the cohomology ring
$H^*(\mathrm{Conf}_k(\mathbb{R}^n); \mathbb{Q})$ is generated by
degree-$(n-1)$ classes $\omega_{ij}$ ($1 \leq i < j \leq k$)
subject to: (a)~$\omega_{ij} = (-1)^n \omega_{ji}$;
(b)~$\omega_{ij}^2 = 0$; (c)~the generalized Arnold relation
$\omega_{ij}\omega_{jk} + \omega_{jk}\omega_{ki} +
\omega_{ki}\omega_{ij} = 0$~\cite{Totaro96,Coh76}.  These replace
the classical Arnold relations (which govern $n = 1, 2$) with
their higher-dimensional analogues; the key difference is that the
generators sit in odd degree when $n$ is even and even degree when
$n$ is odd.

\emph{Step~3: Graph complex connection.}
Kontsevich's graph complex $\mathsf{GC}_n$ computes
$H^*(\mathrm{Aut}(\mathsf{E}_n))$, and the bar differential on
$\mathsf{E}_n$-algebras factors through $\mathsf{GC}_n$.
The passage from $n = 2$ to $n \geq 3$ is governed by the
transition from the Lie graph complex ($n = 1$) through the
associative graph complex ($n = 2$) to the commutative graph complex
($n \to \infty$); the obstruction theory at each stage is controlled
by the parity of the generators.

\emph{Step~4: Synthesis.}
The abstract construction is complete: formality (Step~1) transfers
the bar-cobar adjunction to homology; the Totaro presentation
(Step~2) provides explicit generators; Knudsen's higher enveloping
algebras~\cite{Knudsen18} identify the bar cohomology with
$\mathsf{E}_n$-operadic Koszul duality; and Idrissi's
Part~(i)~\cite{Idrissi22} ensures $d^2 = 0$ at the chain level.
What remains as a \emph{computational refinement} is writing down
the explicit bar differential formula in Totaro generators at
each~$n \geq 3$---the natural candidate propagator is the
Lambrechts--Voli\'c propagator~\cite{LambrechtsVolic14}, and the
parity of the generators (odd for even~$n$) produces sign patterns
that must be tracked through the differential.
\end{remark}

\subsection{Vassiliev invariants from Feynman transform}
\label{subsec:vassiliev}
\index{Vassiliev invariants!from Feynman transform}

\begin{conjecture}[Vassiliev invariants from bar complex;
\ClaimStatusConjectured]\label{conj:vassiliev-bar}
\index{Kontsevich integral!bar complex}
For $\cA = \widehat{\fg}_k$, the Feynman transform identification
$\barB^{\mathrm{full}}(\cA) \simeq \mathrm{FT}_{\mathcal{M}\mathrm{od}}(\cA)$
\textup{(}Theorem~\textup{\ref{thm:prism-higher-genus}(ii))}
produces universal Vassiliev invariants when specialized to
configuration spaces of circles:
\begin{enumerate}[label=\textup{(\roman*)}]
\item The restriction
  $\overline{C}_n(X) \rightsquigarrow C_n(S^1)$
  \textup{(}from complex curve to real circle\textup{)} sends the
  holomorphic propagator $\omega_{ij} = d\log(z_i - z_j)$ to the
  Kontsevich propagator $\omega_{ij}^K = \frac{1}{2\pi} d\arg(t_i - t_j)$.
\item Under this restriction, the bar complex weight systems
  $W_\Gamma\colon \fg^{\otimes n} \to \mathbb{C}$ match the
  classical weight systems of Vassiliev theory, with the
  Lie algebra $\fg$ providing the weight system as in
  Bar-Natan~\cite{BarNatan95}.
\item The genus expansion
  $\sum_g g_s^{2g-2} \barB^{(g)}$ specializes to the
  loop expansion of the Kontsevich integral.
\end{enumerate}
\end{conjecture}

\begin{proposition}[Genus-\texorpdfstring{$0$}{0} weight systems from bar complex;
\ClaimStatusProvedHere]\label{prop:vassiliev-genus0}
\index{weight system!from bar complex}
At genus~$0$, the bar complex of $\widehat{\fg}_k$ produces
classical weight systems:
\begin{enumerate}[label=\textup{(\roman*)}]
\item For each chord diagram $\Gamma$ with $n$ chords, the
  degree-$n$ bar differential $d^{(0)}\colon \barB^n \to \barB^{n-1}$
  induces a linear functional
  $W_\Gamma\colon \fg^{\otimes 2n} \to \bC$ via the residue
  pairing of the FM propagator on $\overline{C}_{2n}(\mathbb{P}^1)$.
\item These coincide with the Bar-Natan weight systems of the
  Lie algebra~$\fg$ with the Killing form at level~$k$: $W_\Gamma$
  assigns to each chord the contraction $\kappa^{ab}$ and to each
  vertex the structure constants~$f^a_{bc}$.
\end{enumerate}
\end{proposition}

\begin{proof}
At genus~$0$, the bar differential is given by residues of
the propagator $\omega_{ij} = d\log(z_i - z_j)$ along collision
divisors $D_I \subset \overline{C}_n(\mathbb{P}^1)$
(Definition~\ref{def:bar-differential-complete}).  For a chord
diagram $\Gamma$ with chord $(i,j)$, the residue
$\mathrm{Res}_{z_i \to z_j} \omega_{ij}$ extracts the
OPE coefficient $J^a(z_i) J^b(z_j) \sim k \kappa^{ab}/(z_i-z_j)^2
+ f^{ab}_c J^c/(z_i-z_j)$: the double pole contributes the
Killing contraction $k\kappa^{ab}$ and the simple pole the
structure constants~$f^{ab}_c$.  These are precisely the
assignments in Bar-Natan's construction of weight systems from
Lie algebras~\cite{BarNatan95}.
\end{proof}

\begin{remark}[Scope]\label{rem:vassiliev-scope}
Proposition~\ref{prop:vassiliev-genus0} establishes the genus-$0$
weight system extraction.  The Feynman transform of $\mathrm{Com}$
recovers the full graph complex computing Vassiliev invariants
(Getzler--Kapranov \cite{GeK98}), and our identification
$\barB^{\mathrm{full}} \simeq \mathrm{FT}$
(Theorem~\ref{thm:prism-higher-genus}) extends this to Koszul
chiral algebras.  The gap at higher genus is the passage from
complex curves to real $1$-manifolds: our propagators are
holomorphic, while Kontsevich's are real.  Comparing requires an
analytic continuation argument from holomorphic to topological
Feynman transforms.
The correct bridge between holomorphic and topological frameworks is Chern--Simons gauge equivalence, not merely restriction from complex curves to $S^1$: Cattaneo--Mnev~\cite{CattaneoMnev10} establish that configuration space integrals on $3$-manifolds produce Vassiliev invariants via BV integration, and the comparison map involves the CS partition function (not a pointwise restriction of propagators).

\emph{Homotopy template} (Convention~\ref{conv:hms-levels}): Type~VII (physics-dictionary).
The conjecture asserts that the dg Lie algebras controlling
holomorphic (chiral bar) and topological (Kontsevich integral)
deformation problems are quasi-isomorphic.
(Contributing to Conjecture~\ref{conj:master-bv-brst}.)
\end{remark}

\subsection{Anomaly cancellation and Koszul structure}
\label{subsec:anomaly-koszul}
\index{anomaly!Koszul-theoretic}

\begin{theorem}[Anomaly cancellation as Koszul constraint;
\ClaimStatusProvedHere]\label{thm:anomaly-koszul}
\index{anomaly cancellation!Koszul constraint}
For the matter-ghost system
$\cA_{\mathrm{tot}} = \cA_{\mathrm{matter}} \otimes \cA_{\mathrm{ghost}}$:
\begin{enumerate}[label=\textup{(\roman*)}]
\item The obstruction coefficient is additive:
  $\kappa(\cA_{\mathrm{tot}}) = \kappa(\cA_{\mathrm{matter}}) + \kappa(\cA_{\mathrm{ghost}})$
  \textup{(}Corollary~\textup{\ref{cor:kappa-additivity})}.
\item The total bar complex is uncurved
  ($d_{\mathrm{bar}}^2 = 0$) if and only if
  $\kappa(\cA_{\mathrm{matter}}) + \kappa(\cA_{\mathrm{ghost}}) = 0$.
\item For the Virasoro matter-ghost system,
  $\kappa(\mathrm{Vir}_c) = c/2$ and
  $\kappa(\mathrm{ghost}) = c_{\mathrm{ghost}}/2 = -13$, so
  $\kappa_{\mathrm{tot}} = 0$ precisely at $c = 26$.
\end{enumerate}
\end{theorem}

\begin{proof}
Part (i) is Corollary~\ref{cor:kappa-additivity}: the singular OPE
of the tensor product is the sum of the individual singular OPEs
(the fields act on disjoint sectors), so the genus-$1$ curvature
is additive.  Part~(ii) follows from
$\dfib^{\,2} = \kappa \cdot \omega_1 \cdot \mathrm{id}$
(Theorem~\ref{thm:genus-universality},
Convention~\ref{conv:higher-genus-differentials}): the total bar complex is
uncurved if and only if $\kappa_{\mathrm{tot}} = 0$.  Part (iii)
is the computation $\kappa(\mathrm{Vir}_c) = c/2$
(Computation~\ref{comp:virasoro-curvature}) combined with
$c_{\mathrm{ghost}} = -26$, so
$\kappa_{\mathrm{tot}} = c/2 - 13 = 0$ at $c = 26$.
\end{proof}

\begin{theorem}[Physical anomaly cancellation, genus~\texorpdfstring{$0$}{0};
\ClaimStatusProvedHere]\label{thm:anomaly-physical-genus0}
The condition $\kappa_{\mathrm{tot}} = 0$
\textup{(}Theorem~\textup{\ref{thm:anomaly-koszul})} is equivalent,
at genus~$0$, to the BRST anomaly cancellation
$c_{\mathrm{matter}} + c_{\mathrm{ghost}} = 0$.  The quasi-isomorphism
$\Phi \colon (\cA_{\mathrm{tot}}^{\mathrm{rel}}, Q_{\mathrm{BRST}})
\xrightarrow{\sim} (\barB^{\mathrm{ch}}_0(\cA_{\mathrm{tot}}),
d_{\mathrm{bar}})$ of Theorem~\textup{\ref{thm:brst-bar-genus0}}
identifies both anomaly cancellation conditions.
\end{theorem}

\begin{proof}
This is Corollary~\ref{cor:anomaly-physical-genus0}.
\end{proof}

\begin{theorem}[Physical anomaly cancellation for KM and \texorpdfstring{$\mathcal{W}$}{W}-algebras;
\ClaimStatusProvedHere]\label{thm:anomaly-physical-km-w}
\index{anomaly cancellation!KM/W-algebras}
For Kac--Moody algebras $\widehat{\fg}_k$ at $k \neq -h^\vee$
and W-algebras $\mathcal{W}^k(\fg)$ at non-critical level,
the bar-BRST identification and anomaly cancellation hold at
genus~$0$ \emph{without restriction on the central charge}:
\begin{enumerate}[label=\textup{(\roman*)}]
\item The bar complex computes semi-infinite \textup{(}BRST\textup{)}
  cohomology:
  $H^*(\barB(\cA)) \cong H^{\infty/2+*}(\cA)$
  \textup{(}Theorems~\textup{\ref{thm:bar-semi-infinite-km}}
  and~\textup{\ref{thm:bar-semi-infinite-w})}.
\item The anomaly duality $\kappa + \kappa' = 0$ holds for
  Feigin--Frenkel dual pairs
  \textup{(}Corollaries~\textup{\ref{cor:anomaly-duality-km}}
  and~\textup{\ref{cor:anomaly-duality-w})}.
  This is anomaly cancellation: the matter-ghost system has
  $\kappa_{\mathrm{tot}} = 0$ precisely when
  $c = c_{\mathrm{crit}}$.
\item At genus~$0$, the Chern--Simons equations of motion
  are recovered from the Maurer--Cartan equation in the bar
  complex \textup{(}Theorem~\textup{\ref{thm:cs-koszul-km})}.
\end{enumerate}
\end{theorem}

\begin{proof}
Parts~(i) and~(ii) are restated from the cited theorems and
corollaries, which are proved using the PBW filtration,
Arkhipov functor, and Eilenberg--Moore spectral sequence
(Theorem~\ref{thm:bar-semi-infinite-km}) or the
Drinfeld--Sokolov reduction cascade
(Theorem~\ref{thm:bar-semi-infinite-w}).
Part~(iii) follows from Theorem~\ref{thm:cs-koszul-km}:
the bar--semi-infinite identification transports the MC equation
to the CS flatness equation via the WZW/CS boundary correspondence.
\end{proof}

\subsection{Noncommutative Hodge theory from genus tower}
\label{subsec:nc-hodge}
\index{noncommutative Hodge theory}

\begin{definition}[Chiral Hodge numbers]\label{def:chiral-hodge-numbers}
\index{Hodge numbers!chiral}
For a Koszul chiral algebra $\cA$ and genus $g \geq 1$, the
\emph{chiral Hodge numbers} are
\[
h^{p,q}_g(\cA)
\;:=\;
\dim E_2^{p,q}(\barB^{(g)}(\cA))
\]
where $E_2^{p,q}$ is the $E_2$ page of the genus spectral
sequence (Spectral Sequence~\ref{ss:genus}).
\end{definition}

\begin{proposition}[Hodge symmetry from complementarity;
\ClaimStatusProvedHere]\label{prop:nc-hodge-symmetry}
\index{Hodge symmetry!chiral}
For a Koszul chiral algebra $\cA$ with dual $\cA^!$:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \textup{(Hodge symmetry)}
  $h^{p,q}_g(\cA) = h^{q,p}_g(\cA^!)$ for all $g, p, q$.
\item \textup{(Hodge-to-de~Rham degeneration)}
  The genus spectral sequence degenerates at $E_2$
  for Koszul algebras, so that the total chiral Hodge number
  $h^n_g(\cA) = \sum_{p+q=n} h^{p,q}_g(\cA)$
  equals the hypercohomology dimension
  $\dim \mathbb{H}^n(\overline{\mathcal{M}}_g,
  \barB^{(g)}_\bullet(\cA))$.
\item \textup{(Self-dual Hodge diamond)}
  If $\cA \simeq \cA^!$ \textup{(}self-dual\textup{)},
  then $h^{p,q}_g(\cA) = h^{q,p}_g(\cA)$.
\end{enumerate}
\end{proposition}

\begin{proof}
Part (i): Theorem~\ref{thm:quantum-complementarity-main}
provides the complementarity isomorphism
$E_2^{p,q}(\cA) \cong E_2^{q,p}(\cA^!)^\vee$, which gives
$h^{p,q}_g(\cA) = h^{q,p}_g(\cA^!)$.  Part (ii): the $E_2$
collapse for Koszul algebras is
Theorem~\ref{thm:bar-cobar-spectral-sequence}.  Part (iii)
is immediate from (i).
\end{proof}

\begin{remark}[Hodge symmetry from universal MC duality]
\label{rem:hodge-theta}
The Hodge symmetry $h^{p,q}_g(\cA) = h^{q,p}_g(\cA^!)$
has a transparent origin in the universal Maurer--Cartan
class (Theorem~\ref{thm:explicit-theta}): Verdier duality
sends $\Theta_\cA \mapsto -\Theta_{\cA^!}$, which
exchanges the $E_2$ pages of $\cA$ and $\cA^!$ via
$\mathbb{D}(\barB^{(g)}(\cA)) \cong \barB^{(g)}(\cA^!)$.
For self-dual algebras ($\cA \simeq \cA^!$, so
$\Theta_\cA = -\Theta_\cA = 0$, i.e.\ $\kappa = 0$),
the chiral Hodge diamond is symmetric and the bar complex
is uncurved --- the strongest form of Hodge symmetry.
\end{remark}

\begin{remark}[Computability of the weight filtration]\label{rem:weight-filtration-computable}
The weight filtration on $H^*(\overline{C}_n(X))$ that enters the genus spectral sequence $E_2^{p,q}$ is explicitly computable by combinatorial methods.  Bibby--Gadish~\cite{BG24} establish effective algorithms for computing the mixed Hodge structure on configuration space cohomology via intersection lattice data.  For our chiral Hodge numbers $h^{p,q}_g(\cA)$, this means the $E_2$ page can in principle be computed from the combinatorial data of the Orlik--Solomon algebra (for genus~$0$) and the augmented OS algebra (for genus~$g \geq 1$), without needing the full geometric machinery of mixed Hodge modules.
\end{remark}

\begin{conjecture}[Twistor interpretation of genus expansion;
\ClaimStatusConjectured]\label{conj:nc-hodge}
The genus expansion variable $g_s$ in the total bar complex
$\barB^{\mathrm{total}}(\cA) = \sum_g g_s^{2g-2}\, \barB^{(g)}(\cA)$
plays the role of the twistor parameter $\lambda$ in the
Kontsevich--Soibelman noncommutative Hodge
theory~\cite{KontsevichSoibelman}: there exists a
$\mathbb{C}^*$-action on the de~Rham complex of the genus tower
such that the weight-$k$ subspace at genus~$g$ has dimension
$h^{k,n-k}_g(\cA)$.
\end{conjecture}

\begin{remark}[Scope]\label{rem:nc-hodge-scope}
The Hodge numbers and their symmetry are proved
(Proposition~\ref{prop:nc-hodge-symmetry}), and the $E_2$
degeneration is established
(Theorem~\ref{thm:bar-cobar-spectral-sequence}).  The remaining
conjecture is the identification with the Kontsevich--Soibelman
$\mathbb{C}^*$-action, which requires establishing a twistor
structure on the genus tower.  There is currently no framework
in which to formulate this precisely for chiral algebras.

\emph{Homotopy template} (Convention~\ref{conv:hms-levels}): Type~III
(shifted-symplectic/Lagrangian enhancement).  The conjecture
upgrades the genus spectral sequence to a $\mathbb{C}^*$-equivariant
shifted-symplectic structure on the total deformation complex.
(Part of the MC2 theorem package; Theorem~\ref{thm:mc2-full-resolution}.)
\end{remark}

\subsection{The nine programmes: synthesis and interdependence}
\label{subsec:programme-synthesis}
\index{research programmes!interdependence}

Programmes~I (higher-dimensional Koszul duality,
\S\ref{subsec:higher-dim-kd}) and~II (Vassiliev invariants,
\S\ref{subsec:vassiliev}) share the critical-level bar complex
as common foundation, but diverge in application: Programme~I
computes $\mathsf{E}_n$-operadic Koszul duals via configuration
space integrals on $n$-manifolds, while Programme~II specializes
from complex curves to real circles, extracting knot invariants via
the Kontsevich integral.  At $n = 3$, the two programmes converge:
the $\mathsf{E}_3$-bar complex on a $3$-manifold computes
perturbative Chern--Simons invariants, and the restriction to
configuration spaces of circles in $S^3$ produces precisely the
Vassiliev weight systems of Programme~II
(Proposition~\ref{prop:vassiliev-genus0}).

The anomaly cancellation programme
(\S\ref{subsec:anomaly-koszul}) is the physical interpretation of
the curvature theory: $\kappa_{\mathrm{tot}} = 0$ is both the
mathematical condition $d_{\mathrm{bar}}^2 = 0$ and the
physical condition for consistent quantization
(Theorem~\ref{thm:anomaly-koszul}).  This feeds directly into the
AdS$_3$/CFT$_2$ programme (\S\ref{subsec:ads-cft-koszul}), where
the curvature $m_0 = \kappa \cdot \mathbf{1}$ plays the role of
the cosmological constant: the bar-cobar adjunction
$\barB(\cA) \dashv \Omega(\barB(\cA))$ models the
boundary-to-bulk and bulk-to-boundary maps of holographic duality.
The $3$d mirror symmetry programme (\S\ref{subsec:3d-mirror})
extends this physical circle from $2$d to $3$d, with
$\Eone$-chiral Koszul duality
(Theorem~\ref{thm:e1-chiral-koszul-duality}) replacing the
$\Etwo$-chiral theory and Coulomb--Higgs exchange replacing
bar-cobar inversion.

The noncommutative Hodge programme (\S\ref{subsec:nc-hodge})
reinterprets the entire genus tower as a twistor family: the
genus expansion variable $g_s$ plays the role of the
Kontsevich--Soibelman twistor parameter, and the chiral Hodge
numbers $h^{p,q}_g(\cA)$ provide the Hodge-theoretic data.
This programme depends on the complementarity theorem
(Theorem~\ref{thm:quantum-complementarity-main}), which supplies
the Hodge symmetry $h^{p,q}_g(\cA) = h^{q,p}_g(\cA^!)$.

\begin{remark}[Unifying vision: modular Koszul duality for
factorization algebras]\label{rem:unifying-vision}
\index{modular Koszul duality!unifying conjecture}
All of the programmes above are facets of a single conjecture:
for a factorization algebra~$\mathcal{F}$ on an oriented

$\mathrm{FT}(\mathcal{F})$ computes the Koszul dual

The evidence is the following hierarchy of established and
conjectural instances:
\begin{enumerate}[label=\textup{(\roman*)}]
\item At $n = 1$ \textup{(}$A_\infty$-algebras\textup{)}:
  classical bar-cobar duality
  \textup{(}Loday--Vallette~\cite{LV12}\textup{)}.  This is the
  foundational case, with the bar and cobar constructions as mutual
  inverses on the Koszul locus.
\item At $n = 2$ with complex structure
  \textup{(}chiral algebras\textup{)}: the Main Theorems~A, B, C

  \textup{(}Theorems~\ref{thm:bar-cobar-isomorphism-main},
  \ref{thm:higher-genus-inversion},
  \ref{thm:quantum-complementarity-main}\textup{)}.  The Feynman
  transform is realized by Verdier duality on FM compactifications,
  and the genus expansion provides the quantum corrections.
\item At $n = 2$ topological
  \textup{(}$\mathsf{E}_2$-algebras\textup{)}: Kontsevich
  formality~\cite{Kon03}, which identifies the $\mathsf{E}_2$-bar
  complex with the deformation complex of the associative operad.
\item At $n = 3$
  \textup{(}$\mathsf{E}_3$-algebras\textup{)}:
  perturbative Chern--Simons invariants, with Vassiliev weight

  \textup{(}Conjecture~\ref{conj:vassiliev-bar}\textup{)}.
\item At $n = \infty$
  \textup{(}$\mathsf{E}_\infty$-algebras\textup{)}:
  factorization homology duality of

\end{enumerate}
This monograph provides case~(ii) in full detail; the passage

\subsection{The coderived Ran-space formalism}
\label{subsec:coderived-ran}

The bar-cobar adjunction is most naturally an equivalence of
$\infty$-categories---the categorical $\exp/\log$ correspondence
(Remark~\ref{rem:nilpotence-periodicity}):
\[
\bar{B}_X \colon
\operatorname{Alg}_{\mathrm{aug}}(\operatorname{Fact}(X))
\;\rightleftarrows\;
\operatorname{CoAlg}_{\mathrm{conil}}(\operatorname{Fact}(X))
\,:\!\Omega_X,
\]
where $\operatorname{Fact}(X)$ denotes factorization algebras on
$\operatorname{Ran}(X)$ (cf.\ Appendix~\ref{app:existence-criteria}).

On the Koszul locus this is an equivalence: $\exp \circ \log = \mathrm{id}$
on the convergence domain.  When curvature is present---that is,
when the categorical logarithm acquires non-trivial monodromy---the
correct ambient category is not the ordinary derived category but the
\emph{coderived category} in the sense of
Positselski~\cite{Positselski11}: the analytic continuation that
extends the $\exp/\log$ correspondence past the convergence radius.

Curved objects produce acyclic-but-not-contractible complexes that the
derived category erases too brutally; the coderived category retains
the monodromy data that the ordinary derived category discards, just
as the multi-valued logarithm retains branch information that a
single-valued determination loses.

The provisional coderived category
$\mathsf{D}^{\mathrm{co}}_{\mathrm{prov}}(X)$
(Definition~\ref{def:provisional-coderived},
Proposition~\ref{prop:coderived-adequacy})
provides a working substitute within the scope of this monograph.
The Koszul locus---where bar-cobar inversion

\subsection{The Lagrangian form of complementarity}

Theorem~C (Theorem~\ref{thm:quantum-complementarity-main}) establishes
the direct-sum decomposition
$Q_g(\cA) \oplus Q_g(\cA^!)
= H^*(\overline{\mathcal{M}}_g, \mathcal{Z}(\cA))$
via the eigenspace decomposition of the Verdier involution.
The natural strengthening is a Lagrangian-polarization theorem,
now proved:

\begin{theorem}[Lagrangian complementarity;
\ClaimStatusProvedHere]\label{thm:lagrangian-complementarity}
\index{Lagrangian!complementarity|textbf}
For a modular Koszul pair $(\cA, \cA^!)$,
the complex
$C_g := R\Gamma(\overline{\mathcal{M}}_g, \mathcal{Z}_{\cA})$
carries two shifted symplectic structures, each admitting
$Q_g(\cA)$ and $Q_g(\cA^!)$ as complementary Lagrangians in the
sense of Pantev--To\"en--Vaqui\'e--Vezzosi~\cite{PTVV13}:
\begin{enumerate}[label=\textup{(\roman*)}]
\item a $(-(3g{-}3))$-shifted symplectic structure from
  Verdier duality on the center local system
  \textup{(}Proposition~\textup{\ref{prop:ptvv-lagrangian}}\textup{)};
\item a $(-1)$-shifted symplectic structure on the formal moduli
  $\mathrm{Def}_g(\cA)$ from the BV antibracket
  \textup{(}Theorem~\textup{\ref{thm:shifted-symplectic-complementarity}}\textup{)}.
\end{enumerate}
\end{theorem}

\begin{remark}[Degree clarification]\label{rem:lagrangian-conj-scope}
The original formulation of this result stated a single
$(-1)$-shifted symplectic structure on~$C_g$.  The proved version
reveals two structures of different degrees: the Verdier pairing
gives $(-(3g{-}3))$-shifted on~$C_g$ itself, while the BV
antibracket gives $(-1)$-shifted on the formal moduli.  Both support
the same Lagrangian polarization.
See Remark~\ref{rem:ptvv-relation} for the compatibility.
\end{remark}

\subsection{The universal Maurer--Cartan class}

\begin{theorem}[Universal MC class;
\ClaimStatusProvedHere]\label{thm:universal-MC}
\index{Maurer--Cartan class!universal}

symmetry~$\mathfrak{g}$, there exists a class
\[
\Theta_{\cA} \;\in\;
\operatorname{MC}\!\left(
  \operatorname{Def}_{\mathrm{cyc}}(\cA)
  \;\widehat{\otimes}\;
  R\Gamma\!\left(\overline{\mathcal{M}}_{g,\bullet},\, \mathbb{Q}\right)

\begin{enumerate}[label=\textup{(\roman*)}]

\subsection{The discriminant as spectral invariant}
\index{discriminant!spectral invariant}

The shared discriminant $\Delta(x) = (1 - 3x)(1+x)$ across
$\mathfrak{sl}_2$, Virasoro, and $\beta\gamma$ systems
(Theorem~\ref{thm:ds-bar-gf-discriminant}) is not a coincidence
of generating functions.  It is the \emph{first genuinely
nontrivial non-scalar characteristic class} of the modular Koszul
object: where $\kappa(\cA)$ is a scalar measuring the curvature
of $d^2$, the discriminant $\Delta_{\cA}(x)$ records the branch
geometry of the dual Hilbert series.

Distinct chiral theories can lie on the same modular spectral
sheet --- sharing the same $\Delta$ --- even when their local
operator content is entirely different.  The underlying reason
is functorial: Drinfeld--Sokolov reduction changes the growth pole
(i.e.\ the leading OPE data) while preserving the discriminant
family.  This is compatible with the observation that
$\Delta_{\cA}$ depends only on the quadratic OPE data, which is
the Koszul-dual-determining datum.

\begin{theorem}[Discriminant as spectral determinant --- verified cases;
\ClaimStatusProvedHere]
\label{thm:discriminant-spectral-verified}
\index{discriminant!spectral determinant}
For $\cA \in \{\widehat{\mathfrak{sl}}_2,\, \mathrm{Vir},\,
\beta\gamma\}$, the bar-cohomology discriminant satisfies
\[
\Delta_{\cA}(x) \;=\; \det(1 - x\, T_{\mathrm{br}})
\]
where $T_{\mathrm{br}}$ is a $d \times d$ matrix whose eigenvalues
are the reciprocals of the branch points of the algebraic Koszul
dual Hilbert series $P_{\cA^!}(x)$, with $d$ the degree of
algebraicity.
In all three cases $d = 2$, the branch points are $x = 1/3$
and $x = -1$, and
$\Delta(x) = (1 - 3x)(1 + x)$.
\end{theorem}

\begin{proof}
Each of the three Hilbert series involves $\sqrt{1 - 2x - 3x^2}$
(Theorem~\ref{thm:ds-bar-gf-discriminant}):
$P_{\widehat{\mathfrak{sl}}_2}$ and $P_{\mathrm{Vir}}$ satisfy
quadratic equations with discriminant $(1+x)(1-3x)$, while
$P_{\beta\gamma}^2 = (1+x)/(1-3x)$ has the same branch locus.

The branch points of any algebraic function $P(x)$ satisfying
$a(x)P^2 + b(x)P + c(x) = 0$ are the zeros of
$\mathrm{Disc}_P = b(x)^2 - 4a(x)c(x)$.  For
$\widehat{\mathfrak{sl}}_2$, the algebraic equation is
$x(1+x)P^2 - (1+x)P + 1 = 0$
(Theorem~\ref{thm:ds-bar-gf-discriminant}), giving
\[
\mathrm{Disc}_P = (1+x)^2 - 4x(1+x) = (1+x)(1-3x).
\]
The branch points are $x_1 = -1$ and $x_2 = 1/3$.
Setting $\alpha_i = 1/x_i$, we obtain $\alpha_1 = -1$ and
$\alpha_2 = 3$, and
\[
\det(1 - x\,T_{\mathrm{br}}) = (1 - \alpha_1 x)(1 - \alpha_2 x)
= (1+x)(1-3x) = \Delta(x).
\]
Since $P_{\mathrm{Vir}}$ and $P_{\beta\gamma}$ involve the same
radical $\sqrt{1-2x-3x^2}$, they share the same branch locus,
hence the same~$\Delta$.
\end{proof}

\begin{theorem}[Spectral discriminant --- general case;
\ClaimStatusProvedHere]\label{thm:discriminant-spectral}
\index{discriminant!transfer matrix}
\index{discriminant!spectral determinant!general case}
Let $\cA$ be a Koszul chiral algebra whose Koszul dual Hilbert
series $P_{\cA^!}(x) = \sum_{n \geq 1} \dim(\cA^!)_n\, x^n$
is either:
\begin{enumerate}[label=\textup{(\alph*)}]
\item \emph{Rational}: $P_{\cA^!}(x) = N(x)/D(x)$ with
$\gcd(N,D) = 1$, $D(0) = 1$, and $\deg D = r$; or
\item \emph{Algebraic of degree $d \geq 2$}: satisfying an
irreducible $f(x,y) \in \mathbb{Q}[x,y]$, $\deg_y f = d$,
with discriminant $\Delta(x) = \operatorname{Disc}_y(f)$
normalized so that $\Delta(0) = 1$.
\end{enumerate}
Define the \emph{transfer matrix} $T_{\mathrm{rec}}$ as follows.

\emph{Rational case.}
The coefficients $a_n = \dim(\cA^!)_n$ satisfy a
constant-coefficient linear recurrence of order~$r$
(the degree of the denominator).  Let $T_{\mathrm{rec}}$
be its companion matrix.  Then $\Delta_{\cA}(x) := D(x)$ and
\begin{equation}\label{eq:discriminant-spectral-rational}
\Delta_{\cA}(x) \;=\; \det\!\left(1 - x\, T_{\mathrm{rec}}\right).
\end{equation}

\emph{Algebraic case.}
The coefficients $a_n$ satisfy a $P$-recursive recurrence
of order~$d$:
\begin{equation}\label{eq:p-recursive-recurrence}
q_d(n)\, a_n + q_{d-1}(n)\, a_{n-1} + \cdots + q_0(n)\, a_{n-d} = 0
\end{equation}
where $q_i(n) \in \mathbb{Q}[n]$.  Let $c_i = \lim_{n \to \infty}
q_i(n)/n^{\deg q_d}$ be the leading asymptotic coefficient of~$q_i$
\textup{(}so $c_d = 1$ after normalization\textup{)}.
The asymptotic companion matrix is
\[
T_{\mathrm{rec}} =
\begin{pmatrix}
-c_{d-1} & -c_{d-2} & \cdots & -c_0 \\
1 & 0 & \cdots & 0 \\
\vdots & & \ddots & \vdots \\
0 & \cdots & 1 & 0
\end{pmatrix}
\in \mathrm{Mat}_{d \times d}(\mathbb{Q}).
\]
Then the bar-cohomology discriminant satisfies
\begin{equation}\label{eq:discriminant-spectral-general}
\Delta(x) \;=\; \det\!\left(1 - x\, T_{\mathrm{rec}}\right).
\end{equation}
In both cases, the eigenvalues of $T_{\mathrm{rec}}$ are the
reciprocals of the singularities \textup{(}poles or branch
points\textup{)} of~$P_{\cA^!}$, and the spectral
radius of $T_{\mathrm{rec}}$ equals $\dim\mathfrak{g}$ for
Kac--Moody algebras
\textup{(}Theorem~\textup{\ref{thm:dominant-branch-point})}.
\end{theorem}

\begin{proof}
\emph{Rational case.}
Write $D(x) = \prod_{i=1}^{r}(1 - \alpha_i x)$ with
$\alpha_i = 1/x_i$ the reciprocal poles.
The partial-fraction decomposition of
$P = N/D$ shows that $a_n = [x^n]P$ satisfies a
constant-coefficient linear recurrence of order~$r$
with characteristic polynomial
$\chi(t) = \prod(t - \alpha_i)$.
The companion matrix $T_{\mathrm{rec}}$ has $\chi$ as
its characteristic polynomial, so
$\det(1 - x\,T_{\mathrm{rec}}) = \prod(1 - \alpha_i x)
= D(x) = \Delta_{\cA}(x)$.

\emph{Algebraic case.}
The argument combines three classical results from analytic
combinatorics.

\emph{Step~1: Algebraic $\Rightarrow$ $P$-recursive.}
An algebraic power series is $D$-finite
(\cite{FlajoletSedgewick}, Proposition~VI.1), hence its
coefficients satisfy a $P$-recursive (holonomic) recurrence
of order~$d$, giving~\eqref{eq:p-recursive-recurrence}.

\emph{Step~2: Singularities of the generating function.}
Since $P(x)$ is algebraic of degree~$d$ with minimal polynomial
$f(x,y) = 0$, its singularities as an analytic function are
the branch points: the roots $x_1, \ldots, x_m$ of
$\Delta(x) = \operatorname{Disc}_y(f)$
(\cite{FlajoletSedgewick}, Theorem~VII.8).

\emph{Step~3: Poincar\'e--Perron theorem.}
For the $P$-recursive recurrence~\eqref{eq:p-recursive-recurrence}
with asymptotic companion matrix~$T_{\mathrm{rec}}$, the
Poincar\'e--Perron theorem
(\cite{FlajoletSedgewick}, Theorem~VIII.4)
identifies the eigenvalues of $T_{\mathrm{rec}}$ with the
asymptotic growth rates of the solutions: provided
$c_0 \neq 0$ and the eigenvalues $\alpha_1, \ldots, \alpha_d$
have distinct absolute values, the fundamental system of
solutions satisfies
$a_n^{(i)}/a_{n-1}^{(i)} \to \alpha_i$ as $n \to \infty$.

\emph{Step~4: Growth rates = reciprocals of branch points.}
By the transfer theorem for algebraic singularities
(\cite{FlajoletSedgewick}, Theorem~VI.3), the contribution
of a simple branch point~$x_j$ to the coefficients is
$[x^n]\, P(x) \sim C_j \cdot x_j^{-n} \cdot n^{-3/2}$.
The growth rates are therefore $\alpha_i = 1/x_i$, the
reciprocals of the branch points.

\emph{Synthesis.}
Combining Steps~3 and~4, the eigenvalues of $T_{\mathrm{rec}}$
are $\alpha_i = 1/x_i$.  Since $x_1, \ldots, x_m$ are the
roots of~$\Delta$ (Step~2) and $\Delta(0) = 1$:
\[
\det(1 - x\, T_{\mathrm{rec}})
= \prod_{i=1}^d (1 - \alpha_i x)
= \prod_{i=1}^d \Bigl(1 - \frac{x}{x_i}\Bigr)
= \Delta(x). \qedhere
\]
\end{proof}

\begin{remark}[Scope and verification]%
\label{rem:discriminant-spectral-scope}
\index{discriminant!Poincar\'e--Perron}
The rational case is unconditional (given rationality of~$P$);
the algebraic case additionally requires the Poincar\'e--Perron
non-degeneracy condition ($c_0 \neq 0$, eigenvalues of distinct
absolute value).  Both conditions hold for all bar-cohomology
sequences computed in this work.  The residual conjecture is
that $P_{\cA^!}$ is algebraic or rational for all Koszul chiral
algebras associated to simple~$\mathfrak{g}$.

\emph{Algebraic verification $($degree~$2)$.}
For $\widehat{\mathfrak{sl}}_2$, Virasoro, and $\beta\gamma$
(Theorem~\ref{thm:discriminant-spectral-verified}), $P$ is
algebraic of degree~$2$.
The Riordan recurrence
$(n{+}1) a_n = (2n{-}2) a_{n-1} + (3n{-}6) a_{n-2}$
gives leading coefficients $c_2 = 1$, $c_1 = -2$, $c_0 = -3$,
and $T_{\mathrm{rec}} = \bigl(\begin{smallmatrix} 2 & 3 \\ 1 &
0 \end{smallmatrix}\bigr)$ with eigenvalues $3$ and $-1$.
Check: $\det(1 - x\,T_{\mathrm{rec}}) = 1 - 2x - 3x^2
= (1-3x)(1+x) = \Delta(x)$.  The eigenvalue
$3 = \dim(\mathfrak{sl}_2)$ is the exponential growth rate
(Theorem~\ref{thm:dominant-branch-point});
$-1$ reflects the alternating correction from the central
extension.

\emph{Rational prediction for $\widehat{\mathfrak{sl}}_3$.}
Conjecture~\ref{conj:sl3-bar-gf} predicts that
$P_{\widehat{\mathfrak{sl}}_3}$ is \emph{rational} with
denominator $(1-8x)(1-3x-x^2)$, so the theorem applies
in its rational form with $r = 3$ poles.
The companion matrix of the recurrence~\eqref{eq:sl3-bar-recurrence}
has characteristic polynomial $(t-8)(t^2-3t-1)$, giving
eigenvalues $8$, $(3+\sqrt{13})/2$, $(3-\sqrt{13})/2$.
Check: $\det(1 - x\,T_{\mathrm{rec}})
= (1-8x)(1-3x-x^2) = \Delta_{\mathfrak{sl}_3}(x)$.
The spectral radius is $8 = \dim(\mathfrak{sl}_3)$,
consistent with Theorem~\ref{thm:dominant-branch-point}.
The remaining eigenvalues
$(3 \pm \sqrt{13})/2$ are the roots of the DS discriminant
factor $t^2 - 3t - 1$, shared with the $\mathcal{W}_3$ generating
function (Remark~\ref{rem:two-discriminant-families}).

\emph{Why not the Kodaira--Spencer operator on $H^1$?}
A natural first guess is
$\Delta(x) = \det(1 - x\, T_{\mathrm{KS}} \mid H^1)$
where $T_{\mathrm{KS}}$ acts on
$H^1(\bar{B}(\cA)) \cong \mathfrak{sl}_2$ ($3$-dimensional).
But Schur's lemma forces $T_{\mathrm{KS}}$ to be scalar on the
irreducible adjoint, giving $\det = (1-cx)^3$, degree~$3 \neq 2$.
The correct object is $T_{\mathrm{rec}}$ (dimension~$d$, the
degree of algebraicity), not~$H^1$ (dimension~$\dim\mathfrak{g}$).
Identifying a natural \emph{subquotient} $H^{\mathrm{red}}_1
\subset H^1$ on which a geometric operator recovers
$T_{\mathrm{rec}}$ remains open
(Conjecture~\ref{conj:discriminant-ks-operator}).

The discriminant's role as first non-scalar characteristic class
of the modular characteristic hierarchy parallels $\kappa(\cA)$ as the first
scalar characteristic number of the universal Maurer--Cartan
class $\Theta_{\cA}$ (Theorem~\ref{thm:universal-MC}).
\end{remark}

\begin{definition}[Spectral branch object]
\label{def:spectral-branch-object}
\index{spectral branch object|textbf}
\index{discriminant!branch transport operator}
A \emph{spectral branch object} for a modular Koszul chiral
algebra~$\cA$ is a finite-rank perfect complex~$V_{\cA}^{\mathrm{br}}$
together with a \emph{branch transport operator}
$T_{\mathrm{br},\cA} \colon V_{\cA}^{\mathrm{br}} \to
V_{\cA}^{\mathrm{br}}$ such that
\begin{equation}\label{eq:discriminant-det}
\Delta_{\cA}(x)
\;=\;
\det\!\bigl(1 - x\, T_{\mathrm{br},\cA}\bigr).
\end{equation}
The pair $(V_{\cA}^{\mathrm{br}}, T_{\mathrm{br},\cA})$ satisfies:
\begin{enumerate}[label=\textup{(\roman*)}]
\item $\Delta_{\cA^!} = \Delta_{\cA}$
  (the spectral discriminant is self-dual);
\item exact factorization functors preserving quadratic OPE data
  preserve $T_{\mathrm{br},\cA}$;
\item $\operatorname{rank}(V_{\cA}^{\mathrm{br}})$
  equals the degree of algebraicity of the Koszul-dual
  Hilbert series.
\end{enumerate}

\smallskip\noindent
\emph{Status.}
In the verified cases (Theorem~\ref{thm:discriminant-spectral-verified}),
$V_{\cA}^{\mathrm{br}}$ is constructed as the span of the
companion-matrix eigenvectors of $T_{\mathrm{rec}}$,
and~\eqref{eq:discriminant-det} is proved.  The general
construction of $V_{\cA}^{\mathrm{br}}$ from the bar complex
(rather than from the recurrence) remains open; see
Conjecture~\ref{conj:discriminant-ks-operator} for the
conjectural identification with a subquotient of~$H^1$.
(Part of the MC2 theorem package; Theorem~\ref{thm:mc2-full-resolution}.)
\end{definition}

\subsection{The index theorem for genus expansions}

The appearance of the $\hat{A}$-genus in the Heisenberg genus
expansion (Theorem~\ref{thm:universal-generating-function})
and the Bernoulli asymptotics of $F_g(\cA)$ are not coincidental:
they are the output of a Grothendieck--Riemann--Roch computation
on the universal curve.

\begin{theorem}[Family index theorem for genus expansions;
\ClaimStatusProvedHere]\label{thm:family-index}
\index{index theorem!modular deformation|textbf}
\index{Grothendieck--Riemann--Roch!genus expansion|textbf}
Let $\cA$ be a Koszul chiral algebra with obstruction coefficient
$\kappa(\cA)$, let $\pi \colon \overline{\mathcal{C}}_g \to
\overline{\mathcal{M}}_g$ be the universal curve with relative
dualizing sheaf~$\omega_\pi$, and let
$\mathbb{E} = R^0\pi_*\omega_\pi$ be the Hodge bundle.
Define the \emph{modular deformation complex} of~$\cA$ at genus~$g$ by
\begin{equation}\label{eq:modular-deformation-complex-concordance}
\mathcal{D}_{\cA}^{(g)}
\;:=\;
\kappa(\cA) \cdot \mathbb{E}
\;\in\; K_0(\overline{\mathcal{M}}_g) \otimes_{\mathbb{Z}} \mathbb{Q},
\end{equation}
the $K$-theory class given by $\kappa(\cA)$ copies of the
Hodge bundle.  Then:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{GRR formula.}
The genus-$g$ free energy is the degree-$g$ component of the
Grothendieck--Riemann--Roch pushforward:
\begin{equation}\label{eq:family-index-grr}
F_g(\cA)
\;=\;
\kappa(\cA) \cdot
\int_{\overline{\mathcal{M}}_{g,1}} \psi^{2g-2}\, c_g(\mathbb{E})
\;=\;
\kappa(\cA) \cdot
\left[
  \pi_*\!\left(
    \operatorname{ch}(\omega_\pi) \cdot
    \operatorname{Td}(T_\pi)
  \right)
\right]_g^{\mathrm{eval}},
\end{equation}
where $[\cdot]_g^{\mathrm{eval}}$ denotes evaluation of the degree-$g$ Chern character
$\operatorname{ch}_g(\mathbb{E})$ against $\psi^{2g-2}$ on
$\overline{\mathcal{M}}_{g,1}$.

\item \emph{$\hat{A}$-genus identification.}
The total generating function is
\begin{equation}\label{eq:family-index-generating}
\sum_{g \geq 1} F_g(\cA)\, x^{2g}
\;=\;
\kappa(\cA) \cdot \bigl(\hat{A}(ix) - 1\bigr)
\;=\;
\kappa(\cA) \cdot
\left(\frac{x/2}{\sin(x/2)} - 1\right),
\end{equation}
where the function $(x/2)/\sin(x/2)$ arises from
$\operatorname{Td}(T_\pi) = c_1(\omega_\pi)/(1 - e^{-c_1(\omega_\pi)})$
in the GRR formula, Wick-rotated by the positivity of
tautological intersection numbers on
$\overline{\mathcal{M}}_g$.

\item \emph{Verdier compatibility.}
Under Koszul duality $\cA \leftrightarrow \cA^!$,
$\mathcal{D}_{\cA^!}^{(g)} = \kappa(\cA^!) \cdot \mathbb{E}$,
and the complementarity
$\kappa(\cA) + \kappa(\cA^!) = \kappa_{\mathrm{total}}$
\textup{(}Theorem~\textup{\ref{thm:genus-universality}(ii))}
gives
\[
F_g(\cA) + F_g(\cA^!)
\;=\;
\kappa_{\mathrm{total}} \cdot \lambda_g^{\mathrm{FP}},
\]
the ``index-theoretic complementarity'': the total genus expansion
of a Koszul pair is a fixed multiple of the
Faber--Pandharipande tautological integral.
\end{enumerate}
\end{theorem}

\begin{proof}
The proof assembles four ingredients, each established
independently.

\emph{Step~1: Construction of $\mathcal{D}_{\cA}^{(g)}$.}
The genus universality theorem
(Theorem~\ref{thm:genus-universality}) establishes that the
genus-$g$ obstruction class factors as
$\mathrm{obs}_g(\cA) = \kappa(\cA) \cdot \lambda_g$ in
$H^{2g}(\overline{\mathcal{M}}_g)$, where
$\lambda_g = c_g(\mathbb{E})$ is the top Chern class of the
Hodge bundle.  This factorization determines
$\mathcal{D}_{\cA}^{(g)}$ uniquely in
$K_0(\overline{\mathcal{M}}_g) \otimes \mathbb{Q}$:
the obstruction class $\mathrm{obs}_g$ is the top Chern class of
$\mathcal{D}_{\cA}^{(g)}$, and the universality theorem
shows that $c_g(\mathcal{D}_{\cA}^{(g)}) =
\kappa(\cA) \cdot c_g(\mathbb{E})$, i.e.,
$\mathcal{D}_{\cA}^{(g)} = \kappa(\cA) \cdot \mathbb{E}$
as $K$-theory classes.

The construction is \emph{functorial}: $\kappa(\cA)$ is
determined by the genus-$1$ curvature of the bar complex
(a canonical invariant of~$\cA$), and $\mathbb{E}$ is determined
by the universal curve (independent of~$\cA$).  The product
$\kappa(\cA) \cdot \mathbb{E}$ is therefore a functor from
Koszul chiral algebras to $K$-theory classes on
$\overline{\mathcal{M}}_g$.

\emph{Step~2: Mumford's GRR computation.}
By Grothendieck--Riemann--Roch applied to the universal curve
$\pi \colon \overline{\mathcal{C}}_g \to \overline{\mathcal{M}}_g$
and the relative dualizing sheaf~$\omega_\pi$:
\begin{equation}\label{eq:mumford-grr}
\operatorname{ch}(\mathbb{E})
\;=\;
\operatorname{ch}(R^0\pi_*\omega_\pi)
\;=\;
1 + \pi_*\!\left(
  \operatorname{ch}(\omega_\pi) \cdot
  \operatorname{Td}(T_\pi)
\right),
\end{equation}

Serre duality on the fibers gives
$R^1\pi_*\omega_\pi \cong (R^0\pi_*\mathcal{O})^\vee \cong \mathcal{O}$,
so $\operatorname{ch}(R\pi_*\omega_\pi) = \operatorname{ch}(\mathbb{E}) - 1$.

Write $K = c_1(\omega_\pi)$ for the relative canonical class.  Since
$c_1(T_\pi) = -K$, the Todd class of the relative tangent bundle is
$\operatorname{Td}(T_\pi) = (-K)/(1 - e^K) = K/(e^K - 1)$, and the
pushforward becomes
\begin{equation}\label{eq:mumford-chern-character}
\operatorname{ch}_n(\mathbb{E})
\;=\;
\pi_*\!\left[
  \frac{K \cdot e^K}{e^K - 1}
\right]_{\text{deg } 2n+1}
\;=\;
\frac{B_{2n}}{(2n)!}\,\kappa_{2n-1}
\;+\; \delta_n
\qquad (n \geq 1),
\end{equation}
where $\kappa_{2n-1} = \pi_*(K^{2n})$ are the
Mumford--Morita--Miller classes \cite{Mumford83}
and $\delta_n$ is supported on the boundary
$\partial\overline{\mathcal{M}}_g$.
Indeed,
\[
\frac{K e^K}{e^K-1}
= \frac{K}{1 - e^{-K}}
= \operatorname{Td}(\omega_\pi)
= 1 + \frac{K}{2} + \sum_{n \geq 1} \frac{B_{2n}}{(2n)!} K^{2n},
\]
and $\pi_*(K^{2n}) = \kappa_{2n-1}$ by definition of the tautological
classes.

\emph{Step~3: Faber--Pandharipande evaluation.}
The Faber--Pandharipande formula
(Theorem~\ref{thm:mumford-formula}) evaluates the
$\psi$-$\lambda_g$ integral:
\[
\int_{\overline{\mathcal{M}}_{g,1}} \psi^{2g-2}\,\lambda_g
\;=\;
\frac{2^{2g-1}-1}{2^{2g-1}} \cdot
\frac{|B_{2g}|}{(2g)!}.
\]
This is the content of~\eqref{eq:family-index-grr}: the genus
universality theorem reduces $F_g$ to $\kappa$ times this
integral, and the integral evaluates to the Bernoulli expression
via the tautological intersection theory initiated by
Mumford~\cite{Mumford83} and completed by
Faber--Pandharipande~\cite{FP00}.

\emph{Step~4: Generating function from the Todd class.}
The Bernoulli numbers $B_{2g}$ are the Taylor coefficients of
the Todd generating function
$x/(1 - e^{-x}) = 1 + x/2 + \sum_{g \geq 1}
\frac{B_{2g}}{(2g)!}\, x^{2g}$.
In Step~2, the Todd class $\operatorname{Td}(\omega_\pi)$
produced these Bernoulli numbers as the Chern character
$\operatorname{ch}_g(\mathbb{E})$, i.e., the genus expansion
\emph{is} the GRR pushforward.
Concretely:
\begin{align}
\sum_{g \geq 1} \lambda_g^{\mathrm{FP}}\, x^{2g}
&\;=\;
\sum_{g \geq 1}
\frac{2^{2g-1}-1}{2^{2g-1}} \cdot
\frac{|B_{2g}|}{(2g)!}\, x^{2g}
\notag \\
&\;=\;
\frac{x/2}{\sin(x/2)} - 1
\;=\; \hat{A}(ix) - 1,
\label{eq:bernoulli-a-hat}
\end{align}
where the second equality is the classical identity
$t/\sin t = \sum_{n \geq 0}
\frac{(2^{2n}-2)\,|B_{2n}|}{(2n)!}\,t^{2n}$
evaluated at $t = x/2$ (equivalently, the even part of
the Wick-rotated Todd class).
Multiplying by $\kappa(\cA)$ gives~(ii).

The positivity of all coefficients
($(x/2)/\sin(x/2)$ has all positive Taylor coefficients,
unlike $\hat{A}(x)$ which alternates) reflects the positivity
of $\lambda_g^{\mathrm{FP}} > 0$ for all~$g$, which in turn
is a consequence of the Hodge index theorem on
$\overline{\mathcal{M}}_g$.

\emph{Step~5: Verdier compatibility.}
Part~(iii) follows immediately from
$\mathcal{D}_{\cA^!}^{(g)} = \kappa(\cA^!) \cdot \mathbb{E}$
and the linearity of $F_g$ in~$\kappa$.
\end{proof}

\begin{remark}[The punchline]\label{rem:family-index-punchline}
The content of the theorem is not the generating function identity
(which was already proved as
Theorem~\ref{thm:universal-generating-function}),
but its \emph{origin}: the genus expansion of a Koszul chiral
algebra is the Grothendieck--Riemann--Roch pushforward of the
Todd class of the universal curve, scaled by the bar-complex
curvature.  The $\hat{A}$-genus appears not by accident but because
$\hat{A}$ is the multiplicative genus associated to the formal
group law $F(x,y) = x + y$ --- the trivial (additive) formal
group --- and the bar-complex genus expansion is additive in
$\kappa$.  Equivalently, the Bernoulli numbers governing both the
Todd class and the $\psi$-$\lambda_g$ integrals are the same
Bernoulli numbers because both are computed by the same GRR
calculation on the universal curve.

In physical terms: the genus expansion of a chiral algebra is
an index.  The ``partition function'' $\sum F_g \hbar^{2g-2}$
is the family index of the modular deformation complex on
$\overline{\mathcal{M}}_g$, and anomaly cancellation
($\kappa_{\mathrm{tot}} = 0$) is the vanishing of this index.

The explicit universal class
$\Theta_{\cA}^{\min} = \kappa \cdot \eta \otimes \Lambda$
(Theorem~\ref{thm:explicit-theta}) gives this a precise
homotopy-algebraic meaning: $\hat{A}$ is the character of
$\Theta_{\cA}$ --- it computes how the universal MC element
distributes its weight across genera.  The Bernoulli numbers
appear because they are the Taylor coefficients of
$\operatorname{tr}(\Theta)$, not because of any formal group
coincidence.  The formal group is additive because $\kappa$
itself is additive under tensor product
(Corollary~\ref{cor:kappa-additivity}), which in turn follows
from $\Theta_{\cA \otimes \cB} = \Theta_{\cA} + \Theta_{\cB}$
in the factorization algebra sense.
\end{remark}

\begin{remark}[Scope]\label{rem:family-index-scope}
Steps~1--3 of the proof use only results established
in this monograph (genus universality,
Theorem~\ref{thm:genus-universality}) and imported results
(Mumford's GRR formula \cite{Mumford83}; Faber--Pandharipande
\cite{FP00}).  Step~4 is classical generating function analysis.
The identification $\mathcal{D}_{\cA}^{(g)} =
\kappa(\cA) \cdot \mathbb{E}$ is canonical given the genus
universality theorem; the functoriality of the construction
follows from the functoriality of $\kappa$ (determined by
genus-$1$ curvature) and $\mathbb{E}$ (determined by the
universal curve).

\emph{What remains.}
The theorem identifies $\mathcal{D}_{\cA}^{(g)}$ via its top Chern
class (the obstruction).  A stronger result would construct
$\mathcal{D}_{\cA}^{(g)}$ as a \emph{sheaf} on
$\overline{\mathcal{M}}_g$ (not just a $K$-theory class) and
identify its full Chern character with the genus expansion at all
intermediate degrees, not just the top.  This would require the
coderived Ran-space formalism of
\S\ref{sec:modular-koszul-programme} and is the subject of
Theorem~\ref{thm:universal-MC}.
\end{remark}

\begin{remark}[The GRR theorem as horizon]
\label{rem:grr-horizon}
\index{Grothendieck--Riemann--Roch!as horizon theorem}
In the Chriss--Ginzburg tradition, a foundational monograph
builds toward a single theorem that justifies the entire
development.  For this monograph, that theorem is
Theorem~\ref{thm:family-index}: the genus expansion of a
Koszul chiral algebra is a GRR pushforward.
The master diagram~\eqref{eq:master-square} is the engine;
the modular characteristic hierarchy is the protagonist:
the scalar genus tower and spectral invariants are proved,
while the full quintuple
$\mathcal{C}_{\cA} = (\Theta_{\cA}, \kappa(\cA), \Delta_{\cA},
\Pi_{\cA}, \mathcal{H}_{\cA})$ now has its H-level component
$\Theta_{\cA}$ \emph{proved}
\textup{(}Theorem~\textup{\ref{thm:mc2-full-resolution}},
resolving
Theorem~\textup{\ref{thm:universal-MC}}\textup{)};
and the family index theorem is the punchline.
What remains beyond this monograph ---
$C_{\mathrm{mod}}$ (shifted-symplectic complementarity),
derived Drinfeld--Kohno, the elliptic extension ---
are further consequences of the same master diagram in richer
geometric settings.
\end{remark}

\subsection{The spectral discriminant as characteristic class}
\index{discriminant!characteristic class}

The discriminant $\Delta_{\cA}(x) = \det(1 - x\,T)$ deserves
promotion from a shared generating-function curiosity to
\emph{the first genuinely nontrivial non-scalar characteristic class
of the modular Koszul object}.  Where the scalar invariant
$\kappa(\cA)$ measures the curvature of~$d^2$, the polynomial
$\Delta_{\cA}$ records the full branch geometry of the dual
Hilbert series (Theorem~\ref{thm:ds-bar-gf-discriminant}).

Distinct chiral theories can lie on the same modular spectral
sheet --- sharing the same discriminant --- even when their local
operator content differs entirely.  Drinfeld--Sokolov reduction
changes the growth pole (the leading OPE data) while preserving
the discriminant family; this is a functorial consequence of the
modular structure, not a numerical accident.
The $K$-theoretic framework that organizes the scalar
invariant~$\kappa$ (L1), the spectral discriminant~$\Delta$ (L2),
and the periodicity profile~$\Pi$ (L3) into a hierarchy of
characteristic classes of the virtual bar
family~$\mathcal{V}_{\cA} \in K_0(\overline{\mathcal{M}}_g)$
is developed in Remark~\ref{rem:structural-saturation}; the
intrinsic construction of the L2 operator from Casimir actions
on bar cohomology is Conjecture~\ref{conj:discriminant-ks-operator}.

\subsection{Chain-level / evaluation-locus derived Drinfeld--Kohno}

The Yangian theory (Chapter~\ref{chap:yangians}) establishes bar-cobar
recovery for the $\Eone$-chiral Yangian and the $q \mapsto q^{-1}$
transformation of the quantum parameter.  The chain-level derived
Drinfeld--Kohno theorem is now proved
(Theorems~\ref{thm:derived-dk-affine}
and~\ref{thm:derived-dk-yangian}): the $\Eone$-chiral Koszul duality
functor, composed with the Kazhdan--Lusztig (resp.\ Kazhdan)
equivalence, yields a derived equivalence with braiding reversal
$R_q \mapsto R_{q^{-1}}$ for both affine algebras and Yangians.
The factorization-level statement on the evaluation locus is also
proved (Theorem~\ref{thm:factorization-dk-eval}).

The full factorization-categorical statement remains conjectural, but
the remaining work is staged rather than monolithic:

\begin{conjecture}[Factorization-categorical DK/KL bridge (DK-3);
  \ClaimStatusConjectured]
\label{conj:derived-drinfeld-kohno}
\index{Drinfeld--Kohno!derived}
There exists an equivalence of $\Eone$-factorization categories
\[
\operatorname{Fact}_{\Eone}(\mathcal{Y}(\mathfrak{g}))
\;\simeq\;
\operatorname{Fact}_{\Eone}(U_q(\mathfrak{g}))^{\mathrm{op}}
\]
extending the chain-level equivalence of
Theorem~\textup{\ref{thm:derived-dk-yangian}}
from evaluation modules to all objects of Yangian
category~$\mathcal{O}$, and identifying the Yangian
$\Eone$-factorization structure with the factorisation quantum
group structure of Latyntsev~\cite{Latyntsev23}.
\end{conjecture}

\subsection{Summary}

The historical rhyme is this.  Classical operadic Koszul duality
says: quadratic presentation plus bar-cobar formalism produces
homotopy theory.  The theory sought here says: factorization
presentation on curves plus Verdier duality on modular
compactifications produces \emph{modular homotopy theory}.  Just as
Drinfeld--Kohno related KZ monodromy to quantum groups, the theory
sought here would relate configuration-space/Deligne--Mumford
monodromy to curved chiral Koszul duality.  And just as \'etale
cohomology supplied the correct ambient cohomology theory for
Frobenius and zeta functions, factorization homology on
$\operatorname{Ran}(X)$ together with Verdier duality on
$\overline{\mathcal{M}}_{g,n}$ is the correct ambient cohomology
theory for OPE, monodromy, and bar-cobar duality on curves.
The following subsection tracks the status of the nine programme-level
aspirations that guided this monograph's development
(Definition~\ref{def:modular-homotopy-theory-working};
Remark~\ref{rem:nine-futures-summary}).

\subsection{Programme status: nine futures}\label{subsec:nine-futures}
\index{modular Koszul duality!programme status}

The development of this monograph was guided by nine interrelated
programme-level aspirations, each advertised in earlier stages of the
project.  Some have been realized as proved theorems; some survive only
after reformulation; one is no longer plausible in its original form.
We assess each, precisely separating what is established from what
remains and stating the correct reformulation where the original
vision requires it.

\medskip

\noindent\textbf{Future~1: All-genus $H^1(\mathcal{M}_g)$-parameterized
bar-cobar theory.}
\index{parameter source!H1 reformulation}

\emph{Original vision.}
Genus-$g$ quantum corrections controlled by parameters
$t_g \in H^1(\mathcal{M}_g)$, with the curved differential
$d_g^2 = \sum_k t_{g,k}\,\mathrm{obs}_k$.

\emph{Status: reformulated and proved.}
The original parameterization is incorrect: $H^1(\mathcal{M}_g) = 0$
for $g \geq 2$ by Harer's theorem, so moduli-space cohomology cannot
supply the genus-$g$ correction parameters.  The correct source is
the \emph{fiber} cohomology $H^1(\Sigma_g, \mathbb{C})$, accessed
via the Leray spectral sequence of the universal curve:
\[
t_g \;\in\; H^0\!\bigl(\mathcal{M}_g,\, R^1\pi_*\mathcal{Z}(\cA)\bigr),
\]
i.e., global sections of the first direct image of the center local
system (Convention~\ref{conv:higher-genus-differentials};
Remark~\ref{rem:parameter-source}).  The reformulated theory is
\emph{proved}: the total corrected differential $\Dg{g}$ incorporates
$A$-period parameters from $H^1(\Sigma_g)$ and satisfies
$\Dg{g}^{\,2} = 0$ (Theorem~\ref{thm:quantum-diff-squares-zero}),
while the fiberwise differential $\dfib$ is curved with
$\dfib^{\,2} = \kappa(\cA)\cdot\omega_g$
(Theorem~\ref{thm:quantum-arnold-relations}).  The genus
universality theorem (Theorem~\ref{thm:genus-universality}) then
controls the obstruction tower by the single scalar~$\kappa(\cA)$.

\medskip

\noindent\textbf{Future~2: Modular Koszul programme.}

\emph{Original vision.}
A complete theory of Koszul duality for factorization algebras
over the modular operad of stable curves, with Verdier duality
controlling the bar-cobar exchange at all genera.

\emph{Status: largely realized.}
The status table on p.~\pageref{sec:modular-koszul-programme}
gives the precise accounting.  Theorems~A--D and~H establish the
scalar core; $C_{\mathrm{mod}}$ (Lagrangian complementarity) and the
Index theorem (GRR for genus expansions) are fully proved; the
evaluation-locus derived Drinfeld--Kohno is proved; MC2 is
fully resolved, with the universal Maurer--Cartan class
$\Theta_{\cA}$ proved (Theorem~\ref{thm:universal-MC}).
The remaining gap is the coderived Ran-space formalism
(\S\ref{subsec:coderived-ran}), belonging to Future~3 below.

\medskip

\noindent\textbf{Future~3: Modular homotopy theory for factorization
algebras on curves.}

\emph{Original vision.}
Factorization presentation on curves plus Verdier duality on
modular compactifications produces a new homotopy theory, in which
genus is a deformation variable internal to Koszul duality itself.

\emph{Status: alive and sharpening.}
This future is the mature endpoint of the project.
The structural content is now precise enough for a working definition.
Definition~\ref{def:modular-homotopy-theory-intro} gives the
formal four-part specification; the three-level ladder from
tree-level to modular
(Definitions~\ref{def:tree-level-homotopy}--\ref{def:modular-homotopy-abstract})
makes the combinatorial step explicit; and the quantum versus cyclic
$L_\infty$ distinction
(Remark~\ref{rem:quantum-vs-cyclic-linf}) identifies the obstruction
to lifting.

\begin{definition}[Modular homotopy theory --- working definition]
\label{def:modular-homotopy-theory-working}
\index{modular homotopy theory!working definition}
A \emph{modular homotopy theory} on a smooth curve~$X$ is the
global completion of the nilpotence-periodicity correspondence
(Remark~\ref{rem:nilpotence-periodicity}) from genus~$0$ to all
genera.  Concretely, it consists of:
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{The categorical $\exp/\log$ adjunction.}  An
  $\infty$-categorical bar-cobar adjunction
  $\barB_X \dashv \Omega_X$ on the coderived category
  $D^{\mathrm{co}}(\operatorname{Fact}(X))$ of curved factorization
  algebras on $\operatorname{Ran}(X)$: the categorical logarithm and
  its inverse, analytically continued to the multi-valued regime via
  the coderived framework.
\item \emph{Functoriality of the logarithm over moduli.}
  Functoriality of the adjunction over the modular operad
  $\{\overline{\mathcal{M}}_{g,n}\}_{g,n}$, intertwined with Verdier
  duality:
  $\mathbb{D}_{\operatorname{Ran}}\,\barB_X(\cA)
  \simeq \barB_X(\cA^!)$.  The monodromy of the logarithm varies
  coherently over moduli.
\item \emph{The complete logarithm.}  A universal Maurer--Cartan class
  $\Theta_{\cA} \in
  \operatorname{MC}(\operatorname{Def}_{\mathrm{cyc}}(\cA)
  \,\widehat{\otimes}\,
  R\Gamma(\overline{\mathcal{M}}_{g,\bullet}, \mathbb{Q}))$
  controlling the genus tower as a single deformation: the full
  logarithmic datum from which the characteristic hierarchy
  (Remark~\ref{rem:characteristic-hierarchy}) and
  its exponential---the modular automorphy---are derived.
\item \emph{The Lagrangian branch structure.}  A shifted-symplectic
  pairing on
  $R\Gamma(\overline{\mathcal{M}}_g, \mathcal{Z}_{\cA})$
  for which the deformation and obstruction complexes
  $Q_g(\cA)$, $Q_g(\cA^!)$ are complementary Lagrangians: the
  branch structure of the categorical logarithm, with each
  branch (algebra vs.\ Koszul dual) contributing a Lagrangian
  half of the total monodromy.
\end{enumerate}
\end{definition}

\noindent
Of these four requirements: (ii) and (iv) are proved
(Theorems~A, C; Proposition~\ref{prop:ptvv-lagrangian}),
where (ii) is established as functorial compatibility over

\end{smallmatrix}\bigr)$

(Theorem~\ref{thm:wn-obstruction})
decomposes along strong generators:
the channel vector
$\tilde{\kappa}_N = (c_2, c_3, \ldots, c_N)$ with $c_s = c/s$
carries one entry per strong generator~$W^{(s)}$ (cross-terms
vanish by conformal weight parity).
The usual scalar $\kappa$ is the sum $\sum_s c_s$.

$\sum_{s=2}^N 1/s \sim \log N$, so the scalar package
\emph{diverges} as $N \to \infty$.  The $\mathcal{W}_\infty$ target
cannot live on the scalar line; it must retain the full channel
vector before summation.  The correct MC4 target for
$\mathcal{W}_\infty$ is a separated complete channel target

$\mathcal{W}_N$ package; the ordinary scalar $\Theta$ is obtained
only after a renormalized projection $\sum_s e_s \rightsquigarrow 1$.

channel pair $(\Theta_{\mathrm{dp}}, \Theta_{\mathrm{sp}})$
(double-pole and simple-pole contributions) transforms by the
shear $\bigl(\begin{smallmatrix} -1 & -2 \\ 0 & 1
\end{smallmatrix}\bigr)$, and the total class
$\Theta = \Theta_{\mathrm{dp}} + \Theta_{\mathrm{sp}}$ is
the $(-1)$-eigenvector.
The analogous multichannel completion is needed on the Yangian side.

\operatorname{Rep}_{\mathrm{fd}}(\mathfrak{sl}_N)$ along
$\mathrm{ev}_a\colon Y_\hbar(\mathfrak{sl}_N) \to
U(\mathfrak{sl}_N)$).
By Lemma~\ref{lem:fd-thick-closure}, the thick closure of
finite-length objects in $D^b(\cA)$ equals $D^b(\cA^{\mathrm{fd}})$:
it captures exactly the complexes with finite-length total
cohomology.  Therefore, if $\mathcal{O}$ contains genuine
infinite-length objects, the literal question
$\mathcal{C}_{\mathrm{eval}} = D^b(\mathcal{O})$ has a
\emph{negative} answer.

The correct generation question is therefore:
\[
\mathcal{C}_{\mathrm{eval}} \stackrel{?}{=}
D^b(\mathcal{O}_{\mathrm{poly}})
\]
where $\mathcal{O}_{\mathrm{poly}}$ is the polynomial subcategory
(all simples finite-dimensional).  This holds for $N = 2$
unconditionally and for $\mathcal{O}_{\mathrm{poly}}$ at all~$N$.
For $\operatorname{Rep}_{\mathrm{fd}}$ in place of
$\mathcal{O}_{\mathrm{poly}}$,
the answer is \emph{yes}
(Proposition~\ref{prop:dk2-thick-generation-typeA}): every
finite-dimensional irreducible is a subquotient of a tensor product
of fundamental evaluation modules.

For modules beyond $\mathcal{O}_{\mathrm{poly}}$, the correct
framework is \emph{localizing} (not thick) generation in a
completed/Ind-enhanced category, or enlargement of the
generator set to include prefundamental or asymptotic modules.

\emph{What is known.}

\begin{enumerate}[label=\textup{(\alph*)}]
\item \emph{Finite-dimensional type~$A$}: resolved
  (Theorem~\ref{thm:dk-fd-typeA}).  Every finite-dimensional simple
  module lies in $\mathcal{C}_{\mathrm{eval}}$.
\item \emph{$N = 2$ (Yangian of $\mathfrak{sl}_2$)}: \textbf{resolved}
  (Theorem~\ref{thm:catO-thick-generation}).
  Category~$\mathcal{O}$ for $Y_\hbar(\mathfrak{sl}_2)$ is controlled
  by the Kirillov--Reshetikhin modules and their limits: every module
  has a finite composition series with finite-dimensional factors
  \cite{ChariPressley94}.  Since every finite-dimensional simple
  lies in $\mathcal{D}_\hbar^{\mathrm{eval}}$
  (Proposition~\ref{prop:dk2-thick-generation-typeA}),
  Lemma~\ref{lem:composition-thick-generation} yields
  $D^b(\mathcal{O}_{Y_\hbar(\mathfrak{sl}_2)})
  = \mathcal{D}_\hbar^{\mathrm{eval}}$.
  More generally, thick generation holds for the polynomial
  subcategory $\mathcal{O}_{\mathrm{poly}}$ at all~$N$
  (Remark~\ref{rem:catO-thick-generation-verification}).
\item \emph{Lattice bypass}: for lattice vertex algebras,
  DK-1$\frac{1}{2}$ (Theorem~\ref{thm:lattice:factorization-koszul})
  gives the factorization equivalence without any thick generation
  argument, via sectorwise finiteness of the $\Lambda$-graded bar complex.
  At simply-laced level~$1$, the FKS identification
  yields an \emph{unconditional} factorization DK statement

\emph{Candidate strategies.}  All four are developed in detail in
\S\ref{sec:cat-O-strategies} of the Yangian chapter.
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Highest-weight filtration + resolution}
  (Proposition~\ref{prop:bgg-criterion}).
  If every Verma module $M(\lambda, a)$ admits a finite resolution
  by finite-dimensional evaluation modules, thick generation follows.

\item \emph{t-structure argument}

  \textbf{Proved} for $\mathfrak{sl}_2$ and
  $\mathcal{O}_{\mathrm{poly}}$ at all~$N$.  The generation

  Lemma~\ref{lem:composition-thick-generation}.

  \textup{(}Theorem~\ref{thm:h-level-factorization-kd} and

\emph{Assessment.}
The generation problem is now resolved for

\noindent\textbf{Future~7: Periodicity and tautological depth.}
\index{periodicity!programme status}

\emph{Original vision.}
Modular, quantum, and geometric periodicities unify into a single
coherent classification, with clean reciprocity laws under Koszul
duality.

\emph{Status: conceptually corrected; two of three inputs resolved.}
The principal correction is that the ``geometric periodicity'' of
the original formulation is not periodic at all: the geometric operator
$\sigma_{\mathrm{geom}} = \kappa(\lambda)$ is \emph{nilpotent}
($\sigma^N = 0$, providing bounded tautological depth) rather than
\emph{periodic} ($\sigma^N = \mathrm{id}$).  The corrected
framework has two genuine periodic sources and one nilpotent
stabilization mechanism:

\begin{enumerate}[label=\textup{(\alph*)}]
\item \emph{Modular periodicity} (Type~I).  Conjectured for minimal
  models (Conjecture~\ref{conj:modular-periodicity-minimal}) and WZW
  models at all positive integer levels
  (Conjecture~\ref{conj:modular-periodicity-wzw}), with $T$-matrix
  periodicity proved unconditionally in both cases; conjectured for
  general rational chiral algebras
  (Conjecture~\ref{conj:modular-periodicity}).  Natural next cases:
  $\mathcal{W}$-algebra minimal models and rational orbifolds.
\item \emph{Quantum periodicity} (Type~II).
  Proved: the quantum-profile input coming from quantum integer,
  quantum-dimension, and fusion-data periodicity is unconditional at
  admissible level.  Its transport under Koszul duality is governed by
  $q \mapsto q^{-1}$
  (Proposition~\ref{prop:periodicity-exchange-koszul}).
\item \emph{Geometric tautological depth} (Type~III).
  \textbf{Proved}: the Kodaira--Spencer image of the Hodge class is
  nilpotent with index $\leq g - 1$ on smooth moduli
  (Theorem~\ref{thm:geometric-depth-smooth}, via Graber--Vakil
  vanishing~\cite{GraVak05}), and $\leq 3g - 2$ on
  $\overline{\mathcal{M}}_g$
  (Theorem~\ref{thm:geometric-periodicity-weak}).  At genus~$2$, the
  Hodge class contribution vanishes entirely.  The boundary-extended
  amplitude is bounded by the Mumford relation
  (Conjecture~\ref{conj:geometric-periodicity}).  The geometric
  contribution controls the stabilization \emph{threshold}---the
  conformal weight beyond which modular/quantum periodicity holds---not
  a periodic factor.
\end{enumerate}
The stratified periodicity package
(page~\pageref{conj:complete-periodicity-classification}) therefore
gives
$\mathrm{Period}(\cA) \mid \mathrm{lcm}(N_{\mathrm{mod}},
N_{\mathrm{quant}})$, with the geometric tautological depth
bounding the stabilization threshold.  The modular input remains the
sole conjectural periodic ingredient.
The reflected periodicity conjecture
(Conjecture~\ref{conj:reflected-modular-periodicity}), which
would express how modular periods transform under Koszul duality
via the dual central charge, depends on
Conjecture~\ref{conj:modular-periodicity}.
Within the monograph itself, the only explicit periodic bar calculation
at theorem level beyond the general exchange theorem is the rank-$1$
$\widehat{\mathfrak{sl}}_2$ pattern of
Proposition~\ref{prop:sl2-periodicity-dl}; higher-rank periodic CDG
packages remain frontier statements and should not be used as already
proved transport data in MC3.

What replaces the original ``unified doctrine'' is a
\emph{stratified periodicity framework} organized by the
\emph{nilpotence-periodicity correspondence}
(Remark~\ref{rem:nilpotence-periodicity}): every periodic phenomenon
in bar cohomology arises as the exponential of a nilpotent datum, and
conversely.  The geometric contribution is correctly classified as
nilpotent depth (proved): the operator $\kappa(\lambda)$ is nilpotent
with index $\leq g-1$ (Theorem~\ref{thm:geometric-depth-smooth}),
and its exponential $\exp(\kappa(\lambda))$ is \emph{unipotent}, not
periodic---geometry provides a stabilization threshold, not a period.

The quantum periodicity is unconditional: the screening operators
$Q_i^2 = 0$ (nilpotent) exponentiate to the periodic representation
theory of quantum groups at roots of unity.  The modular periodicity
of bar cohomology remains conjectural even for the flagship rational
families, but the T-matrix period formula
$N = 24q'/\gcd(p',24)$ for $c = p'/q'$ is the direct consequence of
$\exp(2\pi i N c/24) = 1$: the period is determined by when the
exponential of the central charge (the scalar nilpotent datum) returns
to identity.  The sharp modular formula $N_{\mathrm{mod}}$ is the
principal remaining programme-level target; the
nilpotence-periodicity correspondence predicts that it will be
derived from~$\kappa(\cA)$ via exponentiation.

Thus the seed image rank $3903$ is exact and
\[
\dim H^{1,3}_2 = 66.
\]

The live issue is therefore the periodic/coderived shadow and
categorical meaning of this exact $66$-dimensional packet.  It already
has the rigid support split
\[
F \otimes 48 \;\oplus\; E \otimes 15 \;\oplus\; (K-1)\otimes 3
\]
and palindromic total root-weight profile
\[
1,4,8,12,16,12,8,4,1
\]
across weights $-4,\dots,4$.

Equivalently, this is an exact unit-step
interval staircase with multiplicities
$1,3,4,4,4$ on supports $[-4,4],[-3,3],[-2,2],[-1,1],\{0\}$.
This staircase is also not the image of the first admissible $N=3$
packet under any exact symmetric convolution transport at all: support
forces every radius-$R\ge2$ kernel coefficient at $\pm R$ to vanish,
reducing any exact transport to the radius-$1$ case, and the exact
radius-$0$ and radius-$1$ systems are already inconsistent for both one
flavor and the paired packet.  Signed radius-$4$ kernels do exist on
the restricted window $[-4,4]$, but they necessarily spill outside that
window and so are not genuine transports.  So the next categorical test
is no longer a dimension match; it must explain how the sparse first
$N=3$ packet relates, if at all, to this rigid $N=4$ staircase by a
genuinely non-convolutional shadow, or show that the $N=4$ packet is
genuinely new.

There is, however, now a first structured finite-window candidate on the
S-level: the paired $N=3$ profile is recovered exactly from the
$N=4$ staircase on $[-4,4]$ by the Dirichlet discrete Schr\"odinger
operator $a\Delta + V(w)$ with
$a=-3/140$ and even sextic potential
\[
V(w)=\frac{4}{35}
-\frac{2519}{16800}w^2
+\frac{503}{13440}w^4
-\frac{17}{9600}w^6.
\]
No such operator exists with lower even potential degree, and pure
multiplication requires degree~$8$.  So the first surviving structured
route is already second-order and finite-window, not convolutional.
