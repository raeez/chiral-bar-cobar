\chapter{Explicit computations via NAP duality}
\label{chap:NAP-computations}

\index{Poincare duality!explicit computations}

Non-abelian Poincar\'e duality converts the global statement of
Theorem~A, an adjunction between factorization homology and
cohomology on Ran$(X)$, into a Verdier duality computation on
individual strata, where the integral reduces to iterated residues
against explicit propagators. Given any Koszul pair $(\cA, \cA^!)$, the NAP recipe
produces the bar complex differentials, the cobar coproducts, and
the genus corrections from a single configuration-space integral.

Three worked examples cover the standard families and serve as independent
verification: the free fermion (where the simple pole yields a
non-degenerate bar differential), the $bc$-$\beta\gamma$ system
(where statistics exchange implements Koszul duality), and affine
Kac--Moody algebras (where the Feigin--Frenkel level shift
$k \mapsto -k - 2h^\vee$ emerges from the Verdier pairing). In
each case, the NAP computation reproduces the bar-cobar output of
earlier chapters by an independent route, confirming both the
general theorem and the specific invariants.

More precisely, the NAP theorem (Theorem~\ref{thm:main-NAP-resolution}) identifies $\int_X \mathcal{A}$ with the Verdier dual of $\int_{-X} \mathcal{A}^!$. The bar construction is factorization homology; the cobar construction is factorization cohomology via the Verdier pairing. The standard Koszul pairs now exhibit how curvature $m_0$, level shift $k \mapsto -k-2h^\vee$, and genus corrections emerge from the NAP machinery.

\subsection{Heisenberg algebra}

\begin{example}[Heisenberg via NAP]\label{ex:heisenberg-NAP}
The Heisenberg algebra $\mathcal{H}_k$ with OPE $J(z)J(w) = k/(z-w)^2$ has vanishing bracket differential $d_{\mathrm{bracket}} = 0$ (no simple pole) but nonzero curvature differential $d_{\mathrm{curvature}} = k$ (see~\S\ref{sec:frame-bar-all}), collision residue $r^{\mathrm{coll}}(z) = k/z$, and Koszul dual $\mathcal{H}_k^! = \mathrm{Sym}^{ch}(V^*)$ (Theorem~\ref{thm:heisenberg-not-self-dual}). From the NAP perspective, the level $k$ central extension becomes the curvature $m_0 = k\omega_X$ in the curved $A_\infty$ structure on the dual:
\[\int_X \mathcal{H}_k \simeq \mathbb{D}\left(\int_{-X} \mathrm{Sym}^{ch}(V^*)\right) \quad \text{with curvature } m_0 = k\omega_X\]
\end{example}

\begin{remark}[Heisenberg Koszul duality]\label{rem:heisenberg-koszul}
The duality $\mathcal{H}_k^! = \mathrm{Sym}^{ch}(V^*)$ is \emph{not} level-rank duality $SU(N)_k \leftrightarrow SU(k)_N$, which exchanges different algebras. It is operadic: $\mathrm{chirLie}^! = \mathrm{chirCom}$, realized at the factorization algebra level by Verdier duality on configuration spaces.
\end{remark}

\subsection{Free fermions: correct Koszul pair}

\begin{example}[Free fermions via NAP]\label{ex:fermions-NAP}
The free fermion chiral algebra $\mathcal{F}$ has:

The generators are $\psi(z), \psi^*(z)$ with conformal weight $h = 1/2$, and the OPE is:
\[\psi(z) \psi^*(w) = \frac{1}{z-w} + \text{regular}\]
\[\psi(z) \psi(w) = 0, \quad \psi^*(z) \psi^*(w) = 0\]

\emph{Step~1: Identify underlying classical Koszul pair.}

As a graded algebra:
\[\mathcal{F} \cong \text{Exterior algebra } \Lambda(V)\]

where $V = \text{span}\{\psi, \psi^*\}$.

Classical Koszul theory gives:
\[\Lambda(V)^! = \text{Sym}(V^*)\]

\emph{Step~2: Chiral enhancement via configuration spaces.}

The bar construction computes:
\[\bar{B}^{\text{ch}}(\mathcal{F}) = \sum_{n \geq 0} \int_{\overline{C}_{n+1}(X)} \mathcal{F}^{\boxtimes(n+1)} \otimes \Omega^*_{\log}\]

The fermionic OPE $\psi \psi^* \sim (z-w)^{-1}$ gives residues:
\[\text{Res}_{D_{ij}}(\psi_i \otimes \psi^*_j \otimes \eta_{ij}) = 1\]

These assemble into the coproduct of the bar construction $\bar{B}^{\mathrm{ch}}(\mathcal{F})$:
\[\Delta(\psi^*) = 0, \quad \Delta(\psi) = 0 \quad \text{(primitives)}\]
\[\Delta(\psi \psi^*) = \psi \otimes \psi^* - \psi^* \otimes \psi\]
(the minus sign arises from the Koszul rule for transposing two odd elements). After desuspension $s^{-1}$, the generators become even and the coproduct acquires the symmetric (positive) sign, yielding the coalgebra structure of $\mathrm{Sym}^c(V^*)$

\emph{Step~3: Verdier duality verification.}

The NAP identity:
\[\int_X \mathcal{F} \xleftrightarrow{\mathbb{D}} \int_{-X} \beta\gamma\]

where $\beta\gamma$ is the boson system (which has $\text{Sym}$ structure).

Hence $\mathcal{F}^! \simeq \beta\gamma$.

This is the chiral Koszul duality $\Lambda(V)^! \simeq \mathrm{Sym}(V^*)$ lifted to the vertex algebra setting. Note: this is \emph{not} the boson-fermion correspondence (bosonization), which is an \emph{isomorphism} $\mathcal{F} \cong V_{\mathbb{Z}}$ via lattice VOA extension, not a duality.
\end{example}

\subsection{Affine Kac--Moody at critical level}

\begin{example}[Affine Lie algebras via NAP]\label{ex:kac-moody-NAP}
For a simple Lie algebra $\mathfrak{g}$, the affine Kac--Moody algebra at level $k$ is:
\[\widehat{\mathfrak{g}}_k = (\mathfrak{g} \otimes \mathbb{C}[t, t^{-1}]) \oplus \mathbb{C} K\]

with central extension determined by $k$.

\emph{At critical level $k = -h^\vee$.}

The center becomes infinite-dimensional:
\[\mathcal{Z}(\widehat{\mathfrak{g}}_{-h^\vee}) \cong \text{Fun}(\text{Op}_{{}^L\fg}(D^\times))\]
(the algebra of functions on the space of ${}^L\fg$-opers on the punctured disk, by the Feigin--Frenkel theorem; these map to the Hitchin base but carry additional structure)

\emph{NAP duality (orientation reversal).}
\[\int_X \widehat{\mathfrak{g}}_{-h^\vee} \simeq \mathbb{D}\left(\int_{-X} \widehat{\mathfrak{g}}_{-h^{\vee}}\right)\]

This is pure NAP duality: Verdier duality on the factorization algebra associated to the \emph{same} chiral algebra on the orientation-reversed curve. The Langlands dual $\mathfrak{g}^\vee$ enters separately, through the Feigin--Frenkel theorem:

\emph{The geometric Langlands connection.}
The Feigin--Frenkel theorem identifies $\mathcal{Z}(\widehat{\mathfrak{g}}_{-h^\vee})$ with ${}^L\mathfrak{g}$-opers, providing a bridge to the Langlands dual; $\widehat{\mathfrak{g}}_{-h^\vee}$ describes D-modules on $\text{Bun}_G(X)$; and the geometric Langlands correspondence (conjecturally) relates D-modules on $\text{Bun}_G(X)$ to those on $\text{Bun}_{G^\vee}(X)$, a deeper phenomenon than NAP duality.

\begin{conjecture}[Critical-level affine \texorpdfstring{$\to$}{to} Yangian bridge;
\ClaimStatusConjectured]\label{conj:critical-affine-yangian}
Following Costello's 4d Chern--Simons framework, one expects
\[
(\widehat{\mathfrak{g}}_{-h^\vee})^! \;\simeq\; Y(\mathfrak{g}).
\]
\end{conjecture}

\begin{remark}[Scope]
Conjecture~\ref{conj:critical-affine-yangian} is not used as input in
this chapter. It records a possible bridge from the proved NAP
computations to the geometric Langlands / quantum-group interface.
\end{remark}
\end{example}

\subsection{Virasoro algebra}

\begin{example}[Virasoro via NAP]\label{ex:virasoro-NAP}
The Virasoro algebra has generator $T(z)$ with OPE:
\[T(z) T(w) = \frac{c/2}{(z-w)^4} + \frac{2T(w)}{(z-w)^2} + \frac{\partial T(w)}{z-w} + \text{reg}\]

\emph{Problem.} This is highly non-quadratic; completion is required.

\emph{Step~1: Classical limit.}

Set $c = 0$ (classical Virasoro):
\[\mathcal{V}_0 = \text{Lie}(\text{Vect}(S^1)) \quad \text{(vector fields on circle)}\]

Classical Koszul duality for Lie algebras:
\[U(\mathcal{V}_0)^! \simeq CE^*(\mathcal{V}_0) \quad \text{(Chevalley--Eilenberg)}\]

\emph{Step~2: Quantum deformation at $c \neq 0$.}

The central charge $c$ enters as curvature of the bar complex:
\[\kappa(\text{Vir}_c) = \frac{c}{2}\]

where the Virasoro modular characteristic is the genus-independent curvature parameter
\textup{(}Theorem~\ref{thm:genus-universality}\textup{)}, arising from the Virasoro
2-cocycle $\omega(L_m, L_n) = \frac{1}{12}m(m^2-1)\delta_{m+n,0}$.

The Koszul dual becomes a \emph{curved} algebra:
\[m_0 = \kappa(\mathrm{Vir}_c) \cdot \omega_{\text{Vir}} \neq 0\]

\emph{Step~3: NAP interpretation.}

At genus $g$, the bar construction gives:
\[\bar{B}^{(g)}(\mathcal{V}_c) = \bigoplus_{n \geq 0}\, \Gamma\!\left(\overline{C}_n(\Sigma_g),\, \mathcal{V}_c^{\boxtimes n} \otimes \Omega^*_{\log}\right)\]

The modular characteristic $\kappa(\mathrm{Vir}_c) = c/2$ couples to the Chern class of the Hodge bundle:
\[(c/2) \cdot \lambda_1 \in H^2(\overline{\mathcal{M}}_g)\]

This is the \emph{geometric origin of central charge} from the NAP perspective.

\begin{proposition}[Virasoro NAP duality at \texorpdfstring{$c=26$}{c=26};
\ClaimStatusProvedHere]\label{prop:virasoro-c26-selfdual}
At $c = 26$, the Koszul dual is $\mathrm{Vir}_{26}^! \simeq \mathrm{Vir}_0$
(the uncurved algebra), so NAP gives
\[
\int_X \mathcal{V}_{26}
\;\simeq\;
\mathbb{D}\!\left(\int_{-X} \mathcal{V}_0\right).
\]
This is \emph{not} self-duality: the self-dual point under
the complementarity $c + c' = 26$
(Theorem~\textup{\ref{thm:central-charge-complementarity}})
is $c = 13$.
\end{proposition}

\begin{proof}
By Example~\ref{ex:virasoro-koszul-dual},
$\mathrm{Vir}_c^! \simeq \mathrm{Vir}_{26-c}$. Setting $c = 26$
gives $\mathrm{Vir}_{26}^! \simeq \mathrm{Vir}_0$.
NAP (Theorem~\ref{thm:main-NAP-resolution}) then gives the stated duality.
\end{proof}

\begin{remark}[Scope]
The self-dual point $c = 13$ corresponds to the unique central charge
at which $\mathrm{Vir}_c^! \simeq \mathrm{Vir}_c$. Whether the
factorization homology at $c = 13$ carries additional structure
(e.g., a non-degenerate self-pairing) is an open question.
\end{remark}
\end{example}

\section{\texorpdfstring{Computing $\mathcal{A}^!$ via non-abelian Poincar\'e duality}{Computing the Koszul dual via NAP}}

\subsection{Procedure}
\label{alg:NAP-koszul}

Given a chiral algebra $\mathcal{A}$ on $X$ with generators $\{a_i\}$ and OPE
\[
a_i(z) a_j(w) = \sum_{k,m}\frac{C^k_{ij,m}}{(z-w)^m}a_k(w) + \text{descendants},
\]
the Koszul dual chiral coalgebra $\mathcal{A}^!$ is constructed as follows.

\emph{Step~1: Identify generators of $\mathcal{A}^!$.}
For each generator $a_i \in \mathcal{A}$ of conformal weight $h_i$, introduce a dual
generator $a_i^* \in \mathcal{A}^!$ with $|a_i^*| = -h_i$.

\emph{Step~2: Compute coproduct from OPE.}
For each OPE term $a_i(z)a_j(w) \sim \frac{C^k_{ij,m}}{(z-w)^m} a_k(w)$, define the coproduct component:
\[
\Delta(a_k^*)=\sum_{i,j,m}C^k_{ij,m}(a_i^* \otimes a_j^*)+\text{primitive part}.
\]

\emph{Step~3: Handle composite fields.}
If the OPE involves composite fields (e.g., $:T \cdot T:$ in W-algebras),
add generators $(\text{composite})^*$ to $\mathcal{A}^!$,
compute their coproducts from residue formulas,
and take the $I$-adic completion $\widehat{\mathcal{A}^!}$.

\emph{Step~4: Define differential from Verdier pairing.}
The coproduct $\Delta\colon \mathcal{A}^! \to \mathcal{A}^! \otimes \mathcal{A}^!$ is
dual to the chiral product:
\[
\langle \Delta(a_i^*), a_j \otimes a_k \rangle = \langle a_i^*, \mu(a_j \otimes a_k)\rangle.
\]

\emph{Step~5: Verify coalgebra axioms.}
Check coassociativity $(\Delta \otimes \mathrm{id}) \circ \Delta = (\mathrm{id} \otimes \Delta) \circ \Delta$,
the coderivation property $\Delta \circ d = (d \otimes \mathrm{id} + \mathrm{id} \otimes d) \circ \Delta$,
and conilpotency $\Delta^{(N)} = 0$ for $N$ sufficiently large.

\emph{Step~6: Identify Koszul partner.}
If the Koszul dual coalgebra of $\mathcal{A}$ is quasi-isomorphic to the bar coalgebra of another chiral algebra $\mathcal{B}$, i.e.,
$\bar{B}^{\mathrm{ch}}(\mathcal{A})^{\vee} \simeq \bar{B}^{\mathrm{ch}}(\mathcal{B})^{\vee}$,
then $(\mathcal{B}, \mathcal{A})$ form a Koszul pair. Verify:
\[
\Omega^{\mathrm{ch}}(\bar{B}^{\mathrm{ch}}(\mathcal{A})) \simeq \mathcal{A}
\quad\text{and}\quad
\bar{B}^{\mathrm{ch}}(\mathcal{B})^{\vee} \simeq \mathcal{A}^!.
\]

\subsection{\texorpdfstring{Worked example: $\beta\gamma$ system}{Worked example: system}}

\begin{computation}[\texorpdfstring{$\beta\gamma$}{beta-gamma} Koszul dual]\label{comp:betagamma-dual}
The $\beta\gamma$ chiral algebra has generators $\beta(z), \gamma(z)$ with OPE:
\[\beta(z) \gamma(w) = \frac{1}{z-w} + \text{regular}\]
\[\beta(z) \beta(w) = 0, \quad \gamma(z) \gamma(w) = 0\]

We apply the procedure of \S\ref{alg:NAP-koszul}.

\emph{Step~1: Dual generators.}
\[\beta^* \in (\beta\gamma)^!, \quad |\beta^*| = \text{weight of } \beta = \lambda\]
\[\gamma^* \in (\beta\gamma)^!, \quad |\gamma^*| = \text{weight of } \gamma = 1-\lambda\]

\emph{Step~2: Coproduct from OPE.}

The OPE $\beta \gamma \sim (z-w)^{-1}$ gives:
\[\Delta(\beta^*) = 0 \quad \text{(primitive)}\]
\[\Delta(\gamma^*) = 0 \quad \text{(primitive)}\]

Since $\beta^*, \gamma^*$ are cogenerators of $\bar{B}(\beta\gamma)$, the relevant coproduct is the deconcatenation on the bar complex. The $\beta\gamma$ system is not a Koszul pair with itself; rather:
\[(\beta\gamma)^! = \text{free fermions } \mathcal{F}\]

This is the fermionization of bosons.

\emph{Step~3: Verification via NAP.}

The factorization homology and Verdier duality give:
\[\int_X \beta\gamma = \mathrm{Sym}^*(V), \qquad \mathbb{D}\!\left(\int_{-X} bc\right) = \Lambda^*(V^*)\]

The Koszul duality $\mathrm{Sym}^*(V)^! = \Lambda^*(V^*)$ (classical Koszul duality of quadratic algebras) confirms:
\[(\beta\gamma)^! = bc \quad \text{and} \quad (bc)^! = \beta\gamma\]

They are Koszul duals. (Note: the boson-fermion correspondence $\mathrm{Sym}^*(V) \simeq \Lambda^*(V)$ as vector spaces is \emph{different} from Koszul duality $\mathrm{Sym}^! = \Lambda^*$; the former is a representation isomorphism, the latter an operadic duality. See Theorem~\ref{thm:betagamma-fermion-koszul}.)
\end{computation}

\section{Higher genus corrections via NAP}

\subsection{Genus expansion of factorization homology}

\begin{definition}[Genus-graded NAP duality]\label{def:genus-NAP}
At genus $g$, factorization homology decomposes as
\[\int_{\Sigma_g} \mathcal{A} = \bigoplus_{n \geq 0} \int_{C_n(\Sigma_g)} \mathcal{A}^{\otimes n}.\]
Verdier duality at genus $g$,
\[\mathbb{D}^{(g)}: H^*(\overline{C}_n(\Sigma_g)) \xrightarrow{\sim} H^{d-*}(C_n(\Sigma_g)),\]
receives contributions from modular forms on $\mathcal{M}_g$, period integrals over $\Sigma_g$, and genus-dependent orientation bundles. The \emph{genus-$g$ Koszul dual} is
\[\mathcal{A}^!_{(g)} = \mathbb{D}^{(g)}(\mathcal{A}).\]
This differs from $(\mathcal{A}^!)_{(g)}$, the genus-$g$ component of $\mathcal{A}^!$: the former is the genus-$g$ deformation of the dual, while the latter is determined by the genus-$0$ duality alone.
\end{definition}

\begin{theorem}[Genus complementarity; \ClaimStatusProvedHere]\label{thm:genus-complementarity}
For a Koszul pair $(\mathcal{A}_1, \mathcal{A}_2)$ at genus $g$:
\[Q_g(\mathcal{A}_1) \oplus Q_g(\mathcal{A}_2) \cong H^*(\mathcal{M}_g, Z(\mathcal{A}_1))\]

where $Q_g(\mathcal{A}_i)$ denotes genus-$g$ quantum corrections and $Z(\mathcal{A}_1)$ is the center of $\mathcal{A}_1$, viewed as a local system on $\mathcal{M}_g$. What $\mathcal{A}_1$ sees as deformation, $\mathcal{A}_2$ sees as obstruction.
\end{theorem}

\begin{proof}
This is a restatement of Theorem~\ref{thm:quantum-complementarity-main} (Koszul Complementarity at Higher Genus) in the NAP framework. The proof proceeds via the Kodaira--Spencer map (Theorem~\ref{thm:kodaira-spencer-chiral-complete}), which provides a center action $\rho: Z(\mathcal{A}_1) \to \mathrm{End}(H^*(\overline{\mathcal{M}}_g, Z(\mathcal{A}_1)))$, and the Verdier involution $\sigma$ (Lemma~\ref{lem:verdier-involution-moduli}), which satisfies $\sigma^2 = \mathrm{id}$. Since $\mathrm{char}(\mathbb{C}) \neq 2$, the eigenspace decomposition $H^*(\overline{\mathcal{M}}_g, Z(\mathcal{A}_1)) = V^+ \oplus V^-$ with $V^+ = Q_g(\mathcal{A}_1)$ and $V^- = Q_g(\mathcal{A}_2)$ gives the stated direct sum.
\end{proof}
