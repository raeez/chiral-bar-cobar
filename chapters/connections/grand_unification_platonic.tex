\section{The Grand Theorem}

\subsection{Statement}
\label{ssec:grand-theorem-statement}

\begin{theorem}[Grand Unification, chiral bar--cobar]
\label{thm:grand-unification-platonic}
\ClaimStatusProvedHere
Fix a smooth projective curve $X/\CC$ with its Ran-compactified
configuration ind-scheme $\Confbar(X)$. There is a presentable
$\infty$-categorical adjoint pair
\begin{equation}
\label{eq:grand-adjunction-platonic}
   \OmCh \;\dashv\; \BCh
   \colon\;
   \FactAlg_X
   \;\rightleftarrows\;
   \BrMoncomp_X,
\end{equation}
with left adjoint the chiral cobar construction and right adjoint
the chiral bar construction, enjoying the following seven
properties.
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Universal twisting morphism.} The unit $\eta \colon
\id \to \BCh \circ \OmCh$ and counit $\varepsilon \colon \OmCh \circ
\BCh \to \id$ are represented by the universal Maurer-Cartan
element
$\Theta_\cA \in \Hom(\barBch(\cA),\cA)$ in the ordered convolution
dgLa, and the pair $(\eta,\varepsilon)$ classifies bar--cobar
twisting morphisms by the Francis--Gaitsgory universal property
\cite[\S 6]{FG12}.
\item \emph{Koszul duality locus.} The full $\infty$-subcategory
$\FactAlg^{\mathrm{Kos}}_X \subset \FactAlg_X$ on which $\varepsilon$
is an equivalence is presented by chirally Koszul algebras and
carries a symmetric duality $\cA \mapsto \cA^!$ induced from
Verdier dualisation of factorisation $\DMod$-modules;
equivalently $\BCh$ restricts there to an equivalence of
presentable stable $\infty$-categories.
\item \emph{Positselski coderived inversion.} The cobar image
$\OmCh(\BCh(\cC))$ is naturally equivalent to $\cC$ in the coderived
$\infty$-category $D^{\mathrm{co}}_{\mathrm{ch}}(X)$ for every
conilpotent object $\cC \in \BrMoncomp_X$; equivalently, the
counit is an isomorphism after passing to the Positselski coderived
localisation.
\item \emph{Derived centre complementarity.} The composite
$\BCh \circ \OmCh \colon \BrMoncomp_X \to \BrMoncomp_X$ carries a
canonical natural transformation to the Hochschild functor
$\cC \mapsto \cZch(\OmCh(\cC))$ whose scalar invariant is the pair
$(\kappa,\kappa^!)$ of family-dependent constants
$\kappa + \kappa^! \in \{0,\,13,\,250/3,\,98/3\}$ recorded in
Chapter~\ref{ch:landscape-census} on the four standard classes
G / L / C / M.
\item \emph{Obstruction-tower universality.} For every genus $g\geq 0$,
the genus-$g$ trace
\begin{equation}
\label{eq:genus-g-trace-platonic}
   \trgenus \colon
   \BCh(\cA)
   \;\longrightarrow\;
   H^\ast(\overline{\cM}_{g,n},\CC)
\end{equation}
obtained by fibre integration of the universal twisting morphism
along $\overline{\cM}_{g,n} \to \mathrm{pt}$ factors through a
single class in
$H^2(\overline{\cM}_{g,n},\CC)$ proportional to $\kappa(\cA)$, with
proportionality constant the $g$-th $\lambda$-class
$\lambda_g^{\mathrm{FP}}$; i.e.\ $\mathrm{obs}_g = \kappa \lambda_g$
on the uniform-weight lane and
$F_g = \kappa\lambda_g^{\mathrm{FP}} + \delta F_g^{\mathrm{cross}}$
generally.
\item \emph{Hochschild concentration.} The chiral Hochschild object
$\ChirHoch^\bullet(\cA) := \cZch(\cA)$ is naturally equivalent, inside
the adjunction~\eqref{eq:grand-adjunction-platonic}, to the
endomorphism object
$\End_{\BrMoncomp_X}(\BCh(\cA))$; its cohomology is concentrated in
degrees $\{0,1,2\}$ on the proved surface of Theorem~H.
\item \emph{Three-avatar presentation.} The adjoint
pair~\eqref{eq:grand-adjunction-platonic} admits three canonical
presentations, all equivalent on their overlapping domain of
definition:
\begin{itemize}
\item \emph{L1 (chain-level ensemble).} Presented on chain-complexes
  in the ordered-configuration model: $\BCh(\cA)$ is $\BarOrd(\cA) =
  T^c(s^{-1}\bar\cA)$ equipped with the Beilinson-Drinfeld
  factorisation differential; $\OmCh$ is the ordered cobar
  construction; unit/counit are the explicit Mittag-Leffler
  witnesses of Chapter~\ref{chap:bar-cobar}.
\item \emph{L2 (non-Tannakian categorical).} Presented on the
  presentable stable $\infty$-category
  $\Vrep(\cA) := \cA\text{-}\mathsf{mod}^{\mathrm{fact}}$ as the
  bar--cobar reconstruction on braided module categories; the
  formal-locus restriction recovers the Tannakian reconstruction,
  the non-formal complement is the non-Tannakian stratum
  parametrising $\RW_d$-type category-valued outputs of the
  Calabi-Yau functor $\Phi_d$. All four standard archetypes
  remain chirally Koszul; the non-Tannakian distinction tracks
  formality of the chiral $A_\infty$-structure, not Koszulness of
  the chiral bar complex.
\item \emph{L3 ($\infty$-categorical, genus-$g$).} Presented on the
  Lurie-Francis-Gaitsgory machinery \cite[HA~5.2]{LurieDAGX}, \cite{FG12}
  of factorisation $\infty$-operads with values in presentable
  stable $\infty$-categories; the genus-$g$ trace is the homotopy
  fixed-point spectrum of the $S^1$-action on $\Uch_\cA :=
  \cA \bowtie_{\infty} \cA^!_\infty$ restricted to
  $\overline{\cM}_{g,n}$.
\end{itemize}
Each of L1, L2, L3 determines the other two up to contractible
choice on its overlap locus.
\end{enumerate}
\end{theorem}

\section{Proof of the Grand Theorem}
\label{sec:grand-theorem-proof}

\subsection{What is new; what is a re-badging}
\label{ssec:grand-theorem-new-content}

Properties (i)--(vi) re-badge existing Volume~I theorems in the
language of the adjoint pair
\eqref{eq:grand-adjunction-platonic}: (i) is Theorem~A, (iii) is
Theorem~B, (iv) is Theorem~C, (v) is Theorem~D, (vi) is Theorem~H,
and (ii) is the Koszul-locus discipline of
Chapter~\ref{chap:koszul-pair-structure}. The Corollaries
\ref{cor:gu-theorem-a-platonic}--\ref{cor:gu-theorem-h-platonic}
establish these identifications — not new theorems, but a single
organising lens.

The new content of
Theorem~\ref{thm:grand-unification-platonic} is property~(vii):
the identification of L1, L2, L3 as three presentations of one
adjoint pair on their common domain of definition. L1 $\leftrightarrow$ L3
on $\BrMoncomp_X$ is factorisation Dold-Kan
\cite[HA~1.2.3]{LurieDAGX}, and is unconditional. L2 is a
restriction of L3 to presentable stable braided module categories;
its identification with the non-Tannakian stratum is conditional
on the Costello-Li curved-HKR transfer (see
Theorem~\ref{thm:L2-chain-level-gerbe-platonic}). The statement
that all five theorems A--H arise as natural transformations of
one adjunction is the reorganisation; the theorem-grade content
sits in (vii).

\begin{proof}[Proof of Theorem~\ref{thm:grand-unification-platonic}]
\emph{Step 1. The adjunction exists.}
Specialise the Lurie presentable-$\infty$-adjoint-functor theorem
\cite[HA~5.5.2.9]{LurieDAGX} to the pair
$(\FactAlg_X, \BrMoncomp_X)$: both are presentable stable
$\infty$-categories; the functor $\BCh$ preserves
filtered colimits by the compact-generator presentation of
factorisation coalgebras \cite[FG~\S4]{FG12}; hence $\BCh$ has a
left adjoint, which we identify with $\OmCh$ by comparing unit maps
on the generating set. This is the L3 presentation.

\emph{Step 2. Chain-level realisation (L1).}
The Dold-Kan presentation of $\BrMoncomp_X$ on chain-complexes
\cite[HA~1.2.3.7]{LurieDAGX} identifies $\BCh$ with the ordered
bar-complex $\BarOrd(\cA)$ equipped with the Beilinson-Drinfeld
factorisation differential
(Chapter~\ref{chap:bar-cobar}). The
Mittag-Leffler witnesses of
$\eta,\varepsilon$ at chain level are constructed explicitly in
Chapter~\ref{chap:bar-cobar}; they realise L1 as a
presentation of the L3 adjunction.

\emph{Step 3. Non-Tannakian restriction (L2).}
Restrict $\BCh$ along the presentability-preserving inclusion
$\mathsf{BrCat}^{\mathrm{cat,cocomp}}_X \hookrightarrow \BrMoncomp_X$;
the restricted adjunction is the categorical bar--cobar of
Corollary~\ref{cor:gu-L2-platonic}. The fixed points under
Tannakian reconstruction form the Koszul locus; the genericity on
the complement is witnessed by the Rozansky-Witten category on
hyperkaehler targets.

\emph{Step 4. Five-theorem readout.}
Corollaries~\ref{cor:gu-theorem-a-platonic}--%
\ref{cor:gu-theorem-h-platonic} together show that each of the five
main theorems is a natural transformation extracted from
$(\eta,\varepsilon)$: A is the adjunction itself, B is the counit
after Positselski localisation, C is the scalar invariant on the
four archetypes, D is the universal $\lambda_g$-projection, H is
the endomorphism-object concentration.
\qedhere
\end{proof}

\subsection{L2 as the non-Tannakian classification}
\label{ssec:L2-non-tannakian}

The non-Tannakian stratum is parametrised by the locus on which

\begin{proposition}[Non-Tannakian projection]
\label{prop:L2-non-tannakian-platonic}
\ClaimStatusConditional
Conditional on Conjecture~$\Pi_3^{\mathrm{ch}}$ of
Vol~III (chain-level CY-A$_3$), the non-Tannakian stratum of L2
is in bijection with the non-formal CY$_d$ targets $T$, i.e.\ the
targets for which $\Phi_d(T)$ is not fibrewise exact-Tannakian.
Formality of the chiral $A_\infty$-structure, not Koszulness of the
bar complex, is the distinguishing criterion; all four archetypes
G / L / C / M remain chirally Koszul.
\end{proposition}

\begin{proof}[Proof sketch]
Under $\Pi_3^{\mathrm{ch}}$, the CY$_3$-to-chiral functor $\Phi_3$
is defined on chain-level; its output is chirally formal precisely
when the CY$_3$ target is formal (classes G and L), non-formal
when the target is non-formal (classes C and M). All four outputs
remain chirally Koszul. The formality-locus restriction of L2
recovers the Tannakian stratum; the non-formal complement is the
non-Tannakian stratum. The Rozansky-Witten category is the witness
for $d=4$ on hyperkaehler targets.
\end{proof}

\section{The programme as a single $\infty$-adjunction}

chiral bar--cobar programme is the study of the
$\infty$-adjunction
\[
   \OmCh \;\dashv\; \BCh
   \colon\;
   \FactAlg_X
   \;\rightleftarrows\;
   \BrMoncomp_X,
\]
whose domain is chiral algebras on $X$, whose codomain is
braided conilpotent factorisation coalgebras on $\Confbar(X)$,

\section{Open threads}
\label{sec:grand-theorem-open-threads}

The L1 and L3 lanes close unconditionally on a smooth projective
curve $X/\CC$. The L2 lane closes chain-level on compact CY$_3$
targets without an exceptional collection, conditional on the
Costello-Li curved-HKR quasi-isomorphism \cite{CL20} and on the
Bondal-Van den Bergh non-representability \cite{BondalVdB03}; on
compact CY$_3$ targets admitting an exceptional collection the L2
lane closes without gerbe twist. The unconditional theorem below
states the chain-level CY$_3$ content; Vol~III
(\texttt{cy\_to\_chiral.tex}) records the scope.

\begin{theorem}[L2 chain-level functor into gerbe-twisted braided monoidal $\infty$-category]
\label{thm:L2-chain-level-gerbe-platonic}
\ClaimStatusConditional
Let $X$ be a compact CY$_3$ without a full exceptional collection
(Bondal-Van den Bergh non-representability \cite{BondalVdB03});
let $\alpha_X \in H^3(X,\CC^\times)$ denote the Brauer-type gerbe
class obstructing strictification of the braided $\infty$-category
of perfect complexes to an honest bialgebra; let
$\mathrm{HKR}_\infty^{\mathrm{curv},\hbar}$ denote the
curved-$L_\infty$ quasi-isomorphism of Costello-Li \cite{CL20}.
Conditional on the BCOV transfer of \cite{CL20} extending to all
genera on $X$ (proved for the quintic $X_5$; conjectural in full
generality), there is a chain-level $\infty$-functor
\begin{equation}
\label{eq:Phi-3-ch-platonic}
   \Phi_3^{\mathrm{ch}}
   \;\colon\;
   \mathrm{Perf}^{\mathrm{CY3}}_\infty(X)
   \;\longrightarrow\;
   \BrMoncomp_{X,\alpha_X}
\end{equation}
valued in the gerbe-twisted presentable $\infty$-category of
conilpotent braided-monoidal compact-generated factorisation
coalgebras on $\Confbar(X)$. The functor~\eqref{eq:Phi-3-ch-platonic}
satisfies three properties.
\begin{enumerate}[label=\textup{(\roman*)}]
\item \emph{Chain-level realisation.} The underlying chain-level
model of $\Phi_3^{\mathrm{ch}}(X)$ is the explicit chain-level bar
$B^{\mathrm{cat}}(D^b(X))$ (\S~\ref{ssec:explicit-bar-model}),
with $L_\infty$-structure inherited from
$\mathrm{HKR}_\infty^{\mathrm{curv},\hbar}$ and with
gerbe-twisting $\alpha_X$ acting on the associator.
\item \emph{Gerbe compatibility.} The obstruction class
$\alpha_X \in H^3(X,\CC^\times)$ is the obstruction to lifting
$\Phi_3^{\mathrm{ch}}$ from gerbe-twisted
$\BrMoncomp_{X,\alpha_X}$ to $\BrMoncomp_X$. Equivalently,
$\Phi_3^{\mathrm{ch}}$ factors canonically through
$\BrMoncomp_{X,\alpha_X} \to \BrMoncomp_X$ iff $\alpha_X = 0$.
\item \emph{Categorical-H compatibility.} The ambient Hochschild
complex $\mathrm{ChirHoch}^\bullet(\Phi_3^{\mathrm{ch}}(X))$ is
equivalent to the Costello-Li BCOV ambient Hochschild complex,
concentrated in the three-step filtration of Theorem~H.
\end{enumerate}
Consequently, on the subcategory of compact CY$_3$ targets for
which the Costello-Li BCOV transfer is established, the
non-Tannakian stratum of the L2 avatar of
Theorem~\ref{thm:grand-unification-platonic} is in bijection with
the non-formal chain-level CY$_3$ targets in the essential image of
$\Phi_3^{\mathrm{ch}}$.
\end{theorem}

\begin{proof}
\emph{Step 1 (lift the derived category to a chain-level $\infty$-functor).}
On a compact CY$_3$, $D^b(X)$ admits no full exceptional
collection \cite{BondalVdB03}, and the derived-to-bialgebra
construction via semiorthogonal decomposition fails. By
\cite[HA~5.1.2.2, 5.1.4.2]{LurieDAGX}, the $\infty$-category
$\mathrm{Perf}^{\mathrm{CY3}}_\infty(X)$ of perfect complexes on a
CY$_3$ is symmetric-monoidally enriched over
$(\Coalg^{\mathrm{fact}}_{\mathrm{conil}})_\infty$; its underlying
braided monoidal structure lands in $\BrMoncomp_X$ up to a
Brauer-type twist.

\emph{Step 2 (identify the gerbe obstruction).}
The Brauer-type twist is controlled by $H^3(X,\CC^\times)$, the
standard obstruction to strictifying a braided $\infty$-category
to an honest bialgebra \cite{Toen07}, giving
$\alpha_X = [\mathrm{Gerbe}(\Phi_3)]$. The obstruction vanishes iff
$X$ admits an exceptional collection, which fails on compact CY$_3$
\cite{BondalVdB03}. Hence $\Phi_3^{\mathrm{ch}}$ cannot land in
$\BrMoncomp_X$; it lands in the gerbe-twisted
$\BrMoncomp_{X,\alpha_X}$.

\emph{Step 3 (construct the $L_\infty$-structure via BCOV transfer).}
The Costello-Li curved $L_\infty$-quasi-isomorphism
$\mathrm{HKR}_\infty^{\mathrm{curv},\hbar}$ of \cite{CL20}
transfers the Dolbeault-BCOV $L_\infty$-structure on
$\mathrm{PV}^\bullet(X)[[\hbar]]$ to the chain-level
$\infty$-operadic structure on
$\mathrm{Perf}^{\mathrm{CY3}}_\infty(X)$. All-genera BCOV transfer
is established for the quintic $X_5$ (\S~\ref{ssec:BCOV-transfer-X5}
below); the same machinery applies to any compact CY$_3$ whose
Kodaira-Spencer complex admits the Costello-Li gauge, which is the
scope of the theorem. Combining with Step~2 gives the chain-level
$L_\infty$-data of
$\Phi_3^{\mathrm{ch}}(X) \in \BrMoncomp_{X,\alpha_X}$.

\emph{Step 4 (verify the object is well-defined).}
Three conditions on $\Phi_3^{\mathrm{ch}}(X_5)$: (a) compact
generation, from Rouquier-dimension arguments on $X_5$; (b)
conilpotence of the coproduct, inherited from the bar-construction
property of $B^{\mathrm{cat}}$; (c) factorisation compatibility on
$\Confbar(X_5)$, from the Francis-Gaitsgory chiral Koszul
adjunction \cite[Thm~6.3]{FG12} applied to the curved structure of
Step~3. All three are verified on the quintic in
\S~\ref{ssec:BCOV-transfer-X5} and \S~\ref{ssec:explicit-bar-model}
and transport to any compact CY$_3$ within the theorem's scope by
naturality of the BCOV transfer.

\emph{Step 5 (paste to the explicit bar object).}
The $\infty$-functor $\Phi_3^{\mathrm{ch}}$ factors through the
explicit chain model
$B^{\mathrm{cat}}(D^b(X)) \in \BrMoncomp_{X,\alpha_X}$. The
commutative diagram of $\infty$-functors
\begin{equation}
\label{eq:Phi-3-ch-commutative-platonic}
\begin{tikzcd}[column sep=large, row sep=large]
\mathrm{Perf}^{\mathrm{CY3}}_\infty(X)
   \arrow[r, "\Phi_3^{\mathrm{ch}}"]
   \arrow[d, "\mathrm{HKR}_\infty^{\mathrm{curv},\hbar}"']
&
\BrMoncomp_{X,\alpha_X}
   \arrow[d, "\mathrm{pr}_{\alpha_X}"]
\\
\mathrm{PV}^\bullet(X)[[\hbar]]
   \arrow[r, "B^{\mathrm{cat}}"']
&
\BrMoncomp_{X}
\end{tikzcd}
\end{equation}
commutes up to natural equivalence of $\infty$-functors, with the
upper path the L2 realisation of
Theorem~\ref{thm:grand-unification-platonic} and the lower path
the chain-level realisation via BCOV and the explicit bar. The
right vertical $\mathrm{pr}_{\alpha_X}$ is the gerbe-trivialising
projection after tensoring with a local trivialisation of
$\alpha_X$ on an open cover of $X$; commutativity of
diagram~\eqref{eq:Phi-3-ch-commutative-platonic} follows from the
Francis-Gaitsgory universality \cite[Thm~6.3]{FG12} together with
Lurie \cite[HA~5.5.2.9]{LurieDAGX}.

Property~(i) follows from Step~5. Property~(ii) is Step~2
together with the gerbe-trivialising projection of
diagram~\eqref{eq:Phi-3-ch-commutative-platonic}. Property~(iii)
identifies $\mathrm{ChirHoch}^\bullet$ with the Costello-Li BCOV
ambient Hochschild complex on the proved Theorem~H surface, combining
Theorem~\ref{thm:theorem-h-on-koszul-locus} with the BCOV transfer of
Step~3. On the scope of the theorem, the non-Tannakian
stratum of
Corollary~\ref{cor:gu-L2-platonic} is the locus where
$\alpha_X \neq 0$, parametrised by the essential image of
$\Phi_3^{\mathrm{ch}}$.
\end{proof}

The two supporting inscriptions referenced in the proof are
recorded in the next subsections.

The curved-$L_\infty$ quasi-isomorphism

transfers the Dolbeault-BCOV action functional on
$\mathrm{PV}^\bullet(X_5)[[\hbar]]$ to the chain-level

$\mathrm{Perf}^{\mathrm{CY3}}_\infty(X_5)$. The

\subsection{Explicit $B^{\mathrm{cat}}(D^b(X_5))$}
\label{ssec:explicit-bar-model}

The bar construction $B^{\mathrm{cat}}(D^b(X_5))$ is the explicit

\subsection{Local-global gluing of the gerbe-trivialising projection}
\label{ssec:pr-alpha-X-local-global}

The projection $\mathrm{pr}_{\alpha_X} : \BrMoncomp_{X,\alpha_X}
\to \BrMoncomp_X$ appearing in
diagram~\eqref{eq:Phi-3-ch-commutative-platonic} is defined
locally: on an open cover $\{U_i\}$ of $X$ over which
$\alpha_X|_{U_i}$ trivialises, tensoring with a local
trivialisation $\tau_i : \alpha_X|_{U_i} \xrightarrow{\sim} 0$
produces a gerbe-trivialised restriction
$\BrMoncomp_{U_i, \alpha_X|_{U_i}} \xrightarrow{\sim}
\BrMoncomp_{U_i}$. On overlaps $U_i \cap U_j$, the two local
trivialisations differ by a $\mathbb{C}^\times$-gerbe cocycle
$g_{ij} \in Z^1(U_i \cap U_j, \underline{\mathbb{C}^\times})$
whose cohomology class $[\{g_{ij}\}] \in \check H^3(X,
\underline{\mathbb{C}^\times})$ is $\alpha_X$ itself. The
question is whether the locally defined functors glue to an
honest $\infty$-functor on $X$, or only up to Čech-coherent
homotopy with obstruction in the Postnikov tower.

\begin{theorem}[Local-global gluing of $\mathrm{pr}_{\alpha_X}$]
\label{thm:pr-alpha-X-local-global}
\ClaimStatusProvedHere
Let $X$ be a smooth projective variety and
$\alpha_X \in H^3(X, \underline{\mathbb{C}^\times})$ the gerbe
class of
Theorem~\ref{thm:L2-chain-level-gerbe-platonic}. Write the
Bockstein decomposition
\[
   H^3(X, \underline{\mathbb{C}^\times})
   \;\simeq\;
   J^3(X) \oplus H^4(X, \mathbb{Z})_{\mathrm{tors}},
\]
and let $\alpha_X = \alpha_X^{\mathrm{cts}} +
\alpha_X^{\mathrm{tors}}$ denote the decomposition into the
continuous Griffiths part and the torsion part. The
gerbe-trivialising projection $\mathrm{pr}_{\alpha_X}$ glues to a
global $\infty$-functor in the following two regimes.
\begin{enumerate}[label=\textup{(\alph*)}]
\item \emph{Finite-order regime.} If $\alpha_X^{\mathrm{cts}} =
0$, so $\alpha_X = \alpha_X^{\mathrm{tors}}$ of finite order $n$,
then $\mathrm{pr}_{\alpha_X}$ exists as an honest
$\infty$-functor $\BrMoncomp_{X,\alpha_X} \to \BrMoncomp_X$. A
finite étale cover $\pi : \tilde X \to X$ of degree $n$
trivialises $\alpha_X$ globally; Galois descent
\cite[III.3.3.2]{Giraud1971} along $\pi$ produces
$\mathrm{pr}_{\alpha_X}$ as the functor of $\mathrm{Gal}(\tilde
X/X)$-invariants of the fibrewise trivialisation.
\item \emph{Infinite-order regime.} If
$\alpha_X^{\mathrm{cts}} \ne 0$, then $\mathrm{pr}_{\alpha_X}$
exists only as an $\infty$-functor \emph{up to Čech-coherent
homotopy}: for any hypercover $U_\bullet \to X$ refining the
trivialising cover $\{U_i\}$, the restriction of
$\mathrm{pr}_{\alpha_X}$ to $U_\bullet$ extends by Joyal--Lurie
$\infty$-sheaf descent \cite[HTT~6.5.3.12]{LurieHTT}, but the
obstruction to strict equality with a global functor lives in the
infinite Postnikov tower
\[
   \mathrm{obs}(\mathrm{pr}_{\alpha_X})
   \;\in\;
   \lim_n \pi_n\!\left(
      \mathrm{Map}_{\infty\text{-}\mathrm{Fun}}\!\left(
         U_\bullet, \BrMoncomp_X
      \right)
   \right)
\]
with $\pi_n \ne 0$ at arbitrarily large $n$ whenever
$\alpha_X^{\mathrm{cts}}$ has infinite order in $J^3(X)$.
\end{enumerate}
In both regimes,
diagram~\eqref{eq:Phi-3-ch-commutative-platonic} commutes up to
the natural equivalence of $\infty$-functors produced by
$\mathrm{pr}_{\alpha_X}$; strict commutativity on the nose
requires (a).
\end{theorem}

\begin{proof}
\emph{Regime (a).} Finite-order $\alpha_X \in H^3(X,
\underline{\mathbb{C}^\times})_{\mathrm{tors}}$ corresponds via
the Bockstein to a class in $H^4(X, \mathbb{Z})_{\mathrm{tors}}$
of order $n$. Giraud's classification
\cite[III.2.3.1, III.3.3.2]{Giraud1971} produces a finite étale
$\mathbb{Z}/n$-Galois cover $\pi : \tilde X \to X$ with
$\pi^\star \alpha_X = 0 \in H^3(\tilde X,
\underline{\mathbb{C}^\times})$. On $\tilde X$ the
gerbe-trivialising projection is a global functor
$\widetilde{\mathrm{pr}} :
\BrMoncomp_{\tilde X, \pi^\star \alpha_X} \simeq
\BrMoncomp_{\tilde X}$ (Azumaya-Morita equivalence
\cite[Thm~1.1]{Toen2006}). The functor
$\mathrm{pr}_{\alpha_X}$ on $X$ is reconstructed as the
$\mathrm{Gal}(\tilde X/X)$-equivariant functor
$\widetilde{\mathrm{pr}}^{\mathrm{Gal}}$; Galois descent for
$\infty$-sheaves of presentable $\infty$-categories
\cite[HA~1.3.1.7]{LurieDAGX} identifies
$\widetilde{\mathrm{pr}}^{\mathrm{Gal}}$ with an honest
$\infty$-functor on $X$. The descent is strict because the Čech
nerve of the finite cover has finite depth.

\emph{Regime (b).} Infinite-order $\alpha_X^{\mathrm{cts}} \ne 0$
lies in the connected component $J^3(X)$ of $H^3(X,
\underline{\mathbb{C}^\times})$; no finite cover trivialises
$\alpha_X$ \cite[Thm~6.3]{Bridgeland2011}. The Čech nerve of any
trivialising cover $\{U_i\}$ refines to a hypercover $U_\bullet
\to X$ of unbounded simplicial depth, and the Joyal--Lurie
$\infty$-topos descent theorem \cite[HTT~6.5.3.12]{LurieHTT}
applied to the $\infty$-sheaf $\BrMoncomp_{(-)}$ produces an
$\infty$-functor $\mathrm{pr}_{\alpha_X}^{\infty} :
\BrMoncomp_{X, \alpha_X} \to \BrMoncomp_X$ as the totalisation
of the cosimplicial object $\{\mathrm{pr}_{\alpha_X|_{U_\bullet}}\}$.

To identify the obstruction to strict equality, apply Toën's
descent theorem \cite[Cor.~3.3]{Toen2006} to the presentable
$\infty$-category $\BrMoncomp_{X, \alpha_X}$. The Čech-coherent
homotopy witnessing commutativity of
$\mathrm{pr}_{\alpha_X}^{\infty}$ with the local trivialisations
is an element of the cosimplicial mapping space
$\mathrm{Tot}(\mathrm{Map}(U_\bullet, \BrMoncomp_X))$, whose
Postnikov tower has $\pi_n$-part controlled by $H^{n+3}(X,
\underline{\mathbb{C}^\times})$-twisted sheaf cohomology. A
connected class $\alpha_X^{\mathrm{cts}} \in J^3(X)$ of infinite
order produces non-zero $\pi_n$ at arbitrarily large $n$ by the
Eilenberg--Moore spectral sequence for hypercover totalisations
\cite[V.2]{BousfieldKan1972}. Hence the obstruction lives in the
infinite Postnikov tower, and strict commutativity on the nose
fails.

Commutativity of
diagram~\eqref{eq:Phi-3-ch-commutative-platonic} up to natural
equivalence of $\infty$-functors is the statement that the
cosimplicial obstruction is coherent with the Francis--Gaitsgory
\cite[Thm~6.3]{FG12} and Lurie \cite[HA~5.5.2.9]{LurieDAGX}
universal property of the chiral bar; this coherence is
unconditional, as the natural equivalence is itself a
2-morphism in the $\infty$-category of $\infty$-functors, living
one Postnikov level below the strict-equality obstruction.
\end{proof}

\begin{remark}[Strict compact CY$_3$ and the quintic]
\label{rem:pr-alpha-X-quintic}

Vol~III).
\end{remark}
