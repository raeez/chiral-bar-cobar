
The bar complex of the Heisenberg algebra $\mathcal{H}_\kappa$
reproduces free-boson Feynman rules at all genera
(Theorem~\ref{thm:bar-cobar-path-integral-heisenberg}).
For interacting theories the identification is proved at genus~$0$
for Kac--Moody and $\mathcal{W}$-algebras (Vol~I); beyond that locus
it is conjectural (Conjecture~\ref{conj:bar-cobar-path-integral}).

The Feynman--bar dictionary
(Theorem~\ref{thm:feynman-bar-bridge}) identifies:
propagator $= \eta_{ij}$, vertex $=$ collision residue,
loop $=$ genus increment, UV divergence $=$ FM boundary stratum.
This section verifies the dictionary for the free boson at each
genus.

\begin{construction}[Feynman transform as stable-graph MC decomposition]
\label{constr:feynman-stable-graph}
\index{Feynman transform!stable-graph decomposition|textbf}
\index{stable graph!MC decomposition|textbf}
\index{MC element!stable-graph sum}
The Feynman transform $\mathrm{FT}(\mathrm{FCom})$ of the commutative modular operad decomposes the MC element $\Theta_\cA$ as a stable-graph sum:
\[
\Theta_\cA^{(g,n)} = \sum_{\Gamma \in \mathsf{Gr}^{\mathrm{st}}_{g,n}} \frac{1}{|\mathrm{Aut}(\Gamma)|} \cdot W_\Gamma(\cA),
\]
where $W_\Gamma(\cA)$ is the $\Gamma$-amplitude (Construction~\ref{constr:stable-graph-decomposition}).  Feynman diagrams in physical QFT are the genus-$0$ tree-level truncation of this sum; loop corrections are the higher-genus contributions.  The MC equation $D\Theta + \tfrac{1}{2}[\Theta,\Theta] = 0$ is the codimension-$2$ face cancellation in $\overline{\mathcal{M}}_{g,n}$ (Remark~\ref{rem:mc-codimension-two}).
\end{construction}

\section{The free boson at genus~\texorpdfstring{$0$}{0}}

The Heisenberg vertex algebra $\mathcal{H}_\kappa$ has OPE
$a(z)\,a(w) \sim \kappa\,(z-w)^{-2}$.  The Feynman rules are:
propagator $G(z,w) = \kappa/(z-w)^2$; no vertices (free theory);
each insertion of $a(z_i)$ is an external leg.

\begin{center}
\begin{tikzpicture}
\node at (0,0) {$z_1$};
\node at (3,0) {$z_2$};
\draw[thick] (0.3,0) -- (2.7,0);
\node at (1.5, 0.3) {$\frac{\kappa}{(z_1-z_2)^2}$};
\end{tikzpicture}
\end{center}

The genus-$0$ bar complex is
$\bar{B}^{(0)}_n = \int_{C_n(\mathbb{P}^1)} \omega \otimes a(z_1) \otimes \cdots \otimes a(z_n)$,
with differential computing tree-level factorization:
\begin{equation}
d^{(0)}[a \otimes a] = \text{Res}_{z_1 \to z_2} \frac{a(z_1) a(z_2)}{z_1 - z_2}.
\end{equation}
The propagator $\kappa(z_1-z_2)^{-2}$ combined with
$\eta_{12} \sim (z_1-z_2)^{-1}dz_1$ yields a triple pole whose
residue vanishes: $\text{Res}_{z_1 \to z_2} (z_1-z_2)^{-3} = 0$.

\section{The free boson at genus~\texorpdfstring{$1$}{1}}

The tadpole diagram (self-contraction on an elliptic curve
$E_\tau = \mathbb{C}/(\mathbb{Z} + \tau\mathbb{Z})$) gives:
\begin{equation}
A_{\text{tadpole}}^{(E_\tau)} = \kappa \sideset{}{'}\sum_{(n,m) \in \mathbb{Z}^2} \frac{1}{(n + m\tau)^2}
= \kappa \cdot E_2(\tau),
\end{equation}
up to normalization.  The modular anomaly
$E_2(-1/\tau) = \tau^2 E_2(\tau) + 6\tau/(\pi i)$
is the one-loop quantum correction to modularity, proportional
to~$\kappa$.

The tadpole corresponds to $[\text{Tr}(a \otimes a)]^{(1)} \in \bar{B}^{(1)}_1$,
with $d^{(0)}[\text{Tr}(a \otimes a)]^{(1)} = \kappa \cdot [1]^{(1)}$.

\subsection{Loop-genus correspondence}

\begin{theorem}[Loop-genus correspondence; \ClaimStatusProvedElsewhere{} \cite{costello-renormalization}]
\label{thm:loop-genus-correspondence}
A connected Feynman diagram $\Gamma$ with $V$ vertices, $E$ edges,
and $F$ faces has loop number $L = E - V + 1$.  The genus of the
ribbon graph satisfies
$\chi = V - E + F = 2 - 2g$, hence $g = (L + 1 - F)/2 \leq L/2$,
with equality when $F = 1$.
\end{theorem}

The full free boson correlation function expands as
$\langle a(z_1) \cdots a(z_n) \rangle_{\text{all genera}}
= \sum_{g=0}^\infty \kappa^{n/2+g-1}
\langle a(z_1) \cdots a(z_n) \rangle^{(g)}_{\text{conn.}}$,
and the bar filtration weights genus~$g$ by $\kappa^g$.

\section{One-loop partition function}

Free boson on $E_\tau$: the Gaussian path integral gives
$Z_{E_\tau} = (\mathrm{Im}\,\tau)^{-1/2} |\eta(\tau)|^{-2}$,
which is modular invariant as a density on
$\mathcal{M}_{1,0}$.  The bar complex interpretation:
$Z_{E_\tau} = H_0(\bar{B}^{(1)})$; the modular anomaly
$-c\log|\tau|$ is the scalar trace
$\kappa(\cA)\cdot\deg(\lambda_1) = c/24$
(Theorems~\ref{thm:genus-universality}
and~\ref{thm:explicit-theta}).

\section{Path integral = bar complex}

\begin{theorem}[Bar complex = path integral for the free boson; \ClaimStatusProvedHere]\label{thm:bar-cobar-path-integral-heisenberg}
For the Heisenberg vertex algebra $\mathcal{H}_\kappa$
\textup{(}free boson at level~$\kappa \neq 0$\textup{)} on a
compact Riemann surface $\Sigma_g$ of genus~$g \geq 0$:
\begin{equation}\label{eq:bar-partition-heisenberg}
Z_g^{\mathcal{H}_\kappa}(\Sigma_g)
= H^0(\bar{B}^{(g)}(\mathcal{H}_\kappa)) = \mathbb{C}
\end{equation}
\begin{equation}\label{eq:bar-npoint-heisenberg}
\langle \alpha(z_1) \cdots \alpha(z_n) \rangle_{\Sigma_g}
= H^n(\bar{B}^{(g)}(\mathcal{H}_\kappa))
/ H^0(\bar{B}^{(g)}(\mathcal{H}_\kappa))
\end{equation}
\end{theorem}

\begin{proof}
\emph{Step~1: Bar complex cohomology.}
By Theorem~\ref{thm:heisenberg-higher-genus}, the bar complex
cohomology of $\mathcal{H}_\kappa$ on~$\Sigma_g$ is:
\[
H^n(\bar{B}^{(g)}(\mathcal{H}_\kappa)) = \begin{cases}
\mathbb{C} & n = 0, \\
\mathbb{C}^{2g} & n = 1, \\
\mathbb{C} \oplus \mathbb{C} \cdot c_\kappa^{(g)} & n = 2, \\
0 & n > 2.
\end{cases}
\]

\emph{Step~2: Free boson path integral.}
The path integral is Gaussian with normalized partition function
$Z_g = 1$ (zero-mode volume removed), matching
$H^0 = \mathbb{C}$.  By Wick's theorem:
$H^1 \cong H^1(\Sigma_g, \mathbb{C})$ (zero modes);
the $2$-point function has residue $\kappa$ matching the bar
differential; $H^n = 0$ for $n > 2$ (free theory).

\emph{Step~3: Factorization homology.}
By Theorem~\ref{thm:verdier-AF-compat},
$H^*(\bar{B}^{(g)}(\mathcal{H}_\kappa))$ computes the
factorization homology $\int_{\Sigma_g}\mathcal{H}_\kappa$,
which for the free boson equals the Gaussian BV
cohomology~\cite{CG17}.
\end{proof}

\begin{conjecture}[Bar-cobar as worldsheet path integral: general case; \ClaimStatusConjectured]\label{conj:bar-cobar-path-integral}
For a general chiral algebra $\cA$ on $\Sigma_g$, the bar complex computes:
\begin{equation}
\exp\left( \sum_{g,n} \frac{1}{n!} \int_{C_n(\Sigma_g)} \langle a(z_1) \cdots a(z_n) \rangle_g \right) = \det(\mathbf{1} + \bar{B}(\cA))
\end{equation}
where the right side is the Fredholm determinant of the bar complex.
Proved for $\mathcal{H}_\kappa$ at all genera
\textup{(}Theorem~\textup{\ref{thm:bar-cobar-path-integral-heisenberg})};
conjectural for interacting $\cA$.
Contributing to Conjecture~\ref{conj:master-bv-brst}.
\end{conjecture}

\begin{remark}[Shadow depth decomposition in the Feynman picture]
\label{rem:feynman-depth-decomposition}
The loop-genus correspondence (Theorem~\ref{thm:loop-genus-correspondence})
underlies the depth decomposition $d = 1 + d_{\mathrm{arith}} + d_{\mathrm{alg}}$
of Theorem~\ref{thm:depth-decomposition} (Chapter~\ref{chap:arithmetic-shadows}):
$d_{\mathrm{arith}}$ counts critical lines of the spectral $L$-function
(Feynman-integral periods) and $d_{\mathrm{alg}}$ records the OPE vertex
complexity visible as higher-loop skeleton subdivergences.
\end{remark}
