\begin{abstract}
The fourth slot of the Open Beilinson tower is occupied by the
algebra of line and brane operators on the factorisation dg-category
$\cC^{\mathrm{op}}$ over a tangential log curve $(X, D, \tau)$. The
identical slot in the CY quadrant is occupied by the chiral quantum
group $\GX = D(\Yplus(X))$, the Drinfeld double of the chiral Hall
algebra of the cohomological Hall algebra of $\cC^{\mathrm{op}}$. We
state the level-$4$ vertical holographic equivalence on
$(X, D, \tau)$ as the composite of five named operations on the
positive half: the Hall pairing on
$\Yplus(X)\otimes \Yplus(X)^{*\mathrm{cop}}$; weight completion in
the Hall topology; passage to an integral form preserving the
Cartan; transport along the Maulik--Okounkov stable-envelope
cobordism; descent from the formal locus on $X^\circ$ to the global
factorisation site of $(X, D, \tau)$. The composite is the unique
strict $E_2$-equivalence on the modular-Koszul, integral, completed,
stable-envelope-transported, descended sublocus. Each step carries a
named hypothesis package; failure to display any one of the five
collapses MA-$8$. The frontier obstruction at $g\geq 1$ is the elliptic
2-cocycle
$\obs^{(1)}_{\mathrm{double}}\in
H^{2}(\grtell, \sptt(\cA)\otimes\sptt(\cA^!))$,
whose vanishing is biconditional to $\rmax(\cA)=2$ (Class $\mathbf G$
only).
\end{abstract}

\section{The level-4 vertical, stated}
\label{sec:level-4-vertical-statement}

\begin{convention}[Five-object firewall, level-$4$ refinement]
\label{conv:level-4-five-object}
At level~$4$ four typed objects must be kept distinct:
\[
\Yplus(X) \;\subsetneq\; \Yplus(X)\bowtie \Yplus(X)^{*\mathrm{cop}}
\;=\; D(\Yplus(X)) \;=\; \GX,
\]
together with the line/brane operator algebra
$\Hbar^{\mathrm{line}}_\cC$ on $\cC^{\mathrm{op}}$. The first three
are objects in the CY quadrant; the fourth is the level-$4$ Open
object. Calling $\Yplus(X)$ "the chiral quantum group", or
identifying $\Yplus(X)$ with the line algebra
$\Hbar^{\mathrm{line}}_\cC$ before passing through the four-step
bridge below, is the standard form of MA-$8$.
\end{convention}

\begin{theorem}[Level-$4$ vertical equivalence]
\label{thm:level-4-vertical}
Let $(X,D,\tau)$ be a tangential log curve and $\cC^{\mathrm{op}}$
its factorisation dg-category at level~$0$. Fix a vacuum
$b\in\cC^{\mathrm{op}}$ and let $\cA = A_b = \End_\cC(b)$ be the
chart algebra at level~$1$. Assume the following hypothesis package:
\begin{enumerate}[label=\textup{(H\arabic*)},leftmargin=*]
\item \emph{Modular-Koszul locus} for $\cA$
(Theorem~\textup{A} on $\mathrm{Kosz}(X)$);
\item \emph{Hall-pairing convergence}: the residue pairing
$\langle{-},{-}\rangle_{\mathrm{Hall}}\colon
\Yplus(X)\otimes \Yplus(X)^{*\mathrm{cop}}\to k((\hbar))$
is non-degenerate after completion;
\item \emph{Weight completion}: the Hall product extends along the
weight-graded $J$-adic Mittag--Leffler tower
$\{\Yplus(X)/J^{n}\}_{n\geq 0}$ to the Pro-object
$\widehat{\Yplus}(X) = \varprojlim_n \Yplus(X)/J^{n}$;
\item \emph{Integral form} $\Yplus(X)_{\mathbb Z}\subset
\Yplus(X)\otimes_{\mathbb Z}k$ preserving the Cartan after restriction
of scalars (Drinfeld--Lusztig integral form, with PBW basis indexed
by positive root data of $\cA$);
\item \emph{Stable-envelope transport}: a Maulik--Okounkov stable
envelope $\Stab\colon
K_T(\mathrm{Fix}^T)\to K_T(\Yplus(X)\text{-mod}^{\heartsuit})$
matched with the Hall pairing under the equivariant Euler-class
normalisation;
\item \emph{Global descent}: the formal-disk presentation of
$\Yplus(X)$ glues along $\check{C}\bigl(\cC^{\mathrm{op}}\bigr)$ on
the factorisation site of $(X,D,\tau)$, with quantum master equation
satisfied on global sections and locality of the descent obstruction.
\end{enumerate}
Then there is a strict $E_2$-equivalence of factorisation algebras
on $(X,D,\tau)$
\begin{equation}\label{eq:level-4-vertical}
\Hbar^{\mathrm{line}}_\cC \;\simeq\;
\mathrm{LineMod}\bigl(\GX\bigr) \;=\;
\mathrm{LineMod}\bigl(D\bigl(\Yplus(X)\bigr)\bigr),
\end{equation}
under which the transverse multiplication of line operators in
$\cC^{\mathrm{op}}$ is the Drinfeld pairing on $\GX$, and the
half-braiding on the boundary representation category is the
universal $R$-matrix of $\GX$.
\end{theorem}

This is the Open ↔ CY level-$4$ holographic vertical: the line/brane
operator algebra at level~$4$ on the Open side is the level-$3$
Drinfeld-double output of the level-$4$ Hall input $\Yplus(X)$ on the
CY side, modulo the four named operations of completion, integral
form, stable envelope, and descent.

\section{The five operations}
\label{sec:five-operations}

\begin{construction}[Hall pairing]
\label{constr:hall-pairing}
On the modular-Koszul locus, the binary collision residue
$r_\cA(z) = \operatorname{Res}^{\mathrm{coll}}_{0,2}(\Theta_\cC)$ of
the master concordance specialises to the Hall pairing
\[
\langle\xi_+, \xi_-\rangle_{\mathrm{Hall}}
\;=\; \operatorname{Res}_{z=0}\,\Tr\bigl(\xi_+(z)\,\xi_-(0)\bigr)\,dz,
\qquad \xi_\pm \in \Yplus(X)^{\pm},
\]
non-degenerate after completion. Convergence relies on Hall-shuffle
finiteness of the K-theoretic CoHA (Schiffmann--Vasserot 2013,
Theorem~3.1) and the modular-Koszul condition that the
Maurer--Cartan twisting datum $\Theta_\cC$ has finite-dimensional
graded pieces.
\end{construction}

\begin{construction}[Weight completion]
\label{constr:weight-completion}
The Hall product is graded by the weight lattice of $\cA$ and is
finite on each fixed weight class but not on $\Yplus(X)$ as a whole;
the Drinfeld-double construction requires inverting the universal
$R$-matrix, which mixes infinitely many weight classes. Completion
is the passage to
$\widehat{\Yplus}(X) = \varprojlim_n \Yplus(X)/J^{n}$ along the
weight-graded $J$-adic filtration with Mittag--Leffler control
($\varprojlim^{1}$-vanishing). Skipping completion produces a
formally divergent $R$-matrix and is the source of the
"$\Yplus = \GX$" form of MA-$8$.
\end{construction}

\begin{construction}[Integral form]
\label{constr:integral-form}
The Drinfeld--Lusztig integral form
$\Yplus(X)_{\mathbb Z}\subset \widehat{\Yplus}(X)\otimes_{\mathbb Z}k$
is the $\mathbb Z[\hbar,\hbar^{-1}]$-submodule generated by divided
powers $E_\alpha^{(n)}/[n]_q!$ of root vectors and Cartan
$K_\alpha^{\pm 1}$. Preservation of the Cartan after restriction of
scalars to $k$ requires the PBW basis to be indexed by positive root
data of $\cA$; failure produces a non-Hopf-algebra integral
$\hbar$-completion which kills the Drinfeld pairing.
\end{construction}

\begin{construction}[Stable-envelope transport]
\label{constr:stable-envelope}
For toric CY$_3$ targets and for $K3\times E$, the Maulik--Okounkov
stable-envelope cobordism
$\Stab\colon K_T(\mathrm{Fix}^T)\to K_T(\Yplus(X)\text{-mod})$
transports the Hall pairing of
Construction~\ref{constr:hall-pairing} to the geometric pairing on
$T$-fixed points. The Maulik--Okounkov $R$-matrix
$R^{\mathrm{MO}}(u) = \Stab^{-1}_{\mathfrak C_+}\circ
\Stab_{\mathfrak C_-}$ from chamber comparison agrees, on the modular
Koszul locus, with the chiral $R$-matrix of the level-$2$ bar
construction up to $\mathrm{GRT}_1(\mathbb Q)$-torsor (Vol~III,
Theorem~\textup{thm:mo-chiral-rmatrix}). Stable-envelope transport
without the $\mathrm{GRT}_1$-torsor matching is the residue path's
genuine open frontier off the $\C^{3}, K3\times E$ locus.
\end{construction}

\begin{construction}[Global descent]
\label{constr:global-descent}
The formal-disk Drinfeld double extends to a global factorisation
algebra on $(X, D, \tau)$ provided the four-mechanism
Costello--Gwilliam package: descent on the factorisation site, the
quantum master equation on global sections, anomaly cancellation
through matched-conventions target data, and locality of the descent
obstruction. The output is the global chiral quantum group $\GX$ on
$(X, D, \tau)$ as a factorisation algebra, with line-operator
representation $\mathrm{LineMod}(\GX) = \Hbar^{\mathrm{line}}_\cC$.
\end{construction}

\begin{remark}[The MA-$8$ collapse pattern]
\label{rem:level-4-MA-8}
Each of \textup{Constructions \ref{constr:hall-pairing}--\ref{constr:global-descent}} is its own theorem with hypothesis package; the
identification $\Yplus(X) = \GX$ collapses all five into one
unsupported equality. The collapses $\mathrm{CoHA}(\C^3) = \mathcal
W_{1+\infty}$, $\Yplus = $ "the chiral quantum group", and
"line operators are the CoHA" are three forms of MA-$8$ at distinct
hypothesis points. The cure is the five-step bridge above.
\end{remark}

\section{Frontier-five alignment: the elliptic obstruction}
\label{sec:level-4-f5-alignment}

\begin{theorem}[F5 obstruction at $g\geq 1$;
biconditional to $\rmax(\cA)=2$]
\label{thm:level-4-F5-obstruction}
Let $\cA$ be a chart algebra in one of the archetype classes
$\mathbf{G}, \mathbf{L}, \mathbf{C}, \mathbf{M}, \mathbf{B}$ with
shadow depths
$\rmax(\cA)\in\{2,3,4,\infty,5\}$ respectively. Over
$\overline{M}_{g,n}$ at $g\geq 1$, the global Drinfeld double
$D(\Yplus(X))$ exists as a factorisation algebra over
$(X, D, \tau)$ if and only if a class
\[
\obs^{(1)}_{\mathrm{double}}
\;\in\; H^{2}\bigl(\grtell,\,\sptt(\cA)\otimes\sptt(\cA^!)\bigr)
\]
vanishes. The class is represented by the elliptic two-cocycle
$\omega_{\mathrm{ell}}(\tau) = \wp(\tau)\cdot\langle{-},{-}\rangle_{\sptt}$,
with $\wp$ the Weierstrass $\wp$-function and $\sptt$ the
shadow-depth Lie algebra. We have
\[
\obs^{(1)}_{\mathrm{double}}=0
\quad\Longleftrightarrow\quad
\rmax(\cA) = 2,
\]
i.e.\ vanishing holds iff $\cA$ is in Class $\mathbf G$. Classes
$\mathbf{L}, \mathbf{C}, \mathbf{M}, \mathbf{B}$ are obstructed at
$g\geq 1$ via Enriquez's elliptic
$\mathrm{GRT}_1^{\mathrm{ell}}(\mathbb Q)$-cohomology with non-zero
cocycle.
\end{theorem}

\begin{proof}[Proof structure]
At $g=0$ the Drinfeld--Kohno solution of the KZ equation
trivialises the obstruction (Drinfeld 1989/1991,
$\mathrm{GRT}_1^{\mathrm{fin}}(\mathbb Q)$-torsor). Globalisation to
$g=1$ runs through Enriquez's elliptic associator with
$H^{2}(\grtell,\,\sptt(\cA)\otimes\sptt(\cA^!))$
classifying the obstructions; the elliptic cocycle
$\omega_{\mathrm{ell}}(\tau)$ is the unique non-trivial class for
$\rmax(\cA)\geq 3$ by Calaque--Ene\-quez--Etingof~2009. The
shadow-depth biconditional follows from the universal chain-homotopy
identity $h_{\cA}=h_{\mathrm{LV}}/\mathcal N(\cA)$ of Vol~I
(with $\mathcal N(\cA)=\kappa(\cA)+\kappa^!_{\mathrm{LV}}(\cA)$ the
level-$2$ Verdier-Ran-on-bar norm, not the algebra-level
$K^\kappa$), which
identifies the elliptic obstruction class with the higher-Massey
elliptic shadow at depth $\rmax(\cA)$. Class $\mathbf G$ has
$\rmax=2$ and trivial higher Massey; the obstruction class is the
zero element. Classes $\mathbf{L}, \mathbf{C}, \mathbf{M}, \mathbf{B}$
have $\rmax\geq 3$ and non-trivial Massey lift to elliptic
$\wp$-cocycle. Full obstruction at $g\geq 2$ is the open frontier of
F$5$.
\end{proof}

\begin{corollary}[Stratification of the level-$4$ vertical]
\label{cor:level-4-stratification}
The level-$4$ vertical
equivalence~\eqref{eq:level-4-vertical} holds globally over
$\overline{M}_{g,n}$ at $g=0$ unconditionally; at $g=1$ for all
five archetypes provided the Enriquez elliptic associator is
chosen; at $g\geq 2$ unconditionally only for Class $\mathbf G$,
and conditionally on F$5$ resolution for the remaining four
classes.
\end{corollary}

\section{Hypothesis package and heal path}
\label{sec:level-4-hypothesis-heal}

\begin{remark}[Hypothesis package]
\label{rem:level-4-hypothesis-package}
The level-$4$ vertical equivalence carries the hypothesis package
\textup{(H1)--(H6)} of
Theorem~\ref{thm:level-4-vertical}. Each $H_i$ has its own
non-collapsing companion theorem:
\begin{itemize}[leftmargin=*]
\item (H1) Theorem~A on $\mathrm{Kosz}(X)$
(Vol~I, modular-Koszul locus);
\item (H2) Hall-pairing convergence
(Schiffmann--Vasserot~2013 Theorem~3.1; Vol~III
Theorem~\textup{thm:drinfeld-center-coha} part (i));
\item (H3) Weight-completion Mittag--Leffler
($J$-adic, $\varprojlim^{1}$-vanishing per
Vol~I~Appendix~Q);
\item (H4) Drinfeld--Lusztig integral form
(Lusztig~1990; Drinfeld~1985 \S 9 for the rational case);
\item (H5) MO stable-envelope cobordism
(Maulik--Okounkov~2012 Theorem~10.1.1; chamber-compatibility
Vol~III~Theorem~\textup{thm:mo-chiral-rmatrix});
\item (H6) Costello--Gwilliam globalisation:
descent + QME + anomaly + locality.
\end{itemize}
\end{remark}

\begin{remark}[Heal path]
\label{rem:level-4-heal-path}
The standard collapse "$\Yplus(X) = \GX$" is repaired by:
\textup{(i)} verify that the Hall pairing of
Construction~\ref{constr:hall-pairing} is convergent on the
modular-Koszul locus; \textup{(ii)} apply weight completion of
Construction~\ref{constr:weight-completion}; \textup{(iii)} pass to
the integral form of Construction~\ref{constr:integral-form};
\textup{(iv)} transport via the stable envelope of
Construction~\ref{constr:stable-envelope};
\textup{(v)} descend globally via
Construction~\ref{constr:global-descent}. Skipping any step at
$g\geq 1$ produces $\obs^{(1)}_{\mathrm{double}}\neq 0$ outside Class
$\mathbf G$ and the collapse fails. Inscribing the four-step bridge
explicitly is the unique heal path.
\end{remark}

\begin{remark}[Three verification paths for the F5 biconditional]
\label{rem:level-4-three-paths}
The biconditional
$\obs^{(1)}_{\mathrm{double}}=0 \Leftrightarrow \rmax(\cA)=2$ is
verified by three independent paths:
\begin{enumerate}[label=\textup{(\roman*)},leftmargin=*]
\item \emph{Universal chain-homotopy.}

identically, for $\mathbf{L}, \mathbf{C}, \mathbf{M}, \mathbf{B}$ it

associator; the F5 obstruction is the unique higher-Massey extension

dictionary, to Face~5's $\Phi_{1,5}$-component, which is non-zero
exactly when $\rmax\geq 3$.
\end{enumerate}
\end{remark}

\section{Closure of the level-$4$ vertical}
\label{sec:level-4-closure}

The level-$4$ Open object — the line and brane operator algebra on
$\cC^{\mathrm{op}}$ over $(X,D,\tau)$ — is the level-$3$ Drinfeld
double $\GX = D(\Yplus(X))$ of the level-$4$ CoHA Hall input
$\Yplus(X)$, in the unique presentation produced by the five-step
bridge. The bridge's hypothesis package is non-degenerate: each step
is its own theorem, and skipping any step is the standard form of
MA-$8$. At $g\geq 1$ the global descent obstruction is the elliptic
class $\obs^{(1)}_{\mathrm{double}}\in H^{2}(\grtell, \sptt(\cA)\otimes\sptt(\cA^!))$, which vanishes biconditionally to $\rmax(\cA)=2$
and is the F5 open frontier for the four obstructed classes.

The level-$4$ vertical closes the Beilinson-tower hierarchy of
Vertical equivalences between Vol~I (Open) and Vol~III (CY): together
with the level-$0$ factorisation dg-cat~$\leftrightarrow$~6d hCS
quartic anomaly equivalence, the level-$1$ chart $A_b$~$\leftrightarrow$~$\Phi_d^{\mathrm{FA}}$ equivalence, the level-$2$
bar~$\leftrightarrow$~$\mathrm{Sp}^{\mathrm{ch}}_{\Sigma_{d-1}, C}$
equivalence, the level-$3$ derived centre~$\leftrightarrow$~$\mathsf{SC}^{\mathrm{ch,top}}$-brace equivalence (Vol~II), and the
level-$5$ modular trace~$\leftrightarrow$~$\kappa_{\mathrm{BKM}}(\Phi_N) = c_N(0)/2$ equivalence (Vol~III), this completes the
Open ↔ CY holographic dictionary at every level of the Beilinson
tower with named reconstruction theorem and named hypothesis package
at each step.
