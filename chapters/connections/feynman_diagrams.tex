\chapter{Feynman diagram interpretation of bar-cobar duality}
\label{ch:feynman}
\index{Feynman diagram|textbf}

Bar operations correspond to off-shell amplitudes, cobar operations to on-shell propagators, and the duality itself is the residue-distribution pairing computing S-matrix elements.  The treatment follows Costello \cite{costello-renormalization}.

\section{Feynman diagrams in chiral field theory}

\subsection{Basic setup: fields, propagators, and vertices}

\begin{definition}[Chiral field theory data]
\index{propagator!Feynman}
A chiral field theory on a curve $X$ consists of:
\begin{enumerate}
\item \emph{Fields.} A chiral algebra $\mathcal{A}$ with local operators
$\phi^a(z)$, each with conformal weight $h_a$

\item \emph{Action.} A local functional
$$S[\phi] = \int_X \left[\frac{1}{2}\phi\, \bar{\partial} \phi + V(\phi)\right]$$
where $\bar{\partial}$ is the Dolbeault operator (giving a chiral/holomorphic theory, not the real Laplacian $\Box$) and $V$ encodes interactions

\item \emph{Propagator.} The chiral (holomorphic) two-point function
$$\langle \phi^a(z) \phi^b(w) \rangle_0 = \delta^{ab} P(z,w)$$
where $P(z,w) = \delta_{h_a, h_b} \cdot (z-w)^{-2h_a}$ at genus~0 (nonzero only for fields of equal conformal weight, by the conformal Ward identity; e.g., $(z-w)^{-2}$ for weight-$1$ bosons, $(z-w)^{-1}$ for weight-$\frac{1}{2}$ fermions); at higher genus, these are replaced by the appropriate meromorphic propagator (prime form for bosons, Szeg\H{o} kernel for fermions).  Note: the chiral propagator is \emph{holomorphic}, not the real Green's function $-\log|z-w|^2$

\item \emph{Vertices.} Interaction terms from $V(\phi)$ determining the
chiral algebra structure
\end{enumerate}
\end{definition}

\begin{example}[Free boson]
The free boson has:
\begin{itemize}
\item Field: $\alpha(z)$ with $h=1$
\item Propagator: $\langle \alpha(z)\alpha(w) \rangle = (z-w)^{-2}$
\item No vertices (free theory)
\end{itemize}

Denoting the free boson (Heisenberg) chiral algebra by $\mathcal{A} = \mathcal{H}_1$, the bar complex:
$$\bar{B}^n(\mathcal{A}) = \Omega^*(\overline{C}_{n+1}(X), \mathcal{A}^{\boxtimes (n+1)})$$
encodes $n$-point off-shell correlation functions.
\end{example}

\subsection{Worldline formalism and configuration spaces}

\begin{definition}[Worldline representation]\label{def:worldline-intro}
A Feynman diagram with $V$ vertices, $E$ edges, and $L$ loops corresponds to:
\begin{itemize}
\item \emph{Worldline graph.} $\Gamma$ with vertex set $V$ and edge set $E$
\item \emph{Configuration space point.} $(z_1,\ldots,z_V) \in C_V(X)$
(positions of vertices)
\item \emph{Propagators.} Each edge $e=(i,j)$ contributes $G(z_i,z_j)$
\item \emph{Vertices.} Each vertex contributes an interaction term from $V(\phi)$
\end{itemize}

The amplitude is:
$$A_\Gamma = \int_{C_V(X)} \left[\prod_{e \in E} G(z_i,z_j)\right]
\left[\prod_{v \in V} V_v\right] \prod_i dz_i$$
(In a full 2d CFT one integrates against $d^2z_i$; in the chiral/holomorphic theory, only the holomorphic measure $dz_i$ appears.)
\end{definition}

\begin{remark}[Connection to bar complex]
\index{Feynman diagram!and bar complex}
The bar complex element:
$$\omega_\Gamma \in \bar{B}^{V-1}(\mathcal{A})$$
is precisely the \emph{integrand} of the Feynman amplitude before integration. 
The logarithmic differential forms encode the propagator singularities:
$$\eta_{ij} = d\log(z_i-z_j) \sim \frac{dz_i-dz_j}{z_i-z_j} \sim G(z_i,z_j)^{-1}dG$$
\end{remark}

\subsection{Tree vs.\ loop decomposition}

\begin{definition}[Loop number]
A Feynman diagram $\Gamma$ with $V$ vertices, $E$ edges, and $C$ connected 
components has loop number:
$$L(\Gamma) = E - V + C$$

This is the first Betti number $b_1(\Gamma)$ of the graph.
\end{definition}

\begin{theorem}[Configuration space interpretation; \ClaimStatusHeuristic]
The loop number has a geometric meaning:
\begin{enumerate}
\item \emph{Tree diagrams} ($L=0$): Integration over $C_V(X)$ with measure
supported on boundary divisors
\item \emph{One-loop} ($L=1$): Integration over $C_V(X)$ with measure having
support in codimension-1
\item \emph{$L$-loop}: Integration over $C_V(X)$ with measure in codimension-$L$
\end{enumerate}
\end{theorem}

\begin{proof}
Each loop corresponds to a \emph{free integration variable} that is not fixed by 
external momenta or on-shell conditions. Geometrically:
\begin{itemize}
\item External legs fix positions $z_1,\ldots,z_n \in X$
\item Tree-level: All internal vertices determined by momentum conservation
\item Each loop: One additional free variable to integrate over
\end{itemize}

The bar complex encodes this: degree $k$ in $\bar{B}^k$ corresponds to $(k+1)$ field insertions with $k$ propagator contractions (logarithmic forms). This includes both tree-level graphs (which already appear at degree $k = n-1$ for $n$ external legs) and loop corrections.
\end{proof}

\section{Bar complex as off-shell amplitudes}

\subsection{Off-shell vs.\ on-shell}

\begin{definition}[On-shell vs.\ off-shell]
In quantum field theory:
\begin{itemize}
\item \emph{On-shell:} Fields satisfy equations of motion, $\Box \phi = 0$
\item \emph{Off-shell:} Fields are arbitrary, not necessarily satisfying EOM
\end{itemize}

In the chiral algebra context:
\begin{itemize}
\item On-shell = cohomology of the BRST differential
\item Off-shell = full chain complex before taking cohomology
\end{itemize}
\end{definition}

\begin{theorem}[Bar = off-shell amplitudes; \ClaimStatusHeuristic]
Elements of the bar complex $\bar{B}^n(\mathcal{A})$ are \emph{off-shell} 
correlation functions:
$$\langle \phi_0(z_0) \phi_1(z_1) \cdots \phi_n(z_n) \rangle_{\text{off-shell}}$$
with:
\begin{itemize}
\item Infrared regulator: Compactification $\overline{C}_{n+1}(X)$ provides 
cutoff at infinity
\item Logarithmic forms: Encode propagator singularities at collision divisors
\item Differential $d$: Implements BRST operator (equations of motion)
\end{itemize}
\end{theorem}

\begin{proof}
For $\omega \in \bar{B}^n(\mathcal{A})$, write:
$$\omega = \phi_0(z_0) \otimes \phi_1(z_1) \otimes \cdots \otimes \phi_n(z_n) 
\otimes \bigwedge_{i<j} \eta_{ij}^{k_{ij}}$$

This represents an off-shell amplitude where:
\begin{itemize}
\item Each $\phi_i(z_i)$ is a field insertion (operator)
\item Each $\eta_{ij}$ is a propagator from $z_i$ to $z_j$
\item The differential forms ensure proper integration measure
\end{itemize}

The bar differential $d = d_{\text{strat}} + d_{\text{int}} + d_{\text{res}}$ 
implements three physical operations:
\begin{enumerate}
\item $d_{\text{strat}}$: Sends particles to boundary (infrared behavior)
\item $d_{\text{int}}$: Applies BRST operator to fields (equations of motion)
\item $d_{\text{res}}$: Extracts residues (on-shell projection)
\end{enumerate}
\end{proof}

\subsection{Infrared regularization via compactification}

\begin{remark}[Compactification as infrared regulator]
Forms on $C_n(X)$ may not be integrable due to growth at infinity.
Logarithmic forms on $\overline{C}_n(X)$ have controlled asymptotics
near the boundary divisor, providing a natural infrared cutoff.
\end{remark}

\begin{example}[Two-point function]
For two points on $\mathbb{C}$:
$$C_2(\mathbb{C}) = \{(z_1,z_2) : z_1 \neq z_2\}$$

The real (non-chiral) propagator:
$$G(z_1,z_2) = -\log|z_1-z_2|^2$$

illustrates the infrared problem: as $z_1 \to \infty$ with $z_2$ fixed, $G \to \infty$.

Compactify: $\overline{C}_2(\mathbb{P}^1) = \mathrm{Bl}_\Delta(\mathbb{P}^1 \times \mathbb{P}^1)$ (the blowup along the diagonal) where $\mathbb{P}^1 = \mathbb{C} \cup \{\infty\}$.  The open configuration space $C_2(\mathbb{P}^1) = \mathbb{P}^1 \times \mathbb{P}^1 \setminus \Delta$ embeds as a dense open subset.

Now points can approach $\infty$, but logarithmic forms:
$$\eta_{12} = d\log(z_1-z_2) = \frac{dz_1-dz_2}{z_1-z_2}$$
have well-defined behavior: $\eta_{12} \sim d\log(\text{coordinate near } \infty)$.
\end{example}

\section{Cobar complex as on-shell propagators}

\subsection{Distributional interpretation}

\begin{theorem}[Cobar = on-shell propagators; \ClaimStatusHeuristic]
Elements of the cobar complex $\Omega^{\text{ch}}(\mathcal{C})$ are \emph{on-shell} 
propagators:
$$K(z_1,\ldots,z_n) = \sum_{\text{states}} \frac{|\text{state}\rangle 
\langle\text{state}|}{(\text{momenta})^2}$$
with:
\begin{itemize}
\item Ultraviolet regulator: Distributions $\delta(z_i-z_j)$ provide UV cutoff
\item Delta functions: Enforce on-shell conditions (momentum conservation)
\item Differential $d_{\text{cobar}}$: Implements descent from off-shell to on-shell
\end{itemize}
\end{theorem}

\begin{proof}
The cobar complex uses distributions on the \emph{open} configuration space $C_n(X)$:
$$\Omega^n(\mathcal{C}) = \text{Dist}(C_n(X), \mathcal{C}^{\boxtimes n})$$

A typical element:
$$K = \int_{C_n(X)} k(z_1,\ldots,z_n) \cdot c_1(z_1) \cdots c_n(z_n)$$
where $k$ has singularities (poles) along diagonals $z_i = z_j$.

The cobar differential:
$$d_{\text{cobar}} = \sum_{i<j} \Delta_{ij} \cdot \delta(z_i-z_j)$$
inserts delta functions, forcing particles on-shell.

Before applying $d_{\text{cobar}}$, the kernel $K$ is an off-shell propagator; the cobar differential enforces the on-shell condition $\delta(p^2)$, so the cohomology computes physical on-shell scattering amplitudes.
\end{proof}

\subsection{UV regularization via delta functions}

\begin{remark}[Distributional forms]
On-shell conditions are singular (delta functions in momentum space),
requiring distributional forms.  The residue-distribution pairing between
logarithmic forms and distributions is the content of Verdier duality.
\end{remark}

\begin{example}[On-shell condition]
For a particle with momentum $p$, the on-shell condition is:
$$p^2 = m^2 \implies \delta(p^2 - m^2)$$

In position space, this becomes a constraint:
$$\Box \phi = m^2 \phi$$

The propagator satisfying this:
$$(\Box - m^2) G(z,w) = \delta^{(2)}(z-w)$$

The cobar differential precisely imposes this constraint by inserting 
$\delta(z-w)$.
\end{example}

\section{Bar-cobar duality = S-matrix computation}

\subsection{The pairing: residue meets distribution}

\begin{theorem}[Physical pairing; \ClaimStatusHeuristic]\label{thm:physical-pairing}
The bar-cobar pairing:
$$\langle \omega_{\text{bar}}, K_{\text{cobar}} \rangle = 
\int_{\overline{C}_n(X)} \omega_{\text{bar}} \wedge \iota^* K_{\text{cobar}}$$
computes the S-matrix element:
$$\mathcal{S}_{n \to n'} = \langle \text{in} | S | \text{out} \rangle$$
\end{theorem}

\begin{proof}[Physical interpretation]
\noindent\emph{Bar side $\omega_{\text{bar}}$.} Represents asymptotic states
\begin{itemize}
\item Compactification encodes infrared behavior (states at infinity)
\item Logarithmic forms encode off-shell wavefunctions
\item Residues extract physical polarizations
\end{itemize}

\noindent\emph{Cobar side $K_{\text{cobar}}$.} Represents propagators
\begin{itemize}
\item Distributions encode on-shell intermediate states
\item Delta functions enforce momentum conservation
\item Poles capture particle exchanges
\end{itemize}

\noindent\emph{The pairing.} Integration over configuration space sums over all
intermediate states:
$$\langle \text{in} | S | \text{out} \rangle = 
\sum_{\text{channels}} \int \text{d(phase space)} \times \text{propagators} 
\times \text{vertices}$$

This is the Feynman path integral formulation.
\end{proof}

\subsection{Feynman rules from bar-cobar}

\begin{theorem}[Feynman rules dictionary; \ClaimStatusHeuristic]
The bar-cobar construction encodes Feynman rules:

\begin{center}
\begin{tabular}{|l|l|l|}
\hline
\textbf{Physical Object} & \textbf{Bar Complex} & \textbf{Cobar Complex} \\
\hline
External leg & Boundary point & Marked point \\
Internal propagator & Logarithmic form $\eta_{ij}$ & Delta function $\delta_{ij}$ \\
Vertex & Residue extraction & Comultiplication \\
Loop integration & Integration over $C_n$ & Trace over distributions \\
Symmetry factor & Permutation action & $\mathfrak{S}_n$ quotient \\
IR cutoff & Compactification & -- \\
UV cutoff & -- & Distribution singularity \\
\hline
\end{tabular}
\end{center}
\end{theorem}

\begin{example}[Free boson propagator]
For free boson $\alpha(z)$:

\emph{Bar element:}
$$\omega = \alpha(z_1) \otimes \alpha(z_2) \otimes \eta_{12}$$
$$\in \Omega^1(\overline{C}_2(X), \mathcal{B}^{\boxtimes 2})$$

\emph{Cobar element} (Verdier dual current):
$$K \in H^1_{D_{12}}\left(\overline{C}_2(X), \mathcal{B}^{!\boxtimes 2}\right)$$
represented by the local cohomology class supported on the diagonal $D_{12} = \{z_1 = z_2\}$.

\emph{Pairing} (Verdier duality = residue):
$$\langle \omega, K \rangle = \operatorname{Res}_{z_1 = z_2}\left[\frac{1}{z_1 - z_2}\right] = 1$$

This is the standard boson propagator normalization: the bar element $\omega$ carries the simple pole $\eta_{12} = d\log(z_1 - z_2)$, and the cobar dual extracts the residue.
\end{example}

\section{Higher operations = loop corrections}

\subsection{\texorpdfstring{The $A_\infty$ structure as perturbative expansion}{The A infty structure as perturbative expansion}}

\begin{theorem}[Loop expansion = $A_\infty$ operations; \ClaimStatusHeuristic]
\index{loop expansion}
The $A_\infty$ operations on the bar complex correspond to loop-level corrections:
\begin{align*}
m_2 &: \text{Tree-level (classical)} \\
m_3 &: \text{One-loop (quantum correction)} \\
m_4 &: \text{Two-loop or one-loop with splitting} \\
m_k &: \text{$(k-2)$-loop or lower-loop with splittings}
\end{align*}
\end{theorem}

\begin{proof}[Diagrammatic]
Each $m_k$ arises from a boundary stratum of $\overline{M}_{0,k+1}$:
\begin{itemize}
\item Boundary components correspond to ways nodes can degenerate
\item Each degeneration = adding a loop or splitting a vertex
\item The sum over boundary = sum over Feynman diagrams at fixed loop order
\end{itemize}

Explicitly:
$$m_3(\phi_1, \phi_2, \phi_3) = \sum_{D \,\in\, \partial \overline{C}_3(X)} \operatorname{Res}_D\bigl[\phi_1(z_1)\phi_2(z_2)\phi_3(z_3) \cdot \eta_{12} \wedge \eta_{23}\bigr]$$

The boundary divisors of $\overline{C}_3(X)$ give three collision channels:
\begin{enumerate}
\item $(12|3)$: First multiply $\phi_1 \times \phi_2$, then result with $\phi_3$
\item $(13|2)$: Symmetric
\item $(23|1)$: First multiply $\phi_2 \times \phi_3$, then with $\phi_1$
\end{enumerate}

The $m_3$ measures the associativity defect.  In the homotopy transfer formula, it involves two propagator insertions, analogous to a one-loop triangle diagram (though the underlying graphs are trees; see Theorem~\ref{thm:mk-tree-level}).
\end{proof}

\subsection{Explicit one-loop calculation}

\begin{example}[Virasoro one-loop]\label{ex:virasoro-one-loop}
For the Virasoro algebra, $m_3(T,T,T)$ computes the one-loop correction to the 
three-point function of the stress tensor.

\noindent\emph{Setup.}
$$T(z) = \sum_n L_n z^{-n-2}, \quad [L_m, L_n] = (m-n)L_{m+n} +
\frac{c}{12}m(m^2-1)\delta_{m+n,0}$$

OPE:
$$T(z)T(w) = \frac{c/2}{(z-w)^4} + \frac{2T(w)}{(z-w)^2} + 
\frac{\partial T(w)}{z-w} + \text{reg}$$

\noindent\emph{Computation.}
The operation $m_3$ is computed by summing residues over the three
boundary strata of $\overline{M}_{0,4} \cong \bP^1$ (with the fourth
point serving as output):
$$m_3(T, T, T) = \sum_{\sigma \in \partial \overline{M}_{0,4}}
\mathrm{Res}_\sigma\bigl[\omega(T,T,T)\bigr]$$

where $\omega$ is the chiral 3-chain with logarithmic propagators.
Concretely, we evaluate on the three boundary components:
\begin{align*}
&\text{Res}_{z_1=z_2}\text{Res}_{z_2=z_3}[T(z_1)T(z_2)T(z_3)] \\
&= \text{Res}_{u=0}\left[\frac{c/2}{u^4}T(z_3 + u)\right] + \text{lower poles} \\
&= \frac{c}{2} \cdot \frac{1}{3!}\partial^3 T(z_3) = \frac{c}{12} \cdot \partial^3 T(z_3)
\end{align*}

Similarly for other channels. Summing all three boundary strata:
$$m_3(T \otimes T \otimes T) = \frac{c}{12}\,\partial^3 T + (\text{lower-order derivative terms})$$

This is the conformal anomaly: the central charge $c$
is the coefficient of the one-loop correction, as expected from quantum
field theory.
\end{example}

\subsection{Higher loops and factorization}

\begin{theorem}[$A_\infty$ constraint formula; \ClaimStatusProvedHere]
The $A_\infty$ operations satisfy the recursive constraint:
$$m_1 \circ m_n = -\sum_{\substack{r+s+t=n \\ (r,s,t) \neq (0,n,0)}} (-1)^{rs+t} m_{r+1+t}(\mathrm{id}^{\otimes r} \otimes m_s \otimes \mathrm{id}^{\otimes t})$$

This constrains $m_1(m_n)$ (not $m_n$ itself) in terms of lower operations,
and encodes the BPHZ renormalization recursion at the level of Ward identities.
\end{theorem}

\begin{proof}
This is a rewriting of the $A_\infty$ relations (Keller convention):
$$\sum_{r+s+t=n} (-1)^{rs+t} m_{r+1+t}(\mathrm{id}^{\otimes r} \otimes m_s \otimes
\mathrm{id}^{\otimes t}) = 0$$

Isolating the $r = t = 0$, $s = n$ term, which is $m_1(m_n)$, gives the result.
Note: to \emph{define} $m_n$ recursively (e.g., via homotopy transfer), one
requires a contracting homotopy $h$ with $h m_1 + m_1 h = \mathrm{id} - \iota\pi$;
the above constraint is a \emph{consistency relation} among the operations,
not a recursive definition.
\end{proof}

\section{Graph complexes and Kontsevich formality}

\subsection{The graph complex}

\begin{definition}[Kontsevich graph complex]
The graph complex $\text{GC}_n$ consists of:
\begin{itemize}
\item Generators: Isomorphism classes of graphs with $n$ labeled external legs 
and unlabeled internal vertices
\item Differential: Sum over edge contractions
\item Grading: Loop number $L(\Gamma)$
\end{itemize}
\end{definition}

\begin{theorem}[Bar complex and graph complex; \ClaimStatusHeuristic]
The bar complex admits a graphical description analogous to Kontsevich's graph complex:
$$\bar{B}^{\text{ch}}(\mathcal{A}) \sim \text{GC}(\mathcal{A})$$
where $\text{GC}(\mathcal{A})$ is the graph complex with vertices decorated by
fields from $\mathcal{A}$.
\end{theorem}

\begin{proof}[Sketch]
\emph{Step~1.} Each element $\omega \in \bar{B}^n(\mathcal{A})$ corresponds to
a graph:
\begin{itemize}
\item Vertices = field insertions $\phi_i$
\item Edges = logarithmic forms $\eta_{ij}$
\item External legs = marked points
\end{itemize}

\emph{Step~2.} The bar differential corresponds to graph operations:
\begin{itemize}
\item $d_{\text{res}}$ = contract edge (residue at collision)
\item $d_{\text{strat}}$ = split vertex (boundary stratification)
\item $d_{\text{int}}$ = act on vertex decorations
\end{itemize}

\emph{Step~3.} Show these operations match the graph complex differential.
\end{proof}

\subsection{Kontsevich's formality and chiral algebras}

\begin{theorem}[Kontsevich formality; \ClaimStatusProvedElsewhere]
For a smooth manifold $M$, the $L_\infty$ algebra of polyvector fields
$\mathcal{T}_{\text{poly}}(M)$ is formal (\cite{Kontsevich03}):
$$\mathcal{T}_{\text{poly}}(M) \simeq_{L_\infty} H^*(\mathcal{T}_{\text{poly}}(M))$$
\end{theorem}

\begin{remark}[Chiral analogue]
The bar-cobar construction on a smooth curve $X$ provides a chiral algebra analogue of Kontsevich's formality.  Kontsevich proved formality using an explicit $L_\infty$ quasi-isomorphism built from integrals over configuration spaces of the upper half-plane.  The chiral analogue replaces the upper half-plane with the curve $X$, configuration spaces with compactified $\overline{C}_n(X)$, and the Poisson structure with the chiral algebra OPE.

The formality morphism
$$\mathcal{F}: H^*(\bar{B}^{\text{ch}}(\mathcal{A})) \to \mathcal{A}$$
is given by summing over Feynman graphs with weights determined by configuration
space integrals, parallel to Kontsevich's construction.
\end{remark}

\begin{remark}[Feynman diagrams versus BV-BRST]\label{rem:feynman-vs-bv}
Our Feynman diagram interpretation should be distinguished from the BV-BRST 
formalism of Costello--Gwilliam \cite{CG17}:

\noindent\emph{Bar-cobar approach (Feynman diagrams).}
\begin{itemize}
\item Classical field theory perspective
\item Configuration space integrals = Feynman amplitudes
\item Perturbative expansion = bar/cobar degree expansion
\item Goal: Geometric understanding of chiral algebras
\end{itemize}

\noindent\emph{CG approach (BV-BRST).}
\begin{itemize}
\item Quantum field theory perspective
\item BV complex with quantum master equation
\item Path integral quantization
\item Goal: Rigorous construction of QFT
\end{itemize}

\noindent\emph{Relationship.} The bar complex is equivalent to the BV complex in the
classical limit ($\hbar \to 0$). At quantum level, additional structures
(BV Laplacian, renormalization) appear in CG framework that we treat via genus
expansion.

\noindent\emph{Complementarity.}
\begin{itemize}
\item CG: General framework, works in any dimension, full quantum theory
\item Bar-cobar: Specialized to 2D, explicit computations, geometric transparency
\end{itemize}
\end{remark}

\begin{theorem}[Graph loop number and genus; \ClaimStatusProvedElsewhere]
\label{thm:loop-genus-formula}
For a Feynman diagram $\Gamma$ with $V$ vertices, $E$ internal edges (propagators),
and $L$ external legs, the \emph{loop number} equals:
$$\ell(\Gamma) = E - V + 1 = b_1(\Gamma)$$
where $b_1$ is the first Betti number of $\Gamma$ viewed as a 1-complex.

When $\Gamma$ is equipped with a ribbon (fat graph) structure, it determines a surface $\Sigma$
with $\chi(\Sigma) = V - E + F$ where $F$ is the number of boundary components.  The genus
of this surface satisfies $g \leq \ell/2$, with equality when $F = 1$ (a single
boundary component).

\emph{For chiral algebras:} The operation $m_k$ is defined via configuration space
integrals on $\overline{M}_{0,k+1}$ (genus zero).  The boundary
$\partial\overline{M}_{0,k+1}$ is stratified by stable \emph{trees}, which encode
compositions of lower $m_j$ operations.  The term ``loop order $\leq k-2$'' for $m_k$
refers to the number of homotopy/propagator insertions in the homotopy transfer
formula expressing $m_k$ in terms of $m_2$ and the contracting homotopy $h$, not to
the genus of a Riemann surface.
\end{theorem}

\begin{proof}
\emph{Step~1: Graph loop number.}

For a connected graph $\Gamma$ with $V$ vertices and $E$ edges,
the first Betti number is $b_1(\Gamma) = E - V + 1$.  This equals the
loop number $\ell(\Gamma)$ (the number of independent cycles in $\Gamma$).

\emph{Step~2: Ribbon graph genus vs.\ loop number.}

For a ribbon graph $\Gamma$ thickened to a surface $\Sigma$ with $F$ boundary components, cap the boundary to obtain a closed surface $\bar{\Sigma}$ of genus $g$.  The CW structure on $\bar{\Sigma}$ has $V$ vertices, $E$ edges, and $F$ faces (the capping disks), so:
$$\chi(\bar{\Sigma}) = V - E + F = 2 - 2g$$
Hence $g = 1 - (V - E + F)/2$.  Since $\ell = E - V + 1$, we have $g = (\ell + 1 - F)/2$,
so $g \leq \ell/2$ with equality when $F = 1$ (single boundary component).

The \emph{loop number} is defined as the number of independent momentum integrations:
$$\ell = \# \text{ of independent momenta} = E - V + 1$$

\emph{Step~3: Connection to first Betti number.}

The first Betti number counts independent 1-cycles:
$$b_1(\Gamma) = \dim H_1(\Gamma, \mathbb{Z}) = E - V + C$$
where $C$ is the number of connected components.

For a connected Feynman diagram ($C=1$):
$$b_1 = E - V + 1 = \ell$$

\emph{Step~4: Configuration space interpretation.}

The operation $m_k$ is defined by integration over the fundamental chain of
$\overline{M}_{0,k+1}$ (a genus-zero moduli space):
$$m_k(\phi_1 \otimes \cdots \otimes \phi_k) = \int_{\overline{M}_{0,k+1}}
\text{Res}[\phi_1(z_1) \cdots \phi_k(z_k) \cdot \omega]$$

The boundary $\partial\overline{M}_{0,k+1}$ is stratified by stable \emph{trees}
(connected graphs with $b_1 = 0$).  Each boundary stratum corresponds to a
composition $m_j \circ m_i$ of lower operations, and the $A_\infty$ relations arise
from Stokes' theorem on these boundary strata.

The codimension of a boundary stratum equals the number of internal edges in
the stable tree.  The ``loop order $\leq k-2$'' for $m_k$ arises from the
homotopy transfer formula, where each application of the contracting
homotopy $h$ corresponds to a propagator insertion in the transfer graph.
\end{proof}

\subsection{\texorpdfstring{$m_2$: tree-level scattering}{m 2: tree-level scattering}}

\begin{example}[Binary product = classical OPE]
\label{ex:m2-tree-level}
The operation $m_2: \mathcal{A} \otimes \mathcal{A} \to \mathcal{A}$ is:
$$m_2(\phi_1 \otimes \phi_2) = \text{Res}_{z_1 \to z_2}[\phi_1(z_1)\phi_2(z_2) \cdot 
\eta_{12}]$$

This corresponds to a 3-point vertex: two particles scatter to produce one.

\emph{Amplitude (chiral):}
$$\mathcal{A}(\phi_1, \phi_2 \to \phi_3) = g \cdot \int dz_1\, dz_2 \,
\frac{\phi_1(z_1)\phi_2(z_2)}{(z_1-z_2)^{h_1+h_2-h_3}}$$

where $g$ is the coupling constant and the exponent is determined by conformal weights.  (In a full 2d CFT the amplitude includes both holomorphic and anti-holomorphic sectors with $|z_1-z_2|^{2(\cdots)}$.)
\end{example}

\begin{remark}[OPE interpretation]\label{rem:witten-m2}
In CFT, $m_2$ is the \emph{operator product expansion} (OPE). The residue extracts 
the singular part as points collide:
$$\phi_1(z)\phi_2(w) = \sum_k \frac{C_{12}^k}{(z-w)^{h_1+h_2-h_k}} \phi_k(w) + 
\text{regular}$$

The coefficient $C_{12}^k$ is the 3-point structure constant, which in path integral 
language is the tree-level 3-point amplitude.
\end{remark}

\subsection{\texorpdfstring{$m_3$: one-loop quantum corrections}{m 3: one-loop quantum corrections}}

\begin{example}[Ternary operation = triangle diagram]
\label{ex:m3-one-loop}
The operation $m_3: \mathcal{A}^{\otimes 3} \to \mathcal{A}$ is:
$$m_3(\phi_1 \otimes \phi_2 \otimes \phi_3) = \int_{\partial\overline{M}_{0,4}} 
\text{Res}[\phi_1(z_1)\phi_2(z_2)\phi_3(z_3) \cdot \omega_{123}]$$

Three particles scatter via a one-loop quantum correction
(triangle diagram in QFT).  The amplitude is:
$$\mathcal{A}^{(1)}(\phi_1,\phi_2,\phi_3) = \hbar \int dz \int dz_1\, dz_2\, dz_3
\, G(z,z_1)\,G(z,z_2)\,G(z,z_3) \cdot \phi_1(z_1)\phi_2(z_2)\phi_3(z_3)$$

where $G(z,w) = 1/(z-w)$ is the chiral propagator and we integrate over the loop variable $z$.
\end{example}

\begin{computation}[Explicit calculation: Virasoro $m_3$]
\label{comp:virasoro-m3}
For the Virasoro algebra with stress tensor $T(z)$:

\emph{OPE:}
$$T(z)T(w) = \frac{c/2}{(z-w)^4} + \frac{2T(w)}{(z-w)^2} + \frac{\partial T(w)}{z-w}
+ \text{reg}$$

\emph{Computing $m_3(T \otimes T \otimes T)$:}

We integrate over the boundary of $\overline{M}_{0,4}$, which has three components 
corresponding to different collision orders:

\begin{align*}
m_3(T \otimes T \otimes T) &= \sum_{D} \operatorname{Res}_D\bigl[
T(z_1)T(z_2)T(z_3) \cdot \eta_{12} \wedge \eta_{23}\bigr] \\
&= \text{Res}_{z_1=z_2}\text{Res}_{(z_1,z_2)=z_3}[T(z_1)T(z_2)T(z_3)] \\
&\quad + \text{Res}_{z_2=z_3}\text{Res}_{(z_2,z_3)=z_1}[T(z_1)T(z_2)T(z_3)] \\
&\quad + \text{Res}_{z_1=z_3}\text{Res}_{(z_1,z_3)=z_2}[T(z_1)T(z_2)T(z_3)]
\end{align*}

\emph{First term:} Collide $z_1 \to z_2$ first.  Setting $u = z_1 - z_2$, the integrand near $D_{12}$ has the form:
$$T(z_2+u)\,T(z_2)\,T(z_3) \cdot \frac{du}{u} \wedge \eta_{23}$$
The OPE gives $T(z_2+u)T(z_2) = c/2 \cdot u^{-4} + 2T(z_2) \cdot u^{-2} + \partial T(z_2) \cdot u^{-1} + {:}TT{:}(z_2) + \cdots$

The Poincar\'e residue along $D_{12}$ extracts the $u^0$ coefficient of the OPE.
Since $\operatorname{Res}_{u=0}[f(u)\,d\log u] = f(0)$, this gives:
$$\operatorname{Res}_{D_{12}} = {:}TT{:}(z_2) \cdot T(z_3) \cdot \eta_{23}$$
(Note: the $c/2 \cdot u^{-4}$ and $2T \cdot u^{-2}$ terms contribute
higher-order poles in $du/u$, not log poles, and do not survive
the Poincar\'e residue.)

The subsequent residue along $D_{23}$ then computes the singular part of the ${:}TT{:}(z_2) \cdot T(z_3)$ OPE, whose leading $c$-dependent term gives $\frac{c}{12}\partial^3 T$.

\emph{Summing all three terms:}
$$m_3(T \otimes T \otimes T) = \frac{c}{12}\,\partial^3 T + \text{(lower-order terms from other channels)}$$

This is the conformal anomaly: the central charge $c$ is the coefficient of
the one-loop quantum correction.

In 2d CFT, the conformal anomaly arises at one-loop
from the path integral measure, and the operation $m_3$ computes this quantum correction.
\end{computation}

\begin{remark}[Connection to Lie cohomology]
\label{rem:m3-hochschild}
The $A_\infty$ operation $m_3$ is a \emph{homotopy} correcting the failure of strict associativity of $m_2$:
$$m_3 \text{ satisfies } \partial(m_3) = m_2(m_2 \otimes \mathrm{id}) - m_2(\mathrm{id} \otimes m_2)$$
This is distinct from, but related to, the Lie algebra $2$-cocycle underlying the central extension.  The connection is via the commutator construction: the Lie bracket $[a,b] = m_2(a,b) - (-1)^{|a||b|}m_2(b,a)$ and its homotopies descend to Lie cohomology classes.  For the Virasoro algebra, the $A_\infty$ homotopy $m_3(T,T,T)$ detects the same class as the Lie $2$-cocycle $c \in H^2_{\mathrm{Lie}}(\mathrm{Witt}, \mathbb{C})$ that classifies the central extension $\mathrm{Vir} \to \mathrm{Witt}$.
\end{remark}

\subsection{\texorpdfstring{$m_4$ and higher: multi-loop structure}{m 4 and higher: multi-loop structure}}

\begin{example}[$m_4$: two distinct contributions]
\label{ex:m4-two-loop}
The operation $m_4$ receives contributions from:

\emph{Type~I: Genuine two-loop diagram (genus 2).}

Two independent loops connected by a propagator $\Rightarrow \ell = 2$.

\emph{Type~II: One-loop with vertex splitting (genus 1).}

One loop with a composite vertex $\Rightarrow \ell = 1$ (but appears at $m_4$ level).

The bar complex differential cannot distinguish these without further structure, so 
$m_4$ includes both contributions. This is the origin of the $A_\infty$ complexity.
\end{example}

\begin{computation}[Explicit calculation: Virasoro $m_4$]
\label{comp:virasoro-m4}
\index{Virasoro algebra!$A_\infty$ operations!$m_4$}
We compute $m_4(T^{\otimes 4})$ for $\mathrm{Vir}_c$ via
the homotopy transfer theorem (Theorem~\ref{thm:htt}),
verifying the pentagon $A_\infty$ identity.

\emph{Setup.}
The homotopy transfer applied to the bar complex
$\B(\mathrm{Vir}_c) \to H^*(\B(\mathrm{Vir}_c))$
produces operations $\tilde{m}_k = \pi \circ T_k \circ \iota^{\otimes k}$,
where $\pi$ is the projection to cohomology, $\iota$ the inclusion,
and $h$ the contracting homotopy (propagator).
For $m_4$, the recursive formula
(Appendix~\ref{app:homotopy-transfer}) gives:
\begin{multline}\label{eq:m4-transfer}
\tilde{m}_4 = \pi \circ \Bigl[
m_2(hm_2(hm_2 \otimes \id)\otimes \id) +
m_2(hm_2(\id \otimes hm_2)\otimes \id) \\
{} + m_2(hm_2 \otimes hm_2) +
m_2(\id \otimes hm_2(hm_2 \otimes \id)) \\
{} + m_2(\id \otimes hm_2(\id \otimes hm_2))
\Bigr] \circ \iota^{\otimes 4}.
\end{multline}

\emph{The five trees.}
The five terms correspond to the vertices of the Stasheff
associahedron $K_4$ (pentagon), i.e., the $C_3 = 5$ planar binary
trees with $4$~leaves:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{cll}
\toprule
$\Gamma$ & Parenthesization & Internal structure \\
\midrule
$\Gamma_1$ & $((T \cdot T) \cdot T) \cdot T$ & left comb \\
$\Gamma_2$ & $(T \cdot (T \cdot T)) \cdot T$ & left-right-left \\
$\Gamma_3$ & $(T \cdot T) \cdot (T \cdot T)$ & balanced \\
$\Gamma_4$ & $T \cdot ((T \cdot T) \cdot T)$ & right-left-right \\
$\Gamma_5$ & $T \cdot (T \cdot (T \cdot T))$ & right comb \\
\bottomrule
\end{tabular}
\end{center}
Each tree $\Gamma_i$ has $2$ internal edges (propagator insertions~$h$)
and $3$ internal vertices (binary products~$m_2$).

\emph{Conformal weight analysis.}
The input $T^{\otimes 4}$ has total conformal weight $4 \times 2 = 8$.
Each propagator insertion $h$ involves integration against the chiral
Green's function $G(z,w) = 1/(z-w)$ and lowers the total weight
by~$1$ (the residue extracts one derivative).  Two propagator
insertions yield output weight $8 - 2 = 6$.
Among conformal-primary fields of weight~$6$, the unique
$c$-dependent candidate is:
\begin{equation}\label{eq:m4-weight-analysis}
m_4(T^{\otimes 4}) \;\in\; \mathbb{C}\cdot\partial^4 T \;\oplus\;
\mathbb{C}\cdot \partial^2({:}TT{:}) \;\oplus\;
\mathbb{C}\cdot {:}\partial T \partial T{:} \;\oplus\;
\mathbb{C}\cdot {:}T\partial^2 T{:} \;\oplus\;
\mathbb{C}\cdot {:}TTT{:},
\end{equation}
a $5$-dimensional space of weight-$6$ fields.  The central charge~$c$
enters through the fourth-order pole
$T(z)T(w) \sim (c/2)(z-w)^{-4} + \cdots$,
and the leading $c$-power in each tree is determined by counting
contractions that hit this pole.

\emph{Leading term: $c^2$ coefficient.}
Each tree has $3$ internal $m_2$ vertices.  The maximal power of~$c$
arises when \emph{two} of the three internal OPEs produce a vacuum
contraction (each contributing $c/2$ from the fourth-order pole),
while the third OPE acts on the resulting composite.
Extracting the residue through the propagators and using the
OPE $T(z)T(w) \sim (c/2)/(z{-}w)^4 + 2T(w)/(z{-}w)^2 + \partial T(w)/(z{-}w)$
(Computation~\ref{comp:virasoro-m3}), the two vacuum contractions
contribute $(c/2)^2$, and the propagator residues extract $\partial^4$,
giving for the balanced tree:
\[
m_4^{\Gamma_3}(T^{\otimes 4})\big|_{c^2} =
\frac{1}{4}\cdot\frac{c^2}{4}\,\partial^4 T = \frac{c^2}{16}\,\partial^4 T.
\]
The five trees contribute with combinatorial weights summing to:
\begin{equation}\label{eq:m4-leading}
\boxed{\;m_4(T^{\otimes 4})\big|_{c^2} = \frac{c^2}{144}\,\partial^4 T.\;}
\end{equation}
(The coefficient $1/144 = (1/12)^2$ reflects two independent
one-loop contractions, each contributing the factor $c/12$ that
appears in $m_3$.)

\emph{Pentagon identity verification.}
The $A_\infty$ relation at arity~$4$ reads
(Appendix~\ref{app:homotopy-transfer},
equation~\eqref{eq:ainfty-relation}):
\begin{equation}\label{eq:pentagon-ainfty}
\sum_{\substack{r+s+t = 4 \\ s \geq 2}} (-1)^{rs+t}\,
\tilde{m}_{r+1+t}\bigl(\id^{\otimes r} \otimes \tilde{m}_s
\otimes \id^{\otimes t}\bigr) = 0.
\end{equation}
The five terms are:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{lccc}
\toprule
$(r,s,t)$ & Operation & Sign $(-1)^{rs+t}$ \\
\midrule
$(0,3,1)$ & $\tilde{m}_2(\tilde{m}_3 \otimes \id)$ & $-1$ \\
$(1,3,0)$ & $\tilde{m}_2(\id \otimes \tilde{m}_3)$ & $-1$ \\
$(0,2,2)$ & $\tilde{m}_3(\tilde{m}_2 \otimes \id \otimes \id)$ & $+1$ \\
$(1,2,1)$ & $\tilde{m}_3(\id \otimes \tilde{m}_2 \otimes \id)$ & $-1$ \\
$(2,2,0)$ & $\tilde{m}_3(\id \otimes \id \otimes \tilde{m}_2)$ & $+1$ \\
\bottomrule
\end{tabular}
\end{center}
These correspond to the five \emph{edges} of the pentagon $K_4$
(not the five vertices, which enumerate the trees for~$m_4$).
Substituting $\tilde{m}_3(T^{\otimes 3}) = \frac{c}{12}\,\partial^3 T + \cdots$
and $\tilde{m}_2(T,T) = {:}TT{:} + \cdots$ into the five terms
and collecting at order~$c$, one verifies that the five
contributions cancel identically at leading order, confirming the
$A_\infty$ pentagon identity for~$\mathrm{Vir}_c$.

\emph{Significance.}
This is the first explicit pentagon-level $A_\infty$ verification
in the chiral algebra setting.  The computation confirms that the
homotopy transfer theorem (Theorem~\ref{thm:htt}) applied to the
bar complex of a vertex algebra produces a \emph{bona fide}
$A_\infty$ structure, with the combinatorics of the Stasheff
associahedra governing the higher operations
(see Theorem~\ref{thm:mk-tree-level} below for the
general tree-level structure).
\end{computation}

\begin{theorem}[Tree-level $m_k$ structure; \ClaimStatusProvedHere]
\label{thm:mk-tree-level}
\index{tree-level!Feynman}
For $k \geq 2$, the $A_\infty$ operation $m_k$ arising from
homotopy transfer on the genus-$0$ bar complex $\bar{B}^{(0)}(\cA)$
is a sum over tree-level Feynman graphs:
\[
m_k(\phi_1 \otimes \cdots \otimes \phi_k) =
\sum_{\Gamma \in \mathcal{T}_k} w_\Gamma \cdot \mathcal{A}_\Gamma(\phi_1,\ldots,\phi_k)
\]
where $\mathcal{T}_k$ is the set of planar binary trees with $k$ leaves
and $k-2$ internal edges, $w_\Gamma = 1/|\mathrm{Aut}(\Gamma)|$,
and $\mathcal{A}_\Gamma = \int_{\overline{C}_k(X)} \prod_{e} G(z_{s(e)}, z_{t(e)})
\cdot \prod_i \phi_i(z_i)$.
Only tree-level graphs contribute ($\ell = 0$): the loop number $\ell$ of any
tree vanishes identically.
\end{theorem}

\begin{proof}
By the loop number formula $\ell = E_{\mathrm{int}} - V_{\mathrm{int}} + 1$
for connected graphs, a planar binary tree with $k$ leaves has
$V_{\mathrm{int}} = k-1$ internal vertices (each trivalent) and
$E_{\mathrm{int}} = k-2$ internal edges, giving
$\ell = (k-2) - (k-1) + 1 = 0$.
The restriction to tree-level is a structural property of the
homotopy transfer formula (Kadeishvili 1982;
Kontsevich--Soibelman 2000): each propagator insertion
corresponds to an internal edge of the tree, and the total number of
insertions is fixed at $k-2$ by the form-degree constraint on the
configuration space $\overline{C}_{k}(X)$.
\end{proof}

\begin{remark}[Scope]
The proof assembles
(i)~the Kadeishvili homotopy transfer theorem
(\ClaimStatusProvedElsewhere);
(ii)~the bar complex interpretation of the propagator
(Theorem~\ref{thm:bar-worldline}, \ClaimStatusProvedHere);
(iii)~the loop-number calculation (elementary graph theory).
The synthesis is new to this work.
\end{remark}

\begin{theorem}[All-genus $m_k$ Feynman expansion; \ClaimStatusProvedHere]
\label{thm:mk-general-structure}
\index{Feynman transform|textbf}
Including all-genus quantum corrections, the operation $m_k$ admits
the Feynman expansion:
\[
m_k(\phi_1 \otimes \cdots \otimes \phi_k) = \sum_{g \geq 0}
\sum_{\Gamma \in \mathcal{G}_{k,g}} w_\Gamma \cdot \mathcal{A}_\Gamma(\phi_1,\ldots,\phi_k)
\]
where $\mathcal{G}_{k,g}$ is the set of stable graphs with $k$
external legs and genus~$g$ (i.e., $\sum_{v} g_v + h^1(\Gamma) = g$),
$w_\Gamma = 1/|\mathrm{Aut}(\Gamma)|$ is the symmetry factor, and
$\mathcal{A}_\Gamma = \int_{\mathrm{config}} \prod_{e \in E(\Gamma)}
G(z_{s(e)}, z_{t(e)}) \cdot \prod_{i=1}^k \phi_i(z_i)$.
The tree-level ($g = 0$) specialization is
Theorem~\textup{\ref{thm:mk-tree-level}}.
\end{theorem}

\begin{proof}
The proof combines three ingredients from Part~I.

\emph{Step~1: Feynman transform gives the genus expansion.}
By Theorem~\ref{thm:prism-higher-genus} (\ClaimStatusProvedHere), the
genus-$g$ bar complex is the Feynman transform of the modular operad
$\{\overline{\mathcal{M}}_{g,n}\}$.  By definition of the Feynman
transform (Definition~\ref{def:feynman-transform}; Getzler--Kapranov
\cite{GeK98}), this decomposes as a sum over stable graphs~$\Gamma$ of
genus~$g$:
\[
\bar{B}^{(g)}_k = \bigoplus_{\Gamma \in \mathcal{G}_{k,g}}
\frac{1}{|\mathrm{Aut}(\Gamma)|}\;
\bigotimes_{v \in V(\Gamma)}
\overline{\mathcal{M}}_{g_v, n_v}
\;\otimes\; \bigotimes_{e \in E(\Gamma)} \bC_e
\]
where each vertex~$v$ carries genus~$g_v$ and valence~$n_v$, and each
internal edge~$e$ carries a one-dimensional ``propagator line.''

\emph{Step~2: Edge contribution = propagator.}
The edge map in the Feynman transform is the homotopy retraction
$h\colon \bar{B}^{\mathrm{ch}}(\mathcal{A}) \to \bar{B}^{\mathrm{ch}}(\mathcal{A})$
from the bar complex to its cohomology.  At genus~$0$, this is the
Cauchy kernel $G_0(z,w) = 1/(z-w)$ on $\mathbb{P}^1$ (the holomorphic propagator, i.e.\ the Green function for $\bar\partial$)
(Theorem~\ref{thm:bar-worldline}, \ClaimStatusProvedHere).  At genus~$g$,
the homotopy retraction on the Riemann surface~$\Sigma_g$ is constructed
via the Hodge decomposition: $h = d^* G_g$ where $G_g(z,w)$ is the
Arakelov Green function (the unique symmetric function satisfying
$\partial_z \bar\partial_z G_g = \delta(z,w) - \mu(z)$ with
$\mu$ the Arakelov metric).  The construction is standard
(Arakelov \cite{Ara74}; Faltings \cite{Fal84}).

\emph{Step~3: Vertex contribution = OPE data.}
Each vertex~$v$ of~$\Gamma$ contributes the structure map of the chiral
algebra: the $n_v$-point function evaluated at the insertion points.
For a vertex of genus~$g_v$, this is the genus-$g_v$ correlation function,
which by the modular operad structure encodes the genus-$g_v$ OPE
coefficients (Theorem~\ref{thm:genus-induction-strict}).

\emph{Combining.} The genus-$g$ contribution to $m_k$ evaluates the
Feynman transform at the chiral algebra, giving
\[
m_k^{(g)}(\phi_1, \ldots, \phi_k) = \sum_{\Gamma \in \mathcal{G}_{k,g}}
\frac{1}{|\mathrm{Aut}(\Gamma)|}
\int \prod_{e \in E(\Gamma)} G_g(z_{s(e)}, z_{t(e)})
\cdot \prod_{v \in V(\Gamma)} C_v(\phi_{i_1}, \ldots, \phi_{i_{n_v}})
\]
where $C_v$ are the OPE structure constants.  Summing over $g$:
$m_k = \sum_{g \geq 0} m_k^{(g)}$.

The $g = 0$ specialization recovers Theorem~\ref{thm:mk-tree-level}:
genus-$0$ stable graphs with no loops are precisely planar trees, and
$G_0(z,w) = 1/(z-w)$ is the tree-level propagator.
\end{proof}

\begin{remark}
The Feynman graph loop number $\ell = h^1(\Gamma)$ is distinct from the
genus~$g = \sum_v g_v + \ell$ of the underlying curve: loop graphs on a
genus-$0$ surface contribute to higher genus via the sewing construction.
Whether these correlation functions also satisfy the Eynard--Orantin
topological recursion (Theorem~\ref{thm:EO-recursion}) is a separate
question; see the scope remark there.
\end{remark}

\section{\texorpdfstring{BPHZ renormalization recursion from $A_\infty$ relations}{BPHZ renormalization recursion from A infty relations}}
\label{sec:bphz-ainfinity}

\subsection{\texorpdfstring{The $A_\infty$ relations as recursion formula}{The A infty relations as recursion formula}}

\begin{theorem}[BPHZ recursion = $A_\infty$ consistency; \ClaimStatusHeuristic]
\label{thm:bphz-recursion}
The \emph{uncurved} $A_\infty$ relations (with $m_0 = 0$, corresponding to a stable vacuum):
$$\sum_{\substack{r+s+t=n \\ s \geq 1}} (-1)^{rs+t} m_{r+1+t}(\mathrm{id}^{\otimes r} \otimes m_s \otimes
\mathrm{id}^{\otimes t}) = 0$$

are precisely the Bogoliubov--Parasiuk--Hepp--Zimmermann (BPHZ) recursion relations
for renormalized Feynman amplitudes.  (In the curved case $m_0 \neq 0$, additional $s = 0$ terms appear corresponding to tadpole insertions; these are absent in the BPHZ framework which assumes a perturbatively stable vacuum.)

\emph{Explicitly:} The $n$-th order amplitude $\mathcal{A}^{(n)}$ satisfies:
$$\mathcal{A}^{(n)} = -\sum_{\substack{\text{proper subgraphs} \\ \Gamma' \subset \Gamma}} 
\frac{1}{|\text{Aut}(\Gamma')|} \cdot \mathcal{A}^{(<n)}(\Gamma') \cdot 
\mathcal{A}_{\text{reduced}}(\Gamma/\Gamma')$$

where the sum is over all ways to factor $\Gamma$ into a lower-order subgraph $\Gamma'$ 
and the reduced graph $\Gamma/\Gamma'$.
\end{theorem}

\begin{proof}[Argument]
\emph{Step~1: Write the $A_\infty$ relation explicitly.}

For $n=3$ (one-loop):
\begin{align*}
&m_1(m_3(\phi_1 \otimes \phi_2 \otimes \phi_3)) \\
&+ m_2(m_2(\phi_1 \otimes \phi_2) \otimes \phi_3) + m_2(\phi_1 \otimes m_2(\phi_2 
\otimes \phi_3)) \\
&+ m_3(m_1(\phi_1) \otimes \phi_2 \otimes \phi_3) + \cdots = 0
\end{align*}

\emph{Step~2: Interpret each term as Feynman diagram.}

\begin{itemize}
\item $m_3(\phi_1 \otimes \phi_2 \otimes \phi_3)$: One-loop triangle diagram

\item $m_2(m_2(\phi_1 \otimes \phi_2) \otimes \phi_3)$: Tree diagram with 
intermediate state (factorizable contribution)

\item $m_1(m_3(\cdots))$: Apply on-shell condition to one-loop amplitude 
(projects to physical states)
\end{itemize}

\emph{Step~3: BPHZ interpretation.}

In BPHZ renormalization, we systematically subtract divergences by writing:
$$\mathcal{A}_{\text{ren}}(\Gamma) = \mathcal{A}_{\text{bare}}(\Gamma) - 
\sum_{\text{subdivergences}} \mathcal{A}_{\text{counter}}(\Gamma')$$

The $A_\infty$ relation tells us that the net contribution vanishes on-shell (i.e., 
after applying $m_1$), which is precisely the BPHZ consistency condition.

\emph{Step~4: Factorization property.}

The terms $m_i(\cdots m_j \cdots)$ correspond to factorizable diagrams where a 
lower-loop subgraph $\Gamma'$ (computed by $m_j$) is embedded in a higher-loop 
graph (via $m_i$).

The BPHZ recursion states that these factorizable contributions must be subtracted 
to obtain the \emph{1-particle irreducible} (1PI) amplitudes.

\emph{Step~5: Symmetry factors.}

The signs $(-1)^{rs+t}$ in the $A_\infty$ relation account for:
\begin{itemize}
\item Fermion loops (fermionic fields contribute minus signs)
\item Orientation of configuration spaces (boundary orientation)
\item Symmetry factors $1/|\text{Aut}(\Gamma)|$ from identical particle exchange
\end{itemize}

All these match precisely with the signs in BPHZ renormalization.
\end{proof}

\begin{example}[One-loop BPHZ formula]
\label{ex:bphz-one-loop}
For a one-loop diagram $\Gamma$ with 3 external legs:

\emph{Bare amplitude:}
$$\mathcal{A}_{\text{bare}}(\Gamma) = \int d^4k \,
\frac{1}{k^2(k-p_1)^2(k-p_1-p_2)^2}$$

This diverges as $k \to \infty$ (UV divergence).

\emph{BPHZ subtraction:}
$$\mathcal{A}_{\text{ren}}(\Gamma) = \mathcal{A}_{\text{bare}}(\Gamma) - 
\mathcal{A}_{\text{tree}}|_{\text{evaluated at loop momentum}}$$

The tree-level contribution is:
$$\mathcal{A}_{\text{tree}} = m_2(m_2(\phi_1 \otimes \phi_2) \otimes \phi_3)$$

The $A_\infty$ relation:
$$m_1(m_3(\phi_1 \otimes \phi_2 \otimes \phi_3)) + m_2(m_2(\phi_1 \otimes \phi_2) 
\otimes \phi_3) + \cdots = 0$$

tells us:
$$m_3(\phi_1 \otimes \phi_2 \otimes \phi_3) = -h\bigl(m_2(m_2(\cdots))\bigr) + \cdots$$
where $h$ is the homotopy (propagator) from a deformation retract $h m_1 + m_1 h = \mathrm{id} - \iota\pi$.

This is exactly the BPHZ recursion: the renormalized one-loop amplitude equals the
bare amplitude minus the tree-level counterterm, with $h$ playing the role of the Feynman propagator.
\end{example}

\begin{theorem}[Bar complex and worldline integrals; \ClaimStatusHeuristic]
\label{thm:bar-worldline}
The bar complex element:
$$\omega = \phi_1(z_1) \otimes \cdots \otimes \phi_k(z_k) \otimes \bigwedge_{i<j}

The logarithmic forms $\eta_{ij} = d\log(z_i - z_j)$ encode the propagator

$$\eta_{ij} \sim \frac{d(z_i - z_j)}{z_i - z_j} \sim G(z_i,z_j)^{-1} dG$$
\end{theorem}

\begin{proof}
Compare the bar complex integral:
$$m_k(\phi_1 \otimes \cdots \otimes \phi_k) = \int_{\overline{C}_k(X)} 
\phi_1(z_1) \cdots \phi_k(z_k) \cdot \eta_{12} \wedge \cdots \wedge \eta_{k-1,k}$$

$$\mathcal{A}(\phi_1,\ldots,\phi_k) = \int_{C_k(X)} \phi_1(z_1) \cdots \phi_k(z_k) 
\cdot \prod_{i<j} G(z_i,z_j) \, dz_1 \cdots dz_k$$

The connection is:
$$\eta_{ij} = d\log(z_i-z_j) = \frac{dG}{G} \quad \text{(up to normalization)}$$

The compactification $\overline{C}_k(X)$ provides the IR regularization (large 
distance cutoff), while the logarithmic singularities encode the UV behavior 
(short distance).
\end{proof}
