\end{tabular}
\end{center}
This table is only a mnemonic. The manuscript does not prove an

The proved bar-side content is the following. Theorem~\ref{thm:bar-cobar-isomorphism-main}

\section{Feynman diagrams in chiral field theory}

\begin{definition}[Chiral field theory data]
\label{def:worldline-intro}
\index{propagator!Feynman}
\index{Feynman diagram!and bar complex}
A \emph{chiral field theory} on a curve~$X$ consists of a chiral
algebra~$\cA$ (the fields), the logarithmic propagator
$\eta(z,w) = d\log(z-w)$ at genus~$0$ (replaced by $d\log E(z,w)$
at higher genus, where $E$ is the prime form), and the
OPE structure constants (the vertices). The propagator has weight~$1$
regardless of the conformal weight of the fields
(Remark~\ref{rem:propagator-weight-universality}).
The bar complex element
$\omega_\Gamma \in \barB^{V-1}(\cA)$ is the Feynman integrand
before integration over the configuration space.

A Feynman diagram~$\Gamma$ with $V$~vertices and $E$~edges
determines a configuration space amplitude:
\[
A_\Gamma = \int_{C_V(X)}
\prod_{e=(i,j) \in E(\Gamma)} G(z_i,z_j)
\cdot \prod_{v \in V(\Gamma)} C_v
\cdot \prod_i dz_i,
\]
where $G(z_i,z_j) = 1/(z_i - z_j)$ is the chiral propagator and
each~$C_v$ is a vertex factor from the OPE. The logarithmic forms
$\eta_{ij} = d\log(z_i-z_j)$ encode these propagator singularities.
\end{definition}

\subsection{Tree vs.\ loop decomposition}

\begin{definition}[Loop number]
A Feynman diagram $\Gamma$ with $V$ vertices, $E$ edges, and $C$ connected 
components has loop number:
\[L(\Gamma) = E - V + C\]

This is the first Betti number $b_1(\Gamma)$ of the graph.
\end{definition}

\begin{remark}[Configuration space interpretation]
Each loop in~$\Gamma$ corresponds to a free integration variable not
fixed by external positions: tree diagrams ($L=0$) have all internal
vertices determined by collision residues, while an $L$-loop diagram
carries $L$ additional integrations over~$C_V(X)$. In the bar complex,
degree~$k$ in $\barB^k$ corresponds to $(k+1)$ field insertions with
$k$ propagator contractions.
\end{remark}

\section{Off-shell/on-shell duality}

In QFT, \emph{off-shell} fields are unconstrained; \emph{on-shell}
fields satisfy the equations of motion. In the chiral algebra setting,
this distinction becomes bar versus cobar:

\begin{conjecture}[Feynman rules dictionary; \ClaimStatusConjectured]
\label{conj:physical-pairing}
The bar-cobar construction models Feynman rules. The bar complex
carries off-shell amplitudes (logarithmic forms on the
compactification~$\overline{C}_n(X)$ provide infrared regularization);
the cobar complex carries on-shell propagator templates (distributions
on the open configuration space, with delta functions enforcing on-shell
conditions); and the bar-cobar pairing models the S-matrix:
\[
\langle \omega_{\text{bar}}, K_{\text{cobar}} \rangle
= \int_{\overline{C}_n(X)}
\omega_{\text{bar}} \wedge \iota^* K_{\text{cobar}}
\;\sim\;
\langle \text{in} \,|\, S \,|\, \text{out} \rangle.
\]

\begin{center}
\begin{tabular}{lll}
\toprule
\textbf{Physical Object} & \textbf{Bar Complex} & \textbf{Cobar Complex} \\
\midrule
External leg & Boundary point & Marked point \\
Internal propagator & Logarithmic form $\eta_{ij}$ & Delta function $\delta_{ij}$ \\
Vertex & Residue extraction & Comultiplication \\
Loop integration & Integration over $C_n$ & Trace over distributions \\
IR cutoff & Compactification & -- \\
UV cutoff & -- & Distribution singularity \\
\bottomrule
\end{tabular}
\end{center}
\end{conjecture}

\begin{example}[Free boson propagator]
For free boson $\alpha(z)$, the bar element
$\omega = \alpha(z_1) \otimes \alpha(z_2) \otimes \eta_{12}
\in \Omega^1(\overline{C}_2(X), \cA^{\boxtimes 2})$
carries the simple pole $\eta_{12} = d\log(z_1-z_2)$. The Verdier dual
cobar element is the local cohomology class
$K \in H^1_{D_{12}}(\overline{C}_2(X), \cA^{!\boxtimes 2})$
supported on the diagonal $D_{12} = \{z_1 = z_2\}$. The pairing
$\langle \omega, K \rangle
= \operatorname{Res}_{z_1 = z_2}[(z_1 - z_2)^{-1}] = 1$
recovers the standard propagator normalization.
\end{example}

\section{Higher operations from boundary strata}

\subsection{Explicit one-loop calculation}

\begin{example}[Virasoro one-loop]\label{ex:virasoro-one-loop}
For the Virasoro algebra, $m_3(T,T,T)$ computes the one-loop correction to the 
three-point function of the stress tensor.

\noindent\emph{Setup.}
\[T(z) = \sum_n L_n z^{-n-2}, \quad [L_m, L_n] = (m-n)L_{m+n} +
\frac{c}{12}m(m^2-1)\delta_{m+n,0}\]

OPE:
\[T(z)T(w) = \frac{c/2}{(z-w)^4} + \frac{2T(w)}{(z-w)^2} + 
\frac{\partial T(w)}{z-w} + \text{reg}\]

\noindent\emph{Computation.}
The operation $m_3$ is computed by summing residues over the three
boundary strata of $\overline{M}_{0,4} \cong \bP^1$ (with the fourth
point serving as output):
\[m_3(T, T, T) = \sum_{\sigma \in \partial \overline{M}_{0,4}}
\mathrm{Res}_\sigma\bigl[\omega(T,T,T)\bigr]\]

where $\omega$ is the chiral 3-chain with logarithmic propagators.
Concretely, we evaluate on the three boundary components:
\begin{align*}
&\text{Res}_{z_1=z_2}\text{Res}_{z_2=z_3}[T(z_1)T(z_2)T(z_3)] \\
&= \text{Res}_{u=0}\left[\frac{c/2}{u^4}T(z_3 + u)\right] + \text{lower poles} \\
&= \frac{c}{2} \cdot \frac{1}{3!}\partial^3 T(z_3) = \frac{c}{12} \cdot \partial^3 T(z_3)
\end{align*}

Similarly for other channels. Summing all three boundary strata:
\[m_3(T \otimes T \otimes T) = \frac{c}{12}\,\partial^3 T + (\text{lower-order derivative terms})\]

The central charge $c$ appears as the coefficient of
the one-loop correction. In the universal MC framework
(Theorem~\ref{thm:explicit-theta}), this $c/12$ is
$\operatorname{tr}(\Theta_{\mathrm{Vir}_c}^{(0)})$: the
genus-$0$ scalar trace of $\Theta$. The higher $A_\infty$
operations $m_4, m_5, \ldots$ arise as matrix components
of $\Theta$ beyond this scalar trace, each controlled
by the same curvature parameter $\kappa(\mathrm{Vir}_c) = c/2$.
\end{example}

\subsection{Higher loops and factorization}

\begin{theorem}[\texorpdfstring{$A_\infty$}{A-infinity} constraint formula; \ClaimStatusProvedHere]
\label{thm:ainfty-constraint-formula}
The $A_\infty$ operations satisfy the recursive constraint:
\[m_1 \circ m_n = -\sum_{\substack{r+s+t=n \\ (r,s,t) \neq (0,n,0)}} (-1)^{rs+t} m_{r+1+t}(\mathrm{id}^{\otimes r} \otimes m_s \otimes \mathrm{id}^{\otimes t})\]

This constrains $m_1(m_n)$ (not $m_n$ itself) in terms of lower operations,
and encodes the BPHZ renormalization recursion at the level of Ward identities.
\end{theorem}

\begin{proof}
This is a rewriting of the $A_\infty$ relations (Keller convention):
\[\sum_{r+s+t=n} (-1)^{rs+t} m_{r+1+t}(\mathrm{id}^{\otimes r} \otimes m_s \otimes
\mathrm{id}^{\otimes t}) = 0\]

Isolating the $r = t = 0$, $s = n$ term, which is $m_1(m_n)$, gives the result.
Note: to \emph{define} $m_n$ recursively (e.g., via homotopy transfer), one
requires a contracting homotopy $h$ with $h m_1 + m_1 h = \mathrm{id} - \iota\pi$;
the above constraint is a \emph{consistency relation} among the operations,
not a recursive definition.
\end{proof}

\section{Graph complexes and Kontsevich formality}

\subsection{The graph complex}

\begin{definition}[Kontsevich graph complex]
The graph complex $\text{GC}_n$ consists of:
\begin{itemize}
\item Generators: Isomorphism classes of graphs with $n$ labeled external legs 
and unlabeled internal vertices
\item Differential: Sum over edge contractions
\item Grading: Loop number $L(\Gamma)$
\end{itemize}
\end{definition}

\begin{conjecture}[Bar complex and graph complex; \ClaimStatusConjectured]
\label{conj:v1-bar-graph-complex}
The chiral bar complex admits a quasi-isomorphism to the
$\cA$-decorated Kontsevich graph complex:
$\barB^{\mathrm{ch}}(\cA) \simeq \mathrm{GC}(\cA)$,
where vertices correspond to field insertions, edges to logarithmic
forms~$\eta_{ij}$, and the bar differential decomposes as edge
contraction (residue), vertex splitting (boundary stratification),
and decoration action.
\end{conjecture}

\subsection{Kontsevich's formality and chiral algebras}

\begin{theorem}[Kontsevich formality; \ClaimStatusProvedElsewhere]
\label{thm:kontsevich-formality-feynman}
For a smooth manifold $M$, the $L_\infty$ algebra of polyvector fields
$\mathcal{T}_{\text{poly}}(M)$ is formal (\cite{Kon03}):
\[\mathcal{T}_{\text{poly}}(M) \simeq_{L_\infty} H^*(\mathcal{T}_{\text{poly}}(M))\]
\end{theorem}

\begin{remark}[Chiral analogue of Kontsevich formality]
\label{rem:kontsevich-graphs}
\label{sec:connections-other-feynman}
The bar-cobar construction on a curve~$X$ is a chiral analogue of
Kontsevich formality~\cite{Kon99}: polyvector fields become chiral
algebras, the upper half-plane becomes~$X$, angle forms $d\varphi$
become logarithmic forms~$\eta$, and the $L_\infty$
quasi-isomorphism becomes bar-cobar duality. The formality morphism
$\mathcal{F}\colon H^*(\barB^{\mathrm{ch}}(\cA)) \to \cA$ sums over
Feynman graphs with configuration space integral weights.
\end{remark}

\begin{conjecture}[Disk-local perturbative/FM comparison;
 \ClaimStatusConjectured]\label{conj:v1-disk-local-perturbative-fm}
Let $U \subset X$ be a holomorphic disk chart on a curve, and suppose a
holomorphic-topological perturbative field theory on~$U$ produces the
same local chiral algebra model $\cA|_U$.\linebreak[1]
Then the regularized perturbative BRST-anomaly brackets
of Gaiotto--Kulp--Wu~\cite{GKW24} agree, on that disk chart, with
the local bar operations cut out by the Fulton--MacPherson collision
strata:
\begin{enumerate}[label=\textup{(\roman*)}]
\item the binary perturbative bracket agrees with the bar residue/OPE
 operation on~$C_2(U)$;
\item the ternary perturbative bracket agrees with the sum of bar
 residues on the three codimension-$1$ collision strata of
 $\overline{C}_3(U)$, hence with the local $m_3$-operation; and
\item the quadratic identities satisfied by those perturbative brackets
 agree with the Stokes relations among the corresponding
 codimension-$1$ FM strata after passage to the compactification.
\end{enumerate}
\end{conjecture}

\begin{remark}[MC5 local packet]\label{rem:mc5-local-packet}
Conjecture~\ref{conj:v1-disk-local-perturbative-fm} isolates the first
exact MC5 comparison package. It is finite and local: one binary
collision calculation on~$C_2$, one ternary calculation on the three
boundary divisors of~$\overline{C}_3$, and the matching of the
quadratic/Wess--Zumino identities with the Stokes relations among those
divisors.

Once that local packet is in place, the next step is no longer a
physics-dictionary comparison but an analytic extension problem:
upgrade the local formulas to the FM boundary calculus used by the bar
construction and invoke Stokes regularity on FM compactifications
\textup{(}Proposition~\textup{\ref{prop:stokes-regularity-FM}}\textup{)}.
Only after that analytic compactification step does the remaining MC5
content become the genuswise clutching/renormalization lift.
\end{remark}

\begin{proposition}[Binary-ternary reduction for the disk-local packet;
 \ClaimStatusProvedHere]\label{prop:disk-local-binary-ternary-reduction}
Fix a holomorphic disk chart $U \subset X$, and interpret the
perturbative BRST-anomaly brackets and the local bar operations on the
same local chain model underlying~$\cA|_U$.
Assume:
\begin{enumerate}[label=\textup{(\roman*)}]
\item the binary perturbative bracket agrees with the local bar
 residue/OPE operation on~$C_2(U)$; and
\item the ternary perturbative bracket agrees with the local
 $m_3$-operation, equivalently with the sum of the three
 codimension-$1$ residues on~$\overline{C}_3(U)$.
\end{enumerate}
Then the quadratic/Wess--Zumino identities on the perturbative side
agree automatically with the quadratic $A_\infty$/Stokes relation on
the bar side. In particular, there is no third independent local
coefficient calculation beyond the binary and ternary comparisons.
\end{proposition}

\begin{proof}
On the perturbative side, Gaiotto--Kulp--Wu~\cite{GKW24} prove that the
regularized BRST-anomaly brackets satisfy the required quadratic
axioms/Wess--Zumino consistency relations.

On the bar side, the corresponding degree-$3$ consistency relation is
already built into the local $A_\infty$ structure: by
Theorem~\ref{thm:ainfty-constraint-formula} specialized to $n=3$, the
bar operations satisfy the quadratic relation expressing $m_1m_3$ in
terms of composites of lower operations; geometrically this is the
Stokes relation among the codimension-$1$ boundary strata of the
compactified three-point configuration space.

Once the binary and ternary operations are
identified on the common local chain model, the two quadratic packages
compare the same formal relation, leaving no third
independent local coefficient problem.
\end{proof}

\begin{remark}[Application to the disk-local FM comparison]
Applied to Conjecture~\ref{conj:v1-disk-local-perturbative-fm},
Proposition~\ref{prop:disk-local-binary-ternary-reduction} shows that
the only independent local input is the binary comparison on $C_2(U)$
and the ternary comparison on $\overline{C}_3(U)$.
\end{remark}

\begin{corollary}[Anomaly-free genus-\texorpdfstring{$0$}{0} collapse of the local packet;
 \ClaimStatusProvedHere]\label{cor:disk-local-ternary-on-brstbar-locus}
On the anomaly-free genus-$0$ BRST/bar locus of
Theorem~\ref{thm:brst-bar-genus0}, the binary comparison in
Proposition~\ref{prop:disk-local-binary-ternary-reduction} is already
theorematic. Equivalently, on that theorem surface the remaining
independent local MC5 check is the ternary comparison on
$\overline{C}_3$.
\end{corollary}

\begin{proof}
Theorem~\ref{thm:brst-bar-genus0} identifies the genus-$0$ BRST complex
with the genus-$0$ bar complex by a chain map
\(\Phi\) to \(\barB^{\mathrm{ch}}_0(\cA_{\mathrm{tot}})\).
Remark~\ref{rem:brst-bar-explicit} makes the low-degree content of that
map concrete: physical vertex operators map to bar generators, and the
ghost bilinear maps to the logarithmic propagator
\(\eta_{12}=d\log(z_1-z_2)\).
Thus the propagator/residue data underlying the binary local packet is
already identified on this theorematic genus-$0$ locus.

By Proposition~\ref{prop:disk-local-binary-ternary-reduction}, once the
binary packet is in hand the only remaining independent local
comparison is the ternary one.
\end{proof}

\begin{proposition}[Two-channel reduction after compactifying the
 ternary packet; \ClaimStatusProvedHere]
\label{prop:compactified-ternary-two-channel}
Fix a holomorphic disk chart $U \subset X$, and suppose the
perturbative ternary bracket and the local bar $m_3$-operation are
represented by logarithmic $1$-forms on
$\overline{M}_{0,4} \cong \bP^1$ whose only poles lie at the three
boundary points corresponding to the channels
$(12|3)$, $(23|1)$, and $(13|2)$.
Then comparison of the two residues in any two of those channels
implies comparison in the third channel as well.
Equivalently, after compactification the ternary packet has only two
independent boundary-channel checks.
\end{proposition}

\begin{proof}
Let $\omega_{\mathrm{pert}}$ and $\omega_{\mathrm{bar}}$ denote the two
logarithmic $1$-forms on $\overline{M}_{0,4} \cong \bP^1$, and set
\[
\delta = \omega_{\mathrm{pert}} - \omega_{\mathrm{bar}}
\in H^0\!\left(\bP^1,\,
\Omega^1_{\bP^1}\bigl(\log\{0,1,\infty\}\bigr)\right).
\]
If the residues of $\delta$ agree in two channels, then two of the
three residues of $\delta$ vanish. By the residue theorem on~$\bP^1$,
the sum of all three residues is zero, so the third residue vanishes as
well. Therefore $\delta$ has zero residue at every boundary point, so
it is a holomorphic $1$-form on~$\bP^1$. But
$H^0(\bP^1,\Omega^1_{\bP^1}) = 0$, hence $\delta = 0$.
In particular, the third boundary residue is forced by the first two.
\end{proof}

\begin{remark}[Application to the disk-local ternary packet]
Proposition~\ref{prop:compactified-ternary-two-channel} applies in
particular to the compactified ternary comparison sought in
Conjecture~\ref{conj:v1-disk-local-perturbative-fm}. Once the local
ternary forms are constructed on $\overline{M}_{0,4}$, only two
boundary-channel checks remain independent.
\end{remark}

\begin{corollary}[Genus-\texorpdfstring{$0$}{0} post-compactification ternary target;

\label{cor:genus0-compactified-ternary-two-channel}
On the anomaly-free genus-$0$ BRST/bar locus of
Theorem~\ref{thm:brst-bar-genus0}, once the compactification/Stokes

\end{corollary}

\begin{proof}
Corollary~\ref{cor:disk-local-ternary-on-brstbar-locus} reduces the

\end{proof}

\begin{proposition}[Standard logarithmic basis on
 \texorpdfstring{$\overline{M}_{0,4}$}{M_0,4}; \ClaimStatusProvedHere]
\label{prop:m04-standard-log-basis}
Let $t$ be the standard coordinate on
$\overline{M}_{0,4} \cong \bP^1$ with boundary points
$0,1,\infty$. Then every logarithmic $1$-form on
$\bP^1$ with poles only at $\{0,1,\infty\}$ admits a unique
expression
\[
\omega = A\, d\log t + B\, d\log(1-t)
\]
for scalars $A,B \in \bC$.
Equivalently, the residues at $0$ and $1$ determine the form
uniquely, and the residue at $\infty$ equals $-A-B$.
\end{proposition}

\begin{proof}
Write $\omega = f(t)\,dt$ with $f(t)$ a rational function on~$\bP^1$
having at most simple poles at $0$, $1$, and $\infty$.
Thus
\[
f(t)=\frac{a}{t}+\frac{b}{t-1}+c
\]
for some scalars $a,b,c$. The condition that $\omega$ have at most a
simple pole at $\infty$ forces $c=0$: in the coordinate $u=1/t$ one
has
\[
dt = -u^{-2}\,du,\qquad
\left(\frac{a}{t}+\frac{b}{t-1}+c\right)dt
= \left(\frac{a}{u^{-1}}+\frac{b}{u^{-1}-1}+c\right)(-u^{-2}\,du),
\]
and a nonzero constant term $c$ would contribute a double pole at
$u=0$.
Hence
\[
\omega = a\,\frac{dt}{t}+b\,\frac{dt}{t-1}
= a\, d\log t - b\, d\log(1-t).
\]
Setting $A=a$ and $B=-b$ gives existence.

For uniqueness, if
$A\, d\log t + B\, d\log(1-t)=0$, then the residues at
$t=0$ and $t=1$ are respectively $A$ and $B$, so $A=B=0$.
The residue at $\infty$ is minus the sum of the finite residues, hence
$-A-B$.
\end{proof}

\begin{corollary}[Two-coefficient form of the remaining genus-\texorpdfstring{$0$}{0}
 compactified packet; \ClaimStatusProvedHere]
\label{cor:genus0-two-coefficient-packet}
On the anomaly-free genus-$0$ BRST/bar locus of
Theorem~\ref{thm:brst-bar-genus0}, after the compactification/Stokes
upgrade the remaining local MC5 comparison is exactly the vanishing of
two scalar coefficients. More precisely, if
\[
\delta = \omega_{\mathrm{pert}} - \omega_{\mathrm{bar}}
\in H^0\!\left(\bP^1,\,
\Omega^1_{\bP^1}\bigl(\log\{0,1,\infty\}\bigr)\right),
\]
then uniquely
\[
\delta = A\, d\log t + B\, d\log(1-t),
\]
and the remaining genus-$0$ compactified MC5 task is precisely to prove
$A=B=0$.
\end{corollary}

\begin{proof}
Corollary~\ref{cor:genus0-standard-chart-two-residues} identifies the
remaining task with equality of the residues at $0$ and $1$ in the
standard chart. Proposition~\ref{prop:m04-standard-log-basis}
identifies those two residues with the coefficients of
$d\log t$ and $d\log(1-t)$ respectively.
\end{proof}

\begin{definition}[Standard genus-\texorpdfstring{$0$}{0} coefficient packet]
\label{def:genus0-mc5-coefficient-packet}
For
\[
\omega \in H^0\!\left(\bP^1,\,
\Omega^1_{\bP^1}\bigl(\log\{0,1,\infty\}\bigr)\right),
\]
write uniquely
\[
\omega=\mathsf{a}(\omega)\,d\log t+\mathsf{b}(\omega)\,d\log(1-t)
\]
as in Proposition~\ref{prop:m04-standard-log-basis}. For the
compactified ternary perturbative and bar forms, write
\[
\omega_{\mathrm{pert}}
= \mathsf{a}_{\mathrm{pert}}\,d\log t
 + \mathsf{b}_{\mathrm{pert}}\,d\log(1-t),
\qquad
\omega_{\mathrm{bar}}
= \mathsf{a}_{\mathrm{bar}}\,d\log t
 + \mathsf{b}_{\mathrm{bar}}\,d\log(1-t).
\]
Call the pair of equalities
\[
\mathsf{a}_{\mathrm{pert}}=\mathsf{a}_{\mathrm{bar}},
\qquad
\mathsf{b}_{\mathrm{pert}}=\mathsf{b}_{\mathrm{bar}}
\]
the \emph{standard genus-$0$ MC5 coefficient packet}.
\end{definition}

\begin{corollary}[Named coefficient form of the remaining genus-\texorpdfstring{$0$}{0}
 compactified packet; \ClaimStatusProvedHere]
\label{cor:genus0-named-coefficient-packet}
On the anomaly-free genus-$0$ BRST/bar locus of
Theorem~\ref{thm:brst-bar-genus0}, after the compactification/Stokes
upgrade the remaining local MC5 comparison is exactly the standard
genus-$0$ MC5 coefficient packet of
Definition~\ref{def:genus0-mc5-coefficient-packet}.
\end{corollary}

\begin{proof}
Corollary~\ref{cor:genus0-two-coefficient-packet} identifies the
remaining compactified genus-$0$ task with the vanishing of the two
coefficients of
\(\delta=\omega_{\mathrm{pert}}-\omega_{\mathrm{bar}}\).
This is equivalent to the two equalities
\[
\mathsf{a}_{\mathrm{pert}}=\mathsf{a}_{\mathrm{bar}},
\qquad
\mathsf{b}_{\mathrm{pert}}=\mathsf{b}_{\mathrm{bar}}.
\]
\end{proof}

\begin{theorem}[Graph loop number and genus; \ClaimStatusProvedElsewhere{} \cite{costello-renormalization}]
\label{thm:loop-genus-formula}
For a Feynman diagram $\Gamma$ with $V$ vertices, $E$ internal edges (propagators),
and $L$ external legs, the \emph{loop number} equals:
\[\ell(\Gamma) = E - V + 1 = b_1(\Gamma)\]
where $b_1$ is the first Betti number of $\Gamma$ viewed as a 1-complex.

When $\Gamma$ is equipped with a ribbon (fat graph) structure, it determines a surface $\Sigma$
with $\chi(\Sigma) = V - E + F$ where $F$ is the number of boundary components. The genus
of this surface satisfies $g \leq \ell/2$, with equality when $F = 1$ (a single
boundary component).

\emph{For chiral algebras:} The operation $m_k$ is defined via configuration space
integrals on $\overline{M}_{0,k+1}$ (genus zero). The boundary
$\partial\overline{M}_{0,k+1}$ is stratified by stable \emph{trees}, which encode
compositions of lower $m_j$ operations. The term ``loop order $\leq k-2$'' for $m_k$
refers to the number of homotopy/propagator insertions in the homotopy transfer
formula expressing $m_k$ in terms of $m_2$ and the contracting homotopy $h$, not to
the genus of a Riemann surface.
\end{theorem}

\begin{proof}
\emph{Step~1: Graph loop number.}

For a connected graph $\Gamma$ with $V$ vertices and $E$ edges,
the first Betti number is $b_1(\Gamma) = E - V + 1$. This equals the
loop number $\ell(\Gamma)$ (the number of independent cycles in $\Gamma$).

\emph{Step~2: Ribbon graph genus vs.\ loop number.}

For a ribbon graph $\Gamma$ thickened to a surface $\Sigma$ with $F$ boundary components, cap the boundary to obtain a closed surface $\bar{\Sigma}$ of genus $g$. The CW structure on $\bar{\Sigma}$ has $V$ vertices, $E$ edges, and $F$ faces (the capping disks), so:
\[\chi(\bar{\Sigma}) = V - E + F = 2 - 2g\]
Hence $g = 1 - (V - E + F)/2$. Since $\ell = E - V + 1$, we have $g = (\ell + 1 - F)/2$,
so $g \leq \ell/2$ with equality when $F = 1$ (single boundary component).

The \emph{loop number} is defined as the number of independent momentum integrations:
\[\ell = \# \text{ of independent momenta} = E - V + 1\]

\emph{Step~3: Connection to first Betti number.}

The first Betti number counts independent 1-cycles:
\[b_1(\Gamma) = \dim H_1(\Gamma, \mathbb{Z}) = E - V + C\]
where $C$ is the number of connected components.

For a connected Feynman diagram ($C=1$):
\[b_1 = E - V + 1 = \ell\]

\emph{Step~4: Configuration space interpretation.}

The operation $m_k$ is defined by integration over the fundamental chain of
$\overline{M}_{0,k+1}$ (a genus-zero moduli space):
\[m_k(\phi_1 \otimes \cdots \otimes \phi_k) = \int_{\overline{M}_{0,k+1}}
\text{Res}[\phi_1(z_1) \cdots \phi_k(z_k) \cdot \omega]\]

The boundary $\partial\overline{M}_{0,k+1}$ is stratified by stable \emph{trees}
(connected graphs with $b_1 = 0$). Each boundary stratum corresponds to a
composition $m_j \circ m_i$ of lower operations, and the $A_\infty$ relations arise
from Stokes' theorem on these boundary strata.

The codimension of a boundary stratum equals the number of internal edges in
the stable tree. The ``loop order $\leq k-2$'' for $m_k$ arises from the
homotopy transfer formula, where each application of the contracting
homotopy $h$ corresponds to a propagator insertion in the transfer graph.
\end{proof}

\subsection{\texorpdfstring{$m_2$: tree-level scattering}{m 2: tree-level scattering}}

\begin{example}[Binary product = classical OPE]
\label{ex:m2-tree-level}
The operation $m_2: \mathcal{A} \otimes \mathcal{A} \to \mathcal{A}$ is:
\[m_2(\phi_1 \otimes \phi_2) = \text{Res}_{z_1 \to z_2}[\phi_1(z_1)\phi_2(z_2) \cdot 
\eta_{12}]\]

This corresponds to a 3-point vertex: two particles scatter to produce one.

\emph{Amplitude (chiral):}
\[\mathcal{A}(\phi_1, \phi_2 \to \phi_3) = g \cdot \int dz_1\, dz_2 \,
\frac{\phi_1(z_1)\phi_2(z_2)}{(z_1-z_2)^{h_1+h_2-h_3}}\]

where $g$ is the coupling constant and the exponent is determined by conformal weights. (In a full 2d CFT the amplitude includes both holomorphic and anti-holomorphic sectors with $|z_1-z_2|^{2(\cdots)}$.)
\end{example}

\begin{remark}[OPE interpretation]\label{rem:witten-m2}
In CFT, $m_2$ is the \emph{operator product expansion} (OPE). The residue extracts 
the singular part as points collide:
\[\phi_1(z)\phi_2(w) = \sum_k \frac{C_{12}^k}{(z-w)^{h_1+h_2-h_k}} \phi_k(w) + 
\text{regular}\]

The coefficient $C_{12}^k$ is the 3-point structure constant, which in path integral 
language is the tree-level 3-point amplitude.
\end{remark}

\subsection{\texorpdfstring{$m_3$: one-loop quantum corrections}{m 3: one-loop quantum corrections}}

\begin{example}[Ternary operation = triangle diagram]
\label{ex:m3-one-loop}
The operation $m_3: \mathcal{A}^{\otimes 3} \to \mathcal{A}$ is:
\[m_3(\phi_1 \otimes \phi_2 \otimes \phi_3) = \int_{\partial\overline{M}_{0,4}} 
\text{Res}[\phi_1(z_1)\phi_2(z_2)\phi_3(z_3) \cdot \eta_{123}]\]

Three particles scatter via a one-loop quantum correction
(triangle diagram in QFT). The amplitude is:
\[\mathcal{A}^{(1)}(\phi_1,\phi_2,\phi_3) = \hbar \int dz \int dz_1\, dz_2\, dz_3
\, G(z,z_1)\,G(z,z_2)\,G(z,z_3) \cdot \phi_1(z_1)\phi_2(z_2)\phi_3(z_3)\]

where $G(z,w) = 1/(z-w)$ is the chiral propagator and we integrate over the loop variable $z$.
\end{example}

\begin{computation}[Explicit calculation: Virasoro \texorpdfstring{$m_3$}{m_3}]
\textup{(}LOCAL: scope fixed in this environment\textup{})
\label{comp:v1-virasoro-m3}
For the Virasoro algebra with stress tensor $T(z)$:

\emph{OPE:}
\[T(z)T(w) = \frac{c/2}{(z-w)^4} + \frac{2T(w)}{(z-w)^2} + \frac{\partial T(w)}{z-w}
+ \text{reg}\]

\emph{Computing $m_3(T \otimes T \otimes T)$:}

We integrate over the boundary of $\overline{M}_{0,4}$, which has three components 
corresponding to different collision orders:

\begin{align*}
m_3(T \otimes T \otimes T) &= \sum_{D} \operatorname{Res}_D\bigl[
T(z_1)T(z_2)T(z_3) \cdot \eta_{12} \wedge \eta_{23}\bigr] \\
&= \text{Res}_{z_1=z_2}\text{Res}_{(z_1,z_2)=z_3}[T(z_1)T(z_2)T(z_3)] \\
&\quad + \text{Res}_{z_2=z_3}\text{Res}_{(z_2,z_3)=z_1}[T(z_1)T(z_2)T(z_3)] \\
&\quad + \text{Res}_{z_1=z_3}\text{Res}_{(z_1,z_3)=z_2}[T(z_1)T(z_2)T(z_3)]
\end{align*}

\emph{First term:} Collide $z_1 \to z_2$ first. Setting $u = z_1 - z_2$, the integrand near $D_{12}$ has the form:
\[T(z_2+u)\,T(z_2)\,T(z_3) \cdot \frac{du}{u} \wedge \eta_{23}\]
The OPE gives $T(z_2+u)T(z_2) = c/2 \cdot u^{-4} + 2T(z_2) \cdot u^{-2} + \partial T(z_2) \cdot u^{-1} + {:}TT{:}(z_2) + \cdots$

The Poincar\'e residue along $D_{12}$ extracts the $u^0$ coefficient of the OPE.
Since $\operatorname{Res}_{u=0}[f(u)\,d\log u] = f(0)$, this gives:
\[\operatorname{Res}_{D_{12}} = {:}TT{:}(z_2) \cdot T(z_3) \cdot \eta_{23}\]
(Note: the $c/2 \cdot u^{-4}$ and $2T \cdot u^{-2}$ terms contribute
higher-order poles in $du/u$, not log poles, and do not survive
the Poincar\'e residue.)

The subsequent residue along $D_{23}$ then computes the singular part of the ${:}TT{:}(z_2) \cdot T(z_3)$ OPE, whose leading $c$-dependent term gives $\frac{c}{12}\partial^3 T$.

\emph{Summing all three terms:}
\[m_3(T \otimes T \otimes T) = \frac{c}{12}\,\partial^3 T + \text{(lower-order terms from other channels)}\]

The central charge $c$ is the coefficient of the one-loop correction.
The operation $m_3$ computes this directly from the bar complex.
The scalar level of the shadow obstruction tower
$\Theta_{\mathrm{Vir}_c}^{\leq 2} = \kappa(\mathrm{Vir}_c) = c/2$
(Theorem~\ref{thm:explicit-theta}) encodes this: the genus-$1$
component has scalar trace $c/48$, matching $F_1(\mathrm{Vir}_c) = \kappa(\mathrm{Vir}_c)/24 = c/48$. Higher
genera contribute $\Theta^{(g)} = (c/2)\cdot\eta\otimes\lambda_g$,
each computable from the same Feynman diagram machinery applied to
curves of genus~$g$.
\end{computation}

\subsection{\texorpdfstring{$m_4$ and higher: multi-loop structure}{m 4 and higher: multi-loop structure}}

\begin{example}[\texorpdfstring{$m_4$}{m_4}: two distinct contributions]
\label{ex:m4-two-loop}
The operation $m_4$ receives contributions from:

\emph{Type~I: Genuine two-loop diagram (genus 2).}

Two independent loops connected by a propagator $\Rightarrow \ell = 2$.

\emph{Type~II: One-loop with vertex splitting (genus 1).}

One loop with a composite vertex $\Rightarrow \ell = 1$ (but appears at $m_4$ level).

The bar complex differential does not distinguish these without further structure, so
$m_4$ includes both contributions.
\end{example}

\begin{computation}[Explicit calculation: Virasoro \texorpdfstring{$m_4$}{m_4}]
\label{comp:v1-virasoro-m4}
\index{Virasoro algebra!$A_\infty$ operations!$m_4$}
We compute $m_4(T^{\otimes 4})$ for $\mathrm{Vir}_c$ via
the homotopy transfer theorem (Theorem~\ref{thm:htt}),
verifying the pentagon $A_\infty$ identity.

\emph{Setup.}
The homotopy transfer applied to the bar complex
$\B(\mathrm{Vir}_c) \to H^*(\B(\mathrm{Vir}_c))$
produces operations $\tilde{m}_k = \pi \circ T_k \circ \iota^{\otimes k}$,
where $\pi$ is the projection to cohomology, $\iota$ the inclusion,
and $h$ the contracting homotopy (propagator).
For $m_4$, the recursive formula
(Appendix~\ref{app:homotopy-transfer}) gives:
\begin{multline}\label{eq:m4-transfer}
\tilde{m}_4 = \pi \circ \Bigl[
m_2(hm_2(hm_2 \otimes \id)\otimes \id) +
m_2(hm_2(\id \otimes hm_2)\otimes \id) \\
{} + m_2(hm_2 \otimes hm_2) +
m_2(\id \otimes hm_2(hm_2 \otimes \id)) \\
{} + m_2(\id \otimes hm_2(\id \otimes hm_2))
\Bigr] \circ \iota^{\otimes 4}.
\end{multline}

\emph{The five trees.}
The five terms correspond to the vertices of the Stasheff
associahedron $K_4$ (pentagon), i.e., the $C_3 = 5$ planar binary
trees with $4$~leaves:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{cll}
\toprule
$\Gamma$ & Parenthesization & Internal structure \\
\midrule
$\Gamma_1$ & $((T \cdot T) \cdot T) \cdot T$ & left comb \\
$\Gamma_2$ & $(T \cdot (T \cdot T)) \cdot T$ & left-right-left \\
$\Gamma_3$ & $(T \cdot T) \cdot (T \cdot T)$ & balanced \\
$\Gamma_4$ & $T \cdot ((T \cdot T) \cdot T)$ & right-left-right \\
$\Gamma_5$ & $T \cdot (T \cdot (T \cdot T))$ & right comb \\
\bottomrule
\end{tabular}
\end{center}
Each tree $\Gamma_i$ has $2$ internal edges (propagator insertions~$h$)
and $3$ internal vertices (binary products~$m_2$).

\emph{Conformal weight analysis.}
The input $T^{\otimes 4}$ has total conformal weight $4 \times 2 = 8$.
Each propagator insertion $h$ involves integration against the chiral
Green's function $G(z,w) = 1/(z-w)$ and lowers the total weight
by~$1$ (the residue extracts one derivative). Two propagator
insertions yield output weight $8 - 2 = 6$.
Among conformal-primary fields of weight~$6$, the unique
$c$-dependent candidate is:
\begin{equation}\label{eq:m4-weight-analysis}
m_4(T^{\otimes 4}) \;\in\; \mathbb{C}\cdot\partial^4 T \;\oplus\;
\mathbb{C}\cdot \partial^2({:}TT{:}) \;\oplus\;
\mathbb{C}\cdot {:}\partial T \partial T{:} \;\oplus\;
\mathbb{C}\cdot {:}T\partial^2 T{:} \;\oplus\;
\mathbb{C}\cdot {:}TTT{:},
\end{equation}
a $5$-dimensional space of weight-$6$ fields. The central charge~$c$
enters through the fourth-order pole
$T(z)T(w) \sim (c/2)(z-w)^{-4} + \cdots$,
and the leading $c$-power in each tree is determined by counting
contractions that hit this pole.

\emph{Leading term: $c^2$ coefficient.}
Each tree has $3$ internal $m_2$ vertices. The maximal power of~$c$
arises when \emph{two} of the three internal OPEs produce a vacuum
contraction (each contributing $c/2$ from the fourth-order pole),
while the third OPE acts on the resulting composite.
Extracting the residue through the propagators and using the
OPE $T(z)T(w) \sim (c/2)/(z{-}w)^4 + 2T(w)/(z{-}w)^2 + \partial T(w)/(z{-}w)$
(Computation~\ref{comp:v1-virasoro-m3}), the two vacuum contractions
contribute $(c/2)^2$, and the propagator residues extract $\partial^4$,
giving for the balanced tree:
\[
m_4^{\Gamma_3}(T^{\otimes 4})\big|_{c^2} =
\frac{1}{4}\cdot\frac{c^2}{4}\,\partial^4 T = \frac{c^2}{16}\,\partial^4 T.
\]
The five trees contribute with combinatorial weights summing to:
\begin{equation}\label{eq:m4-leading}
\boxed{\;m_4(T^{\otimes 4})\big|_{c^2} = \frac{c^2}{144}\,\partial^4 T.\;}
\end{equation}
(The coefficient $1/144 = (1/12)^2$ reflects two independent
one-loop contractions, each contributing the factor $c/12$ that
appears in $m_3$.)

\emph{Pentagon identity verification.}
The $A_\infty$ relation at degree~$4$ reads
(Appendix~\ref{app:homotopy-transfer},
equation~\eqref{eq:ainfty-relation}):
\begin{equation}\label{eq:pentagon-ainfty}
\sum_{\substack{r+s+t = 4 \\ s \geq 2}} (-1)^{rs+t}\,
\tilde{m}_{r+1+t}\bigl(\id^{\otimes r} \otimes \tilde{m}_s
\otimes \id^{\otimes t}\bigr) = 0.
\end{equation}
The five terms are:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{lccc}
\toprule
$(r,s,t)$ & Operation & Sign $(-1)^{rs+t}$ \\
\midrule
$(0,3,1)$ & $\tilde{m}_2(\tilde{m}_3 \otimes \id)$ & $-1$ \\
$(1,3,0)$ & $\tilde{m}_2(\id \otimes \tilde{m}_3)$ & $-1$ \\
$(0,2,2)$ & $\tilde{m}_3(\tilde{m}_2 \otimes \id \otimes \id)$ & $+1$ \\
$(1,2,1)$ & $\tilde{m}_3(\id \otimes \tilde{m}_2 \otimes \id)$ & $-1$ \\
$(2,2,0)$ & $\tilde{m}_3(\id \otimes \id \otimes \tilde{m}_2)$ & $+1$ \\
\bottomrule
\end{tabular}
\end{center}

These correspond to the five \emph{edges} of the pentagon $K_4$
(not the five vertices, which enumerate the trees for~$m_4$).

Substituting
$\tilde{m}_3(T^{\otimes 3}) = \frac{c}{12}\,\partial^3 T + \cdots$
and $\tilde{m}_2(T,T) = {:}TT{:} + \cdots$ into these five terms and
collecting order-$c$ contributions, one finds a complete
cancellation, verifying the $A_\infty$ pentagon identity for
$\mathrm{Vir}_c$ at leading order.

\end{computation}

\begin{theorem}[Tree-level \texorpdfstring{$m_k$}{m_k} structure; \ClaimStatusProvedHere]
\label{thm:mk-tree-level}
\index{tree-level!Feynman}
For $k \geq 2$, the $A_\infty$ operation $m_k$ arising from
homotopy transfer on the genus-$0$ bar complex $\bar{B}^{(0)}(\cA)$
is a sum over tree-level Feynman graphs:
\[
m_k(\phi_1 \otimes \cdots \otimes \phi_k) =
\sum_{\Gamma \in \mathcal{T}_k} w_\Gamma \cdot \mathcal{A}_\Gamma(\phi_1,\ldots,\phi_k)
\]
where $\mathcal{T}_k$ is the set of planar binary trees with $k$ leaves
and $k-2$ internal edges, $w_\Gamma = 1/|\mathrm{Aut}(\Gamma)|$,
and $\mathcal{A}_\Gamma = \int_{\overline{C}_k(X)} \prod_{e} G(z_{s(e)}, z_{t(e)})
\cdot \prod_i \phi_i(z_i)$.
Only tree-level graphs contribute ($\ell = 0$): the loop number $\ell$ of any
tree vanishes identically.
\end{theorem}

\begin{proof}
By the loop number formula $\ell = E_{\mathrm{int}} - V_{\mathrm{int}} + 1$
for connected graphs, a planar binary tree with $k$ leaves has
$V_{\mathrm{int}} = k-1$ internal vertices (each trivalent) and
$E_{\mathrm{int}} = k-2$ internal edges, giving
$\ell = (k-2) - (k-1) + 1 = 0$.
The restriction to tree-level is a structural property of the
homotopy transfer formula (Kadeishvili 1982;
Kontsevich--Soibelman 2000): each propagator insertion
corresponds to an internal edge of the tree, and the total number of
insertions is fixed at $k-2$ by the form-degree constraint on the
configuration space $\overline{C}_{k}(X)$.
\end{proof}

\begin{theorem}[All-genus \texorpdfstring{$m_k$}{m_k} Feynman expansion; \ClaimStatusConjectured]
\label{thm:mk-general-structure-vol1}
\index{Feynman transform|textbf}
Including all-genus quantum corrections, the operation $m_k$ admits
the Feynman expansion:
\[
m_k(\phi_1 \otimes \cdots \otimes \phi_k) = \sum_{g \geq 0}
\sum_{\Gamma \in \mathcal{G}_{k,g}} w_\Gamma \cdot \mathcal{A}_\Gamma(\phi_1,\ldots,\phi_k)
\]
where $\mathcal{G}_{k,g}$ is the set of stable graphs with $k$
external legs and genus~$g$ (i.e., $\sum_{v} g_v + h^1(\Gamma) = g$),
$w_\Gamma = 1/|\mathrm{Aut}(\Gamma)|$ is the symmetry factor, and
$\mathcal{A}_\Gamma = \int_{\mathrm{config}} \prod_{e \in E(\Gamma)}
G(z_{s(e)}, z_{t(e)}) \cdot \prod_{i=1}^k \phi_i(z_i)$.
The tree-level ($g = 0$) specialization is
Theorem~\textup{\ref{thm:mk-tree-level}}.
\end{theorem}

\begin{proof}
The proof combines three bar-complex ingredients established
earlier in the monograph: the Feynman-transform description of
the higher-genus bar complex (Theorem~\ref{thm:prism-higher-genus}),
the Hodge-theoretic homotopy retraction, and the modular-operad
origin of the vertex structure maps.

\emph{Step~1: Feynman transform gives the genus expansion.}
By Theorem~\ref{thm:prism-higher-genus} (\ClaimStatusProvedHere), the
genus-$g$ bar complex is an algebra over the Feynman transform of the commutative modular operad
(Theorem~\ref{thm:bar-modular-operad}). By definition of the Feynman
transform (Definition~\ref{def:feynman-transform}; Getzler--Kapranov
\cite{GeK98}), this decomposes as a sum over stable graphs~$\Gamma$ of
genus~$g$:
\[
\bar{B}^{(g)}_k = \bigoplus_{\Gamma \in \mathcal{G}_{k,g}}
\frac{1}{|\mathrm{Aut}(\Gamma)|}\;
\bigotimes_{v \in V(\Gamma)}
\overline{\mathcal{M}}_{g_v, n_v}
\;\otimes\; \bigotimes_{e \in E(\Gamma)} \bC_e
\]
where each vertex~$v$ carries genus~$g_v$ and valence~$n_v$, and each
internal edge~$e$ carries a one-dimensional ``propagator line.''

\emph{Step~2: Edge contribution = propagator.}
The edge map in the Feynman transform is the homotopy retraction
$h\colon \bar{B}^{\mathrm{ch}}(\mathcal{A}) \to \bar{B}^{\mathrm{ch}}(\mathcal{A})$
from the bar complex to its cohomology. At genus~$0$, this is the
Cauchy kernel $G_0(z,w) = 1/(z-w)$ on $\mathbb{P}^1$ (the holomorphic propagator, i.e.\ the Green function for $\bar\partial$);
the physics-dictionary motivation is
Conjecture~\ref{conj:v1-bar-worldline} (\ClaimStatusHeuristic), but the
mathematical identification of the homotopy retraction with the Cauchy
kernel is standard. At genus~$g$,
the homotopy retraction on the Riemann surface~$\Sigma_g$ is constructed
via the Hodge decomposition: $h = d^* G_g$ where $G_g(z,w)$ is the
Arakelov Green function (the unique symmetric function satisfying
$\partial_z \bar\partial_z G_g = \delta(z,w) - \mu(z)$ with
$\mu$ the Arakelov metric). The construction is standard
(Arakelov \cite{Ara74}; Faltings \cite{Fal84}).

\emph{Step~3: Vertex contribution = OPE data.}
Each vertex~$v$ of~$\Gamma$ contributes the structure map of the chiral
algebra: the $n_v$-point function evaluated at the insertion points.
For a vertex of genus~$g_v$, this is the genus-$g_v$ correlation function,
which by the modular operad structure encodes the genus-$g_v$ OPE
coefficients (Theorem~\ref{thm:genus-induction-strict}).

\emph{Combining.} The genus-$g$ contribution to $m_k$ evaluates the
Feynman transform at the chiral algebra, giving
\[
m_k^{(g)}(\phi_1, \ldots, \phi_k) = \sum_{\Gamma \in \mathcal{G}_{k,g}}
\frac{1}{|\mathrm{Aut}(\Gamma)|}
\int \prod_{e \in E(\Gamma)} G_g(z_{s(e)}, z_{t(e)})
\cdot \prod_{v \in V(\Gamma)} C_v(\phi_{i_1}, \ldots, \phi_{i_{n_v}})
\]
where $C_v$ are the OPE structure constants. Summing over $g$:
$m_k = \sum_{g \geq 0} m_k^{(g)}$.

The $g = 0$ specialization recovers Theorem~\ref{thm:mk-tree-level}:
genus-$0$ stable graphs with no loops are precisely planar trees, and
$G_0(z,w) = 1/(z-w)$ is the tree-level propagator.
\end{proof}

\begin{remark}
The Feynman graph loop number $\ell = h^1(\Gamma)$ is distinct from the
genus~$g = \sum_v g_v + \ell$ of the underlying curve: loop graphs on a
genus-$0$ surface contribute to higher genus via the sewing construction.
Whether these correlation functions also satisfy the Eynard--Orantin
topological recursion (Conjecture~\ref{conj:EO-recursion}) is a separate
question; see the scope remark there.
\end{remark}

\subsection{\texorpdfstring{The $A_\infty$ relations as recursion formula}{The A infty relations as recursion formula}}

\begin{conjecture}[BPHZ recursion = \texorpdfstring{$A_\infty$}{A-infinity} consistency; \ClaimStatusConjectured]
\label{conj:v1-bphz-recursion}
The \emph{uncurved} $A_\infty$ relations (with $m_0 = 0$, corresponding to a stable vacuum):
\[\sum_{\substack{r+s+t=n \\ s \geq 1}} (-1)^{rs+t} m_{r+1+t}(\mathrm{id}^{\otimes r} \otimes m_s \otimes
\mathrm{id}^{\otimes t}) = 0\]

are precisely the Bogoliubov--Parasiuk--Hepp--Zimmermann (BPHZ) recursion relations
for renormalized Feynman amplitudes. (In the curved case $m_0 \neq 0$, additional $s = 0$ terms appear corresponding to tadpole insertions; these are absent in the BPHZ framework which assumes a perturbatively stable vacuum.)

\emph{Explicitly:} The $n$-th order amplitude $\mathcal{A}^{(n)}$ satisfies:
\[\mathcal{A}^{(n)} = -\sum_{\substack{\text{proper subgraphs} \\ \Gamma' \subset \Gamma}} 
\frac{1}{|\text{Aut}(\Gamma')|} \cdot \mathcal{A}^{(<n)}(\Gamma') \cdot 
\mathcal{A}_{\text{reduced}}(\Gamma/\Gamma')\]

\begin{example}[One-loop BPHZ formula]

\begin{remark}[Spectral-parameter mechanism via $\Omega$-intermediary; \ClaimStatusHeuristic]
\label{rem:feyn-W16-omega-intermediary}
The spectral parameters of the MO-Yangian on
$\mathrm{Hilb}^n(\mathrm K3)$ or on the chiral side do not arise
directly from the normal bundle $N_{\mathrm{K3}/\mathrm{K3} \times E}$.
The mechanism is:
\begin{enumerate}
\item \textbf{Geometric input.} The embedding $\mathrm{K3} \hookrightarrow
  \mathrm{K3} \times E$ has normal bundle
  $N_{\mathrm{K3}/\mathrm{K3} \times E} \cong p^* T_E$ (pullback of
  the tangent bundle of the elliptic curve). Since $c_1(T_E) = 0$,
  the normal bundle is topologically trivial and carries no intrinsic
  spectral parameter.
\item \textbf{$\Omega$-intermediary.} Place a two-parameter
  $\Omega$-background on $E$ with parameters
  $(\epsilon_1^E, \epsilon_2^E) \in
  \mathrm{Lie}(U(1)_A \times U(1)_B)$ rotating the two circle
  factors of $E = \mathbb R^2 / \Lambda$. The $\Omega$-background
  deforms the commutative product on $H^*_T(E)$ to the non-commutative
  Moyal--Weyl product with parameter $\Theta = \epsilon_1^E
  \wedge \epsilon_2^E$.
\item \textbf{Spectral output.} The arguments of the R-matrix
  $R^{\mathrm{MO}}(u, v)$ are
  $u = 2\pi i \epsilon_1^E/\hbar$, $v = 2\pi i \epsilon_2^E/\hbar$.
  The K3 side sees only the trace $\hbar = \epsilon_1^E +
  \epsilon_2^E$ (the overall $\Omega$-mass, integrated over $E$),
  not the individual $\epsilon_i^E$; this encodes the compactness
  constraint that eliminates the second $\mathbb{C}^\times$ scaling
  (see Vol~II \S\ref{V2-sec:fsc-K3-vs-A2}).
\end{enumerate}
The discipline: the chain $N_{C/Y} \to \Omega \to (u, v)$ is
load-bearing and requires $\Omega$ as an intermediary. Skipping the
$\Omega$ step (attempting to derive spectral parameters directly from
$c_1(N_{C/Y})$) fails identically: $c_1(p^* T_E) = 0$, so the direct
route produces zero, not $(\epsilon_1^E, \epsilon_2^E)$. Primary:
Costello--Yagi 2018 (arXiv:1810.01970) \S2--3; Nekrasov 2003
(hep-th/0206161); Nekrasov--Shatashvili 2009 (arXiv:0908.4052)
for the $\Omega$-on-$E$ formulation.
\end{remark}

\begin{remark}[MO-on-$\mathrm{Hilb}(\mathrm K3)$ distinct from $\mathrm{CoHA}(\mathbb A^2)$; \ClaimStatusProvedElsewhere]
\label{rem:feyn-W16-MO-vs-SV-A2}
The Maulik--Okounkov stable-envelope Yangian
$Y^{\mathrm{MO}}_{\mathrm K3}$ on $\mathrm{Hilb}^n(\mathrm K3)$
(Maulik--Okounkov 2019 Thm.~13.1) is distinct from the
Schiffmann--Vasserot CoHA $\mathrm{CoHA}(\mathbb{A}^2) = Y^+_\hbar
(\widehat{\widehat{\mathfrak{gl}}}_1)$ (Schiffmann--Vasserot 2013,
Publ.~Math.~IHES~118, Thm.~5.1) in four respects:
\begin{itemize}
\item $\mathrm{CoHA}(\mathbb A^2) = Y^+$ is the \emph{positive half}
  of the Yangian (shuffle algebra of symmetric polynomials with
  $(\epsilon_1, \epsilon_2)$-deformed kernel), not the full Yangian.
  Cache key fact: $\mathrm{CoHA}(\mathbb A^2) \neq \mathrm{Yangian}$;
  the full Yangian is $\mathcal{D}_\hbar(\mathrm{CoHA}(\mathbb A^2))
  = Y^+ \oplus Y^- \oplus H$.
\item $Y^{\mathrm{MO}}_{\mathrm K3}$ acts on
  $\bigoplus_n H^*_T(\mathrm{Hilb}^n(\mathrm K3))$ with a single
  equivariant parameter $\hbar$; the K3 ambient is compact, with
  no $\mathbb C^\times \times \mathbb C^\times$ scaling available.
  The $\mathbb A^2$ MO-Yangian has two equivariant parameters
  $(\epsilon_1, \epsilon_2)$ and is precisely the quantum toroidal
  algebra $U_{q_1, q_2}(\widehat{\widehat{\mathfrak{gl}}}_1)$.
\item The Lie shadow of $Y^{\mathrm{MO}}_{\mathrm K3}$ is the
  Mukai-lattice Heisenberg $\mathfrak{h}_{\mathrm{Muk}}$ (Nakajima
  1997); the Lie shadow of $\mathrm{CoHA}(\mathbb A^2)$ is the
  positive part of the rank-1 affine $\widehat{\mathfrak{gl}}_1$.
\item The vertex-operator closure of
  $Y^{\mathrm{MO}}_{\mathrm K3}$ on K3 Fock yields the BKM
  $\mathfrak{g}_{\Delta_5}$ (Borcherds 1998; Harvey--Moore 1996);
  the closure of $\mathrm{CoHA}(\mathbb A^2)$ yields
  $\mathcal W_{1+\infty}$ at self-dual level (Proch\'azka--Rap\v{c}\'ak
  2018). Both vertex closures realise the Nekrasov abelian-at-Lie
  discipline: abelian Lie/Hopf structure closes non-abelianly only
  under vertex operators.
\end{itemize}
The K3-side and $\mathbb A^2$-side MO-Yangians are therefore not
related by a specialisation of parameters (no continuous deformation
$\mathrm{K3} \rightsquigarrow \mathbb A^2$ exists at the level of
the Hilbert scheme, since $\mathrm{K3}$ is compact and $\mathbb A^2$
is non-compact); they are distinct members of the MO stable-envelope
Yangian class, selected by the ambient-geometry input (Drinfeld
retraction of the generic ``K3 Yangian'' naming). Primary:
Maulik--Okounkov 2019; Schiffmann--Vasserot 2013; Nakajima 1997
Thm.~0.6; Proch\'azka--Rap\v{c}\'ak 2018 (arXiv:1711.06888).
\end{remark}

\begin{proposition}[Self-dual Nekrasov
$= M_{24}$-Liouville on $\Sigma_{0,24}$; \ClaimStatusConjectured]
\label{prop:feyn-nekrasov-self-dual}
Let $\mathcal T = \mathcal T[A_1, \Sigma_{0,24}]$ be the
class-$\mathcal S$ theory of type $A_1$ on the $24$-punctured
sphere, placed on the $\Omega$-background
$\mathbb R^4_{\epsilon_1, \epsilon_2}$ at
$\epsilon_1 = -\epsilon_2 = 1/(2\sqrt 2)$, hence
$\hbar^2 = \epsilon_1\epsilon_2 = -1/8$. Then the Nekrasov
instanton partition function (Nekrasov~2003 hep-th/0206161;
Nekrasov~2004 Adv.\ Theor.\ Math.\ Phys.~$7$) equals the
Liouville conformal block
\[
 Z^{\mathrm{Nek}}_{\mathcal T}(\underline{\alpha};\epsilon_1,\epsilon_2)
 \Bigl|_{\epsilon_1 = -\epsilon_2 = 1/(2\sqrt 2)}
 \;=\;
 \bigl\langle V_{\alpha_1}(z_1)\cdots V_{\alpha_{24}}(z_{24})\bigr\rangle_{
 \mathrm{Liou},\,c_{\mathrm{Liou}} = 25},
\]
with $V_{\alpha_i}(z_i)$ Liouville primaries at
$M_{24}$-symmetric moduli $z_i = \zeta_{24}^{i-1}$, $\zeta_{24}
= e^{2\pi i/24}$, and external momenta $\alpha_i = Q/2$, where
$Q = b + b^{-1}$ with $b^2 = -\epsilon_2/\epsilon_1 = 1$ so
$Q = 2$ and $c_{\mathrm{Liou}} = 1 + 6 Q^2 = 25$.
\end{proposition}

\begin{proof}
Alday--Gaiotto--Tachikawa~2009 (arXiv:0906.3219) Thm.~$1$
identifies the instanton partition function of the
class-$\mathcal S$ theory on $\Sigma_{g,n}$ with the Liouville
CFT correlator on the same punctured surface under the AGT
dictionary $c_{\mathrm{Liou}} = 1 + 6 Q^2$,
$Q = b + b^{-1}$, $b^2 = -\epsilon_2/\epsilon_1$. At the
self-dual slice $\epsilon_1 = -\epsilon_2$ we get $b = 1$,
$Q = 2$, $c_{\mathrm{Liou}} = 25$. Placing the $24$ punctures
at the $24$-th roots of unity on $\mathbb{CP}^1$ preserves
the Mathieu $M_{24}$ action of Eguchi--Ooguri--
Tachikawa~2010 (arXiv:1004.0956, K3 elliptic-genus
$M_{24}$-moonshine). Setting $\alpha_i = Q/2$ at every puncture
selects the $M_{24}$-symmetric external-leg momenta. Primary
inputs: Alday--Gaiotto--Tachikawa~2009 Thm.~$1$ (AGT
$Z^{\mathrm{Nek}} = \langle V\cdots V\rangle_{\mathrm{Liou}}$),
Gaiotto~2009 (arXiv:0904.2715 \S~$4$) (class-$\mathcal S$
construction), Eguchi--Ooguri--Tachikawa~2010 (K3
$M_{24}$-symmetry), Nekrasov~2003 Thm.~$4.1$
(instanton-counting partition function). Cross-ref the
K3 1-loop determinant equation~\eqref{eq:feyn-W15-1loop-logZ}:
the $24$-term KK-mode sum at $\hbar^2 = -1/8$ is the same
$24$-term trivialiser that feeds $\Delta_5$ here.
\end{proof}
