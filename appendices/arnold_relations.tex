\section{Arnold relations: historical development and attribution}
\label{sec:arnold-historical}
\index{Arnold relations!detailed|textbf}

\subsection{Historical context}

Arnold~\cite{Arnold69} proved that $H^*(\mathrm{Conf}_n(\mathbb{C}), \mathbb{Z})$ is generated by classes $\omega_{ij}$ subject to
\begin{equation}\label{eq:arnold-original}
\omega_{ij} \omega_{jk} + \omega_{jk} \omega_{ki} + \omega_{ki} \omega_{ij} = 0
\end{equation}
for distinct $i, j, k$.  Brieskorn~\cite{Brieskorn73} extended this to arbitrary hyperplane arrangements via the deletion-restriction exact sequence.  The Orlik--Solomon theorem~\cite{OS80} gives the definitive algebraic form: $H^*(M(\mathcal{A}), \mathbb{Z}) \cong A(\mathcal{A})$, the exterior algebra on arrangement generators modulo Arnold relations.

The connection to chiral algebras runs through Kontsevich's formality theorem~\cite{Kontsevich97} (configuration space integrals satisfying $d^2 = 0$ via Arnold relations), Beilinson--Drinfeld~\cite{BD04} (Arnold relations in the chiral bar construction), Costello--Gwilliam~\cite{CG17} (Arnold relations as locality in factorization algebras), and recent extensions to derived categories and higher genus~\cite{GLZ22, FG12}.

\subsection{Our contribution: geometric realization at all genera}

\begin{remark}[Novel contributions]
While Arnold (1969), Brieskorn (1973), and Orlik--Solomon (1980) established the relations 
for configuration spaces in $\mathbb{C}$ (genus 0), we extend their work to:

\noindent\emph{Higher genus} (Theorem \ref{thm:arnold-higher-genus}).
Arnold relations hold on configuration spaces $\text{Conf}_n(X)$ for curves $X$ of ANY genus $g$:
$$\eta_{ij} \wedge \eta_{jk} + \eta_{jk} \wedge \eta_{ki} + \eta_{ki} \wedge \eta_{ij} = 0 
\quad \text{in } H^2(\text{Conf}_n(X))$$

\noindent\emph{Quantum corrections} (Theorem \ref{thm:arnold-quantum}).
With central charge $\mu_0 \in Z(\mathcal{A})$, the modified relations:
$$\Dg{g}(\eta_{ij} \wedge \eta_{jk}) + \Dg{g}(\eta_{jk} \wedge \eta_{ki})
+ \Dg{g}(\eta_{ki} \wedge \eta_{ij}) = \mu_0 \otimes \omega_g$$
still ensure $\Dg{g}^{\,2} = 0$ strictly.

\noindent\emph{Chiral algebra interpretation} (\S\ref{sec:arnold-chiral}).
We give a complete dictionary between:
\begin{itemize}
\item Arnold's topological relations $\leftrightarrow$ Bar differential nilpotence
\item Brieskorn's deletion-restriction sequence $\leftrightarrow$ Spectral sequence of bar complex
\item Orlik--Solomon algebra $\leftrightarrow$ Cohomology of bar construction
\end{itemize}
\end{remark}

\section{Arnold relations}

\subsection{Arnold relations for hyperplane arrangements}

The cohomology ring $H^*(C_n(\mathbb{C}), \mathbb{Z})$ is generated by logarithmic $1$-forms
$$\omega_{ij} = \frac{1}{2\pi i} d\log(z_i - z_j)$$
subject to the quadratic relations discovered by Arnold~\cite{Arnold69}.

\section{The relations: elementary statement and first examples}

\subsection{The fundamental identity}

\begin{theorem}[Arnold relations \cite{Arn69}; \ClaimStatusProvedElsewhere]
\label{thm:arnold-relations-appendix}
For any configuration of points $z_1, \ldots, z_n$ in $\mathbb{C}$, define the logarithmic 1-forms:
$$\eta_{ij} = d\log(z_i - z_j) = \frac{dz_i - dz_j}{z_i - z_j}$$

Then for any three distinct indices $i, j, k \in \{1, \ldots, n\}$:
$$\eta_{ij} \wedge \eta_{jk} + \eta_{jk} \wedge \eta_{ki} + \eta_{ki} \wedge \eta_{ij} = 0$$
These three-term relations generate the complete ideal of relations in the cohomology algebra $H^*(C_n(\mathbb{C}), \mathbb{C})$, which is therefore the Orlik--Solomon algebra of the braid arrangement.
\end{theorem}

\subsection{Examples}

For three points, the Arnold relation reads
$\eta_{ij} \wedge \eta_{jk} + \eta_{jk} \wedge \eta_{ki} + \eta_{ki} \wedge \eta_{ij} = 0$,
proved by the partial fractions identity $(z_k - z_i) + (z_i - z_j) + (z_j - z_k) = 0$.
For four points, the $\binom{4}{3} = 4$ Arnold relations generate all relations in $H^*(C_4(\mathbb{C}))$; higher-degree relations are consequences of the quadratic ones by the Orlik--Solomon theorem.

\section{The first proof: elementary combinatorics}\label{sec:first-complete-proof}

\begin{proof}[Proof of Theorem~\ref{thm:arnold-relations-appendix}]
The partial fractions identity
$\frac{1}{(z_i - z_j)(z_j - z_k)} + \frac{1}{(z_j - z_k)(z_k - z_i)} + \frac{1}{(z_k - z_i)(z_i - z_j)} = 0$
(numerator: $(z_k - z_i) + (z_i - z_j) + (z_j - z_k) = 0$) directly implies the Arnold relation, since
$\eta_{ij} \wedge \eta_{jk} = \frac{(dz_i - dz_j) \wedge (dz_j - dz_k)}{(z_i - z_j)(z_j - z_k)}$
and the three numerator $2$-forms coincide.  Completeness: the quotient algebra has Poincar\'{e} polynomial $\prod_{j=0}^{n-1}(1+jt)$, matching Arnold's computation~\cite{Arnold69}.
\end{proof}

\section{The second proof: topology and integration}

\begin{remark}[Topological heuristic]\label{rem:topological-heuristic}
Consider the map $\Phi: S^1 \times C_{|S|}(\mathbb{C}) \to C_{|S|+2}(\mathbb{C})$ defined by $\Phi(e^{i\theta}, w_1, \ldots, w_{|S|}) = (z_i, z_i + \epsilon e^{i\theta}, w_1, \ldots, w_{|S|})$, placing $z_j$ on a small circle around $z_i$.

The Arnold relation for three points $z_i, z_j, z_k$ follows from the identity $\eta_{ij} = d\log(z_i - z_j)$ by computing the $S^1$-integral:
$$\frac{1}{2\pi i}\oint_{|z_j - z_i| = \epsilon} \eta_{ij} \wedge \eta_{jk} = \text{Res}_{z_j = z_i}\left[\frac{dz_j}{z_j - z_i} \wedge \frac{dz_j}{z_j - z_k}\right] = \eta_{ik}$$
The Arnold relation then arises from the partial fractions decomposition of the integrand.  This is a \emph{heuristic motivation}: the rigorous proof is the partial fractions identity given in \S\ref{sec:first-complete-proof} below.
\end{remark}

\section{The third proof: operadic structure}

\begin{definition}[Configuration space operad]
The collection $\{\mathcal{C}_n = \overline{C}_n(\mathbb{C})\}_{n \geq 0}$ forms an operad with:
\begin{itemize}
\item $\mathcal{C}_n$ represents ``$n$-ary operations''
\item Composition $\gamma_i: \mathcal{C}_n \times \mathcal{C}_m \to \mathcal{C}_{n+m-1}$ given by inserting configurations
\item Unit $1 \in \mathcal{C}_1$ is the identity operation
\end{itemize}
\end{definition}

\begin{proposition}[Operadic equivalence: Arnold relations $\Leftrightarrow$ $d^2 = 0$]\label{prop:operadic-equivalence-arnold}
The Arnold relations for the forms $\eta_{ij}$ are \emph{equivalent} to the condition $d^2 = 0$ on the configuration space operad.
\end{proposition}

\begin{proof}
The configuration space operad has a natural differential $d = \sum_{i<j} \partial_{ij}$ where $\partial_{ij}$ corresponds to taking the residue at the collision divisor $D_{ij}$.  Computing:
\begin{align}
d^2 &= \sum_{i<j} \partial_{ij}^2 + \sum_{\substack{(i,j) \neq (k,l) \\ \{i,j\} \cap \{k,l\} = \emptyset}} \partial_{ij} \partial_{kl} + \sum_{i<j<k} (\partial_{ij}\partial_{jk} + \partial_{ij}\partial_{ik} + \partial_{jk}\partial_{ik})
\end{align}
The first sum vanishes ($\partial_{ij}^2 = 0$: a simple pole has no iterated residue).  For disjoint index pairs, $\partial_{ij}$ and $\partial_{kl}$ have degree~$1$ (cohomological) and act on independent factors, so they \emph{anticommute}: $\partial_{ij}\partial_{kl} + \partial_{kl}\partial_{ij} = 0$.  The remaining triple-index terms give $d^2 = 0$ if and only if $\partial_{ij}\partial_{jk} + \partial_{jk}\partial_{ki} + \partial_{ki}\partial_{ij} = 0$ for all triples $\{i,j,k\}$.

Under the correspondence $\partial_{ij} \leftrightarrow \text{Res}_{D_{ij}}$ and composition $\leftrightarrow$ wedge product, this triple condition is equivalent to the Arnold relation $\eta_{ij} \wedge \eta_{jk} + \eta_{jk} \wedge \eta_{ki} + \eta_{ki} \wedge \eta_{ij} = 0$.  In one direction, assuming the Arnold relation, the triple residue terms vanish and $d^2 = 0$ follows.  In the other direction, $d^2 = 0$ applied to appropriate test forms forces the Arnold relation.
\end{proof}

\section{Consequences for the bar complex}

\begin{theorem}[Bar differential squares to zero; \ClaimStatusProvedHere]
The bar differential
$$d = d_{\text{internal}} + d_{\text{residue}} + d_{\text{de Rham}}$$
satisfies $d^2 = 0$ if and only if the Arnold relations hold.
\end{theorem}

\begin{proof}
The key term is $d_{\text{residue}}^2$. Computing:
\begin{align}
d_{\text{residue}}^2 &= \left(\sum_{i<j} \text{Res}_{D_{ij}}\right)^2 \\
&= \sum_{i<j<k} \left(\text{Res}_{D_{ij}} \circ \text{Res}_{D_{jk}} + \text{cyclic}\right)
\end{align}

Each triple term corresponds to an Arnold relation with $|S| = 1$. The vanishing of $d_{\text{residue}}^2$ is equivalent to:
$$\text{Res}_{D_{ij}}[\text{Res}_{D_{jk}}[\omega]] + \text{cyclic} = 0$$

This is what the Arnold relations guarantee.
\end{proof}

\end{remark}

\section{Arnold relations: deletion-restriction sequence}
\label{sec:arnold-nine-term}

\begin{theorem}[Arnold relations and braid arrangement cohomology; \ClaimStatusProvedElsewhere]\label{thm:arnold-complete-exact}
The relations among logarithmic 1-forms on configuration spaces are completely 
characterized by the cohomology of the complement of the braid arrangement, as 
first established by Arnold \cite{Arnold69}.

For $n$ points, the cohomology $H^1(C_n(\mathbb{C}), \mathbb{C})$ is generated by 
logarithmic forms $\eta_{ij}$ subject to Arnold relations.
\end{theorem}

\begin{proof}

We provide three complementary proofs:

\emph{Proof~1: Combinatorial (\`a la Arnold).}

Arnold's original proof \cite{Arnold69} uses the Orlik--Solomon algebra.

\begin{definition}[Orlik--Solomon algebra]\label{def:orlik-solomon-arnold}
\index{Orlik--Solomon!algebra}
For the braid arrangement $\mathcal{A} = \{H_{ij}\}$ where $H_{ij} = \{z_i = z_j\}$, 
the Orlik--Solomon algebra is:
$$\text{OS}(\mathcal{A}) = \mathbb{C}\langle e_{ij} \mid i < j \rangle / I$$
where $I$ is the ideal generated by:
\begin{enumerate}
\item $e_{ij}^2 = 0$
\item $e_{ij} \wedge e_{jk} + e_{jk} \wedge e_{ki} + e_{ki} \wedge e_{ij} = 0$ (Arnold relation)
\end{enumerate}
\end{definition}

\begin{lemma}[OS computes cohomology \cite{OS80}; \ClaimStatusProvedElsewhere]\label{lem:OS-cohomology-arnold}
$$H^*(C_n(\mathbb{C}), \mathbb{C}) \simeq \text{OS}(\mathcal{A})$$
\end{lemma}

\begin{proof}[Proof of Lemma]
The logarithmic forms $\eta_{ij} = d\log(z_i - z_j)$ generate $H^1(C_n(\mathbb{C}))$.

They satisfy:
\begin{enumerate}
\item $\eta_{ij} \wedge \eta_{ij} = 0$ (antisymmetry)
\item $\eta_{ij} \wedge \eta_{jk} + \eta_{jk} \wedge \eta_{ki} + \eta_{ki} \wedge \eta_{ij} = 0$ 
(Arnold relation)
\end{enumerate}

These are exactly the relations defining OS$(\mathcal{A})$.

The isomorphism $e_{ij} \mapsto \eta_{ij}$ is proven by induction on $n$ using the 
long exact sequence for the pair $(C_n(\mathbb{C}), C_n(\mathbb{C}) \setminus H_{12})$.
\end{proof}

\emph{Verification of Arnold relation -- explicit computation.}

For points $z_1, z_2, z_3 \in \mathbb{C}$, we verify:
$$\eta_{12} \wedge \eta_{23} + \eta_{23} \wedge \eta_{31} + \eta_{31} \wedge \eta_{12} = 0$$

All three numerator 2-forms $(dz_i - dz_j) \wedge (dz_j - dz_k)$ coincide,
so the sum reduces to a common numerator times the partial fraction sum:
$$\frac{1}{(z_1-z_2)(z_2-z_3)} + \frac{1}{(z_2-z_3)(z_3-z_1)} + \frac{1}{(z_3-z_1)(z_1-z_2)} = 0$$

This identity holds because multiplying through by $(z_1-z_2)(z_2-z_3)(z_3-z_1)$ gives
the telescoping sum $(z_3-z_1) + (z_1-z_2) + (z_2-z_3) = 0$.  QED for Proof 1.

\emph{Proof~2: Geometric (\`a la Kontsevich).}

Kontsevich's proof uses configuration space compactification.

\begin{lemma}[Residue exact sequence \cite{Har77}; \ClaimStatusProvedElsewhere]\label{lem:residue-exact-arnold}
For the compactified configuration space $\overline{C}_n(X)$ with boundary divisor $D$:
$$0 \to \Omega^1(C_n(X)) \to \Omega^1_{\log}(\overline{C}_n(X)) \xrightarrow{\text{Res}} 
\bigoplus_{i<j} \mathcal{O}_{D_{ij}} \to 0$$
is exact.
\end{lemma}

\begin{proof}[Proof of Lemma]
This follows from the exact sequence for logarithmic forms:
$$0 \to \Omega^1_X \to \Omega^1_X(\log D) \xrightarrow{\text{Res}} \bigoplus_i \mathcal{O}_{D_i} 
\to 0$$

For configuration spaces, $D = \bigcup_{i<j} D_{ij}$ has normal crossings, so the 
sequence remains exact.
\end{proof}

The Arnold relations are precisely the kernel of the residue map, matching Proof 1. QED for Proof 2.

\emph{Proof~3: Homotopy-Theoretic.}

View $C_n(\mathbb{C})$ as a $K(\pi, 1)$ space for the pure braid group $P_n$.

\begin{lemma}[Braid group cohomology \cite{Arn69, Coh76}; \ClaimStatusProvedElsewhere]\label{lem:braid-cohomology-arnold}
The cohomology ring of the configuration space $C_n(\mathbb{C})$ is the quotient of the exterior algebra on degree-$1$ generators $\omega_{ij}$ ($1 \leq i < j \leq n$) by the Arnold relations:
$$H^*(C_n(\mathbb{C}), \mathbb{Z}) \cong \bigwedge[\omega_{ij}] \big/ \bigl(\omega_{ij}\omega_{jk} + \omega_{jk}\omega_{ki} + \omega_{ki}\omega_{ij}\bigr)$$
In particular, $H^1(P_n, \mathbb{Z}) \cong \mathbb{Z}^{\binom{n}{2}}$ (free abelian, generated by the classes $\omega_{ij}$), and the Arnold relations generate the ideal of relations in degree~$2$ and above.
\end{lemma}

\begin{proof}[Proof of Lemma - Sketch]
The pure braid group $P_n = \pi_1(C_n(\mathbb{C}))$ has generators $A_{ij}$ (loops around the diagonal $D_{ij}$).  The abelianization is free abelian:
$$P_n^{ab} = H_1(P_n, \mathbb{Z}) \cong \mathbb{Z}^{\binom{n}{2}}$$
(the braid relations become trivial upon abelianization).  By universal coefficients, $H^1(P_n, \mathbb{Z}) = \mathrm{Hom}(\mathbb{Z}^{\binom{n}{2}}, \mathbb{Z}) \cong \mathbb{Z}^{\binom{n}{2}}$.  The Arnold relations constrain the \emph{cup product} structure in $H^2$ and higher, not $H^1$ itself.
\end{proof}

All three approaches (combinatorial, geometric, homotopy-theoretic) yield the same
result: the Arnold relations completely characterize the cohomology of configuration
spaces.

\end{proof}

\begin{remark}[Systematic verification]\label{rem:nine-term-detailed-arnold}
The Arnold relations can be verified systematically for each triple $\{i,j,k\} \subset \{1,\ldots,n\}$:
\begin{itemize}
\item $n=3$: $\binom{3}{3} = 1$ relation (verified above)
\item $n=4$: $\binom{4}{3} = 4$ relations (all follow from the $n=3$ case by restriction)
\item $n=5$: $\binom{5}{3} = 10$ relations (similarly)
\end{itemize}
\end{remark}

\begin{corollary}[Bar differential squares to zero; \ClaimStatusProvedHere]\label{cor:bar-d-squared-zero-arnold}
The Arnold relations ensure $d^2 = 0$ for the geometric bar differential:
$$d^2 = \sum_{\text{cycles}} [\operatorname{Res}_{D_i}, \operatorname{Res}_{D_j}] = 0$$
because the residue commutators sum to zero by Arnold relations.
\begin{proof}
This is the content of Theorem~\ref{thm:bar-nilpotency-complete}, where the nine pairwise compositions of the three bar differential components are shown to cancel.  The Arnold relations enter in the $d_{\mathrm{res}} \circ d_{\mathrm{res}}$ component.
\end{proof}
\end{corollary}

\section{Arnold relations in bar differential nilpotency}
\label{sec:arnold-in-bar-nilpotency}

\begin{theorem}[Arnold relations $\Leftrightarrow$ $d_{\text{residue}}^2 = 0$; \ClaimStatusProvedHere]\label{thm:arnold-iff-nilpotent}
The following are equivalent:
\begin{enumerate}
\item The Arnold relations hold for all triples $(i,j,k)$:
$$\eta_{ij} \wedge \eta_{jk} + \eta_{jk} \wedge \eta_{ki} + \eta_{ki} \wedge \eta_{ij} = 0$$

\item The residue differential is nilpotent:
$$d_{\text{residue}}^2 = 0$$

\item The composition of residues satisfies:
$$\text{Res}_{D_{ij}} \circ \text{Res}_{D_{jk}} + \text{Res}_{D_{jk}} \circ \text{Res}_{D_{ki}}
+ \text{Res}_{D_{ki}} \circ \text{Res}_{D_{ij}} = 0$$
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{(1) $\Rightarrow$ (3).}

Start with the Arnold relation for forms:
$$\eta_{ij} \wedge \eta_{jk} + \eta_{jk} \wedge \eta_{ki} + \eta_{ki} \wedge \eta_{ij} = 0$$

Apply the residue operator $\text{Res}_{D_{ij}}$ to the whole relation. Since $\eta_{ij} = d\log(z_i - z_j)$ has a simple pole along $D_{ij}$ while $\eta_{jk}$ is smooth there, the residue of their wedge product is:
$$\text{Res}_{D_{ij}}[\eta_{ij} \wedge \eta_{jk}] = \text{Res}_{D_{ij}}[\eta_{ij}] \cdot \eta_{jk}\big|_{D_{ij}} = 1 \cdot \eta_{jk}\big|_{D_{ij}} = \eta_{jk}\big|_{D_{ij}}$$
(This is the standard formula: for a form $\omega = f\,d\log(t) \wedge \alpha$ with $\alpha$ smooth across $\{t=0\}$, one has $\text{Res}_{t=0}[\omega] = f|_{t=0} \cdot \alpha|_{t=0}$.)

Similarly for the other terms. Applying $\text{Res}_{D_{ij}}$ to the Arnold relation yields:
$$\eta_{jk}|_{D_{ij}} + \text{Res}_{D_{ij}}[\eta_{jk} \wedge \eta_{ki}] + 
\text{Res}_{D_{ij}}[\eta_{ki} \wedge \eta_{ij}] = 0$$

This is precisely the composition formula we need.

\emph{(3) $\Rightarrow$ (2).}

The square of $d_{\text{residue}}$ expands as:
$$d_{\text{residue}}^2 = \sum_{i<j} \sum_{k<\ell} \text{Res}_{D_{ij}} \circ \text{Res}_{D_{k\ell}}$$

Terms with disjoint pairs $(i,j)$ and $(k,\ell)$ \emph{anticommute} (the residue operators have cohomological degree~$1$, and they act on independent factors), so $\text{Res}_{D_{ij}} \circ \text{Res}_{D_{k\ell}} + \text{Res}_{D_{k\ell}} \circ \text{Res}_{D_{ij}} = 0$.

Terms with one shared index give triples, which cancel by (3).

Terms with two shared indices are diagonal ($\text{Res}_D^2 = 0$).

Therefore $d_{\text{residue}}^2 = 0$.

\emph{(2) $\Rightarrow$ (1).}

Assume $d_{\text{residue}}^2 = 0$. Apply this to a specific test form
$\omega = \eta_{ij} \wedge \eta_{jk} \wedge \alpha$ where $\alpha$ is any $(n-2)$-form
with no poles.

Computing:
$$d_{\text{residue}}(\omega) = \text{Res}_{D_{ij}}[\eta_{ij} \wedge \eta_{jk} \wedge \alpha]
+ \text{Res}_{D_{jk}}[\eta_{ij} \wedge \eta_{jk} \wedge \alpha] + \cdots$$

Applying $d_{\text{residue}}$ again to each term and collecting: the coefficient of the
$0$-form $\alpha|_{\text{triple collision}}$ in $d_{\text{residue}}^2(\omega)$ equals
$$\text{Res}_{D_{ij}} \text{Res}_{D_{jk}}[\eta_{ij} \wedge \eta_{jk}]
+ \text{Res}_{D_{jk}} \text{Res}_{D_{ki}}[\eta_{jk} \wedge \eta_{ki}]
+ \text{Res}_{D_{ki}} \text{Res}_{D_{ij}}[\eta_{ki} \wedge \eta_{ij}].$$
Since $d_{\text{residue}}^2 = 0$ (from $\partial^2 = 0$ on the compactification), this
sum must vanish.  Taking into account the Koszul signs from the ordering of iterated
residues, this yields (cf.\ Computation~\ref{comp:explicit-residues}):
$$\text{Res}_{D_{ij}} \circ \text{Res}_{D_{jk}} + \text{Res}_{D_{jk}} \circ
\text{Res}_{D_{ki}} + \text{Res}_{D_{ki}} \circ \text{Res}_{D_{ij}} = 0$$
which is precisely the Arnold relation on the 2-forms
$\eta_{ij} \wedge \eta_{jk}$, $\eta_{jk} \wedge \eta_{ki}$,
$\eta_{ki} \wedge \eta_{ij}$.
\end{proof}

\subsection{Explicit residue calculations}

\begin{computation}[Residues of logarithmic forms]\label{comp:explicit-residues}
Consider the 2-form:
$$\omega = \eta_{12} \wedge \eta_{23} = d\log(z_1 - z_2) \wedge d\log(z_2 - z_3)$$

\emph{Residue along $D_{12}$ (where $z_1 \to z_2$).}

Near $D_{12}$, use coordinates:
\begin{align*}
u &= z_2 \quad \text{(center)}\\
\epsilon &= z_1 - z_2 \quad \text{(separation)}\\
v &= z_3 \quad \text{(other point)}
\end{align*}

Then:
$$\eta_{12} = d\log(\epsilon) = \frac{d\epsilon}{\epsilon}$$
$$\eta_{23} = d\log(u - v)$$

The 2-form becomes:
$$\omega = \frac{d\epsilon}{\epsilon} \wedge d\log(u - v) = \frac{d\epsilon}{\epsilon} 
\wedge \frac{du - dv}{u - v}$$

Taking the residue (integrating over $\epsilon$):
$$\text{Res}_{D_{12}}[\omega] = \oint_{\epsilon=0} \frac{d\epsilon}{\epsilon} \wedge 
\frac{du - dv}{u - v} = \frac{du - dv}{u - v} = d\log(z_2 - z_3)|_{z_1 = z_2} = \eta_{23}|_{D_{12}}$$

\emph{Double residue along $D_{12}$ then $D_{23}$.}

Now take residue of $\eta_{23}|_{D_{12}}$ along $D_{23}$ (where $z_2 \to z_3$):
$$\text{Res}_{D_{23}}[\eta_{23}|_{D_{12}}] = \text{Res}_{D_{23}}\left[\frac{du}{u-v}\right]
= 1$$

\emph{Arnold cancellation.}

The Arnold relation with signs is:
$$\eta_{12} \wedge \eta_{23} + \eta_{23} \wedge \eta_{31} + \eta_{31} \wedge \eta_{12} = 0$$
where $\eta_{31} = \eta_{13}$ as differential forms (since $d\log(-1) = 0$; the Orlik--Solomon generators $e_{ij}$ satisfy $e_{ji} = e_{ij}$, indexed by unordered pairs corresponding to hyperplanes $H_{ij} = H_{ji}$).

Taking iterated residues along the corresponding divisors, the Koszul signs from the order of residue operations yield:
$$\text{Res}_{D_{12}} \circ \text{Res}_{D_{23}} + \text{Res}_{D_{23}} \circ \text{Res}_{D_{31}}
+ \text{Res}_{D_{31}} \circ \text{Res}_{D_{12}} = 1 - 1 + 0 = 0 \quad \checkmark$$
\end{computation}

\subsection{\texorpdfstring{Arnold relations for $n=4$: the four triple relations}{Arnold relations for n=4: the four triple relations}}

\begin{computation}[All Arnold relations for four points]\label{comp:arnold-n4}
For $n=4$, we have $\binom{4}{3} = 4$ triples, each giving an Arnold relation:

\emph{Triple $(1,2,3)$.}
$$\eta_{12} \wedge \eta_{23} + \eta_{23} \wedge \eta_{31} + \eta_{31} \wedge \eta_{12} = 0$$

\emph{Triple $(1,2,4)$.}
$$\eta_{12} \wedge \eta_{24} + \eta_{24} \wedge \eta_{41} + \eta_{41} \wedge \eta_{12} = 0$$

\emph{Triple $(1,3,4)$.}
$$\eta_{13} \wedge \eta_{34} + \eta_{34} \wedge \eta_{41} + \eta_{41} \wedge \eta_{13} = 0$$

\emph{Triple $(2,3,4)$.}
$$\eta_{23} \wedge \eta_{34} + \eta_{34} \wedge \eta_{42} + \eta_{42} \wedge \eta_{23} = 0$$

Each relation ensures that $d_{\text{residue}}^2 = 0$ for the corresponding triple collision.

\emph{Consistency check.} These four Arnold relations are \emph{linearly independent}
in $\bigwedge^2 H^1(\overline{C}_4(\mathbb{C}))$. They span the $\binom{4}{3} = 4$-dimensional
Arnold ideal, reducing $\dim \bigwedge^2 H^1 = \binom{6}{2} = 15$ to $\dim H^2 = 15 - 4 = 11$.
\end{computation}

\subsection{\texorpdfstring{General pattern for $n$ points}{General pattern for n points}}

\begin{theorem}[Arnold relations for $n$ points \cite{Arn69, OS80}; \ClaimStatusProvedElsewhere]\label{thm:arnold-general-n}
For $n$ points, there are $\binom{n}{3}$ Arnold relations, one for each triple $(i,j,k)$.

These relations are:
\begin{itemize}
\item Linearly independent in $\bigwedge^2 H^1(\overline{C}_n(X))$ (they are zero in $H^2$, constituting the defining relations of the Orlik--Solomon algebra in degree~2)
\item Sufficient to ensure $d_{\text{residue}}^2 = 0$
\item Equivalent to the vanishing of triple compositions of residues
\end{itemize}

The degree-1 cohomology has dimension
$$\dim H^1(C_n(\mathbb{C})) = \binom{n}{2}$$
(one generator $\eta_{ij}$ for each pair $i < j$).
The degree-2 cohomology is the quotient of $\bigwedge^2 H^1$ by the Arnold ideal:
$$\dim H^2(C_n(\mathbb{C})) = \binom{\binom{n}{2}}{2} - \binom{n}{3}$$
More precisely, the Poincar\'e polynomial is $P_t(C_n(\mathbb{C})) = \prod_{k=1}^{n-1}(1+kt)$,
so $\dim H^k = e_k(1, 2, \ldots, n-1)$ where $e_k$ denotes the $k$-th elementary symmetric polynomial.
\end{theorem}

\begin{proof}
By the Orlik--Solomon theorem (\cite{OS80}~Theorem~3.4), the cohomology
algebra $H^*(C_n(\mathbb{C}); \mathbb{Q})$ is the quotient of the exterior
algebra on generators $\eta_{ij}$ ($1 \le i < j \le n$) by the Arnold
ideal.  The dimension formula
$\dim H^k = e_k(1, 2, \ldots, n-1)$ follows from the Poincar\'e
polynomial $P_t(C_n(\mathbb{C})) = \prod_{k=1}^{n-1}(1+kt)$, which
is proved in \cite{OS80}~Corollary~3.5.
\end{proof}

\subsection{Physical interpretation: operator product associativity}

\begin{remark}[OPE associativity]\label{persp:ope-arnold}
In conformal field theory, the Arnold relations encode associativity of the
operator product expansion: the three ways of computing a three-point function by
successive OPEs must agree up to monodromy.  On logarithmic forms, this reads
$(\phi_i \times \phi_j) \times \phi_k = \phi_i \times (\phi_j \times \phi_k)$,
and the Arnold relations ensure that the correlation function
$\langle \phi_i(z_i) \phi_j(z_j) \phi_k(z_k) \rangle$ is single-valued
after accounting for all branch cuts.
\end{remark}

\section{Global boundary relations on configuration spaces}
\label{sec:global-boundary-relations}

The local Arnold relations of the preceding sections control the cancellation of
codimension-$2$ contributions one triple at a time.  The
\emph{global} version follows: a systematic accounting of all codimension-$2$ strata
of $\overline{C}_n(X)$ and their signed contributions.  This provides the
rigorous foundation for $d_{\mathrm{res}}^2 = 0$ as a consequence of the
boundary-of-boundary identity $\partial^2 = 0$ on the stratified space
$\overline{C}_n(X)$.

\subsection{Stratification of the Fulton--MacPherson compactification}

\begin{theorem}[Configuration space boundary relations; \ClaimStatusProvedHere]
\label{thm:config-boundary-relations}
Let $X$ be a smooth algebraic curve and
$\overline{C}_n(X)$ the Fulton--MacPherson compactification of the
configuration space $C_n(X)$.  The boundary $\partial\overline{C}_n(X)$ is a
normal crossings divisor whose strata are indexed by \emph{nested partitions}
(equivalently, rooted trees with $n$ labeled leaves).  The following hold:

\begin{enumerate}[label=(\roman*)]
\item \emph{Codimension-$1$ strata.}  For each subset
  $S \subseteq \{1, \ldots, n\}$ with $|S| \geq 2$, there is a boundary
  divisor $D_S$ parametrizing configurations where the points labeled by
  $S$ have collided.  As a variety,
  \[
    D_S \cong \overline{C}_{|S|}(T_{x}X) \times \overline{C}_{n - |S| + 1}(X)
  \]
  where $T_x X \cong \mathbb{A}^1$ is the tangent space at the collision point.

\item \emph{Codimension-$2$ strata.}  Each codimension-$2$ stratum arises
  from one of two configurations:
  \begin{enumerate}[label=(\alph*)]
  \item \emph{Nested collisions.} $S' \subset S$ with $|S'| \geq 2$,
    $|S| \geq 3$.  The stratum $D_{S'} \cap D_S$ parametrizes a two-stage
    collision: the points in $S'$ collide first within the group $S$, then $S$
    collides as a whole.
  \item \emph{Disjoint collisions.} $S_1 \cap S_2 = \emptyset$ with
    $|S_1|, |S_2| \geq 2$.  The stratum $D_{S_1} \cap D_{S_2}$ parametrizes
    two independent collision events.
  \end{enumerate}

\item \emph{The boundary-of-boundary identity.}
  Let $\partial = \sum_{S} \epsilon_S\, \mathrm{Res}_{D_S}$ denote the signed
  boundary operator, where $\epsilon_S = (-1)^{\sigma(S)}$ is the orientation
  sign from the normal crossings structure (determined by the ordering of
  boundary divisors). Then:
  \begin{equation}\label{eq:boundary-squared-zero}
    \partial^2 = 0.
  \end{equation}
  Concretely: for every codimension-$2$ stratum $Z$, the two codimension-$1$
  divisors $D_1, D_2$ containing $Z$ contribute opposite signs,
  \[
    \epsilon_{D_1 \to Z} + \epsilon_{D_2 \to Z} = 0,
  \]
  and the contributions cancel in pairs.
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{Step~1: Normal crossings structure.}
The Fulton--MacPherson compactification $\overline{C}_n(X)$ is a smooth
variety with boundary a normal crossings divisor
\cite{FM94}.  The irreducible components of the boundary are
the divisors $D_S$ for $|S| \geq 2$, and distinct components $D_S$,
$D_{S'}$ meet transversally if and only if either $S \subset S'$,
$S' \subset S$, or $S \cap S' = \emptyset$.  When $S$ and $S'$ are
``crossing'' ($S \cap S' \neq \emptyset$, neither contained in the other),
$D_S \cap D_{S'} = \emptyset$.  This is the key property that distinguishes
the FM compactification from the naive closure.

\emph{Step~2: Orientation conventions.}
We fix a total ordering on the set of boundary divisors (e.g., lexicographic
on the subsets $S$).  The normal crossings condition provides, for each
codimension-$k$ stratum $D_{S_1} \cap \cdots \cap D_{S_k}$, an orientation
determined by the outward normal vectors to $D_{S_1}, \ldots, D_{S_k}$ in
that order.  Reordering introduces a sign $(-1)^{\mathrm{sgn}(\sigma)}$ from
the permutation.

\emph{Step~3: Codimension-$2$ cancellation.}
We verify $\partial^2 = 0$ by showing that each codimension-$2$ stratum
contributes zero.  There are two cases.

\emph{Case (a): Nested collisions ($S' \subset S$).}
The stratum $D_{S'} \cap D_S$ is reached via two paths:
\begin{itemize}
\item First restrict to $D_S$ (the group $S$ collides), then restrict further
  to $D_{S'}$ within $D_S$ (the subgroup $S'$ collides within the already-
  collapsed cluster).
\item First restrict to $D_{S'}$ (the subgroup $S'$ collides), then restrict
  to $D_S$ (the remaining points of $S$ join the collapsed subgroup).
\end{itemize}
In the normal crossings chart with coordinates $(t_S, t_{S'}, \ldots)$,
the first path gives $\mathrm{Res}_{t_{S'}=0} \circ \mathrm{Res}_{t_S=0}$
and the second gives $\mathrm{Res}_{t_S=0} \circ \mathrm{Res}_{t_{S'}=0}$.
The two orderings of the normal directions differ by a transposition, hence
contribute with opposite signs.  When combined with the local Arnold relation
(Theorem~\ref{thm:arnold-iff-nilpotent}), which controls the algebraic
content of each residue, the contributions cancel:
\[
  \mathrm{Res}_{D_S} \circ \mathrm{Res}_{D_{S'}} +
  \mathrm{Res}_{D_{S'}} \circ \mathrm{Res}_{D_S} = 0
  \quad \text{on } D_{S'} \cap D_S.
\]

\emph{Case (b): Disjoint collisions ($S_1 \cap S_2 = \emptyset$).}
Since $D_{S_1}$ and $D_{S_2}$ involve disjoint sets of points,
the residues along $D_{S_1}$ and $D_{S_2}$ act on independent tensor
factors of $\mathcal{A}^{\boxtimes n}$.  The two orderings give:
\[
  \mathrm{Res}_{D_{S_1}} \circ \mathrm{Res}_{D_{S_2}} =
  (-1)^{|S_1| \cdot |S_2|}\, \mathrm{Res}_{D_{S_2}} \circ \mathrm{Res}_{D_{S_1}}
\]
by the Koszul sign rule (the residue maps have degrees determined by the
codimension).  In the boundary operator $\partial$, the orientation signs
$\epsilon_{S_1}$ and $\epsilon_{S_2}$ are chosen so that this Koszul sign
precisely compensates the reordering, giving:
\[
  \epsilon_{S_1}\epsilon_{S_2}\, \mathrm{Res}_{D_{S_1}} \circ
  \mathrm{Res}_{D_{S_2}} +
  \epsilon_{S_2}\epsilon_{S_1}\, \mathrm{Res}_{D_{S_2}} \circ
  \mathrm{Res}_{D_{S_1}} = 0.
\]

\emph{Case (c): Non-transversal strata.}
When $S \cap S' \neq \emptyset$ and neither $S \subset S'$ nor
$S' \subset S$, we have $D_S \cap D_{S'} = \emptyset$ by the FM
compactification structure.  These pairs contribute nothing to
$\partial^2$.

\emph{Step~4: Conclusion.}
Every codimension-$2$ stratum falls into case (a), (b), or (c), and in
each case the contribution to $\partial^2$ vanishes.  Therefore
$\partial^2 = 0$, completing the proof.
\end{proof}

\begin{corollary}[$d_{\mathrm{res}}^2 = 0$ globally; \ClaimStatusProvedHere]\label{cor:dres-squared-global}
The residue differential $d_{\mathrm{res}} = \sum_{|S|=2} \mathrm{Res}_{D_S}$
on the bar complex $\bar{B}^{\mathrm{ch}}(\mathcal{A})$ satisfies
$d_{\mathrm{res}}^2 = 0$ as a global identity on $\overline{C}_n(X)$, for
any smooth curve $X$ and any genus.
\end{corollary}

\begin{proof}
The residue differential is a summand of the boundary operator $\partial$
(restricting to codimension-$1$ strata $D_S$ with $|S| = 2$, which
correspond to two-point collisions).  The identity $\partial^2 = 0$
(Theorem~\ref{thm:config-boundary-relations}) implies that the
$d_{\mathrm{res}}^2$ terms cancel against the nested and disjoint
codimension-$2$ strata exactly as described above.  The nine-term
cancellation of Theorem~\ref{thm:bar-nilpotency-complete} is recovered
as the special case $n = 3$, $S = \{i,j\}$, $S' = \{j,k\}$,
$S'' = \{k,i\}$.
\end{proof}

\begin{remark}[Relationship to the nine-term verification]
\label{rem:nine-term-global}
The nine-term cancellation in the proof of
Theorem~\ref{thm:bar-nilpotency-complete} is the local manifestation of
the global identity $\partial^2 = 0$.  For $n = 3$ points, the
codimension-$2$ strata are triple collision loci $D_{ijk}$, and the three
codimension-$1$ divisors $D_{ij}$, $D_{jk}$, $D_{ki}$ meeting at $D_{ijk}$
produce exactly the nine terms (three squared terms and six cross terms) that
cancel by the Arnold relation.  For $n > 3$, the global theorem subsumes all
such local cancellations simultaneously, providing a uniform proof that does
not require case-by-case checking.
\end{remark}
