\chapter{Homotopy transfer}
\label{app:homotopy-transfer}

\index{homotopy transfer theorem|textbf}

The homotopy
transfer theorem is the mechanism by which classical Koszul duality
acquires quantum corrections.  Passing from a chain complex to its
cohomology via a deformation retract necessarily introduces higher
operations $m_k$ for $k \geq 3$; these are precisely the genus-$0$
quantum corrections of the bar complex
(Theorem~\ref{thm:bar-ainfty-complete}).

\section{Historical origins}
\label{sec:ainfty-historical}

$A_\infty$ algebras were introduced by Stasheff~\cite{Sta63} to encode the homotopy-coherent associativity of loop space multiplication.  The operadic viewpoint identifies them as algebras over the homology of the little intervals operad.  For the role of bar-cobar duality in the chiral setting, see Chapter~\ref{chap:higher-genus}.

\section{The homotopy transfer theorem}
\label{sec:htt}

\begin{theorem}[Homotopy transfer theorem \cite{LV12, Kadeishvili80}; \ClaimStatusProvedElsewhere]
\label{thm:htt}
Let $(V, d_V)$ and $(W, d_W)$ be chain complexes with:
\begin{enumerate}[label=(\roman*)]
\item A chain map $p: V \to W$ (projection).
\item A chain map $\iota: W \to V$ (inclusion).
\item A chain homotopy $h: V \to V[-1]$ satisfying $\id_V - \iota p = d_V h + h d_V$.
\item The \emph{side conditions} (compatibility of $h$ with $p, \iota$): $p \iota = \id_W$, $h \iota = 0$, $p h = 0$, $h^2 = 0$.
\end{enumerate}
Such data is called a \emph{strong deformation retract} (SDR).

If $V$ carries a $\cP_\infty$-algebra structure (for $\cP$ a Koszul operad), 
then $W$ inherits a $\cP_\infty$-algebra structure such that $p$ and $\iota$ 
extend to $\infty$-quasi-isomorphisms of $\cP_\infty$-algebras.
\end{theorem}

\begin{proof}
The transferred structure on $W$ is constructed via the \emph{tensor trick}
(\cite{LV12}~\S10.3).
For an $\Ainf$-algebra structure on $V$ with operations $m_n: V^{\otimes n} \to V$,
define $T_n: V^{\otimes n} \to V$ recursively by:
\[
T_n = m_n + \sum_{\substack{r+s+t = n \\ s < n}} m_{r+1+t}(\id^{\otimes r} \otimes h\, T_s \otimes \id^{\otimes t}).
\]
This converges since $T_s$ with $s < n$ is already defined and $h$ absorbs
the degree.  The transferred operations are $\tilde{m}_n := p \circ T_n \circ
\iota^{\otimes n}$.  Explicitly:
\begin{align*}
\tilde{m}_1 &= p \, d_V \, \iota = d_W, \\
\tilde{m}_2 &= p \, m_2 \, \iota^{\otimes 2}, \\
\tilde{m}_3 &= p \, m_3 \, \iota^{\otimes 3} + p \, m_2 (h m_2 \otimes \id) \, \iota^{\otimes 3}
              + p \, m_2 (\id \otimes h m_2) \, \iota^{\otimes 3}.
\end{align*}

\emph{Verification of $\Ainf$-relations.}
The $\Ainf$-relation at arity~$n$ for $\{\tilde{m}_k\}$ requires:
\begin{equation}\label{eq:ainfty-relation}
\sum_{\substack{r+s+t=n}} (-1)^{rs+t}\, \tilde{m}_{r+1+t}(\id^{\otimes r} \otimes \tilde{m}_s \otimes \id^{\otimes t}) = 0.
\end{equation}
Substituting $\tilde{m}_k = p \circ T_k \circ \iota^{\otimes k}$ and using the
SDR identity $\iota p = \id_V - (d_V h + h d_V)$, each internal composition
$\iota p$ between an outer $T_{r+1+t}$ and inner $T_s$ decomposes as:
\begin{itemize}
\item a term $\id_V$, which together with the $\Ainf$-relations for $\{m_k\}$
  gives cancelling contributions;
\item a term $-d_V h - h d_V$, which telescopes by the side conditions
  $h^2 = 0$, $p h = 0$, $h \iota = 0$.
\end{itemize}
Concretely, the terms involving~$\id_V$ reassemble into $p \circ (\text{$\Ainf$
relation for } \{m_k\}) \circ \iota^{\otimes n} = 0$, while the terms
involving $d_V h + h d_V$ cancel pairwise: $h^2 = 0$ prevents double
homotopy insertions, and $ph = h\iota = 0$ kill boundary terms.  Thus
$\{\tilde{m}_n\}$ satisfies the $\Ainf$-relations.

The maps $p$ and $\iota$ extend to $\infty$-morphisms $\tilde{p}_n$ and
$\tilde{\iota}_n$ by the same tree formula with $p$ (resp.~$\iota$) at
the root (resp.~leaves), and the composite $\tilde{p} \circ \tilde{\iota}
= \id_W$ by $p\iota = \id_W$, while $\tilde{\iota} \circ \tilde{p}
\sim \id_V$ via the homotopy built from~$h$.
\end{proof}

\begin{definition}[Strong deformation retract data]
\label{def:sdr}
\index{strong deformation retract|textbf}
\index{SDR|see{strong deformation retract}}
A \emph{strong deformation retract} (SDR) from $V$ to $W$ is a tuple 
$(V, W, p, \iota, h)$ satisfying the conditions of Theorem~\ref{thm:htt}. 
We denote this by:
\[
\begin{tikzcd}[column sep=large]
(V, d_V) \arrow[r, shift left, "p"] & (W, d_W) \arrow[l, shift left, "\iota"]
\end{tikzcd}
\quad h: V \to V[-1], \quad \id - \iota p = dh + hd.
\]
The diagram commutes up to homotopy $h$.
\end{definition}

\begin{lemma}[Existence of SDR; \ClaimStatusProvedHere]
\label{lem:sdr-existence}
If $V \xrightarrow{\sim} W$ is a quasi-isomorphism of chain complexes over a 
field $k$, then:
\begin{enumerate}[label=(\roman*)]
\item There exists an SDR from $V$ to $H_*(V) = H_*(W)$.
\item If $V$ and $W$ are both semi-free (or projective as graded modules), 
there exists an SDR between them.
\end{enumerate}
\end{lemma}

\begin{proof}
(i) Choose a splitting $V = H_*(V) \oplus B \oplus C$ where $d: C \xrightarrow{\sim} B$ 
is an isomorphism (decomposition into homology, boundaries, and ``extra'' cycles 
that become boundaries). Define:
\[
\iota: H_*(V) \hookrightarrow V, \quad p: V \twoheadrightarrow H_*(V), \quad 
h: V \to V[-1]
\]
where $h$ sends $B$ to $C$ via the inverse of $d|_C$, and is zero on $H_*(V) \oplus C$.
We verify each condition.
\emph{(a)} $p \circ \iota = \mathrm{id}_{H_*(V)}$: the inclusion $\iota$ maps $H_*(V)$ into $V$ as a direct summand, and $p$ projects back.
\emph{(b)} $\iota \circ p - \mathrm{id}_V = d \circ h + h \circ d$: on $H_*(V)$, $dh + hd = 0$ (both $d$ and $h$ vanish) and $\iota p - \mathrm{id} = 0$; on $B$, $dh(b) = d(0) = 0$ (since $h|_B = 0$) while $hd(b) = h(d(b)) \in C$ and $\iota p(b) - b = 0 - b = -b$, so $hd(b) = -b$ as needed since $h$ inverts $d|_C: C \xrightarrow{\sim} B$; on $C$, $dh(c) = d(d^{-1}c) = c$ while $hd(c) = 0$ (since $d(C) \subset H_* \oplus B$ and $h|_{H_*} = 0$, $h|_B = 0$), and $\iota p(c) - c = 0 - c$, confirming $dh + hd = \iota p - \mathrm{id}$.
\emph{(c)} $h^2 = 0$: $h$ maps $B \to C$ and $h|_C = 0$; $h \circ \iota = 0$: $\iota$ lands in $H_*(V)$ and $h|_{H_*} = 0$; $p \circ h = 0$: $h$ maps into $C$ and $p|_C = 0$.

(ii) The Comparison Theorem for projective resolutions provides the maps; 
standard homological algebra constructs the homotopy.
\end{proof}


\section{Explicit formulas for transferred structures}
\label{sec:transfer-formulas}

\begin{construction}[Transferred \texorpdfstring{$\Ainf$}{A-infinity}-structure]
\label{constr:transfer-ainf}
Let $(A, \{m_n\})$ be an $\Ainf$-algebra and $(A, H, p, \iota, h)$ an SDR to 
the homology $H = H_*(A)$. The transferred $\Ainf$-structure $\{\tilde{m}_n\}$ 
on $H$ is given by:

\emph{$\tilde{m}_1 = 0$.} The differential on homology vanishes.

\emph{$\tilde{m}_2$.} The induced product:
\[
\tilde{m}_2(a, b) = p \, m_2(\iota(a), \iota(b)).
\]

\emph{$\tilde{m}_3$.} The \emph{Massey product} or $\Ainf$-associator:
\begin{align*}
\tilde{m}_3(a, b, c) &= p \, m_3(\iota(a), \iota(b), \iota(c)) \\
&\quad + p \, m_2(h \, m_2(\iota(a), \iota(b)), \iota(c)) \\
&\quad + (-1)^{|a|}\, p \, m_2(\iota(a), h \, m_2(\iota(b), \iota(c))).
\end{align*}
The sign $(-1)^{|a|}$ arises from commuting the homotopy $h$ (degree $-1$) past $\iota(a)$ (degree $|a|$) via the Koszul sign rule.

\emph{$\tilde{m}_n$} (general). Sum over planar rooted trees $T$ with $n$ 
leaves:
\[
\tilde{m}_n = \sum_{T \in \mathrm{PRT}_n} p \, T(m_\bullet, h, \iota)
\]
where $T(m_\bullet, h, \iota)$ places $\iota$ at leaves, $m_k$ at vertices of 
valence $k$, and $h$ on internal edges.
\end{construction}

\begin{theorem}[Tree formula for transferred operations \cite{LV12}; \ClaimStatusProvedElsewhere]
\label{thm:tree-formula}
\index{tree formula!for transfer}
The transferred $n$-ary operation is:
\[
\tilde{m}_n = \sum_{T \in \mathrm{PRT}_n} \epsilon(T) \cdot p \circ \prod_{v \in V(T)} 
m_{|v|} \circ \prod_{e \in E_{\mathrm{int}}(T)} h \circ \iota^{\otimes n}
\]
where:
\begin{enumerate}[label=(\roman*)]
\item $\mathrm{PRT}_n$ is the set of planar rooted trees with $n$ leaves.
\item $V(T)$ is the set of internal vertices; $|v|$ is the valence of $v$.
\item $E_{\mathrm{int}}(T)$ is the set of internal edges.
\item $\epsilon(T) \in \{\pm 1\}$ is a sign determined by the Koszul rule.
\end{enumerate}
\end{theorem}

\begin{remark}[Tree-level only]\label{rem:tree-level}
Only tree-level graphs contribute to homotopy transfer: the sum in Theorem~\ref{thm:tree-formula} runs over trees (acyclic connected graphs), never over graphs with loops.  This is because the three side conditions $h^2 = 0$, $ph = 0$, and $h\iota = 0$ together ensure that any graph with a cycle vanishes: $h^2 = 0$ kills paths with consecutive homotopies, $ph = 0$ kills graphs where $h$ appears at the output edge, and $h\iota = 0$ kills graphs where $h$ appears at an input edge.  In the chiral algebra setting, this means the transferred operations are algebraic (finite sums), not analytic, and no renormalization issues arise.
\end{remark}

\begin{example}[Trees for \texorpdfstring{$\tilde{m}_4$}{m4-tilde}]
\label{ex:trees-m4}
The trees contributing to $\tilde{m}_4$ are:

\begin{enumerate}[label=(\arabic*)]
\item One vertex of valence 4:
\[
\begin{tikzpicture}[scale=0.5]
\node at (0,1) {$\bullet$};
\draw (0,1) -- (-1.5,0) (-0.5,0) (0.5,0) (1.5,0);
\draw (-1.5,0) node[below] {$1$};
\draw (-0.5,0) node[below] {$2$};
\draw (0.5,0) node[below] {$3$};
\draw (1.5,0) node[below] {$4$};
\end{tikzpicture}
\quad \leadsto \quad p \, m_4 \, \iota^{\otimes 4}
\]

\item One vertex of valence 3, one of valence 2 (5 configurations --- the binary vertex can be a child of the ternary in 3 ways, or the ternary can be a child of the binary in 2 ways):
\[
p \, m_3(h \, m_2 \otimes \id \otimes \id) \, \iota^{\otimes 4}, \quad
p \, m_2(h \, m_3 \otimes \id) \, \iota^{\otimes 4}, \quad \ldots
\]

\item Three vertices of valence 2 (5 configurations --- the Catalan number $C_3 = 5$ binary trees with 4 leaves).
\end{enumerate}
Total: $1 + 5 + 5 = 11$ planar rooted trees with 4 leaves and all internal vertices of arity $\geq 2$.
\end{example}

\begin{proposition}[Sign computation \cite{LV12}; \ClaimStatusProvedElsewhere]
\label{prop:transfer-signs}
The sign $\epsilon(T)$ in the tree formula is:
\[
\epsilon(T) = (-1)^{\sum_{e \in E_{\mathrm{int}}(T)} (|e|_{\mathrm{left}} + 1)}
\]
where $|e|_{\mathrm{left}}$ is the sum of degrees of inputs to the left of 
edge $e$ (counting the edge as separating left from right at its source vertex).
\end{proposition}


\section{Applications to minimal models}
\label{sec:minimal-models}

\begin{definition}[Minimal model]
\label{def:minimal-model}
\index{minimal model|textbf}
A \emph{minimal model} of a dg-algebra $A$ is an $\Ainf$-algebra $M$ with:
\begin{enumerate}[label=(\roman*)]
\item $M$ has zero differential: $m_1 = 0$.
\item There is an $\Ainf$-quasi-isomorphism $M \xrightarrow{\sim} A$.
\end{enumerate}
A minimal model is unique up to $\Ainf$-isomorphism.
\end{definition}

\begin{theorem}[Existence of minimal models \cite{Kadeishvili80}; \ClaimStatusProvedElsewhere]
\label{thm:minimal-model-existence}
Every $\Ainf$-algebra over a field admits a minimal model. Explicitly, if 
$A$ is an $\Ainf$-algebra:
\begin{enumerate}[label=(\roman*)]
\item Take $M = H_*(A)$ as a graded vector space.
\item Choose an SDR $(A, H_*(A), p, \iota, h)$.
\item Apply the homotopy transfer theorem to get $\{m_n\}_{n \geq 2}$ on $M$.
\item The resulting $(M, 0, m_2, m_3, \ldots)$ is the minimal model.
\end{enumerate}
\end{theorem}

\begin{corollary}[Formality \cite{DGMS75, Kon99}; \ClaimStatusProvedElsewhere]
\label{cor:formality}
An $\Ainf$-algebra $A$ is \emph{formal} if its minimal model has $m_n = 0$
for all $n \geq 3$, i.e., the minimal model is a graded associative
algebra with trivial differential and no higher operations.

Equivalently, $A$ is formal if $A \simeq H_*(A)$ as $\Ainf$-algebras, where 
$H_*(A)$ has the trivial $\Ainf$-structure from its product.
\end{corollary}

\begin{example}[Minimal model of de Rham complex]
\label{ex:minimal-derham}
For $X = S^1$ (circle), the de Rham complex is:
\[
(\Omega^*(S^1), d) = (k \xrightarrow{0} k, \quad 1 \mapsto 0, \quad d\theta \mapsto 0)
\]
with cohomology $H^* = k \oplus k[-1]$.

The minimal model has:
\begin{enumerate}[label=(\roman*)]
\item $M = k \cdot 1 \oplus k \cdot x$ with $|1| = 0$, $|x| = 1$.
\item $m_2(1, 1) = 1$, $m_2(1, x) = m_2(x, 1) = x$, $m_2(x, x) = 0$.
\item $m_n = 0$ for $n \geq 3$ (the circle is formal).
\end{enumerate}
\end{example}

\begin{application}[Minimal model for chiral algebras]
\label{app:minimal-chiral}
For a chiral algebra $\cA$, the homotopy transfer theorem provides:
\begin{enumerate}[label=(\roman*)]
\item A minimal $\Ainf$-chiral structure on $H^{\mathrm{ch}}_*(\cA)$.
\item The higher operations $\{m_n\}_{n \geq 3}$ encode obstructions to formality.
\item For Koszul chiral algebras, the minimal model simplifies (higher operations vanish).
\end{enumerate}

For the Heisenberg algebra $\cH$:
\[
H^{\mathrm{ch}}_*(\cH) = k
\]
is 1-dimensional (Koszul), and the minimal model is the ground field with 
trivial structure.
\end{application}

\begin{theorem}[Homotopy transfer for operadic algebras \cite{LV12}; \ClaimStatusProvedElsewhere]
\label{thm:htt-operadic}
Let $\cP$ be a Koszul operad and $(A, W, p, \iota, h)$ an SDR with $A$ a 
$\cP$-algebra. The transferred $\cP_\infty$-structure on $W$ satisfies:
\begin{enumerate}[label=(\roman*)]
\item The $\cP_\infty$-relations hold strictly (not just up to homotopy).
\item The maps $\iota$ and $p$ extend to $\infty$-morphisms.
\item If $A$ is already $\cP_\infty$ (not just $\cP$), the transfer still works.
\end{enumerate}
\end{theorem}


\section{Extended formulas and computational techniques}
\label{sec:extended-htt}

Extended formulas for homotopy transfer essential for
explicit computations in chiral Koszul duality follow.

\begin{construction}[Homotopy transfer for \texorpdfstring{$\Linf$}{L-infinity}-algebras]
\label{constr:htt-linf}
Let $(\fg, \{l_n\})$ be an $\Linf$-algebra and $(\fg, H, p, \iota, h)$ an SDR. 
The transferred $\Linf$-structure on $H$ has brackets:

\emph{$\tilde{l}_1 = 0$.} Differential vanishes on homology.

\emph{$\tilde{l}_2$.} The induced Lie bracket:
\[
\tilde{l}_2(a, b) = p \, l_2(\iota(a), \iota(b)).
\]

\emph{$\tilde{l}_3$.} The \emph{Massey--Lie bracket} or Jacobiator:
\begin{align*}
\tilde{l}_3(a, b, c) &= p \, l_3(\iota(a), \iota(b), \iota(c)) \\
&\quad + p \, l_2(h \, l_2(\iota(a), \iota(b)), \iota(c)) \\
&\quad + (-1)^{|b||c|} \, p \, l_2(h \, l_2(\iota(a), \iota(c)), \iota(b)) \\
&\quad + (-1)^{|a|(|b|+|c|)} \, p \, l_2(h \, l_2(\iota(b), \iota(c)), \iota(a)).
\end{align*}

The signs arise from the graded antisymmetry of the Lie bracket.
\end{construction}

\begin{proposition}[\texorpdfstring{$\Linf$}{L-infinity}-relations for transferred structure \cite{LV12}; \ClaimStatusProvedElsewhere]
\label{prop:linf-relations}
The transferred brackets $\{\tilde{l}_n\}$ satisfy the $\Linf$-relations:
\[
\sum_{i+j = n+1} \sum_{\sigma \in \mathrm{Sh}(i, n-i)} \epsilon(\sigma) 
\tilde{l}_j(\tilde{l}_i(x_{\sigma(1)}, \ldots, x_{\sigma(i)}), x_{\sigma(i+1)}, \ldots, x_{\sigma(n)}) = 0
\]
where $\mathrm{Sh}(i, n-i)$ denotes $(i, n-i)$-unshuffles and $\epsilon(\sigma)$ 
is the Koszul sign.
\end{proposition}

\begin{computation}[Explicit \texorpdfstring{$\tilde{l}_4$}{l4-tilde} formula]
\label{comp:l4-formula}
The 4-ary transferred bracket is:
\begin{align*}
\tilde{l}_4(a, b, c, d) &= p \, l_4(\iota a, \iota b, \iota c, \iota d) \\
&\quad + \sum_{\sigma \in S_4} \epsilon_\sigma \, p \, l_3(h l_2(\iota x_{\sigma_1}, \iota x_{\sigma_2}), \iota x_{\sigma_3}, \iota x_{\sigma_4}) \\
&\quad + \sum_{\text{pairings}} \epsilon \, p \, l_2(h l_2(\iota a, \iota b), h l_2(\iota c, \iota d)) \\
&\quad + \sum_{\sigma} \epsilon_\sigma \, p \, l_2(h l_3(\iota x_{\sigma_1}, \iota x_{\sigma_2}, \iota x_{\sigma_3}), \iota x_{\sigma_4}) \\
&\quad + \text{(trees with two internal edges)}
\end{align*}
The full formula involves 25 terms corresponding to trees with 4 leaves.
\end{computation}

\begin{theorem}[Uniqueness of minimal \texorpdfstring{$\Linf$}{L-infinity}-model \cite{LV12}; \ClaimStatusProvedElsewhere]
\label{thm:linf-minimal-unique}
If $(\fg, \{l_n\})$ is an $\Linf$-algebra over a field:
\begin{enumerate}[label=(\roman*)]
\item A minimal $\Linf$-model $(H, \{\tilde{l}_n\})$ exists with $\tilde{l}_1 = 0$.
\item Any two minimal $\Linf$-models are $\Linf$-isomorphic.
\item The isomorphism is unique up to $\Linf$-homotopy.
\end{enumerate}
\end{theorem}

\begin{construction}[Homotopy transfer for coalgebras]
\label{constr:htt-coalg}
For a dg-coalgebra $(C, \Delta, d)$ and SDR $(C, H, p, \iota, h)$, the transferred 
$\Ainf$-coalgebra structure on $H$ has coproducts:

\emph{$\tilde{\Delta}_1 = 0$.} No differential on homology.

\emph{$\tilde{\Delta}_2$.} The induced coproduct:
\[
\tilde{\Delta}_2(c) = (p \otimes p) \Delta(\iota(c)).
\]

\emph{$\tilde{\Delta}_n$ for $n \geq 3$.} Higher coproducts from trees:
\[
\tilde{\Delta}_n = \sum_{T \in \mathrm{Trees}_n} \epsilon(T) \cdot (p^{\otimes n}) \circ T(\Delta, h) \circ \iota
\]
where trees now have edges ``pointing down'' (toward outputs) rather than up.
\end{construction}

\begin{example}[Transfer for symmetric coalgebra]
\label{ex:transfer-sym-coalg}
Let $C = S^c(V)$ be the symmetric coalgebra on a chain complex $(V, d_V)$. 
The coproduct is deconcatenation:
\[
\Delta(v_1 \cdots v_n) = \sum_{I \sqcup J = [n]} v_I \otimes v_J
\]
where $v_I = \prod_{i \in I} v_i$ (in order).

If $V \xrightarrow{p} H_*(V) =: W$ is a deformation retract:
\begin{enumerate}[label=(\roman*)]
\item $\tilde{\Delta}_2$ on $S^c(W)$ is the standard deconcatenation (symmetric coalgebra).
\item $\tilde{\Delta}_3$ involves Massey products: corrections arise when $d_V$ is nontrivial.
\item If $V$ is formal (quasi-isomorphic to $H_*(V)$ with zero differential), then $\tilde{\Delta}_n = 0$ for $n \geq 3$.
\end{enumerate}
\end{example}


\section{Applications to chiral algebras}
\label{sec:htt-chiral}


\begin{theorem}[Chiral homotopy transfer; \ClaimStatusProvedHere]
\label{thm:chiral-htt}
\index{$A_\infty$-structure!transferred}
Let $\cA$ be an $\Eone$-chiral algebra on a curve $X$ and suppose we have an 
SDR of the underlying D-module:
\[
(\cA, H, p, \iota, h) \quad \text{with } H = H^{\mathrm{ch}}_*(\cA).
\]
Then:
\begin{enumerate}[label=(\roman*)]
\item $H$ inherits an $\Ainf$-chiral algebra structure.
\item The higher operations $\{m_n^{\mathrm{ch}}\}_{n \geq 3}$ are ``Massey products'' 
for the chiral structure.
\item If $\cA$ is Koszul, then $m_n^{\mathrm{ch}} = 0$ for $n \geq 3$.
\end{enumerate}
\end{theorem}

\begin{proof}
We apply the general homotopy transfer theorem (Theorem~\ref{thm:htt}) to the $\Eone$-chiral operad.

\emph{Step~1} (D-module compatibility). Since $\cA$ is a chiral algebra, it carries a $\cD_X$-module structure.  The SDR maps $(p, \iota, h)$ must be morphisms of $\cD_X$-modules (i.e., commute with the action of $\cD_X$).  This holds because chiral homology $H = H^{\mathrm{ch}}_*(\cA)$ is defined as a derived functor in the category of $\cD_X$-modules, so $p$ and $\iota$ are automatically $\cD_X$-linear, and $h$ can be chosen $\cD_X$-linearly by the splitting argument of Lemma~\ref{lem:sdr-existence}(i) applied within the $\cD_X$-module category (which is abelian with enough projectives over a smooth curve~$X$).

\emph{Step~2} (Transferred binary operation). The transferred product is $\tilde{m}_2 = p \circ \mu^{\mathrm{ch}} \circ (\iota \otimes \iota)$, exactly as in Construction~\ref{constr:transfer-ainf}.  Since $\mu^{\mathrm{ch}}$ is a chiral product (defined via the $j_*j^*$ functor on $X \times X$), $\tilde{m}_2$ inherits $\cD_X$-linearity from Step~1.

\emph{Step~3} (Higher operations via trees). The transferred operations $\tilde{m}_n^{\mathrm{ch}}$ for $n \geq 3$ are sums over planar rooted trees with $n$ leaves (as in Construction~\ref{constr:transfer-ainf}), where each internal edge contributes a factor of $h$ and each internal vertex contributes the chiral product $\mu^{\mathrm{ch}}$.  Since only tree-level graphs contribute (no loops; cf.\ Remark~\ref{rem:tree-level}), each $\tilde{m}_n^{\mathrm{ch}}$ is a finite sum.

\emph{Step~4} ($\Ainf$-relations). The transferred operations $\{\tilde{m}_n^{\mathrm{ch}}\}$ satisfy the $\Ainf$-relations by the same algebraic argument as in the general theorem: the identity $\sum_{r+s+t=n}(-1)^{rs+t}\tilde{m}_{r+1+t}(\mathrm{id}^{\otimes r} \otimes \tilde{m}_s \otimes \mathrm{id}^{\otimes t}) = 0$ follows from substituting the tree formulas and using the $\Eone$-relations on $\cA$ together with the SDR identities $p\iota = \mathrm{id}$, $\mathrm{id} - \iota p = dh + hd$, $h^2 = 0$.

\emph{Koszulness} (iii). If $\cA$ is Koszul, the bar spectral sequence degenerates at $E_2$ with $E_\infty$ concentrated on the Koszul line $q = 0$ (Proposition~\ref{prop:degen-koszul}).  This means $\mathrm{Ext}^{p,q}_{\cA}(k,k) = 0$ for $q \neq 0$: the bar resolution is minimal.  One may therefore choose the SDR for $\Bbar(\cA) \to H^*(\Bbar(\cA))$ compatible with the bigrading, so that the homotopy $h$ preserves bar degree and shifts internal degree.  Every tree contributing to $\tilde{m}_n^{\mathrm{ch}}$ for $n \geq 3$ would produce an operation of bar-degree shift incompatible with the Koszul line concentration, forcing $\tilde{m}_n^{\mathrm{ch}} = 0$.  Equivalently, for a minimal resolution all Massey products vanish.
\end{proof}

\begin{example}[Kac--Moody minimal model]
\label{ex:km-minimal}
For the affine Kac--Moody algebra $\hat{\fg}_k$ at level $k$ and a smooth projective curve $X$ of genus $g$:
\begin{enumerate}[label=(\roman*)]
\item The chiral homology $H^{\mathrm{ch}}_*(X, \hat{\fg}_k)$ depends on both $k$ and $g$.
\item At generic $k$ (irrational, or more precisely not a non-negative integer and not equal to $-h^\vee$): the higher chiral homology $H^{\mathrm{ch}}_i(X, \hat{\fg}_k) = 0$ for $i > 0$, while $H^{\mathrm{ch}}_0(X, \hat{\fg}_k)$ reduces to the space of conformal blocks, which is one-dimensional for the vacuum module \cite{BD04}.
\item At non-negative integral $k$: the chiral homology $H^{\mathrm{ch}}_*(X, \hat{\fg}_k)$ is related to spaces of conformal blocks for integrable representations, and the minimal model encodes this representation theory.
\item The transferred higher operations $m_n^{\mathrm{ch}}$ for $n \geq 3$
vanish by Koszulness of $\hat{\fg}_k$ (at non-critical level $k \neq -h^\vee$).
\end{enumerate}
\end{example}

\begin{proposition}[Transferred structure and bar complex \cite{LV12}; \ClaimStatusProvedElsewhere]
\label{prop:transfer-bar}
The homotopy transfer of chiral structures is compatible with the bar construction:
\[
\Bbar^{\mathrm{geom}}(H, \{\tilde{m}_n\}) \simeq \Bbar^{\mathrm{geom}}(\cA, \{m_n\})
\]
as geometric bar complexes. The quasi-isomorphism is induced by the SDR data.
\end{proposition}

\begin{proof}
The bar construction is functorial for $\Ainf$-morphisms \cite{LV12}: an $\Ainf$-quasi-isomorphism $f: A \xrightarrow{\sim} B$ induces a quasi-isomorphism $\Bbar(f): \Bbar(A) \xrightarrow{\sim} \Bbar(B)$ of bar complexes. Since the SDR maps $(\iota, p)$ extend to $\Ainf$-quasi-isomorphisms by Theorem~\ref{thm:htt}, the bar complexes $\Bbar(\cA)$ and $\Bbar(H, \{\tilde{m}_n\})$ are quasi-isomorphic. The geometric realization is compatible because $\Bbar^{\mathrm{geom}}$ is computed as the factorization homology of $\Bbar$, and factorization homology preserves quasi-isomorphisms between factorization algebras on $X$.
\end{proof}


\section{Homotopy transfer and the main theorems}
\label{sec:htt-main-theorems}

The homotopy transfer theorem \emph{is} the mechanism by which classical Koszul duality
acquires quantum corrections.  We make this precise.

\subsection{The bar-cobar counit as HTT output}

\begin{theorem}[Bar-cobar inversion via homotopy transfer;
\ClaimStatusProvedHere]
\label{thm:bar-cobar-htt}
\index{homotopy transfer theorem!bar-cobar inversion}
Let $\cA$ be a chiral Koszul algebra on a smooth curve~$X$.
The bar-cobar counit
\[
\psi \colon \Omega(\Bbar(\cA)) \to \cA
\]
of Theorem~\textup{\ref{thm:bar-cobar-isomorphism-main}} is the
\emph{canonical} $\Ainf$-quasi-isomorphism produced by the homotopy
transfer theorem (Theorem~\ref{thm:htt}), applied to the SDR
\[
\bigl(\Omega(\Bbar(\cA)),\; \cA,\; \psi,\; \iota_{\mathrm{counit}},\;
h_{\mathrm{Hodge}}\bigr)
\]
where $\iota_{\mathrm{counit}}$ is the canonical inclusion and
$h_{\mathrm{Hodge}}$ is the contracting homotopy from the
Hodge decomposition of $\Omega(\Bbar(\cA))$ relative to the bar
filtration.
\end{theorem}

\begin{proof}
By construction, the cobar complex $\Omega(\Bbar(\cA))$ is the
cofree chiral algebra on $s\,\overline{\cA}^!$ with differential
$d_{\mathrm{cobar}} = d_{\mathrm{coop}} + d_{\mathrm{tw}}$, where
$d_{\mathrm{tw}}$ encodes the twisting morphism
$\tau \colon \Bbar(\cA) \to \cA$.

\emph{Step~1} (SDR from bar filtration).
The bar filtration $F_p\Omega(\Bbar(\cA))$ gives a spectral sequence
with $E_1^{p,q} = H^q(\mathrm{gr}_p^F\,\Omega(\Bbar(\cA)))$.
By Koszulness, $E_1$ is concentrated on $q = 0$
(Proposition~\ref{prop:degen-koszul}).  The $E_1$-degeneration
produces an SDR: the projection~$\psi$ to
$E_\infty \cong \cA$ and the inclusion~$\iota$ from $\cA$ back to the
cobar complex, with contracting homotopy~$h$ lowering the $q$-degree.
The side conditions $\psi\iota = \mathrm{id}$, $h^2 = 0$, $\psi h = 0$,
$h\iota = 0$ follow from the bigrading.

\emph{Step~2} (HTT produces the same map).
Applying Theorem~\ref{thm:htt} to the strict $\Eone$-chiral algebra
structure on $\Omega(\Bbar(\cA))$ and the SDR from Step~1 produces
transferred operations $\{\tilde{m}_n\}_{n \geq 2}$ on $\cA$
and an $\Ainf$-quasi-isomorphism $f_\infty \colon \cA \to
\Omega(\Bbar(\cA))$.  The component $f_1 = \iota$ and the
binary product $\tilde{m}_2 = \psi \circ \mu \circ
(\iota \otimes \iota) = \mu_{\cA}$ (the original product on $\cA$,
since $\psi\iota = \mathrm{id}$).

\emph{Step~3} (Koszulness forces vanishing).
For $n \geq 3$, each tree $T$ contributing to $\tilde{m}_n$
(Construction~\ref{constr:transfer-ainf}) inserts at least one
homotopy~$h$ on an internal edge.  Since $h$ lowers the filtration
degree by~$1$ and the Koszul-line concentration forces all
non-trivial cohomology to live on $q = 0$, the composition
$\psi \circ (\text{tree})\circ \iota^{\otimes n}$ vanishes: the
output lives in $q < 0$, which is empty.  Therefore
$\tilde{m}_n = 0$ for $n \geq 3$, and $f_\infty = \psi$ is a
strict quasi-isomorphism of dg algebras, not merely of $\Ainf$-algebras.

This is the chain-level content of the bar-cobar inversion
(Theorem~\ref{thm:bar-cobar-isomorphism-main}): on the Koszul locus,
the bar-cobar counit is the HTT quasi-isomorphism, and Koszulness is
the vanishing of all Massey products.
\end{proof}

\begin{remark}[Koszulness as formality]
\label{rem:koszulness-formality-htt}
\index{formality!Koszulness}
The preceding proof identifies three equivalent conditions
(cf.\ Remark~\ref{rem:homotopy-native-b}):
\begin{enumerate}[label=(\roman*)]
\item $\cA$ is chiral Koszul (bar spectral sequence degenerates at
$E_2$, Koszul-line concentration).
\item The transferred $\Ainf$-structure on $H^*(\Bbar(\cA))$ is
\emph{formal}: $\tilde{m}_n = 0$ for $n \geq 3$.
\item The bar-cobar counit $\psi$ is a \emph{strict}
quasi-isomorphism (not merely $\Ainf$).
\end{enumerate}
The equivalence (i) $\Leftrightarrow$ (ii) is the homotopy-algebraic
meaning of Koszulness: a Koszul algebra is one whose derived
structure contains no hidden higher information beyond the binary
product.  Condition~(iii) is the geometric meaning: the
configuration space integrals computing $\psi$ collapse to residue
extraction alone, with no contributions from higher-codimension
boundary strata.
\end{remark}

\subsection{Trees and boundary strata: the geometric HTT}

The tree formula of Theorem~\ref{thm:tree-formula} has a direct
geometric realization via the Fulton--MacPherson compactification.

\begin{proposition}[Trees as boundary strata;
\ClaimStatusProvedHere]
\label{prop:trees-boundary-strata}
\index{Fulton--MacPherson compactification!boundary strata}
\index{homotopy transfer theorem!geometric realization}
Let $T$ be a planar rooted tree with $n$ leaves contributing to the
transferred operation $\tilde{m}_n$
(Construction~\ref{constr:transfer-ainf}).  The corresponding
term in the tree formula equals the integral
\[
\tilde{m}_n^T(a_1, \ldots, a_n) \;=\;
\int_{\partial_T \overline{C}_n(X)}
\mathrm{ev}^*(\bar{a}_1 \otimes \cdots \otimes \bar{a}_n)
\;\wedge\;
\bigwedge_{e \in E_{\mathrm{int}}(T)} \omega_e\,,
\]
where:
\begin{enumerate}[label=(\roman*)]
\item $\partial_T \overline{C}_n(X) \hookrightarrow
\overline{C}_n(X)$ is the boundary stratum of the
Fulton--MacPherson compactification indexed by~$T$: the locus
where the points cluster according to the vertex structure of~$T$;
\item $\omega_e = d\log(z_{s(e)} - z_{t(e)})$ is the logarithmic
propagator on the internal edge~$e$ with source $s(e)$ and
target $t(e)$;
\item the homotopy $h$ of the SDR corresponds to integration over
the fiber direction in $\partial_T\overline{C}_n(X) \to
\overline{C}_{|V(T)|}(X)$, i.e., integration over the
``infinitesimal screen'' at each vertex cluster.
\end{enumerate}
The full transferred operation is the sum over all trees:
\[
\tilde{m}_n = \sum_{T \in \mathrm{PRT}_n}
\int_{\partial_T \overline{C}_n(X)}
\mathrm{ev}^*(\bar{a}_1 \otimes \cdots \otimes \bar{a}_n)
\;\wedge\;\omega_T\,.
\]
\end{proposition}

\begin{proof}
The Fulton--MacPherson compactification
$\overline{C}_n(X) = \mathrm{Bl}_{\Delta}(X^n)$ has a stratification
indexed by planar rooted trees with $n$ leaves: each stratum
$\partial_T$ records a pattern of coincidences among the $n$ points
(cf.\ \cite{FM94}).

The bar differential $d_{\mathrm{bar}}$ on
$\Bbar^{\mathrm{geom}}_n(\cA)$ is the residue along the codimension-1
boundary $\partial_{ij}\overline{C}_n(X)$ where points $z_i, z_j$
collide.  This is the contribution from trees with a single
internal edge.  Higher-codimension strata correspond to multiple
simultaneous collisions, i.e., trees with multiple internal edges.

The homotopy $h$ in the SDR integrates ``inward'' along fibers of the
boundary projection $\partial_T \to \partial_{T/e}$, which contracts
an internal edge $e$ of the tree~$T$.  This is the geometric
content of the formula $h = (\text{fiber integral over screen
direction})$: when two points have already collided (codimension-$1$
stratum), the homotopy $h$ integrates over the residual angular
direction to produce a new composite element on the coarser stratum.

Composing: the tree formula $p \circ \prod_v m_{|v|} \circ
\prod_e h \circ \iota^{\otimes n}$ places propagators $\omega_e$
on internal edges, OPE operations $m_{|v|}$ at vertices, and
integrates over the fiber directions via $h$.  The result is
precisely the integral over $\partial_T\overline{C}_n(X)$ of the
product of logarithmic propagators.
\end{proof}

\begin{remark}[Feynman diagrams as transferred operations]
\label{rem:feynman-htt}
\index{Feynman diagrams!homotopy transfer}
Proposition~\ref{prop:trees-boundary-strata} establishes the following
dictionary between the algebraic HTT and the physical Feynman diagram
expansion:

\medskip
\begin{center}
\renewcommand{\arraystretch}{1.15}
\begin{tabular}{@{} l @{\;\;$\longleftrightarrow$\;\;} l @{}}
\toprule
\textbf{HTT (algebraic)} & \textbf{Feynman (geometric)} \\
\midrule
Planar rooted tree $T$ & Tree-level Feynman diagram \\
Internal edge of $T$ & Propagator insertion $\eta_{ij}$ \\
Vertex of valence $k$ & $k$-valent interaction vertex (OPE) \\
Homotopy $h$ & Integration over loop momentum (fiber) \\
SDR projection $p$ & External state projection \\
Transferred $\tilde{m}_n$ & $n$-point tree-level amplitude \\
Massey product $\tilde{m}_3 \neq 0$ & Non-trivial 3-point function \\
Koszulness ($\tilde{m}_n = 0$, $n \geq 3$) & Free field theory (Gaussian) \\
\bottomrule
\end{tabular}
\end{center}

\medskip
\noindent
The identification ``Koszulness = free theory'' has precise content:
for a Koszul chiral algebra, the bar complex is a minimal resolution,
all Massey products vanish, and the Feynman expansion truncates at
tree level with only 2-valent vertices (propagators).  Non-Koszul
algebras have non-trivial Massey products, corresponding to genuine
higher-point interactions that cannot be reduced to pairwise
propagation.
\end{remark}

\subsection{Genus corrections as transferred operations}
\label{subsec:genus-corrections-htt}

At genus $g \geq 1$, the bar complex acquires curvature
$\dfib^{\,2} = \kappa(\cA) \cdot \omega_g$
(Convention~\ref{conv:higher-genus-differentials}), and the
HTT framework extends to the curved setting.

\begin{proposition}[Genus-\texorpdfstring{$1$}{1} curvature as \texorpdfstring{$m_0$}{m0};
\ClaimStatusProvedHere]
\label{prop:genus1-curvature-m0}
\index{curvature!as $m_0$}
The genus-$1$ obstruction $\kappa(\cA) \in k$ of
Theorem~\textup{\ref{thm:genus-universality}} is the nullary
operation $m_0$ of the curved $\Ainf$-structure transferred from
the genus-$1$ bar complex:
\[
m_0 \;=\; \kappa(\cA) \cdot [\omega_1]
\;\in\; H^2(\overline{\mathcal{M}}_{1,1},\, \mathbb{Q})\,.
\]
More precisely, let $\Bbar^{(1)}(\cA)$ be the genus-$1$ bar complex
with fiberwise differential $\dfib$ satisfying $\dfib^{\,2} =
\kappa(\cA) \cdot \lambda_1 \cdot \mathrm{id}$ where $\lambda_1$ is the
Hodge class.  Then the HTT applied to the SDR from the bar
filtration produces:
\begin{enumerate}[label=(\roman*)]
\item A nullary operation $m_0 = \kappa(\cA) \cdot \lambda_1$,
measuring the failure of $d^2 = 0$ at genus~$1$.
\item A unary operation $m_1$ with $m_1^2(a) = [m_0, a]$
(the commutator identity for curved $\Ainf$-algebras).
\item Binary and higher operations $m_n$ ($n \geq 2$) modified
by the curvature: $m_2$ acquires corrections from the
Eisenstein propagator on the elliptic curve, and $m_3$
acquires contributions from the genus-$1$ Massey product.
\end{enumerate}
\end{proposition}

\begin{proof}
The genus-$1$ bar complex on an elliptic curve $E_\tau$ has
differential $\dfib = d^{(0)} + d^{(1)}$ where $d^{(0)}$ is the
genus-$0$ collision differential and $d^{(1)} = \kappa(\cA) \cdot
\mathrm{tr}$ inserts the trace element
(Theorem~\ref{thm:concrete-quantum-differential}).
Then $(d^{(0)})^2 = 0$ and $\{d^{(0)}, d^{(1)}\} = 0$
by the decomposition of
$\Dg{1}^2 = 0$ (Theorem~\ref{thm:genus-differential}).

The SDR from the bar filtration gives
$(\Bbar^{(1)}(\cA), H^*(\Bbar^{(0)}(\cA)), p, \iota, h)$
at the associated graded level.  Applying HTT with the perturbed
differential $d^{(1)}$ (the ``perturbation lemma'' variant
\cite{LV12}), the transferred structure acquires:
\begin{itemize}
\item $m_0 = p \circ d^{(1)} \circ \iota(1) = \kappa(\cA) \cdot
p(\mathrm{tr}) = \kappa(\cA) \cdot \lambda_1$ (the trace element
projects to the Hodge class),
\item $m_1(a) = p \circ d^{(1)} \circ \iota(a) + p \circ d^{(0)}
\circ h \circ d^{(1)} \circ \iota(a) + \cdots$
(geometric series in $h \circ d^{(1)}$, converging
because $d^{(1)}$ raises filtration degree),
\item $m_2(a,b) = p \circ \mu \circ (\iota(a) \otimes \iota(b))
+ (\text{corrections from } h \circ d^{(1)})$.
\end{itemize}
The relation $m_1^2(a) = m_2(m_0, a) - m_2(a, m_0) = [m_0, a]$
follows from the general curved $\Ainf$-identity with $r+s+t = 1$
and $s = 0$.
\end{proof}

\begin{remark}[Higher genus: moduli cohomology as
transferred operations]
\label{rem:higher-genus-htt}
\index{moduli space!homotopy transfer}
At genus $g \geq 2$, the bar complex $\Bbar^{(g)}(\cA)$ is a family
over $\overline{\mathcal{M}}_g$ with fiberwise differential $\dfib$
satisfying $\Dg{g}^2 = 0$ (Theorem~\ref{thm:genus-differential}).
The HTT applied fiberwise produces a curved $\Ainf$-algebra over
$\overline{\mathcal{M}}_g$ whose operations are sections of bundles
over moduli space.  The key identification is:

\medskip
\emph{The $n$-ary transferred operation $\tilde{m}_n^{(g)}$ is a
section of
\[
\mathcal{H}om(\cA^{\boxtimes n},\, \cA) \otimes
H^*(\overline{\mathcal{M}}_{g,n},\, \mathbb{Q})\,.
\]
The dependence on $H^*(\overline{\mathcal{M}}_{g,n})$ is the
``quantum correction'' from the genus-$g$ moduli geometry.}

\medskip
\noindent
This gives a homotopy-algebraic interpretation of Theorem~C
(Deformation-Obstruction Complementarity,
Theorem~\ref{thm:quantum-complementarity-main}):
the quantum obstruction $Q_g(\cA)$ and the dual obstruction
$Q_g(\cA^!)$ together span $H^*(\overline{\mathcal{M}}_g, Z(\cA))$
\emph{because they are the complementary components of the
transferred $\Ainf$-structure on $H^*(\Bbar^{(g)}(\cA))$, restricted
to the center and projected along the Koszul line and its complement}.
The complementarity $Q_g(\cA) + Q_g(\cA^!) =
H^*(\overline{\mathcal{M}}_g, Z(\cA))$ is the statement that
\emph{the full transferred structure exhausts the moduli
cohomology, with nothing left over}.

This interpretation elevates Theorem~C from a numerical identity to a
structural statement: the bar-cobar machine, viewed through HTT,
\emph{decomposes} moduli-space cohomology into the piece
controlled by $\cA$ and the piece controlled by $\cA^!$, with the
Koszul duality functor as the sorting mechanism.
\end{remark}

\begin{remark}[Physical content: renormalization vs.\ homotopy transfer]
\label{rem:renormalization-htt}
\index{renormalization!homotopy transfer}
Homotopy transfer and
perturbative renormalization in quantum field theory are analogous, and in the chiral algebra setting they coincide:

\begin{enumerate}[label=(\roman*)]
\item \textbf{Tree-level HTT = classical field theory.}
The tree formula (Theorem~\ref{thm:tree-formula}) sums over
tree-level Feynman diagrams.  The condition $h^2 = 0$ in the SDR
(the ``no-loop'' condition from Remark~\ref{rem:tree-level})
ensures that only trees contribute, not graphs with loops.
Classical field theory is determined by its tree-level amplitudes.

\item \textbf{Genus-$g$ curvature = $g$-loop quantum corrections.}
The curvature $\kappa(\cA) \cdot \omega_g$ at genus~$g$ is the
analogue of the $g$-loop counterterm in perturbative
renormalization.  The perturbation lemma that extends HTT to
the curved setting (Proposition~\ref{prop:genus1-curvature-m0})
corresponds to absorbing the counterterm into the effective
action, order by order in $\hbar \sim \kappa(\cA)$.

\item \textbf{Koszulness = absence of anomalies.}
A Koszul chiral algebra has $\tilde{m}_n = 0$ for $n \geq 3$
(Theorem~\ref{thm:chiral-htt}(iii)), meaning no higher-point
interactions are generated by integrating out degrees of freedom.
In physics, this is the statement that the theory is
\emph{anomaly-free at the level of OPE structure}: the classical
OPE data determine the quantum theory completely, with no need
for higher-order counterterms.  The genus-$g$ curvature $\kappa(\cA)$
is the \emph{only} quantum correction, and it is universal
(Theorem~\ref{thm:genus-universality}).
\end{enumerate}
\end{remark}
