\chapter{Level-2 vertical holographic equivalence: $\bar B^{\mathrm{ch}}(A_b)$ and $\SpCh_{\Sigma_{d-1},C}(\PhiFA_d(\cC))$}
\label{app:vertical-equivalence-level-2}

The Beilinson tower carries a vertical equivalence at level~$2$:
the Open-quadrant chiral bar coalgebra $\Bbarch_X(A_b)$ at the chart
algebra $A_b=\End_{\cC^{\mathrm{op}}}(b)$ on the tangential log curve
$(X,D,\tau)$ matches the second stage of the CY-quadrant two-stage
factorisation
\[
 \Phi_d^{(\Sigma_{d-1},C)}\;=\;\SpCh_{\Sigma_{d-1},C}\circ\PhiFA_d ,
\]
on the verified loci, after a chosen specialisation datum
$(\Sigma_{d-1},C)$ and on the Koszul / Kontsevich--Tamarkin formality
locus on which both sides are unambiguous. The two-stage factorisation
is constitutive, never one-stage: $\PhiFA_d$ produces an $E_d$-holomorphic
factorisation algebra on the CY$_d$ target, and $\SpCh_{\Sigma_{d-1},C}$
integrates along an admissible $(d{-}1)$-cycle and restricts to a
reference curve $C\subset X$. The level-$2$ vertical is the assertion
that $\Bbarch_X(A_b)$ and the Stage-$2$ output represent the same
$E_n$-twisting coalgebra on $C$.

\section{Statement}
\label{sec:vert-level-2-statement}

\paragraph{Type signature.}
\textit{Quadrant}: Open $\leftrightarrow$ CY.
\textit{Presentation}: $\mathrm{Cat}$ (factorisation $\infty$-category
on the Open side; CY$_d$-category on the CY side).
\textit{Level}: $2$.
\textit{Hypothesis package}: on the Open side, augmentation,
augmentation-ideal completeness, and Koszul-locus residence
$A_b\in\Kosz(X)$ (hypotheses (H1)--(H3) of
Theorem~\ref{thm:koszul-reflection}); on the CY side, a fixed
Kontsevich--Tamarkin $E_d$-formality torsor point at the relevant $d$,
a witnessed admissible specialisation datum
$(\Sigma_{d-1},C)$ in the sense of Vol~III's
witnessed-admissibility definition, and a chosen Calabi--Yau form
$\Omega_X\in H^0(X,K_X)$.

\begin{theorem}[Level-$2$ vertical equivalence; \ClaimStatusConditional]
\label{thm:vertical-level-2}
Fix $d\in\{1,2,3\}$ and a CY$_d$ category $\cC$ satisfying the
Vol~III hypotheses (H1)--(H4) of
Proposition~\ref{V3-prop:phi-fa-three-step-assembly}. Let
$X$ be a smooth complex CY$_d$ variety with chosen Calabi--Yau form
$\Omega_X$ and witnessed admissible specialisation datum
$(\Sigma_{d-1},C)$; on the Open side, let $C$ be the underlying
tangential log curve $(X_{\mathrm{Open}},D,\tau)=(C,D_C,\tau_C)$ and
$A_b=\End_{\cC^{\mathrm{op}}}(b)$ a Koszul chart algebra at a
chosen vacuum $b$. The two presentations agree at level~$2$:
\[
 \Bbarch_C(A_b)\;\simeq\;
 \SpCh_{\Sigma_{d-1},C}\!\bigl(\PhiFA_d(\cC)\bigr)
 \qquad
 \text{in $\EnHolFA(C)$ at $n=n_{\mathrm{native}}(d)$,}
\]
where $n_{\mathrm{native}}(d)\in\{\infty,2,1\}$ for $d\in\{1,2,3\}$ by
Proposition~\ref{V3-prop:native-en-level} of Vol~III, and the
identification is canonical after the formality torsor and the
admissible datum are fixed.
\end{theorem}

The equivalence is not a tautology: the left-hand side is a chiral
$E_n$-coassociative coalgebra on $C$ built from an associative chart
$A_b$ by the chiral bar construction, while the right-hand side is the
restriction-after-pushforward of an $E_d$-holomorphic factorisation
algebra on the higher-dimensional ambient $X$. The theorem asserts
that the two constructions agree on the verified locus.

\section{The two stages}
\label{sec:vert-level-2-two-stages}

\paragraph{Stage~1: $\PhiFA_d$.}
The native holomorphic factorisation algebra
$\PhiFA_d(\cC)\in\EdHolFA(X)$ of Vol~III is assembled
from the Gerstenhaber bracket of degree $1-d$ on $\HH^\bullet(\cC)$ by
the conjunction of Kontsevich $E_d$-formality, Tamarkin
deformation-quantisation, and Costello--Gwilliam--Li holomorphic
locality on $X$. It satisfies factorisation over $d$-dimensional opens.
The construction depends on a torsor point in $\mathrm{Form}_d(\Q)$
controlled by $\GRT_1(\Q)$ at $d=3$; before fixing this point, the
output is defined only up to the formality torsor.

\paragraph{Stage~2: $\SpCh_{\Sigma_{d-1},C}$.}
The chiral specialisation functor
\[
 \SpCh_{\Sigma_{d-1},C}\colon \EdHolFA(X)\longrightarrow \EnHolFA(C),
 \qquad
 \cF\mapsto\Bigl(\textstyle\int_{\Sigma_{d-1}}\cF\Bigr)\Big|_C ,
\]
integrates along the admissible $(d{-}1)$-cycle (factorisation homology
in the sense of Lurie and Francis--Gaitsgory) and restricts to the
reference curve $C\subset X$. The native operadic level on $C$ is
$n_{\mathrm{native}}(d)$; the residual chiral structure on $C$ is
controlled by Serre duality and the degree-$(1-d)$ bracket.

\paragraph{Compatibility with Theorem~A.}
On the Koszul locus, Theorem~A
(Theorem~\ref{thm:koszul-reflection}) gives the inversion
$\Omega^{\mathrm{ch}}_C\Bbarch_C(A_b)\xrightarrow{\sim}A_b$. The
level-$2$ vertical converts this into the identification
$\SpCh_{\Sigma_{d-1},C}(\PhiFA_d(\cC))\simeq \Bbarch_C(A_b)$ as the
twisting coalgebra of $A_b$. The bar plays the role of the Stage-$2$
chiral shadow; the chart algebra $A_b$ is recovered at level~$1$ by
chiral cobar inversion. Both sides represent the same $E_n$-twisting
coalgebra; neither side is the bulk
$Z^{\mathrm{der}}_{\mathrm{ch}}(A_b)$, which lives at level~$3$ via
chiral Hochschild cochains (Theorem~B and the level-$3$ vertical).

\section{The typed two-stage factorisation}
\label{sec:vert-level-2-two-stage-typed}

A one-stage expression
``$\Phi_d\colon\CY_d\text{-}\Cat\to\ChirAlg$'' has no typed target:
it suppresses the distinction between the holomorphic factorisation
algebra on $X$ and its chiral shadow on $C$. The typed statement is the
two-stage factorisation
$\Phi_d^{(\Sigma_{d-1},C)}=\SpCh_{\Sigma_{d-1},C}\circ\PhiFA_d$ with
the specialisation datum named explicitly. The functor terminology
itself is improper for the $d$-indexed family $\{\Phi_d\}$, since the
target $E_{n(d)}$-chiral category depends on $d$ and the family is a
correspondence family rather than a single functor (Vol~III's
correspondence-family convention).

\paragraph{Typed replacement.}
A statement of the form
``$\Bbarch(A_b)$ is the CY-to-chiral image of $\cC$ under $\Phi_d$''
without naming the specialisation datum forgets the source, the
specialisation cycle, and the residual operadic level. The typed form is:
``$\Bbarch_C(A_b)\simeq \SpCh_{\Sigma_{d-1},C}(\PhiFA_d(\cC))$
on the verified locus, with $(\Sigma_{d-1},C)$ a witnessed admissible
specialisation datum and the Kontsevich--Tamarkin formality torsor
fixed.'' The $E_n$-level on $C$ is $n_{\mathrm{native}}(d)$; the
$E_d$-level on $X$ is the operadic native level of Stage~$1$.

\section{Stratification by dimension}
\label{sec:vert-level-2-stratification}

The level-$2$ vertical equivalence is proved at distinct strengths
across the dimensions $d$. The stratification mirrors the open
frontier of Vol~III's two-stage factorisation theorem
(Theorem~\ref{V3-thm:phi-two-stage-factorisation}; the level-$2$
extension problem for $d\geq 4$).

\begin{description}
 \item[$d=1$.] Trivial: the bracket on $\HH^\bullet(\cC)$ vanishes,
 the $E_d$-stage is $E_\infty$-commutative, and Stage~$2$ reduces to
 ordinary restriction. The equivalence with $\Bbarch_C(A_b)$ is
 functor-level on the entirety of $\CY_1\text{-}\Cat$ and reduces
 to the standard commutative bar coalgebra of $A_b$.

 \item[$d=2$.] Functor-level: Kontsevich $E_2$-formality is
 unconditional over $\Q$, and the $\bS^2$-framing of step~(c) of
 Proposition~\ref{V3-prop:phi-fa-three-step-assembly} proceeds
 unconditionally. The level-$2$ vertical
 holds as a natural transformation of symmetric-monoidal
 $(\infty,1)$-functors. K3 is the principal instance: the K3 chart
 algebra in the reduced+$\mathrm{Aut}$ stratum produces, at
 Stage~$2$, the Mukai--Heisenberg $\Bbarch$-shadow.

 \item[$d=3$.] Object-level on the verified K3-Class~A locus
 (Frontier~F9): the chain-level framing, anomaly-completion, Costello
 correction, filtered Hochschild comparison, and witnessed
 specialisation data of the Vol~III $\CY$-A$_3$ existence-and-rigidity
 theorem give an explicit chain-level identification. Functor-level
 across all of $\CY_3\text{-}\Cat$ remains conditional on
 category-level $E_3$-formality of $\HH^\bullet(\cC)$.

 \item[$d\geq 4$.] Open: Stage~$1$ for $d\geq 4$ is the conditional
 template of Definition~\ref{V3-def:phi-fa-and-sp}, awaiting the
 corresponding holomorphic-twist, framing, and completion witnesses;
 the level-$2$ vertical is correspondingly conjectural.
\end{description}

\section{Three independent verification paths}
\label{sec:vert-level-2-verification}

\paragraph{Path~1: $\beta\gamma$ at $d=2$ on local $\bC^2$.}
Stage~$1$ produces the Mukai-doubling shadow on local $\bC^2$ via the
Hochschild $E_2$-bracket of degree $-1$; Stage~$2$ reduces, on a
reference curve $C\subset\bC^2$, to the standard $\beta\gamma_\lambda$
chart algebra. The chiral bar $\Bbarch_C(\beta\gamma_\lambda)$ matches
the $\SpCh$-output by direct computation, recovering the class-$\mathsf{C}$
shadow profile $S_2=6\lambda^2-6\lambda+1$, $S_3=0$, $S_4=-5/12$,
$S_r=0$ for $r\geq 5$ of \S\ref{sec:master-concordance-classification}.

\paragraph{Path~2: $K3$ at $d=2$, Heisenberg--Mukai stratum.}
On the reduced+$\mathrm{Aut}$ stratum, Stage~$1$ produces an
$E_2$-holomorphic FA on K3 controlled by the Mukai pairing; Stage~$2$
descends, on a reference elliptic curve $E\subset K3$, to the
Heisenberg--Mukai chiral algebra $\mathcal H^{\mathrm{Heis}}_{\mathrm{Muk}}$
of Vol~III's level-$2$ row. The chiral bar of the chart agrees: the
class-$\mathsf{B}$ row $(0,0,3,5,24)$ of the $5\times5$
$\kappa$-stratification matrix is reproduced from both sides, with
$\kappa_{\mathrm{BKM}}=5$ at the paramodular $\Delta_5$ input by the
universal Borcherds-weight identity $\kappa_{\mathrm{BKM}}(\Phi_N)=c_N(0)/2$.

\paragraph{Path~3: $K3\times E$ at $d=3$, witnessed Class~A locus.}
Stage~$1$ at $d=3$ on $X=K3\times E$ produces $\PhiFA_3(D^b\mathrm{Coh}(K3\times E))$;
six routes to $G(K3\times E)$ are six $(\Sigma_2,C)$-specialisations
of one Stage~$1$ datum, not six $\Phi_3$-applications. Stage~$2$ on a
specific $(\Sigma_2,C)$ produces, on the verified Class~A locus, the
Mukai--Heisenberg parent
$\SpCh_{K3,E}(\PhiFA_3(\mathrm{Perf}(K3\times E)))\to\mathcal H^{\mathrm{Heis}}_{\mathrm{Muk}}\xrightarrow{\mathrm{Bor}}U_{\mathrm{ch}}(\mathfrak g_{\Delta_5})$,
whose chiral bar agrees with the Open-side $\Bbarch_C(A_b)$ at the
chosen vacuum. The K3-Yangian Mukai-self-mirror branch
$Y_\hbar(\mathfrak{so}(4|20))$ supplies a second internal cross-check
on the same row.

\section{Hypothesis package and extension locus}
\label{sec:vert-level-2-hypothesis}

\paragraph{Hypothesis package.}
\begin{itemize}
 \item[(\textsc{Open})] $A_b=\End_{\cC^{\mathrm{op}}}(b)$ at a chosen
 vacuum on a factorisation dg-category $\cC^{\mathrm{op}}$ on
 $(C,D_C,\tau_C)$, augmented and complete; $A_b\in\Kosz(C)$ in the
 sense of Theorem~\ref{thm:koszul-reflection}.
 \item[(\textsc{CY})] $\cC\in\CY_d\text{-}\Cat$ smooth proper over a
 field of characteristic zero, $\HH^0(\cC)=k$, non-degenerate cyclic
 pairing in degree $-d$.
 \item[(\textsc{Datum})] $X$ a smooth complex CY$_d$ variety with chosen
 $\Omega_X\in H^0(X,K_X)$; an admissible $(d{-}1)$-cycle
 $\Sigma_{d-1}\subset X$ and a reference curve $C\subset X$ forming a
 witnessed specialisation datum.
 \item[(\textsc{Formality})] A fixed Kontsevich--Tamarkin
 $E_d$-formality torsor point; at $d=3$, a fixed
 $\GRT_1(\Q)$-torsor representative.
\end{itemize}

\paragraph{Extension locus.}
For $d=1$ and $d=2$ the equivalence is unconditional at functor level.
For $d=3$ it is object-level after the witnessed Class~A locus is
extended through the chain-level framing, anomaly-completion, and
filtered Hochschild comparison of Vol~III's $\CY$-A$_3$ theorem. For
$d\geq 4$ the open obligations are: construction of the holomorphic
twist and framing witnesses for Stage~$1$; category-level $E_d$-formality
of $\HH^\bullet(\cC)$ beyond the operadic-level statement; and a
chain-level identification of the Stage-$2$ output with $\Bbarch_C(A_b)$
on the extended locus.
