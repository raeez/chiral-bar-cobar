criterion.
\end{abstract}

\section{The existence problem}
\label{sec:existence-problem-statement}

\subsection{Motivation and setup}

\begin{question}[Central question]\label{q:existence-central}
Given a chiral algebra $\mathcal{A}$ on $X$, when does there exist a
Koszul dual coalgebra $\mathcal{A}^!$ such that:
\begin{enumerate}
\item The bar construction resolves the dual coalgebra:
      $\bar{B}(\mathcal{A}) \xrightarrow{\sim} \mathcal{A}^!$
\item The cobar of the bar recovers $\mathcal{A}$:
      $\Omega(\bar{B}(\mathcal{A})) \xrightarrow{\sim} \mathcal{A}$
\item The module and comodule categories are equivalent (with
      completeness/conilpotence conditions):
      \[\mathrm{Mod}^{\mathrm{compl}}(\mathcal{A}) \simeq
      \mathrm{CoMod}^{\mathrm{conil}}(\mathcal{A}^!)\]
\end{enumerate}
When conditions (1)--(3) hold, the Koszul dual \emph{algebra} is
obtained by linear dualization:
$(\mathcal{A}^!)^{\vee} = \bar{B}(\mathcal{A})^{\vee}$.
\end{question}

\subsection{Preliminary definitions}

\begin{definition}[Koszul dual: quadratic case (Priddy)]\label{def:koszul-dual-quadratic}
Let $\mathcal{A} = T_{\mathcal{D}}(V) / (R)$ be a \emph{quadratic} chiral algebra,
where $V$ is a graded $\mathcal{D}_X$-module of generators and
$R \subseteq V \otimes_{\mathcal{D}} V$ is a subspace of quadratic relations.
The \emph{Koszul dual} of $\mathcal{A}$ is the quadratic coalgebra
\begin{equation}\label{eq:koszul-dual-quadratic}
\mathcal{A}^! \;\hookrightarrow\; \text{Cofree}_{\mathcal{D}}(sV^*),
\end{equation}
defined as the maximal sub-coalgebra of $\text{Cofree}_{\mathcal{D}}(sV^*)$ cogenerated by $sR^\perp$,
where:
\begin{itemize}
\item $V^* = \text{Hom}_{\mathcal{D}}(V, \mathcal{D}_X)$ is the $\mathcal{D}$-linear dual,
\item $R^\perp = \{ f \in (V^*)^{\otimes 2} : \langle f, r \rangle = 0
      \text{ for all } r \in R \}$ is the annihilator of $R$ under the
      natural pairing $(V^*)^{\otimes 2} \otimes V^{\otimes 2} \to \mathbb{C}$,
\item $s$ denotes the cohomological suspension (degree shift by $+1$).
\end{itemize}
This definition is intrinsic: $\mathcal{A}^!$ is constructed directly from the
presentation $(V, R)$ without reference to the bar construction or any
resolution.  In the chiral setting, the quadratic dual is defined analogously:
for a chiral algebra $\mathcal{A}$ presented by generators and chiral OPE
relations, $\mathcal{A}^!$ is generated by the dual generators with orthogonal
complement relations.
\end{definition}

\begin{remark}[Relationship to the bar construction]\label{rem:bar-vs-quadratic-dual}
The bar construction $\bar{B}(\mathcal{A})$ is defined \emph{independently} of the
Koszul dual: it is the cofree coalgebra on the desuspension of the augmentation
ideal $\bar{\mathcal{A}} = \ker(\varepsilon)$,
\[\bar{B}(\mathcal{A}) \;=\; \bigl(T^c(s^{-1}\bar{\mathcal{A}}),\; d\bigr),\]
equipped with the differential $d$ induced by the multiplication of $\mathcal{A}$.
No knowledge of $\mathcal{A}^!$ is required to define $\bar{B}(\mathcal{A})$.

The \emph{Koszulness condition} (Definition~\ref{def:koszul-property-existence}
below) is precisely the statement that the natural projection
$\bar{B}(\mathcal{A}) \twoheadrightarrow \mathcal{A}^!$ is a quasi-isomorphism
of dg coalgebras.
Thus the bar construction provides a dg coalgebra \emph{resolution} of
$\mathcal{A}^!$ when $\mathcal{A}$ is Koszul, not its definition.
To obtain the Koszul dual algebra from the coalgebra~$\mathcal{A}^!$,
one takes the graded linear dual $(\mathcal{A}^!)^{\vee}$; this
requires finite-dimensionality of each graded component
of~$\mathcal{A}^!$.
\end{remark}

\begin{definition}[Koszul property]\label{def:koszul-property-existence}
Let $\mathcal{A} = T_{\mathcal{D}}(V)/(R)$ be a quadratic chiral algebra with
Koszul dual $\mathcal{A}^!$ as in Definition~\ref{def:koszul-dual-quadratic}.
Then $\mathcal{A}$ is \emph{Koszul} if:
\begin{enumerate}
\item The natural projection from the bar coalgebra to the Koszul dual
      coalgebra is a quasi-isomorphism of dg coalgebras:
      \[\bar{B}(\mathcal{A}) \xrightarrow{\;\sim\;} \mathcal{A}^!\]
      (Dualizing both sides then gives
      $\bar{B}(\mathcal{A})^{\vee} \simeq (\mathcal{A}^!)^{\vee}$,
      recovering the Koszul dual \emph{algebra} as the linear dual of
      the Koszul dual coalgebra; see
      Remark~\ref{rem:bar-vs-quadratic-dual}.)
\item Equivalently, the cobar of the bar recovers $\mathcal{A}$:
      \[\Omega(\bar{B}(\mathcal{A})) \xrightarrow{\;\sim\;} \mathcal{A}\]
\item The module-comodule categories are equivalent (with appropriate completeness/conilpotence):
      \[\mathrm{Mod}^{\mathrm{compl}}(\mathcal{A}) \simeq \mathrm{CoMod}^{\mathrm{conil}}(\mathcal{A}^!)\]
\end{enumerate}
Conditions (1) and (2) are equivalent for conilpotent coalgebras
(see \cite{LV12}, Theorem 2.3.3).  Note that $\mathcal{A}^!$ always exists for
quadratic algebras; the content of the Koszul property is that the bar
construction provides a \emph{resolution} of $\mathcal{A}^!$, rather than merely
surjecting onto it.
\end{definition}


\section{Quadratic case: constructive existence proof}
\label{sec:quadratic-existence}

\subsection{Statement of result}

\begin{theorem}[Quadratic algebras have duals \cite{LV12, Priddy70}; \ClaimStatusProvedElsewhere]\label{thm:quadratic-have-duals}
Let $\mathcal{A} = \text{Free}_{\mathcal{D}}(V) / (R)$ be a \emph{quadratic} 
chiral algebra, meaning:
\begin{enumerate}
\item $V$ is a graded vector space (the generators)
\item $R \subseteq V^{\otimes 2}$ (the relations are quadratic)
\item $R$ satisfies certain regularity conditions (specified below)
\end{enumerate}

Then $\mathcal{A}$ admits a Koszul dual coalgebra $\mathcal{A}^!$, which can be 
constructed explicitly.
\end{theorem}

\begin{proof}[Proof sketch]
The proof is constructive and has four steps.

\emph{Step~1. Construct the dual space.}
Define:
\[V^* = \text{Hom}(V, \mathbb{C})\]
with dual grading: $|v^*| = -|v|$.

\emph{Step~2. Define the dual relations.}
The relations $R \subseteq V^{\otimes 2}$ induce:
\[R^\perp \subseteq (V^*)^{\otimes 2}\]
defined by:
\[R^\perp = \{f \in (V^*)^{\otimes 2} : f(r) = 0 \text{ for all } r \in R\}\]

\emph{Step~3. Build the dual coalgebra.}
Define $\mathcal{A}^!$ as the maximal sub-coalgebra of $\text{Cofree}_{\mathcal{D}}(sV^*)$ cogenerated by $sR^\perp$:
\[\mathcal{A}^! \hookrightarrow \text{Cofree}_{\mathcal{D}}(sV^*)\]
where $s$ denotes suspension (degree shift by 1).  (Note: the dual \emph{coalgebra} is a sub-object of the cofree coalgebra, not a quotient; the quotient construction $\text{Free}(sV^*)/(sR^\perp)$ gives the dual \emph{algebra}.)

\emph{Step~4. Verify the Koszul property.}
One verifies that $\mathcal{A}^!$ has a well-defined coalgebra structure, that the bar coalgebra resolves the dual coalgebra ($\bar{B}(\mathcal{A}) \xrightarrow{\sim} \mathcal{A}^!$ as dg coalgebras), and that the cobar of the bar recovers $\mathcal{A}$ ($\Omega(\bar{B}(\mathcal{A})) \xrightarrow{\sim} \mathcal{A}$).  The coalgebra structure on $\mathcal{A}^!$ follows from the universal property of the cofree coalgebra: the maximal sub-coalgebra cogenerated by $sR^\perp$ inherits coassociativity from $\text{Cofree}_{\mathcal{D}}(sV^*)$; see \cite{LV12}*{Theorem~7.1.2}.
\end{proof}

\subsection{Explicit construction}

\begin{construction}[Building \texorpdfstring{$\mathcal{A}^!$}{A!} from \texorpdfstring{$\mathcal{A}$}{A}]\label{const:quadratic-dual-criteria}
Given quadratic $\mathcal{A} = \text{Free}(V) / (R)$, construct $\mathcal{A}^!$ as follows:

\emph{Input.}
Generators $V = \text{span}\{v_1, \ldots, v_n\}$ with degrees $|v_i|$, and relations $R = \text{span}\{r_1, \ldots, r_m\}$ where each $r_j \in V^{\otimes 2}$.

\emph{Step~1. Dual generators.}
\[V^* = \text{span}\{v_1^*, \ldots, v_n^*\} \quad \text{with } |v_i^*| = -|v_i| + 1\]

(The shift by 1 is the suspension $s$.)

\emph{Step~2. Write relations in coordinates.}
Express each relation as:
\[r_j = \sum_{i,k} a_{ik}^j \, v_i \otimes v_k\]

\emph{Step~3. Orthogonal complement of relations.}
Under the natural pairing $\langle v_i^* \otimes v_k^*, v_l \otimes v_m \rangle
= \delta_{il}\delta_{km}$, define:
\[R^\perp = \{f \in (V^*)^{\otimes 2} : \langle f, r \rangle = 0
\;\text{for all}\; r \in R\}\]
This is a subspace of codimension $\dim R$ in $(V^*)^{\otimes 2}$.
\emph{Warning.} $R^\perp$ is the \emph{annihilator} of $R$, not the image of $R$
under the basis identification $V \cong V^*$

\emph{Step~4. Koszul dual coalgebra.}
\[\mathcal{A}^! = \text{sub-coalgebra of } \text{Cofree}(sV^*) \text{ cogenerated by } R^\perp\]

That is, $\mathcal{A}^!$ is the maximal sub-coalgebra of $\text{Cofree}(sV^*)$ whose degree-2 component lies in $R^\perp \subset (sV^*)^{\otimes 2}$.  Here $\text{Cofree}(sV^*)$ denotes the cofree coalgebra:
\[\text{Cofree}(V^*) = \bigoplus_{n=0}^\infty (V^*)^{\otimes n}\]
with comultiplication:
\[\Delta(v_1^* \otimes \cdots \otimes v_n^*) = \sum_{i=0}^n 
(v_1^* \otimes \cdots \otimes v_i^*) \otimes (v_{i+1}^* \otimes \cdots \otimes v_n^*)\]
\end{construction}

\begin{example}[Heisenberg algebra]\label{ex:heisenberg-quadratic-dual}
The Heisenberg chiral algebra is:
\[\mathcal{H} = \text{Free}(a, a^*) / ([a, a^*] - 1)\]

\emph{Generators.} $V = \text{span}\{a, a^*\}$ with $|a| = 1$, $|a^*| = 1$.

\emph{Relation (quadratic part).} $R = \operatorname{span}\{a \otimes a^* - a^* \otimes a\} \subset V^{\otimes 2}$

(The full relation $[a, a^*] = \kappa$ is inhomogeneous; the constant $\kappa$
contributes curvature $m_0$ in the bar complex, not a quadratic relation.
For Koszul duality, we work with the quadratic part.)

\emph{Dual generators.} $V^* = \operatorname{span}\{a^{\dagger}, (a^*)^{\dagger}\}$ with
$|a^{\dagger}| = 0$, $|(a^*)^{\dagger}| = 0$ (after suspension).

\emph{Orthogonal complement.} $R$ is the antisymmetric part of $V^{\otimes 2}$,
so its annihilator is the symmetric part:
\[R^\perp = \operatorname{Sym}^2(V^*) = \operatorname{span}\{(a^{\dagger})^2,\;
((a^*)^{\dagger})^2,\; a^{\dagger}(a^*)^{\dagger} + (a^*)^{\dagger}a^{\dagger}\}\]

\emph{Dual coalgebra.}
\[\mathcal{H}^! \hookrightarrow \text{Cofree}(a^{\dagger}, (a^*)^{\dagger})\]
(the maximal sub-coalgebra cogenerated by $R^\perp$)

This has comultiplication:
\[\Delta(a^{\dagger}) = a^{\dagger} \otimes 1 + 1 \otimes a^{\dagger}\]
\[\Delta((a^*)^{\dagger}) = (a^*)^{\dagger} \otimes 1 + 1 \otimes (a^*)^{\dagger}\]

(These are primitive.)

\emph{Verification.} The bar construction resolves the dual coalgebra,
and the cobar of the bar recovers the original algebra:
\[\bar{B}(\mathcal{H}) \xrightarrow{\sim} \mathcal{H}^!
\quad \text{(dg coalgebras)} \quad \text{and} \quad
\Omega(\bar{B}(\mathcal{H})) \xrightarrow{\sim} \mathcal{H}
\quad \text{(chiral algebras)}.\]
\end{example}

\subsection{Regularity conditions}

\begin{definition}[Quadratic regularity]\label{def:quadratic-regularity}
A quadratic presentation $\mathcal{A} = \text{Free}(V) / (R)$ is \emph{regular} if:
\begin{enumerate}
\item \emph{Non-degeneracy.} The pairing:
      \[V \otimes V^* \to \mathbb{C}\]
      is non-degenerate
      
\item \emph{Compatibility.} The relations $R$ and $R^\perp$ satisfy:
      \[(R^\perp)^\perp = R\]
      (the dual of the dual recovers the original)
      
\item \emph{Conilpotency.} The coalgebra $\mathcal{A}^!$ is conilpotent: every element
      of the coaugmentation coideal $\overline{\mathcal{A}^!} = \ker(\varepsilon)$ is
      annihilated by a sufficiently high iterate of the reduced coproduct $\bar{\Delta}$:
      \[\bigcup_{n=0}^\infty \ker(\bar{\Delta}^n) = \overline{\mathcal{A}^!}\]
\end{enumerate}
\end{definition}

\begin{theorem}[Regularity implies Koszul \cite{BGS96, LV12}; \ClaimStatusProvedElsewhere]\label{thm:regular-implies-koszul}
If $\mathcal{A} = \text{Free}(V) / (R)$ is quadratic and regular (Definition 
\ref{def:quadratic-regularity}), then:
\begin{enumerate}
\item $\mathcal{A}^!$ exists and is given by Construction \ref{const:quadratic-dual-criteria}
\item $\mathcal{A}$ is Koszul
\item Bar-cobar quasi-isomorphism holds (Theorem~\ref{thm:bar-cobar-inversion-qi})
\end{enumerate}
\end{theorem}


\section{Non-quadratic case: completion required}
\label{sec:non-quadratic-existence}

\subsection{Necessity of completion}

\begin{remark}[Obstructions in non-quadratic case]\label{rem:non-quadratic-obstructions}
For higher-order relations ($n \ge 3$), the dual relations $r^\perp \in (V^*)^{\otimes n}$ need not form a coalgebra ideal, the bidual map may have infinite-dimensional kernel, and $\mathcal{A}^*$ lacks a natural comultiplication.  All three obstructions are resolved by working with filtered completions.
\end{remark}

\begin{definition}[Filtered completion]\label{def:filtered-completion}
Let $\mathcal{A} = \text{Free}(V) / (R)$ with $R$ containing higher-order relations. 
Define the \emph{filtered completion} $\widehat{\mathcal{A}}$ as:

\emph{Step~1. Choose filtration.}
Filter $\mathcal{A}$ by order of monomials:
\[F_0\mathcal{A} \subseteq F_1\mathcal{A} \subseteq F_2\mathcal{A} \subseteq \cdots\]
where $F_n\mathcal{A}$ = elements of order $\leq n$.

\emph{Step~2. Complete.}
\[\widehat{\mathcal{A}} = \varprojlim_n \mathcal{A} / F_n\mathcal{A}\]

This is the \emph{inverse limit} of finite-dimensional quotients.
\end{definition}

\begin{theorem}[Completed Koszul dual \cite{Positselski11}; \ClaimStatusProvedElsewhere]\label{thm:completed-koszul-dual}
For a non-quadratic chiral algebra $\mathcal{A}$ with regular filtration, the 
completed dual:
\[\widehat{\mathcal{A}^!} = \varprojlim_n (\mathcal{A} / F_n\mathcal{A})^!\]
exists and satisfies:
\begin{enumerate}
\item $\widehat{\mathcal{A}^!}$ is a completed chiral coalgebra
\item Bar-cobar constructions converge in completion:
      \[\Omega(\bar{B}(\widehat{\mathcal{A}})) \simeq \widehat{\mathcal{A}} \quad \text{and} \quad
      \bar{B}(\widehat{\mathcal{A}}) \simeq \widehat{\mathcal{A}^!}\]
\item Derived categories equivalent (completed version)
\end{enumerate}
\end{theorem}

\subsection{I-adic completion}

\begin{definition}[I-adic completion]\label{def:i-adic-completion}
Let $I \subseteq \mathcal{A}$ be an ideal (the ``augmentation ideal''). The 
\emph{$I$-adic completion} is:
\[\widehat{\mathcal{A}}_I = \varprojlim_n \mathcal{A} / I^n\]

\emph{Geometric meaning.} This is completion at the ``point at infinity'' or 
``augmentation point'' of the spectrum.
\end{definition}

\begin{example}[Virasoro algebra]\label{ex:virasoro-i-adic}
The Virasoro algebra is:
\[\text{Vir} = \text{span}\{L_n, c : n \in \mathbb{Z}\} / 
([L_m, L_n] - (m-n)L_{m+n} - \frac{c}{12}(m^3 - m)\delta_{m+n,0})\]

This is \emph{inhomogeneous quadratic}: the relation $[L_m, L_n] = (m-n)L_{m+n} + \frac{c}{12}(m^3-m)\delta_{m+n,0}$ is bilinear in generators plus a constant (the central charge defines a Lie algebra 2-cocycle in $H^2(\mathrm{Witt}, \mathbb{C})$).

\emph{Augmentation ideal.} $I = \text{span}\{L_n : n \in \mathbb{Z}\}$ (the kernel of the counit $\varepsilon: U(\mathrm{Vir}) \to \mathbb{C}$ sending all $L_n \mapsto 0$ and $c \mapsto c_0$)

\emph{Completion.}
\[\widehat{\text{Vir}} = \varprojlim_n \text{Vir} / I^n\]

In this completion, the Virasoro algebra acquires a completed curved bar
model whose current manuscript-level dual datum is the same-family
shadow $\mathrm{Vir}_{26-c}$ rather than an H-level infinite-generator
object:
\[\widehat{\text{Vir}^!} = \text{completed cofree coalgebra on } L_n^*, c^*\]

\emph{Warning.} Without completion, $\text{Vir}$ does NOT have a Koszul dual in 
the naive sense.
\end{example}

\subsection{Convergence criteria}

\begin{theorem}[Convergence of completion \cite{Positselski11}; \ClaimStatusProvedElsewhere]\label{thm:completion-convergence-criteria}
For a filtered chiral algebra $\mathcal{A}$ with filtration $\{F_n\}$, the 
completion $\widehat{\mathcal{A}} = \varprojlim_n \mathcal{A}/F_n$ converges if:
\begin{enumerate}
\item \emph{Exhaustiveness.} $\bigcup_n F_n = \mathcal{A}$
\item \emph{Separatedness.} $\bigcap_n F_n = 0$
\item \emph{Bounded growth.} $\dim(F_n / F_{n-1}) = O(n^k)$ for some $k$
\end{enumerate}

When these hold, $\widehat{\mathcal{A}}$ is a well-defined topological algebra, 
and $\widehat{\mathcal{A}^!}$ exists.
\end{theorem}

\section{Algorithmic existence test}
\label{sec:algorithmic-test}

\subsection{The algorithm}

\begin{construction}[Existence test for Koszul duals]\label{con:koszul-dual-existence}
Given a chiral algebra $\mathcal{A} = \operatorname{Free}(V) / (R)$, one determines whether the Koszul dual exists as follows.

\emph{Step~1: Quadratic test.}
If all relations in $R$ lie in $V^{\otimes 2}$, proceed to Step~2; otherwise pass to Step~4 (non-quadratic case).

\emph{Step~2: Regularity.}
Compute the dual space $V^*$ and dual relations $R^\perp \subseteq (V^*)^{\otimes 2}$.  If $(R^\perp)^\perp = R$ and the pairing $V \otimes V^* \to \mathbb{C}$ is non-degenerate, proceed; otherwise the dual does not exist.

\emph{Step~3: Construction and conilpotency.}
Form the candidate dual $\mathcal{A}^! \hookrightarrow \operatorname{Cofree}(sV^*)$ as the maximal sub-coalgebra cogenerated by $sR^\perp$.  The dual exists if and only if $\mathcal{A}^!$ is conilpotent.

\emph{Step~4: Non-quadratic case.}
Compute $\max_i \deg(r_i)$.  If relations have bounded degree, define a filtration by order and check the convergence criteria of Theorem~\ref{thm:completion-convergence-criteria}.  The dual exists (after completion) if and only if convergence holds.
\end{construction}

\subsection{Examples of algorithm application}

\begin{example}[Algorithm applied to Heisenberg]\label{ex:algorithm-heisenberg}
Input: $\mathcal{H} = \text{Free}(a, a^*) / ([a, a^*] - 1)$

\emph{Step~1.} Check quadratic. The Heisenberg relation $[a, a^*] = 1$ is inhomogeneous quadratic; the homogeneous part of the relation space is:
\[R = \{a \otimes a^* - a^* \otimes a\} \subseteq V^{\otimes 2}\]
Quadratic (in the inhomogeneous sense of Positselski). Proceed to Step 2.

\emph{Step~2.} Check regularity.
\[V^* = \text{span}\{a^{\dagger}, (a^*)^{\dagger}\}\]
\[R^\perp = \mathrm{Sym}^2(V^*) = \text{span}\{a^{\dagger} \otimes a^{\dagger},\; (a^*)^{\dagger} \otimes (a^*)^{\dagger},\; a^{\dagger} \otimes (a^*)^{\dagger} + (a^*)^{\dagger} \otimes a^{\dagger}\}\]

Check: $(R^\perp)^\perp = R$? Yes.
Check: $V \otimes V^* \to \mathbb{C}$ non-degenerate? Yes.

Regular. Proceed to Step 3.

\emph{Step~3.} Construct dual.
\[\mathcal{H}^! = \text{sub-coalgebra of } \text{Cofree}(sa^{\dagger}, s(a^*)^{\dagger}) \text{ cogenerated by } sR^\perp\]

\emph{Step~4.} Verify conilpotency.
\[\Delta(a^{\dagger}) = a^{\dagger} \otimes 1 + 1 \otimes a^{\dagger}\]

This is primitive, hence conilpotent.

\emph{Output.} YES, dual exists, $\mathcal{H}^!$ as constructed.
\end{example}

\begin{example}[Running algorithm on \texorpdfstring{$\mathcal{W}_\infty$}{W\_infinity}]\label{ex:algorithm-w-infinity}
Input: $\mathcal{W}_\infty$ (W-algebra with infinitely many generators of all weights)

\emph{Step~1.} Check quadratic.
Relations include terms of all orders (quadratic, cubic, quartic, \ldots).
Not quadratic. Proceed to Step~4.

\emph{Step~4.} Non-quadratic case: bar construction and Koszul property.
Check: Are relations bounded degree? 

NO. $\mathcal{W}_\infty$ has relations of arbitrarily high degree.

\emph{Output.} The bar construction $\bar{B}(\mathcal{W}_\infty)$ exists as a dg coalgebra.  Although the naive Koszul property fails (the counit $\Omega(\bar{B}(\mathcal{W}_\infty)) \to \mathcal{W}_\infty$ is \emph{not} a quasi-isomorphism at finite stage), the strong completion-tower theorem (Theorem~\ref{thm:completed-bar-cobar-strong}) establishes a completed Koszul duality on the completion-closed subcategory $\mathrm{CompCl}(\mathcal{F}_{\mathrm{ft}})$.

\emph{Explanation.} The unbounded complexity of $\mathcal{W}_\infty$ (relations of arbitrarily high degree among infinitely many generators) prevents the bar spectral sequence from degenerating at any finite page, but the strong completion tower circumvents this obstruction.
\end{example}

\subsection{Detailed analysis: Kac--Moody}

\begin{proposition}[Kac--Moody Koszul duals \cite{FBZ04, Feigin-Frenkel}; \ClaimStatusProvedElsewhere]\label{prop:kac-moody-koszul-duals}
Let $\widehat{\mathfrak{g}}_k$ be the affine Kac--Moody algebra at level $k \in \mathbb{C}$. 
Then:
\begin{enumerate}
\item For \emph{generic} $k$, $\widehat{\mathfrak{g}}_k$ is Koszul with dual:
      \[(\widehat{\mathfrak{g}}_k)^! \cong \mathrm{CE}^{\mathrm{ch}}(\widehat{\mathfrak{g}}_{-k - 2h^\vee})\]
      i.e., the chiral Chevalley--Eilenberg complex at the shifted level $-k - 2h^\vee$.

      \emph{Relation to Feigin--Frenkel duality.} For simply-laced $\mathfrak{g}$,
      a separate (conjectural) identification relates the Koszul dual to the Langlands dual algebra:
      $\widehat{\mathfrak{g}^L}_{k^L}$ where $(k + h^\vee)(k^L + h^{\vee,L}) = 1$.
      This is Feigin--Frenkel level duality, which is \emph{not} the same as chiral Koszul duality but interacts with it via the geometric Langlands correspondence
      
\item For \emph{special} values ($k = -h^\vee$, critical level), the dual requires 
      completion
      
\item The relationship between chiral Koszul duality (level shift $k \mapsto -k - 2h^\vee$) and Feigin--Frenkel/Langlands duality ($(k+h^\vee)(k^L + h^{\vee,L}) = 1$) is subtle; the two are distinct dualities that interact via the geometric Langlands program
\end{enumerate}
\end{proposition}

\begin{proof}
The Kac--Moody algebra has presentation:
\[\widehat{\mathfrak{g}}_k = \text{Free}(\{J^a_n : a \in \mathfrak{g}, n \in \mathbb{Z}\}) 
/ \text{(Kac--Moody relations)}\]

The relations are quadratic:
\[[J^a_m, J^b_n] = f^{ab}_c J^c_{m+n} + k \delta_{m+n,0} \langle a, b \rangle\]

\emph{Koszul dual generators.} $(J^a_n)^*$ in the dual of the same Lie algebra $\mathfrak{g}$ (not the Langlands dual $\mathfrak{g}^L$).  The Langlands dual appears only for non-simply-laced $\mathfrak{g}$ via the Feigin--Frenkel theorem, which is a deeper statement than Koszul duality.

\emph{Dual relations.} Determined by the orthogonal complement of the quadratic relations with respect to the Koszul pairing on $V \otimes V$.

\emph{Level matching.} The formula relating $k$ and $k^L$ comes from requiring:
\[\langle \widehat{\mathfrak{g}}_k, (\widehat{\mathfrak{g}}_k)^! \rangle\]
to be non-degenerate.

For details, see Chapter~\ref{chap:kac-moody}.
\end{proof}

\subsection{Detailed analysis: $\mathcal{W}$-algebras}

\begin{proposition}[\texorpdfstring{$\mathcal{W}$}{W}-algebra Koszul analysis]\label{prop:w-algebra-koszul}
\begin{enumerate}
\item\emph{(\ClaimStatusProvedHere)}
The universal $\mathcal{W}$-algebra $\mathcal{W}^k(\fg, f_{\mathrm{prin}})$
is chirally Koszul at every level~$k$, for every simple~$\fg$.
The completed Koszul dual $(\mathcal{W}^k)^!$ exists: for finitely generated
$\mathcal{W}_N$ ($N < \infty$) by
Theorem~\textup{\ref{thm:completed-bar-cobar-strong}}, and for
$\mathcal{W}_\infty$ by weight stabilisation
(Theorem~\textup{\ref{thm:stabilized-completion-positive}}).

\item\emph{(Open.)}
For the simple quotient $\mathcal{W}_k(\fg)$ at admissible level~$k$:
$C_2$-cofiniteness holds~\cite{Arakawa15}, and the Zhu algebra is
finite-dimensional, but chiral Koszulness is not established.
The obstruction is the passage from ordinary $\operatorname{Ext}$
(Zhu level) to bar-$\operatorname{Ext}$ (chiral level): vacuum null
vectors in the simple quotient can create bar cohomology outside
the diagonal, and no mechanism to exclude this is known.
\end{enumerate}
\end{proposition}

\begin{proof}[Proof of item~\textup{1}]
$\mathcal{W}^k(\fg, f_{\mathrm{prin}})$ is freely strongly generated
by the Feigin--Frenkel theorem.
Proposition~\ref{prop:pbw-universality} applies.
\end{proof}

\section{Practical computation of Koszul duals}
\label{sec:practical-computation}

\subsection{Step-by-step guide}

\begin{procedure}[Computing \texorpdfstring{$\mathcal{A}^!$}{A!} in practice]\label{proc:compute-koszul-dual}
Given $\mathcal{A} = \text{Free}(V) / (R)$:

\emph{Stage~1. Preparation.}
\begin{enumerate}
\item Write generators $V = \text{span}\{v_1, \ldots, v_n\}$ with explicit degrees
\item Write relations $R = \text{span}\{r_1, \ldots, r_m\}$ in normal form
\item Identify which relations are quadratic, cubic, etc.
\end{enumerate}

\emph{Stage~2. Quadratic part.}
\begin{enumerate}
\item Extract quadratic relations $R_2 \subseteq R$
\item Form dual space $V^* = \{v_1^*, \ldots, v_n^*\}$
\item Compute dual quadratic relations $R_2^\perp \subseteq (V^*)^{\otimes 2}$
\item Build quadratic part of dual:
      \[\mathcal{A}^!_{\text{quad}} = \text{sub-coalgebra of } \text{Cofree}(sV^*) \text{ cogenerated by } sR_2^\perp\]
\end{enumerate}

\emph{Stage~3. Higher relations (if applicable).}
\begin{enumerate}
\item If $R$ contains only quadratic relations: Done, $\mathcal{A}^! = \mathcal{A}^!_{\text{quad}}$
\item If $R$ contains higher relations:
   \begin{enumerate}
   \item Define filtration $F_n\mathcal{A}$ by degree
   \item Compute successive quotients $\mathcal{A}_n = \mathcal{A} / F_n$
   \item Compute duals $\mathcal{A}_n^!$ for each $n$
   \item Take inverse limit: $\widehat{\mathcal{A}^!} = \varprojlim_n \mathcal{A}_n^!$
   \end{enumerate}
\end{enumerate}

\emph{Stage~4. Verification.}
\begin{enumerate}
\item Check conilpotency: $\ker(\varepsilon) = \bigcup_n \ker(\bar{\Delta}^{(n)})$ (every element of the coaugmentation coideal is annihilated by a sufficiently high iterate of the reduced coproduct)
\item Verify bar resolution: Compute $\bar{B}(\mathcal{A})$ and check $\bar{B}(\mathcal{A}) \xrightarrow{\sim} \mathcal{A}^!$ (quasi-isomorphism of dg coalgebras)
\item Verify cobar-bar: Compute $\Omega(\bar{B}(\mathcal{A}))$ and compare to $\mathcal{A}$
\end{enumerate}
\end{procedure}

\subsection{\texorpdfstring{Worked example: free fermion ($bc$ system)}{Worked example: free fermion}}

\begin{example}[Free fermion (\texorpdfstring{$bc$}{bc} ghost) system]\label{ex:free-fermion-koszul-dual}
The free fermion ($bc$ ghost) system is:
\[\mathcal{F} = \text{Free}(b, c) / (b \otimes c + c \otimes b)\]

\emph{Generators.} $V = \text{span}\{b, c\}$ with $|b| = \lambda$,
$|c| = 1 - \lambda$.

\emph{Relation.} $R = \{b \otimes c + c \otimes b\}$ (anticommutation).

\emph{Step~1. Dual space.}
\[V^* = \text{span}\{b^*, c^*\}\]
with $|b^*| = -\lambda + 1$, $|c^*| = \lambda$.

\emph{Step~2. Dual relation.}
\[R^\perp = \operatorname{span}\{b^* \otimes b^*,\; c^* \otimes c^*,\; b^* \otimes c^* - c^* \otimes b^*\}\]

This is 3-dimensional (codimension 1 in $(V^*)^{\otimes 2}$).
Check: $\langle b^* \otimes c^* - c^* \otimes b^*,\, b \otimes c + c \otimes b \rangle = 1 - 1 = 0$.
The symmetric generator $b^* \otimes c^* + c^* \otimes b^*$ has pairing $1 + 1 = 2 \neq 0$, so it is \emph{not} in $R^\perp$.

\emph{Step~3. Construct dual.}
\[\mathcal{F}^! = \text{sub-coalgebra of } \text{Cofree}(sb^*, sc^*) \text{ cogenerated by } sR^\perp\]

Since $R^\perp$ contains the symmetric tensors $b^{*2}$, $c^{*2}$, and $b^*c^* - c^*b^*$,
this is the \emph{symmetric (polynomial) coalgebra}:
\[\mathcal{F}^! = \mathrm{Sym}^{\bullet,c}(sb^*, sc^*)\]

with the cocommutative shuffle comultiplication.
This is the $\beta\gamma$ system (cf.\ Theorem~\ref{thm:betagamma-fermion-koszul}), consistent with the Lie$\leftrightarrow$Com duality: the $bc$ system has antisymmetric (Lie-type) relations, so its Koszul dual has symmetric (Com-type) relations.

\emph{Verification.}
\begin{itemize}
\item Conilpotent? Yes (symmetric coalgebra is conilpotent: the coradical filtration is by polynomial degree).
\item Koszul dual coalgebra? $\bar{B}(\mathcal{F})$ has cohomology concentrated in bar degree~$1$; the linear dual $\bar{B}(\mathcal{F})^\vee \cong \mathcal{F}^! \cong \beta\gamma$ is the Koszul dual \emph{algebra}.
\item Bar-cobar inversion? $\Omega(\bar{B}(\mathcal{F})) \xrightarrow{\sim} \mathcal{F}$ (Koszul complex is acyclic; recovers $\mathcal{F}$ itself).
\end{itemize}

Hence the free fermion ($bc$ ghost) system is Koszul with symmetric coalgebra dual ($\beta\gamma$ system).
\end{example}

\begin{remark}[Comparison of existence paradigms
  \cite{LV12, Positselski11}]
\label{rem:existence-summary}
Three existence paradigms appear in and around the monograph:
\begin{enumerate}
\item \emph{Strict quadratic/conilpotent paradigm.}  In the classical
      and quadratic literature, regularity plus conilpotence give an
      explicit strict dual and bar-cobar resolution
      \textup{(}Construction~\ref{const:quadratic-dual-criteria}\textup{)}.
\item \emph{Completed/filter-complete paradigm.}  In the non-quadratic
      setting one seeks a completed dual under explicit convergence and
      inverse-limit hypotheses; Appendix~\ref{app:nilpotent-completion}
      records this frontier package and its standard towers.
\item \emph{Negative or comparison statements.}  Claims that no dual
      exists depend on the ambient category, the allowed completion, and the
      level at which one asks for inversion.  They should therefore be read
      as regime-specific comparison statements, not as a single monograph
      theorem overriding the family-by-family theorem surface.
\end{enumerate}
Accordingly this appendix explains how the paradigms fit together; it
does not supply a single constitutional ``if and only if'' criterion
for the whole monograph.
\end{remark}

\subsection{Practical recommendations}

\begin{remark}[Alternatives when the dual does not exist]\label{rem:no-dual-alternatives}
If a chiral algebra $\mathcal{A}$ does not have a Koszul dual, alternatives include:

\emph{Option 1: Weaken to $A_\infty$.}
Even without a Koszul dual, $\mathcal{A}$ admits a minimal $A_\infty$ resolution via the bar-cobar construction: the cobar of the bar $\Omega(\bar{B}(\mathcal{A}))$ is always an $A_\infty$ quasi-isomorphic model of~$\mathcal{A}$, regardless of Koszulness.

\emph{Option 2: Work with derived category.}
The derived category $\mathcal{D}^b(\mathcal{A}\text{-mod})$ exists even without 
explicit dual.

\emph{Option 3: Use factorization homology.}
Factorization homology $\int_X \mathcal{A}$ can be computed without Koszul dual 
(Costello--Gwilliam framework).

\emph{Option 4: Restrict to nice subcategory.}
Sometimes a subcategory of $\mathcal{A}\text{-mod}$ has better properties.
\end{remark}
